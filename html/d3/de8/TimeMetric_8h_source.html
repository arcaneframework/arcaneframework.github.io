119  // Met à nul \a rhs pour ne pas qu'il appelle 'endAction'.
120  rhs.m_collector = nullptr;

122  explicit TimeMetricSentry(const TimeMetricAction& action)
123  : m_collector(action.collector())
124  {
125  if (m_collector)
126  m_id = m_collector->beginAction(action);
127  }
128  ~TimeMetricSentry() noexcept(false)
129  {
130  if (m_collector)
131  m_collector->endAction(m_id);
132  }
133  private:

135  TimeMetricId m_id;
