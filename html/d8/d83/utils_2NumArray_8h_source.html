44 template<typename DataType,int RankValue>
45 class MDSpanBase
46 {
47  using UnqualifiedValueType = std::remove_cv_t<DataType>;
48  friend class NumArrayBase<UnqualifiedValueType,RankValue>;
49  // Pour que MDSpan<const T> ait accès à MDSpan<T>
50  friend class MDSpanBase<const UnqualifiedValueType,RankValue>;
51  public:
52  MDSpanBase() = default;
53  ARCCORE_HOST_DEVICE MDSpanBase(DataType* ptr,ArrayExtentsWithOffset<RankValue> extents)
54  : m_ptr(ptr), m_extents(extents)
55  {
56  }
57  // Constructeur MDSpan<const T> à partir d'un MDSpan<T>
58  template<typename X,typename = std::enable_if_t<std::is_same_v<X,UnqualifiedValueType>>>
59  ARCCORE_HOST_DEVICE MDSpanBase(const MDSpanBase<X,RankValue>& rhs)
60  : m_ptr(rhs.m_ptr), m_extents(rhs.m_extents){}
61  public:
62  ARCCORE_HOST_DEVICE DataType* _internalData() { return m_ptr; }
63  ARCCORE_HOST_DEVICE const DataType* _internalData() const { return m_ptr; }
64  public:
65  ArrayExtents<RankValue> extents() const
66  {
67  return m_extents.extents();
68  }
69  ArrayExtentsWithOffset<RankValue> extentsWithOffset() const
70  {
71  return m_extents;
72  }
73  Int64 extent(int i) const { return m_extents(i); }
74  public:
75  ARCCORE_HOST_DEVICE Int64 offset(ArrayBoundsIndex<RankValue> idx) const
76  {
77  return m_extents.offset(idx);
78  }
80  ARCCORE_HOST_DEVICE DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
81  {
82  return m_ptr[offset(idx)];
83  }
85  ARCCORE_HOST_DEVICE DataType* ptrAt(ArrayBoundsIndex<RankValue> idx) const
86  {
87  return m_ptr+offset(idx);
88  }
89  public:
90  MDSpanBase<const DataType,RankValue> constSpan() const
91  { return MDSpanBase<const DataType,RankValue>(m_ptr,m_extents); }
92  Span<DataType> to1DSpan() { return { m_ptr, m_extents.totalNbElement() }; }
93  Span<const DataType> to1DSpan() const { return { m_ptr, m_extents.totalNbElement() }; }
94  private:
95  // Utilisé uniquement par NumArrayBase pour la copie
96  Span2<const DataType> _internalTo2DSpan() const
97  {
98  Int64 dim1_size = m_extents(0);
99  Int64 dim2_size = 1;
100  for (int i=1; i<RankValue; ++i )
101  dim2_size *= m_extents(i);
102  return { m_ptr, dim1_size, dim2_size };
103  }
104  protected:
105  DataType* m_ptr = nullptr;
106  ArrayExtentsWithOffset<RankValue> m_extents;
107 };

365 template<typename DataType,int RankValue>
366 class NumArrayBase
367 {
368  public:
369  using ConstSpanType = MDSpan<const DataType,RankValue>;
370  using SpanType = MDSpan<DataType,RankValue>;
371 
372  public:
374  Int64 totalNbElement() const { return m_total_nb_element; }
375  Int64 extent(int i) const { return m_span.extent(i); }
376  void resize(ArrayExtents<RankValue> extents)
377  {
378  m_span.m_extents.setSize(extents);
379  _resize();
380  }
381  protected:
382  NumArrayBase() : m_data(platform::getDefaultDataAllocator()){}
383  explicit NumArrayBase(ArrayExtents<RankValue> extents)
384  : m_data(platform::getDefaultDataAllocator())
385  {
386  resize(extents);
387  }
388  private:
389  void _resize()
390  {
391  Int64 dim1_size = extent(0);
392  Int64 dim2_size = 1;
393  // TODO: vérifier débordement.
394  for (int i=1; i<RankValue; ++i )
395  dim2_size *= extent(i);
396  m_total_nb_element = dim1_size * dim2_size;
397  m_data.resize(dim1_size,dim2_size);
398  m_span.m_ptr = m_data.to1DSpan().data();
399  }
400  public:
401  void fill(const DataType& v) { m_data.fill(v); }
402  Int32 nbDimension() const { return RankValue; }
403  ArrayExtents<RankValue> extents() const { return m_span.extents(); }
404  ArrayExtentsWithOffset<RankValue> extentsWithOffset() const
405  {
406  return m_span.extentsWithOffset();
407  }
408  public:
409  SpanType span() { return m_span; }
410  ConstSpanType span() const { return m_span.constSpan(); }
411  ConstSpanType constSpan() const { return m_span.constSpan(); }
412  public:
413  Span<const DataType> to1DSpan() const { return m_data.to1DSpan(); }
414  Span<DataType> to1DSpan() { return m_data.to1DSpan(); }
415  void copy(ConstSpanType rhs) { m_data.copy(rhs._internalTo2DSpan()); }
416  const DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
417  {
418  return m_span(idx);
419  }
420  DataType& s(ArrayBoundsIndex<RankValue> idx)
421  {
422  return m_span(idx);
423  }
424  void swap(NumArrayBase<DataType,RankValue>& rhs)
425  {
426  m_data.swap(rhs.m_data);
427  std::swap(m_span,rhs.m_span);
428  std::swap(m_total_nb_element,rhs.m_total_nb_element);
429  }
430  Int64 capacity() const { return m_data.capacity(); }
431  public:
433  DataType* _internalData() { return m_span._internalData(); }
434  protected:
435  MDSpan<DataType,RankValue> m_span;
436  UniqueArray2<DataType> m_data;
437  Int64 m_total_nb_element = 0;
438 };
