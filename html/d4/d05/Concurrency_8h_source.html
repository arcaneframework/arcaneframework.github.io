192 template<typename InstanceType>
193 class TaskFunctor
194 : public ITaskFunctor
195 {
196  public:
197  typedef void (InstanceType::*FunctorType)();
198  public:
199  TaskFunctor(InstanceType* instance,FunctorType func)
200  : m_instance(instance), m_function(func)
201  {
202  }
203  TaskFunctor(const TaskFunctor& rhs) = default;
204  TaskFunctor& operator=(const TaskFunctor& rhs) = delete;
205  public:
207  void executeFunctor(const TaskContext& /*tc*/) override
208  {
209  (m_instance->*m_function)();
210  }
211  ITaskFunctor* clone(void* buffer,Integer size) override
212  {
213  if (sizeof(*this)>(size_t)size)
214  ARCANE_FATAL("INTERNAL: task functor buffer is too small");
215  return new (buffer) TaskFunctor<InstanceType>(*this);
216  }
217  private:
218  InstanceType* m_instance;
219  FunctorType m_function;
220 };

232 {

235  public:
236  TaskFunctorWithContext(InstanceType* instance,FunctorType func)
237  : m_instance(instance), m_function(func)
238  {
239  }
240  public:
242  void executeFunctor(const TaskContext& tc) override
243  {
244  (m_instance->*m_function)(tc);
245  }
246  ITaskFunctor* clone(void* buffer,Integer size) override
247  {
248  if (sizeof(*this)>(size_t)size)
249  ARCANE_FATAL("INTERNAL: task functor buffer is too small");
250  return new (buffer) TaskFunctorWithContext<InstanceType>(*this);
251  }
252  private:
253  InstanceType* m_instance;
254  FunctorType m_function;
255 };
