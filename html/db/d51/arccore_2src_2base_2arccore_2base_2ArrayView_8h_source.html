670 template<typename T> inline bool
671 operator==(ConstArrayView<T> rhs, ConstArrayView<T> lhs)
672 {
673  if (rhs.size()!=lhs.size())
674  return false;
675  Integer s = rhs.size();
676  for( Integer i=0; i<s; ++i ){
677  if (rhs[i]==lhs[i])
678  continue;
679  else
680  return false;
681  }
682  return true;
683 }

717 template<typename T> inline void
718 dumpArray(std::ostream& o,ConstArrayView<T> val,int max_print)
719 {
720  Integer n = val.size();
721  if (max_print>0 && n>max_print){
722  // N'affiche que les (max_print/2) premiers et les (max_print/2) dernièrs
723  // sinon si le tableau est très grand cela peut générer des
724  // sorties listings énormes.
725  Integer z = (max_print/2);
726  Integer z2 = n - z;
727  o << "[0]=\"" << val[0] << '"';
728  for( Integer i=1; i<z; ++i )
729  o << " [" << i << "]=\"" << val[i] << '"';
730  o << " ... ... (skipping indexes " << z << " to " << z2 << " ) ... ... ";
731  for( Integer i=(z2+1); i<n; ++i )
732  o << " [" << i << "]=\"" << val[i] << '"';
733  }
734  else{
735  for( Integer i=0; i<n; ++i ){
736  if (i!=0)
737  o << ' ';
738  o << "[" << i << "]=\"" << val[i] << '"';
739  }
740  }
741 }
