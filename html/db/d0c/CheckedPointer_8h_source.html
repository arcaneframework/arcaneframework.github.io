55 template<class T>
56 class CheckedPointer
57 {
58  protected:
59 
61  const CheckedPointer<T>& operator=(const CheckedPointer<T>& from)
62  {
63  m_value = from.m_value;
64  return (*this);
65  }
66 
67  template <typename T2>
68  const CheckedPointer<T>& operator=(const CheckedPointer<T2>& from)
69  {
70  m_value = from.get();
71  return (*this);
72  }
73 
74 
76  const CheckedPointer<T>& operator=(T* new_value)
77  {
78  m_value = new_value; return (*this);
79  }
80 
82  CheckedPointer(const CheckedPointer<T>& from) : m_value(from.m_value) {}
83 
85  template <typename T2>
86  CheckedPointer(const CheckedPointer<T2>& from) : m_value(from.m_value) {}
87 
88  public:
89 
91  CheckedPointer() : m_value(nullptr) {}
92 
94  explicit CheckedPointer(T* t) : m_value(t) {}
95 
96  public:
97  explicit operator bool() const { return get()!=nullptr; }
98  public:
99 
101  inline T* operator->() const
102  {
103 #ifdef ARCCORE_CHECK
104  if (!m_value)
105  arccoreNullPointerError();
106 #endif
107  return m_value;
108  }
109 
111  inline T& operator*() const
112  {
113 #ifdef ARCCORE_CHECK
114  if (!m_value)
115  arccoreNullPointerError();
116 #endif
117  return *m_value;
118  }
119 
126  inline T* get() const
127  {
128  return m_value;
129  }
130 
131  inline bool isNull() const
132  {
133  return (!m_value);
134  }
135 
136  protected:
137 
138  T* m_value;
139 };
