138  private:
146  struct Int32View
147  {
148  public:
149  Int32 operator[](Int32 index) const
150  {
151 #ifdef ARCANE_CHECK
152  if (index==NULL_ITEM_LOCAL_ID){
153  // Pour l'entité nulle, la taille doit être 0.
154  if (m_size!=0)
155  arcaneRangeError(index,m_size);
156  }
157  else
158  ARCANE_CHECK_AT(index,m_size);
159 #endif
160  return m_data[index];
161  }
162  void operator=(Int32ArrayView v)
163  {
164  m_data = v.data();
165  m_size = v.size();
166  }
167  // Il faut que data[NULL_ITEM_LOCAL_ID] soit valide.
168  void setNull(Int32* data)
169  {
170  m_data = data;
171  m_size = 0;
172  }
173  operator Int32ConstArrayView() const
174  {
175  return Int32ConstArrayView(m_size,m_data);
176  }
177  Int32ArrayView toMutableView() const
178  {
179  return Int32ArrayView(m_size,m_data);
180  }
181  private:
182  Int32 m_size;
183  Int32* m_data;
184  };
185  public:
186  enum {
187  NODE_IDX = 0,
188  EDGE_IDX = 1,
189  FACE_IDX = 2,
190  CELL_IDX = 3,
191  HPARENT_IDX = 4,
192  HCHILD_IDX = 5,
193  MAX_ITEM_KIND = 6
194  };
195  public:
196  static ItemInternalConnectivityList nullInstance;
197  public:
198  ItemInternalConnectivityList()
199  : m_items(nullptr), m_nb_access_all(0), m_nb_access(0)
200  {
201  for( Integer i=0; i<MAX_ITEM_KIND; ++i ){
202  m_indexes_array[i] = nullptr;
203  m_nb_item_array[i] = nullptr;
204  m_nb_item_null_data[i][0] = 0;
205  m_nb_item_null_data[i][1] = 0;
206  m_nb_item[i].setNull(&m_nb_item_null_data[i][1]);
207  }
208  }
209  public:
214  const Int32* itemLocalIds(Int32 item_kind,Int32 lid) const
215  {
216 #ifdef ARCANE_CHECK
217  ++m_nb_access_all;
218 #endif
219  return &(m_list[item_kind][ m_indexes[item_kind][lid] ]);
220  }
225  Int32 itemLocalId(Int32 item_kind,Int32 lid,Integer index) const
226  {
227 #ifdef ARCANE_CHECK
228  ++m_nb_access;
229 #endif
230  return m_list[item_kind][ m_indexes[item_kind][lid] + index];
231  }
233  Int64 nbAccess() const { return m_nb_access; }
235  Int64 nbAccessAll() const { return m_nb_access_all; }
237  void setConnectivityIndex(Int32 item_kind,Int32ArrayView v)
238  {
239  m_indexes[item_kind] = v;
240  }
242  void setConnectivityList(Int32 item_kind,Int32ConstArrayView v)
243  {
244  m_list[item_kind] = v;
245  }
247  void setConnectivityNbItem(Int32 item_kind,Int32ArrayView v)
248  {
249  m_nb_item[item_kind] = v;
250  }
252  Int32ConstArrayView connectivityIndex(Int32 item_kind) const
253  {
254  return m_indexes[item_kind];
255  }
257  Int32ConstArrayView connectivityList(Int32 item_kind) const
258  {
259  return m_list[item_kind];
260  }
262  Int32ConstArrayView connectivityNbItem(Int32 item_kind) const
263  {
264  return m_nb_item[item_kind];
265  }

313  { return itemLocalId(ItemInternalConnectivityList::HCHILD_IDX,lid,index); }
314  public:
315  Int32 _nbNodeV2(Int32 lid) const { return m_nb_item[NODE_IDX][lid]; }
316  Int32 _nbEdgeV2(Int32 lid) const { return m_nb_item[EDGE_IDX][lid]; }
317  Int32 _nbFaceV2(Int32 lid) const { return m_nb_item[FACE_IDX][lid]; }
318  Int32 _nbCellV2(Int32 lid) const { return m_nb_item[CELL_IDX][lid]; }
319  Int32 _nbHParentV2(Int32 lid) const { return m_nb_item[HPARENT_IDX][lid]; }
320  Int32 _nbHChildrenV2(Int32 lid) const { return m_nb_item[HCHILD_IDX][lid]; }
321  private:
322  // Ces deux tableaux utilisent une vue modifiable par compatibilité avec les anciens
323  // mécanismes. Une fois que ceci auront été supprimés alors il sera
324  // constant
325  Int32ArrayView m_indexes[MAX_ITEM_KIND];
326  Int32View m_nb_item[MAX_ITEM_KIND];
327  Int32ConstArrayView m_list[MAX_ITEM_KIND];
328  public:
329  MeshItemInternalList* m_items;
330  private:
331  // Compte le nombre d'accès pour vérification. A supprimer par la suite.
332  mutable Int64 m_nb_access_all;
333  mutable Int64 m_nb_access;
334  public:
335  Int32ArrayView _mutableConnectivityIndex(Int32 item_kind) const
336  {
337  return m_indexes[item_kind];
338  }
339  Int32ArrayView _mutableConnectivityNbItem(Int32 item_kind) const
340  {
341  return m_nb_item[item_kind].toMutableView();
342  }
343  Int32Array* m_indexes_array[MAX_ITEM_KIND];
344  Int32Array* m_nb_item_array[MAX_ITEM_KIND];
345  private:
346  Int32 m_nb_item_null_data[MAX_ITEM_KIND][2];
347 };

407  II_Refine = 1 << 20,

409  II_JustCoarsened = 1 << 22,
410  II_Inactive = 1 << 23,
411  II_CoarsenInactive = 1 << 24,
412
