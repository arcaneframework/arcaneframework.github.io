43 // Pour indiquer que Span2<T>::view() retourne un Array2View
44 // et Span2<const T>::view() retourne un ConstArray2View.
45 template<typename T>
46 class View2TypeT
47 {
48  public:
49  using view_type = Array2View<T>;
50 };
51 template<typename T>
52 class View2TypeT<const T>
53 {
54  public:
55  using view_type = ConstArray2View<T>;
56 };
57 }

69 template<class T>
70 class Span2
71 {
72  public:
73 
74  using ElementType = T;
75  using element_type = ElementType;
76  using value_type = typename std::remove_cv<ElementType>::type;
77  using index_type = Int64;
78  using difference_type = Int64;
79  using pointer = ElementType*;
80  using const_pointer = typename std::add_const<ElementType*>::type;
81  using reference = ElementType&;
82  using view_type = typename detail::View2TypeT<ElementType>::view_type;
83 
84  public:
86  ARCCORE_HOST_DEVICE Span2(ElementType* ptr,Int64 dim1_size,Int64 dim2_size)
87  : m_ptr(ptr), m_dim1_size(dim1_size), m_dim2_size(dim2_size) {}
89  ARCCORE_HOST_DEVICE Span2() : m_ptr(nullptr), m_dim1_size(0), m_dim2_size(0) {}
91  Span2(const Array2View<value_type>& from)
92  : m_ptr(from.m_ptr), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
93  // Constructeur à partir d'un ConstArrayView. Cela n'est autorisé que
94  // si T est const.
95  template<typename X,typename = std::enable_if_t<std::is_same_v<X,value_type>> >
96  Span2(const ConstArray2View<X>& from)
97  : m_ptr(from.data()), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
98  // Pour un Span<const T>, on a le droit de construire depuis un Span<T>
99  template<typename X,typename = std::enable_if_t<std::is_same_v<X,value_type>> >
100  ARCCORE_HOST_DEVICE Span2(const Span2<X>& from)
101  : m_ptr(from.data()), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
103  // contenant \a asize éléments.
104  public:
106  ARCCORE_HOST_DEVICE Int64 dim1Size() const { return m_dim1_size; }
108  ARCCORE_HOST_DEVICE Int64 dim2Size() const { return m_dim2_size; }
110  ARCCORE_HOST_DEVICE Int64 totalNbElement() const { return m_dim1_size*m_dim2_size; }
111  public:
112  ARCCORE_HOST_DEVICE Span<ElementType> operator[](Int64 i) const
113  {
114  ARCCORE_CHECK_AT(i,m_dim1_size);
115  return Span<ElementType>(m_ptr + (m_dim2_size*i),m_dim2_size);
116  }
118  ARCCORE_HOST_DEVICE ElementType item(Int64 i,Int64 j) const
119  {
120  ARCCORE_CHECK_AT(i,m_dim1_size);
121  ARCCORE_CHECK_AT(j,m_dim2_size);
122  return m_ptr[(m_dim2_size*i) + j];
123  }
125  ARCCORE_HOST_DEVICE ElementType setItem(Int64 i,Int64 j,const ElementType& value)
126  {
127  ARCCORE_CHECK_AT(i,m_dim1_size);
128  ARCCORE_CHECK_AT(j,m_dim2_size);
129  m_ptr[(m_dim2_size*i) + j] = value;
130  }
131  public:
135  view_type smallView()
136  {
137  Integer s1 = arccoreCheckArraySize(m_dim1_size);
138  Integer s2 = arccoreCheckArraySize(m_dim2_size);
139  return view_type(m_ptr,s1,s2);
140  }
141 
145  ConstArrayView<value_type> constSmallView() const
146  {
147  Integer s1 = arccoreCheckArraySize(m_dim1_size);
148  Integer s2 = arccoreCheckArraySize(m_dim2_size);
149  return ConstArrayView<value_type>(m_ptr,s1,s2);
150  }
151  public:
155  inline ElementType* unguardedBasePointer() { return m_ptr; }
159  ARCCORE_HOST_DEVICE inline ElementType* data() { return m_ptr; }
163  ARCCORE_HOST_DEVICE inline const ElementType* data() const { return m_ptr; }
164  private:
165  ElementType* m_ptr;
166  Int64 m_dim1_size;
167  Int64 m_dim2_size;
168 };
