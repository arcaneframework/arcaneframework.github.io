39 class ARCANE_CARTESIANMESH_EXPORT DirNode
40 {
41  friend NodeDirectionMng;
42  private:
43  typedef signed char IndexType;
44  static constexpr IndexType NULL_CELL = -1;
45  struct DirNodeCellIndex
46  {
47  public:
48  IndexType operator[](Int32 i) const
49  {
50  ARCANE_CHECK_AT(i,8);
51  return m_indexes[i];
52  }
53  IndexType& operator[](Int32 i)
54  {
55  ARCANE_CHECK_AT(i,8);
56  return m_indexes[i];
57  }
58  public:
59  IndexType m_indexes[8];
60  };
61  private:
62  // Seul NodeDirectionMng à le droit de construire un DirNode.
63  DirNode(ItemInternal* current,Node next,Node prev,DirNodeCellIndex idx)
64  : m_current(current), m_previous(prev), m_next(next), m_cell_index(idx) {}
65  public:
67  Node previous() const { return m_previous; }
69  NodeLocalId previousId() const { return NodeLocalId(m_previous.localId()); }
71  Node next() const { return m_next; }
73  NodeLocalId nextId() const { return NodeLocalId(m_next.localId()); }
81  Int32 cellIndex(Int32 position) const { return m_cell_index[position]; }
88  CellLocalId cellId(Int32 position) const
89  {
90  Int32 x = cellIndex(position);
91  return (x==NULL_CELL) ? CellLocalId(NULL_ITEM_LOCAL_ID) : CellLocalId(m_current->cellLocalId(x));
92  }
99  Cell cell(Int32 position) const
100  {
101  Int32 x = cellIndex(position);
102  return (x==NULL_CELL) ? Cell() : Cell(m_current->internalCell(x));
103  }
104 
106  Cell nextLeftCell() const { return cell(CNP_NextLeft); }
108  Cell nextRightCell() const { return cell(CNP_NextRight); }
110  Cell previousRightCell() const { return cell(CNP_PreviousRight); }
112  Cell previousLeftCell() const { return cell(CNP_PreviousLeft); }
113 
115  CellLocalId nextLeftCellId() const { return cellId(CNP_NextLeft); }
117  CellLocalId nextRightCellId() const { return cellId(CNP_NextRight); }
119  CellLocalId previousRightCellId() const { return cellId(CNP_PreviousRight); }
121  CellLocalId previousLeftCellId() const { return cellId(CNP_PreviousLeft); }
122 
124  Cell topNextLeftCell() const { return cell(CNP_TopNextLeft); }
126  Cell topNextRightCell() const { return cell(CNP_TopNextRight); }
128  Cell topPreviousRightCell() const { return cell(CNP_TopPreviousRight); }
130  Cell topPreviousLeftCell() const { return cell(CNP_TopPreviousLeft); }
131 
133  CellLocalId topNextLeftCellId() const { return cellId(CNP_TopNextLeft); }
135  CellLocalId topNextRightCellId() const { return cellId(CNP_TopNextRight); }
137  CellLocalId topPreviousRightCellId() const { return cellId(CNP_TopPreviousRight); }
139  CellLocalId topPreviousLeftCellId() const { return cellId(CNP_TopPreviousLeft); }
140 
141  private:
142  ItemInternal* m_current;
143  Node m_previous;
144  Node m_next;
145  DirNodeCellIndex m_cell_index;
146 };
