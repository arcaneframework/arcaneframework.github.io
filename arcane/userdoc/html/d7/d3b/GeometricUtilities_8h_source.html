58  Real u = iso.x;

60 
61  Real x0 = (1.0-u) * (1.0-v);
62  Real x1 = (1.0+u) * (1.0-v);
63  Real x2 = (1.0+u) * (1.0+v);
64  Real x3 = (1.0-u) * (1.0+v);
65 
66  return 0.25 * (m_pos[0] * x0 + m_pos[1] * x1 + m_pos[2] * x2 + m_pos[3] * x3);
