63 template<typename DataType>
64 class View1DSetter
65 {
66  // Pour accéder à m_data;
67  friend class View1DGetterSetter<DataType>;
68  public:
69  using ValueType = DataType;
70  using DataTypeReturnReference = View1DSetter<DataType>;
71  explicit ARCCORE_HOST_DEVICE View1DSetter(Span<DataType> data) : m_data(data){}
72  ARCCORE_HOST_DEVICE DataViewSetter<DataType> operator[](Int64 index) const
73  {
74  return DataViewSetter<DataType>(m_data.ptrAt(index));
75  }
76  DataTypeReturnReference& operator=(const View1DSetter<DataType>& rhs) = delete;
77  public:
78  ARCCORE_HOST_DEVICE void copy(Span<const DataType> rhs)
79  {
80  m_data.copy(rhs);
81  }
82  private:
83  Span<DataType> m_data;
84 };

92 template<typename DataType>
93 class View1DGetterSetter
94 : public View1DSetter<DataType>
95 {

103  public:
104  ARCCORE_HOST_DEVICE DataViewGetterSetter<DataType> operator[](Int64 index) const
105  {

107  }

109  ARCCORE_HOST_DEVICE Span<DataType> value() const { return m_data; }
110  ARCCORE_HOST_DEVICE operator Span<DataType>() { return m_data; }
111  ARCCORE_HOST_DEVICE operator Span<const DataType>() const { return m_data; }

119 template<typename ItemType, typename Accessor>
120 class ItemVariableScalarOutViewT
121 : public VariableViewBase
122 {
123  public:
124 
125  using DataType = typename Accessor::ValueType;
126  using DataTypeReturnReference = DataType&;
127  using ItemIndexType = typename ItemTraitsT<ItemType>::LocalIdType;
128 
129  public:
130 
131  ItemVariableScalarOutViewT(RunCommand& command,IVariable* var,Span<DataType> v)
132  : VariableViewBase(command,var), m_values(v.data()),
133  m_size(v.size()){}
134 
136  SimdSetter<DataType> operator[](SimdItemIndexT<ItemType> simd_item) const
137  {
138  return SimdSetter<DataType>(m_values,simd_item.simdLocalIds());
139  }
140 
142  SimdDirectSetter<DataType> operator[](SimdItemDirectIndexT<ItemType> simd_item) const
143  {
144  return SimdDirectSetter<DataType>(m_values+simd_item.baseLocalId());
145  }
146 
148  ARCCORE_HOST_DEVICE Accessor operator[](ItemIndexType i) const
149  {
150  ARCANE_CHECK_AT(i.localId(),m_size);
151  return Accessor(this->m_values + i.localId());
152  }
153 
155  ARCCORE_HOST_DEVICE Accessor value(ItemIndexType i) const
156  {
157  ARCANE_CHECK_AT(i.localId(),m_size);
158  return Accessor(this->m_values + i.localId());
159  }
160 
162  ARCCORE_HOST_DEVICE void setValue(ItemIndexType i,const DataType& v) const
163  {
164  ARCANE_CHECK_AT(i.localId(),m_size);
165  this->m_values[i.localId()] = v;
166  }
167 
168  private:
169  DataType* m_values;
170  Int64 m_size;
171 };
