33/*!
34 * \brief Converti la valeur de la chaîne \a s dans le type basique \a T
35 * et stocke la valeur dans \a v.
36 *
37 * \retval true en cas d'échec.
38 * \retval false en cas de succès
39 */
40template<class T> inline bool
41builtInGetValue(T& v,const String& s)
42{
43 T read_val = T();
44 std::istringstream sbuf(s.localstr());
45 sbuf >> read_val;
46 if (sbuf.fail() || sbuf.bad())
47 return true;
48 if (!sbuf.eof())
49 return true;
50 v = read_val;
51 return false;
52}

106/*!
107 * \brief Converti la valeur de la chaîne \a s en un booléen
108 *
109 * Stocke la valeur convertie dans \a v.
110 *
111 * \retval true en cas d'échec et la valeur de \a v reste inchangée
112 * \retval false en cas de succès
113 */
114inline bool
115builtInGetValue(bool& v,const String& s)
116{
117 if (s.null())
118 return true;
119 if (s=="false" || s=="faux" || s=="0"){
120 v = false;
121 return false;
122 }
123
124 if (s=="true" || s=="vrai" || s=="1"){
125 v = true;
126 return false;
127 }
128 return true;
129}

218 /*!
219 * \brief Convertit \a s en le type \a T.
220 *
221 * Si \a s.empty() est vrai, alors retourne \a default_value.
222 */
223 static std::optional<T>
224 tryParseIfNotEmpty(StringView s,const T& default_value)
225 {

229 /*!
230 * \brief Convertit la valeur de la variable d'environnement \a s en le type \a T.
231 *
232 * Si platform::getEnvironmentVariable(s) est nul, return std::nullopt.
233 * Sinon, retourne cette valeur convertie en le type \a T. Si la conversion
234 * n'est pas possible, retourne std::nullopt si \a throw_if_invalid vaut \a false ou
235 * lève une exception s'il vaut \a true.
236 */
237 ARCANE_UTILS_EXPORT static std::optional<T>
238 tryParseFromEnvironment(StringView s,bool throw_if_invalid);
239};
