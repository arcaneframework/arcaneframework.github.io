30/*!
31 * \brief Converti la valeur de la chaîne \a s dans le type basique \a T
32 * et stocke la valeur dans \a v.
33 *
34 * \retval true en cas d'échec.
35 * \retval false en cas de succès
36 */
37template<class T> inline bool
38builtInGetValue(T& v,const String& s)
39{
40 T read_val = T();
41 std::istringstream sbuf(s.localstr());
42 sbuf >> read_val;
43 if (sbuf.fail() || sbuf.bad())
44 return true;
45 if (!sbuf.eof())
46 return true;
47 v = read_val;
48 return false;
49}

125/*!
126 * \brief Converti la valeur de la chaîne \a s en un booléen
127 *
128 * Stocke la valeur convertie dans \a v.
129 *
130 * \retval true en cas d'échec et la valeur de \a v reste inchangée
131 * \retval false en cas de succès
132 */
133inline bool
134builtInGetValue(bool& v,const String& s)
135{
136 if (s.null())
137 return true;
138 if (s=="false" || s=="faux" || s=="0"){
139 v = false;
140 return false;
141 }
142
143 if (s=="true" || s=="vrai" || s=="1"){
144 v = true;
145 return false;
146 }
147 return true;
148}
