419 /*!
420 * \brief Indique si l'instance est proche de l'instance nulle.
421 *
422 * \retval true si math::isNearlyZero() est vrai pour chaque composante.
423 * \retval false sinon.
424 */
425 inline constexpr ARCCORE_HOST_DEVICE bool isNearlyZero(const Real2& v)
426 {
427 return math::isNearlyZero(v.x) && math::isNearlyZero(v.y);
428 }
429
430 //! Retourne la norme au carré du couple \f$x^2+y^2+z^2\f$
431 inline constexpr ARCCORE_HOST_DEVICE Real squareNormL2(const Real2& v)
432 {
433 return v.x * v.x + v.y * v.y;
434 }
435
436 //! Retourne la norme du couple \f$\sqrt{x^2+y^2+z^2}\f$
437 inline ARCCORE_HOST_DEVICE Real normL2(const Real2& v)
438 {
439 return math::sqrt(math::squareNormL2(v));
440 }
441
442 /*!
443 * \brief Normalise le couple.
444 *
445 * Si le couple est non nul, divise chaque composante par la norme du couple
446 * (abs()), de telle sorte qu'après l'appel à cette méthode, abs() valent \a 1.
447 * Si le couple est nul, ne fait rien.
448 */
449 inline Real2& mutableNormalize(Real2& v)
450 {
451 Real d = math::normL2(v);
452 if (!math::isZero(d))
453 v.divSame(d);
454 return v;
455 }
