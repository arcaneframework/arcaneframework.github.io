81 * \internal

83 * \ingroup Concurrency

85template<typename InstanceType>
86class TaskFunctor
87: public ITaskFunctor
88{
89 public:
90 typedef void (InstanceType::*FunctorType)();
91 public:
92 TaskFunctor(InstanceType* instance,FunctorType func)
93 : m_instance(instance), m_function(func)
94 {
95 }
96 TaskFunctor(const TaskFunctor& rhs) = default;
97 TaskFunctor& operator=(const TaskFunctor& rhs) = delete;
98 public:
99 //! Exécute la méthode associé
100 void executeFunctor(const TaskContext& /*tc*/) override
101 {
102 (m_instance->*m_function)();
103 }
104 ITaskFunctor* clone(void* buffer,Integer size) override
105 {
106 if (sizeof(*this)>(size_t)size)
107 ARCANE_FATAL("INTERNAL: task functor buffer is too small");
108 return new (buffer) TaskFunctor<InstanceType>(*this);
109 }
110 private:
111 InstanceType* m_instance;
112 FunctorType m_function;
113};

605/*!
606 * \brief Applique en concurrence la fonction lambda \a lambda_function
607 * sur l'intervalle d'itération donné par \a loop_ranges.
608 */
609template<int RankValue,typename LambdaType> inline void
610arcaneParallelFor(const ComplexForLoopRanges<RankValue>& loop_ranges,
611 const ParallelLoopOptions& options,
612 const LambdaType& lambda_function)
613{
614 // Modif Arcane 3.7.9 (septembre 2022)
615 // Effectue une copie pour privatiser au thread courant les valeurs de la lambda.
616 // Cela est nécessaire pour que objets comme les reducers soient bien pris
617 // en compte.
618 // TODO: regarder si on pourrait faire la copie uniquement une fois par thread
619 // si cette copie devient couteuse.
620 auto xfunc = [&lambda_function] (const ComplexForLoopRanges<RankValue>& sub_bounds)
621 {
622 using Type = typename std::remove_reference<LambdaType>::type;
623 Type private_lambda(lambda_function);
624 arcaneSequentialFor(sub_bounds,private_lambda);
625 };
626 LambdaMDRangeFunctor<RankValue,decltype(xfunc)> ipf(xfunc);
627 TaskFactory::executeParallelFor(loop_ranges,options,&ipf);
628}
