194 template<typename InstanceType>
195 class TaskFunctor
196 : public ITaskFunctor
197 {
198  public:
199  typedef void (InstanceType::*FunctorType)();
200  public:
201  TaskFunctor(InstanceType* instance,FunctorType func)
202  : m_instance(instance), m_function(func)
203  {
204  }
205  TaskFunctor(const TaskFunctor& rhs) = default;
206  TaskFunctor& operator=(const TaskFunctor& rhs) = delete;
207  public:
209  void executeFunctor(const TaskContext& /*tc*/) override
210  {
211  (m_instance->*m_function)();
212  }
213  ITaskFunctor* clone(void* buffer,Integer size) override
214  {
215  if (sizeof(*this)>(size_t)size)
216  ARCANE_FATAL("INTERNAL: task functor buffer is too small");
217  return new (buffer) TaskFunctor<InstanceType>(*this);
218  }
219  private:
220  InstanceType* m_instance;
221  FunctorType m_function;
222 };

616 arcaneParallelFor(const ComplexLoopRanges<RankValue>& loop_ranges,
617  const ParallelLoopOptions& options,
618  const LambdaType& lambda_function)
619 {
620  auto xfunc = [=] (const ComplexLoopRanges<RankValue>& sub_bounds)

622  arcaneSequentialFor(sub_bounds,lambda_function);
623  };
624  LambdaMDRangeFunctor<RankValue,decltype(xfunc)> ipf(xfunc);

626 }
