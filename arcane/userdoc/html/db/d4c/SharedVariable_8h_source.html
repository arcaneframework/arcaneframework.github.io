38 template<typename ItemTypeT, typename DataTypeT>
39 class SharedMeshVariableScalarRefT {
40 public:
41  typedef ItemTypeT ItemType;
42  typedef DataTypeT DataType;
43  typedef DataTypeT & DataTypeReturnReference;
44  typedef MeshVariableScalarRefT<ItemTypeT,DataTypeT> TrueVariable;
45  typedef SharedMeshVariableScalarRefT<ItemTypeT,DataTypeT> ThisVariable;
46 
47 public:
48  SharedMeshVariableScalarRefT()
49  : m_true_variable(nullptr)
50  , m_family(NULL)
51  , m_parent_family(NULL)
52  , m_direct_access(false)
53  , m_family_depth(0)
54  {
55  ;
56  }
57 
58 // SharedMeshVariableScalarRefT(TrueVariable & v)
59 // : m_true_variable(v)
60 // , m_family(v.itemGroup().itemFamily())
61 // , m_parent_family(NULL)
62 // , m_direct_access(true)
63 // , m_family_depth(0)
64 // {
65 // ;
66 // }
67 
68  SharedMeshVariableScalarRefT(IItemFamily * family, TrueVariable & v)
69  : m_true_variable(v)
70  , m_family(family)
71  , m_parent_family(family->parentFamily())
72  , m_direct_access(false)
73  , m_family_depth(-1)
74  {
75  IItemFamily * variable_family = v.itemGroup().itemFamily();
76 
77  // Ne gère actuellement qu'au plus un niveau d'imbrication
78  // Si parent_family == family, n'utilise pas l'imbrication
79  // Attn alors si item.parent()!=item
80  if (variable_family == m_family)
81  m_direct_access = true;
82  else if (m_parent_family == variable_family)
83  m_family_depth = 0;
84  else
85  throw FatalErrorException(A_FUNCINFO,"Incompatible Family on shared variable");
86  }
87 
88  SharedMeshVariableScalarRefT(const ThisVariable & v)
89  : m_true_variable(v.m_true_variable)
90  , m_family(v.m_family)
91  , m_parent_family(v.m_parent_family)
92  , m_direct_access(v.m_direct_access)
93  , m_family_depth(v.m_family_depth)
94  {
95  ;
96  }
97 
98  ~SharedMeshVariableScalarRefT()
99  {
100  ;
101  }
102 
103  DataTypeReturnReference operator[](const ItemType & i)
104  {
105  ARCANE_ASSERT((m_family!=m_parent_family || i.internal()==i.internal()->parent(m_family_depth)),("Confusion: item parent differs from item"));
106  return m_true_variable.asArray()[(m_direct_access)?i.localId():i.internal()->parentId(m_family_depth)];
107  }
108 
109  DataType operator[](const ItemType & i) const
110  {
111  ARCANE_ASSERT((m_family!=m_parent_family || i.internal()==i.internal()->parent(m_family_depth)),("Confusion: item parent differs from item"));
112  return m_true_variable.asArray()[(m_direct_access)?i.localId():i.internal()->parentId(m_family_depth)];
113  }
114 
115 
116  DataTypeReturnReference operator[](const ItemEnumeratorT<ItemType> & i)
117  {
118  ARCANE_ASSERT((m_family!=m_parent_family || (*i).internal()==i->parent(m_family_depth)),("Confusion: item parent differs from item"));
119  return m_true_variable.asArray()[(m_direct_access)?i.localId():i->internal()->parentId(m_family_depth)];
120  }
121 
122  DataType operator[](const ItemEnumeratorT<ItemType> & i) const
123  {
124  ARCANE_ASSERT((m_family!=m_parent_family || (*i).internal()==i->parent(m_family_depth)),("Confusion: item parent differs from item"));
125  return m_true_variable.asArray()[(m_direct_access)?i.localId():i->internal()->parentId(m_family_depth)];
126  }
127 
128  TrueVariable & trueVariable()
129  {
130  return m_true_variable;
131  }
132 
133  const TrueVariable & trueVariable() const
134  {
135  return m_true_variable;
136  }
137  public:
139  ARCANE_DEPRECATED_240 void operator=(const ThisVariable & v)
140  {
141  m_true_variable.refersTo(v.m_true_variable);
142  m_family = v.m_family;
143  m_parent_family = v.m_parent_family;
144  m_direct_access = v.m_direct_access;
145  m_family_depth = v.m_family_depth;
146  }
147 
148 protected:
149  TrueVariable m_true_variable;
150  IItemFamily * m_family;
151  IItemFamily * m_parent_family;
152  bool m_direct_access;
153  Integer m_family_depth;
154 };
