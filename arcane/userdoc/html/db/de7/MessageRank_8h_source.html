24namespace Arccore::MessagePassing
25{
26/*!
27 * \brief Rang d'un message.
28 *
29 * Le type exact du rang dépend de l'implémentation. Pour être le plus
30 * générique possible, on utilise le type 'Int32' qui est aussi celui
31 * utilisé par MPI.
32 *
33 * Il existe trois valeurs spéciales pour le rang:
34 * - la valeur par défaut
35 * - la valeur procNullRank() qui correspond à MPI_PROC_NULL
36 * - la valeur anySourceRang() qui correspond à MPI_ANY_SOURCE
37 *
38 * \sa PointToPointMessageInfo
39 */
40class ARCCORE_MESSAGEPASSING_EXPORT MessageRank
41{
42 public:
43
44 /*!
45 * \brief Rang par défaut.
46 *
47 * La signification du rang par défaut dépend du type de message.
48 * \sa PointToPointMessageInfo.
49 */
50 MessageRank()
51 : m_rank(A_NULL_RANK)
52 {}
53
54 explicit MessageRank(Int32 rank)
55 : m_rank(rank)
56 {}
57
58 friend bool operator==(const MessageRank& a, const MessageRank& b)
59 {
60 return a.m_rank == b.m_rank;
61 }
62 friend bool operator!=(const MessageRank& a, const MessageRank& b)
63 {
64 return a.m_rank != b.m_rank;
65 }
66 friend bool operator<(const MessageRank& a, const MessageRank& b)
67 {
68 return a.m_rank < b.m_rank;
69 }
70
71 //! Valeur du rang
72 Int32 value() const { return m_rank; }
73
74 //! Positionne la valeur du rang
75 void setValue(Int32 rank) { m_rank = rank; }
76
77 //! Vrai si rang non initialisé correspondant au rang par défaut
78 bool isNull() const { return m_rank == A_NULL_RANK; }
79
80 //! Vrai si rang correspondant à anySourceRank()
81 bool isAnySource() const { return m_rank == A_ANY_SOURCE_RANK; }
82
83 //! Vrai si rang correspondant à procNullRank()
84 bool isProcNull() const { return m_rank == A_PROC_NULL_RANK; }
85
86 //! Rang correspondant à MPI_ANY_SOURCE
87 static MessageRank anySourceRank() { return MessageRank(A_ANY_SOURCE_RANK); }
88
89 //! Rang correspondant à MPI_PROC_NULL
90 static MessageRank procNullRank() { return MessageRank(A_PROC_NULL_RANK); }
91
92 void print(std::ostream& o) const;
93 friend inline std::ostream& operator<<(std::ostream& o, const MessageRank& tag)
94 {
95 tag.print(o);
96 return o;
97 }
98
99 private:
100
101 Int32 m_rank;
102};
