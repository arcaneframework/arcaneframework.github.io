32/*!
33 * \internal
34 * \brief Classe de base d'un objet avec compteur de référence.
35 *
36 * Ces objets sont gérés par compteur de référence.
37 */
38class ARCCORE_COMMON_EXPORT ObjectImpl
39{
40 public:
41
42 ObjectImpl()
43 : m_ref_count(0)
44 {}
45 ObjectImpl(const ObjectImpl& rhs) = delete;
46 virtual ~ObjectImpl() {}
47 ObjectImpl& operator=(const ObjectImpl& rhs) = delete;
48
49 public:
50
51 //! Incrémente le compteur de référence
52 void addRef() { ++m_ref_count; }
53 //! Décrémente le compteur de référence
54 void removeRef()
55 {
56 Int32 r = --m_ref_count;
57 if (r < 0)
58 _noReferenceErrorCallTerminate(this);
59 if (r == 0)
60 deleteMe();
61 }
62 //! Retourne la valeur du compteur de référence
63 Int32 refCount() const { return m_ref_count.load(); }
64
65 public:
66
67 //! Détruit cet objet
68 virtual void deleteMe() { delete this; }
69
70 private:
71
72 std::atomic<Int32> m_ref_count; //!< Nombre de références sur l'objet.
73
74 private:
75
76 static void _noReferenceErrorCallTerminate(const void* ptr);
77};

132/*!
133 * \ingroup Collection
134 * \brief Enumérateur générique.
135 *
136 * Cette classe permet d'itérer de manière générique sur une collection,
137 * sans connaître le type des éléments de la collection. Pour une itération
138 * utilisant un typage fort, il faut utiliser la classe template EnumeratorT.
139 *
140 * Exemple d'utilisation d'un énumérateur:
141 *
142 * \code
143 * VectorT<int> integers;
144 * for( Enumerator i(integers.enumerator()); ++i; )
145 * cout << i.current() << '\n';
146 * \endcode
147 */
148class ARCCORE_COMMON_EXPORT EnumeratorBase
149{
150 public:
151
152 //! Contruit un énumérateur nul.
153 EnumeratorBase() = default;
154
155 /*!
156 * \brief Contruit un énumérateur associé à l'implémentation \a impl.
157 *
158 * L'instance devient propriétaire de l'implémentation qui est détruite
159 * lorsque l'instance est détruite.
160 */
161 explicit EnumeratorBase(EnumeratorImplBase* impl)
162 : m_impl(impl)
163 {}
164
165 public:
166
167 void reset() { m_impl->reset(); }
168 bool moveNext() { return m_impl->moveNext(); }
169 void* current() { return m_impl->current(); }
170 const void* current() const { return m_impl->current(); }
171
172 public:
173
174 //! Avance l'énumérateur sur l'élément suivant.
175 bool operator++() { return moveNext(); }
176
177 protected:
178
179 EnumeratorImplBase* _impl() { return m_impl.get(); }
180 const EnumeratorImplBase* _impl() const { return m_impl.get(); }
181
182 private:
183
184 AutoRef2<EnumeratorImplBase> m_impl; //!< Implémentation
185};

299/*!
300 * \internal
301 * \brief classe de base d'implémentation d'une collection.
302 *
303 * Une collection est un objet contenant des éléments (i.e. un conteneur).
304 *
305 * Il est possible de parcourir les éléments d'une collection au moyen
306 * d'un énumerateur obtenu par enumerator(). L'énumérateur ainsi obtenu
307 * est générique quel que soit le type de la collection. Il est par
308 * conséquent moins performant qu'un énumérateur dédié à un type et il
309 * vaut mieux utiliser ce dernier si cela est possible.
310 *
311 * Une collection génère des événements lorsque des éléments sont supprimés,
312 * insérés ou modifiés. Il est possible d'enregistrer un handler pour
313 * obtenir ces évènements avec change().
314 *
315 * Les opérations constantes sont threadsafe.
316 *
317 * Cette classe est destinée à être dérivée pour chaque implémentation
318 * d'une collection.
319 *
320 * \sa EnumeratorImpl
321 */
322class CollectionImplBase
323: public ObjectImpl
324{
325 public:
326
327 //! Type indexant le tableau
328 typedef Integer size_type;
329 //! Type d'une distance entre itérateur éléments du tableau
330 typedef ptrdiff_t difference_type;
331
332 public:
333
334 //! Construit une collection vide
335 CollectionImplBase() = default;
336 //! Construit une collection avec \a acount éléments
337 explicit CollectionImplBase(Integer acount)
338 : m_count(acount)
339 {}
340 /*!\brief Opérateur de recopie.
341 * les handlers d'évènements ne sont pas recopiés. */
342 CollectionImplBase(const CollectionImplBase& from) = delete;
343
344 public:
345
346 //! Retourne le nombre d'éléments de la collection
347 Integer count() const { return m_count; }
348 //! Supprime tous les éléments de la collection
349 virtual void clear() = 0;
350
351 public:
352
353 //! Evènement envoyé avant de supprimer tous les éléments
354 virtual void onClear() {}
355 //! Evènement envoyé lorsque tous les éléments ont été supprimés
356 virtual void onClearComplete()
357 {
358 _sendEvent(CollectionEventArgs::ClearComplete, 0, 0);
359 }
360 //! Evènement envoyé avant d'insérer un élément
361 virtual void onInsert() {}
362 //! Evènement envoyé après insertion d'un élément
363 virtual void onInsertComplete(void* object, Integer position)
364 {
365 _sendEvent(CollectionEventArgs::InsertComplete, object, position);
366 }
367 //! Evènement envoyé avant de supprimer un élément
368 virtual void onRemove() {}
369 //! Evènement envoyé après supression d'un élément
370 virtual void onRemoveComplete(void* object, Integer position)
371 {
372 _sendEvent(CollectionEventArgs::RemoveComplete, object, position);
373 }
374 virtual void onSet() {}
375 virtual void onSetComplete(void* object, Integer position)
376 {
377 _sendEvent(CollectionEventArgs::SetComplete, object, position);
378 }
379 virtual void onValidate() {}
380
381 public:
382
383 //! Retourne un énumérateur générique sur la collection.
384 virtual EnumeratorImplBase* enumerator() const = 0;
385
386 public:
387
388 CollectionChangeEventHandler& change() { return m_collection_handlers; }
389
390 protected:
391
392 void _setCount(Integer acount) { m_count = acount; }
393
394 private:
395
396 Integer m_count = 0;
397 CollectionChangeEventHandler m_collection_handlers;
398
399 private:
400
401 void _sendEvent(CollectionEventArgs::eAction action, void* object, Integer position)
402 {
403 CollectionEventArgs args(action, object, position);
404 m_collection_handlers.notify(args);
405 }
406};
