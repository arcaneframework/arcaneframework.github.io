27/*!
28 * \ingroup Collection
29 * \brief Vue pour un tableau 4D.
30 *
31 * Cette classe permet d'obtenir une vue 4D à partir d'une zone contigue
32 * en mémoire, comme par exemple celle obtenue via la classe Array.
33 *
34 * La vue peut s'utiliser comme un tableau C classique, par exemple:
35 * \code
36 * Array4View<Real> a;
37 * a[0][1][2][3] = 5.0;
38 * \endcode
39 *
40 * Il est néammoins préférable d'utiliser directement les méthodes
41 * item() ou setItem() (ou l'opérateur operator()) pour accéder en lecture ou
42 * écriture à un élément du tableau.
43 */
44template<class DataType>
45class Array4View
46{
47 public:
48 //! Construit une vue
49 constexpr Array4View(DataType* ptr,Integer dim1_size,Integer dim2_size,
50 Integer dim3_size,Integer dim4_size)
51 : m_ptr(ptr), m_dim1_size(dim1_size), m_dim2_size(dim2_size), m_dim3_size(dim3_size),
52 m_dim4_size(dim4_size), m_dim34_size(dim3_size*dim4_size),
53 m_dim234_size(m_dim34_size*dim2_size)
54 {
55 }
56 //! Construit une vue vide
57 constexpr Array4View()
58 : m_ptr(0), m_dim1_size(0), m_dim2_size(0), m_dim3_size(0), m_dim4_size(0),
59 m_dim34_size(0), m_dim234_size(0)
60 {
61 }
62 public:
63 //! Valeur de la première dimension
64 constexpr Integer dim1Size() const { return m_dim1_size; }
65 //! Valeur de la deuxième dimension
66 constexpr Integer dim2Size() const { return m_dim2_size; }
67 //! Valeur de la troisième dimension
68 constexpr Integer dim3Size() const { return m_dim3_size; }
69 //! Valeur de la quatrième dimension
70 constexpr Integer dim4Size() const { return m_dim4_size; }
71 //! Nombre total d'éléments du tableau
72 constexpr Integer totalNbElement() const { return m_dim1_size*m_dim234_size; }
73 public:
74 constexpr Array3View<DataType> operator[](Integer i)
75 {
76 ARCCORE_CHECK_AT(i,m_dim1_size);
77 return Array3View<DataType>(m_ptr + (m_dim234_size*i),m_dim2_size,m_dim3_size,m_dim4_size);
78 }
79 constexpr ConstArray3View<DataType> operator[](Integer i) const
80 {
81 ARCCORE_CHECK_AT(i,m_dim1_size);
82 return ConstArray3View<DataType>(m_ptr + (m_dim234_size*i),m_dim2_size,m_dim3_size,m_dim4_size);
83 }
84 //! Valeur pour l'élément \a i,j,k,l
85 constexpr DataType item(Integer i,Integer j,Integer k,Integer l) const
86 {
87 ARCCORE_CHECK_AT4(i,j,k,l,m_dim1_size,m_dim2_size,m_dim3_size,m_dim4_size);
88 return m_ptr[(m_dim234_size*i) + m_dim34_size*j + m_dim4_size*k + l];
89 }
90 //! Valeur pour l'élément \a i,j,k,l
91 constexpr const DataType& operator()(Integer i,Integer j,Integer k,Integer l) const
92 {
93 ARCCORE_CHECK_AT4(i,j,k,l,m_dim1_size,m_dim2_size,m_dim3_size,m_dim4_size);
94 return m_ptr[(m_dim234_size*i) + m_dim34_size*j + m_dim4_size*k + l];
95 }
96 //! Valeur pour l'élément \a i,j,k,l
97 constexpr DataType& operator()(Integer i,Integer j,Integer k,Integer l)
98 {
99 ARCCORE_CHECK_AT4(i,j,k,l,m_dim1_size,m_dim2_size,m_dim3_size,m_dim4_size);
100 return m_ptr[(m_dim234_size*i) + m_dim34_size*j + m_dim4_size*k + l];
101 }
102#ifdef ARCCORE_HAS_MULTI_SUBSCRIPT
103 //! Valeur pour l'élément \a i,j,k,l
104 constexpr const DataType& operator[](Integer i,Integer j,Integer k,Integer l) const
105 {
106 ARCCORE_CHECK_AT4(i,j,k,l,m_dim1_size,m_dim2_size,m_dim3_size,m_dim4_size);
107 return m_ptr[(m_dim234_size*i) + m_dim34_size*j + m_dim4_size*k + l];
108 }
109 //! Valeur pour l'élément \a i,j,k,l
110 constexpr DataType& operator[](Integer i,Integer j,Integer k,Integer l)
111 {
112 ARCCORE_CHECK_AT4(i,j,k,l,m_dim1_size,m_dim2_size,m_dim3_size,m_dim4_size);
113 return m_ptr[(m_dim234_size*i) + m_dim34_size*j + m_dim4_size*k + l];
114 }
115#endif
116 //! Positionne la valeur pour l'élément \a i,j,k,l
117 constexpr void setItem(Integer i,Integer j,Integer k,Integer l,const DataType& value)
118 {
119 ARCCORE_CHECK_AT4(i,j,k,l,m_dim1_size,m_dim2_size,m_dim3_size,m_dim4_size);
120 m_ptr[(m_dim234_size*i) + m_dim34_size*j + m_dim4_size*k + l] = value;
121 }
122 public:
123 /*!
124 * \brief Pointeur sur le premier élément du tableau.
125 */
126 constexpr inline DataType* unguardedBasePointer() { return m_ptr; }
127 /*!
128 * \brief Pointeur sur le premier élément du tableau.
129 */
130 constexpr inline DataType* data() { return m_ptr; }
131 private:
132 DataType* m_ptr;
133 Integer m_dim1_size; //!< Taille de la 1ere dimension
134 Integer m_dim2_size; //!< Taille de la 2eme dimension
135 Integer m_dim3_size; //!< Taille de la 3eme dimension
136 Integer m_dim4_size; //!< Taille de la 4eme dimension
137 Integer m_dim34_size; //!< dim3 * dim4
138 Integer m_dim234_size; //!< dim2 * dim3 * dim4
139};
