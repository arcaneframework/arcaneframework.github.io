87/*!
88 * \brief Classe pour gérer le profiling d'une seule exécution d'une boucle.
89 */
90class ARCCORE_BASE_EXPORT ForLoopOneExecStat
91{
92 public:
93
94 /*!
95 * \brief Incrémente le nombre de chunk utilisé.
96 *
97 * Cette méthode peut être appelée simultanément par plusieurs threads.
98 */
99 void incrementNbChunk() { ++m_nb_chunk; }
100
101 //! Positionne le temps de début de la boucle (en nanoseconde)
102 void setBeginTime(Int64 v) { m_begin_time = v; }
103
104 //! Positionne le temps de fin de la boucle en nanoseconde
105 void setEndTime(Int64 v) { m_end_time = v; }
106
107 //! Nombre de chunks
108 Int64 nbChunk() const { return m_nb_chunk; }
109
110 /*!
111 * \brief Temps d'exécution (en nanoseconde).
112 *
113 * La valeur retournée n'est valide que si setBeginTime() et setEndTime()
114 * ont été appelés avant.
115 */
116 Int64 execTime() const { return m_end_time - m_begin_time; }
117
118 void reset()
119 {
120 m_nb_chunk = 0;
121 m_begin_time = 0;
122 m_end_time = 0;
123 }
124
125 private:
126
127 //! Nombre de chunk de décomposition de la boucle (en multi-thread)
128 std::atomic<Int64> m_nb_chunk = 0;
129
130 // Temps de début d'exécution
131 Int64 m_begin_time = 0;
132
133 // Temps de fin d'exécution
134 Int64 m_end_time = 0;
135};

142 * Il est possible d'activer le profilage en appelant setProfilingLevel() avec

146 * impl::ForLoopStatInfoList spécifique au thread en cours d'exécution.
147 */
148class ARCCORE_BASE_EXPORT ProfilingRegistry
149{
150 public:
151
152 /*!

157 static Impl::ForLoopStatInfoList* threadLocalInstance();
158
159 /*!
160 * \brief Positionne le niveau de profilage.
161 *
162 * Si 0, alors il n'y a pas de profilage. Le profilage est actif à partir
163 * du niveau 1.
164 */
165 static void setProfilingLevel(Int32 level);
166
167 //! Niveau de profilage
168 static Int32 profilingLevel() { return m_profiling_level; }
169
170 //! Indique si le profilage est actif.
171 static bool hasProfiling() { return m_profiling_level > 0; }
172
173 /*!
174 * \brief Visite la liste des statistiques des boucles
175 *
176 * Il y a une instance de impl::ForLoopStatInfoList par thread qui a
177 * exécuté une boucle.
178 *
179 * Cette méthode ne doit pas être appelée s'il y a des boucles en cours d'exécution.
180 */
181 static void visitLoopStat(const std::function<void(const Impl::ForLoopStatInfoList&)>& f);
182
183 /*!

185 *

190 */
191 static void visitAcceleratorStat(const std::function<void(const Impl::AcceleratorStatInfoList&)>& f);
192

195 public:
