60/*!
61 * \brief Vue en lecture, écriture ou lecture/écriture sur un 'Span'.
62 */
63template <typename Accessor>
64class SpanView
65: public SpanViewBase
66{
67 public:
68
69 using DataType = typename Accessor::ValueType;
70 using AccessorReturnType = typename Accessor::AccessorReturnType;
71 using SpanType = Span<DataType>;
72
73 public:
74
75 SpanView(RunCommand& command, SpanType v)
76 : SpanViewBase(command)
77 , m_values(v)
78 {}
79
80 constexpr ARCCORE_HOST_DEVICE AccessorReturnType operator()(Int32 i) const
81 {
82 return Accessor::build(m_values.ptrAt(i));
83 }
84
85 constexpr ARCCORE_HOST_DEVICE AccessorReturnType operator[](Int32 i) const
86 {
87 return Accessor::build(m_values.ptrAt(i));
88 }
89
90 private:
91
92 SpanType m_values;
93};
