121  template <class T>
122  bool setSeed(T value)
123  {
124  if (m_seed.size() == 0) {
125  return false;
126  }
127  memcpy(m_seed.data(), &value, std::min(m_seed.size(), (Integer)sizeof(T)));
128  for (Integer i = sizeof(T); i < m_seed.size(); i++) {
129  m_seed[i] = 0x00;
130  }
131  return true;
132  }

144  template <class T>
145  bool seed(T& value, bool without_size_check = true) const
146  {
147  if (m_seed.size() == 0 || (!without_size_check && sizeof(T) != m_seed.size())) {
148  return false;
149  }
150  value = 0;
151  memcpy(&value, m_seed.data(), std::min(m_seed.size(), (Integer)sizeof(T)));
152  return true;
153  }

165  template <class T>
166  bool seed(T* value, bool without_size_check = true) const
167  {
168  if (m_seed.size() == 0 || (sizeof(T) != m_seed.size() && !without_size_check)) {
169  return false;
170  }
