28/*!
29 * \internal
30 * \brief Interface d'un message de sérialisation entre IMessagePassingMng.
31 *
32 * Un message de sérialisation consiste en une série d'octets envoyés
33 * d'un rang source() à un rang destination().
34 * Si isSend() est vrai, c'est source() qui envoie à destination(),
35 * sinon c'est l'inverse.
36 * S'il s'agit d'un message de réception, le serializer() est alloué
37 * et remplit automatiquement. Pour que le parallélisme fonctionne correctement,
38 * il faut qu'à un message d'envoi corresponde un message de réception
39 * complémentaire (envoyé par le rang destination()).
40 *
41 * Le message peut-être non bloquant. Un message peut-être détruit
42 * lorsque sa propriété finished() est vrai.
43 */
44class ARCCORE_MESSAGEPASSING_EXPORT ISerializeMessage
45{
46 public:
47
48 enum eMessageType
49 {
50 MT_Send,
51 MT_Recv,
52 MT_Broadcast
53 };
54 //! Stratégie d'envoi/réception
55 enum class eStrategy
56 {
57 //! Stratégie par défaut.
58 Default,
59 /*!
60 * \brief Stratégie utilisant un seul message si possible.
61 *
62 * Cela suppose d'utiliser la fonction mpProbe()
63 * pour connaitre la taille du message
64 * avant de poster la réception.
65 */
66 OneMessage
67 };
68
69 virtual ~ISerializeMessage() = default; //!< Libère les ressources.
70
71 public:
72
73 //! \a true s'il faut envoyer, \a false s'il faut recevoir
74 virtual bool isSend() const =0;
75
76 //! Type du message
77 virtual eMessageType messageType() const =0;
78
79 /*!
80 * \brief Rang du destinataire (si isSend() est vrai) ou envoyeur.
81 *
82 * Dans le cas d'une réception, il est possible de spécifier n'importe quel
83 * rang en spécifiant A_NULL_RANK.
84 */
85 ARCCORE_DEPRECATED_2020("Use destination() instead")
86 virtual Int32 destRank() const =0;
87
88 /*!
89 * \brief Rang du destinataire (si isSend() est vrai) ou de l'envoyeur.
90 *
91 * Dans le cas d'une réception, le rang peut valoir nul pour indiquer
92 * qu'on souhaite recevoir de n'importe qui.
93 * rang en spécifiant A_NULL_RANK.
94 */
95 virtual MessageRank destination() const =0;
96
97 /*!
98 * \brief Rang de l'envoyeur du message
99 * Voir aussi destRank() pour une interprétation suivant la valeur de isSend()
100 */
101 ARCCORE_DEPRECATED_2020("Use source() instead")
102 virtual Int32 origRank() const =0;
103
104 /*!
105 * \brief Rang de l'envoyeur du message
106 *
107 * Voir aussi destination() pour une interprétation suivant la valeur de isSend()
108 */
109 virtual MessageRank source() const =0;
110
111 //! Sérialiseur
112 virtual ISerializer* serializer() =0;
113
114 //! \a true si le message est terminé
115 virtual bool finished() const =0;
116
117 /*!
118 * \internal
119 * \brief Positionne l'état 'fini' du message.
120 */
121 virtual void setFinished(bool v) =0;
122
123 /*!
124 * \internal
125 */
126 ARCCORE_DEPRECATED_2020("Use setInternalTag() instead")
127 virtual void setTag(Int32 tag) =0;
128
129 /*!
130 * \internal
131 * \brief Positionne un tag interne pour le message.
132 *
133 * Ce tag est utile s'il faut envoyer/recevoir plusieurs messages
134 * à un même couple origin/destination.
135 *
136 * Cette méthode est interne à %Arccore.
137 */
138 virtual void setInternalTag(MessageTag tag) =0;
139
140 /*!
141 * \internal
142 * \brief Tag interne du message.
143 */
144 ARCCORE_DEPRECATED_2020("Use internalTag() instead")
145 virtual Int32 tag() const =0;
146
147 /*!
148 * \internal
149 * \brief Tag interne du message.
150 */
151 virtual MessageTag internalTag() const =0;
152
153 /*!
154 * \internal
155 * \brief Identificant du message
156 */
157 virtual MessageId _internalMessageId() const =0;
158
159 /*!
160 * \brief Positionne la stratégie d'envoi/réception.
161 *
162 * La stratégie utilisée doit être la même pour le message envoyé
163 * et le message de réception sinon le comportement est indéfini.
164 */
165 virtual void setStrategy(eStrategy strategy) =0;
166
167 //! Stratégie utilisée pour les envois/réceptions
168 virtual eStrategy strategy() const =0;
169
170 /*!
171 * \brief Indique si le message a déjà été traité.
172 *
173 * Si le message a déjà été traité, il n'est pas possible de changer certaines
174 * caractéristiques (comme la stratégie ou le tag)
175 */
176 virtual bool isProcessed() const =0;
177};
178

182} // End namespace Arccore::MessagePassing
