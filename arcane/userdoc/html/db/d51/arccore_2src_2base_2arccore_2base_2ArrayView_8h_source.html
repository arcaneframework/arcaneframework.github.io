39/*!
40 * \ingroup Collection
41 * \brief Vue modifiable d'un tableau d'un type \a T.
42 *
43 Cette classe template permet d'accéder et d'utiliser un tableau d'éléments du
44 type \a T de la même manière qu'un tableau C standard. Elle maintient en
45 plus la taille du tableau. La fonction size() permet de connaître le nombre
46 d'éléments du tableau et l'opérateur operator[]() permet d'accéder à un élément donné.
47
48 Il est garanti que tous les éléments de la vue sont consécutifs en mémoire.
49
50 Cette classe ne gère aucune mémoire et c'est le conteneur associé qui la gère.
51 Les conteneurs possibles fournis par %Arccore sont les classes Array,
52 UniqueArray ou SharedArray. Une vue n'est valide que tant que le conteneur associé n'est pas réalloué.
53 De même, le contructeur et l'opérateur de recopie ne font que recopier les pointeurs
54 sans réallouer la mémoire. Il faut donc les utiliser avec précaution.
55
56 Si %Arccore est compilé en mode vérification (ARCCORE_CHECK est défini), les accès
57 par l'intermédiaire de l'opérateurs operator[]() sont vérifiés et une
58 exception IndexOutOfRangeException est lancé si un débordement de
59 tableau a lieu. En attachant une session de debug au processus, il est
60 possible de voir la pile d'appel au moment du débordement.
61
62 Voici des exemples d'utilisation:
63
64 \code
65 Real t[5];
66 ArrayView<Real> a(t,5); // Gère un tableau de 5 réels.
67 Integer i = 3;
68 Real v = a[2]; // Affecte à la valeur du 2ème élément
69 a[i] = 5; // Affecte au 3ème élément la valeur 5
70 \endcode
71
72 Il est aussi possible d'accéder aux éléments du tableau par des
73 itérateurs de la même manière qu'avec les containers de la STL.
74
75 L'exemple suivant créé un itérateur \e i sur le tableau \e a et itère
76 sur tout le tableau (méthode i()) et affiche les éléments:
77
78 \code
79 * for( Real v : a )
80 * cout << v;
81 \endcode
82
83 L'exemple suivant fait la somme des 3 premiers éléments du tableau:
84
85 \code
86 * Real sum = 0.0;
87 * for( Real v : a.subView(0,3) )
88 * sum += v;
89 \endcode
90
91*/

369 /*!
370 * \brief Recopie le tableau \a copy_array dans l'instance.
371 *
372 * Comme aucune allocation mémoire n'est effectuée, le
373 * nombre d'éléments de \a copy_array doit être inférieur ou égal au
374 * nombre d'éléments courant. S'il est inférieur, les éléments du
375 * tableau courant situés à la fin du tableau sont inchangés
376 */
377 template<class U>
378 void copy(const U& copy_array)
379 {
380 auto copy_size = copy_array.size();
381 const_pointer copy_begin = copy_array.data();
382 pointer to_ptr = m_ptr;
383 Integer n = m_size;
384 if (copy_size<m_size)
385 n = (Integer)copy_size;
386 for( Integer i=0; i<n; ++i )
387 to_ptr[i] = copy_begin[i];
388 }
389
390 //! Retourne \a true si le tableau est vide (dimension nulle)
391 constexpr bool empty() const noexcept { return m_size==0; }
392 //! \a true si le tableau contient l'élément de valeur \a v
393 bool contains(const_reference v) const
394 {
395 for( Integer i=0; i<m_size; ++i ){
396 if (m_ptr[i]==v)
397 return true;
398 }
399 return false;
400 }
