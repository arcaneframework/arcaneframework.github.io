145 template<typename ItemType, typename Accessor>
146 class ItemVariableScalarOutViewT
147 : public VariableViewBase
148 {
149  public:
150 
151  using DataType = typename Accessor::ValueType;
152  using DataTypeReturnReference = DataType&;
153  using ItemIndexType = typename ItemTraitsT<ItemType>::LocalIdType;
154 
155  public:
156 
157  ItemVariableScalarOutViewT(IVariable* var,Span<DataType> v)
158  : VariableViewBase(var), m_values(v.data()),
159  m_size(v.size()){}
160 
162  SimdSetter<DataType> operator[](SimdItemIndexT<ItemType> simd_item) const
163  {
164  return SimdSetter<DataType>(m_values,simd_item.simdLocalIds());
165  }
166 
168  SimdDirectSetter<DataType> operator[](SimdItemDirectIndexT<ItemType> simd_item) const
169  {
170  return SimdDirectSetter<DataType>(m_values+simd_item.baseLocalId());
171  }
172 
174  Accessor operator[](ItemIndexType i) const
175  {
176  ARCANE_CHECK_AT(i.localId(),m_size);
177  return this->m_values + i.localId();
178  }
179 
181  Accessor value(ItemIndexType i) const
182  {
183  ARCANE_CHECK_AT(i.localId(),m_size);
184  return this->m_values + i.localId();
185  }
186 
188  void setValue(ItemIndexType i,const DataType& v) const
189  {
190  ARCANE_CHECK_AT(i.localId(),m_size);
191  this->m_values[i.localId()] = v;
192  }
193 
194  private:
195  DataType* m_values;
196  Int64 m_size;
197 };
