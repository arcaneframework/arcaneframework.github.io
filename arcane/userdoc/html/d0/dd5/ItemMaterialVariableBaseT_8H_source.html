82 template<typename Traits> void
83 ItemMaterialVariableBase<Traits>::
84 _init(ArrayView<PrivatePartType*> vars)
85 {
86  if (!m_vars.empty())
87  ARCANE_FATAL("Internal error: _init() already called");
88  m_vars.addRange(vars);
89  Integer nb_var = m_vars.size();
90  m_views.resize(nb_var);
91  // Il faut maintenir une référence sur les variables qu'on possède pour
92  // être sur qu'elles sont bien initialisées et qu'elles ne seront pas détruites
93  // tant que cette instance existe.
94  m_p->m_refs.resize(nb_var);
95  for( Integer i=0; i<nb_var; ++i ){
96  // m_vars[i] peut être nul si notre variable est une variable uniquement milieu.
97  m_p->m_refs[i] = nullptr;
98  m_views[i] = ContainerViewType();
99  if (i==0){
100  // Pour la variable globale, il s'agit d'une variable maillage.
101  m_p->m_refs[i] = m_global_variable_ref;
102  }
103  else{
104  // Pour les autres, il s'agit d'une variable tableau.
105  if (m_vars[i])
106  m_p->m_refs[i] = new VariableRefType(m_vars[i]);
107  }
108  // TODO: il faut pouvoir être notifié du changement de cette variable.
109  // pour ensuite mettre à jour les vues (peut-être aussi faire un MyVariableRef).
110  if (m_vars[i])
111  m_views[i] = m_vars[i]->valueView();
112  //std::cout << "INIT_VIEW: " << i << " v=" << m_views[i].size() << '\n';
113  }
114 }

119 template<typename Traits> void
120 ItemMaterialVariableBase<Traits>::
121 buildFromManager(bool is_continue)
122 {
123  IMeshMaterialMng* mat_mng = m_p->materialMng();
124 
125  // Il faut récupérer (ou créér) les autres variables.
126  //ConstArrayView<IMeshMaterial*> materials = mat_mng->materials();
127  ConstArrayView<MeshMaterialVariableIndexer*> indexers = mat_mng->variablesIndexer();
128  Integer nb_indexer = indexers.size();
129  UniqueArray<PrivatePartType*> all_vars(nb_indexer+1);
130  all_vars[0] = m_global_variable;
131  IMesh* mesh = m_global_variable->mesh();
132 
133  // Il faut attacher un observer sur la variable globale pour savoir quand elle change.
134  // Pour les variables partielles, cela se fait automatiquement lorsque le matériau associé change.
135  m_p->m_global_variable_changed_observer = new ObserverT<ThatClass>(this,&ThatClass::_syncFromGlobalVariable);
136  m_global_variable->onSizeChangedObservable()->attachObserver(m_p->m_global_variable_changed_observer);
137 
138  m_p->m_modified_times.resize(nb_indexer);
139  m_p->m_modified_times.fill(0);
140 
141  bool is_env_only = m_p->space()==MatVarSpace::Environment;
142  //TODO: regarder s'il faut stocker les propriétés de cette variable
143  // avant sa création
144  int property = m_global_variable->property();
145  for( Integer i=0; i<nb_indexer; ++i ){
146  MeshMaterialVariableIndexer* indexer = indexers[i];
147  // Ne fait rien si on est uniquement une variable milieu et que l'indexeur
148  // n'est pas associé à un milieu.
149  if (is_env_only && !indexer->isEnvironment()){
150  all_vars[i+1] = nullptr;
151  continue;
152  }
153  // Prend le nom spécifié pour le nom de la variable matériau s'il existe
154  String var_name = indexer->wantedVariableName();
155  bool is_compat = false;
156  // Si un nom est spécifié, il s'agit du mode de compatibilité
157  // avec Troll et dans ce cas la variable n'est pas associée
158  // au maillage.
159  if (var_name.null())
160  var_name = String("Mat")+indexer->name()+"_"+this->name();
161  else{
162  var_name = var_name + "_" + this->name();
163  is_compat = true;
164  }
165 
166 
167  //Note: Ces variables ne sont pas associées au maillage.
168  // Il s'agit juste de variables tableau.
169  IVariable* var = 0;
170  if (is_compat){
171  VariableBuildInfo vbi2(mesh->variableMng(),var_name,property);
172  VariableInfo vi = VariableRefType::_internalVariableInfo(vbi2);
173  var = PrivatePartType::getReference(vbi2,vi);
174  }
175  else{
176  VariableBuildInfo vbi2(mesh,var_name,property);
177  VariableInfo vi = VariableRefType::_internalVariableInfo(vbi2);
178  var = PrivatePartType::getReference(vbi2,vi);
179  }
180  // Tag la variable pour les codes qui voudraient savoir qu'il
181  // s'agit d'une variable matériau.
182  var->addTag("Material","1");
183  PrivatePartType* true_ptr = dynamic_cast<PrivatePartType*>(var);
184  ARCANE_CHECK_POINTER(true_ptr);
185  all_vars[i+1] = true_ptr;
186  if (!is_continue){
187  // TODO: regarder si setUsed() est nécessaire.
188  true_ptr->setUsed(true);
189  // Dimensionne la variable à la taille du tableau des valeurs mixtes.
190  Traits::resizeWithReserve(true_ptr,indexer->maxIndexInMultipleArray());
191  }
192  }
193  this->_init(all_vars);
194 }

199 template<typename Traits> void
200 ItemMaterialVariableBase<Traits>::
201 _syncFromGlobalVariable()
202 {
203  // Cette méthode est appelée lorsque la taille de la variable globale
204  // change. Cette variable est la première variable de la liste.
205  // Il faut ensuite notifier toutes les références de ce changement.
206  m_views[0] = m_vars[0]->valueView();
207 
208  for( MeshMaterialVariableRef::Enumerator i(this); i.hasNext(); ++i ){
209  MeshMaterialVariableRef* ref = *i;
210  ref->updateFromInternal();
211  }

217 template<typename Traits> void
218 ItemMaterialVariableBase<Traits>::
219 syncReferences()
220 {
221  ConstArrayView<MeshMaterialVariableIndexer*> indexers = m_p->materialMng()->variablesIndexer();
222  Integer nb_indexer = indexers.size();
223  for( Integer i=0; i<nb_indexer; ++i ){
224  MeshMaterialVariableIndexer* indexer = indexers[i];
225  PrivatePartType* true_ptr = m_vars[i+1];
226  if (true_ptr)
227  Traits::resizeWithReserve(true_ptr,indexer->maxIndexInMultipleArray());
228  }
229 
230  Integer nb_var = m_vars.size();
231  for( Integer i=0; i<nb_var; ++i ){
232  m_views[i] = ContainerViewType();
233  if (m_vars[i])
234  m_views[i] = m_vars[i]->valueView();
235  }
236 
237  for( MeshMaterialVariableRef::Enumerator i(this); i.hasNext(); ++i ){
238  MeshMaterialVariableRef* ref = *i;
239  ref->updateFromInternal();
240  }

274 template<typename Traits> void
275 ItemMaterialVariableBase<Traits>::
276 _restoreData(IMeshComponent* component,IData* data,Integer data_index,
277  Int32ConstArrayView ids,bool allow_null_id)
278 {
279  if (m_p->space()==MatVarSpace::Environment && !component->isEnvironment())
280  ARCANE_FATAL("Can not restore data on material for environment only variable");
281 
282  auto* true_data = dynamic_cast<ValueDataType*>(data);
283  ARCANE_CHECK_POINTER(true_data);
284  auto values = true_data->view();
285 
286  PrivatePartType* global_var = this->_trueGlobalVariable();
287 
288  // Tableau pour initialiser les valeurs soit à zéro, soit avec
289  // la valeur de la variable globale.
290  UniqueContainerType tmp_values;
291  if (component->materialMng()->isDataInitialisationWithZero()){
292  Integer max_id = m_global_variable->itemFamily()->maxLocalId();
293  Traits::resizeAndFillWithDefault(true_data,tmp_values,max_id);
294  }
295  else
296  tmp_values = global_var->constValueView();
297 
298  if (allow_null_id){
299  for( Integer i=0, n=ids.size(); i<n; ++i ){
300  if (ids[i]!=NULL_ITEM_ID){
301  Traits::setValue(tmp_values[ids[i]],values[data_index+i]);
302  }
303  }
304  }
305  else{
306  for( Integer i=0, n=ids.size(); i<n; ++i ){
307  Traits::setValue(tmp_values[ids[i]] , values[data_index+i] );
308  }
309  }
310 
311  ENUMERATE_COMPONENTCELL(icell,component){
312  ComponentCell ec = *icell;
313  setValue(ec._varIndex(),tmp_values[ec.globalCell().localId()]);
314  }
315 }

336 _copyGlobalToPartial(Int32 var_index,Int32ConstArrayView local_ids,
337  Int32ConstArrayView indexes_in_multiple)

377 template<typename Traits> void
378 ItemMaterialVariableBase<Traits>::
379 _initializeNewItems(const ComponentItemListBuilder& list_builder)
380 {
381  MeshMaterialVariableIndexer* indexer = list_builder.indexer();
382  Integer var_index = indexer->index();
383  PrivatePartType* partial_var = m_vars[var_index+1];
384  if (!_isValidAndUsedAndGlobalUsed(partial_var))
385  return;
386 
387  // Redimensionne le tableau des valeurs multiples si besoin.
388  Traits::resizeWithReserve(partial_var,indexer->maxIndexInMultipleArray());
389  m_views[var_index+1] = partial_var->valueView();
390 
391  bool init_with_zero = m_p->materialMng()->isDataInitialisationWithZero();
392 
393  ContainerConstViewType global_view = m_vars[0]->valueView();
394 
395  ConstArrayView<MatVarIndex> partial_matvar = list_builder.partialMatVarIndexes();
396 
397  Integer nb_partial = partial_matvar.size();
398 
399  // TODO: regarder s'il faut initialiser les mailles pure avec zéro.
400  // En effet, normalement il n'y a pas de maille pure si on ajoute un matériau
401  // (sauf lors de l'init).
402  DataType zero = DataType();
403 
404  if (init_with_zero){
405  for( Integer i=0; i<nb_partial; ++i )
406  this->setFillValue(partial_matvar[i],zero);
407  }
408  else{
409  Int32ConstArrayView partial_local_ids = list_builder.partialLocalIds();
410  for( Integer i=0; i<nb_partial; ++i ){
411  setValue(partial_matvar[i], global_view[partial_local_ids[i]]);
412  }
413  }
414 }

478 template<typename VariableTrueType> IMeshMaterialVariable*
479 MeshMaterialVariableCommonStaticImpl<VariableTrueType>::
480 getReference(const MaterialVariableBuildInfo& v,MatVarSpace mvs)
481 {
482  typedef typename VariableTrueType::PrivatePartType PrivatePartType;
483  typedef typename VariableTrueType::VariableRefType VariableRefType;
484 
485  MeshHandle mesh_handle = v.meshHandle();
486  if (mesh_handle.isNull())
487  ARCANE_FATAL("No mesh handle for material variable");
488 
489  IMeshMaterialMng* mat_mng = v.materialMng();
490 
491  //TODO: regarder si verrou necessaire
492  if (!mat_mng)
493  mat_mng = IMeshMaterialMng::getReference(mesh_handle,true);
494 
495  if (!mat_mng)
496  ARCANE_FATAL("No IMaterialMng for mesh");
497 
498  if (!v.itemFamilyName().null())
499  ARCANE_FATAL("Can not declare material variable for specific family");
500 
501  if (!v.itemGroupName().null())
502  ARCANE_FATAL("Can not declare partial material variable");
503 
504  // Regarde si l'option qui force la création des variables matériaux
505  // même si on est une variable milieu est active.
506 
507  if (VariableTrueType::TraitsType::dimension()==0)
508  if (mat_mng->isAllocateScalarEnvironmentVariableAsMaterial())
509  mvs = MatVarSpace::MaterialAndEnvironment;
510 
511  VariableBuildInfo gvar_bi(v);
512 
513  {
514  Mutex::ScopedLock sl(mat_mng->variableLock());
515 
517  VariableInfo gvar_vi = VariableRefType::_internalVariableInfo(gvar_bi);
518  IVariable* g_var = PrivatePartType::getReference(gvar_bi,gvar_vi);
519 
520  PrivatePartType* true_ptr = dynamic_cast<PrivatePartType*>(g_var);
521 
522  // TODO: regarder s'il faut mettre aussi le nom du m_material_mng.
523  // Si on le fait, cela pose un problème avec la variable globale
524  // associée. Si on ne le fait pas, on ne peut à priori n'avoir
525  // qu'un seul IMeshMaterialMng.
526  IMeshMaterialVariable* mat_var = mat_mng->checkVariable(g_var);
527  VariableTrueType* true_mat_var = 0;
528  // \a mat_var est non nul si la variable associée existe déjà.
529  if (mat_var)
530  true_mat_var = dynamic_cast<VariableTrueType*>(mat_var);
531  else{
532  VariableRefType* var_ref = new VariableRefType(true_ptr);
533  MaterialVariableBuildInfo mvbi(mat_mng,v);
534  true_mat_var = new VariableTrueType(mvbi,true_ptr,var_ref,mvs);
535  mat_mng->addVariable(true_mat_var);
536  // Le gestionnaire est déjà créé, on peut allouer toutes les infos
537  if (!mat_mng->variablesIndexer().empty())
538  true_mat_var->buildFromManager(false);
539  }
540  ARCANE_CHECK_POINTER(true_mat_var);
541  if (mvs!=true_mat_var->space())
542  ARCANE_FATAL("Incoherent space for variable '{0}' wanted_space={1} existing_space={2}.\n"
543  " Check all instances of the variables are of the same space\n"
544  " (i.e. all variables are MaterialVariable or EnvironmentVariable)",
545  true_mat_var->name(),(int)mvs,(int)true_mat_var->space());
546  // Incrémente le compteur de référence.
547  // Il faut le faire ici et pas dans addVariableRef() car en multithreading
548  // il est possible de créer plusieurs références sur la même variable en même
549  // temps et de détruire une des références avant que le addVariableRef() de l'autre
550  // référence ne soit appelé ce qui provoque la destruction de true_mat_var.
551  ++true_mat_var->m_p->m_nb_reference;
552  return true_mat_var;
553  }
554 }
