77 /*!
78 * \brief Effectue le partitionnement d'une liste en deux parties.
79 */
80 template <typename SelectLambda, typename InputIterator, typename OutputIterator>
81 void apply(GenericPartitionerBase& s, Int32 nb_item, InputIterator input_iter, OutputIterator output_iter,
82 const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
83 {
84 RunQueue queue = s.m_queue;
85 eExecutionPolicy exec_policy = queue.executionPolicy();
86 RunCommand command = makeCommand(queue);
87 command << trace_info;
88 Impl::RunCommandLaunchInfo launch_info(command, nb_item);
89 launch_info.beginExecute();
90 switch (exec_policy) {
91#if defined(ARCCORE_COMPILING_CUDA)
92 case eExecutionPolicy::CUDA: {
93 size_t temp_storage_size = 0;
94 cudaStream_t stream = Impl::CudaUtils::toNativeStream(&queue);
95 // Premier appel pour connaitre la taille pour l'allocation
96 int* nb_list1_ptr = nullptr;
97 ARCCORE_CHECK_CUDA(::cub::DevicePartition::If(nullptr, temp_storage_size,
98 input_iter, output_iter, nb_list1_ptr, nb_item,
99 select_lambda, stream));
100
101 s.m_algo_storage.allocate(temp_storage_size);
102 nb_list1_ptr = s.m_device_nb_list1_storage.allocate();
103 ARCCORE_CHECK_CUDA(::cub::DevicePartition::If(s.m_algo_storage.address(), temp_storage_size,
104 input_iter, output_iter, nb_list1_ptr, nb_item,
105 select_lambda, stream));
106 s.m_device_nb_list1_storage.copyToAsync(s.m_host_nb_list1_storage, queue);
107 } break;
108#endif
109#if defined(ARCCORE_COMPILING_HIP)
110 case eExecutionPolicy::HIP: {
111 size_t temp_storage_size = 0;
112 // Premier appel pour connaitre la taille pour l'allocation
113 hipStream_t stream = Impl::HipUtils::toNativeStream(&queue);
114 int* nb_list1_ptr = nullptr;
115 ARCCORE_CHECK_HIP(rocprim::partition(nullptr, temp_storage_size, input_iter, output_iter,
116 nb_list1_ptr, nb_item, select_lambda, stream));
117
118 s.m_algo_storage.allocate(temp_storage_size);
119 nb_list1_ptr = s.m_device_nb_list1_storage.allocate();
120
121 ARCCORE_CHECK_HIP(rocprim::partition(s.m_algo_storage.address(), temp_storage_size, input_iter, output_iter,
122 nb_list1_ptr, nb_item, select_lambda, stream));
123 s.m_device_nb_list1_storage.copyToAsync(s.m_host_nb_list1_storage, queue);
124 } break;
125#endif
126#if defined(ARCCORE_COMPILING_SYCL)
127 case eExecutionPolicy::SYCL: {
128#if defined(ARCCORE_HAS_ONEDPL)
129 // Seulement implémenté pour DPC++.
130 // Actuellement (dpc++ 2025.0), il n'y a pas l'équivalent avec SYCL de
131 // la méthode de partition de cub ou rocprim.
132 // Utilise la fonction 'stable_partition'. Cependant, cette fonction
133 // ne supporte pas si la mémoire uniquement sur accélérateur. De plus,
134 // il faut que InputIterator et OutputIterator remplissent le concept
135 // std::random_access_iterator ce qui n'est pas le cas (notamment car il n'y a
136 // pas les opérateurs de copie ni de constructeur vide à cause de la lambda).
137 // Pour éviter tous ces problèmes, on alloue donc des tableaux temporaires
138 // pour l'appel à 'stable_sort' et on recopie les valeurs en sortie.
139 using InputDataType = typename InputIterator::value_type;
140 using DataType = typename OutputIterator::value_type;
141 NumArray<DataType, MDDim1> tmp_output_numarray(nb_item);
142 NumArray<bool, MDDim1> tmp_select_numarray(nb_item);
143 auto tmp_output = tmp_output_numarray.to1DSmallSpan();
144 auto tmp_select = tmp_select_numarray.to1DSmallSpan();
145 {
146 auto command = makeCommand(queue);
147 command << RUNCOMMAND_LOOP1(iter, nb_item)
148 {
149 auto [i] = iter();
150 tmp_output[i] = input_iter[i];
151 };
152 }
153 auto tmp_select_lambda = [=](Int32 i) { return tmp_select[i]; };
154 sycl::queue sycl_queue = Impl::SyclUtils::toNativeStream(queue);
155 auto policy = oneapi::dpl::execution::make_device_policy(sycl_queue);
156 auto output_after = oneapi::dpl::stable_partition(policy, tmp_output.begin(), tmp_output.end(), select_lambda);
157 queue.barrier();
158 Int32 nb_list1 = (output_after - tmp_output.begin());
159 Int32 nb_list2 = nb_item - nb_list1;
160 s.m_host_nb_list1_storage[0] = nb_list1;
161 //std::cerr << "NbList1=" << nb_list1 << " NbList2=" << nb_list2 << "\n";
162 {
163 // Recopie dans output les valeurs filtrées.
164 // Pour être cohérent avec 'cub' et 'rocprim', il faut inverser l'ordre des
165 // des valeurs de la liste pour les éléments ne remplissant pas la condition.
166 // Pour cela, on fait une boucle de taille (nb_list1 + nb_list/2) et chaque
167 // itération pour i>=nb_list1 gère deux éléments.
168 auto command = makeCommand(queue);
169 Int32 nb_iter2 = (nb_list2 / 2) + (nb_list2 % 2);
170 //std::cout << "NB_ITER2=" << nb_iter2 << "\n";
171 command << RUNCOMMAND_LOOP1(iter, (nb_list1 + nb_iter2))
172 {
173 auto [i] = iter();
174 if (i >= nb_list1) {
175 // Partie de la liste pour les valeurs ne remplissant par le critère.
176 Int32 j = i - nb_list1;
177 Int32 reverse_i = (nb_item - (j + 1));
178 auto x1 = tmp_output[i];
179 auto x2 = tmp_output[reverse_i];
180 output_iter[i] = tmp_output[reverse_i];
181 output_iter[reverse_i] = tmp_output[i];
182 }
183 else
184 output_iter[i] = tmp_output[i];
185 };
186 }
187 queue.barrier();
188#else // ARCCORE_HAS_ONEDPL
189 ARCCORE_THROW(NotImplementedException, "Partition is only implemented for SYCL back-end using oneDPL");
190#endif // ARCCORE_HAS_ONEDPL
191 } break;
192#endif // ARCCORE_COMPILING_SYCL
193 case eExecutionPolicy::Thread:
194 // Pas encore implémenté en multi-thread
195 [[fallthrough]];
196 case eExecutionPolicy::Sequential: {
197 auto saved_output_iter = output_iter;
198 auto output2_iter = output_iter + nb_item;
199 for (Int32 i = 0; i < nb_item; ++i) {
200 auto v = *input_iter;
201 if (select_lambda(v)) {
202 *output_iter = v;
203 ++output_iter;
204 }
205 else {
206 --output2_iter;
207 *output2_iter = v;
208 }
209 ++input_iter;
210 }
211 Int32 nb_list1 = static_cast<Int32>(output_iter - saved_output_iter);
212 s.m_host_nb_list1_storage[0] = nb_list1;
213 } break;
214 default:
215 ARCCORE_FATAL(getBadPolicyMessage(exec_policy));
216 }
217 launch_info.endExecute();
218 }

220 /*!
221 * \brief Effectue le partitionnement d'une liste en trois parties.
222 */
223 template <typename Select1Lambda, typename Select2Lambda,
224 typename InputIterator, typename FirstOutputIterator,
225 typename SecondOutputIterator, typename UnselectedIterator>
226 void apply3(GenericPartitionerBase& s, Int32 nb_item,
227 InputIterator input_iter,
228 FirstOutputIterator first_output_iter,
229 SecondOutputIterator second_output_iter,
230 UnselectedIterator unselected_iter,
231 const Select1Lambda& select1_lambda,
232 const Select2Lambda& select2_lambda,
233 const TraceInfo& trace_info = TraceInfo())
234 {
235 RunQueue queue = s.m_queue;
236 eExecutionPolicy exec_policy = queue.executionPolicy();
237 RunCommand command = makeCommand(queue);
238 command << trace_info;
239 Impl::RunCommandLaunchInfo launch_info(command, nb_item);
240 launch_info.beginExecute();
241 switch (exec_policy) {
242#if defined(ARCCORE_COMPILING_CUDA)
243 case eExecutionPolicy::CUDA: {
244 size_t temp_storage_size = 0;
245 cudaStream_t stream = Impl::CudaUtils::toNativeStream(&queue);
246 // Premier appel pour connaitre la taille pour l'allocation
247 int* nb_list1_ptr = nullptr;
248 ARCCORE_CHECK_CUDA(::cub::DevicePartition::If(nullptr, temp_storage_size,
249 input_iter, first_output_iter, second_output_iter,
250 unselected_iter, nb_list1_ptr, nb_item,
251 select1_lambda, select2_lambda, stream));
252
253 s.m_algo_storage.allocate(temp_storage_size);
254 nb_list1_ptr = s.m_device_nb_list1_storage.allocate();
255 ARCCORE_CHECK_CUDA(::cub::DevicePartition::If(s.m_algo_storage.address(), temp_storage_size,
256 input_iter, first_output_iter, second_output_iter,
257 unselected_iter, nb_list1_ptr, nb_item,
258 select1_lambda, select2_lambda, stream));
259 s.m_device_nb_list1_storage.copyToAsync(s.m_host_nb_list1_storage, queue);
260 } break;
261#endif
262#if defined(ARCCORE_COMPILING_HIP)
263 case eExecutionPolicy::HIP: {
264 size_t temp_storage_size = 0;
265 // Premier appel pour connaitre la taille pour l'allocation
266 hipStream_t stream = Impl::HipUtils::toNativeStream(&queue);
267 int* nb_list1_ptr = nullptr;
268 using namespace rocprim;
269 ARCCORE_CHECK_HIP(::rocprim::partition_three_way(nullptr, temp_storage_size, input_iter, first_output_iter,
270 second_output_iter, unselected_iter,
271 nb_list1_ptr, nb_item, select1_lambda, select2_lambda, stream));
272
273 s.m_algo_storage.allocate(temp_storage_size);
274 nb_list1_ptr = s.m_device_nb_list1_storage.allocate();
275
276 ARCCORE_CHECK_HIP(partition_three_way(s.m_algo_storage.address(), temp_storage_size, input_iter, first_output_iter,
277 second_output_iter, unselected_iter, nb_list1_ptr, nb_item,
278 select1_lambda, select2_lambda, stream));
279 s.m_device_nb_list1_storage.copyToAsync(s.m_host_nb_list1_storage, queue);
280 } break;
281#endif
282#if defined(ARCCORE_COMPILING_SYCL)
283 case eExecutionPolicy::SYCL: {
284 ARCCORE_THROW(NotImplementedException, "3-way partition is not implemented for SYCL back-end");
285 } break;
286#endif
287 case eExecutionPolicy::Thread:
288 // Pas encore implémenté en multi-thread
289 [[fallthrough]];
290 case eExecutionPolicy::Sequential: {
291 Int32 nb_first = 0;
292 Int32 nb_second = 0;
293 for (Int32 i = 0; i < nb_item; ++i) {
294 auto v = *input_iter;
295 bool is_1 = select1_lambda(v);
296 bool is_2 = select2_lambda(v);
297 if (is_1) {
298 *first_output_iter = v;
299 ++first_output_iter;
300 ++nb_first;
301 }
302 else {
303 if (is_2) {
304 *second_output_iter = v;
305 ++second_output_iter;
306 ++nb_second;
307 }
308 else {
309 *unselected_iter = v;
310 ++unselected_iter;
311 }
312 }
313 // Incrémenter l'itérateur à la fin car il est utilisé pour le positionnement
314 ++input_iter;
315 }
316 s.m_host_nb_list1_storage[0] = nb_first;
317 s.m_host_nb_list1_storage[1] = nb_second;
318 } break;
319 default:
320 ARCCORE_FATAL(getBadPolicyMessage(exec_policy));
321 }
322 launch_info.endExecute();
323 }
324};

336/*!
337 * \brief Algorithme générique de partitionnement d'une liste.
338 *
339 * Cette classe fournit des algorithmes pour partitionner une liste en deux
340 * ou trois parties selon un critère fixé par l'utilisateur.
341 */
342class GenericPartitioner
343: private Impl::GenericPartitionerBase
344{
345 public:
346
347 explicit GenericPartitioner(const RunQueue& queue)
348 : Impl::GenericPartitionerBase(queue)
349 {
350 _allocate();
351 }
352
353 public:
354
355 /*!
356 * \brief Effectue un partitionnement d'une liste en deux parties.
357 *
358 * Le nombre de valeurs de la liste est donné par \a nb_value.
359 * Les deux fonctions lambda \a select_lambda et \a setter_lambda permettent
360 * de partitionner et de positionner les valeurs de la liste.
361 *
362 * Après exécution, il est possible de récupérer le nombre d'éléments
363 * de la première partie de la liste via la méthode \a nbFirstPart().
364 *
365 * \snippet AcceleratorPartitionerUnitTest.cc SampleListPartitionerTwoWayIndex
366 */
367 template <typename SelectLambda, typename SetterLambda>
368 void applyWithIndex(Int32 nb_value, const SetterLambda& setter_lambda,
369 const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
370 {
371 if (_checkEmpty(nb_value))
372 return;
373 _setCalled();
374 Impl::GenericPartitionerBase* base_ptr = this;
375 Impl::GenericPartitionerIf gf;
376
377 Impl::IndexIterator input_iter;
378 Impl::SetterLambdaIterator<SetterLambda> out(setter_lambda);
379 gf.apply(*base_ptr, nb_value, input_iter, out, select_lambda, trace_info);
380 }
381
382 /*!
383 * \brief Effectue un partitionnement d'une liste en deux parties.
384 *
385 * Le nombre de valeurs de la liste est donné par \a nb_value.
386 * Les valeurs en entrée sont fournies par l'itérateur \a input_iter et
387 * les valeurs en sorties par l'itérateur \a output_iterator. La fonction
388 * lambda \a select_lambda permet de sélectionner la partition utilisée :
389 * si le retour est \a true, la valeur sera dans la première partie de la liste,
390 * sinon elle sera dans la seconde. En sortie les valeurs de la deuxième
391 * partie sont rangées en ordre inverse de la liste d'entrée.
392 *
393 * Après exécution, il est possible de récupérer le nombre d'éléments
394 * de la première partie de la liste via la méthode \a nbFirstPart().
395 *
396 * \snippet AcceleratorPartitionerUnitTest.cc SampleListPartitionerTwoWayIf
397 */
398 template <typename InputIterator, typename OutputIterator, typename SelectLambda>
399 void applyIf(Int32 nb_value, InputIterator input_iter, OutputIterator output_iter,
400 const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
401 {
402 if (_checkEmpty(nb_value))
403 return;
404 _setCalled();
405 Impl::GenericPartitionerBase* base_ptr = this;
406 Impl::GenericPartitionerIf gf;
407 gf.apply(*base_ptr, nb_value, input_iter, output_iter, select_lambda, trace_info);
408 }
409
410 /*!
411 * \brief Effectue un partitionnement d'une liste en trois parties.
412 *
413 * Le nombre de valeurs de la liste est donné par \a nb_value.
414 * Les deux fonctions lambda \a select1_lambda et \a select2_lambda permettent
415 * de partitionner la liste avec l'algorithme suivant:
416 * - si select1_lambda() est vrai, la valeur sera positionnée via \a setter1_lambda,
417 * - sinon si select2_lambda() est vrai, la valeur sera positionnée via \a setter2_lambda,
418 * - sinon la valeur sera positionnée via \a unselected_setter_lambda.
419 *
420 * Les listes en sortie sont dans le même ordre qu'en entrée.
421 *
422 * Après exécution, il est possible de récupérer le nombre d'éléments
423 * de la première et de la deuxième liste la méthode \a nbParts().
424 *
425 * \snippet AcceleratorPartitionerUnitTest.cc SampleListPartitionerThreeWayIndex
426 */
427 template <typename Setter1Lambda, typename Setter2Lambda, typename UnselectedSetterLambda,
428 typename Select1Lambda, typename Select2Lambda>
429 void applyWithIndex(Int32 nb_value,
430 const Setter1Lambda setter1_lambda,
431 const Setter2Lambda setter2_lambda,
432 const UnselectedSetterLambda& unselected_setter_lambda,
433 const Select1Lambda& select1_lambda,
434 const Select2Lambda& select2_lambda,
435 const TraceInfo& trace_info = TraceInfo())
436 {
437 if (_checkEmpty(nb_value))
438 return;
439 _setCalled();
440 Impl::GenericPartitionerBase* base_ptr = this;
441 Impl::GenericPartitionerIf gf;
442 Impl::IndexIterator input_iter;
443 Impl::SetterLambdaIterator<Setter1Lambda> setter1_wrapper(setter1_lambda);
444 Impl::SetterLambdaIterator<Setter2Lambda> setter2_wrapper(setter2_lambda);
445 Impl::SetterLambdaIterator<UnselectedSetterLambda> unselected_setter_wrapper(unselected_setter_lambda);
446 gf.apply3(*base_ptr, nb_value, input_iter, setter1_wrapper, setter2_wrapper,
447 unselected_setter_wrapper, select1_lambda, select2_lambda, trace_info);
448 }
449
450 /*!
451 * \brief Effectue un partitionnement d'une liste en trois parties.
452 *
453 * Le nombre de valeurs de la liste est donné par \a nb_value.
454 * Les deux fonctions lambda \a select1_lambda et \a select2_lambda permettent
455 * de partitionner la liste avec l'algorithme suivant:
456 * - si select1_lambda() est vrai, la valeur ajoutée \a first_output_iter,
457 * - sinon si select2_lambda() est vrai, la valeur sera ajoutée à \a second_output_iter,
458 * - sinon la valeur sera ajoutée à \a unselected_iter.
459 *
460 * Les listes en sortie sont dans le même ordre qu'en entrée.
461 *
462 * Après exécution, il est possible de récupérer le nombre d'éléments
463 * de la première et de la deuxième liste la méthode \a nbParts().
464 *
465 * \snippet AcceleratorPartitionerUnitTest.cc SampleListPartitionerThreeWayIf
466 */
467 template <typename InputIterator, typename FirstOutputIterator,
468 typename SecondOutputIterator, typename UnselectedIterator,
469 typename Select1Lambda, typename Select2Lambda>
470 void applyIf(Int32 nb_value, InputIterator input_iter,
471 FirstOutputIterator first_output_iter,
472 SecondOutputIterator second_output_iter,
473 UnselectedIterator unselected_iter,
474 const Select1Lambda& select1_lambda,
475 const Select2Lambda& select2_lambda,
476 const TraceInfo& trace_info = TraceInfo())
477 {
478 if (_checkEmpty(nb_value))
479 return;
480 _setCalled();
481 Impl::GenericPartitionerBase* base_ptr = this;
482 Impl::GenericPartitionerIf gf;
483 gf.apply3(*base_ptr, nb_value, input_iter, first_output_iter, second_output_iter,
484 unselected_iter, select1_lambda, select2_lambda, trace_info);
485 }
486
487 /*!
488 * \brief Nombre d'éléments de la première partie de la liste.
489 */
490 Int32 nbFirstPart()
491 {
492 m_is_already_called = false;
493 return _nbFirstPart();
494 }
495
496 /*!
497 * \brief Nombre d'éléments de la première et deuxième partie de la liste.
498 *
499 * Retourne une vue de deux valeurs. La première valeur contient le nombre
500 * d'éléments de la première liste et la seconde valeur le
501 * nombre d'éléments de la deuxième liste.
502 *
503 * Cette méthode n'est valide qu'après avoir appelé une méthode de partitionnement
504 * en trois parties.
505 */
506 SmallSpan<const Int32> nbParts()
507 {
508 m_is_already_called = false;
509 return _nbParts();
510 }
511
512 private:
513
514 bool m_is_already_called = false;
515
516 private:
517
518 void _setCalled()
519 {
520 if (m_is_already_called)
521 ARCCORE_FATAL("apply() has already been called for this instance");
522 m_is_already_called = true;
523 }
524 bool _checkEmpty(Int32 nb_value)
525 {
526 if (nb_value == 0) {
527 _resetNbPart();
528 return true;
529 }
530 return false;
531 }
532};
