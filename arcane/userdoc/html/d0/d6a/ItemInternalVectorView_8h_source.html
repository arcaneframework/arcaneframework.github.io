36 class ItemInternalVectorViewConstIterator
37 {
38  friend class ItemInternalVectorView;
39  typedef ItemInternal* ItemInternalPtr;
40  ItemInternalVectorViewConstIterator(const ItemInternalPtr* items,
41  const Int32* ARCANE_RESTRICT local_ids,
42  Integer index)
43  : m_items(items), m_local_ids(local_ids), m_index(index){}
44  public:
45  // Pas directement utilisé mais est nécessaire pour ICC 17.0
46  ItemInternalVectorViewConstIterator()
47  : m_items(nullptr), m_local_ids(nullptr), m_index(0){}
48  public:
49  typedef ItemInternalVectorViewConstIterator ThatClass;
50  public:
51  typedef std::random_access_iterator_tag iterator_category;
53  typedef const ItemInternalPtr* pointer;
55  typedef const ItemInternalPtr& reference;
57  typedef ItemInternalPtr value_type;
59  typedef Integer size_type;
61  typedef Integer difference_type;
62  public:
63  value_type operator*() const { return m_items[ m_local_ids[m_index] ]; }
64  value_type operator->() const { return m_items[ m_local_ids[m_index] ]; }
65  ThatClass& operator++() { ++m_index; return (*this); }
66  ThatClass& operator--() { --m_index; return (*this); }
67  void operator+=(difference_type v) { m_index += v; }
68  void operator-=(difference_type v) { m_index -= v; }
69  friend Integer operator-(const ThatClass& a,const ThatClass& b)
70  {
71  return a.m_index - b.m_index;
72  }
73  friend ThatClass operator-(const ThatClass& a,difference_type v)
74  {
75  Integer index = a.m_index - v;
76  return ThatClass(a.m_items,a.m_local_ids,index);
77  }
78  friend ThatClass operator+(const ThatClass& a,difference_type v)
79  {
80  Integer index = a.m_index + v;
81  return ThatClass(a.m_items,a.m_local_ids,index);
82  }
83  friend bool operator<(const ThatClass& lhs,const ThatClass& rhs)
84  {
85  return lhs.m_index<=rhs.m_index;
86  }
87  friend bool operator==(const ThatClass& lhs,const ThatClass& rhs)
88  {
89  // TODO: regarder si cela ne pose pas de problemes de performance
90  // de faire ces trois comparaions. Si c'est le cas on peut
91  // faire uniquement la dernière.
92  return lhs.m_items==rhs.m_items && lhs.m_local_ids==rhs.m_local_ids && lhs.m_index==rhs.m_index;
93  }
94  friend bool operator!=(const ThatClass& lhs,const ThatClass& rhs)
95  {
96  return !(lhs==rhs);
97  }
98  private:
99  const ItemInternalPtr* m_items;
100  const Int32* ARCANE_RESTRICT m_local_ids;
101  Integer m_index;
102 };
