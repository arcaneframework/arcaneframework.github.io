50void MatCellVector::
51_build(CellVectorView view)
52{
53 FixedArray<UniqueArray<ConstituentItemIndex>,2> internals;
54 FixedArray<UniqueArray<MatVarIndex>,2> matvar_indexes;
55 FixedArray<UniqueArray<Int32>,2> local_ids;
56 IMeshComponent* my_component = _component();
57
58 ENUMERATE_ALLENVCELL(iallenvcell,_materialMng()->view(view)){
59 AllEnvCell all_env_cell = *iallenvcell;
60 ENUMERATE_CELL_ENVCELL(ienvcell,all_env_cell){
61 ENUMERATE_CELL_MATCELL(imatcell,(*ienvcell)){
62 MatCell mc = *imatcell;
63 if (mc.component()==my_component){
64 MatVarIndex idx = mc._varIndex();
65 if (idx.arrayIndex()==0){
66 internals[0].add(mc._constituentItemIndex());
67 matvar_indexes[0].add(idx);
68 local_ids[0].add(mc.globalCell().localId());
69 }
70 else{
71 internals[1].add(mc._constituentItemIndex());
72 matvar_indexes[1].add(idx);
73 local_ids[1].add(mc.globalCell().localId());
74 }
75 }
76 }
77 }
78 }
79 this->_setItems(internals[0],internals[1]);
80 this->_setMatVarIndexes(matvar_indexes[0],matvar_indexes[1]);
81 this->_setLocalIds(local_ids[0],local_ids[1]);
82}
