47/*!
48 * \internal
49 * \brief Conserve une liste de taille fixe maximale de StackFrame.
50 */
51class FixedStackFrameArray
52{
53 public:
54 static const int MAX_FRAME = 32;
55 FixedStackFrameArray() : m_nb_frame(0){}
56 public:
57 ConstArrayView<StackFrame> view() const
58 {
59 return ConstArrayView<StackFrame>(m_nb_frame,m_addresses);
60 }
61 /*!
62 * \brief Ajoute \a frame à la liste des frames. Si nbFrame() est supérieur
63 * ou égal à MAX_FRAME, aucune opération n'est effectuée.
64 */
65 void addFrame(const StackFrame& frame)
66 {
67 if (m_nb_frame<MAX_FRAME){
68 m_addresses[m_nb_frame] = frame;
69 ++m_nb_frame;
70 }
71 }
72 Integer nbFrame() const { return m_nb_frame; }
73 private:
74 //! Liste des adresses de la pile d'appel. Stocke au plus 32 appels.
75 StackFrame m_addresses[MAX_FRAME];
76 Integer m_nb_frame;
77};
