30/*!
31 * \brief Implémentation d'un buffer de sérialisation contigu en mémoire.
32 *
33 * Cette implémentation permet de sérialiser les données dans une zone
34 * contigue en mémoire et ainsi de l'envoyer en une seule fois
35 * via MPI par exemple.
36 * Le buffer est composé d'un première partie contenant NB_SIZE_ELEM
37 * objets de type Integer. Cette partie contient le nombre d'élément
38 * sérialisé de chaque type (Real, Int64, Int32, Int16, Byte). La deuxième
39 * partie contient les données sérialisées proprement dites.
40 *
41 * Lorsque cette classe est utilisée dans le cadre d'un appel MPI,
42 * on utilise le tableau \a m_size_copy_buffer pour envoyer les tailles.
43 * lors du premier message lorsque le message complet est envoyé en plusieurs fois.
44 * La norme MPI indique effectivement qu'un buffer utilisé lors d'un appel MPI ne doit plus être
45 * utilisé tant que cet appel n'est pas terminé.
46 */

365 SizeInfo getPaddingSize(Int64 nb_elem,Int64 elem_size)
366 {
367 if (nb_elem<0)
368 ARCCORE_FATAL("Bad number of element '{0}' (should be >=0)",nb_elem);
369 if (elem_size<=0)
370 ARCCORE_FATAL("Bad elem_size '{0}'",elem_size);
371 Int64 s = nb_elem * elem_size;
372 Int64 pad = s % ALIGN_SIZE;
373 if (pad==0)
374 pad = ALIGN_SIZE;
375 Int64 new_size = s + (ALIGN_SIZE-pad);
376 if ( (new_size%ALIGN_SIZE)!=0 )
377 ARCCORE_FATAL("Bad padding {0}",new_size);
378 //std::cout << " nb_elem=" << nb_elem << " elem_size=" << elem_size << " s=" << s << " new_size=" << new_size << '\n';
379 return { s, new_size };
380 }

382 /*!
383 * \brief Remplit avec une valeur fixe les zones correspondantes au padding.
384 * Cela permet d'éviter d'avoir des valeurs non initialisées.
385 *
386 * Il faut avoir appeler _allocBuffer() avant
387 */
388 void _fillPadding(Int64 position,SizeInfo size_info)
389 {
390 Int64 begin = position + size_info.m_original_size;
391 Int64 end = position + size_info.m_padded_size;
392 _fillPadding(m_buffer_view.subspan(begin,end-begin));
393 }
394
395 void _fillPadding(Span<Byte> buf)
396 {
397 // Utilise une valeur non nulle pour repérer plus facilement
398 // les zones de padding si besoin.
399 constexpr Byte v = (Byte)(250);
400 for( Int64 i=0, s=buf.size(); i<s; ++i )
401 buf[i] = v;
402 }

447
448 void _allocBuffer(Int64 size)
449 {
450 if (size<1024)
451 size = 1024;
452 m_buffer.resize(size+ALIGN_SIZE*4);
453 Int64 addr = (Int64)(&m_buffer[0]);
454 Int64 padding = addr % ALIGN_SIZE;
455 Int64 position = 0;
456 if (padding!=0){
457 position = ALIGN_SIZE - padding;
458 }
459 // La taille doit être un multiple de ALIGN_SIZE;
460 Int64 new_size = (size + ALIGN_SIZE) - (size%ALIGN_SIZE);
461 m_buffer_view = m_buffer.span().subspan(position,new_size);
462
463 // Initialise les valeurs de la zone tampon
464 auto padding_view = m_buffer.span().subspan(position+size,new_size-size);
465 _fillPadding(padding_view);
466 }

495void BasicSerializer::Impl2::
496reserve(eDataType dt,Int64 n,Int64 nb_put)
497{
498 ARCCORE_ASSERT((m_mode==ModeReserve),("Bad mode"));
499 switch(dt){
500 case DT_Real: m_real.m_reserved_size += n; break;
501 case DT_Int64: m_int64.m_reserved_size += n; break;
502 case DT_Int32: m_int32.m_reserved_size += n; break;
503 case DT_Int16: m_int16.m_reserved_size += n; break;
504 case DT_Byte: m_byte.m_reserved_size += n; break;
505 case DT_Int8: m_int8.m_reserved_size += n; break;
506 case DT_Float16: m_float16.m_reserved_size += n; break;
507 case DT_BFloat16: m_bfloat16.m_reserved_size += n; break;
508 case DT_Float32: m_float32.m_reserved_size += n; break;
509 default:
510 ARCCORE_THROW(ArgumentException,"bad datatype v={0}",(int)dt);
511 }
512 if (m_is_serialize_typeinfo)
513 // Pour le type de la donnée.
514 m_byte.m_reserved_size += nb_put;
515}
