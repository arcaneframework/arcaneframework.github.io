332  void _resize(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
333  {
334  if (new_size2==m_md->dim2_size){
335  _resize(new_size1,rb);
336  }
337  else if (totalNbElement()==0){
338  _resizeFromEmpty(new_size1,new_size2,rb);
339  }
340  else if (new_size2<m_md->dim2_size){
341  _resizeSameDim1ReduceDim2(new_size2,rb);
342  _resize(new_size1,rb);
343  }
344  else if (new_size2>m_md->dim2_size){
345  _resizeSameDim1IncreaseDim2(new_size2,rb);
346  _resize(new_size1,rb);
347  }
348  else
349  throw NotImplementedException("Array2::resize","already sized");
350  }
351  void _resizeFromEmpty(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
352  {
353  _resize2(new_size1,new_size2,rb);
354  m_md->dim1_size = new_size1;
355  m_md->dim2_size = new_size2;
356  _arccoreCheckSharedNull();
357  }
358  void _resizeSameDim1ReduceDim2(Int64 new_size2,InitBehaviour rb)
359  {
360  ARCCORE_ASSERT((new_size2<m_md->dim2_size),("Bad Size"));
361  Int64 n = m_md->dim1_size;
362  Int64 n2 = m_md->dim2_size;
363  for( Int64 i=0; i<n; ++i ){
364  for( Int64 j=0; j<new_size2; ++j )
365  m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
366  }
367  _resize2(n,new_size2,rb);
368  m_md->dim2_size = new_size2;
369  _arccoreCheckSharedNull();
370  }
371  void _resizeSameDim1IncreaseDim2(Int64 new_size2,InitBehaviour rb)
372  {
373  ARCCORE_ASSERT((new_size2>m_md->dim2_size),("Bad Size"));
374  Int64 n = m_md->dim1_size;
375  Int64 n2 = m_md->dim2_size;
376  _resize2(n,new_size2,rb);
377  // Recopie en partant de la fin pour éviter tout écrasement
378  for( Int64 i=(n-1); i>=0; --i ){
379  for( Int64 j=(n2-1); j>=0; --j )
380  m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
381  }
382  m_md->dim2_size = new_size2;
383  _arccoreCheckSharedNull();
384  }
385  void _resize2(Int64 d1,Int64 d2,InitBehaviour rb)
386  {
387  Int64 new_size = d1*d2;
388  // Si la nouvelle taille est nulle, il faut tout de meme faire une allocation
389  // pour stocker les valeurs dim1_size et dim2_size (sinon, elle seraient
390  // dans TrueImpl::shared_null
391  if (new_size==0)
392  this->_reserve(4);
393  if (rb==IB_InitWithDefault)
394  Base::_resize(new_size,DataType());
395  else if (rb==IB_NoInit)
396  Base::_resize(new_size);
397  else
398  throw NotSupportedException("Array2::_resize2","invalid value InitBehaviour");
399  }
