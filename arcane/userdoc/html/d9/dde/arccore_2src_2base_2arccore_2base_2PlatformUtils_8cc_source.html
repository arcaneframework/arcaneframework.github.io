273{
274 can_create = true;
275 struct stat dirstat;

277 int stat_val = ::stat(dirname,&dirstat);
278 if (stat_val==0){
279 if (dirstat.st_mode & S_IFDIR)

282 can_create = false;
283 return false;
284 }
285 return false;

292extern "C++" ARCCORE_BASE_EXPORT bool Platform::
293recursiveCreateDirectory(const String& dir_name)
294{
295 //cerr << "** REC CRE DIR " << dir_name << '\n';
296 bool can_create = false;
297 bool is_dir = isDirectoryExist(dir_name,can_create);
298 if (is_dir)
299 return false;
300 if (!can_create)
301 return true;
302 Int64 pos = 0;
303 const char* dir_name_str = dir_name.localstr();
304 for( Int64 i=0, is=dir_name.length(); i<is; ++i )
305 if (dir_name_str[i]=='/')
306 pos = i;
307 if (pos!=0){
308 String parent_dir(std::string_view(dir_name_str,pos));
309 bool is_bad = recursiveCreateDirectory(parent_dir);
310 if (is_bad)
311 return true;
312 }
313 //cerr << "** REAL DIR " << dir_name << '\n';
314#ifdef ARCCORE_OS_WIN32
315 int ret = mkdir(dir_name_str);
316#else
317 int ret = mkdir(dir_name_str,S_IRWXU|S_IRWXG);
318#endif
319 if (ret==(-1))
320 return true;
321 return false;
322}

353extern "C++" ARCCORE_BASE_EXPORT String Platform::
354getFileDirName(const String& file_name)
355{
356 // Sous windows, regarde s'il y a des '/'. Dans ce cas on prend ce caractère comme
357 // séparateur. Sinon, on prend '\\'.
358 char separator = '/';
359 const char* file_name_str = file_name.localstr();
360#if defined(ARCCORE_OS_WIN32)
361 {
362 bool has_slash = false;
363 for( Int64 i=0, n=file_name.length(); i<n; ++i )
364 if (file_name_str[i]=='/'){
365 has_slash = true;
366 break;
367 }
368 if (!has_slash)
369 separator = '\\';
370 }
371#endif
372 Int64 pos = 0;
373 for( Int64 i=0, n=file_name.length(); i<n; ++i )
374 if (file_name_str[i]==separator)
375 pos = i;
376
377 if (pos==0)
378 return String(".");
379
380 return std::string_view(file_name_str,pos);
381}

494#ifdef ARCCORE_OS_LINUX
495static long _getRSSMemoryLinux()
496{
497 FILE* f = ::fopen("/proc/self/stat","r");
498 if (!f)
499 return 0;
500
501 // See proc(1), category 'stat'
502 int z_pid;
503 char z_comm[4096];
504 char z_state;
505 int z_ppid;
506 int z_pgrp;
507 int z_session;
508 int z_tty_nr;
509 int z_tpgid;
510 unsigned long z_flags;
511 unsigned long z_minflt;
512 unsigned long z_cminflt;
513 unsigned long z_majflt;
514 unsigned long z_cmajflt;
515 unsigned long z_utime;
516 unsigned long z_stime;
517 long z_cutime;
518 long z_cstime;
519 long z_priority;
520 long z_nice;
521 long z_zero;
522 long z_itrealvalue;
523 long z_starttime;
524 unsigned long z_vsize;
525 long z_rss;
526 int r = fscanf(f,"%d %s %c %d %d %d %d %d"
527 "%lu %lu %lu %lu %lu %lu %lu"
528 "%ld %ld %ld %ld %ld %ld %ld %lu %ld",
529 &z_pid,z_comm,&z_state,&z_ppid,&z_pgrp,&z_session,&z_tty_nr,&z_tpgid,
530 &z_flags,&z_minflt,&z_cminflt,&z_majflt,&z_cmajflt,&z_utime,&z_stime,
531 &z_cutime,&z_cstime,&z_priority,&z_nice,&z_zero,&z_itrealvalue,
532 &z_starttime,&z_vsize,&z_rss);
533 if (r!=24)
534 z_rss = 0;
535
536 ::fclose(f);
537 return z_rss;
538}
539#endif

562extern "C++" ARCCORE_BASE_EXPORT Int64 Platform::
563getCPUTime()
564{
565#if defined(ARCCORE_OS_LINUX)
566 struct timespec ts;
567 if (clock_gettime(CLOCK_PROCESS_CPUTIME_ID, &ts) == 0){
568 return (ts.tv_sec*1000000) + (ts.tv_nsec / 1000);
569 }
570 return 1;
571#else
572 // TODO Supprimer l'usage de clock() pour Win32 et MacOS.
573 static Int64 orig_clock = 0;
574 static clock_t call_clock = 0;
575 clock_t current_clock = ::clock();
576 if (orig_clock==0 && call_clock==0){
577 call_clock = current_clock;
578 }
579 else{
580 if (current_clock<call_clock){
581 // On a depasse la valeur max de clock
582 //cout << " WARNING: CLOCK depasse INT_MAX: " << call_clock
583 // << " current=" << current_clock << '\n';
584 Int64 v = ARCCORE_INT64_MAX;
585 v *= 2;
586 orig_clock += v;
587 }
588 call_clock = current_clock;
589 }
590 //cout << " INFO: CLOCK orig=" << orig_clock << " call=" << call_clock
591 // << " current=" << current_clock << '\n';
592 return orig_clock + call_clock;
593#endif
594}

612extern "C++" ARCCORE_BASE_EXPORT String Platform::
613timeToHourMinuteSecond(Real t)
614{
615 Int64 s = (Int64)(t);
616 Int64 hour = s / 3600;
617 Int64 remaining_hour = s - (3600 * hour);
618 Int64 minute = remaining_hour / 60;
619 Int64 second = remaining_hour - (60 * minute);
620
621 StringBuilder sb;
622 sb += hour;
623 sb += "h";
624 sb += minute;
625 sb += "m";
626 sb += second;
627 sb += "s";
628
629 return sb.toString();
630}

758bool _getHasColorTerminal()
759{
760#ifdef ARCCORE_OS_UNIX
761 String force_color = Platform::getEnvironmentVariable("ARCCORE_COLORTERM");
762 if (!force_color.null())
763 return true;
764
765 if (!::isatty(::fileno(stdout)))
766 return false;
767
768 String term = Platform::getEnvironmentVariable("TERM");
769 //std::cout << "CHECK ENV term=" << term << '\n';
770
771 if (term=="xterm-color" || term=="xterm-256color")
772 return true;
773
774 if (term=="xterm" || term=="rxvt" || term=="rxvt-unicode"){
775 String color_term = Platform::getEnvironmentVariable("COLORTERM");
776 //std::cout << "CHECK ENV color_term=" << color_term << '\n';
777 if (!color_term.null())
778 return true;
779 }
780#endif
781 return false;
782}

787void Platform::
788platformInitialize(bool enable_fpe)
789{
790 // Pour l'instant, la seule initialisation spécifique dépend
791 // des processeurs i386. Elle consiste à changer la valeur par
792 // défaut des flags de la FPU pour générer une exception
793 // lors d'une erreur arithmétique (comme les divisions par zéro).
794 if (enable_fpe)
795 enableFloatingException(true);
796
797 getCPUTime();
798
799 global_has_color_console = _getHasColorTerminal();
800
801 if (getEnvironmentVariable("ARCCORE_PAUSE_ON_EXCEPTION")=="1")
802 arccoreSetPauseOnException(true);
803 if (getEnvironmentVariable("ARCCORE_PRINT_ON_EXCEPTION")=="1")
804 arccoreCallExplainInExceptionConstructor(true);
805}

838getLoadedSharedLibraryFullPath(const String& dll_name)
839{
840 String full_path;
841 if (dll_name.null())
842 return full_path;
843#if defined(ARCCORE_OS_LINUX)
844 {
845 std::ifstream ifile("/proc/self/maps");
846 String v;
847 String true_name = "lib" + dll_name + ".so";
848 while (ifile.good()){
849 ifile >> v;
850 Span<const Byte> vb = v.bytes();
851 if (vb.size()>0 && vb[0]=='/'){
852 if (v.endsWith(true_name)){
853 full_path = v;
854 //std::cout << "V='" << v << "'\n";
855 break;
856 }
857 }
858 }
859 }
860#elif defined(ARCCORE_OS_WIN32)
861 HMODULE hModule = GetModuleHandleA(dll_name.localstr());
862 if (!hModule)
863 return full_path;
864 TCHAR dllPath[_MAX_PATH];
865 GetModuleFileName(hModule, dllPath, _MAX_PATH);
866 full_path = StringView(dllPath);
867#elif defined(ARCCORE_OS_MACOS)
868 {
869 String true_name = "lib" + dll_name + ".dylib";
870 uint32_t count = _dyld_image_count();
871 for (uint32_t i = 0; i < count; i++) {
872 const char* image_name = _dyld_get_image_name(i);
873 if (image_name) {
874 String image_path(image_name);
875 if (image_path.endsWith(true_name)) {
876 full_path = image_path;
877 break;
878 }
879 }
880 }
881 }
882#else
883 throw NotSupportedException(A_FUNCINFO);
884//#error "platform::getSymbolFullPath() not implemented for this platform"
885#endif
886 return full_path;
887}
