33/*!
34 * \internal
35 * \brief Tableau avec allocateur virtuel.
36 *
37 * C'est à la classe virtuelle de détruire les objets dans le
38 * destructeur virtuel.
39 */
40template <class T>
41class ListImplBase
42: public CollectionImplT<T>
43{
44 public:
45
46 typedef CollectionImplT<T> BaseClass;
47
48 public:
49
50 typedef const T& ObjectRef;
51
52 //! Type des éléments du tableau
53 typedef T value_type;
54 //! Type de l'itérateur sur un élément du tableau
55 typedef value_type* iterator;
56 //! Type de l'itérateur constant sur un élément du tableau
57 typedef const value_type* const_iterator;
58 //! Type pointeur d'un élément du tableau
59 typedef value_type* pointer;
60 //! Type pointeur constant d'un élément du tableau
61 typedef const value_type* const_pointer;
62 //! Type référence d'un élément du tableau
63 typedef value_type& reference;
64 //! Type référence constante d'un élément du tableau
65 typedef const value_type& const_reference;
66
67 //! Type d'un itérateur sur tout le tableau
68 typedef IterT<ListImplBase<T>> iter;
69 //! Type d'un itérateur constant sur tout le tableau
70 typedef ConstIterT<ListImplBase<T>> const_iter;
71
72 public:
73
74 //! Construit un tableau vide.
75 ListImplBase() = default;
76
77 public:
78
79 //! Recopie le tableau \a s.
80 void assign(const ListImplBase<T>& s)
81 {
82 _arrayCopy(s);
83 }
84 //! Recopie le tableau \a s.
85 void assign(const ConstArrayView<T>& s)
86 {
87 _arrayCopy(s);
88 }
89 //! Recopie le tableau \a s.
90 void assign(const ArrayView<T>& s)
91 {
92 _arrayCopy(s);
93 }
94
95 public:
96
97 /*!
98 * \brief i-ème élément du tableau.
99 *
100 * En mode \a check, vérifie les débordements.
101 */
102 T& operator[](Integer i)
103 {
104 return m_array[i];
105 }
106
107 /*!
108 * \brief i-ème élément du tableau.
109 *
110 * En mode \a check, vérifie les débordements.
111 */
112 const T& operator[](Integer i) const
113 {
114 return m_array[i];
115 }
116
117 //! Retourne un iterateur sur le premier élément du tableau
118 iterator begin() override
119 {
120 return m_array.data();
121 }
122 //! Retourne un iterateur sur le premier élément après la fin du tableau
123 iterator end() override
124 {
125 return m_array.data() + this->count();
126 }
127 //! Retourne un iterateur constant sur le premier élément du tableau
128 const_iterator begin() const override
129 {
130 return m_array.data();
131 }
132 //! Retourne un iterateur constant sur le premier élément après la fin du tableau
133 const_iterator end() const override
134 {
135 return m_array.data() + this->count();
136 }
137
138 //! Retourne un iterateur sur le premier élément du tableau
139 T* begin2() const override
140 {
141 return const_cast<T*>(m_array.data());
142 }
143
144 //! Retourne un iterateur sur le premier élément après la fin du tableau
145 T* end2() const override
146 {
147 return begin2() + this->count();
148 }
149
150 public:
151
152 //! Applique le fonctor \a f à tous les éléments du tableau
153 template <class Function> Function
154 each(Function f)
155 {
156 std::for_each(begin(), end(), f);
157 return f;
158 }
159
160 public:
161
162 /*! \brief Signale qu'il faut réserver de la mémoire pour \a new_capacity éléments
163 * Il s'agit juste d'une indication. La classe dérivée est libre de ne
164 * pas en tenir compte.
165 */
166 void reserve(Integer new_capacity)
167 {
168 m_array.reserve(new_capacity);
169 }
170
171 /*!
172 * \brief Retourne le nombre d'éléments alloués du tableau.
173 *
174 * Il s'agit juste d'une indication. La classe dérivée est libre de ne
175 * pas en tenir compte.
176 */
177 Integer capacity() const
178 {
179 return m_array.capacity();
180 }
181
182 void clear() override
183 {
184 this->onClear();
185 m_array.clear();
186 this->_setCount(0);
187 this->onClearComplete();
188 }
189
190 //! Ajoute l'élément \a elem à la fin du tableau
191 void add(ObjectRef elem) override
192 {
193 this->onInsert();
194 Integer s = this->count();
195 m_array.add(elem);
196 this->_setCount(s + 1);
197 this->onInsertComplete(_ptr() + s, s);
198 }
199
200 bool remove(ObjectRef element) override
201 {
202 Integer i = 0;
203 Integer s = this->count();
204 for (; i < s; ++i)
205 if (m_array[i] == element) {
206 _removeAt(i);
207 return true;
208 }
209 throwOutOfRangeException();
210 return false;
211 }
212
213 void removeAt(Integer index) override
214 {
215 Integer s = this->count();
216 if (index >= s)
217 throwOutOfRangeException();
218 _removeAt(index);
219 }
220
221 const_iterator find(ObjectRef element) const
222 {
223 Integer s = this->count();
224 for (Int32 i = 0; i < s; ++i)
225 if (m_array[i] == element)
226 return begin() + i;
227 return end();
228 }
229
230 bool contains(ObjectRef element) const override
231 {
232 const_iterator i = find(element);
233 return (i != end());
234 }
235
236 EnumeratorImplBase* enumerator() const override;
237
238 protected:
239
240 void _removeAt(Integer index)
241 {
242 Integer s = this->count();
243 T* ptr = _ptr();
244 T* remove_ob = ptr + index;
245 this->onRemove();
246 m_array.remove(index);
247 this->_setCount(s - 1);
248 // TODO: supprimer utilisation de 'remove_ob' car ce n'est pas le bon objet
249 this->onRemoveComplete(remove_ob, index);
250 }
251
252 public:
253
254 /*!\brief Modifie la taille du tableau.
255 * \a new_size est le nouveau nombre d'éléments du tableau.
256 */
257 void resize(Integer new_size)
258 {
259 m_array.resize(new_size);
260 this->_setCount(new_size);
261 }
262
263 protected:
264
265 Integer _capacity() const
266 {
267 return m_array.capacity();
268 }
269
270 protected:
271
272 void _arrayCopy(const ListImplBase<T>& array)
273 {
274 _arrayCopy(array.begin2(), array.count());
275 }
276 void _arrayCopy(const ConstArrayView<T>& array)
277 {
278 _arrayCopy(array.data(), array.size());
279 }
280 void _arrayCopy(const ArrayView<T>& array)
281 {
282 _arrayCopy(array.data(), array.size());
283 }
284 void _arrayCopy(const T* from_ptr, Integer from_size)
285 {
286 if (from_size == 0) {
287 clear();
288 return;
289 }
290 m_array.copy(ConstArrayView<T>(from_size, from_ptr));
291 this->_setCount(from_size);
292 }
293
294 protected:
295
296 /*!
297 * \brief Retourne un pointeur sur le tableau
298 *
299 * \warning Il est préférable de ne pas utiliser cette méthode pour
300 * accéder à un élément du tableau car
301 * ce pointeur peut être invalidé par un redimensionnement du tableau.
302 * De plus, accéder aux éléments du tableau par ce pointeur ne permet
303 * aucune vérification de dépassement, même en mode DEBUG.
304 */
305 inline T* _ptr()
306 {
307 return m_array.data();
308 }
309
310 inline const T* _ptr() const
311 {
312 return m_array.data();
313 }
314
315 private:
316
317 UniqueArray<T> m_array;
318
319 private:
320};

621/*!
622 * \brief Implémentation d'une collection d'éléments sous forme de vecteur.

624template <typename T>
625class List
626: public Collection<T>
627{
628 public:
629
630 typedef Collection<T> BaseClass;
631
632 private:
633
634 typedef ListImplT<T> Impl;
635
636 public:
637
638 typedef Impl Vec;
639 typedef typename Vec::value_type value_type;
640 typedef typename Vec::iterator iterator;
641 typedef typename Vec::const_iterator const_iterator;
642 typedef typename Vec::pointer pointer;
643 typedef typename Vec::const_pointer const_pointer;
644 typedef typename Vec::reference reference;
645 typedef typename Vec::const_reference const_reference;
646
647 //! Type d'un itérateur constant sur tout le tableau
648 typedef ListEnumeratorT<T> Enumerator;
649
650 typedef ConstIterT<List<T>> const_iter;
651
652 public:
653
654 List()
655 : BaseClass(new Impl())
656 {}
657 List(const ConstArrayView<T>& from)
658 : BaseClass(new Impl(from))
659 {}
660 List(const ArrayView<T>& from)
661 : BaseClass(new Impl(from))
662 {}
663 explicit List(const EnumeratorT<T>& from)
664 : BaseClass(new Impl(from))
665 {}
666
667 public:
668
669 void resize(Integer new_size)
670 {
671 _cast().resize(new_size);
672 }
673
674 T& operator[](Integer i)
675 {
676 return _cast()[i];
677 }
678
679 const T& operator[](Integer i) const
680 {
681 return _cast()[i];
682 }
683
684 //! Clone la collection \a base
685 void clone(const Collection<T>& base)
686 {
687 //Impl* n = new Impl(base);
688 //BaseClass::_setRef(n);
689 _cast().assign(base);
690 }
691
692 //! Clone la collection
693 List<T> clone() const
694 {
695 return List<T>(new Impl(*this));
696 }
697
698 public:
699
700 const_iterator begin() const { return _cast().begin(); }
701 const_iterator end() const { return _cast().end(); }
702 iterator begin() { return _cast().begin(); }
703 iterator end() { return _cast().end(); }
704
705 T* begin2() const { return _cast().begin2(); }
706 T* end2() const { return _cast().end2(); }
707
708 //! Applique le fonctor \a f à tous les éléments du tableau
709 template <typename Function> Function
710 each(Function f)
711 {
712 return _cast().each(f);
713 }
714
715 template <typename Function> iterator
716 find_if(Function f)
717 {
718 iterator _end = end2();
719 iterator i = std::find_if(begin2(), _end, f);
720 if (i != _end)
721 return i;
722 return _end;
723 }
724
725 template <typename Function> const_iterator
726 find_if(Function f) const
727 {
728 const_iterator _end = end();
729 const_iterator i = std::find_if(begin(), _end, f);
730 if (i != _end)
731 return i;
732 return _end;
733 }
734
735 protected:
736
737 List(Impl* from)
738 : BaseClass(from)
739 {}
740
741 private:
742
743 Impl& _cast() { return *static_cast<Impl*>(this->_noNullRef()); }
744 const Impl& _cast() const { return *static_cast<const Impl*>(this->_noNullRef()); }
745};
