27/*!
28 * \internal
29 * \brief Dictionnaire de chaînes unicode.
30 *
31 * Maintient une liste des couples (clé,valeur) permettant d'associer une
32 * chaîne de caractère à une autre. Ce type de dictionnaire est
33 * utilisé par exemple pour les traductions, auquel cas la clé est
34 * le langage et la valeur la traduction correspondante.
35 */
36class ARCCORE_COMMON_EXPORT StringDictionary
37{
38 private:
39
40 class Impl; //!< Implémentation
41
42 public:
43
44 //! Construit un dictionnaire
45 StringDictionary();
46 //! Construit un dictionnaire
47 StringDictionary(const StringDictionary& rhs);
48 ~StringDictionary(); //!< Libère les ressources
49
50 public:
51
52 /*! \brief Ajoute le couple (key,value) au dictionnaire.
53 *
54 * Si une valeur existe déjà pour \a key, elle est remplacée par
55 * la nouvelle.
56 */
57 void add(const String& key, const String& value);
58
59 /*! \brief Supprime la valeur associée à \a key.
60 *
61 * Si aucune valeur n'était associée à \a key, rien ne se passe.
62 * \return la valeur supprimée s'il y en a une.
63 */
64 String remove(const String& key);
65
66 /*! \brief Retourne la valeur associée à \a key.
67 *
68 * Si aucune valeur n'est associée à \a key, la chaîne nulle est retournée.
69 * Il n'est pas possible de faire la différence entre une valeur
70 * correspondant à la chaîne nulle et une valeur non trouvée sauf si
71 * \a throw_exception est vrai, auquel cas une exception est renvoyée
72 * s'il n'existe pas de valeur correspondant à \a key.
73 */
74 String find(const String& key, bool throw_exception = false) const;
75
76 //! Remplit \a keys et \a values avec les valeurs correspondantes du dictionnaire
77 void fill(StringList& param_names, StringList& values) const;
78
79 private:
80
81 Impl* m_p; //!< Implémentation
82};
