171/*!
172 * \brief Teste si un message est disponible.
173 *
174 * Cette fonction permet de savoir si un message issu du couple (rang,tag)
175 * est disponible. \a message doit avoir été initialisé avec un couple (rang,tag)
176 * (message.isRankTag() doit être vrai).
177 *
178 * Retourne une instance de \a MessageId.
179 *
180 * En mode non bloquant, si aucun message n'est disponible, alors
181 * MessageId::isValid() vaut \a false pour l'instance retournée.
182 *
183 * La sémantique est identique à celle de MPI_Mprobe. Le message retourné est enlevé
184 * de la liste des messages et donc un appel ultérieur à cette méthode avec les mêmes
185 * paramètres retournera un autre message ou un message nul. Si on souhaite un
186 * comportement identique à MPI_Iprobe()/MPI_Probe() alors il faut utiliser mpLegacyProbe().
187 */
188ARCCORE_MESSAGEPASSING_EXPORT MessageId
189mpProbe(IMessagePassingMng* pm, const PointToPointMessageInfo& message);
190

192/*---------------------------------------------------------------------------*/
193/*!
194 * \brief Teste si un message est disponible.
195 *
196 * Cette fonction permet de savoir si un message issu du couple (rang,tag)
197 * est disponible. \a message doit avoir été initialisé avec un couple (rang,tag)
198 * (message.isRankTag() doit être vrai).
199 *
200 * Retourne une instance de \a MessageSourceInfo. En mode non bloquant, si aucun message
201 * n'est disponible, alors MessageSourceInfo::isValid() vaut \a false pour
202 * l'instance retournée.
203 *
204 * La sémantique est identique à celle de MPI_Probe. Il est donc possible
205 * si on appelle plusieurs fois cette fonction de retourner le même message.
206 * Il n'est pas garanti non plus si on fait un mpReceive() avec l'instance retournée
207 * d'avoir le même message. Pour toutes ces raisons il est préférable d'utiliser
208 * la fonction mpProbe().
209 */
210ARCCORE_MESSAGEPASSING_EXPORT MessageSourceInfo
211mpLegacyProbe(IMessagePassingMng* pm, const PointToPointMessageInfo& message);
