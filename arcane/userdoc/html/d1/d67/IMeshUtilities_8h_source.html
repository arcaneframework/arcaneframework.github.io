57 /*!
58 * \brief Recherche les identifiants locaux des faces à partir
59 * de leur connectivité.
60 *
61 * Prend en entrée une liste d'entités décrite par les identifiants uniques
62 * (Item::uniqueId()) de leurs noeuds et recherche les identifiants locaux (Item::localId())
63 * de ces entités.
64 *
65 * \param items_type tableau des ItemTypeId des entités
66 * \param items_connectivity tableau contenant les indices uniques des noeuds des entités.
67 * \param local_ids en retour, contient les identifiants locaux des
68 * entités. Le nombre d'éléments de \a local_ids doit être égal à
69 * celui de \a items_nb_node.
70 *
71 * Le tableau \a items_connectivity contient les identifiants des noeuds des faces,
72 * rangés consécutivement. Par exemple, si \c items_type[0]==IT_Triangle3 et
73 * \c items_type[1]==IT_Quad4, alors \a items_connectivity[0..2] contiendra les
74 * noeuds de l'entité 0, et items_connectivity[3..6] ceux de l'entité 1.
75 *
76 * Si \a allow_null est faux, une erreur fatale est générée si
77 * une entité n'est pas trouvée, sinon NULL_ITEM_LOCAL_ID est
78 * retourné pour l'entité correspondante.
79 */
80 virtual void getFacesLocalIdFromConnectivity(ConstArrayView<ItemTypeId> items_type,
81 ConstArrayView<Int64> items_connectivity,
82 ArrayView<Int32> local_ids,
83 bool allow_null = false) = 0;

85 /*!
86 * \brief Calcule la normale d'un groupe de face.
87 *
88 * Cette méthode calcule la normale à un groupe de face en considérant que
89 * cette surface est un plan. Pour le calcul, l'algorithme essaie de
90 * déterminer les noeuds aux extrémités de cette surface, et calcule une
91 * normale à partir de ces noeuds. L'orientation de la normale (rentrante
92 * ou sortante) est indéfinie.
93 *
94 * Si la surface n'est pas plane, le résultat est indéfini.
95 *
96 * L'algorithme actuel ne fonctionne pas toujours sur une surface composée
97 * uniquement de triangles.
98 *
99 * Cette méthode est collective. L'algorithme utilisé garantit les
100 * mêmes résultats en séquentiel et en parallèle.
101 *
102 * La variable \a nodes_coord est utilisée comme coordonnées pour les noeuds.
103 * En général, il s'agit de IMesh::nodesCoordinates().
104 */
105 virtual Real3 computeNormal(const FaceGroup& face_group,
106 const VariableNodeReal3& nodes_coord) =0;
107
108 /*!
109 * \brief Calcule le vecteur directeur d'une ligne.
110 *
111 * Cette méthode calcule le vecteur directeur d'un groupe de noeuds
112 * en considérant qu'il forme une ligne. Pour le calcul, l'algorithme essaie de
113 * déterminer les noeuds aux extrémités de cette ligne, et calcule un
114 * vecteur à partir de ces noeuds. Le sens du vecteur est indéfini.
115 *
116 * Si le groupe ne forme pas une ligne, le résultat est indéfini.
117 *
118 * Cette méthode est collective. L'algorithme utilisé garantit les
119 * mêmes résultats en séquentiel et en parallèle.
120 *
121 * Si \a n1 et \a n2 ne sont pas nuls, ils contiendront en sortie
122 * les coordonnées extrèmes à partir desquelles la direction est calculée.
123 *
124 * La variable \a nodes_coord est utilisée comme coordonnées pour les noeuds.
125 * En général, il s'agit de IMesh::nodesCoordinates().
126 */
127 virtual Real3 computeDirection(const NodeGroup& node_group,
128 const VariableNodeReal3& nodes_coord,
129 Real3* n1,Real3* n2) =0;

202 /*!
203 * \brief Repartitionne et échange le maillage en gérant la réplication.
204 *
205 * Cette méthode effectue un repartitionnement du maillage via
206 * l'appel à IMeshPartitioner::partitionMesh(bool) et procède à l'échange
207 * des entités via IPrimaryMesh::exchangeItems().
208 *
209 * Elle mais gère aussi la réplication en s'assurant que tous les réplica
210 * ont le même maillage.
211 * Le principe est le suivant:
212 * - seul le réplica maître effectue le repartitionnement en
213 * appelant IMeshPartitioner::partitionMesh() avec \a partitioner comme partitionneur
214 * - les valeurs des IItemFamily::itemsNewOwner() sont ensuite
215 * synchronisées avec les autres réplicas.
216 * - les échanges d'entités sont effectués via IPrimaryMesh::exchangeItems().
217 *
218 * Cette méthode est collective sur l'ensemble des réplicas.
219 *
220 * \pre Tous les réplicas doivent avoir le même maillage, c'est à dire
221 * que toutes les familles d'entités doivent être identiques à l'exception
222 * des familles de particules qui ne sont pas concernées.
223 * \pre Le maillage doit être une instance de IPrimaryMesh.
224 *
225 * \post Tous les réplicas ont le même maillage à l'exception des familles
226 * de particules.
227 *
228 * \param partitioner Instance du partitionneur à utiliser
229 * \param initial_partition Indique s'il s'agit du partitionnement initial.
230 */
231 virtual void partitionAndExchangeMeshWithReplication(IMeshPartitionerBase* partitioner,
232 bool initial_partition) =0;

240 this->mergeNodes(nodes_local_id, nodes_to_merge_local_id, false);
241 }

243 /*!
244 * \brief Fusionne des nœuds.
245 *
246 * Fusionne deux à deux les nœuds de \a nodes_to_merge_local_id avec ceux
247 * de \a nodes_local_id. Chaque nœud \a nodes_to_merge_local_id[i] est
248 * fusionné avec \a nodes_local_id[i].
249 *
250 * Les nœuds \a nodes_to_merge_local_id sont détruits après fusion. Les entités
251 * reposant entièrement sur ces nœuds fusionnés sont aussi détruites.
252 *
253 * Il est interdit de fusionner deux nœuds d'une même maille ou d'une même face
254 * (après fusion, une face ou une maille ne peut pas avoir deux fois le
255 * même nœud).
256 *
257 * Une fois la fusion effectuée, les faces contenant les nœuds fusionnés
258 * (\a nodes_to_merge_local_id) sont détruites. Si \a allow_non_corresponding_face
259 * est faux, alors pour chaque face détruite doit correspondre une face
260 * existante avec les nœuds fusionnés (\a nodes_local_id).
261 */
262 virtual void mergeNodes(Int32ConstArrayView nodes_local_id,
263 Int32ConstArrayView nodes_to_merge_local_id,
264 bool allow_non_corresponding_face) = 0;
265 /*!
266 * \brief Recalcule les uniqueId() des arêtes, faces et mailles en fonction
267 * des uniqueId() des noeuds.
268 *
269 * \warning Cette méthode est expérimentale et ne doit être utilisée que
270 * dans Arcane. Elle suppose que les uniqueId() des entités sont construit
271 * à partir de generateHashUniqueId().
272 */
273 virtual void recomputeItemsUniqueIdFromNodesUniqueId() =0;
274};
