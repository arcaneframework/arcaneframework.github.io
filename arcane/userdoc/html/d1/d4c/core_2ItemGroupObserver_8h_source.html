27class IItemGroupObserver
28{
29 public:
30
31 template<typename T>
32 struct FuncTraits
33 {
34 //! Type du pointeur sur la méthode avec Infos
35 typedef void (T::*FuncPtrWithInfo)(const Int32ConstArrayView * info);
36
37 //! Type du pointeur sur la méthode sans Infos
38 typedef void (T::*FuncPtr)();
39 };
40
41 public:
42
43 //! Destructeur
44 virtual ~IItemGroupObserver() { }
45
46 //! Execute l'action associée à l'extension
47 /*! \param info liste des localIds ajoutés
48 * Suppose qu'il n'y a pas de changement d'ordre ou de renumérotation
49 *
50 * Cette méthode ne peut pas être parallèle.
51 */
52 virtual void executeExtend(const Int32ConstArrayView * info) =0;
53
54 //! Execute l'action associée à l'extension
55 /*! \param info liste des positions supprimées dans l'ancien groupe
56 * Suppose qu'il n'y a pas de changement d'ordre ou de renumérotation
57 * Cette approche par rapport à la liste des localIds est motivée par
58 * la contrainte dans PartialVariable qui n'a pas connaissance des localIds
59 * qu'il héberge.
60 * \param info2 liste des localIds des éléments supprimés. Potentiellement redondant
61 * avec \a info, mais inévitable pour certaines structures changeant l'ordre par rapport
62 * au groupe de référence (ex: ItemGroupDynamicMeshObserver) (DEPRECATED)
63 *
64 * Cette méthode ne peut pas être parallèle.
65 */
66 virtual void executeReduce(const Int32ConstArrayView * info) =0;
67 // virtual void executeReduce(const Int32ConstArrayView * info, const Int32ConstArrayView * info2) = 0;
68
69 //! Execute l'action associée au compactage
70 /*! \param info liste des permutations dans le sens old->new
71 * Suppose qu'il n'y a pas de changement de taille
72 */
73 virtual void executeCompact(const Int32ConstArrayView * info) =0;
74
75 //! Execute l'action associée à l'invalidation
76 /*! Aucune information de transition disponible */
77 virtual void executeInvalidate() =0;
78
79 //! Indique si l'observer aura besoin d'information de transition
80 /*! Cette information ne doit pas changer après le premier appel à cet fonction */
81 virtual bool needInfo() const =0;
82};

88class ItemGroupObserverWithInfoT
89: public IItemGroupObserver
90{
91 public:
92
93 ItemGroupObserverWithInfoT(T* object,
94 typename FuncTraits<T>::FuncPtrWithInfo extend_funcptr,
95 typename FuncTraits<T>::FuncPtrWithInfo reduce_funcptr,
96 typename FuncTraits<T>::FuncPtrWithInfo compact_funcptr,
97 typename FuncTraits<T>::FuncPtr invalidate_funcptr)
98 : m_object(object)
99 , m_extend_function(extend_funcptr)
100 , m_reduce_function(reduce_funcptr)
101 , m_compact_function(compact_funcptr)
102 , m_invalidate_function(invalidate_funcptr) { }
103
104 public:
105
106 void executeExtend(const Int32ConstArrayView * info)
107 {
108 (m_object->*m_extend_function)(info);
109 }
110
111 void executeReduce(const Int32ConstArrayView * info)
112 {
113 (m_object->*m_reduce_function)(info);
114 }
115
116 void executeCompact(const Int32ConstArrayView * info)
117 {
118 (m_object->*m_compact_function)(info);
119 }
120
121 void executeInvalidate()
122 {
123 (m_object->*m_invalidate_function)();
124 }
125
126 bool needInfo() const
127 {
128 return true;
129 }
130
131 private:
132
133 T* m_object; //!< Objet associé.
134 typename FuncTraits<T>::FuncPtrWithInfo m_extend_function; //!< Pointeur vers la méthode associée.
135 typename FuncTraits<T>::FuncPtrWithInfo m_reduce_function; //!< Pointeur vers la méthode associée.
136 typename FuncTraits<T>::FuncPtrWithInfo m_compact_function; //!< Pointeur vers la méthode associée.
137 typename FuncTraits<T>::FuncPtr m_invalidate_function; //!< Pointeur vers la méthode associée.
138};
