63/*!
64 * \internal
65 * \brief Classe de base pour effectuer une réduction.
66 *
67 * Contient les arguments nécessaires pour effectuer une réduction.
68 */
69template <typename DataType>
70class GenericReducerBase
71{
72 friend class GenericReducerIf<DataType>;
73
74 public:
75
76 GenericReducerBase(const RunQueue& queue)
77 : m_queue(queue)
78 {}
79
80 protected:
81
82 DataType _reducedValue() const
83 {
84 m_queue.barrier();
85 return m_host_reduce_storage[0];
86 }
87
88 void _allocate()
89 {
90 eMemoryResource r = eMemoryResource::HostPinned;
91 if (m_host_reduce_storage.memoryRessource() != r)
92 m_host_reduce_storage = NumArray<DataType, MDDim1>(r);
93 m_host_reduce_storage.resize(1);
94 }
95
96 protected:
97
98 RunQueue m_queue;
99 GenericDeviceStorage m_algo_storage;
100 DeviceStorage<DataType> m_device_reduce_storage;
101 NumArray<DataType, MDDim1> m_host_reduce_storage;
102};

106/*!
107 * \internal
108 * \brief Classe pour effectuer un partitionnement d'une liste.
109 *
110 * La liste est partitionnée en deux listes.
111 *
112 * \a DataType est le type de donnée.
113 */
114template <typename DataType>
115class GenericReducerIf
116{
117 // TODO: Faire le malloc sur le device associé à la queue.
118 // et aussi regarder si on peut utiliser mallocAsync().
119
120 public:
121
122 template <typename InputIterator, typename ReduceOperator>
123 void apply(GenericReducerBase<DataType>& s, Int32 nb_item, const DataType& init_value,
124 InputIterator input_iter, ReduceOperator reduce_op, const TraceInfo& trace_info)
125 {
126 RunQueue& queue = s.m_queue;
127 RunCommand command = makeCommand(queue);
128 command << trace_info;
129 Impl::RunCommandLaunchInfo launch_info(command, nb_item);
130 launch_info.beginExecute();
131 eExecutionPolicy exec_policy = queue.executionPolicy();
132 switch (exec_policy) {
133#if defined(ARCCORE_COMPILING_CUDA)
134 case eExecutionPolicy::CUDA: {
135 size_t temp_storage_size = 0;
136 cudaStream_t stream = Impl::CudaUtils::toNativeStream(queue);
137 DataType* reduced_value_ptr = nullptr;
138 // Premier appel pour connaitre la taille pour l'allocation
139 ARCCORE_CHECK_CUDA(::cub::DeviceReduce::Reduce(nullptr, temp_storage_size, input_iter, reduced_value_ptr,
140 nb_item, reduce_op, init_value, stream));
141
142 s.m_algo_storage.allocate(temp_storage_size);
143 reduced_value_ptr = s.m_device_reduce_storage.allocate();
144 ARCCORE_CHECK_CUDA(::cub::DeviceReduce::Reduce(s.m_algo_storage.address(), temp_storage_size,
145 input_iter, reduced_value_ptr, nb_item,
146 reduce_op, init_value, stream));
147 s.m_device_reduce_storage.copyToAsync(s.m_host_reduce_storage, queue);
148 } break;
149#endif
150#if defined(ARCCORE_COMPILING_HIP)
151 case eExecutionPolicy::HIP: {
152 size_t temp_storage_size = 0;
153 hipStream_t stream = Impl::HipUtils::toNativeStream(queue);
154 DataType* reduced_value_ptr = nullptr;
155 // Premier appel pour connaitre la taille pour l'allocation
156 ARCCORE_CHECK_HIP(rocprim::reduce(nullptr, temp_storage_size, input_iter, reduced_value_ptr, init_value,
157 nb_item, reduce_op, stream));
158
159 s.m_algo_storage.allocate(temp_storage_size);
160 reduced_value_ptr = s.m_device_reduce_storage.allocate();
161
162 ARCCORE_CHECK_HIP(rocprim::reduce(s.m_algo_storage.address(), temp_storage_size, input_iter, reduced_value_ptr, init_value,
163 nb_item, reduce_op, stream));
164 s.m_device_reduce_storage.copyToAsync(s.m_host_reduce_storage, queue);
165 } break;
166#endif
167#if defined(ARCCORE_COMPILING_SYCL)
168 case eExecutionPolicy::SYCL: {
169 {
170 RunCommand command2 = makeCommand(queue);
171 using ReducerType = typename ReduceOperatorToReducerTypeTraits<DataType, ReduceOperator>::ReducerType;
172 ReducerType reducer(command2);
173 command2 << RUNCOMMAND_LOOP1(iter, nb_item, reducer)
174 {
175 auto [i] = iter();
176 reducer.combine(input_iter[i]);
177 };
178 queue.barrier();
179 s.m_host_reduce_storage[0] = reducer.reducedValue();
180 }
181 } break;
182#endif
183 case eExecutionPolicy::Thread:
184 // Pas encore implémenté en multi-thread
185 [[fallthrough]];
186 case eExecutionPolicy::Sequential: {
187 DataType reduced_value = init_value;
188 for (Int32 i = 0; i < nb_item; ++i) {
189 reduced_value = reduce_op(reduced_value, *input_iter);
190 ++input_iter;
191 }
192 s.m_host_reduce_storage[0] = reduced_value;
193 } break;
194 default:
195 ARCCORE_FATAL(getBadPolicyMessage(exec_policy));
196 }
197 launch_info.endExecute();
198 }
199};

214/*!
215 * \brief Algorithme générique de réduction sur accélérateur.
216 *
217 * La réduction se fait via les appels à applyMin(), applyMax(), applySum(),
218 * applyMinWithIndex(), applyMaxWithIndex() ou applySumWithIndex(). Ces
219 * méthodes sont asynchrones. Après réduction, il est possible récupérer la
220 * valeur réduite via reducedValue(). L'appel à reducedValue() bloque tant
221 * que la réduction n'est pas terminée.
222 *
223 * Les instances de cette classe peuvent être utilisées plusieurs fois.
224 *
225 * Voici un exemple pour calculer la somme d'un tableau de 50 éléments:
226 *
227 * \code
228 * using namespace Arcane;
229 * const Int32 nb_value(50);
230 * Arcane::NumArray<Real, MDDim1> t1(nv_value);
231 * Arcane::SmallSpan<const Real> t1_view(t1);
232 * Arcane::RunQueue queue = ...;
233 * Arcane::Accelerator::GenericReducer<Real> reducer(queue);
234 *
235 * // Calcul direct
236 * reducer.applySum(t1_view);
237 * std::cout << "Sum is '" << reducer.reducedValue() << "\n";
238 *
239 * // Calcul avec lambda
240 * auto getter_func = [=] ARCCORE_HOST_DEVICE(Int32 index) -> Real
241 * {
242 * return t1_view[index];
243 * }
244 * reducer.applySumWithIndex(nb_value,getter_func);
245 * std::cout << "Sum is '" << reducer.reducedValue() << "\n";
246 * \endcode
247 */
248template <typename DataType>
249class GenericReducer
250: private Impl::GenericReducerBase<DataType>
251{
252 public:
253
254 explicit GenericReducer(const RunQueue& queue)
255 : Impl::GenericReducerBase<DataType>(queue)
256 {
257 this->_allocate();
258 }
259
260 public:
261
262 //! Applique une réduction 'Min' sur les valeurs \a values
263 void applyMin(SmallSpan<const DataType> values, const TraceInfo& trace_info = TraceInfo())
264 {
265 _apply(values.size(), values.data(), Impl::MinOperator<DataType>{}, trace_info);
266 }
267
268 //! Applique une réduction 'Max' sur les valeurs \a values
269 void applyMax(SmallSpan<const DataType> values, const TraceInfo& trace_info = TraceInfo())
270 {
271 _apply(values.size(), values.data(), Impl::MaxOperator<DataType>{}, trace_info);
272 }
273
274 //! Applique une réduction 'Somme' sur les valeurs \a values
275 void applySum(SmallSpan<const DataType> values, const TraceInfo& trace_info = TraceInfo())
276 {
277 _apply(values.size(), values.data(), Impl::SumOperator<DataType>{}, trace_info);
278 }
279
280 //! Applique une réduction 'Min' sur les valeurs sélectionnées par \a select_lambda
281 template <typename SelectLambda>
282 void applyMinWithIndex(Int32 nb_value, const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
283 {
284 _applyWithIndex(nb_value, select_lambda, Impl::MinOperator<DataType>{}, trace_info);
285 }
286
287 //! Applique une réduction 'Max' sur les valeurs sélectionnées par \a select_lambda
288 template <typename SelectLambda>
289 void applyMaxWithIndex(Int32 nb_value, const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
290 {
291 _applyWithIndex(nb_value, select_lambda, Impl::MaxOperator<DataType>{}, trace_info);
292 }
293
294 //! Applique une réduction 'Somme' sur les valeurs sélectionnées par \a select_lambda
295 template <typename SelectLambda>
296 void applySumWithIndex(Int32 nb_value, const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
297 {
298 _applyWithIndex(nb_value, select_lambda, Impl::SumOperator<DataType>{}, trace_info);
299 }
300
301 //! Valeur de la réduction
302 DataType reducedValue()
303 {
304 m_is_already_called = false;
305 return this->_reducedValue();
306 }
307
308 private:
309
310 bool m_is_already_called = false;
311
312 private:
313
314 template <typename InputIterator, typename ReduceOperator>
315 void _apply(Int32 nb_value, InputIterator input_iter, ReduceOperator reduce_op, const TraceInfo& trace_info)
316 {
317 _setCalled();
318 Impl::GenericReducerBase<DataType>* base_ptr = this;
319 Impl::GenericReducerIf<DataType> gf;
320 DataType init_value = reduce_op.defaultValue();
321 gf.apply(*base_ptr, nb_value, init_value, input_iter, reduce_op, trace_info);
322 }
323
324 template <typename GetterLambda, typename ReduceOperator>
325 void _applyWithIndex(Int32 nb_value, const GetterLambda& getter_lambda,
326 ReduceOperator reduce_op, const TraceInfo& trace_info)
327 {
328 _setCalled();
329 Impl::GenericReducerBase<DataType>* base_ptr = this;
330 Impl::GenericReducerIf<DataType> gf;
331 Impl::GetterLambdaIterator<DataType, GetterLambda> input_iter(getter_lambda);
332 DataType init_value = reduce_op.defaultValue();
333 gf.apply(*base_ptr, nb_value, init_value, input_iter, reduce_op, trace_info);
334 }
335
336 void _setCalled()
337 {
338 if (m_is_already_called)
339 ARCCORE_FATAL("apply() has already been called for this instance");
340 m_is_already_called = true;
341 }
342};
