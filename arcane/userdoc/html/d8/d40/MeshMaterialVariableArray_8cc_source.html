48template<typename DataType> void
49MaterialVariableArrayTraits<DataType>::
50copyTo(SmallSpan2<const DataType> input, SmallSpan<const Int32> input_indexes,
51 SmallSpan2<DataType> output, SmallSpan<const Int32> output_indexes,
52 RunQueue& queue)
53{
54 // TODO: vérifier tailles des indexes et des dim2Size() identiques
55 Integer nb_value = input_indexes.size();
56 Integer dim2_size = input.dim2Size();
57
58 auto command = makeCommand(queue);
59 command << RUNCOMMAND_LOOP1(iter, nb_value)
60 {
61 auto [i] = iter();
62 auto xo = output[output_indexes[i]];
63 auto xi = input[ input_indexes[i] ];
64 for( Integer j=0; j<dim2_size; ++j )
65 xo[j] = xi[j];
66 };
67}

87template<typename DataType> void
88MaterialVariableArrayTraits<DataType>::
89saveData(IMeshComponent* component,IData* data,
90 Array<ContainerViewType>& cviews)
91{
92 // Pour les tableaux 2D, on redimensionne directement
93 // \a values en ajoutant le nombre d'éléments pour notre
94 // component.
95 ConstArrayView<Array2View<DataType>> views = cviews;
96 if (views.empty())
97 return;
98 auto* true_data = dynamic_cast<ValueDataType*>(data);
99 ARCANE_CHECK_POINTER(true_data);
100 ContainerType& values = true_data->_internal()->_internalDeprecatedValue();
101 ComponentItemVectorView component_view = component->view();
102
103 Integer dim2_size = views[0].dim2Size();
104 Integer current_dim1_size = values.dim1Size();
105 Integer added_dim1_size = component_view.nbItem();
106 values.resizeNoInit(current_dim1_size+added_dim1_size,dim2_size);
107 ConstArrayView<MatVarIndex> mvi_indexes = component_view._matvarIndexes();
108
109 for( Integer i=0, n=mvi_indexes.size(); i<n; ++i ){
110 MatVarIndex mvi = mvi_indexes[i];
111 values[i+current_dim1_size].copy(views[mvi.arrayIndex()][mvi.valueIndex()]);
112 }
113}

151template<typename DataType> void
152ItemMaterialVariableArray<DataType>::
153serialize(ISerializer* sbuf,Int32ConstArrayView ids)
154{
155 // TODO: essayer de fusionner cette methode avec la variante scalaire.
156 IItemFamily* family = m_global_variable->itemFamily();
157 if (!family)
158 return;
159 IMeshMaterialMng* mat_mng = m_p->materialMng();
160 const Int32 nb_count = DataTypeTraitsT<DataType>::nbBasicType();
161 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
162 const eDataType data_type = DataTypeTraitsT<BasicType>::type();
163 ItemVectorView ids_view(family->view(ids));
164 Int32 dim2_size = m_global_variable->valueView().dim2Size();
165 bool has_mat = this->space()!=MatVarSpace::Environment;
166 switch(sbuf->mode()){
167 case ISerializer::ModeReserve:
168 {
169 Int64 nb_val = 0;
170 ENUMERATE_ALLENVCELL(iallenvcell,mat_mng,ids_view){
171 ++nb_val; // 1 valeur pour le milieu
172 ENUMERATE_CELL_ENVCELL(ienvcell,(*iallenvcell)){
173 EnvCell envcell = *ienvcell;
174 if (has_mat)
175 nb_val += envcell.nbMaterial(); // 1 valeur par matériau du milieu.
176 }
177 }
178 sbuf->reserve(m_global_variable->fullName());
179 sbuf->reserve(DT_Int64,1); // Pour le nombre de valeurs.
180 Int64 nb_basic_value = nb_val * nb_count * dim2_size;
181 sbuf->reserveSpan(data_type,nb_basic_value); // Pour le nombre de valeurs.
182 }
183 break;
184 case ISerializer::ModePut:
185 {
186 UniqueArray<DataType> values;
187 ENUMERATE_ALLENVCELL(iallenvcell,mat_mng,ids_view){
188 ENUMERATE_CELL_ENVCELL(ienvcell,(*iallenvcell)){
189 values.addRange(value(ienvcell._varIndex()));
190 if (has_mat){
191 ENUMERATE_CELL_MATCELL(imatcell,(*ienvcell)){
192 values.addRange(value(imatcell._varIndex()));
193 }
194 }
195 }
196 }
197 Int64 nb_value = values.largeSize();
198 Span<const BasicType> basic_values(reinterpret_cast<BasicType*>(values.data()),nb_value*nb_count);
199 sbuf->put(m_global_variable->fullName());
200 sbuf->putInt64(nb_value);
201 sbuf->putSpan(basic_values);
202 }
203 break;
204 case ISerializer::ModeGet:
205 {
206 UniqueArray<BasicType> basic_values;
207 String ref_name;
208 sbuf->get(ref_name);
209 if (m_global_variable->fullName()!=ref_name)
210 ARCANE_FATAL("Bad serialization expected={0} found={1}",m_global_variable->fullName(),ref_name);
211 Int32 nb_value = CheckedConvert::toInt32(sbuf->getInt64());
212 Int32 nb_basic_value = nb_value * nb_count;
213 basic_values.resize(nb_basic_value);
214 sbuf->getSpan(basic_values);
215 if (dim2_size!=0){
216 SmallSpan2<DataType> data_values(reinterpret_cast<DataType*>(basic_values.data()),nb_value,dim2_size);
217 Int32 index = 0;
218 ENUMERATE_ALLENVCELL(iallenvcell,mat_mng,ids_view){
219 ENUMERATE_CELL_ENVCELL(ienvcell,(*iallenvcell)){
220 EnvCell envcell = *ienvcell;
221 setValue(ienvcell._varIndex(),data_values[index]);
222 ++index;
223 if (has_mat){
224 ENUMERATE_CELL_MATCELL(imatcell,envcell){
225 setValue(imatcell._varIndex(),data_values[index]);
226 ++index;
227 }
228 }
229 }
230 }
231 }
232 }
233 break;
234 default:
235 ARCANE_THROW(NotSupportedException,"Invalid serialize");
236 }
237}

298template<typename DataType> void
299ItemMaterialVariableArray<DataType>::
300_copyToBufferLegacy(SmallSpan<const MatVarIndex> matvar_indexes,Span<std::byte> bytes) const
301{
302 const Integer one_data_size = dataTypeSize();
303 Integer dim2_size = m_vars[0]->valueView().dim2Size();
304
305 // TODO: Vérifier que la taille est un multiple de 'one_data_size' et que
306 // l'alignement est correct.
307 const Int32 value_size = CheckedConvert::toInt32(bytes.size() / one_data_size);
308 Array2View<DataType> values(reinterpret_cast<DataType*>(bytes.data()),value_size,dim2_size);
309 for( Integer z=0; z<value_size; ++z ){
310 values[z].copy(value(matvar_indexes[z]));
311 }
312}
