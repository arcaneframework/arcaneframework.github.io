49  // Wrapper de Arccore::Array pour la classe NumArray
50  template<typename DataType>
51  class NumArrayContainer
52  : private Arccore::Array<DataType>
53  {
54  private:
55  using BaseClass = Arccore::Array<DataType>;
56  public:
57  using BaseClass::capacity;
58  using BaseClass::fill;
59  public:
60  explicit NumArrayContainer(IMemoryAllocator* a)
61  {
62  this->_initFromAllocator(a,0);
63  }
64  NumArrayContainer(const NumArrayContainer<DataType>& rhs) : BaseClass()
65  {
66  this->_initFromSpan(rhs.to1DSpan());
67  }
68  NumArrayContainer<DataType>& operator=(const NumArrayContainer<DataType>& rhs)
69  {
70  if (this!=&rhs){
71  BaseClass::_copy(rhs.data());
72  }
73  return (*this);
74  }
75  public:
76  void resize(Int64 new_size) { BaseClass::_resize(new_size); }
77  Span<DataType> to1DSpan() { return BaseClass::span(); }
78  Span<const DataType> to1DSpan() const { return BaseClass::constSpan(); }
79  Span<std::byte> bytes() { return asWritableBytes(BaseClass::span()); }
80  Span<const std::byte> bytes() const { return asBytes(BaseClass::constSpan()); }
81  void swap(NumArrayContainer<DataType>& rhs) { BaseClass::_swap(rhs); }
82  void copy(Span<const DataType> rhs) { BaseClass::_copy(rhs.data()); }
83  void copyInitializerList(std::initializer_list<DataType> alist)
84  {
85  Span<DataType> s = to1DSpan();
86  Int64 s1 = s.size();
87  Int32 index = 0;
88  for( auto x : alist ){
89  s[index] = x;
90  ++index;
91  // S'assure qu'on ne déborde pas
92  if (index>=s1)
93  break;
94  }
95  }
96  };
97 }

125 template<typename DataType,int RankValue,typename LayoutType>
126 class NumArrayBase
127 : public NumArrayBaseCommon
128 {
129  public:
130 
131  using ConstSpanType = MDSpan<const DataType,RankValue,LayoutType>;
132  using SpanType = MDSpan<DataType,RankValue,LayoutType>;
133  using ArrayWrapper = impl::NumArrayContainer<DataType>;
134 
135  public:
136 
138  Int64 totalNbElement() const { return m_total_nb_element; }
140  Int32 extent(int i) const { return m_span.extent(i); }
145  void resize(ArrayExtents<RankValue> extents)
146  {
147  m_span.m_extents.setSize(extents);
148  _resize();
149  }
150  protected:
151 
152  NumArrayBase() : m_data(_getDefaultAllocator()){}
153  explicit NumArrayBase(eMemoryRessource r) : m_data(_getDefaultAllocator(r)), m_memory_ressource(r){}
154  explicit NumArrayBase(ArrayExtents<RankValue> extents)
155  : m_data(_getDefaultAllocator()), m_memory_ressource(eMemoryRessource::UnifiedMemory)
156  {
157  resize(extents);
158  }
159  NumArrayBase(ArrayExtents<RankValue> extents,eMemoryRessource r)
160  : m_data(_getDefaultAllocator(r)), m_memory_ressource(r)
161  {
162  resize(extents);
163  }
164 
165  private:
166  void _resize()
167  {
168  Int32 dim1_size = extent(0);
169  Int32 dim2_size = 1;
170  // TODO: vérifier débordement.
171  for (int i=1; i<RankValue; ++i )
172  dim2_size *= extent(i);
173  m_total_nb_element = dim1_size * dim2_size;
174  m_data.resize(m_total_nb_element);
175  m_span.m_ptr = m_data.to1DSpan().data();
176  }
177  public:
178  void fill(const DataType& v) { m_data.fill(v); }
179  Int32 nbDimension() const { return RankValue; }
180  ArrayExtents<RankValue> extents() const { return m_span.extents(); }
181  ArrayExtentsWithOffset<RankValue,LayoutType> extentsWithOffset() const
182  {
183  return m_span.extentsWithOffset();
184  }
185  public:
186  SpanType span() { return m_span; }
187  ConstSpanType span() const { return m_span.constSpan(); }
188  ConstSpanType constSpan() const { return m_span.constSpan(); }
189  public:
190  Span<const DataType> to1DSpan() const { return m_span.to1DSpan(); }
191  Span<DataType> to1DSpan() { return m_span.to1DSpan(); }
192  void copy(ConstSpanType rhs)
193  {
194  _checkHost(m_memory_ressource);
195  m_data.copy(rhs.to1DSpan());
196  }
197  void copy(const NumArrayBase<DataType,RankValue,LayoutType>& rhs)
198  {
199  this->resize(rhs.extents());
200  _copy(asBytes(rhs.to1DSpan()),rhs.m_memory_ressource,
201  asWritableBytes(to1DSpan()),m_memory_ressource);
202  }
203  const DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
204  {
205  return m_span(idx);
206  }
207  DataType& operator()(ArrayBoundsIndex<RankValue> idx)
208  {
209  return m_span(idx);
210  }
211  DataType& s(ArrayBoundsIndex<RankValue> idx)
212  {
213  return m_span(idx);
214  }
215  void swap(NumArrayBase<DataType,RankValue>& rhs)
216  {
217  m_data.swap(rhs.m_data);
218  std::swap(m_span,rhs.m_span);
219  std::swap(m_total_nb_element,rhs.m_total_nb_element);
220  std::swap(m_memory_ressource,rhs.m_memory_ressource);
221  }
222  Int64 capacity() const { return m_data.capacity(); }
223  eMemoryRessource memoryRessource() const { return m_memory_ressource; }
224  Span<std::byte> bytes() { return asWritableBytes(to1DSpan()); }
225  Span<const std::byte> bytes() const { return asBytes(to1DSpan()); }
226  public:
228  DataType* _internalData() { return m_span._internalData(); }
229  protected:
230  SpanType m_span;
231  ArrayWrapper m_data;
232  Int64 m_total_nb_element = 0;
233  eMemoryRessource m_memory_ressource = eMemoryRessource::Unknown;
234 };
