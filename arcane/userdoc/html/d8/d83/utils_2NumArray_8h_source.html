51  class NumArrayContainer
52  : private Arccore::Array<DataType>
53  {
54  private:
55  using BaseClass = Arccore::Array<DataType>;
56  using ThatClass = NumArrayContainer<DataType>;
57  public:
58  using BaseClass::capacity;
59  using BaseClass::fill;
60  public:
61  explicit NumArrayContainer(IMemoryAllocator* a)
62  {
63  this->_initFromAllocator(a,0);
64  }
65  NumArrayContainer(const ThatClass& rhs) : BaseClass()
66  {
67  this->_initFromSpan(rhs.to1DSpan());
68  }
69  NumArrayContainer(ThatClass&& rhs)
70  : BaseClass(std::move(rhs))
71  {
72  }
73  ThatClass& operator=(const ThatClass& rhs)
74  {
75  if (this!=&rhs){
76  BaseClass::_copy(rhs.data());
77  }
78  return (*this);
79  }
80  ThatClass& operator=(ThatClass&& rhs)
81  {
82  this->_move(rhs);
83  return (*this);
84  }
85 
86  public:
87  void resize(Int64 new_size) { BaseClass::_resize(new_size); }
88  Span<DataType> to1DSpan() { return BaseClass::span(); }
89  Span<const DataType> to1DSpan() const { return BaseClass::constSpan(); }
90  Span<std::byte> bytes() { return asWritableBytes(BaseClass::span()); }
91  Span<const std::byte> bytes() const { return asBytes(BaseClass::constSpan()); }
92  void swap(NumArrayContainer<DataType>& rhs) { BaseClass::_swap(rhs); }
93  void copy(Span<const DataType> rhs) { BaseClass::_copy(rhs.data()); }
94  void copyInitializerList(std::initializer_list<DataType> alist)
95  {
96  Span<DataType> s = to1DSpan();
97  Int64 s1 = s.size();
98  Int32 index = 0;
99  for( auto x : alist ){
100  s[index] = x;
101  ++index;
102  // S'assure qu'on ne déborde pas
103  if (index>=s1)
104  break;
105  }
106  }
107  };
108 }

136 template<typename DataType,A_MDRANK_TYPE(RankValue),typename LayoutType>
137 class NumArrayBase
138 : public NumArrayBaseCommon
139 {
140  public:
141 
142  using ConstSpanType = MDSpan<const DataType,RankValue,LayoutType>;
143  using SpanType = MDSpan<DataType,RankValue,LayoutType>;
144  using ArrayWrapper = impl::NumArrayContainer<DataType>;
145  using ArrayBoundsIndexType = typename SpanType::ArrayBoundsIndexType;
146  using ThatClass = NumArrayBase<DataType,RankValue,LayoutType>;
147 
148  public:
149 
151  Int64 totalNbElement() const { return m_total_nb_element; }
153  Int32 extent(int i) const { return m_span.extent(i); }
158  void resize(ArrayExtents<RankValue> extents)
159  {
160  m_span.m_extents.setSize(extents);
161  _resize();
162  }
163  protected:
164 
165  NumArrayBase() : m_data(_getDefaultAllocator()){}
166  explicit NumArrayBase(eMemoryRessource r) : m_data(_getDefaultAllocator(r)), m_memory_ressource(r){}
167  explicit NumArrayBase(ArrayExtents<RankValue> extents)
168  : m_data(_getDefaultAllocator()), m_memory_ressource(eMemoryRessource::UnifiedMemory)
169  {
170  resize(extents);
171  }
172  NumArrayBase(ArrayExtents<RankValue> extents,eMemoryRessource r)
173  : m_data(_getDefaultAllocator(r)), m_memory_ressource(r)
174  {
175  resize(extents);
176  }
177  NumArrayBase(const ThatClass&) = default;
178  NumArrayBase(ThatClass&&) = default;
179  ThatClass& operator=(ThatClass&&) = default;
180  ThatClass& operator=(const ThatClass&) = default;
181 
182  private:
183  void _resize()
184  {
185  Int32 dim1_size = extent(0);
186  Int32 dim2_size = 1;
187  // TODO: vérifier débordement.
188  for (int i=1; i< A_MDRANK_RANK_VALUE(RankValue) ; ++i )
189  dim2_size *= extent(i);
190  m_total_nb_element = dim1_size * dim2_size;
191  m_data.resize(m_total_nb_element);
192  m_span.m_ptr = m_data.to1DSpan().data();
193  }
194  public:
195  void fill(const DataType& v) { m_data.fill(v); }
196  constexpr Int32 nbDimension() const { return A_MDRANK_RANK_VALUE(RankValue); }
197  ArrayExtents<RankValue> extents() const { return m_span.extents(); }
198  ArrayExtentsWithOffset<RankValue,LayoutType> extentsWithOffset() const
199  {
200  return m_span.extentsWithOffset();
201  }
202  public:
203  SpanType span() { return m_span; }
204  ConstSpanType span() const { return m_span.constSpan(); }
205  ConstSpanType constSpan() const { return m_span.constSpan(); }
206  public:
207  Span<const DataType> to1DSpan() const { return m_span.to1DSpan(); }
208  Span<DataType> to1DSpan() { return m_span.to1DSpan(); }
209  void copy(ConstSpanType rhs)
210  {
211  _checkHost(m_memory_ressource);
212  m_data.copy(rhs.to1DSpan());
213  }
214  void copy(const NumArrayBase<DataType,RankValue,LayoutType>& rhs)
215  {
216  this->resize(rhs.extents());
217  _copy(asBytes(rhs.to1DSpan()),rhs.m_memory_ressource,
218  asWritableBytes(to1DSpan()),m_memory_ressource);
219  }
220  const DataType& operator()(ArrayBoundsIndexType idx) const
221  {
222  return m_span(idx);
223  }
224  DataType& operator()(ArrayBoundsIndexType idx)
225  {
226  return m_span(idx);
227  }
228  DataType& s(ArrayBoundsIndexType idx)
229  {
230  return m_span(idx);
231  }
232  void swap(NumArrayBase<DataType,RankValue>& rhs)
233  {
234  m_data.swap(rhs.m_data);
235  std::swap(m_span,rhs.m_span);
236  std::swap(m_total_nb_element,rhs.m_total_nb_element);
237  std::swap(m_memory_ressource,rhs.m_memory_ressource);
238  }
239  Int64 capacity() const { return m_data.capacity(); }
240  eMemoryRessource memoryRessource() const { return m_memory_ressource; }
241  Span<std::byte> bytes() { return asWritableBytes(to1DSpan()); }
242  Span<const std::byte> bytes() const { return asBytes(to1DSpan()); }
243  public:
245  DataType* _internalData() { return m_span._internalData(); }
246  protected:
247  SpanType m_span;
248  ArrayWrapper m_data;
249  Int64 m_total_nb_element = 0;
250  eMemoryRessource m_memory_ressource = eMemoryRessource::Unknown;
251 };
