31/*!
32 * \ingroup ArcaneMaterials
33 * \brief Table de connectivité des 'Cell' vers leur(s) 'AllEnvCell' destinée
34 * à une utilisation sur accélérateur.
35 *
36 * Classe qui conserve la connectivité de toutes les mailles
37 * \a Cell vers toutes leurs mailles \a AllEnvCell.
38 *
39 * On crée une instance via la méthode create().
40 *
41 * Le coût de l'initialisation est cher, il faut allouer la mémoire et remplir les
42 * structures. On parcours toutes les mailles et pour chaque maille on fait
43 * appel au CellToAllEnvCellConverter.
44 *
45 * Une fois l'instance créée, elle doit être mise à jour à chaque fois que
46 * la topologie des matériaux/environnements change (ce qui est également cher).
47 *
48 * Cette classe est une classe interne et ne doit pas être manipulée directement.
49 * Il faut passer par les helpers associés dans le IMeshMaterialMng et
50 * la classe CellToAllEnvCellAccessor.
51 */
52class ARCANE_MATERIALS_EXPORT AllCellToAllEnvCell
53{
54 friend class CellToAllEnvCellAccessor;
55 friend class CellToAllComponentCellEnumerator;
56 friend AllCellToAllEnvCellContainer;
57
58 private:
59
60 //! Méthode d'accès à la table de "connectivité" cell -> all env cells
61 ARCCORE_HOST_DEVICE Span<ComponentItemLocalId> operator[](Int32 cell_id) const
62 {
63 return m_allcell_allenvcell_ptr[cell_id];
64 }
65
66 private:
67
68 Span<Span<ComponentItemLocalId>> m_allcell_allenvcell_ptr;
69};
70

73/*!
74 * \ingroup ArcaneMaterials
75 * \brief Classe d'encapsulation pour accéder à la connectivité équivalente
76 * cell -> allenvcell. Destinée à être utilisée avec l'API accélérateur
77 * via les RUNCOMMAND_...
78 * \note Aucun interet en soit, mis à part le fait d'obliger l'utilisateur à créer
79 * cet objet en amout de l'appel à un RUNCOMMAND_ENUMERATE_CELL_ALLENVCELL et donc
80 * de garantir la copie du pointeur AllCellToAllEnvCell pour la lambda à executer sur
81 * l'accélérateur
82 */

85 friend class CellToAllComponentCellEnumerator;

87 public:

101 private:

116class ARCANE_MATERIALS_EXPORT CellToAllComponentCellEnumerator
117{
118 friend class EnumeratorTracer;
119
120 public:
121
122 using index_type = Span<ComponentItemLocalId>::index_type;
123 using size_type = Span<ComponentItemLocalId>::size_type;
124
125 public:
126
127 // La version CPU permet de vérifier qu'on a bien fait l'init avant l'ENUMERATE
128 ARCCORE_HOST_DEVICE explicit CellToAllComponentCellEnumerator(Int32 cell_id, const CellToAllEnvCellAccessor& acc)
129 {
130 AllCellToAllEnvCell all_env_view = acc._getAllCellToAllEnvCell();
131 m_ptr = all_env_view[cell_id];
132 }
133 ARCCORE_HOST_DEVICE void operator++()
134 {
135 ++m_index;
136 }
137
138 ARCCORE_HOST_DEVICE bool hasNext() const
139 {
140 return m_index < m_ptr.size();
141 }
142
143 ARCCORE_HOST_DEVICE ComponentItemLocalId operator*() const
144 {
145 return m_ptr[m_index];
146 }
147
148 private:
149
150 index_type m_index = 0;
151 Span<ComponentItemLocalId> m_ptr;
152};
