34/*!
35 * \ingroup ArcaneMaterials
36 * \brief Table de connectivité des 'Cell' vers leur(s) 'AllEnvCell' destinée
37 * à une utilisation sur accélérateur.
38
39 Classe qui stoque la connectivité de toutes les mailles
40 \a Cell vers toutes leurs mailles \a AllEnvCell.
41
42 On crée une instance via la méthode create().
43
44 Le coût de l'initialisation est cher, il faut allouer la mémoire et remplir les
45 structures. On parcours toutes les mailles et pour chaque maille on fait
46 appel au CellToAllEnvCellConverter.
47
48 Une fois l'instance créée, elle doit être mise à jour à chaque fois que
49 la topologie des matériaux/environnements change (ce qui est également cher).
50
51 Cette classe est une classe interne et ne doit pas être manipulée directement.
52 Il faut passer par les helpers associés dans le IMeshMaterialMng et
53 la classe CellToAllEnvCellAccessor.
54
55 */
56class ARCANE_MATERIALS_EXPORT AllCellToAllEnvCell
57{
58 public:
59
60 AllCellToAllEnvCell() = default;
61
62 //! Copies interdites
63 AllCellToAllEnvCell(const AllCellToAllEnvCell&) = delete;
64 AllCellToAllEnvCell& operator=(const AllCellToAllEnvCell&) = delete;
65
66 /*!
67 * Fonction de création alternative. Il faut attendre que les données
68 * relatives aux matériaux soient finalisées.
69 * La différence réside dans la gestion de la mémoire.
70 * Ici, on applique un compromis sur la taille de la table cid -> envcells
71 * où la taille du tableau pour ranger les envcells d'une cell est égale à la taille
72 * max du nb d'environnement présent à un instant t dans un maille.
73 * Celà permet de ne pas faire les allocations mémoire dans la boucle interne et de
74 * façon systématique.
75 * => Gain de perf à évaluer.
76 */
77 static AllCellToAllEnvCell* create(IMeshMaterialMng* mm, IMemoryAllocator* alloc);
78 ///! Fonction de destruction
79 static void destroy(AllCellToAllEnvCell* instance);
80
81 /*!
82 * Méthode d'accès à la table de "connectivité" cell -> all env cells
83 */
84 ARCCORE_HOST_DEVICE Span<ComponentItemLocalId>* internal() const
85 {
86 return m_allcell_allenvcell;
87 }
88
89 /*! Méthode pour donner le nombre maximal d'environnements
90 * présents sur une maille à l'instant t.
91 * Le fait d'effectuer cette opération à un instant donné, permet
92 * d'avoir une valeur max <= au nombre total d'environnement présents
93 * dans le jdd (et donc d'économiser un peu de mémoire)
94 */
95 Int32 maxNbEnvPerCell() const;
96
97 /* On regarde si le nb max d'env par cell à l'instant t à changer,
98 * et si c'est le cas, on force la reconstruction de la table.
99 * Est appelé par le forceRecompute du ImeshMaterialMng
100 */
101 void bruteForceUpdate();
102
103
104 private:
105 void reset();
106
107 private:
108 IMeshMaterialMng* m_material_mng = nullptr;
109 IMemoryAllocator* m_alloc = nullptr;
110 Integer m_size = 0;
111 Span<ComponentItemLocalId>* m_allcell_allenvcell = nullptr;
112 ComponentItemLocalId* m_mem_pool = nullptr;
113 Int32 m_current_max_nb_env = 0;
114};

120 * \brief Classe d'encapsulation pour accéder à la connectivité équivalente
121 * cell -> allenvcell. Destinée à être utilisée avec l'API accélérateur

124 * cet objet en amout de l'appel à un RUNCOMMAND_ENUMERATE_CELL_ALLENVCELL et donc
125 * de garantir la copie du pointeur AllCellToAllEnvCell pour la lambda à executer sur
126 * l'accélérateur

140 {
141 }

157 return *this;

159

161 {
162 return m_cell_allenvcell;

164

170 private:

172 AllCellToAllEnvCell* m_cell_allenvcell = nullptr;
173};

178class ARCANE_MATERIALS_EXPORT CellToAllComponentCellEnumerator
179{
180 friend class EnumeratorTracer;
181
182 public:
183
184 using index_type = Span<ComponentItemLocalId>::index_type;
185 using size_type = Span<ComponentItemLocalId>::size_type;
186
187 public:
188 // La version CPU permet de vérifier qu'on a bien fait l'init avant l'ENUMERATE
189 ARCCORE_HOST_DEVICE explicit CellToAllComponentCellEnumerator(Integer cell_id, const CellToAllEnvCellAccessor& acc)
190 : m_cid(cell_id)
191#if defined(__CUDA_ARCH__) || defined(__HIP_DEVICE_COMPILE__)
192 , m_ptr(&(acc.getAllCellToAllEnvCell()->internal()[cell_id]))
193 , m_size((*m_ptr).size())
194 {
195 }
196#else
197 {
198 if (acc.getAllCellToAllEnvCell()) {
199 m_ptr = &(acc.getAllCellToAllEnvCell()->internal()[cell_id]);
200 m_size = (*m_ptr).size();
201 } else {
202 ARCANE_FATAL("Must create AllCellToAllEnvCell before using ENUMERATE_ALLENVCELL");
203 }
204 }
205#endif
206
207 ARCCORE_HOST_DEVICE void operator++() { ++m_index; }
208
209 ARCCORE_HOST_DEVICE bool hasNext() const { return m_index<m_size; }
210
211 ARCCORE_HOST_DEVICE const ComponentItemLocalId& operator*() const { return (*m_ptr)[m_index]; }
212
213 private:
214
215 Integer m_cid = 0;
216 index_type m_index = 0;
217 Span<ComponentItemLocalId>* m_ptr = nullptr;
218 size_type m_size = 0;
219};
