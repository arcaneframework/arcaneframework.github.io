28/*!
29 * \ingroup StandardService
30 * \brief Interface d'un objet transformant les courbes d'historiques.
31 *
32 * Les classes implémentant cette interface peuvent transformer les
33 * courbes des historiques en temps. Cela permet par exemple de modifier
34 * les points des courbes ou d'en enlever.
35 *
36 * L'utilisation se fait via l'appel à ITimeHistoryMng::applyTransformation() qui
37 * va appeler La méthode transform() de cette instance pour chaque courbe.
38 *
39 * Il est permis de changer le nombre d'éléments de la courbe, mais
40 * infos.iterations.size()*infos.sub_size et values.size() doivent avoir le même nombre
41 * d'éléments. Il n'est pas permis de changer le nom de la courbe ni
42 * le nombre de valeurs par itération (sub_size).
43 */
44class ITimeHistoryTransformer
45{
46 public:
47
48 //! Infos communes à chaque courbe
49 class CommonInfo
50 {
51 public:
52 //! Nom de la courbe
53 String name;
54 //! Liste des itérations
55 Int32SharedArray iterations;
56 //! Nombre de valeurs par courbe
57 Integer sub_size;
58 };
59
60 public:
61
62 virtual ~ITimeHistoryTransformer(){} //!< Libère les ressources
63
64 public:
65
66 //! Applique la transformation pour une courbe avec des valeurs \a Real
67 virtual void transform(CommonInfo& infos,RealSharedArray values) =0;
68
69 //! Applique la transformation pour une courbe avec des valeurs \a Int32
70 virtual void transform(CommonInfo& infos,Int32SharedArray values) =0;
71
72 //! Applique la transformation pour une courbe avec des valeurs \a Int64
73 virtual void transform(CommonInfo& infos,Int64SharedArray values) =0;
74};
