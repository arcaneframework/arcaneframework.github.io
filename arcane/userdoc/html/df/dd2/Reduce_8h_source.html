81 template<>
82 class ReduceAtomicSum<double>
83 {
84  public:
85  static double apply(std::atomic<double>* ptr,double v)
86  {
87  double old = ptr->load(std::memory_order_consume);
88  double wanted = old + v;
89  while (!ptr->compare_exchange_weak(old,wanted,std::memory_order_release,std::memory_order_consume))
90  wanted = old + v;
91  return wanted;
92  }
93 };
94 template<>
95 class ReduceAtomicSum<Int64>
96 {
97  public:
98  static Int64 apply(std::atomic<Int64>* ptr,Int64 v)
99  {
100  Int64 x = std::atomic_fetch_add(ptr,v);
101  return x+v;
102  }
103 };

137 template<typename DataType>
138 class ReduceFunctorMax
139 {
140  public:
141  static ARCCORE_DEVICE
142  DataType applyDevice(DataType* ptr,DataType v,DataType identity)
143  {
144  _applyDevice(ptr,v,identity);
145  return *ptr;
146  }
147  static DataType apply(std::atomic<DataType>* ptr,DataType v)
148  {
149  DataType prev_value = *ptr;
150  while(prev_value < v && !ptr->compare_exchange_weak(prev_value, v)) {}
151  return *ptr;
152  }
153  public:
154  ARCCORE_HOST_DEVICE static constexpr
155  DataType identity() { return impl::ReduceIdentity<DataType>::maxValue(); }
156  private:
157  static ARCCORE_DEVICE
