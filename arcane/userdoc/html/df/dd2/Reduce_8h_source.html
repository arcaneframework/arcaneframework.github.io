113 class ReduceAtomicSum;
114 
115 template<>
116 class ReduceAtomicSum<double>
117 {
118  public:
119  static double apply(std::atomic<double>* ptr,double v)
120  {
121  double old = ptr->load(std::memory_order_consume);
122  double wanted = old + v;
123  while (!ptr->compare_exchange_weak(old,wanted,std::memory_order_release,std::memory_order_consume))
124  wanted = old + v;
125  return wanted;
126  }
127 };
128 template<>
129 class ReduceAtomicSum<Int64>
130 {
131  public:
132  static Int64 apply(std::atomic<Int64>* ptr,Int64 v)
133  {
134  Int64 x = std::atomic_fetch_add(ptr,v);
135  return x+v;
136  }
137 };
138 template<>
139 class ReduceAtomicSum<Int32>
140 {
141  public:
142  static Int32 apply(std::atomic<Int32>* ptr,Int32 v)
143  {
144  Int32 x = std::atomic_fetch_add(ptr,v);
145  return x+v;
146  }
147 };

171 template<typename DataType>
172 class ReduceFunctorMax
173 {
174  public:
175  static ARCCORE_DEVICE
176  DataType applyDevice(const ReduceDeviceInfo<DataType>& dev_info)
177  {
178  _applyDevice(dev_info);
179  return *(dev_info.m_final_ptr);
180  }
181  static DataType apply(std::atomic<DataType>* ptr,DataType v)
182  {
183  DataType prev_value = *ptr;
184  while(prev_value < v && !ptr->compare_exchange_weak(prev_value, v)) {}
185  return *ptr;
186  }
187  public:
188  ARCCORE_HOST_DEVICE static constexpr
189  DataType identity() { return impl::ReduceIdentity<DataType>::maxValue(); }
190  private:
191  static ARCCORE_DEVICE
192  void _applyDevice(const ReduceDeviceInfo<DataType>& dev_info);
193 };
