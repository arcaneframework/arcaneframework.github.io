54 // NOTE: cette classe devrait être privée mais ce n'est pas possible avec CUDA.
55 //! Itérateur sur la lambda via un index
56 template <typename SetterLambda>
57 class SetterLambdaIterator
58 {
59 public:
60
61 //! Permet de positionner un élément de l'itérateur de sortie
62 class Setter
63 {
64 public:
65
66 ARCCORE_HOST_DEVICE explicit Setter(const SetterLambda& s, Int32 output_index)
67 : m_output_index(output_index)
68 , m_lambda(s)
69 {}
70 ARCCORE_HOST_DEVICE void operator=(Int32 input_index)
71 {
72 m_lambda(input_index, m_output_index);
73 }
74 Int32 m_output_index = 0;
75 SetterLambda m_lambda;
76 };
77
78 using value_type = Int32;
79 using iterator_category = std::random_access_iterator_tag;
80 using reference = Setter;
81 using difference_type = ptrdiff_t;
82 using pointer = void;
83
84 using ThatClass = SetterLambdaIterator<SetterLambda>;
85
86 public:
87
88 ARCCORE_HOST_DEVICE SetterLambdaIterator(const SetterLambda& s)
89 : m_lambda(s)
90 {}
91 ARCCORE_HOST_DEVICE explicit SetterLambdaIterator(const SetterLambda& s, Int32 v)
92 : m_lambda(s)
93 , m_index(v)
94 {}
95
96 public:
97
98 ARCCORE_HOST_DEVICE SetterLambdaIterator<SetterLambda>& operator++()
99 {
100 ++m_index;
101 return (*this);
102 }
103 ARCCORE_HOST_DEVICE reference operator*() const
104 {
105 return Setter(m_lambda, m_index);
106 }
107 ARCCORE_HOST_DEVICE reference operator[](Int32 x) const { return Setter(m_lambda, m_index + x); }
108 ARCCORE_HOST_DEVICE friend ThatClass operator+(Int32 x, const ThatClass& iter)
109 {
110 return ThatClass(iter.m_lambda, iter.m_index + x);
111 }
112 ARCCORE_HOST_DEVICE friend ThatClass operator+(const ThatClass& iter, Int32 x)
113 {
114 return ThatClass(iter.m_lambda, iter.m_index + x);
115 }
116 ARCCORE_HOST_DEVICE Int32 operator-(const ThatClass& x) const
117 {
118 return m_index - x.m_index;
119 }
120 ARCCORE_HOST_DEVICE friend bool operator<(const ThatClass& iter1, const ThatClass& iter2)
121 {
122 return iter1.m_index < iter2.m_index;
123 }
124
125 private:
126
127 Int32 m_index = 0;
128 SetterLambda m_lambda;
129 };
130
131 public:
132
133 GenericFilteringBase();
134
135 protected:
136
137 Int32 _nbOutputElement() const;
138 void _allocate();
139
140 protected:
141
142 RunQueue* m_queue = nullptr;
143 GenericDeviceStorage m_algo_storage;
144 DeviceStorage<int> m_device_nb_out_storage;
145 NumArray<Int32, MDDim1> m_host_nb_out_storage;
146};

152//! Implémentation pour SYCL
153class SyclGenericFilteringImpl
154{
155 public:
156
157 template <typename SelectLambda, typename InputIterator, typename OutputIterator>
158 static void apply(GenericFilteringBase& s, Int32 nb_item, InputIterator input_iter,
159 OutputIterator output_iter, SelectLambda select_lambda)
160 {
161 RunQueue* queue = s.m_queue;
162 using DataType = std::iterator_traits<OutputIterator>::value_type;
163#if defined(ARCANE_USE_SCAN_ONEDPL) && defined(__INTEL_LLVM_COMPILER)
164 sycl::queue true_queue = impl::SyclUtils::toNativeStream(queue);
165 auto policy = oneapi::dpl::execution::make_device_policy(true_queue);
166 auto out_iter = oneapi::dpl::copy_if(policy, input_iter, input_iter + nb_item, output_iter, select_lambda);
167 Int32 nb_output = out_iter - output_iter;
168 s.m_host_nb_out_storage[0] = nb_output;
169#else
170 NumArray<Int32, MDDim1> scan_input_data(nb_item);
171 NumArray<Int32, MDDim1> scan_output_data(nb_item);
172 SmallSpan<Int32> in_scan_data = scan_input_data.to1DSmallSpan();
173 SmallSpan<Int32> out_scan_data = scan_output_data.to1DSmallSpan();
174 {
175 auto command = makeCommand(queue);
176 command << RUNCOMMAND_LOOP1(iter, nb_item)
177 {
178 auto [i] = iter();
179 in_scan_data[i] = select_lambda(input_iter[i]) ? 1 : 0;
180 };
181 }
182 queue->barrier();
183 SyclScanner<false /*is_exclusive*/, Int32, ScannerSumOperator<Int32>> scanner;
184 scanner.doScan(*queue, in_scan_data, out_scan_data, 0);
185 // La valeur de 'out_data' pour le dernier élément (nb_item-1) contient la taille du filtre
186 Int32 nb_output = out_scan_data[nb_item - 1];
187 s.m_host_nb_out_storage[0] = nb_output;
188
189 const bool do_verbose = false;
190 if (do_verbose && nb_item < 1500)
191 for (int i = 0; i < nb_item; ++i) {
192 std::cout << "out_data i=" << i << " out_data=" << out_scan_data[i]
193 << " in_data=" << in_scan_data[i] << " value=" << input_iter[i] << "\n ";
194 }
195 // Copie depuis 'out_data' vers 'in_data' les indices correspondant au filtre
196 // Comme 'output_iter' et 'input_iter' peuvent se chevaucher, il
197 // faut faire une copie intermédiaire
198 // TODO: détecter cela et ne faire la copie que si nécessaire.
199 NumArray<DataType,MDDim1> out_copy(eMemoryRessource::Device);
200 out_copy.resize(nb_output);
201 auto out_copy_view = out_copy.to1DSpan();
202 {
203 auto command = makeCommand(queue);
204 command << RUNCOMMAND_LOOP1(iter, nb_item)
205 {
206 auto [i] = iter();
207 if (in_scan_data[i] == 1)
208 out_copy_view[out_scan_data[i] - 1] = input_iter[i];
209 };
210 }
211 {
212 auto command = makeCommand(queue);
213 command << RUNCOMMAND_LOOP1(iter, nb_output)
214 {
215 auto [i] = iter();
216 output_iter[i] = out_copy_view[i];
217 };
218 }
219 // Obligatoire à cause de 'out_copy'. On pourra le supprimer avec une
220 // allocation temporaire.
221 queue->barrier();
222#endif
223 }
224};

229/*!
230 * \internal
231 * \brief Classe pour effectuer un filtrage
232 *
233 * \a DataType est le type de donnée.
234 * \a FlagType est le type du tableau de filtre.
235 */
236template <typename DataType, typename FlagType, typename OutputDataType>
237class GenericFilteringFlag
238{
239 public:
240
241 void apply(GenericFilteringBase& s, SmallSpan<const DataType> input, SmallSpan<OutputDataType> output, SmallSpan<const FlagType> flag)
242 {
243 const Int32 nb_item = input.size();
244 if (output.size() != nb_item)
245 ARCANE_FATAL("Sizes are not equals: input={0} output={1}", nb_item, output.size());
246 [[maybe_unused]] const DataType* input_data = input.data();
247 [[maybe_unused]] DataType* output_data = output.data();
248 [[maybe_unused]] const FlagType* flag_data = flag.data();
249 eExecutionPolicy exec_policy = eExecutionPolicy::Sequential;
250 RunQueue* queue = s.m_queue;
251 ARCANE_CHECK_POINTER(queue);
252 exec_policy = queue->executionPolicy();
253 switch (exec_policy) {
254#if defined(ARCANE_COMPILING_CUDA)
255 case eExecutionPolicy::CUDA: {
256 size_t temp_storage_size = 0;
257 cudaStream_t stream = impl::CudaUtils::toNativeStream(queue);
258 // Premier appel pour connaitre la taille pour l'allocation
259 int* nb_out_ptr = nullptr;
260 ARCANE_CHECK_CUDA(::cub::DeviceSelect::Flagged(nullptr, temp_storage_size,
261 input_data, flag_data, output_data, nb_out_ptr, nb_item, stream));
262
263 s.m_algo_storage.allocate(temp_storage_size);
264 nb_out_ptr = s.m_device_nb_out_storage.allocate();
265 ARCANE_CHECK_CUDA(::cub::DeviceSelect::Flagged(s.m_algo_storage.address(), temp_storage_size,
266 input_data, flag_data, output_data, nb_out_ptr, nb_item, stream));
267 s.m_device_nb_out_storage.copyToAsync(s.m_host_nb_out_storage, queue);
268 } break;
269#endif
270#if defined(ARCANE_COMPILING_HIP)
271 case eExecutionPolicy::HIP: {
272 size_t temp_storage_size = 0;
273 // Premier appel pour connaitre la taille pour l'allocation
274 hipStream_t stream = impl::HipUtils::toNativeStream(queue);
275 int* nb_out_ptr = nullptr;
276 ARCANE_CHECK_HIP(rocprim::select(nullptr, temp_storage_size, input_data, flag_data, output_data,
277 nb_out_ptr, nb_item, stream));
278
279 s.m_algo_storage.allocate(temp_storage_size);
280 nb_out_ptr = s.m_device_nb_out_storage.allocate();
281
282 ARCANE_CHECK_HIP(rocprim::select(s.m_algo_storage.address(), temp_storage_size, input_data, flag_data, output_data,
283 nb_out_ptr, nb_item, stream));
284 s.m_device_nb_out_storage.copyToAsync(s.m_host_nb_out_storage, queue);
285 } break;
286#endif
287#if defined(ARCANE_COMPILING_SYCL)
288 case eExecutionPolicy::SYCL: {
289 impl::IndexIterator iter2(0);
290 auto filter_lambda = [=](Int32 input_index) -> bool { return flag[input_index] != 0; };
291 auto setter_lambda = [=](Int32 input_index, Int32 output_index) { output[output_index] = input[input_index]; };
292 GenericFilteringBase::SetterLambdaIterator<decltype(setter_lambda)> out(setter_lambda);
293 SyclGenericFilteringImpl::apply(s, nb_item, iter2, out, filter_lambda);
294 } break;
295#endif
296 case eExecutionPolicy::Thread:
297 // Pas encore implémenté en multi-thread
298 [[fallthrough]];
299 case eExecutionPolicy::Sequential: {
300 Int32 index = 0;
301 for (Int32 i = 0; i < nb_item; ++i) {
302 if (flag[i] != 0) {
303 output[index] = input[i];
304 ++index;
305 }
306 }
307 s.m_host_nb_out_storage[0] = index;
308 } break;
309 default:
310 ARCANE_FATAL(getBadPolicyMessage(exec_policy));
311 }
312 }
313};

317/*!
318 * \internal
319 * \brief Classe pour effectuer un filtrage
320 *
321 * \a DataType est le type de donnée.
322 * \a FlagType est le type du tableau de filtre.
323 */
324class GenericFilteringIf
325{
326 public:
327
328 template <typename SelectLambda, typename InputIterator, typename OutputIterator>
329 void apply(GenericFilteringBase& s, Int32 nb_item, InputIterator input_iter, OutputIterator output_iter,
330 const SelectLambda& select_lambda, const TraceInfo& trace_info)
331 {
332 eExecutionPolicy exec_policy = eExecutionPolicy::Sequential;
333 RunQueue* queue = s.m_queue;
334 ARCANE_CHECK_POINTER(queue);
335 exec_policy = queue->executionPolicy();
336 RunCommand command = makeCommand(*queue);
337 command << trace_info;
338 impl::RunCommandLaunchInfo launch_info(command, nb_item);
339 launch_info.beginExecute();
340 switch (exec_policy) {
341#if defined(ARCANE_COMPILING_CUDA)
342 case eExecutionPolicy::CUDA: {
343 size_t temp_storage_size = 0;
344 cudaStream_t stream = impl::CudaUtils::toNativeStream(queue);
345 // Premier appel pour connaitre la taille pour l'allocation
346 int* nb_out_ptr = nullptr;
347 ARCANE_CHECK_CUDA(::cub::DeviceSelect::If(nullptr, temp_storage_size,
348 input_iter, output_iter, nb_out_ptr, nb_item,
349 select_lambda, stream));
350
351 s.m_algo_storage.allocate(temp_storage_size);
352 s.m_device_nb_out_storage.allocate();
353 nb_out_ptr = s.m_device_nb_out_storage.address();
354 ARCANE_CHECK_CUDA(::cub::DeviceSelect::If(s.m_algo_storage.address(), temp_storage_size,
355 input_iter, output_iter, nb_out_ptr, nb_item,
356 select_lambda, stream));
357 s.m_device_nb_out_storage.copyToAsync(s.m_host_nb_out_storage, queue);
358 } break;
359#endif
360#if defined(ARCANE_COMPILING_HIP)
361 case eExecutionPolicy::HIP: {
362 size_t temp_storage_size = 0;
363 // Premier appel pour connaitre la taille pour l'allocation
364 hipStream_t stream = impl::HipUtils::toNativeStream(queue);
365 int* nb_out_ptr = nullptr;
366 ARCANE_CHECK_HIP(rocprim::select(nullptr, temp_storage_size, input_iter, output_iter,
367 nb_out_ptr, nb_item, select_lambda, stream));
368
369 s.m_algo_storage.allocate(temp_storage_size);
370 nb_out_ptr = s.m_device_nb_out_storage.allocate();
371 ARCANE_CHECK_HIP(rocprim::select(s.m_algo_storage.address(), temp_storage_size, input_iter, output_iter,
372 nb_out_ptr, nb_item, select_lambda, 0));
373 s.m_device_nb_out_storage.copyToAsync(s.m_host_nb_out_storage, queue);
374 } break;
375#endif
376#if defined(ARCANE_COMPILING_SYCL)
377 case eExecutionPolicy::SYCL: {
378 SyclGenericFilteringImpl::apply(s, nb_item, input_iter, output_iter, select_lambda);
379 } break;
380#endif
381 case eExecutionPolicy::Thread:
382 // Pas encore implémenté en multi-thread
383 [[fallthrough]];
384 case eExecutionPolicy::Sequential: {
385 Int32 index = 0;
386 for (Int32 i = 0; i < nb_item; ++i) {
387 if (select_lambda(*input_iter)) {
388 *output_iter = *input_iter;
389 ++index;
390 ++output_iter;
391 }
392 ++input_iter;
393 }
394 s.m_host_nb_out_storage[0] = index;
395 } break;
396 default:
397 ARCANE_FATAL(getBadPolicyMessage(exec_policy));
398 }
399 launch_info.endExecute();
400 }
401};

514/*!
515 * \brief Algorithme générique de filtrage sur accélérateur.
516 */
517class GenericFilterer
518: private impl::GenericFilteringBase
519{
520
521 public:
522
523 /*!
524 * \brief Créé une instance.
525 *
526 * \pre queue!=nullptr
527 */
528 explicit GenericFilterer(RunQueue* queue)
529 {
530 ARCANE_CHECK_POINTER(queue);
531 m_queue = queue;
532 _allocate();
533 }
534
535 public:
536
537 /*!
538 * \brief Applique un filtre.
539 *
540 * Filtre tous les éléments de \a input pour lesquels \a flag est différent de 0 et
541 * remplit \a output avec les valeurs filtrées. \a output doit avoir une taille assez
542 * grande pour contenir tous les éléments filtrés.
543 *
544 * L'algorithme séquentiel est le suivant:
545 *
546 * \code
547 * Int32 index = 0;
548 * for (Int32 i = 0; i < nb_item; ++i) {
549 * if (flag[i] != 0) {
550 * output[index] = input[i];
551 * ++index;
552 * }
553 * }
554 * return index;
555 * \endcode
556 *
557 * Il faut appeler la méthode nbOutputElement() pour obtenir le nombre d'éléments
558 * après filtrage.
559 */
560 template <typename InputDataType, typename OutputDataType, typename FlagType>
561 void apply(SmallSpan<const InputDataType> input, SmallSpan<OutputDataType> output, SmallSpan<const FlagType> flag)
562 {
563 const Int32 nb_value = input.size();
564 if (output.size() != nb_value)
565 ARCANE_FATAL("Sizes are not equals: input={0} output={1}", nb_value, output.size());
566 if (flag.size() != nb_value)
567 ARCANE_FATAL("Sizes are not equals: input={0} flag={1}", nb_value, flag.size());
568
569 _setCalled();
570 if (_checkEmpty(nb_value))
571 return;
572 impl::GenericFilteringBase* base_ptr = this;
573 impl::GenericFilteringFlag<InputDataType, FlagType, OutputDataType> gf;
574 gf.apply(*base_ptr, input, output, flag);
575 }
576
577 /*!
578 * \brief Applique un filtre.
579 *
580 * Filtre tous les éléments de \a input pour lesquels \a select_lambda vaut \a true et
581 * remplit \a output avec les valeurs filtrées. \a output doit avoir une taille assez
582 * grande pour contenir tous les éléments filtrés.
583 *
584 * \a select_lambda doit avoir un opérateur `ARCCORE_HOST_DEVICE bool operator()(const DataType& v) const`.
585 *
586 * Par exemple la lambda suivante permet de ne garder que les éléments dont
587 * la valeur est supérieure à 569.
588 *
589 * \code
590 * auto filter_lambda = [] ARCCORE_HOST_DEVICE (const DataType& x) -> bool {
591 * return (x > 569.0);
592 * };
593 * \endcode
594 *
595 * L'algorithme séquentiel est le suivant:
596 *
597 * \code
598 * Int32 index = 0;
599 * for (Int32 i = 0; i < nb_item; ++i) {
600 * if (select_lambda(input[i])) {
601 * output[index] = input[i];
602 * ++index;
603 * }
604 * }
605 * return index;
606 * \endcode
607 *
608 * Il faut appeler la méthode nbOutputElement() pour obtenir le nombre d'éléments
609 * après filtrage.
610 */
611 template <typename DataType, typename SelectLambda>
612 void applyIf(SmallSpan<const DataType> input, SmallSpan<DataType> output,
613 const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
614 {
615 const Int32 nb_value = input.size();
616 if (output.size() != nb_value)
617 ARCANE_FATAL("Sizes are not equals: input={0} output={1}", nb_value, output.size());
618
619 _setCalled();
620 if (_checkEmpty(nb_value))
621 return;
622 impl::GenericFilteringBase* base_ptr = this;
623 impl::GenericFilteringIf gf;
624 gf.apply(*base_ptr, nb_value, input.data(), output.data(), select_lambda, trace_info);
625 }
626
627 /*!
628 * \brief Applique un filtre.
629 *
630 * Cette méthode est identique à Filterer::applyIf(SmallSpan<const DataType> input,
631 * SmallSpan<DataType> output, const SelectLambda& select_lambda) mais permet de spécifier un
632 * itérateur \a input_iter pour l'entrée et \a output_iter pour la sortie.
633 * Le nombre d'entité en entrée est donné par \a nb_value.
634 */
635 template <typename InputIterator, typename OutputIterator, typename SelectLambda>
636 void applyIf(Int32 nb_value, InputIterator input_iter, OutputIterator output_iter,
637 const SelectLambda& select_lambda, const TraceInfo& trace_info = TraceInfo())
638 {
639 _setCalled();
640 if (_checkEmpty(nb_value))
641 return;
642 impl::GenericFilteringBase* base_ptr = this;
643 impl::GenericFilteringIf gf;
644 gf.apply(*base_ptr, nb_value, input_iter, output_iter, select_lambda, trace_info);
645 }
646
647 /*!
648 * \brief Applique un filtre avec une sélection suivant un index.
649 *
650 * Cette méthode permet de filtrer en spécifiant une fonction lambda à la fois
651 * pour la sélection et l'affection. Le prototype de ces lambda est:
652 *
653 * \code
654 * auto select_lambda = [=] ARCCORE_HOST_DEVICE (Int32 index) -> bool;
655 * auto setter_lambda = [=] ARCCORE_HOST_DEVICE (Int32 input_index,Int32 output_index) -> void;
656 * \endcode
657 *
658 * Par exemple, si on souhaite recopier dans \a output le tableau \a input dont les
659 * valeurs sont supérieures à 25.0.
660 *
661 * \code
662 * SmallSpan<Real> input = ...;
663 * SmallSpan<Real> output = ...;
664 * auto select_lambda = [=] ARCCORE_HOST_DEVICE (Int32 index) -> bool
665 * {
666 * return input[index] > 25.0;
667 * };
668 * auto setter_lambda = [=] ARCCORE_HOST_DEVICE (Int32 input_index,Int32 output_index)
669 * {
670 * output[output_index] = input[input_index];
671 * };
672 * Arcane::Accelerator::RunQueue* queue = ...;
673 * Arcane::Accelerator::GenericFilterer filterer(queue);
674 * filterer.applyWithIndex(input.size(), select_lambda, setter_lambda);
675 * Int32 nb_out = filterer.nbOutputElement();
676 * \endcode
677 */
678 template <typename SelectLambda, typename SetterLambda>
679 void applyWithIndex(Int32 nb_value, const SelectLambda& select_lambda,
680 const SetterLambda& setter_lambda, const TraceInfo& trace_info = TraceInfo())
681 {
682 _setCalled();
683 if (_checkEmpty(nb_value))
684 return;
685 impl::GenericFilteringBase* base_ptr = this;
686 impl::GenericFilteringIf gf;
687 impl::IndexIterator input_iter;
688 SetterLambdaIterator<SetterLambda> out(setter_lambda);
689 gf.apply(*base_ptr, nb_value, input_iter, out, select_lambda, trace_info);
690 }
691
692 //! Nombre d'éléments en sortie.
693 Int32 nbOutputElement()
694 {
695 m_is_already_called = false;
696 return _nbOutputElement();
697 }
698
699 private:
700
701 bool m_is_already_called = false;
702
703 private:
704
705 void _setCalled()
706 {
707 if (m_is_already_called)
708 ARCANE_FATAL("apply() has already been called for this instance");
709 m_is_already_called = true;
710 }
711 bool _checkEmpty(Int32 nb_value)
712 {
713 if (nb_value == 0) {
714 m_host_nb_out_storage[0] = 0;
715 return true;
716 }
717 return false;
718 }
719};
