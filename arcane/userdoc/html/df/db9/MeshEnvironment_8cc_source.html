58class MeshEnvironmentObserver
59: public TraceAccessor
60, public IItemGroupObserver
61{
62 public:
63
64 MeshEnvironmentObserver(MeshEnvironment* env, ITraceMng* tm)
65 : TraceAccessor(tm)
66 , m_environment(env)
67 {}
68
69 public:
70
71 void executeExtend(const Int32ConstArrayView* info1) override
72 {
73 if (info1) {
74 //info(4) << "EXTEND_ENV " << m_environment->name() << " ids=" << (*info1);
75 if (m_environment->materialMng()->isInMeshMaterialExchange())
76 info() << "EXTEND_ENV_IN_LOADBALANCE " << m_environment->name()
77 << " ids=" << (*info1);
78 }
79 }
80 void executeReduce(const Int32ConstArrayView* info1) override
81 {
82 if (info1) {
83 //info(4) << "REDUCE_ENV " << m_environment->name() << " ids=" << (*info1);
84 if (m_environment->materialMng()->isInMeshMaterialExchange())
85 info() << "REDUCE_ENV_IN_LOADBALANCE " << m_environment->name()
86 << " ids=" << (*info1);
87 }
88 }
89 void executeCompact(const Int32ConstArrayView* info1) override
90 {
91 info(4) << "COMPACT_ENV " << m_environment->name();
92 if (!info1)
93 ARCANE_FATAL("No info available");
94 Int32ConstArrayView old_to_new_ids(*info1);
95 m_environment->notifyLocalIdsChanged(old_to_new_ids);
96 }
97 void executeInvalidate() override
98 {
99 info() << "WARNING: invalidate() is invalid on an environment group! partial values may be corrupted"
100 << " env=" << m_environment->name();
101 }
102 bool needInfo() const override { return true; }
103
104 private:
105
106 MeshEnvironment* m_environment;
107};

131void MeshEnvironment::
132build()
133{
134 IMesh* mesh = m_material_mng->mesh();
135 IItemFamily* cell_family = mesh->cellFamily();
136 String group_name = m_material_mng->name() + "_" + name();
137 CellGroup cells = cell_family->findGroup(group_name, true);
138 cells._internalApi()->setAsConstituentGroup();
139
140 if (m_material_mng->isMeshModificationNotified()) {
141 m_group_observer = new MeshEnvironmentObserver(this, traceMng());
142 cells.internal()->attachObserver(this, m_group_observer);
143 }
144
145 m_data._setItems(cells);
146}

161void MeshEnvironment::
162setVariableIndexer(MeshMaterialVariableIndexer* idx)
163{
164 m_data._setVariableIndexer(idx);
165 idx->setCells(m_data.items());
166 idx->setIsEnvironment(true);
167
168 // S'il n'y qu'un matériau, le variable indexer de ce matériau est
169 // aussi 'idx' mais avec un autre groupe associé. Pour que tout soit
170 // cohérent, il faut être sur que ce matériau a aussi le même groupe.
171 // TODO: pour garantir la cohérence, il faudrait supprimer
172 // dans m_data le groupe d'entité.
173 if (m_true_materials.size() == 1)
174 m_true_materials[0]->componentData()->_setItems(m_data.items());
175 m_data._buildPartData();
176 for (MeshMaterial* mat : m_true_materials)
177 mat->componentData()->_buildPartData();
178}

183void MeshEnvironment::
184computeNbMatPerCell()
185{
186 info(4) << "ComputeNbMatPerCell env=" << name();
187 Integer nb_mat = m_materials.size();
188 Integer total = 0;
189 for (Integer i = 0; i < nb_mat; ++i) {
190 IMeshMaterial* mat = m_materials[i];
191 CellGroup mat_cells = mat->cells();
192 total += mat_cells.size();
193 }
194 m_total_nb_cell_mat = total;
195}

201 *
202 * Cette méthode est appelée par le MeshMaterialMng et doit être appelée
203 * une fois que les m_items_internal ont été mis à jour et
204 * computeNbMatPerCell() et computeItemListForMaterials() ont été appelées
205 */
206void MeshEnvironment::
207computeMaterialIndexes(ComponentItemInternalData* item_internal_data, RunQueue& queue)
208{
209 info(4) << "Compute (V2) indexes for environment name=" << name();
210 const bool is_mono_mat = isMonoMaterial();
211 if (is_mono_mat) {
212 _computeMaterialIndexesMonoMat(item_internal_data, queue);
213 }
214 else
215 _computeMaterialIndexes(item_internal_data, queue);
216}

218/*---------------------------------------------------------------------------*/
219/*---------------------------------------------------------------------------*/
220
221void MeshEnvironment::
222_computeMaterialIndexes(ComponentItemInternalData* item_internal_data, RunQueue& queue)
223{
224 IItemFamily* cell_family = cells().itemFamily();
225 Integer max_local_id = cell_family->maxLocalId();
226 const Int16 env_id = componentId();
227
228 ComponentItemInternalRange mat_items_internal_range = item_internal_data->matItemsInternalRange(id());
229
230 UniqueArray<Int32> cells_index(platform::getDefaultDataAllocator());
231 cells_index.resize(max_local_id);
232 UniqueArray<Int32> cells_pos(platform::getDefaultDataAllocator());
233 cells_pos.resize(max_local_id);
234
235 // TODO: regarder comment supprimer ce tableau cells_env qui n'est normalement pas utile
236 // car on doit pouvoir directement utiliser les m_items_internal
237 UniqueArray<ConstituentItemIndex> cells_env(platform::getDefaultDataAllocator());
238 cells_env.resize(max_local_id);
239
240 Int32ConstArrayView local_ids = variableIndexer()->localIds();
241 ConstituentItemLocalIdListView constituent_item_list_view = m_data.constituentItemListView();
242 const bool do_old = false;
243 if (do_old) {
244 Integer cell_index = 0;
245 for (Integer z = 0, nb = local_ids.size(); z < nb; ++z) {
246 Int32 lid = local_ids[z];
247 matimpl::ConstituentItemBase env_item = constituent_item_list_view._constituenItemBase(z);
248 Int32 nb_mat = env_item.nbSubItem();
249 cells_index[lid] = cell_index;
250 cell_index += nb_mat;
251 }
252 }
253 else {
254 // Calcul l'indice de la première MatCell pour chaque Cell.
255 Int32 nb_id = local_ids.size();
256 {
257 Accelerator::GenericScanner scanner(queue);
258 auto cells_index_view = viewOut(queue, cells_index);
259
260 auto getter = [=] ARCCORE_HOST_DEVICE(Int32 index) -> Int32 {
261 return constituent_item_list_view._constituenItemBase(index).nbSubItem();
262 };
263 auto setter = [=] ARCCORE_HOST_DEVICE(Int32 index, Int32 value) {
264 Int32 lid = local_ids[index];
265 cells_index_view[lid] = value;
266 };
267 Accelerator::ScannerSumOperator<Int32> op;
268 scanner.applyWithIndexExclusive(nb_id, 0, getter, setter, op, A_FUNCINFO);
269 }
270 }
271 {
272 auto command = makeCommand(queue);
273 auto cells_env_view = viewOut(command, cells_env);
274 auto cells_index_view = viewIn(command, cells_index);
275 auto cells_pos_view = viewOut(command, cells_pos);
276 Int32 nb_id = local_ids.size();
277
278 command << RUNCOMMAND_LOOP1(iter, nb_id)
279 {
280 auto [z] = iter();
281 Int32 lid = local_ids[z];
282 matimpl::ConstituentItemBase env_item = constituent_item_list_view._constituenItemBase(z);
283 Int32 nb_mat = env_item.nbSubItem();
284 Int32 cell_index = cells_index_view[lid];
285 cells_pos_view[lid] = cell_index;
286 if (nb_mat != 0) {
287 env_item._setFirstSubItem(mat_items_internal_range[cell_index]);
288 }
289 cells_env_view[lid] = env_item.constituentItemIndex();
290 };
291 }
292 {
293 Integer nb_mat = m_true_materials.size();
294 for (Integer i = 0; i < nb_mat; ++i) {
295 MeshMaterial* mat = m_true_materials[i];
296 Int16 mat_id = mat->componentId();
297 const MeshMaterialVariableIndexer* var_indexer = mat->variableIndexer();
298 CellGroup mat_cells = mat->cells();
299 info(4) << "COMPUTE (V2) mat_cells mat=" << mat->name() << " nb_cell=" << mat_cells.size()
300 << " mat_id=" << mat_id << " index=" << var_indexer->index();
301
302 mat->resizeItemsInternal(var_indexer->nbItem());
303
304 auto command = makeCommand(queue);
305 auto matvar_indexes = viewIn(command, var_indexer->matvarIndexes());
306 auto local_ids = viewIn(command, var_indexer->localIds());
307 SmallSpan<Int32> cells_pos_view(cells_pos);
308 auto cells_env_view = viewIn(command, cells_env);
309 ComponentItemSharedInfo* mat_shared_info = item_internal_data->matSharedInfo();
310 ComponentItemInternalRange mat_item_internal_range(item_internal_data->matItemsInternalRange(env_id));
311 SmallSpan<ConstituentItemIndex> mat_id_list = mat->componentData()->m_constituent_local_id_list.mutableLocalIds();
312 const Int32 nb_id = local_ids.size();
313 Span<Int32> mat_cells_local_id = mat_cells._internalApi()->itemsLocalId();
314 command << RUNCOMMAND_LOOP1(iter, nb_id)
315 {
316 auto [z] = iter();
317 MatVarIndex mvi = matvar_indexes[z];
318 Int32 lid = local_ids[z];
319 Int32 pos = cells_pos_view[lid];
320 ++cells_pos_view[lid];
321 ConstituentItemIndex cii = mat_item_internal_range[pos];
322 matimpl::ConstituentItemBase ref_ii(mat_shared_info, cii);
323 mat_id_list[z] = cii;
324 ref_ii._setSuperAndGlobalItem(cells_env_view[lid], ItemLocalId(lid));
325 ref_ii._setComponent(mat_id);
326 ref_ii._setVariableIndex(mvi);
327 // Le rang 0 met à jour le padding SIMD du groupe associé au matériau
328 if (z == 0)
329 ArraySimdPadder::applySimdPaddingView(mat_cells_local_id);
330 };
331 mat_cells._internalApi()->notifySimdPaddingDone();
332 }
333 }
334}

338/*!
339 * \brief Calcul les infos sur les matériaux en mono-matériaux.
340 *
341 * Spécialisation pour le cas où le milieu n'a qu'un matériau.
342 */
343void MeshEnvironment::
344_computeMaterialIndexesMonoMat(ComponentItemInternalData* item_internal_data, RunQueue& queue)
345{
346 const Int16 env_id = componentId();
347
348 ConstituentItemLocalIdListView constituent_item_list_view = m_data.constituentItemListView();
349
350 MeshMaterial* mat = m_true_materials[0];
351 const Int16 mat_id = mat->componentId();
352 const MeshMaterialVariableIndexer* var_indexer = mat->variableIndexer();
353 CellGroup mat_cells = mat->cells();
354 info(4) << "COMPUTE (V2) mat_cells mat=" << mat->name() << " nb_cell=" << mat_cells.size()
355 << " mat_id=" << mat_id << " index=" << var_indexer->index();
356
357 mat->resizeItemsInternal(var_indexer->nbItem());
358
359 auto command = makeCommand(queue);
360 auto matvar_indexes = viewIn(command, var_indexer->matvarIndexes());
361 auto local_ids = viewIn(command, var_indexer->localIds());
362 ComponentItemSharedInfo* mat_shared_info = item_internal_data->matSharedInfo();
363 ComponentItemInternalRange mat_item_internal_range(item_internal_data->matItemsInternalRange(env_id));
364 SmallSpan<ConstituentItemIndex> mat_id_list = mat->componentData()->m_constituent_local_id_list.mutableLocalIds();
365 const Int32 nb_id = local_ids.size();
366 Span<Int32> mat_cells_local_id = mat_cells._internalApi()->itemsLocalId();
367 command << RUNCOMMAND_LOOP1(iter, nb_id)
368 {
369 auto [z] = iter();
370 MatVarIndex mvi = matvar_indexes[z];
371 const Int32 lid = local_ids[z];
372 const Int32 pos = z;
373 matimpl::ConstituentItemBase env_item = constituent_item_list_view._constituenItemBase(z);
374 ConstituentItemIndex cii = mat_item_internal_range[pos];
375 env_item._setFirstSubItem(cii);
376
377 matimpl::ConstituentItemBase ref_ii(mat_shared_info, cii);
378 mat_id_list[z] = cii;
379 ref_ii._setSuperAndGlobalItem(env_item.constituentItemIndex(), ItemLocalId(lid));
380 ref_ii._setComponent(mat_id);
381 ref_ii._setVariableIndex(mvi);
382 // Le rang 0 met à jour le padding SIMD du groupe associé au matériau
383 if (z == 0)
384 ArraySimdPadder::applySimdPaddingView(mat_cells_local_id);
385 };
386 mat_cells._internalApi()->notifySimdPaddingDone();
387}

391/*!
392 * \brief Calcul pour les mailles des matériaux du milieu leur emplacement
393 * dans le tableau d'indexation des variables.
394 */
395void MeshEnvironment::
396computeItemListForMaterials(const ConstituentConnectivityList& connectivity_list)
397{
398 info(4) << "ComputeItemListForMaterials (V2)";
399 ConstArrayView<Int16> nb_env_per_cell = connectivity_list.cellsNbEnvironment();
400 const Int16 env_id = componentId();
401 // Calcul pour chaque matériau le nombre de mailles mixtes
402 // TODO: a faire dans MeshMaterialVariableIndexer
403 for (MeshMaterial* mat : m_true_materials) {
404 MeshMaterialVariableIndexer* var_indexer = mat->variableIndexer();
405 CellGroup cells = var_indexer->cells();
406 Integer var_nb_cell = cells.size();
407
408 ComponentItemListBuilderOld list_builder(var_indexer, 0);
409
410 info(4) << "MAT_INDEXER mat=" << mat->name() << " NB_CELL=" << var_nb_cell << " name=" << cells.name();
411 ENUMERATE_CELL (icell, cells) {
412 Int32 lid = icell.itemLocalId();
413 // On ne prend l'indice global que si on est le seul matériau et le seul
414 // milieu de la maille. Sinon, on prend un indice multiple
415 if (nb_env_per_cell[lid] > 1 || connectivity_list.cellNbMaterial(icell, env_id) > 1)
416 list_builder.addPartialItem(lid);
417 else
418 list_builder.addPureItem(lid);
419 }
420
421 if (traceMng()->verbosityLevel() >= 5)
422 info() << "MAT_NB_MULTIPLE_CELL (V2) mat=" << var_indexer->name()
423 << " nb_in_global=" << list_builder.pureMatVarIndexes().size()
424 << " (ids=" << list_builder.pureMatVarIndexes() << ")"
425 << " nb_in_multiple=" << list_builder.partialMatVarIndexes().size()
426 << " (ids=" << list_builder.partialLocalIds() << ")";
427 var_indexer->endUpdate(list_builder);
428 }
429}

434void MeshEnvironment::
435notifyLocalIdsChanged(Int32ConstArrayView old_to_new_ids)
436{
437 // NOTE:
438 // Cette méthode est appelée lorsqu'il y a un compactage du maillage
439 // et le groupe d'entité associé à ce milieu vient d'être compacté.
440 // Comme actuellement il n'y a pas d'observeurs pour l'ajout
441 // ou la suppression de mailles du groupe, il est possible
442 // lorsque cette méthode est appelée que les infos des milieux et
443 // matériaux ne soient pas à jour (par exemple, la liste des local_ids
444 // du m_variable_indexer n'a pas les mêmes valeurs que cells().
445 // Pour l'instant ce n'est pas très grave car tout est écrasé après
446 // chaque modif sur un matériau ou un milieu.
447 // A terme, il faudra prendre cela en compte lorsque l'ajout
448 // où la suppression de mailles matériaux sera optimisée.
449 info(4) << "Changing (V3) local ids references env=" << name();
450 info(4) << "CurrentCells name=" << cells().name()
451 << " n=" << cells().view().localIds().size();
452 info(4) << "MatVarIndex name=" << cells().name()
453 << " n=" << variableIndexer()->matvarIndexes().size();
454 Integer nb_mat = m_true_materials.size();
455 info(4) << "NotifyLocalIdsChanged env=" << name() << " nb_mat=" << nb_mat
456 << " old_to_new_ids.size=" << old_to_new_ids.size();
457
458 // Si le milieu n'a qu'un seul matériau, ils partagent le même variable_indexer
459 // donc il ne faut changer les ids qu'une seule fois. Par contre, le
460 // tableau m_items_internal n'est pas partagé entre le matériau
461 // et le milieu donc il faut recalculer les infos séparément.
462 // Il faut le faire pour le milieu avant de mettre à jour les infos du matériau car
463 // une fois ceci fait la valeur m_variable_indexer->m_local_ids_in_indexes_view
464 // aura changé et il ne sera plus possible de déterminer la correspondance
465 // entre les nouveaux et les anciens localId
466
467 if (nb_mat == 1) {
468 m_data._changeLocalIdsForInternalList(old_to_new_ids);
469 MeshMaterial* true_mat = m_true_materials[0];
470 _changeIds(true_mat->componentData(), old_to_new_ids);
471 }
472 else {
473 // Change les infos des matériaux
474 for (Integer i = 0; i < nb_mat; ++i) {
475 MeshMaterial* true_mat = m_true_materials[i];
476 info(4) << "ChangeIds MAT i=" << i << " MAT=" << true_mat->name();
477 _changeIds(true_mat->componentData(), old_to_new_ids);
478 }
479 // Change les infos du milieu
480 _changeIds(componentData(), old_to_new_ids);
481 }
482
483 // Reconstruit les infos sur les mailles pures et mixtes.
484 // Il faut le faire une fois que tous les valeurs sont à jour.
485 {
486 RunQueue& queue = m_material_mng->_internalApi()->runQueue();
487 for (Integer i = 0; i < nb_mat; ++i) {
488 MeshMaterial* true_mat = m_true_materials[i];
489 true_mat->componentData()->_rebuildPartData(queue);
490 }
491 componentData()->_rebuildPartData(queue);
492 }
493
494 checkValid();
495}
