79  if (m_value)
80  m_value->addRef();
81  }
83  void _removeRef()
84  {
85  if (m_value)
86  m_value->removeRef();
87  }
89  void _changeValue(T* new_value)
90  {
91  if (m_value==new_value)
92  return;
93  _removeRef();
94  m_value = new_value;
95  _addRef();
96  }
