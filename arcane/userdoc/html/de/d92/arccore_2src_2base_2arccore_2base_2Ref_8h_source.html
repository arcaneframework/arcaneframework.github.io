25/*!
26 * \brief Wrapper autour d'une classe gérant son propre compteur de référence.
27 *
28 * La classe \a InstanceType doit gérer son propre compteur de référence et
29 * sa propre destruction.
30 */
31template <typename InstanceType>
32class ReferenceCounterWrapper
33{
34 //! Vérifie que la classe 'InstanceType' a bien un typedef sur ReferenceCounterTag (\sa Ref)
35 inline static void _checkHasReferenceCounterTag()
36 {
37 static_assert(std::is_same_v<typename InstanceType::ReferenceCounterTagType, ReferenceCounterTag>, "Bad tag");
38 }
39
40 public:
41
42 //! Constructeur avec un deleter vide. Dans ce cas pas besoin de le conserver
43 ReferenceCounterWrapper(InstanceType* ptr, const RefBase::BasicDeleterBase&)
44 : m_instance(ptr)
45 {
46 _checkHasReferenceCounterTag();
47 }
48 template <typename U> // U doit dériver de \a RefBase::DeleterBase
49 ReferenceCounterWrapper(InstanceType* ptr, const U& deleter_base)
50 : m_instance(ptr)
51 {
52 // Ce constructeur est appelé lorqu'il y a une ExternalRef associé ou si on spécifie
53 // que l'objet doit être détruit manuellement.
54 _checkHasReferenceCounterTag();
55 m_instance->_internalReferenceCounter()->_setExternalDeleter(new RefBase::DeleterBase(deleter_base));
56 }
57 explicit ReferenceCounterWrapper(InstanceType* ptr)
58 : m_instance(ptr)
59 {
60 _checkHasReferenceCounterTag();
61 }
62 //! Autorise à convertir si 'T*' et 'InstanceType*' sont convertibles
63 template <typename T,
64 typename X = typename std::is_convertible<T*, InstanceType*>::type>
65 explicit ReferenceCounterWrapper(const ReferenceCounterWrapper<T>& r)
66 : m_instance(r.get())
67 {
68 _checkHasReferenceCounterTag();
69 }
70 ReferenceCounterWrapper() = default;
71
72 public:
73
74 //! Retourne l'instance
75 InstanceType* get() const { return m_instance.get(); }
76
77 //! Supprime la référence associée actuellement.
78 void reset() { m_instance = nullptr; }
79
80 RefBase::DeleterBase* getDeleter()
81 {
82 return m_instance.get()->_internalReferenceCounter()->_externalDeleter();
83 }
84
85 private:
86
87 Arccore::ReferenceCounter<InstanceType> m_instance;
88};

121/*!
122 * \brief Référence à une instance.
123 *
124 * Cette classe utilise un compteur de référence pour gérer la durée de vie
125 * d'une instance C++. Elle fonctionne de manière similaire à std::shared_ptr.
126 *
127 * Lorsque la dernière instance de cette classe est détruite, l'instance
128 * référencée est détruite. La manière de détruire l'instance associée
129 * est spécifié lors de la création de la première référence via l'appel
130 * à une des méthodes create() ou createWithHandle().
131 *
132 * Il existe deux implémentation possibles pour compter les références.
133 * Par défaut, on utilise 'std::shared_ptr'. Il est aussi possible
134 * d'utiliser un compteur de référence interne à la classe ce qui
135 * permet d'être compatible avec la classe ReferenceCounter et aussi de
136 * pouvoir récupérer une référence à partir de l'instance elle même. Cette
137 * deuxième implémentation est accessible si le type \a InstanceType
138 * définit un type ReferenceCounterTagType valant ReferenceCounterTag.
139 */
140template <typename InstanceType, int ImplTagId>
141class Ref
142: public RefBase
143{
144 public:
145
146 typedef Ref<InstanceType, ImplTagId> ThatClass;
147 typedef typename RefTraitsTagId<InstanceType, ImplTagId>::ImplType ImplType;
148
149 private:
150
151 template <typename... _Args>
152 using _IsRefConstructible = typename std::enable_if<std::is_constructible<ImplType, _Args...>::value>::type;
153
154 using RefBase::DeleterBase;
155 class Deleter : public DeleterBase
156 {
157 public:
158
159 Deleter() = default;
160 Deleter(Internal::ExternalRef h)
161 : DeleterBase(h)
162 {}
163 Deleter(Internal::ExternalRef h, bool no_destroy)
164 : DeleterBase(std::move(h), no_destroy)
165 {}
166 void operator()(InstanceType* tt)
167 {
168 if (m_no_destroy)
169 return;
170 bool is_destroyed = this->_destroyHandle(tt, m_handle);
171 if (!is_destroyed)
172 delete tt;
173 }
174 };
175
176 class BasicDeleter
177 : public BasicDeleterBase
178 {
179 public:
180
181 bool hasExternal() const { return false; }
182 void operator()(InstanceType* tt)
183 {
184 delete tt;
185 }
186 };
187
188 public:
189
190 static constexpr int RefType = RefTraitsTagId<InstanceType, ImplTagId>::RefType;
191
192 private:
193
194 explicit Ref(InstanceType* t)
195 : m_instance(t, _createBasicDeleter((ImplType*)nullptr)) //BasicDeleter{})
196 {}
197 Ref(InstanceType* t, Internal::ExternalRef handle)
198 : m_instance(t, Deleter(handle))
199 {}
200 Ref(InstanceType* t, bool no_destroy)
201 : m_instance(t, Deleter(nullptr, no_destroy))
202 {}
203
204 private:
205
206 Ref(ImplType&& t)
207 : m_instance(t)
208 {}
209
210 public:
211
212 /*!
213 * \brief Construit une référence issue d'une autre référence sur un type compatible.
214 *
215 * La conversion est autorisée si on peut construire une instance de 'ImplType'
216 * à partir de celle de celle de Ref<T>::ImplType.
217 */
218 template <typename T, typename = _IsRefConstructible<typename Ref<T>::ImplType>>
219 Ref(const Ref<T>& rhs) noexcept
220 : m_instance(rhs._internalInstance())
221 {}
222 Ref() = default;
223 Ref(const ThatClass& rhs) = default;
224 ThatClass& operator=(const ThatClass& rhs) = default;
225 ~Ref() = default;
226
227 public:
228
229 /*!
230 * \internal
231 * \brief Créé une référence à partir de l'instance \a t.
232 *
233 * Cette méthode est interne à %Arccore.
234 *
235 * L'instance \a t doit avoir été créée par l'opérateur 'operator new'
236 * et sera détruite par l'opérateur 'operator delete'
237 */
238 static ThatClass create(InstanceType* t)
239 {
240 return ThatClass(t);
241 }
242
243 template <typename PointerType, typename... Args>
244 static inline Ref<InstanceType>
245 createRef(Args&&... args)
246 {
247 PointerType* pt = new PointerType(std::forward<Args>(args)...);
248 return Ref<InstanceType>(pt);
249 }
250
251 /*!
252 * \internal
253 * \brief Créé une référence à partir d'une instance ayant une
254 * référence externe.
255 */
256 static ThatClass createWithHandle(InstanceType* t, Internal::ExternalRef handle)
257 {
258 return ThatClass(t, handle);
259 }
260
261 static ThatClass _createNoDestroy(InstanceType* t)
262 {
263 return ThatClass(t, true);
264 }
265
266 public:
267
268 friend inline bool operator==(const ThatClass& a, const ThatClass& b)
269 {
270 return a.get() == b.get();
271 }
272
273 friend inline bool operator!=(const ThatClass& a, const ThatClass& b)
274 {
275 return a.get() != b.get();
276 }
277
278 friend inline bool operator<(const ThatClass& a, const ThatClass& b)
279 {
280 return a.get() < b.get();
281 }
282
283 friend inline bool operator!(const ThatClass& a)
284 {
285 return a.isNull();
286 }
287
288 operator bool() const { return (!isNull()); }
289
290 public:
291
292 //! Instance associée ou `nullptr` si aucune
293 InstanceType* get() const { return m_instance.get(); }
294 //! Indique si le compteur référence une instance non nulle.
295 bool isNull() const { return m_instance.get() == nullptr; }
296 InstanceType* operator->() const { return m_instance.get(); }
297 //! Positionne l'instance au pointeur nul.
298 void reset() { m_instance.reset(); }
299 /*!
300 * \internal
301 * \brief Libère le pointeur du compteur de référence sans le détruire.
302 * Cela n'est autorisé que si l'implémentation utiliser 'std::shared_ptr'.
303 */
304 template<typename T = ThatClass, typename std::enable_if_t<T::RefType == REF_TAG_SHARED_PTR, bool> = true>
305 InstanceType* _release()
306 {
307 // Relâche l'instance. Pour cela, on indique au destructeur
308 // de ne pas détruire l'instance de 'm_instance' et on
309 // retourne cette dernière.
310 auto* r = _getDeleter(m_instance);
311 if (r)
312 r->setNoDestroy(true);
313 InstanceType* t = m_instance.get();
314 m_instance.reset();
315 return t;
316 }
317 const ImplType& _internalInstance() const { return m_instance; }
318
319 private:
320
321 ImplType m_instance;
322
323 private:
324
325 static Deleter _createBasicDeleter(std::shared_ptr<InstanceType>*)
326 {
327 return {};
328 }
329 static BasicDeleterBase _createBasicDeleter(impl::ReferenceCounterWrapper<InstanceType>*)
330 {
331 return {};
332 }
333 static Deleter* _getDeleter(std::shared_ptr<InstanceType>& v)
334 {
335 return std::get_deleter<Deleter>(v);
336 }
337 static DeleterBase* _getDeleter(impl::ReferenceCounterWrapper<InstanceType>& v)
338 {
339 return v.getDeleter();
340 }
341};

348/*!
349 * \brief Créé une référence sur un pointeur.
350 *
351 * La pointeur \a t doit avoir été alloué par l'operateur 'operator new' et
352 * sera détruit par l'opérateur 'operator delete' lorsqu'il n'y aura plus
353 * de référence dessus.
354 */
355template <typename InstanceType> inline auto
356makeRef(InstanceType* t) -> Ref<InstanceType>
357{
358 return Ref<InstanceType>::create(t);
359}
360

362/*---------------------------------------------------------------------------*/
363/*!
364 * \brief Récupère une référence sur le pointeur \a t.
365 *
366 * Cette méthode n'est disponible que si la classe InstanceType utilise un
367 * compteur de réference (ImplTagId==REF_TAG_REFERENCE_COUNTER).
368 *
369 * \code
370 * class A {};
371 * class B : public A {};
372 * Ref<B> rb = ...;
373 * B* b = rb.get();
374 * Ref<A> ra = makeRefFromInstance<A>(b);
375 * \endcode
376 */
377template <typename InstanceType,
378 typename InstanceType2,
379 typename std::enable_if_t<Ref<InstanceType>::RefType, int> = REF_TAG_REFERENCE_COUNTER>
380inline Ref<InstanceType>
381makeRefFromInstance(InstanceType2* t)
382{
383 return Ref<InstanceType>::create(t);
384}

388/*!
389 * \brief Créé une instance de type \a TrueType avec les arguments \a Args
390 * et retourne une référence dessus.
391 */
392template <typename TrueType, class... Args> inline Ref<TrueType>
393createRef(Args&&... args)
394{
395 return makeRef<TrueType>(new TrueType(std::forward<Args>(args)...));
