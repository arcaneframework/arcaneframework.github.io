34 class ItemVectorViewConstIterator
35 {
36  protected:
37  friend class ItemVectorView;
38  typedef ItemInternal* ItemInternalPtr;
39  ItemVectorViewConstIterator(const ItemInternalPtr* items,
40  const Int32* ARCANE_RESTRICT local_ids,
41  Integer index)
42  : m_items(items), m_local_ids(local_ids), m_index(index){}
43  public:
44  // Pas directement utilisé mais est nécessaire pour ICC 17.0
45  ItemVectorViewConstIterator()
46  : m_items(nullptr), m_local_ids(nullptr), m_index(0){}
47  public:
48  typedef ItemVectorViewConstIterator ThatClass;
49  public:
50  typedef std::random_access_iterator_tag iterator_category;
52  typedef const Item* pointer;
54  typedef const Item& reference;
56  typedef Item value_type;
58  typedef Integer size_type;
60  typedef Integer difference_type;
61  public:
62  Item operator*() const { return m_items[ m_local_ids[m_index] ]; }
63  Item operator->() const { return m_items[ m_local_ids[m_index] ]; }
64  ThatClass& operator++() { ++m_index; return (*this); }
65  ThatClass& operator--() { --m_index; return (*this); }
66  void operator+=(difference_type v) { m_index += v; }
67  void operator-=(difference_type v) { m_index -= v; }
68  Integer operator-(const ThatClass& b) const
69  {
70  return this->m_index - b.m_index;
71  }
72  friend ThatClass operator-(const ThatClass& a,difference_type v)
73  {
74  Integer index = a.m_index - v;
75  return ThatClass(a.m_items,a.m_local_ids,index);
76  }
77  friend ThatClass operator+(const ThatClass& a,difference_type v)
78  {
79  Integer index = a.m_index + v;
80  return ThatClass(a.m_items,a.m_local_ids,index);
81  }
82  friend bool operator<(const ThatClass& lhs,const ThatClass& rhs)
83  {
84  return lhs.m_index<=rhs.m_index;
85  }
86  friend bool operator==(const ThatClass& lhs,const ThatClass& rhs)
87  {
88  return lhs.m_items==rhs.m_items && lhs.m_local_ids==rhs.m_local_ids && lhs.m_index==rhs.m_index;
89  }
90  friend bool operator!=(const ThatClass& lhs,const ThatClass& rhs)
91  {
92  return !(lhs==rhs);
93  }
94  protected:
95  const ItemInternalPtr* m_items;
96  const Int32* ARCANE_RESTRICT m_local_ids;
97  Integer m_index;
98 };
