272  void _init()
273  {
274  m_is_not_end = (m_index<m_count);
275  if (m_is_not_end){
276  Int32 lid = m_local_ids[m_index];
277  // Vérifie qu'on n'indexe pas 'm_items' avec un localId() nul.
278  // Cela n'est pas possible avec les groupes d'entités mais ca l'est avec
279  // les ItemVector par exemple.
280  // De même, 'm_items' peut-être nul si tous les localId() sont nuls.
281  // Si c'est le cas, alors on prend comme ItemSharedInfo l'instance nulle.
282  // A terme le ItemSharedInfo sera renseigné directement dans le constructeur de
283  // cette classe.
284  Int32 idx = lid;
285  if (idx==NULL_ITEM_LOCAL_ID)
286  idx = 0;
287  ItemSharedInfo* isi = (m_items) ? m_items[idx]->sharedInfo() : ItemSharedInfo::nullInstance();
288  m_base = ItemType(ItemBaseBuildInfo(lid,isi));
289  }
290  }
