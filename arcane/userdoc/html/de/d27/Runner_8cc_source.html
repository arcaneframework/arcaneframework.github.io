40 inline impl::IRunnerRuntime*
41 _getRuntimeNoCheck(eExecutionPolicy p)
42 {
43 impl::IRunnerRuntime* runtime = nullptr;
44 switch (p) {
45 case eExecutionPolicy::None:
46 ARCANE_FATAL("No runtime for eExecutionPolicy::None");
47 case eExecutionPolicy::HIP:
48 return impl::getHIPRunQueueRuntime();
49 case eExecutionPolicy::CUDA:
50 return impl::getCUDARunQueueRuntime();
51 case eExecutionPolicy::Sequential:
52 return impl::getSequentialRunQueueRuntime();
53 case eExecutionPolicy::Thread:
54 return impl::getThreadRunQueueRuntime();
55 }
56 return runtime;
57 }
58
59 inline impl::IRunnerRuntime*
60 _getRuntime(eExecutionPolicy p)
61 {
62 auto* x = _getRuntimeNoCheck(p);
63 if (!x)
64 ARCANE_FATAL("No runtime is available for execution policy '{0}'", p);
65 return x;
66 }
67
68 inline void
69 _stopProfiling(eExecutionPolicy p)
70 {
71 auto* x = _getRuntimeNoCheck(p);
72 if (x)
73 x->stopProfiling();
74 }

88void RunnerImpl::
89initialize(Runner* runner, eExecutionPolicy v, DeviceId device)
90{
91 if (m_is_init)
92 ARCANE_FATAL("Runner is already initialized");
93 if (v == eExecutionPolicy::None)
94 ARCANE_THROW(ArgumentException, "executionPolicy should not be eExecutionPolicy::None");
95 if (device.isHost() || device.isNull())
96 ARCANE_THROW(ArgumentException, "device should not be Device::hostDevice() or Device::nullDevice()");
97
98 m_execution_policy = v;
99 m_device_id = device;
100 m_runtime = _getRuntime(v);
101 m_is_init = true;
102 m_is_auto_prefetch_command = false;
103
104 // Pour test
105 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_ACCELERATOR_PREFETCH_COMMAND", true))
106 m_is_auto_prefetch_command = (v.value() != 0);
107
108 // Il faut initialiser le pool à la fin car il a besoin d'accéder à \a m_runtime
109 m_run_queue_pool = new RunQueueImplStack(runner->_impl());
110}

159
160impl::RunQueueImpl* RunnerImpl::
161_internalCreateOrGetRunQueueImpl()
162{
163 _checkIsInit();
164
165 auto pool = getPool();
166
167 {
168 impl::RunnerImpl::Lock my_lock(this);
169 if (!pool->empty()) {
170 impl::RunQueueImpl* p = pool->top();
171 pool->pop();
172 return p;
173 }
174 }
175
176 return pool->createRunQueue(RunQueueBuildInfo{});
177}
178
179/*---------------------------------------------------------------------------*/
180/*---------------------------------------------------------------------------*/
181
182impl::RunQueueImpl* RunnerImpl::
183_internalCreateOrGetRunQueueImpl(const RunQueueBuildInfo& bi)
184{
185 _checkIsInit();
186 // Si on utilise les paramètres par défaut, on peut utilier une RunQueueImpl
187 // issue du pool.
188 if (bi.isDefault())
189 return _internalCreateOrGetRunQueueImpl();
190 impl::IRunnerRuntime* runtime = m_runtime;
191 ARCANE_CHECK_POINTER(runtime);
192 auto* queue = new impl::RunQueueImpl(this, 0, bi);
193 return queue;
194}

417const DeviceInfo& Runner::
418deviceInfo() const
419{
420 _checkIsInit();
421 const IDeviceInfoList* dlist = deviceInfoList(executionPolicy());
422 Int32 nb_device = dlist->nbDevice();
423 if (nb_device == 0)
424 ARCANE_FATAL("Internal error: no device available");
425 Int32 device_id = deviceId().asInt32();
426 if (device_id >= nb_device)
427 device_id = 0;
428 return dlist->deviceInfo(device_id);
