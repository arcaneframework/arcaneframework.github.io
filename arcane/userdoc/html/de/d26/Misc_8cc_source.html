276{
277 // \n écrit en meme temps pour éviter des écritures intermédiares parasites
278 char buffer[256];
279 va_list ap;
280 va_start(ap,format);
281 vsnprintf(buffer,256,format,ap);
282 va_end(ap);
283 cerr << buffer << "\n";
284 cout << "*E* " << buffer << "\n";
285}

402extern "C++" ARCANE_UTILS_EXPORT void
403arcaneRedirectSignals(fSignalFunc sig_func)
404{
405 setSignalFunc(sig_func);
406#if STD_SIGNAL_TYPE == 1
407
408#ifdef USE_SIGACTION
409 struct sigaction sa;
410 sa.sa_flags = SA_SIGINFO;
411 sigemptyset(&sa.sa_mask);
412 sa.sa_sigaction = _MiscSigactionFunc;
413
414 sigaction(SIGSEGV, &sa, nullptr); // Segmentation fault.
415 sigaction(SIGFPE, &sa, nullptr); // Floating Exception.
416 sigaction(SIGBUS, &sa, nullptr); // Bus Error.
417 sigaction(SIGALRM, &sa, nullptr); // Signal alarm (ITIMER_REAL)
418 sigaction(SIGVTALRM, &sa, nullptr); // Signal alarm (ITIMER_VIRTUAL)
419#else
420 default_signal_func_sigsegv = sigset(SIGSEGV,_MiscSigFunc); // Segmentation fault.
421 default_signal_func_sigfpe = sigset(SIGFPE ,_MiscSigFunc); // Floating Exception.
422 default_signal_func_sigbus = sigset(SIGBUS ,_MiscSigFunc); // Bus Error.
423 //default_signal_func_sigsys = sigset(SIGSYS ,_MiscSigFunc); // Bad argument in system call.
424 //default_signal_func_sigpipe = sigset(SIGPIPE,_MiscSigFunc); // Pipe error.
425 default_signal_func_sigalrm = sigset(SIGALRM,_MiscSigFunc); // Signal alarm (ITIMER_REAL)
426 default_signal_func_sigvtalrm = sigset(SIGVTALRM,_MiscSigFunc); // Signal alarm (ITIMER_VIRTUAL)
427#endif // USE_SIGACTION
428#endif // STD_SIGNAL_TYPE == 1

431extern "C++" ARCANE_UTILS_EXPORT void
432arcaneCallDefaultSignal(int val)
433{
434#if STD_SIGNAL_TYPE == 1
435
436 //fSignalFunc func = 0;
437 SignalException::eSignalType signal_type = SignalException::ST_Unknown;
438
439 switch(val){
440 case SIGSEGV:
441 //func = default_signal_func_sigsegv;
442 signal_type = SignalException::ST_SegmentationFault;
443 break;
444 case SIGFPE:
445 //func = default_signal_func_sigfpe;
446 signal_type = SignalException::ST_FloatingException;
447 break;
448 case SIGBUS:
449 //func = default_signal_func_sigbus;
450 signal_type = SignalException::ST_BusError;
451 break;
452 case SIGALRM:
453 case SIGVTALRM:
454 // func = default_signal_func_sigalrm;
455 signal_type = SignalException::ST_Alarm;
456 break;
457 }
458
459 //cerr << "** SIGVAL " << func << ' ' << SIG_DFL << ' '
460 //<< SIG_IGN << ' ' << global_already_in_signal << '\n';
461
462 //if (val==SIGSEGV || val==SIGBUS)
463 //::abort();
464
465 // En cas de nouveau signal alors qu'on est déja dans un handler, ou
466 // s'il s'agit d'un signal d'erreur memoire (SIGBUS ou SIGSEGV), on fait un abort.
467 if (global_already_in_signal)
468 ::abort();
469 global_already_in_signal = true;
470
471 arcaneDebugPause("arcaneCallDefaultSignal");
472
473 // Pour éviter certains problèmes lors de la récupération de la pile
474 // qui peuvent provoquer une exception, on récupère d'abord la pile
475 // d'appel et on la passe au constructeur de l'exception.
476 StackTrace stack_trace;
477 IStackTraceService* stack_service = platform::getStackTraceService();
478 if (stack_service){
479 stack_trace = stack_service->stackTrace();
480 cerr << " Signal exception Stack: " << stack_trace.toString() << '\n';
481 }
482 else
483 cerr << " No stack trace service\n";
484
485 if (signal_type==SignalException::ST_Alarm){
486 global_already_in_signal = false;
487 throw TimeoutException("Arcane.Signal",stack_trace);
488 }
489 else if (signal_type==SignalException::ST_FloatingException){
490 global_already_in_signal = false;
491 cerr << "** THROW ARITHMETIC EXCEPTION\n";
492 // Réactive les exceptions flottantes pour le prochain coup
493 platform::enableFloatingException(true);
494 throw ArithmeticException(A_FUNCINFO,stack_trace);
495 }
496 throw SignalException("Arcane.Signal",stack_trace,signal_type,val);
497#endif
498}
