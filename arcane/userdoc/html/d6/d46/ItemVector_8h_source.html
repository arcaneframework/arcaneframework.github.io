29/*!
30 * \brief Vecteur d'entités.
31 *
32 * Le fonctionnement du vecteur d'entité est similaire à celui
33 * du groupe d'entité ItemGroup à la différence suivante:
34 * - le vecteur est local au sous-domaine
35 * - le vecteur est invalidé si la famille (IItemFamily) associée évolue
36 * (après appel à IItemFamily::endUpdate() si le compactage ou le tri est
37 * actif)
38 * - le vecteur peut contenir plusieurs fois la même entité.
39 *
40 * Un vecteur est intéressant pour construire une liste temporaire
41 * d'entités. Il reprend les fonctionnalités similaires à la classe
42 * Array et il est donc possible par exemple d'ajouter des éléments un par un,
43 * soit via un localId(), soit via une entité.
44 *
45 * Un vecteur doit nécessairement être associée à une famille (setFamily())
46 * avant d'être utilisé.
47 */
48class ARCANE_CORE_EXPORT ItemVector
49{
50 public:
51
52 typedef Item ItemType;
53
54 public:
55
56 //! Créé un vecteur vide associé à la famille \a family.
57 explicit ItemVector(IItemFamily* afamily);
58
59 //! Créé un vecteur associé à la famille \a family et contenant les entités \a local_ids.
60 ItemVector(IItemFamily* afamily, Int32ConstArrayView local_ids);
61
62 //! Créé un vecteur pour \a size éléments associé à la famille \a family.
63 ItemVector(IItemFamily* afamily, Integer asize);
64
65 //! Créé un vecteur nul. Il faudra ensuite appeler setFamily() pour l'utiliser
66 ItemVector() = default;
67
68 public:
69
70 //! Operateur de cast vers ItemVectorView
71 operator ItemVectorView() const { return view(); }
72
73 public:
74
75 /*!
76 * \brief Positionne la famille associée
77 *
78 * Le vecteur est vidé de ses éléments
79 */
80 void setFamily(IItemFamily* afamily);
81
82 //! Ajoute une entité de numéro local \a local_id à la fin du vecteur
83 void add(Int32 local_id) { m_local_ids.add(local_id); }
84
85 //! Ajoute une liste d'entité de numéros locaux \a local_ids à la fin du vecteur
86 void add(ConstArrayView<Int32> local_ids) { m_local_ids.addRange(local_ids); }
87
88 //! Ajoute une entité de numéro local \a local_id à la fin du vecteur
89 void addItem(ItemLocalId local_id) { m_local_ids.add(local_id); }
90
91 //! Ajoute une entité à la fin du vecteur
92 void addItem(Item item) { m_local_ids.add(item.localId()); }
93
94 //! Nombre d'éléments du vecteur
95 Int32 size() const { return m_local_ids.size(); }
96
97 //! Réserve la mémoire pour \a capacity entités
98 void reserve(Integer capacity) { m_local_ids.reserve(capacity); }
99
100 //! Supprime toutes les entités du vecteur.
101 void clear() { m_local_ids.clear(); }
102
103 //! Vue sur le vecteur
104 ItemVectorView view() const { return ItemVectorView(m_shared_info, m_local_ids, 0); }
105
106 //! Vue sur les numéros locaux
107 ArrayView<Int32> viewAsArray() { return m_local_ids.view(); }
108
109 //! Vue constante sur les numéros locaux
110 ConstArrayView<Int32> viewAsArray() const { return m_local_ids.constView(); }
111
112 //! Supprime l'entité à l'index \a index
113 void removeAt(Int32 index) { m_local_ids.remove(index); }
114
115 /*!
116 * \brief Positionne le nombre d'éléments du tableau.
117 *
118 * Si la nouvelle taille est supérieure à l'ancienne, les
119 * éléments ajoutés sont indéfinis.
120 */
121 void resize(Integer new_size) { m_local_ids.resize(new_size); }
122
123 //! Clone ce vecteur
124 ItemVector clone() { return ItemVector(m_family, m_local_ids.constView()); }
125
126 //! Entité à la position \a index du vecteur
127 Item operator[](Int32 index) const { return Item(m_local_ids[index], m_shared_info); }
128
129 //! Famille associée au vecteur
130 IItemFamily* family() const { return m_family; }
131
132 //! Enumérateur
133 ItemEnumerator enumerator() const { return ItemEnumerator(m_shared_info, m_local_ids); }
134
135 protected:
136
137 SharedArray<Int32> m_local_ids;
138 IItemFamily* m_family = nullptr;
139 ItemSharedInfo* m_shared_info = ItemSharedInfo::nullInstance();
140
141 private:
142
143 void _init();
144};
