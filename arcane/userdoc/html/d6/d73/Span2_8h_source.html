62 template<typename T,typename SizeType>
63 class Span2Impl
64 {
65  public:
66 
67  using ElementType = T;
68  using element_type = ElementType;
69  using value_type = typename std::remove_cv<ElementType>::type;
70  using index_type = SizeType;
71  using difference_type = SizeType;
72  using size_type = SizeType;
73  using pointer = ElementType*;
74  using const_pointer = typename std::add_const<ElementType*>::type;
75  using reference = ElementType&;
76  using const_reference = const ElementType&;
77  using view_type = typename detail::View2TypeT<ElementType>::view_type;
78 
80  template<typename X>
81  using is_same_const_type = std::enable_if_t<std::is_same_v<X,T> || std::is_same_v<std::add_const_t<X>,T>>;
82 
83  public:
84 
86  ARCCORE_HOST_DEVICE Span2Impl(pointer ptr,SizeType dim1_size,SizeType dim2_size)
87  : m_ptr(ptr), m_dim1_size(dim1_size), m_dim2_size(dim2_size) {}
89  ARCCORE_HOST_DEVICE Span2Impl() : m_ptr(nullptr), m_dim1_size(0), m_dim2_size(0) {}
90  // Constructeur à partir d'un ConstArrayView. Cela n'est autorisé que
91  // si T est const.
92  template<typename X,typename = std::enable_if_t<std::is_same_v<X,value_type>> >
93  Span2Impl(const ConstArray2View<X>& from)
94  : m_ptr(from.data()), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
95  // Pour un Span<const T>, on a le droit de construire depuis un Span<T>
96  template<typename X,typename = std::enable_if_t<std::is_same_v<X,value_type>> >
97  ARCCORE_HOST_DEVICE Span2Impl(const Span2<X>& from)
98  : m_ptr(from.data()), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
99 
100  public:
101 
103  constexpr ARCCORE_HOST_DEVICE SizeType dim1Size() const { return m_dim1_size; }
105  constexpr ARCCORE_HOST_DEVICE SizeType dim2Size() const { return m_dim2_size; }
107  constexpr ARCCORE_HOST_DEVICE SizeType totalNbElement() const { return m_dim1_size*m_dim2_size; }
108 
109  public:
110 
111  constexpr ARCCORE_HOST_DEVICE SpanImpl<ElementType,SizeType> operator[](SizeType i) const
112  {
113  ARCCORE_CHECK_AT(i,m_dim1_size);
114  return SpanImpl<ElementType,SizeType>(m_ptr + (m_dim2_size*i),m_dim2_size);
115  }
116 
117  constexpr ARCCORE_HOST_DEVICE SpanImpl<ElementType,SizeType> operator()(SizeType i) const
118  {
119  ARCCORE_CHECK_AT(i,m_dim1_size);
120  return SpanImpl<ElementType,SizeType>(m_ptr + (m_dim2_size*i),m_dim2_size);
121  }
122 
123  constexpr ARCCORE_HOST_DEVICE reference operator()(SizeType i,SizeType j) const
124  {
125  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
126  return m_ptr[(m_dim2_size*i) + j];
127  }
128 
129 #ifdef ARCCORE_HAS_MULTI_SUBSCRIPT
130  constexpr ARCCORE_HOST_DEVICE reference operator[](SizeType i,SizeType j) const
131  {
132  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
133  return m_ptr[(m_dim2_size*i) + j];
134  }
135 #endif
136 
138  constexpr ARCCORE_HOST_DEVICE ElementType item(SizeType i,SizeType j) const
139  {
140  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
141  return m_ptr[(m_dim2_size*i) + j];
142  }
143 
145  constexpr ARCCORE_HOST_DEVICE ElementType setItem(SizeType i,SizeType j,const ElementType& value)
146  {
147  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
148  m_ptr[(m_dim2_size*i) + j] = value;
149  }
150 
151  public:
152 
156  constexpr view_type smallView()
157  {
158  Integer s1 = arccoreCheckArraySize(m_dim1_size);
159  Integer s2 = arccoreCheckArraySize(m_dim2_size);
160  return view_type(m_ptr,s1,s2);
161  }
162 
166  constexpr ConstArrayView<value_type> constSmallView() const
167  {
168  Integer s1 = arccoreCheckArraySize(m_dim1_size);
169  Integer s2 = arccoreCheckArraySize(m_dim2_size);
170  return ConstArrayView<value_type>(m_ptr,s1,s2);
171  }
172 
173  public:
174 
176  constexpr ElementType* unguardedBasePointer() { return m_ptr; }
177 
179  constexpr ARCCORE_HOST_DEVICE ElementType* data() { return m_ptr; }
180 
182  constexpr ARCCORE_HOST_DEVICE const ElementType* data() const { return m_ptr; }
183 
184  protected:
185 
186  ElementType* m_ptr;
187  SizeType m_dim1_size;
188  SizeType m_dim2_size;
189 };
