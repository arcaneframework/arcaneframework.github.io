57 template<class T>
58 class Span2
59 {
60  public:
61 
62  using ElementType = T;
63  using element_type = ElementType;
64  using value_type = typename std::remove_cv<ElementType>::type;
65  using index_type = Int64;
66  using difference_type = Int64;
67  using pointer = ElementType*;
68  using const_pointer = typename std::add_const<ElementType*>::type;
69  using reference = ElementType&;
70  using view_type = typename detail::View2TypeT<ElementType>::view_type;
71 
72  public:
74  ARCCORE_HOST_DEVICE Span2(ElementType* ptr,Int64 dim1_size,Int64 dim2_size)
75  : m_ptr(ptr), m_dim1_size(dim1_size), m_dim2_size(dim2_size) {}
77  ARCCORE_HOST_DEVICE Span2() : m_ptr(nullptr), m_dim1_size(0), m_dim2_size(0) {}
79  Span2(const Array2View<value_type>& from)
80  : m_ptr(from.m_ptr), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
81  // Constructeur à partir d'un ConstArrayView. Cela n'est autorisé que
82  // si T est const.
83  template<typename X,typename = std::enable_if_t<std::is_same_v<X,value_type>> >
84  Span2(const ConstArray2View<X>& from)
85  : m_ptr(from.data()), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
86  // Pour un Span<const T>, on a le droit de construire depuis un Span<T>
87  template<typename X,typename = std::enable_if_t<std::is_same_v<X,value_type>> >
88  ARCCORE_HOST_DEVICE Span2(const Span2<X>& from)
89  : m_ptr(from.data()), m_dim1_size(from.dim1Size()),m_dim2_size(from.dim2Size()) {}
91  // contenant \a asize éléments.
92  public:
94  ARCCORE_HOST_DEVICE Int64 dim1Size() const { return m_dim1_size; }
96  ARCCORE_HOST_DEVICE Int64 dim2Size() const { return m_dim2_size; }
98  ARCCORE_HOST_DEVICE Int64 totalNbElement() const { return m_dim1_size*m_dim2_size; }
99  public:
100  ARCCORE_HOST_DEVICE Span<ElementType> operator[](Int64 i) const
101  {
102  ARCCORE_CHECK_AT(i,m_dim1_size);
103  return Span<ElementType>(m_ptr + (m_dim2_size*i),m_dim2_size);
104  }
106  ARCCORE_HOST_DEVICE ElementType item(Int64 i,Int64 j) const
107  {
108  ARCCORE_CHECK_AT(i,m_dim1_size);
109  ARCCORE_CHECK_AT(j,m_dim2_size);
110  return m_ptr[(m_dim2_size*i) + j];
111  }
113  ARCCORE_HOST_DEVICE ElementType setItem(Int64 i,Int64 j,const ElementType& value)
114  {
115  ARCCORE_CHECK_AT(i,m_dim1_size);
116  ARCCORE_CHECK_AT(j,m_dim2_size);
117  m_ptr[(m_dim2_size*i) + j] = value;
118  }
119  public:
123  view_type smallView()
124  {
125  Integer s1 = arccoreCheckArraySize(m_dim1_size);
126  Integer s2 = arccoreCheckArraySize(m_dim2_size);
127  return view_type(m_ptr,s1,s2);
128  }
129 
133  ConstArrayView<value_type> constSmallView() const
134  {
135  Integer s1 = arccoreCheckArraySize(m_dim1_size);
136  Integer s2 = arccoreCheckArraySize(m_dim2_size);
137  return ConstArrayView<value_type>(m_ptr,s1,s2);
138  }
139  public:
143  inline ElementType* unguardedBasePointer() { return m_ptr; }
147  ARCCORE_HOST_DEVICE inline ElementType* data() { return m_ptr; }
151  ARCCORE_HOST_DEVICE inline const ElementType* data() const { return m_ptr; }
152  private:
153  ElementType* m_ptr;
154  Int64 m_dim1_size;
155  Int64 m_dim2_size;
156 };
