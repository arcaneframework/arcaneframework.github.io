56/*!
57 * \brief Fichier ou flux de traces.
58 */
59class FileTraceStream
60: public ITraceStream
61{
62 public:
63 explicit FileTraceStream(const String& filename)
64 : m_nb_ref(0), m_stream(nullptr), m_need_destroy(true)
65 {
66 m_stream = new std::ofstream(filename.localstr());
67 }
68 FileTraceStream(std::ostream* stream,bool need_destroy)
69 : m_nb_ref(0), m_stream(stream), m_need_destroy(need_destroy)
70 {
71 }
72 ~FileTraceStream() override
73 {
74 if (m_need_destroy)
75 delete m_stream;
76 }
77 public:
78 void addReference() override { ++m_nb_ref; }
79 void removeReference() override
80 {
81 // Décrémente et retourne la valeur d'avant.
82 // Si elle vaut 1, cela signifie qu'on n'a plus de références
83 // sur l'objet et qu'il faut le détruire.
84 Int32 v = std::atomic_fetch_add(&m_nb_ref,-1);
85 if (v==1)
86 delete this;
87 }
88 std::ostream* stream() override { return m_stream; }
89 private:
90 std::atomic<Int32> m_nb_ref;
91 std::ostream* m_stream;
92 bool m_need_destroy;
93};

764void TraceMng::
765_writeTimeString(std::ostream& out)
766{
767 char trace_timer_buffer[256];
768 unsigned long t = static_cast<unsigned long>(m_trace_timer.getTime());
769 const unsigned long hour = t/3600; t -= hour*3600;
770 const unsigned long min = t/60; t -= min*60;
771 // TODO utiliser une méthode portable.
772#ifdef ARCCORE_OS_WIN32
773 _snprintf(trace_timer_buffer,256,"%04lu:%02lu:%02lu",hour,min,t);
774#else
775 snprintf(trace_timer_buffer,256,"%04lu:%02lu:%02lu",hour,min,t);
776#endif
777 out << trace_timer_buffer;
778}

780/*---------------------------------------------------------------------------*/
781/*---------------------------------------------------------------------------*/
782
783void TraceMng::
784_putTraceMessage(std::ostream& out,Trace::eMessageType id,Span<const Byte> msg_str)
785{
786 if (m_want_trace_timer || (m_current_class_flags & Trace::PF_ElapsedTime)){
787 _writeTimeString(out);
788 out << " ";
789 }
790 switch(id){
791 case Trace::Info:
792 if (!(m_current_class_flags & Trace::PF_NoClassName)){
793 out << "*I-";
794 out.width(10);
795 out.flags(std::ios::left);
796 out << _currentTraceClassName();
797 out << ' ';
798 }
799 _putStream(out,msg_str);
800 break;
801 case Trace::Warning:
802 flush();
803 out << "*W* Code:---------------------------------------------\n";
804 out << "*W* WARNING";
805 _putTraceId(out);
806 out << ": ";
807 _putStream(out,msg_str);
808 break;
809 case Trace::Error:
810 flush();
811 out << "*E* Code:---------------------------------------------\n";
812 out << "*E* ERROR";
813 _putTraceId(out);
814 out << ": ";
815 _putStream(out,msg_str);
816 break;
817 case Trace::Log:
818 out << "*L-";
819 out.width(8);
820 out.flags(std::ios::left);
821 out << _currentTraceClassName();
822 out << ' ';
823 _putStream(out,msg_str);
824 break;
825 case Trace::Fatal:
826 flush();
827 out << "*F* Code:---------------------------------------------\n";
828 out << "*F* FATAL";
829 _putTraceId(out);
830 out << ": ";
831 _putStream(out,msg_str);
832 break;
833 case Trace::ParallelFatal:
834 flush();

836 out << "*F* FATAL";
837 _putTraceId(out);
838 out << ": ";
839 _putStream(out,msg_str);

843 out.flags(std::ios::left);
844 out.width(10);
845 out << _currentTraceClassName();
846 out << ' ';
847 _putStream(out,msg_str);

849 case Trace::Normal:
850 break;
851 case Trace::Null:

856}

862 {
863 "30",
864 "31", "32", "33", "34", "35", "36", "37",
865 "1;31", "1;32", "1;33", "1;34", "1;35", "1;36", "1;37"

868void TraceMng::
869_writeColor(std::ostream& output,Span<const Byte> input,int color, bool do_flush)
870{
871 if (color>Trace::Color::LAST_COLOR || color<0)
872 color = 0;
873 // Pour être sur que les message sont écrits en une seule fois,
874 // il ne faut faire qu'un seul write.
875 if (color!=0){
876 Mutex::ScopedLock sl(m_trace_mutex);
877 output << "\33[" << color_fmt[color] << "m";
878 Int64 len = input.size();
879 // Le message se termine toujours par un '\n'. On écrit la fin de la couleur
880 // avant de '\n'
881 if (len>0)
882 --len;
883 _writeSpan(output,input.subspan(0,len));
884 output << "\33[0m\n";
885 }
886 else
887 _writeSpan(output,input);
888
889 if (do_flush)
890 output.flush();
891}

895/*!
896 * \brief Écrit le listing.
897 *
898 * Le listing peut sortir à la fois sur std::cout et dans un
899 * ITraceStream si *m_listing_stream* est non nul. Le niveau de verbosité
900 * peut être différent dans les deux cas.
901 */
902void TraceMng::
903_writeListing(Span<const Byte> input,Int32 level,int color,bool do_flush)
904{
905 std::ostream* listing_stream = (m_listing_stream) ? m_listing_stream->stream() : nullptr;
906 if (!m_has_color)
907 color = 0;
908
909 // Regarde si le niveau de verbosité souhaité est suffisant pour afficher
910 // le message.
911 Int32 message_level = level;
912
913 // Sortie ITraceStream.
914 if (listing_stream){
915 Int32 verbosity_level = m_current_class_verbosity_level;
916 if (verbosity_level==Trace::UNSPECIFIED_VERBOSITY_LEVEL)
917 verbosity_level = m_verbosity_level;
918 // Pas de couleur si on redirige les sorties
919 int listing_color = 0;
920 bool is_printed = (message_level <= verbosity_level);
921 if (is_printed)
922 _writeColor(*listing_stream,input,listing_color,do_flush);
923 }
924
925 // Sortie std::cout
926 if (m_is_master || !listing_stream){
927 Int32 verbosity_level = m_current_class_verbosity_level;
928 if (verbosity_level==Trace::UNSPECIFIED_VERBOSITY_LEVEL)
929 verbosity_level = (listing_stream) ? m_stdout_verbosity_level : m_verbosity_level;
930 bool is_printed = (message_level <= verbosity_level);
931 if (is_printed)
932 _writeColor(std::cout,input,color,do_flush);
933 }
934}

1025/*---------------------------------------------------------------------------*/
1026
1027void TraceMng::
1028_endTrace(const TraceMessage* msg)
1029{
1030 Trace::eMessageType id = msg->type();
1031 TraceMngStreamList* ts = _getStreamList();
1032 const std::string& str = ts->m_str_list[id]->str();
1033 const Byte* str_data = reinterpret_cast<const Byte*>(str.data());
1034 std::vector<Byte> msg_str_copy(str_data,str_data+str.length());
1035 Span<const Byte> msg_str(msg_str_copy.data(),msg_str_copy.size());
1036
1037 ts->m_str_list[id]->str(std::string());
1038 if (msg_str.empty())
1039 return;
1040
1041 ts->m_tmp_buf.str(std::string());
1042 _putTraceMessage(ts->m_tmp_buf,id,msg_str);
1043 const std::string& tmp_buf_str = ts->m_tmp_buf.str();
1044 const Byte* tmp_buf_str_data = reinterpret_cast<const Byte*>(tmp_buf_str.data());
1045 Span<const Byte> buf_array(tmp_buf_str_data,tmp_buf_str.length());
1046
1047 if (_sendToProxy2(msg,buf_array))
1048 return;
1049
1050 _writeDirect(msg,buf_array,msg_str);
1051}

1053/*---------------------------------------------------------------------------*/
1054/*---------------------------------------------------------------------------*/
1055
1056void TraceMng::
1057_writeDirect(const TraceMessage* msg,Span<const Byte> buf_array,
1058 Span<const Byte> orig_message)
1059{
1060 std::ostream* listing_stream = (m_listing_stream) ? m_listing_stream->stream() : nullptr;
1061 std::ostream& def_out = (listing_stream) ? (*listing_stream) : std::cout;
1062
1063 Int32 print_level = -1;
1064 Trace::eMessageType id = msg->type();
1065 int color = msg->color();
1066 // TODO Rendre paramétrable.
1067 const bool write_stack_trace_for_error = false;
1068 switch(id){
1069 case Trace::Normal:
1070 _writeListing(buf_array,print_level,color,false);
1071 _checkFlush();
1072 break;
1073 case Trace::Info:
1074 _writeListing(buf_array,msg->level(),color,false);
1075 _checkFlush();
1076 break;
1077 case Trace::Log:
1078 _write(_logStream(),buf_array);
1079 _checkFlush();
1080 break;
1081 case Trace::Warning:
1082 case Trace::Error:
1083 {
1084 auto error_stream = _errorStream();
1085 auto log_stream = _logStream();
1086 auto tc_color = (id==Trace::Warning) ? Trace::Color::DarkYellow : Trace::Color::DarkRed;
1087 _writeListing(buf_array,print_level,tc_color,true);
1088 _write(error_stream,buf_array,true);
1089 String stack_trace = Platform::getStackTrace();
1090 if (write_stack_trace_for_error)
1091 _writeStackTrace(&def_out,stack_trace);
1092 _write(log_stream,buf_array,true);
1093 _writeStackTrace(log_stream,stack_trace);
1094 if (listing_stream){
1095 _write(std::cerr,buf_array);
1096 }
1097 }
1098 break;
1099 case Trace::Fatal:
1100 case Trace::ParallelFatal:
1101 if (m_is_master || id==Trace::Fatal){
1102 auto error_stream = _errorStream();
1103 auto log_stream = _logStream();
1104 _writeListing(buf_array,print_level,Trace::Color::Red,true);
1105 _write(error_stream,buf_array,true);
1106 _write(log_stream,buf_array,true);
1107 String stack_trace = Platform::getStackTrace();
1108 _writeStackTrace(log_stream,stack_trace);
1109 if (listing_stream)
1110 _write(std::cerr,buf_array);
1111 }
1112 {
1113 String s1(orig_message);
1114 FatalErrorException ex("TraceMng::endTrace()",s1);
1115 if (id==Trace::ParallelFatal)
1116 ex.setCollective(true);
1117 throw ex;
1118 }
1119 case Trace::Debug:
1120 _writeListing(buf_array,print_level,color,true);
1121 break;
1122 case Trace::Null:
1123 break;
1124 }
1125}
