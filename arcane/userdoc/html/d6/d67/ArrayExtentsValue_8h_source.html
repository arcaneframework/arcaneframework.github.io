38template <class T> constexpr ARCCORE_HOST_DEVICE
39T fastmod(T a , T b)
40{
41 return a < b ? a : a-b*(a/b);
42}

44/*!
45 * \brief Informations pour une dimension fixe connue à la compilation.
46 *
47 * La valeur de la dimension est donnée en paramètre template

51{
52 public:
53 static constexpr Int64 size() { return Size; };
54 static constexpr Int32 v = Size;
55};

399 // Compute base indices

403 fac *= m_extent2.v;
404 Int32 i1 = impl::fastmod(i/fac,m_extent1.v);

406 Int32 i0 = i /fac;
407 return { i0, i1, i2, i3 };

417 {
418 std::array<Int32, ExtentsType::nb_dynamic> x = {};
419 Int32 i = 0;
420 if constexpr (X0 == DynExtent)
