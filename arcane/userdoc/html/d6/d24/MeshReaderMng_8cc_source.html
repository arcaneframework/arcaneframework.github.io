98IMesh* MeshReaderMng::
99readMesh(const String& mesh_name,const String& file_name,IParallelMng* parallel_mng)
100{
101 m_p->checkInit();
102 String extension;
103 {
104 // Cherche l'extension du fichier et la conserve dans \a extension
105 std::string_view fview = file_name.toStdStringView();
106 std::size_t extension_pos = fview.find_last_of('.');
107 if (extension_pos==std::string_view::npos)
108 ARCANE_FATAL("file name '{0}' has no extension",file_name);
109 fview.remove_prefix(extension_pos+1);
110 extension = fview;
111
112 }
113 // TODO: à terme, créer le maillage par le lecteur.
114 ISubDomain* sd = m_p->m_sub_domain;
115 IParallelMng* pm = parallel_mng;
116 IPrimaryMesh* mesh = sd->mainFactory()->createMesh(sd,pm,mesh_name);
117 mesh->properties()->setBool("dump", false);
118
119 String use_unit_str = (m_p->m_is_use_unit) ? "true" : "false";
120 String use_unit_xml = "<?xml version=\"1.0\"?><file use-unit='"+use_unit_str+"' />";
121
122 ITraceMng* tm = sd->traceMng();
123 ScopedPtrT<IXmlDocumentHolder> xml_doc(IXmlDocumentHolder::loadFromBuffer(use_unit_xml.bytes(), String(),tm));
124 XmlNode mesh_xml_node = xml_doc->documentNode().documentElement();
125
126 String dir_name;
127 bool is_bad = true;
128 bool use_internal_partition = pm->isParallel();
129 for( auto& reader_ref : m_p->readers() ){
130 IMeshReader* reader = reader_ref.get();
131 if (!reader->allowExtension(extension))
132 continue;
133
134 auto ret = reader->readMeshFromFile(mesh,mesh_xml_node,
135 file_name,dir_name,
136 use_internal_partition);
137 if (ret==IMeshReader::RTOk){
138 is_bad = false;
139 break;
140 }
141 if (ret==IMeshReader::RTError){
142 ARCANE_FATAL("Can not read mesh file '{0}'",file_name);
143 }
144 }
145
146 if (is_bad)
147 ARCANE_FATAL("No mesh reader is available for mesh file '{0}'",file_name);
148
149 return mesh;
150}
