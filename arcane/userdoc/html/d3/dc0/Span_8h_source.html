47/*!
48 * \ingroup Collection
49 * \brief Vue d'un tableau d'éléments de type \a T.
50 *
51 * Cette classe ne doit pas être utilisée directement. Il faut utiliser
52 * Span ou SmallSpan à la place.
53 *
54 * La vue est non modifiable si l'argument template est de type 'const T'.
55 * Cette classe permet d'accéder et d'utiliser un tableau d'éléments du
56 * type \a T de la même manière qu'un tableau C standard. \a SizeType est le
57 * type utilisé pour conserver le nombre d'éléments du tableau. Cela peut
58 * être 'Int32' ou 'Int64'.
59*/
60template<typename T,typename SizeType,SizeType Extent>
61class SpanImpl
62{
63 public:
64
65 using ThatClass = SpanImpl<T,SizeType,Extent>;
66 using size_type = SizeType;
67 using ElementType = T;
68 using element_type = ElementType;
69 using value_type = typename std::remove_cv_t<ElementType>;
70 using const_value_type = typename std::add_const_t<value_type>;
71 using index_type = SizeType;
72 using difference_type = SizeType;
73 using pointer = ElementType*;
74 using const_pointer = const ElementType*;
75 using reference = ElementType&;
76 using const_reference = const ElementType&;
77 using iterator = ArrayIterator<pointer>;
78 using const_iterator = ArrayIterator<const_pointer>;
79 using view_type = typename detail::ViewTypeT<ElementType>::view_type;
80 using reverse_iterator = std::reverse_iterator<iterator>;
81 using const_reverse_iterator = std::reverse_iterator<const_iterator>;
82
83 //! Indique si on peut convertir un 'X' ou 'const X' en un 'T'
84 template<typename X>
85 using is_same_const_type = std::enable_if_t<std::is_same_v<X,T> || std::is_same_v<std::add_const_t<X>,T>>;
86
87 public:
88
89 //! Construit une vue vide.
90 constexpr ARCCORE_HOST_DEVICE SpanImpl() noexcept : m_ptr(nullptr), m_size(0) {}
91
92 //! Constructeur de recopie depuis une autre vue
93 // Pour un Span<const T>, on a le droit de construire depuis un Span<T>
94 template<typename X,SizeType XExtent = Extent, typename = std::enable_if_t<std::is_same_v<X,value_type>> >
95 constexpr ARCCORE_HOST_DEVICE SpanImpl(const SpanImpl<X,SizeType,XExtent>& from) noexcept
96 : m_ptr(from.data()), m_size(from.size()) {}
97
98 //! Construit une vue sur une zone mémoire commencant par \a ptr et contenant \a asize éléments.
99 constexpr ARCCORE_HOST_DEVICE SpanImpl(pointer ptr,SizeType asize) noexcept
100 : m_ptr(ptr), m_size(asize) {}
101
102 //! Construit une vue depuis un std::array
103 template<std::size_t N,typename X,typename = is_same_const_type<X> >
104 constexpr ARCCORE_HOST_DEVICE SpanImpl(std::array<X,N>& from)
105 : m_ptr(from.data()), m_size(ArraySizeChecker<SizeType>::check(from.size())) {}
106
107 //! Opérateur de recopie
108 template<std::size_t N,typename X,typename = is_same_const_type<X> >
109 constexpr ARCCORE_HOST_DEVICE ThatClass& operator=(std::array<X,N>& from)
110 {
111 m_ptr = from.data();
112 m_size = ArraySizeChecker<SizeType>::check(from.size());

117

248 /*!
249 * \brief Sous-vue à partir de l'élément \a abegin
250 * et contenant \a asize éléments.
251 *
252 * Si `(abegin+asize` est supérieur à la taille du tableau,
253 * la vue est tronquée à cette taille, retournant éventuellement une vue vide.

260 return {m_ptr+abegin,asize};

311 /*!
312 * \brief Recopie le tableau \a copy_array dans l'instance.
313 *
314 * Comme aucune allocation mémoire n'est effectuée, le
315 * nombre d'éléments de \a copy_array doit être inférieur ou égal au
316 * nombre d'éléments courant. S'il est inférieur, les éléments du
317 * tableau courant situés à la fin du tableau sont inchangés
318 */
319 template<class U> ARCCORE_HOST_DEVICE
320 void copy(const U& copy_array)
321 {
322 Int64 n = copy_array.size();
323 Int64 size_as_int64 = m_size;
324 arccoreCheckAt(n,size_as_int64+1);
325 const_pointer copy_begin = copy_array.data();
326 pointer to_ptr = m_ptr;
327 // On est sur que \a n tient sur un 'SizeType' car il est plus petit
328 // que \a m_size
329 SizeType n_as_sizetype = static_cast<SizeType>(n);
330 for( SizeType i=0; i<n_as_sizetype; ++i )
331 to_ptr[i] = copy_begin[i];
332 }
333
334 //! Retourne \a true si le tableau est vide (dimension nulle)
335 constexpr ARCCORE_HOST_DEVICE bool empty() const noexcept { return m_size==0; }
336 //! \a true si le tableau contient l'élément de valeur \a v
337 ARCCORE_HOST_DEVICE bool contains(const_reference v) const
338 {
339 for( SizeType i=0; i<m_size; ++i ){
340 if (m_ptr[i]==v)
341 return true;
342 }
343 return false;
344 }
345
346 /*!
347 * /brief Position du premier élément de valeur \a v
348 *
349 * /param v La valeur à trouver.
350 * /return La position du premier élément de valeur \a v si présent, std::nullopt sinon.
351 */
352 std::optional<SizeType> findFirst(const_reference v) const
353 {
354 for( SizeType i=0; i<m_size; ++i ){
355 if (m_ptr[i]==v)
356 return i;
357 }
358 return std::nullopt;
359 }

385 template<typename X,SizeType Extent2,typename = std::enable_if_t<std::is_same_v<X,value_type>>> friend bool
386 operator==(const SpanImpl<T,SizeType,Extent>& rhs, const SpanImpl<X,SizeType,Extent2>& lhs)

393 operator!=(const SpanImpl<T,SizeType,Extent>& rhs, const SpanImpl<X,SizeType,Extent2>& lhs)
394 {
395 return !operator==(rhs,lhs);
396 }
397
398 //! Opérateur d'égalité
399 template<SizeType Extent2> friend bool
400 operator==(const SpanImpl<T,SizeType,Extent>& rhs, const SpanImpl<T,SizeType,Extent2>& lhs)
401 {
402 return impl::areEqual(SpanImpl<T,SizeType>(rhs),SpanImpl<T,SizeType>(lhs));
403 }
404
405 //! Opérateur d'inégalité
406 template<SizeType Extent2> friend bool
407 operator!=(const SpanImpl<T,SizeType,Extent>& rhs, const SpanImpl<T,SizeType,Extent2>& lhs)
408 {
409 return !operator==(rhs,lhs);
410 }

412 friend inline std::ostream& operator<<(std::ostream& o, const ThatClass& val)
413 {
414 impl::dumpArray(o,Span<const T>(val),500);
415 return o;
416 }

418 protected:

421 * \brief Modifie le pointeur et la taille du tableau.
422 *
423 * C'est à la classe dérivée de vérifier la cohérence entre le pointeur
424 * alloué et la dimension donnée.

426 constexpr void _setArray(pointer v,SizeType s) noexcept { m_ptr = v; m_size = s; }

444 private:

454 }

459/*!

769 * Remplit \a result avec les valeurs du tableau \a values correspondantes
770 * aux indices \a indexes.
771 *
772 * \pre results.size() >= indexes.size();
773 */
774template<typename DataType,typename IntegerType,typename SizeType> inline void
775_sampleSpan(SpanImpl<const DataType,SizeType> values,
776 SpanImpl<const IntegerType,SizeType> indexes,
777 SpanImpl<DataType,SizeType> result)
778{
779 const Int64 result_size = indexes.size();
780 [[maybe_unused]] const Int64 my_size = values.size();
781 const DataType* ptr = values.data();
782 for( Int64 i=0; i<result_size; ++i) {
783 IntegerType index = indexes[i];
784 ARCCORE_CHECK_AT(index,my_size);
785 result[i] = ptr[index];
786 }
787}

851template<typename ByteType, typename DataType,Int64 Extent> inline Span<DataType>
852asSpanInternal(Span<ByteType,Extent> bytes)
853{
854 Int64 size = bytes.size();
855 if (size==0)
856 return {};
857 static constexpr Int64 data_type_size = static_cast<Int64>(sizeof(DataType));
858 static_assert(data_type_size>0,"Bad datatype size");
859 ARCCORE_ASSERT((size%data_type_size)==0,("Size is not a multiple of sizeof(DataType)"));
860 auto* ptr = reinterpret_cast<DataType*>(bytes.data());
861 return { ptr, size / data_type_size };
862}
