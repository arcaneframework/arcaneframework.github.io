289  void copy(const U& copy_array)
290  {
291  Int64 n = copy_array.size();
292  Int64 size_as_int64 = m_size;
293  arccoreCheckAt(n,size_as_int64+1);
294  const_pointer copy_begin = copy_array.data();

296  // On est sur que \a n tient sur un 'SizeType' car il est plus petit

298  SizeType n_as_sizetype = static_cast<SizeType>(n);
299  for( SizeType i=0; i<n_as_sizetype; ++i )
300  to_ptr[i] = copy_begin[i];
301  }

686 _sampleSpan(SpanImpl<const DataType,SizeType> values,
687  SpanImpl<const IntegerType,SizeType> indexes,
688  SpanImpl<DataType,SizeType> result)
689 {
690  const Int64 result_size = indexes.size();
691  [[maybe_unused]] const Int64 my_size = values.size();
692  const DataType* ptr = values.data();
693  for( Int64 i=0; i<result_size; ++i) {
694  IntegerType index = indexes[i];
695  ARCCORE_CHECK_AT(index,my_size);
696  result[i] = ptr[index];
697  }
698 }
