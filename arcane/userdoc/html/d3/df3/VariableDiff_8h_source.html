97  void dump(IVariable* var,int max_print)
98  {
99  ITraceMng* msg = var->subDomain()->traceMng();
100  Integer sid = var->subDomain()->subDomainId();
101  const String& var_name = var->name();
102  Integer nb_diff = m_diffs_info.size();
103  Integer nb_print = nb_diff;
104  if (max_print>=0 && nb_diff>static_cast<Integer>(max_print))
105  nb_print = max_print;
106  OStringStream ostr;
107  ostr().precision(FloatInfo<Real>::maxDigit());
108  ostr() << nb_diff << " entities having different values for the variable "
109  << var_name << '\n';
110  for( Integer i=0; i<nb_print; ++i ){
111  const DiffInfo& di = m_diffs_info[i];
112  if (di.m_unique_id!=NULL_ITEM_UNIQUE_ID){
113  // Il s'agit d'une entité
114  char type = di.m_is_own ? 'O' : 'G';
115  ostr() << "VDIFF: Variable '" << var_name << "'"
116  << " (" << type << ")"
117  << " uid=" << di.m_unique_id
118  << " lid=" << di.m_local_id;
119  if (di.m_sub_index!=NULL_ITEM_ID)
120  ostr() << " [" << di.m_sub_index << "]";
121  ostr() << " val: " << di.m_current
122  << " ref: " << di.m_ref << " rdiff: " << di.m_diff << '\n';
123  }
124  else{
125  // Il s'agit de l'indice d'une variable tableau
126  ostr() << "VDIFF: Variable '" << var_name << "'"
127  << " index=" << di.m_local_id;
128  if (di.m_sub_index!=NULL_ITEM_ID)
129  ostr() << " [" << di.m_sub_index << "]";
130  ostr() << " val: " << di.m_current
131  << " ref: " << di.m_ref << " rdiff: " << di.m_diff << '\n';
132  }
133  }
134  msg->pinfo() << "Processor " << sid << " : " << nb_diff
135  << " values are different on the variable "
136  << var_name << ":\n" << ostr.str();
137  }
138 };
