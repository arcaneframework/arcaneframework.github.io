79  public:
80  virtual void assign(ExpressionResult* res,IVariable* var)
81  {
82  // Notons que la taille de la variable peut être plus importante
83  // que celle du résultat suivant les indices retenus (cf WhereExpression)
84  Integer size = res->size();
85 
86  // recuperation des valeurs du resultat et de la variable
87  ArrayView<T> res_val;
88  res->data()->value(res_val);
89  ExpressionResult var_res(var);
90  ArrayView<T> var_val;
91  var_res.data()->value(var_val);
92  IntegerConstArrayView res_indices = res->indices();
93 
94  for( Integer i=0 ; i<size ; ++i)
95  var_val[res_indices[i]] = res_val[i];
96  }
97 
98  virtual void evaluate(ExpressionResult* res, IVariable* var)
99  {
100  // Notons que la taille de la variable peut être plus importante
101  // que celle du résultat suivant les indices retenus (cf WhereExpression)
102  Integer size = res->size();
103  if (size > var->nbElement())
104  throw BadOperandException("VariableOperatorT::evaluate");
105 
106  // allocation du résultat en fonction du type de la variable
107  VariantBase::eType type = VariantBase::fromDataType(var->dataType());
108  res->allocate(type);
109 
110  // recuperation des valeurs des operandes
111  ArrayView<T> res_val;
112  res->data()->value(res_val);
113  ExpressionResult var_res(var);
114  ArrayView<T> var_val;
115  var_res.data()->value(var_val);
116  IntegerConstArrayView res_indices = res->indices();
117 
118  for( Integer i=0 ; i<size ; ++i)
119  res_val[i] = var_val[res_indices[i]];
120  }
