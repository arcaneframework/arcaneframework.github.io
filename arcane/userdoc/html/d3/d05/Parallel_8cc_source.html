56extern "C++" void MessagePassing::
57namedBarrier(IParallelMng* pm,const String& name)
58{
59 // Copie les 1024 premiers caractères de name dans un tableau de Int32
60 // et fait une réduction 'max' sur ce tableau. Si la réduction est différente
61 // de notre valeur c'est qu'un des rang n'a pas la bonne barrière.
62 ARCANE_CHECK_POINTER(pm);
63
64 const int nsize = 256;
65 Int32 sbuf[nsize];
66 Int32 max_sbuf[nsize];
67 Int32ArrayView buf(nsize,sbuf);
68 Int32ArrayView max_buf(nsize,max_sbuf);
69 ByteArrayView buf_as_bytes(nsize*sizeof(Int32),(Byte*)sbuf);
70 buf.fill(0);
71 Integer len = arcaneCheckArraySize(name.length());
72 ByteConstArrayView name_as_bytes(len,(const Byte*)name.localstr());
73 Integer name_len = math::min(buf_as_bytes.size(),name_as_bytes.size());
74 buf_as_bytes.copy(name_as_bytes.subView(0,name_len));
75 buf[nsize-1] = 0;
76 max_buf.copy(buf);
77 pm->reduce(Parallel::ReduceMax,max_buf);
78
79 if (max_buf!=buf){
80 max_buf[nsize-1] = 0;
81 String max_string((const char*)max_buf.unguardedBasePointer());
82 ARCANE_FATAL("Bad namedBarrier expected='{0}' found='{1}'",
83 name,max_string);
84 }
85}

90extern "C++" void MessagePassing::
91filterCommonStrings(IParallelMng* pm,ConstArrayView<String> input_strings,
92 Array<String>& common_strings)
93{
94 const Int32 nb_string = input_strings.size();
95
96 // Créé un buffer pour sérialiser les noms des variables dont on dispose
97 SerializeBuffer send_buf;
98 send_buf.setMode(ISerializer::ModeReserve);
99 send_buf.reserve(DT_Int32,1);
100 for( Integer i=0; i<nb_string; ++i ){
101 send_buf.reserve(input_strings[i]);
102 }
103
104 send_buf.allocateBuffer();
105 send_buf.setMode(ISerializer::ModePut);
106 send_buf.putInt32(nb_string);
107 for( Integer i=0; i<nb_string; ++i ){
108 send_buf.put(input_strings[i]);
109 }
110
111 // Récupère les infos des autres PE.
112 SerializeBuffer recv_buf;
113 pm->allGather(&send_buf,&recv_buf);
114
115 std::map<String,Int32> string_occurences;
116
117 Int32 nb_rank = pm->commSize();
118 recv_buf.setMode(ISerializer::ModeGet);
119 for( Integer i=0; i<nb_rank; ++i ){
120 Int32 nb_string_rank = recv_buf.getInt32();
121 for( Integer z=0; z<nb_string_rank; ++z ){
122 String x;
123 recv_buf.get(x);
124 auto vo = string_occurences.find(x);
125 if (vo==string_occurences.end())
126 string_occurences.insert(std::make_pair(x,1));
127 else
128 vo->second = vo->second + 1;
129 }
130 }
131
132 // Parcours la liste des chaînes de caractète et range dans \a out_strings
133 // celles qui sont disponibles sur tous les rangs de \a pm
134 std::set<String> common_set;
135 {
136 auto end_iter = string_occurences.end();
137 for( Integer i=0; i<nb_string; ++i ){
138 String str = input_strings[i];
139 auto i_str = string_occurences.find(str);
140 if (i_str==end_iter)
141 // Ne devrait pas arriver
142 continue;
143 if (i_str->second!=nb_rank)
144 continue;
145 common_set.insert(str);
146 }
147 }
148
149 // Créé la liste finale en itérant sur \a common_set
150 // qui est trié alphabétiquement.
151 common_strings.clear();
152 for( const String& s : common_set )
153 common_strings.add(s);
154}
