31CartesianMeshNumberingMng::
32CartesianMeshNumberingMng(IMesh* mesh)
33: TraceAccessor(mesh->traceMng())
34, m_mesh(mesh)
35, m_dimension(mesh->dimension())
36, m_pattern(2)
37, m_max_level(0)
38, m_min_level(0)
39{
40 auto* m_generation_info = ICartesianMeshGenerationInfo::getReference(m_mesh, true);
41
42 Int64ConstArrayView global_nb_cells_by_direction = m_generation_info->globalNbCells();
43 m_nb_cell.x = global_nb_cells_by_direction[MD_DirX];
44 m_nb_cell.y = global_nb_cells_by_direction[MD_DirY];
45 m_nb_cell.z = ((m_dimension == 2) ? 1 : global_nb_cells_by_direction[MD_DirZ]);
46
47 if (m_nb_cell.x <= 0)
48 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirX] (should be >0)", m_nb_cell.x);
49 if (m_nb_cell.y <= 0)
50 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirY] (should be >0)", m_nb_cell.y);
51 if (m_nb_cell.z <= 0)
52 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirZ] (should be >0)", m_nb_cell.z);
53
54 m_p_to_l_level.add(0);
55
56 if (m_dimension == 2) {
57 m_latest_cell_uid = m_nb_cell.x * m_nb_cell.y;
58 m_latest_node_uid = (m_nb_cell.x + 1) * (m_nb_cell.y + 1);
59 m_latest_face_uid = (m_nb_cell.x * m_nb_cell.y) * 2 + m_nb_cell.x * 2 + m_nb_cell.y;
60 }
61 else {
62 m_latest_cell_uid = m_nb_cell.x * m_nb_cell.y * m_nb_cell.z;
63 m_latest_node_uid = (m_nb_cell.x + 1) * (m_nb_cell.y + 1) * (m_nb_cell.z + 1);
64 m_latest_face_uid = (m_nb_cell.z + 1) * m_nb_cell.x * m_nb_cell.y + (m_nb_cell.x + 1) * m_nb_cell.y * m_nb_cell.z + (m_nb_cell.y + 1) * m_nb_cell.z * m_nb_cell.x;
65 }
66
67 m_first_cell_uid_level.add(0);
68 m_first_node_uid_level.add(0);
69 m_first_face_uid_level.add(0);
70}

75void CartesianMeshNumberingMng::
76prepareLevel(Int32 level)
77{
78 if (level <= m_max_level && level >= m_min_level)
79 return;
80 if (level == m_max_level + 1) {
81 m_max_level++;
82 }
83 else if (level == m_min_level - 1) {
84 m_min_level--;
85 }
86 else {
87 ARCANE_FATAL("Level error : {0}", level);
88 }
89 m_p_to_l_level.add(level);
90
91 m_first_cell_uid_level.add(m_latest_cell_uid);
92 m_first_node_uid_level.add(m_latest_node_uid);
93 m_first_face_uid_level.add(m_latest_face_uid);
94
95 const Int64x3 nb_cell(globalNbCellsX(level), globalNbCellsY(level), globalNbCellsZ(level));
96
97 if (m_dimension == 2) {
98 m_latest_cell_uid += nb_cell.x * nb_cell.y;
99 m_latest_node_uid += (nb_cell.x + 1) * (nb_cell.y + 1);
100 m_latest_face_uid += (nb_cell.x * nb_cell.y) * 2 + nb_cell.x * 2 + nb_cell.y;
101 }
102 else {
103 m_latest_cell_uid += nb_cell.x * nb_cell.y * nb_cell.z;
104 m_latest_node_uid += (nb_cell.x + 1) * (nb_cell.y + 1) * (nb_cell.z + 1);
105 m_latest_face_uid += (nb_cell.z + 1) * nb_cell.x * nb_cell.y + (nb_cell.x + 1) * nb_cell.y * nb_cell.z + (nb_cell.y + 1) * nb_cell.z * nb_cell.x;
106 }
107}

112void CartesianMeshNumberingMng::
113updateFirstLevel()
114{
115 Int32 nb_levels_to_add = -m_min_level;
116
117 if (nb_levels_to_add == 0) {
118 return;
119 }
120
121 m_max_level += nb_levels_to_add;
122 m_min_level += nb_levels_to_add;
123
124 for (Int32& i : m_p_to_l_level) {
125 i += nb_levels_to_add;
126 }
127
128 m_nb_cell /= (m_pattern * nb_levels_to_add);
129
130 // ----------
131 // CartesianMeshCoarsening2::_recomputeMeshGenerationInfo()
132 // Recalcule les informations sur le nombre de mailles par direction.
133 auto* cmgi = ICartesianMeshGenerationInfo::getReference(m_mesh, false);
134 if (!cmgi)
135 return;
136
137 {
138 ConstArrayView<Int64> v = cmgi->ownCellOffsets();
139 cmgi->setOwnCellOffsets(v[0] / m_pattern, v[1] / m_pattern, v[2] / m_pattern);
140 }
141 {
142 ConstArrayView<Int64> v = cmgi->globalNbCells();
143 cmgi->setGlobalNbCells(v[0] / m_pattern, v[1] / m_pattern, v[2] / m_pattern);
144 }
145 {
146 ConstArrayView<Int32> v = cmgi->ownNbCells();
147 cmgi->setOwnNbCells(v[0] / m_pattern, v[1] / m_pattern, v[2] / m_pattern);
148 }
149 cmgi->setFirstOwnCellUniqueId(firstCellUniqueId(0));
150 // CartesianMeshCoarsening2::_recomputeMeshGenerationInfo()
151 // ----------
152}

238// Tant que l'on a un unique "pattern" pour x, y, z, pas besoin de trois méthodes.
239Int64 CartesianMeshNumberingMng::
240offsetLevelToLevel(Int64 coord, Integer level_from, Integer level_to) const
241{
242 if (level_from == level_to) {
243 return coord;
244 }
245 else if (level_from < level_to) {
246 return coord * m_pattern * (level_to - level_from);
247 }
248 else {
249 return coord / (m_pattern * (level_from - level_to));
250 }
251}

257cellUniqueIdToCoordX(Int64 uid, Integer level)
258{
259 const Int64 nb_cell_x = globalNbCellsX(level);
260 const Int64 nb_cell_y = globalNbCellsY(level);
261 const Int64 first_cell_uid = firstCellUniqueId(level);
262

264
265 const Int64 to2d = uid % (nb_cell_x * nb_cell_y);
266 return to2d % nb_cell_x;

429/*---------------------------------------------------------------------------*/
430
431void CartesianMeshNumberingMng::
432cellFaceUniqueIds(ArrayView<Int64> uid, Integer level, Int64x3 cell_coord)
433{
434 if (uid.size() != nbFaceByCell())
435 ARCANE_FATAL("Bad size of arrayview");
436
437 const Int64x3 nb_cell(globalNbCellsX(level), globalNbCellsY(level), globalNbCellsZ(level));
438 const Int64x3 nb_face(nb_cell + 1);
439
440 const Int64 first_face_uid = firstFaceUniqueId(level);
441
442 // Numérote les faces
443 // Cet algo n'est pas basé sur l'algo 2D.
444 // Les UniqueIDs générés sont contigües.
445 // Il est aussi possible de retrouver les UniqueIDs des faces
446 // à l'aide de la position de la cellule et la taille du maillage.
447 // De plus, l'ordre des UniqueIDs des faces d'une cellule est toujours le
448 // même (en notation localId Arcane (cell.face(i)) : 0, 3, 1, 4, 2, 5).
449 // Les UniqueIDs générés sont donc les mêmes quelque soit le découpage.
450 /*
451 x z
452 ┌──► │ ┌──►
453 │ │ │
454 y▼12 13 14 │y▼ ┌────┬────┐
455 │ 26 │ 27 │ │ │ 24 │ 25 │
456 └────┴────┘ │ 0 4 8
457 15 16 17 │
458 │ 28 │ 29 │ │ │ │ │
459 └────┴────┘ │ 2 6 10
460 z=0 │ x=0
461 - - - - - - - - - - - - - - - - - -
462 z=1 │ x=1
463 18 19 20 │ ┌────┬────┐
464 │ 32 │ 33 │ │ │ 30 │ 31 │
465 └────┴────┘ │ 1 5 9
466 21 22 23 │
467 │ 34 │ 35 │ │ │ │ │
468 └────┴────┘ │ 3 7 11
469 │
470 */
471 // On a un cube décomposé en huit cellules (2x2x2).
472 // Le schéma au-dessus représente les faces des cellules de ce cube avec
473 // les uniqueIDs que l'algorithme génèrera (sans face_adder).
474 // Pour cet algo, on commence par les faces "xy".
475 // On énumère d'abord en x, puis en y, puis en z.
476 // Une fois les faces "xy" numérotées, on fait les faces "yz".
477 // Toujours le même ordre de numérotation.
478 // On termine avec les faces "zx", encore dans le même ordre.
479 //
480 // Dans l'implémentation ci-dessous, on fait la numérotation
481 // maille par maille.
482
483 const Int64 total_face_xy = nb_face.z * nb_cell.x * nb_cell.y;
484 const Int64 total_face_xy_yz = total_face_xy + nb_face.x * nb_cell.y * nb_cell.z;
485
486 const Int64 nb_cell_before_j = cell_coord.y * nb_cell.x;
487
488 uid[0] = (cell_coord.z * nb_cell.x * nb_cell.y) + nb_cell_before_j + (cell_coord.x);
489
490 uid[3] = uid[0] + nb_cell.x * nb_cell.y;
491
492 uid[1] = (cell_coord.z * nb_face.x * nb_cell.y) + (cell_coord.y * nb_face.x) + (cell_coord.x) + total_face_xy;
493
494 uid[4] = uid[1] + 1;
495
496 uid[2] = (cell_coord.z * nb_cell.x * nb_face.y) + nb_cell_before_j + (cell_coord.x) + total_face_xy_yz;
497
498 uid[5] = uid[2] + nb_cell.x;
499
500 uid[0] += first_face_uid;
501 uid[1] += first_face_uid;
502 uid[2] += first_face_uid;
503 uid[3] += first_face_uid;
504 uid[4] += first_face_uid;
505 uid[5] += first_face_uid;
506}

510
511void CartesianMeshNumberingMng::
512cellFaceUniqueIds(ArrayView<Int64> uid, Integer level, Int64x2 cell_coord)
513{
514 if (uid.size() != nbFaceByCell())
515 ARCANE_FATAL("Bad size of arrayview");
516
517 const Int64 nb_cell_x = globalNbCellsX(level);
518 const Int64 nb_face_x = nb_cell_x + 1;
519 const Int64 first_face_uid = firstFaceUniqueId(level);
520
521 // Numérote les faces
522 // |-0--|--2-|
523 // 4| 6| 8|
524 // |-5--|-7--|
525 // 9| 11| 13|
526 // |-10-|-12-|
527 //
528 // Avec cette numérotation, HAUT < GAUCHE < BAS < DROITE
529 // Mis à part les uniqueIds de la première ligne de face, tous
530 // les uniqueIds sont contigües.
531
532 // HAUT
533 // - "(current_level_nb_face_x + current_level_nb_cell_x)" :
534 // le nombre de faces GAUCHE BAS DROITE au dessus.
535 // - "cell_coord.y * (current_level_nb_face_x + current_level_nb_cell_x)" :
536 // le nombre total de faces GAUCHE BAS DROITE au dessus.
537 // - "cell_coord.x * 2"
538 // on avance deux à deux sur les faces d'un même "coté".
539 uid[0] = cell_coord.x * 2 + cell_coord.y * (nb_face_x + nb_cell_x);
540
541 // BAS
542 // Pour BAS, c'est comme HAUT mais avec un "nombre de face du dessus" en plus.
543 uid[2] = uid[0] + (nb_face_x + nb_cell_x);
544 // GAUCHE
545 // Pour GAUCHE, c'est l'UID de BAS -1.
546 uid[3] = uid[2] - 1;
547 // DROITE
548 // Pour DROITE, c'est l'UID de BAS +1.
549 uid[1] = uid[2] + 1;
550
551 uid[0] += first_face_uid;
552 uid[1] += first_face_uid;
553 uid[2] += first_face_uid;
554 uid[3] += first_face_uid;
555}

583/*---------------------------------------------------------------------------*/
584
585void CartesianMeshNumberingMng::
586cellUniqueIdsAroundCell(ArrayView<Int64> uid, Int64 cell_uid, Int32 level)
587{
588 uid.fill(-1);
589
590 const Int64 coord_cell_x = cellUniqueIdToCoordX(cell_uid, level);
591 const Int64 coord_cell_y = cellUniqueIdToCoordY(cell_uid, level);
592
593 const Int64 nb_cells_x = globalNbCellsX(level);
594 const Int64 nb_cells_y = globalNbCellsY(level);
595
596 if (m_dimension == 2) {
597 ARCANE_ASSERT((uid.size() == 9), ("Size of uid array != 9"));
598
599 for (Integer j = -1; j < 2; ++j) {
600 const Int64 coord_around_cell_y = coord_cell_y + j;
601 if (coord_around_cell_y >= 0 && coord_around_cell_y < nb_cells_y) {
602
603 for (Integer i = -1; i < 2; ++i) {
604 const Int64 coord_around_cell_x = coord_cell_x + i;

678 │ │ │ │ =

684 /*!
685 * \brief Lambda permettant de déterminer la position d'un noeud enfant
686 * dans une maille parent.

689 const Real x = (Real)pos_x / (Real)m_pattern;
690 const Real y = (Real)pos_y / (Real)m_pattern;
691
692 const Real i = (node3.x - node0.x) * y + node0.x;
693 const Real j = (node2.x - node1.x) * y + node1.x;
694
695 const Real k = (node1.y - node0.y) * x + node0.y;
696 const Real l = (node2.y - node3.y) * x + node3.y;
697
698 const Real tx = (j - i) * x + i;
699 const Real ty = (l - k) * y + k;

701 /*
702 info() << "[txty]"
703 << " x : " << x
704 << " -- y : " << y

723 for (Integer i = 0; i < m_pattern; ++i) {

744 const Real3& node4(nodes_coords[parent_cell.node(4)]);
745 const Real3& node5(nodes_coords[parent_cell.node(5)]);

749 /*!
750 * \brief Lambda permettant de déterminer la position d'un noeud enfant
751 * dans une maille parent.
752 */
753 auto txtytz = [&](Integer pos_x, Integer pos_y, Integer pos_z) -> Real3 {
754 const Real x = (Real)pos_x / (Real)m_pattern;
755 const Real y = (Real)pos_y / (Real)m_pattern;
756 const Real z = (Real)pos_z / (Real)m_pattern;
757
758 // Face (m, n, o, p) entre les faces (node0, node1, node2, node3) et (node4, node5, node6, node7).
759 const Real3 m = (node4 - node0) * z + node0;
760 const Real3 n = (node5 - node1) * z + node1;
761 const Real3 o = (node6 - node2) * z + node2;
762 const Real3 p = (node7 - node3) * z + node3;
763
764 // On calcule tx et ty comme en 2D mais sur la face (m, n, o, p).
765 const Real i = (p.x - m.x) * y + m.x;
766 const Real j = (o.x - n.x) * y + n.x;
767
768 const Real tx = (j - i) * x + i;
769
770 const Real k = (n.y - m.y) * x + m.y;
771 const Real l = (o.y - p.y) * x + p.y;
772
773 const Real ty = (l - k) * y + k;
774
775 const Real q = (p.z - m.z) * y + m.z;
776 const Real r = (o.z - n.z) * y + n.z;
777
778 const Real s = (n.z - m.z) * x + m.z;
779 const Real t = (o.z - p.z) * x + p.z;
780
781 const Real tz = (((r - q) * x + q) + ((t - s) * y + s)) * 0.5;
782
783 /*
784 info() << "[txtytz]"
785 << " x : " << x
786 << " -- y : " << y
787 << " -- z : " << z
788 << " -- node0 : " << node0

790 << " -- node2 : " << node2
791 << " -- node3 : " << node3
792 << " -- node4 : " << node4

794 << " -- node6 : " << node6
795 << " -- node7 : " << node7

797 << " -- n : " << n
798 << " -- o : " << o
799 << " -- p : " << p
800 << " -- j : " << j
801 << " -- k : " << k
802 << " -- l : " << l
803 << " -- q : " << q
804 << " -- r : " << r
805 << " -- s : " << s
806 << " -- t : " << t
807 << " -- tx : " << tx
808 << " -- ty : " << ty
809 << " -- tz : " << tz;
810 */
811 return { tx, ty, tz };
812 };
813
814 const Integer node_1d_3d_x[] = { 0, 1, 1, 0, 0, 1, 1, 0 };
815 const Integer node_1d_3d_y[] = { 0, 0, 1, 1, 0, 0, 1, 1 };
816 const Integer node_1d_3d_z[] = { 0, 0, 0, 0, 1, 1, 1, 1 };

818 for (Integer k = 0; k < m_pattern; ++k) {

820 for (Integer i = 0; i < m_pattern; ++i) {
821

827 for (Integer inode = begin; inode < end; ++inode) {
828 nodes_coords[child.node(inode)] = txtytz(i + node_1d_3d_x[inode], j + node_1d_3d_y[inode], k + node_1d_3d_z[inode]);
829 // Real3 pos = txtytz(i + node_1d_3d_x[inode], j + node_1d_3d_y[inode], k + node_1d_3d_z[inode]);
830 // nodes_coords[child.node(inode)] = pos;
831 // info() << "Node uid : " << child.node(inode).uniqueId()
