32CartesianMeshNumberingMng::
33CartesianMeshNumberingMng(IMesh* mesh)
34: TraceAccessor(mesh->traceMng())
35, m_mesh(mesh)
36, m_pattern(2)
37{
38 auto* m_generation_info = ICartesianMeshGenerationInfo::getReference(m_mesh,true);
39
40 Int64ConstArrayView global_nb_cells_by_direction = m_generation_info->globalNbCells();
41 m_nb_cell_x = global_nb_cells_by_direction[MD_DirX];
42 if (m_nb_cell_x <= 0)
43 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirX] (should be >0)", m_nb_cell_x);
44
45 m_nb_cell_y = global_nb_cells_by_direction[MD_DirY];
46 if (m_nb_cell_y <= 0)
47 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirY] (should be >0)", m_nb_cell_y);
48
49 m_nb_cell_z = global_nb_cells_by_direction[MD_DirZ];
50 if (m_nb_cell_z <= 0)
51 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirZ] (should be >0)", m_nb_cell_z);
52
53 m_first_cell_uid_level.add(0);
54 m_first_node_uid_level.add(0);
55 m_first_face_uid_level.add(0);
56}

58Int64 CartesianMeshNumberingMng::
59getFirstCellUidLevel(Integer level)
60{
61 if(level < m_first_cell_uid_level.size()){
62 return m_first_cell_uid_level[level];
63 }
64
65 const Int32 dimension = m_mesh->dimension();
66
67 Int64 level_i_nb_cell_x = getGlobalNbCellsX(m_nb_cell_level.size());
68 Int64 level_i_nb_cell_y = getGlobalNbCellsY(m_nb_cell_level.size());
69 Int64 level_i_nb_cell_z = getGlobalNbCellsZ(m_nb_cell_level.size());
70
71 for(Integer i = m_nb_cell_level.size(); i < level+1; ++i){
72 if(dimension == 2){
73 m_nb_cell_level.add(level_i_nb_cell_x * level_i_nb_cell_y);
74 }
75 else if(dimension == 3){
76 m_nb_cell_level.add(level_i_nb_cell_x * level_i_nb_cell_y * level_i_nb_cell_z);
77 }
78 m_first_cell_uid_level.add(m_first_cell_uid_level[i] + m_nb_cell_level[i]);
79
80 level_i_nb_cell_x *= m_pattern;
81 level_i_nb_cell_y *= m_pattern;
82 level_i_nb_cell_z *= m_pattern;
83 }
84
85 return m_first_cell_uid_level[level];
86}

89Int64 CartesianMeshNumberingMng::
90getFirstNodeUidLevel(Integer level)
91{
92 if(level < m_first_node_uid_level.size()){
93 return m_first_node_uid_level[level];
94 }
95
96 const Int32 dimension = m_mesh->dimension();
97
98 Int64 level_i_nb_cell_x = getGlobalNbCellsX(m_nb_node_level.size());
99 Int64 level_i_nb_cell_y = getGlobalNbCellsY(m_nb_node_level.size());
100 Int64 level_i_nb_cell_z = getGlobalNbCellsZ(m_nb_node_level.size());
101
102 for(Integer i = m_nb_node_level.size(); i < level+1; ++i){
103 if(dimension == 2){
104 m_nb_node_level.add((level_i_nb_cell_x + 1) * (level_i_nb_cell_y + 1));
105 }
106 else if(dimension == 3){
107 m_nb_node_level.add((level_i_nb_cell_x + 1) * (level_i_nb_cell_y + 1) * (level_i_nb_cell_z + 1));
108 }
109 m_first_node_uid_level.add(m_first_node_uid_level[i] + m_nb_node_level[i]);
110
111 level_i_nb_cell_x *= m_pattern;
112 level_i_nb_cell_y *= m_pattern;
113 level_i_nb_cell_z *= m_pattern;
114 }
115
116 return m_first_node_uid_level[level];

120Int64 CartesianMeshNumberingMng::
121getFirstFaceUidLevel(Integer level)
122{
123 if(level < m_first_face_uid_level.size()){
124 return m_first_face_uid_level[level];
125 }
126
127 const Int32 dimension = m_mesh->dimension();
128
129 Int64 level_i_nb_cell_x = getGlobalNbCellsX(m_nb_face_level.size());
130 Int64 level_i_nb_cell_y = getGlobalNbCellsY(m_nb_face_level.size());
131 Int64 level_i_nb_cell_z = getGlobalNbCellsZ(m_nb_face_level.size());
132
133 for(Integer i = m_nb_face_level.size(); i < level+1; ++i){
134 if(dimension == 2){
135 m_nb_face_level.add((level_i_nb_cell_x * level_i_nb_cell_y) * 2 + level_i_nb_cell_x*2 + level_i_nb_cell_y);
136 }
137 else if(dimension == 3){
138 m_nb_face_level.add((level_i_nb_cell_z + 1) * level_i_nb_cell_x * level_i_nb_cell_y
139 + (level_i_nb_cell_x + 1) * level_i_nb_cell_y * level_i_nb_cell_z
140 + (level_i_nb_cell_y + 1) * level_i_nb_cell_z * level_i_nb_cell_x);
141 }
142 m_first_face_uid_level.add(m_first_face_uid_level[i] + m_nb_face_level[i]);
143
144 level_i_nb_cell_x *= m_pattern;
145 level_i_nb_cell_y *= m_pattern;
146 level_i_nb_cell_z *= m_pattern;
147 }
148
149 return m_first_face_uid_level[level];
150}

185uidToCoordX(Int64 uid, Integer level)
186{
187 const Int64 nb_cell_x = getGlobalNbCellsX(level);
188 const Int64 nb_cell_y = getGlobalNbCellsY(level);
189 const Int64 first_cell_uid = getFirstCellUidLevel(level);
190
191 uid -= first_cell_uid;
192
193 const Int64 to2d = uid % (nb_cell_x * nb_cell_y);
194 return to2d % nb_cell_x;

269getNodeUids(ArrayView<Int64> uid, Integer level, Int64 cell_coord_i, Int64 cell_coord_j, Int64 cell_coord_k)
270{
271 if(uid.size() != getNbNode())

273

310void CartesianMeshNumberingMng::
311getFaceUids(ArrayView<Int64> uid, Integer level, Int64 cell_coord_i, Int64 cell_coord_j, Int64 cell_coord_k)
312{
313 if(uid.size() != getNbFace())
314 ARCANE_FATAL("Bad size of arrayview");
315
316 const Int64 nb_cell_x = getGlobalNbCellsX(level);
317 const Int64 nb_cell_y = getGlobalNbCellsY(level);
318 const Int64 nb_cell_z = getGlobalNbCellsZ(level);
319
320 const Int64 nb_face_x = nb_cell_x + 1;
321 const Int64 nb_face_y = nb_cell_y + 1;
322 const Int64 nb_face_z = nb_cell_z + 1;
323
324 const Int64 first_face_uid = getFirstFaceUidLevel(level);
325
326 // Numérote les faces
327 // Cet algo n'est pas basé sur l'algo 2D.
328 // Les UniqueIDs générés sont contigües.
329 // Il est aussi possible de retrouver les UniqueIDs des faces
330 // à l'aide de la position de la cellule et la taille du maillage.
331 // De plus, l'ordre des UniqueIDs des faces d'une cellule est toujours le
332 // même (en notation localId Arcane (cell.face(i)) : 0, 3, 1, 4, 2, 5).
333 // Les UniqueIDs générés sont donc les mêmes quelque soit le découpage.
334 /*
335 x z
336 ┌──► │ ┌──►
337 │ │ │
338 y▼12 13 14 │y▼ ┌────┬────┐
339 │ 26 │ 27 │ │ │ 24 │ 25 │
340 └────┴────┘ │ 0 4 8
341 15 16 17 │
342 │ 28 │ 29 │ │ │ │ │
343 └────┴────┘ │ 2 6 10
344 z=0 │ x=0
345 - - - - - - - - - - - - - - - - - -
346 z=1 │ x=1
347 18 19 20 │ ┌────┬────┐
348 │ 32 │ 33 │ │ │ 30 │ 31 │
349 └────┴────┘ │ 1 5 9
350 21 22 23 │
351 │ 34 │ 35 │ │ │ │ │
352 └────┴────┘ │ 3 7 11
353 │
354 */
355 // On a un cube décomposé en huit cellules (2x2x2).
356 // Le schéma au-dessus représente les faces des cellules de ce cube avec
357 // les uniqueIDs que l'algorithme génèrera (sans face_adder).
358 // Pour cet algo, on commence par les faces "xy".
359 // On énumère d'abord en x, puis en y, puis en z.
360 // Une fois les faces "xy" numérotées, on fait les faces "yz".
361 // Toujours le même ordre de numérotation.
362 // On termine avec les faces "zx", encore dans le même ordre.
363 //
364 // Dans l'implémentation ci-dessous, on fait la numérotation
365 // maille par maille.
366
367 const Int64 total_face_xy = nb_face_z * nb_cell_x * nb_cell_y;
368 const Int64 total_face_xy_yz = total_face_xy + nb_face_x * nb_cell_y * nb_cell_z;
369
370 const Int64 nb_cell_before_j = cell_coord_j * nb_cell_x;
371
372
373
374 uid[0] = (cell_coord_k * nb_cell_x * nb_cell_y)
375 + nb_cell_before_j
376 + (cell_coord_i);
377
378 uid[3] = uid[0] + nb_cell_x * nb_cell_y;
379
380 uid[1] = (cell_coord_k * nb_face_x * nb_cell_y)
381 + (cell_coord_j * nb_face_x)
382 + (cell_coord_i) + total_face_xy;
383
384 uid[4] = uid[1] + 1;
385
386 uid[2] = (cell_coord_k * nb_cell_x * nb_face_y)
387 + nb_cell_before_j
388 + (cell_coord_i) + total_face_xy_yz;
389
390 uid[5] = uid[2] + nb_cell_x;
391
392
393 uid[0] += first_face_uid;
394 uid[1] += first_face_uid;
395 uid[2] += first_face_uid;
396 uid[3] += first_face_uid;
397 uid[4] += first_face_uid;
398 uid[5] += first_face_uid;
399}

402void CartesianMeshNumberingMng::
403getFaceUids(ArrayView<Int64> uid, Integer level, Int64 cell_coord_i, Int64 cell_coord_j)
404{
405 if(uid.size() != getNbFace())
406 ARCANE_FATAL("Bad size of arrayview");
407
408 const Int64 nb_cell_x = getGlobalNbCellsX(level);
409 const Int64 nb_face_x = nb_cell_x + 1;
410 const Int64 first_face_uid = getFirstFaceUidLevel(level);
411
412 // Numérote les faces
413 // |-0--|--2-|
414 // 4| 6| 8|
415 // |-5--|-7--|
416 // 9| 11| 13|
417 // |-10-|-12-|
418 //
419 // Avec cette numérotation, HAUT < GAUCHE < BAS < DROITE
420 // Mis à part les uniqueIds de la première ligne de face, tous
421 // les uniqueIds sont contigües.
422
423 // HAUT
424 // - "(current_level_nb_face_x + current_level_nb_cell_x)" :
425 // le nombre de faces GAUCHE BAS DROITE au dessus.
426 // - "cell_coord_j * (current_level_nb_face_x + current_level_nb_cell_x)" :
427 // le nombre total de faces GAUCHE BAS DROITE au dessus.
428 // - "cell_coord_i * 2"
429 // on avance deux à deux sur les faces d'un même "coté".
430 uid[0] = cell_coord_i * 2 + cell_coord_j * (nb_face_x + nb_cell_x);
431
432 // BAS
433 // Pour BAS, c'est comme HAUT mais avec un "nombre de face du dessus" en plus.
434 uid[2] = uid[0] + (nb_face_x + nb_cell_x);
435 // GAUCHE
436 // Pour GAUCHE, c'est l'UID de BAS -1.
437 uid[3] = uid[2] - 1;
438 // DROITE
439 // Pour DROITE, c'est l'UID de BAS +1.
440 uid[1] = uid[2] + 1;
441
442 uid[0] += first_face_uid;
443 uid[1] += first_face_uid;
444 uid[2] += first_face_uid;
445 uid[3] += first_face_uid;
446}

448void CartesianMeshNumberingMng::
449getCellUidsAround(ArrayView<Int64> uid, Cell cell)
450{
451 uid.fill(-1);
452
453 Int64 coord_cell_x = uidToCoordX(cell);
454 Int64 coord_cell_y = uidToCoordY(cell);
455
456 Integer level = cell.level();
457 Int64 nb_cells_y = getGlobalNbCellsY(level);
458 Int64 nb_cells_x = getGlobalNbCellsX(level);
459
460 if(m_mesh->dimension() == 2){
461 ARCANE_ASSERT((uid.size() == 9), ("Size of uid array != 9"));
462
463 for(Integer j = -1; j < 2; ++j){
464 Int64 coord_around_cell_y = coord_cell_y + j;
465 if(coord_around_cell_y >= 0 && coord_around_cell_y < nb_cells_y){
466
467 for(Integer i = -1; i < 2; ++i){
468 Int64 coord_around_cell_x = coord_cell_x + i;
469 if(coord_around_cell_x >= 0 && coord_around_cell_x < nb_cells_x) {
470 uid[(i+1) + ((j+1) * 3)] = getCellUid(level, coord_around_cell_x, coord_around_cell_y);
471 }
472 }
473 }
474 }
475 }
476
477 else {
478 ARCANE_ASSERT((uid.size() == 27), ("Size of uid array != 27"));
479
480 Int64 coord_cell_z = uidToCoordZ(cell);
481 Int64 nb_cells_z = getGlobalNbCellsZ(level);
482
483 for(Integer k = -1; k < 2; ++k){
484 Int64 coord_around_cell_z = coord_cell_z + k;
485 if(coord_around_cell_z >= 0 && coord_around_cell_z < nb_cells_z) {
486
487 for(Integer j = -1; j < 2; ++j){
488 Int64 coord_around_cell_y = coord_cell_y + j;
489 if(coord_around_cell_y >= 0 && coord_around_cell_y < nb_cells_y){
490
491 for(Integer i = -1; i < 2; ++i){
492 Int64 coord_around_cell_x = coord_cell_x + i;
493 if(coord_around_cell_x >= 0 && coord_around_cell_x < nb_cells_x) {
494 uid[(i+1) + ((j+1) * 3) + ((k+1) * 9)] = getCellUid(level, coord_around_cell_x, coord_around_cell_y, coord_around_cell_z);
495 }
496 }
497 }
498 }
499 }
500 }

504void CartesianMeshNumberingMng::
505setNodeCoordinates(Cell child_cell)
506{
507 if (!(child_cell.itemBase().flags() & ItemFlags::II_JustAdded)) {
508 ARCANE_FATAL("Cell not II_JustAdded");
509 }
510 Cell parent_cell = child_cell.hParent();
511
512 VariableNodeReal3& nodes_coords = m_mesh->nodesCoordinates();
513
514 Real3 pos(
515 Real(uidToCoordX(child_cell) % m_pattern),
516 Real(uidToCoordY(child_cell) % m_pattern),
517 (m_mesh->dimension() == 2 ? 0 : Real(uidToCoordZ(child_cell) % m_pattern))
518 );
519
520 Real3 size_child_cell;
521
522 if(m_mesh->dimension() == 2) {
523 size_child_cell = nodes_coords[parent_cell.node(2)] - nodes_coords[parent_cell.node(0)];
524 }
525 else{
526 size_child_cell = nodes_coords[parent_cell.node(6)] - nodes_coords[parent_cell.node(0)];
527 }
528 size_child_cell /= m_pattern;
529
530 Real3 origin_parent_cell(nodes_coords[parent_cell.node(0)]);
531 Real3 origin_child_cell(origin_parent_cell + (size_child_cell * pos));
532
533 nodes_coords[child_cell.node(0)] = origin_child_cell;
534
535 nodes_coords[child_cell.node(1)] = origin_child_cell;
536 nodes_coords[child_cell.node(1)].x += size_child_cell.x;
537
538 nodes_coords[child_cell.node(2)] = origin_child_cell;
539 nodes_coords[child_cell.node(2)].x += size_child_cell.x;
540 nodes_coords[child_cell.node(2)].y += size_child_cell.y;
541
542 nodes_coords[child_cell.node(3)] = origin_child_cell;
543 nodes_coords[child_cell.node(3)].y += size_child_cell.y;
544
545 if(m_mesh->dimension() == 3) {
546 nodes_coords[child_cell.node(4)] = origin_child_cell;
547 nodes_coords[child_cell.node(4)].z += size_child_cell.z;
