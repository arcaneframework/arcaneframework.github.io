218 constexpr void _setFromList(std::initializer_list<T> v)
219 {
220 auto s = v.size();
221 auto ptr = v.begin();
222 if (s > 0)
223 x = *ptr;
224 ++ptr;
225 if (s > 1)
226 y = *ptr;
227 ++ptr;
228 if (s > 2)
229 z = *ptr;
230 }
