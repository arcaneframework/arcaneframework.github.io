106  protected:
107  ComponentCellEnumerator(ConstArrayView<ComponentItemInternal*> items,
108  ConstArrayView<MatVarIndex> matvar_indexes,
109  IMeshComponent* component)
110  : m_index(0), m_size(items.size()), m_items(items)
111  , m_matvar_indexes(matvar_indexes), m_component(component)
112  {
113 #ifdef ARCANE_CHECK
114  if (m_index<m_size)
115  _check();
116 #endif
117  }
118  public:
119  static ComponentCellEnumerator create(IMeshComponent* component);
120  static ComponentCellEnumerator create(const ComponentItemVector& v);
121  static ComponentCellEnumerator create(ComponentItemVectorView v);
122  public:
123  void operator++()
124  {
125  ++m_index;
126 #ifdef ARCANE_CHECK
127  if (m_index<m_size)
128  _check();
129 #endif
130  }
131  bool hasNext() const { return m_index<m_size; }
132 
133  ComponentCell operator*() const
134  {
135  return ComponentCell(m_items[m_index]);
136  }
137 
138  Integer index() const { return m_index; }
139  MatVarIndex _varIndex() const { return m_matvar_indexes[m_index]; }
140 
141  operator ComponentItemLocalId() const
142  {
143  return ComponentItemLocalId(m_matvar_indexes[m_index]);
144  }
145 
146  protected:
147 
148  Int32 _varArrayIndex() const { return m_matvar_indexes[m_index].arrayIndex(); }
149  Int32 _varValueIndex() const { return m_matvar_indexes[m_index].valueIndex(); }
150 
151  protected:
152  void _check() const
153  {
154  ComponentItemInternal* ii = m_items[m_index];
155  MatVarIndex mvi = ii->variableIndex();
156  Int32 i_var_array_index = mvi.arrayIndex();
157  Int32 mv_array_index = _varArrayIndex();
158  if (i_var_array_index!=mv_array_index)
159  ARCANE_FATAL("Bad 'var_array_index' in ComponentCell matvar='{0}' registered='{1}' index={2}",
160  mvi,m_matvar_indexes[m_index],m_index);
161  Int32 i_var_value_index = mvi.valueIndex();
162  Int32 mv_value_index = _varValueIndex();
163  if (i_var_value_index!=mv_value_index)
164  ARCANE_FATAL("Bad 'var_value_index' for ComponentCell matvar='{0}' registered='{1}' index={2}",
165  mvi,m_matvar_indexes[m_index],m_index);
166  }
167  protected:
168  Integer m_index;
169  Integer m_size;
170  ConstArrayView<ComponentItemInternal*> m_items;
171  ConstArrayView<MatVarIndex> m_matvar_indexes;
172  IMeshComponent* m_component;
173 };
