29/*!
30 * \brief Type d'une donnée de base.
31 *
32 * \note: les valeurs doivent être contigues et tenir sur un octet et ne
33 * doivent pas être modifiées car elles sont utilisés pour la sérialisation
34 * par exemple. Il est cependant possible d'en ajouter.
35 */
36enum class eBasicDataType : unsigned char
37{
38 Unknown = 0, //!< Donnée de type inconnu ou non initialisé
39 Byte = 1, //!< Donnée de type octet
40 Float16 = 2, //!< Donnée de type Float16
41 Float32 = 3, //!< Donnée de type Float32
42 Float64 = 4, //!< Donnée de type Float64
43 Float128 = 5, //!< Donnée de type Float128
44 Int16 = 6, //!< Donnée de type entier 16 bits
45 Int32 = 7, //!< Donnée de type entier 32 bits
46 Int64 = 8, //!< Donnée de type entier 64 bits
47 Int128 = 9, //!< Donnée de type entier 128 bits
48 BFloat16 = 10, //! < Donnée de type BFloat16
49 Int8 = 11, //! Donnée de type entier 8 bits
50 Real = Float64 //! Donnée de type Float64
51};
52//! Nombre de types de base supportés
53constexpr unsigned char NB_BASIC_DATA_TYPE = 12;
