182{

208/*!
209 * \brief Pour une face, positionne à la fois la back cell et la front cell.
210 *
211 * \a back_cell_lid et/ou \a front_cell_lid peuvent valoir NULL_ITEM_LOCAL_ID
212 * ce qui signifie que l'entité n'a pas de back cell ou front cell. Si les
213 * deux valeurs sont nulles, alors la face est considérée comme n'ayant
214 * plus de mailles connectées.
215 *
216 * \note Cette méthode est utilisée uniquement par FaceFamily.
217 */
218void impl::MutableItemBase::
219_setFaceBackAndFrontCells(Int32 back_cell_lid,Int32 front_cell_lid)
220{
221 if (front_cell_lid==NULL_ITEM_LOCAL_ID){
222 // Reste uniquement la back_cell ou aucune maille.
223 Int32 mod_flags = (back_cell_lid!=NULL_ITEM_LOCAL_ID) ? (II_Boundary | II_HasBackCell | II_BackCellIsFirst) : 0;
224 _setFaceInfos(mod_flags);
225 }
226 else if (back_cell_lid==NULL_ITEM_LOCAL_ID){
227 // Reste uniquement la front cell
228 _setFaceInfos(II_Boundary | II_HasFrontCell | II_FrontCellIsFirst);
229 }
230 else{
231 // Il y a deux mailles connectées.
232 _setFaceInfos(II_HasFrontCell | II_HasBackCell | II_BackCellIsFirst);
233 }
234}
