44void VariableUtils::
45prefetchVariableAsync(IVariable* var, RunQueue* queue_or_null)
46{
47 ARCANE_CHECK_POINTER(var);
48 if (!queue_or_null)
49 return;
50 if (!var->isUsed())
51 return;
52 IData* d = var->data();
53 INumericDataInternal* nd = d->_commonInternal()->numericData();
54 if (!nd)
55 return;
56 ConstMemoryView mem_view = nd->memoryView();
57 queue_or_null->prefetchMemory(MemoryPrefetchArgs(mem_view).addAsync());
58}
