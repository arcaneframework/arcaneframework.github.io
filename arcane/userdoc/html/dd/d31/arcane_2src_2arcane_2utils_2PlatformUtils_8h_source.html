398// Définition du pragma pour indiquer l'indépendance des itérations
399
400/*!
401 * \def ARCANE_PRAGMA_IVDEP
402 * Pragma pour indiquer au compilateur que les itérations d'une boucle sont
403 * indépendanntes. Ce pragma se positionne avant une boucle 'for'.
404 */
405
406/*!
407 * \def ARCANE_PRAGMA_IVDEP_VALUE
408 * Valeur du pragma ARCANE_PRAGMA_IVDEP
409 */
410
411// Pour les définitions, il faut finir par GCC car Clang et ICC définissent
412// la macro __GNU__
413// Pour CLANG, il n'y a pas encore d'équivalent au pragma ivdep de ICC.
414// Celui qui s'en approche le plus est:
415// #pragma clang loop vectorize(enable)
416// mais il ne force pas la vectorisation.
417#ifdef __clang__
418# define ARCANE_PRAGMA_IVDEP_VALUE "clang loop vectorize(enable)"
419#else
420# ifdef __INTEL_COMPILER
421# define ARCANE_PRAGMA_IVDEP_VALUE "ivdep"
422# else
423# ifdef __GNUC__
424# if (__GNUC__>=5)
425# define ARCANE_PRAGMA_IVDEP_VALUE "GCC ivdep"
426# endif
427# endif
428# endif
429#endif
430
431#ifdef ARCANE_PRAGMA_IVDEP_VALUE
432#define ARCANE_PRAGMA_IVDEP _Pragma(ARCANE_PRAGMA_IVDEP_VALUE)
433#else
434#define ARCANE_PRAGMA_IVDEP
435#define ARCANE_PRAGMA_IVDEP_VALUE ""
436#endif
