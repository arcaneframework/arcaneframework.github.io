350 // Définition du pragma pour indiquer l'indépendance des itérations
351 
363 // Pour les définitions, il faut finir par GCC car Clang et ICC définissent
364 // la macro __GNU__
365 // Pour CLANG, il n'y a pas encore d'équivalent au pragma ivdep de ICC.
366 // Celui qui s'en approche le plus est:
367 // #pragma clang loop vectorize(enable)
368 // mais il ne force pas la vectorisation.
369 #ifdef __clang__
370 # define ARCANE_PRAGMA_IVDEP_VALUE "clang loop vectorize(enable)"
371 #else
372 # ifdef __INTEL_COMPILER
373 # define ARCANE_PRAGMA_IVDEP_VALUE "ivdep"
374 # else
375 # ifdef __GNUC__
376 # if (__GNUC__>=5)
377 # define ARCANE_PRAGMA_IVDEP_VALUE "GCC ivdep"
378 # endif
379 # endif
380 # endif
381 #endif
382 
383 #ifdef ARCANE_PRAGMA_IVDEP_VALUE
384 #define ARCANE_PRAGMA_IVDEP _Pragma(ARCANE_PRAGMA_IVDEP_VALUE)
385 #else
386 #define ARCANE_PRAGMA_IVDEP
387 #define ARCANE_PRAGMA_IVDEP_VALUE ""
388 #endif
