38 template<typename ItemType,typename Lambda>
39 void _doIndirectThreadLambda(ItemVectorViewT<ItemType> sub_items,Lambda func)
40 {
41  typedef typename ItemType::LocalIdType LocalIdType;
42 
43  auto privatizer = privatize(func);
44  auto& body = privatizer.privateCopy();
45 
46  ENUMERATE_ITEM(iitem,sub_items){

56 template<typename ItemType,typename Lambda> void
57 _applyItems(RunCommand& command,ItemVectorViewT<ItemType> items,Lambda func)
58 {
59  // TODO: fusionner la partie commune avec 'applyLoop'
60  Integer vsize = items.size();
61  if (vsize==0)
62  return;
63  typedef typename ItemType::LocalIdType LocalIdType;
64  impl::RunCommandLaunchInfo launch_info(command);
65  const eExecutionPolicy exec_policy = launch_info.executionPolicy();
66  switch(exec_policy){
67  case eExecutionPolicy::CUDA:
68 #if defined(ARCANE_COMPILING_CUDA)
69  {
70  launch_info.beginExecute();
71  SmallSpan<const Int32> local_ids = items.localIds();
72  auto [b,t] = launch_info.computeThreadBlockInfo(vsize);
73  cudaStream_t* s = reinterpret_cast<cudaStream_t*>(launch_info._internalStreamImpl());
74  // TODO: utiliser cudaLaunchKernel() à la place.
75  impl::doIndirectGPULambda<ItemType,Lambda> <<<b,t,0,*s>>>(local_ids,std::forward<Lambda>(func));
76  }
77 #else
78  ARCANE_FATAL("Requesting CUDA kernel execution but the kernel is not compiled with CUDA compiler");
79 #endif
80  break;
81  case eExecutionPolicy::HIP:
82 #if defined(ARCANE_COMPILING_HIP)
83  {
84  launch_info.beginExecute();
85  SmallSpan<const Int32> local_ids = items.localIds();
86  auto [b,t] = launch_info.computeThreadBlockInfo(vsize);
87  hipStream_t* s = reinterpret_cast<hipStream_t*>(launch_info._internalStreamImpl());
88  auto& loop_func = impl::doIndirectGPULambda<ItemType,Lambda>;
89  hipLaunchKernelGGL(loop_func,b,t,0,*s, local_ids,std::forward<Lambda>(func));
90  }
91 #else
92  ARCANE_FATAL("Requesting HIP kernel execution but the kernel is not compiled with HIP compiler");
93 #endif
94  break;
95  case eExecutionPolicy::Sequential:
96  {
97  launch_info.beginExecute();
98  ENUMERATE_ITEM(iitem,items){
99  func(LocalIdType(iitem.itemLocalId()));
100  }
101  }
102  break;
103  case eExecutionPolicy::Thread:
104  {
105  launch_info.beginExecute();
106  arcaneParallelForeach(items,
107  [&](ItemVectorViewT<ItemType> sub_items)
108  {
109  impl::_doIndirectThreadLambda(sub_items,func);
110  });
111  }
112  break;
113  default:
114  ARCANE_FATAL("Invalid execution policy '{0}'",exec_policy);
115  }
116  launch_info.endExecute();
117 }
