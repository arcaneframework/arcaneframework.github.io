41template<typename ItemType,typename Lambda>
42void _doIndirectThreadLambda(ItemVectorViewT<ItemType> sub_items,const Lambda& func)
43{
44 typedef typename ItemType::LocalIdType LocalIdType;
45
46 auto privatizer = privatize(func);
47 auto& body = privatizer.privateCopy();
48
49 ENUMERATE_ITEM(iitem,sub_items){
50 body(LocalIdType(iitem.itemLocalId()));
51 }
52}

56/*!
57 * \brief Applique l'enumération \a func sur la liste d'entité \a items.
58 */
59template<typename ItemType,typename Lambda> void
60_applyItems(RunCommand& command,ItemVectorViewT<ItemType> items,const Lambda& func)
61{
62 // TODO: fusionner la partie commune avec 'applyLoop'
63 Integer vsize = items.size();
64 if (vsize==0)
65 return;
66 typedef typename ItemType::LocalIdType LocalIdType;
67 impl::RunCommandLaunchInfo launch_info(command,vsize);
68 const eExecutionPolicy exec_policy = launch_info.executionPolicy();
69 launch_info.computeLoopRunInfo(vsize);
70 launch_info.beginExecute();
71 switch(exec_policy){
72 case eExecutionPolicy::CUDA:
73 _applyKernelCUDA(launch_info,ARCANE_KERNEL_CUDA_FUNC(doIndirectGPULambda)<ItemType,Lambda>,func,items.localIds());
74 break;
75 case eExecutionPolicy::HIP:
76 _applyKernelHIP(launch_info,ARCANE_KERNEL_HIP_FUNC(doIndirectGPULambda)<ItemType,Lambda>,func,items.localIds());
77 break;
78 case eExecutionPolicy::Sequential:
79 ENUMERATE_NO_TRACE_(ItemType,iitem,items){
80 func(LocalIdType(iitem.itemLocalId()));
81 }
82 break;
83 case eExecutionPolicy::Thread:
84 arcaneParallelForeach(items,launch_info.loopRunInfo(),
85 [&](ItemVectorViewT<ItemType> sub_items)
86 {
87 impl::_doIndirectThreadLambda(sub_items,func);
88 });
89 break;
90 default:
91 ARCANE_FATAL("Invalid execution policy '{0}'",exec_policy);
92 }
93 launch_info.endExecute();
94}
