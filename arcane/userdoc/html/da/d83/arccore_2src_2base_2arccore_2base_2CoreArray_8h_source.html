160 static Span<const DataType> _constView(const std::vector<DataType>& c)
161 {
162 Int64 s = static_cast<Int64>(c.size());
163 return Span<const DataType>(c.data(),s);
164 }
165 static Span<DataType> _view(std::vector<DataType>& c)
166 {
167 Int64 s = static_cast<Int64>(c.size());
168 return Span<DataType>(c.data(),s);

172 c.push_back(v);

175 {
176 return c.data();
177 }
178 static const DataType* _data(const std::vector<DataType>& c)
179 {
180 return c.data();
181 }
182 private:
183
184 ContainerType m_p;
185};
