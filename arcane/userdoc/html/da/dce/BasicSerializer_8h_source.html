32 template <class DataType>
33 class BasicSerializerDataT
34 {
35  public:
36  BasicSerializerDataT()
37  : m_reserved_size(0)
38  , m_current_position(0)
39  {}
40 
41  public:
42  void put(Span<const DataType> values)
43  {
44  Int64 n = values.size();
45  Int64 cp = m_current_position;
46  Int64 max_size = 1 + m_buffer.size();
47  arccoreCheckAt(n + cp, max_size);
48 
49  DataType* ptr = m_buffer.data() + cp;
50  const DataType* vptr = values.data();
51  for (Int64 i = 0; i < n; ++i)
52  ptr[i] = vptr[i];
53  m_current_position += n;
54  }
55 
56  void get(Span<DataType> values)
57  {
58  Int64 n = values.size();
59  Int64 cp = m_current_position;
60  Int64 max_size = 1 + m_buffer.size();
61  arccoreCheckAt(n + cp, max_size);
62 
63  const DataType* ptr = m_buffer.data() + cp;
64  DataType* vptr = values.data();
65  for (Int64 i = 0; i < n; ++i)
66  vptr[i] = ptr[i];
67  m_current_position += n;
68  }
69 
70  public:
71  Int64 m_reserved_size;
72  Int64 m_current_position;
73  Span<DataType> m_buffer;
74 };
