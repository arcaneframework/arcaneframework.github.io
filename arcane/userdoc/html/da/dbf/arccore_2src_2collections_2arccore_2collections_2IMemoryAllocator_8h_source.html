27/*!
28 * \brief Interface d'un allocateur pour la mémoire.
29 *
30 * Cette classe définit une interface pour l'allocation mémoire utilisée
31 * par les classes tableaux de Arccore (Array, UniqueArray).
32 *
33 * Une instance de cette classe doit rester valide tant qu'il existe
34 * des tableaux l'utilisant. Comme l'allocateur est transféré lors des copies,
35 * il est préférable que les allocateurs soient des objets statiques qui
36 * dont la durée de vie est celle du programme.
37 *
38 * Les allocateurs n'ont pas d'état modifiables spécifiques et doivent fonctionner en
39 * multi-threading.
40 *
41 * Depuis la version 2.3 de Arccore, les méthodes sans MemoryAllocationArgs sont
42 * obsolètes.
43 */
44class ARCCORE_COLLECTIONS_EXPORT IMemoryAllocator
45{
46 public:
47
48 /*!
49 * \brief Détruit l'allocateur.
50 *
51 * Les objets alloués par l'allocateur doivent tous avoir été désalloués.
52 */
53 virtual ~IMemoryAllocator() = default;
54
55 public:
56
57 /*!
58 * \brief Indique si l'allocateur supporte la sémantique de realloc.
59 *
60 * Les allocateurs par défaut du C (malloc/realloc/free) supportent
61 * évidemment le realloc mais ce n'est pas forcément le cas
62 * des allocateurs spécifiques avec alignement mémoire (comme
63 * par exemple posix_memalign).
64 */
65 virtual bool hasRealloc(MemoryAllocationArgs args) const;
66
67 /*!
68 * \brief Alloue de la mémoire pour \a new_size octets et retourne le pointeur.
69 *
70 * La sémantique est équivalent à malloc():
71 * - \a new_size peut valoir zéro et dans ce cas le pointeur retourné
72 * est soit nul, soit une valeur non spécifiée.
73 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
74 */
75 virtual AllocatedMemoryInfo allocate(MemoryAllocationArgs args, Int64 new_size);
76
77 /*!
78 * \brief Réalloue de la mémoire pour \a new_size octets et retourne le pointeur.
79 *
80 * Le pointeur \a current_ptr doit avoir été alloué via l'appel à
81 * allocate() ou reallocate() de cette instance.
82 *
83 * La sémantique de cette méthode est équivalente à realloc():
84 * - \a current_ptr peut-être nul auquel cas cet appel est équivalent
85 * à allocate().
86 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
87 */
88 virtual AllocatedMemoryInfo reallocate(MemoryAllocationArgs args, AllocatedMemoryInfo current_ptr, Int64 new_size);
89
90 /*!
91 * \brief Libère la mémoire dont l'adresse de base est \a ptr.
92 *
93 * Le pointeur \a ptr doit avoir été alloué via l'appel à
94 * allocate() ou reallocate() de cette instance.
95 *
96 * La sémantique de cette méthode équivalente à free() et donc \a ptr
97 * peut être nul auquel cas aucune opération n'est effectuée.
98 */
99 virtual void deallocate(MemoryAllocationArgs args, AllocatedMemoryInfo ptr);
100
101 /*!
102 * \brief Ajuste la capacité suivant la taille d'élément.
103 *
104 * Cette méthode est utilisée pour éventuellement modifié le nombre
105 * d'éléments alloués suivant leur taille. Cela permet par exemple
106 * pour les allocateurs alignés de garantir que le nombre d'éléments
107 * alloués est un multiple de cet alignement.
108 *
109 */
110 virtual Int64 adjustedCapacity(MemoryAllocationArgs args, Int64 wanted_capacity, Int64 element_size) const;
111
112 /*!
113 * \brief Valeur de l'alignement garanti par l'allocateur.
114 *
115 * Cette méthode permet de s'assurer qu'un allocateur a un alignement suffisant
116 * pour certaines opérations comme la vectorisation par exemple.
117 *
118 * S'il n'y a aucune garantie, retourne 0.
119 */
120 virtual size_t guarantedAlignment(MemoryAllocationArgs args) const;
121
122 /*!
123 * \brief Notifie du changement des arguments spécifiques à l'instance.
124 *
125 * \param ptr zone mémoire allouée
126 * \param old_args ancienne valeur des arguments
127 * \param new_args nouvelle valeur des arguments
128 */
129 virtual void notifyMemoryArgsChanged(MemoryAllocationArgs old_args, MemoryAllocationArgs new_args, AllocatedMemoryInfo ptr);
130
131 /*!
132 * \brief Copie la mémoire entre deux zones.
133 *
134 * L'implémentation par défaut utilise std::memcpy().
135 *
136 * \param args arguments de la zone mémoire
137 * \param destination destination de la copie
138 * \param destination source de la copie
139 */
140 virtual void copyMemory(MemoryAllocationArgs args, AllocatedMemoryInfo destination, AllocatedMemoryInfo source);
141
142 public:
143
144 // Méthodes historiques (avant 2023) sans arguments supplémentaires.
145 // Elles sont laissées pour rester compatible avec l'existant mais seront
146 // supprimées à terme.
147 ARCCORE_DEPRECATED_REASON("Y2023: use hasRealloc(MemoryAllocationArgs) instead")
148 virtual bool hasRealloc() const = 0;
149 ARCCORE_DEPRECATED_REASON("Y2023: use allocate(MemoryAllocationArgs,Int64) instead")
150 virtual void* allocate(size_t new_size) = 0;
151 ARCCORE_DEPRECATED_REASON("Y2023: use reallocate(MemoryAllocationArgs,AllocatedMemoryInfo,Int64) instead")
152 virtual void* reallocate(void* current_ptr, size_t new_size) = 0;
153 ARCCORE_DEPRECATED_REASON("Y2023: use deallocate(MemoryAllocationArgs,AllocatedMemoryInfo) instead")
154 virtual void deallocate(void* ptr) = 0;
155 ARCCORE_DEPRECATED_REASON("Y2023: use adjustedCapacity(MemoryAllocationArgs,Int64,Int64) instead")
156 virtual size_t adjustCapacity(size_t wanted_capacity, size_t element_size) = 0;
157 ARCCORE_DEPRECATED_REASON("Y2023: use guarantedAlignment(MemoryAllocationArgs) instead")
158 virtual size_t guarantedAlignment() = 0;
159};

163/*!

185
186 /*!
187 * \brief Alloue de la mémoire pour \a new_size octets et retourne le pointeur.
188 *
189 * La sémantique est équivalent à malloc():
190 * - \a new_size peut valoir zéro et dans ce cas le pointeur retourné
191 * est soit nul, soit une valeur spécifique
192 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
193 */
194 virtual AllocatedMemoryInfo allocate(MemoryAllocationArgs args, Int64 new_size) = 0;
195
196 /*!
197 * \brief Réalloue de la mémoire pour \a new_size octets et retourne le pointeur.
198 *
199 * Le pointeur \a current_ptr doit avoir été alloué via l'appel à
200 * allocate() ou reallocate() de cette instance.
201 *
202 * La sémantique de cette méthode est équivalente à realloc():
203 * - \a current_ptr peut-être nul auquel cas cet appel est équivalent
204 * à allocate().
205 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
206 */
207 virtual AllocatedMemoryInfo reallocate(MemoryAllocationArgs args, AllocatedMemoryInfo current_ptr, Int64 new_size) = 0;
208
209 /*!
210 * \brief Libère la mémoire dont l'adresse de base est \a ptr.
211 *
212 * Le pointeur \a ptr doit avoir été alloué via l'appel à
213 * allocate() ou reallocate() de cette instance.
214 *
215 * La sémantique de cette méthode équivalente à free() et donc \a ptr
216 * peut être nul auquel cas aucune opération n'est effectuée.
217 */
218 virtual void deallocate(MemoryAllocationArgs args, AllocatedMemoryInfo ptr) = 0;
219
220 /*!
221 * \brief Ajuste la capacité suivant la taille d'élément.
222 *
223 * Cette méthode est utilisée pour éventuellement modifié le nombre
224 * d'éléments alloués suivant leur taille. Cela permet par exemple
225 * pour les allocateurs alignés de garantir que le nombre d'éléments
226 * alloués est un multiple de cet alignement.
227 *
228 */
229 virtual Int64 adjustedCapacity(MemoryAllocationArgs args, Int64 wanted_capacity, Int64 element_size) const = 0;
230
231 /*!
232 * \brief Valeur de l'alignement garanti par l'allocateur.
233 *
234 * Cette méthode permet de s'assurer qu'un allocateur a un alignement suffisant
235 * pour certaines opérations comme la vectorisation par exemple.
236 *
237 * S'il n'y a aucune garantie, retourne 0.
238 */
239 virtual size_t guarantedAlignment(MemoryAllocationArgs args) const = 0;
240
241 private:
242
243 bool hasRealloc() const final;
244 void* allocate(size_t new_size) final;
245 void* reallocate(void* current_ptr, size_t new_size) final;
246 void deallocate(void* ptr) final;
247 size_t adjustCapacity(size_t wanted_capacity, size_t element_size) final;
248 size_t guarantedAlignment() final;
249};

326/*!
327 * \brief Allocateur mémoire avec alignement mémoire spécifique.
328 *
329 * Cette classe s'utilise via les deux méthodes publiques Simd()
330 * et CacheLine() qui retournent respectivement un allocateur avec
331 * un alignement adéquat pour autoriser la vectorisation et un allocateur
332 * aligné sur une ligne de cache.
333 *
334 * Cette classe est obsolète. Utiliser 'AlignedMemoryAllocator2' à la place
335 */
336class ARCCORE_COLLECTIONS_EXPORT AlignedMemoryAllocator
337: public IMemoryAllocator
338{
339 public:
340
341 using IMemoryAllocator::adjustCapacity;
342 using IMemoryAllocator::allocate;
343 using IMemoryAllocator::deallocate;
344 using IMemoryAllocator::guarantedAlignment;
345 using IMemoryAllocator::hasRealloc;
346 using IMemoryAllocator::reallocate;
347
348 private:
349
350 static AlignedMemoryAllocator SimdAllocator;
351 static AlignedMemoryAllocator CacheLineAllocator;
352
353 public:
354
355 // TODO: essayer de trouver les bonnes valeurs en fonction de la cible.
356 // 64 est OK pour toutes les architectures x64 à la fois pour le SIMD
357 // et la ligne de cache.
358
359 // IMPORTANT : Si on change la valeur ici, il faut changer la taille de
360 // l'alignement de ArrayImplBase.
361
362 // TODO Pour l'instant seul un alignement sur 64 est autorisé. Pour
363 // autoriser d'autres valeurs, il faut modifier l'implémentation dans
364 // ArrayImplBase.
365
366 // TODO marquer les méthodes comme 'final'.
367
368 //! Alignement pour les structures utilisant la vectorisation
369 static constexpr Integer simdAlignment() { return 64; }
370 //! Alignement pour une ligne de cache.
371 static constexpr Integer cacheLineAlignment() { return 64; }
372
373 /*!
374 * \brief Allocateur garantissant l'alignement pour utiliser
375 * la vectorisation sur la plateforme cible.
376 *
377 * Il s'agit de l'alignement pour le type plus restrictif et donc il
378 * est possible d'utiliser cet allocateur pour toutes les structures vectorielles.
379 */
380 static AlignedMemoryAllocator* Simd()
381 {
382 return &SimdAllocator;
383 }
384
385 /*!
386 * \brief Allocateur garantissant l'alignement sur une ligne de cache.
387 */
388 static AlignedMemoryAllocator* CacheLine()
389 {
390 return &CacheLineAllocator;
391 }
392
393 protected:
394
395 explicit AlignedMemoryAllocator(Integer alignment)
396 : m_alignment((size_t)alignment)
397 {}
398
399 public:
400
401 bool hasRealloc() const override;
402 void* allocate(size_t new_size) override;
403 void* reallocate(void* current_ptr, size_t new_size) override;
404 void deallocate(void* ptr) override;
405 size_t adjustCapacity(size_t wanted_capacity, size_t element_size) override;
406 size_t guarantedAlignment() override { return m_alignment; }
407
408 private:
409
410 size_t m_alignment;
411};
