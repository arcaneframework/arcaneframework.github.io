27/*!
28 * \brief Interface d'un allocateur pour la mémoire.
29 *
30 * Cette classe définit une interface pour l'allocation mémoire utilisée
31 * par les classes tableaux de Arccore (Array, UniqueArray).
32 *
33 * Une instance de cette classe doit rester valide tant qu'il existe
34 * des tableaux l'utilisant. Comme l'allocateur est transféré lors des copies,
35 * il est préférable que les allocateurs soient des objets statiques qui
36 * dont la durée de vie est celle du programme.
37 *
38 * Les allocateurs n'ont pas d'état modifiables spécifiques et doivent fonctionner en
39 * multi-threading.
40 *
41 * Depuis la version 2.3 de Arccore, les méthodes sans MemoryAllocationArgs sont
42 * obsolètes.
43 */
44class ARCCORE_COLLECTIONS_EXPORT IMemoryAllocator
45{
46 public:
47
48 /*!
49 * \brief Détruit l'allocateur.
50 *
51 * Les objets alloués par l'allocateur doivent tous avoir été désalloués.
52 */
53 virtual ~IMemoryAllocator() = default;
54
55 public:
56
57 /*!
58 * \brief Indique si l'allocateur supporte la sémantique de realloc.
59 *
60 * Les allocateurs par défaut du C (malloc/realloc/free) supportent
61 * évidemment le realloc mais ce n'est pas forcément le cas
62 * des allocateurs spécifiques avec alignement mémoire (comme
63 * par exemple posix_memalign).
64 */
65 virtual bool hasRealloc(MemoryAllocationArgs) const { return false; }
66
67 /*!
68 * \brief Alloue de la mémoire pour \a new_size octets et retourne le pointeur.
69 *
70 * La sémantique est équivalent à malloc():
71 * - \a new_size peut valoir zéro et dans ce cas le pointeur retourné
72 * est soit nul, soit une valeur spécifique
73 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
74 */
75 virtual AllocatedMemoryInfo allocate(MemoryAllocationArgs args, Int64 new_size) = 0;
76
77 /*!
78 * \brief Réalloue de la mémoire pour \a new_size octets et retourne le pointeur.
79 *
80 * Le pointeur \a current_ptr doit avoir été alloué via l'appel à
81 * allocate() ou reallocate() de cette instance.
82 *
83 * La sémantique de cette méthode est équivalente à realloc():
84 * - \a current_ptr peut-être nul auquel cas cet appel est équivalent
85 * à allocate().
86 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
87 */
88 virtual AllocatedMemoryInfo reallocate(MemoryAllocationArgs args, AllocatedMemoryInfo current_ptr, Int64 new_size) = 0;
89
90 /*!
91 * \brief Libère la mémoire dont l'adresse de base est \a ptr.
92 *
93 * Le pointeur \a ptr doit avoir été alloué via l'appel à
94 * allocate() ou reallocate() de cette instance.
95 *
96 * La sémantique de cette méthode équivalente à free() et donc \a ptr
97 * peut être nul auquel cas aucune opération n'est effectuée.
98 */
99 virtual void deallocate(MemoryAllocationArgs args, AllocatedMemoryInfo ptr) = 0;
100
101 /*!
102 * \brief Ajuste la capacité suivant la taille d'élément.
103 *
104 * Cette méthode est utilisée pour éventuellement modifié le nombre
105 * d'éléments alloués suivant leur taille. Cela permet par exemple
106 * pour les allocateurs alignés de garantir que le nombre d'éléments
107 * alloués est un multiple de cet alignement.
108 *
109 */
110 virtual Int64 adjustedCapacity(MemoryAllocationArgs args, Int64 wanted_capacity, Int64 element_size) const = 0;
111
112 /*!
113 * \brief Valeur de l'alignement garanti par l'allocateur.
114 *
115 * Cette méthode permet de s'assurer qu'un allocateur a un alignement suffisant
116 * pour certaines opérations comme la vectorisation par exemple.
117 *
118 * S'il n'y a aucune garantie, retourne 0.
119 */
120 virtual size_t guaranteedAlignment(MemoryAllocationArgs args) const =0;
121
122 /*!
123 * \brief Valeur de l'alignement garanti par l'allocateur.
124 *
125 * \sa guaranteedAlignment()
126 */
127 ARCCORE_DEPRECATED_REASON("Y2024: Use guaranteedAlignment() instead")
128 virtual size_t guarantedAlignment(MemoryAllocationArgs args) const;
129
130 /*!
131 * \brief Notifie du changement des arguments spécifiques à l'instance.
132 *
133 * \param ptr zone mémoire allouée
134 * \param old_args ancienne valeur des arguments
135 * \param new_args nouvelle valeur des arguments
136 */
137 virtual void notifyMemoryArgsChanged(MemoryAllocationArgs old_args, MemoryAllocationArgs new_args, AllocatedMemoryInfo ptr);
138
139 /*!
140 * \brief Copie la mémoire entre deux zones.
141 *
142 * L'implémentation par défaut utilise std::memcpy().
143 *
144 * \param args arguments de la zone mémoire
145 * \param destination destination de la copie
146 * \param destination source de la copie
147 */
148 virtual void copyMemory(MemoryAllocationArgs args, AllocatedMemoryInfo destination, AllocatedMemoryInfo source);
149
150 //! Ressource mémoire fournie par l'allocateur
151 virtual eMemoryResource memoryResource() const { return eMemoryResource::Unknown; }
152};

200/*!
201 * \brief Allocateur mémoire avec alignement mémoire spécifique.
202 *
203 * Cette classe s'utilise via les deux méthodes publiques Simd()
204 * et CacheLine() qui retournent respectivement un allocateur avec
205 * un alignement adéquat pour autoriser la vectorisation et un allocateur
206 * aligné sur une ligne de cache.
207 */
208class ARCCORE_COLLECTIONS_EXPORT AlignedMemoryAllocator
209: public IMemoryAllocator
210{
211 private:
212
213 static AlignedMemoryAllocator SimdAllocator;
214 static AlignedMemoryAllocator CacheLineAllocator;
215
216 public:
217
218 // TODO: essayer de trouver les bonnes valeurs en fonction de la cible.
219 // 64 est OK pour toutes les architectures x64 à la fois pour le SIMD
220 // et la ligne de cache.
221
222 // IMPORTANT : Si on change la valeur ici, il faut changer la taille de
223 // l'alignement de ArrayImplBase.
224
225 // TODO Pour l'instant seul un alignement sur 64 est autorisé. Pour
226 // autoriser d'autres valeurs, il faut modifier l'implémentation dans
227 // ArrayImplBase.
228
229 // TODO marquer les méthodes comme 'final'.
230
231 //! Alignement pour les structures utilisant la vectorisation
232 static constexpr Integer simdAlignment() { return 64; }
233 //! Alignement pour une ligne de cache.
234 static constexpr Integer cacheLineAlignment() { return 64; }
235
236 /*!
237 * \brief Allocateur garantissant l'alignement pour utiliser
238 * la vectorisation sur la plateforme cible.
239 *
240 * Il s'agit de l'alignement pour le type plus restrictif et donc il
241 * est possible d'utiliser cet allocateur pour toutes les structures vectorielles.
242 */
243 static AlignedMemoryAllocator* Simd()
244 {
245 return &SimdAllocator;
246 }
247
248 /*!
249 * \brief Allocateur garantissant l'alignement sur une ligne de cache.
250 */
251 static AlignedMemoryAllocator* CacheLine()
252 {
253 return &CacheLineAllocator;
254 }
255
256 protected:
257
258 explicit AlignedMemoryAllocator(Int32 alignment)
259 : m_alignment(static_cast<size_t>(alignment))
260 {}
261
262 public:
263
264 bool hasRealloc(MemoryAllocationArgs) const override { return false; }
265 AllocatedMemoryInfo allocate(MemoryAllocationArgs args, Int64 new_size) override;
266 AllocatedMemoryInfo reallocate(MemoryAllocationArgs args, AllocatedMemoryInfo current_ptr, Int64 new_size) override;
267 void deallocate(MemoryAllocationArgs args, AllocatedMemoryInfo ptr) override;
268 Int64 adjustedCapacity(MemoryAllocationArgs args, Int64 wanted_capacity, Int64 element_size) const override;
269 size_t guaranteedAlignment(MemoryAllocationArgs) const override { return m_alignment; }
270 eMemoryResource memoryResource() const override { return eMemoryResource::Host; }
271
272 private:
273
274 size_t m_alignment;
275};
