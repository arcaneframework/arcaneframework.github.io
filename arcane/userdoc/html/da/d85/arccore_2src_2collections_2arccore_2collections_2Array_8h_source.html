153/*!
154 * \brief Classe de base interne pour les tableaux.
155 *
156 * Cette classe gère uniquement les meta-données pour les tableaux comme
157 * le nombre d'éléments ou la capacité.
158 *
159 * \a m_md est un pointeur contenant les meta-donné du tableau. Si le
160 * tableau est partagé (SharedArray, SharedArray2), alors ce pointeur
161 * est alloué dynamiquement et dans ce cas _isUseOwnMetaData() doit
162 * retourner \a false. Si le tableau n'est pas partagé (UniqueArray ou
163 * UniqueArray2), alors les meta-données sont conservées directement
164 * dans l'instance du tableau pour éviter des allocations inutiles
165 * et \a m_md pointe alors vers \a m_meta_data. Dans tous les cas, il
166 * ne faut pas utiliser \a m_meta_data directement, mais toujours passer
167 * par \a m_md.
168 */

174 {
175 m_md = &m_meta_data;
176 }
177 virtual ~AbstractArrayBase() = default;

179 public:

184 }

186 {
187 return m_md->allocation_options;

190 * \brief Positionne le nom du tableau pour les informations de debug.

194 void setDebugName(const String& name);
195 //! Nom de debug (nul si aucun nom spécifié)
196 String debugName() const;

208 /*!
209 * \brief Indique si \a m_md fait référence à \a m_meta_data.
210 *
211 * C'est le cas pour les UniqueArray et UniqueArray2 mais

222 {

224 std::swap(m_meta_data,rhs.m_meta_data);
225 _checkSetUseOwnMetaData();

229 void _copyMetaData(const AbstractArrayBase& rhs)
230 {
231 // Déplace les meta-données
232 // Attention si on utilise m_meta_data alors il
233 // faut positionner m_md pour qu'il pointe vers notre propre m_meta_data.
234 m_meta_data = rhs.m_meta_data;
235 m_md = rhs.m_md;
236 _checkSetUseOwnMetaData();
237 }
238
239 void _allocateMetaData()
240 {
241#ifdef ARCCORE_CHECK
242 if (m_md->is_not_null)
243 ArrayMetaData::throwNullExpected();
244#endif
245 if (_isUseOwnMetaData()){
246 m_meta_data = ArrayMetaData();
247 m_md = &m_meta_data;
248 }
249 else{
250 m_md = new ArrayMetaData();
251 m_md->is_allocated_by_new = true;
252 }
253 m_md->is_not_null = true;
254 }
255
256 void _deallocateMetaData(ArrayMetaData* md)
257 {
258 if (md->is_allocated_by_new)
259 delete md;
260 else
261 *md = ArrayMetaData();
262 }
263
264 void _checkValidSharedArray()
265 {
266#ifdef ARCCORE_CHECK
267 if (m_md->is_not_null && !m_md->is_allocated_by_new)
268 ArrayMetaData::throwInvalidMetaDataForSharedArray();
269#endif
270 }
271
272 private:
273
274 void _checkSetUseOwnMetaData()
275 {
276 if (!m_md->is_allocated_by_new)
277 m_md = &m_meta_data;
278 }
279};

557 /*!
558 * \brief Réalloue le tableau pour une nouvelle capacité égale à \a new_capacity.
559 *
560 * Si la nouvelle capacité est inférieure à l'ancienne, rien ne se passe.
561 */
562 template<typename PodType>
563 void _internalRealloc(Int64 new_capacity, bool compute_capacity, PodType pod_type, RunQueue* queue = nullptr)
564 {
565 if (_isSharedNull()){
566 if (new_capacity!=0)
567 _internalAllocate(new_capacity,queue);
568 return;
569 }
570
571 Int64 acapacity = new_capacity;
572 if (compute_capacity){
573 acapacity = m_md->capacity;
574 //std::cout << " REALLOC: want=" << wanted_size << " current_capacity=" << capacity << '\n';
575 while (new_capacity>acapacity)
576 acapacity = (acapacity==0) ? 4 : (acapacity + 1 + acapacity / 2);
577 //std::cout << " REALLOC: want=" << wanted_size << " new_capacity=" << capacity << '\n';
578 }
579 // Si la nouvelle capacité est inférieure à la courante,ne fait rien.
580 if (acapacity <= m_md->capacity)
581 return;
582 _internalReallocate(acapacity, pod_type, queue);
583 }

602 //! Réallocation pour un type complexe (non POD)
603 void _internalReallocate(Int64 new_capacity, FalseType, RunQueue* queue)
604 {
605 T* old_ptr = m_ptr;
606 ArrayMetaData* old_md = m_md;
607 AllocatedMemoryInfo old_mem_info = _currentMemoryInfo();
608 Int64 old_size = m_md->size;
609 _directAllocate(new_capacity, queue);
610 if (m_ptr!=old_ptr){
611 for (Int64 i = 0; i < old_size; ++i) {
612 new (m_ptr + i) T(old_ptr[i]);
613 old_ptr[i].~T();
614 }
615 m_md->nb_ref = old_md->nb_ref;
616 m_md->_deallocate(old_mem_info, queue);
617 _updateReferences();
618 }

877 // Copie les éléments déjà existant
