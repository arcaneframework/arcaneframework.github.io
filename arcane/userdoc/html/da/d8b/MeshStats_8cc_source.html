134template<typename T> void MeshStats::
135_dumpStats()
136{
137 Trace::Setter setter(traceMng(),"Mesh");
138
139 const Integer nb_type = ItemTypeMng::nbBasicItemType();
140 const Integer nb_kind = NB_ITEM_KIND;
141
142 const char* name[3] = { "(All)", "(Own)", "(Ghost)" };
143
144 // Pour ne s'occuper de Own et Ghost que en parallèle
145 const Integer nstat = (m_parallel_mng->isParallel()) ? 3 : 1;
146
147 // On regroupe toutes les données dans la même communication
148 const Integer data_by_stat = nb_type+nb_kind;
149 const Integer all_data_count = nstat*data_by_stat;
150 Int64UniqueArray nb_data(all_data_count,0);
151
152 // On remplit maintenant tout le tableau par bloc
153 for( Integer istat=0; istat<nstat; ++istat){
154 const Integer first_type = istat * data_by_stat;
155 const Integer first_kind = first_type + nb_type;
156 // Nombre d'éléments de chaque type sur ce sous-domaine.
157 Int64ArrayView nb_local_type = nb_data.view().subView(first_type,nb_type);
158 // Nombre d'éléments de chaque genre sur de sous-domaine
159 Int64ArrayView nb_local_kind = nb_data.view().subView(first_kind,nb_kind);
160 _computeElementsOnGroup<T>(nb_local_type,nb_local_kind,istat);
161 }
162
163 // Tableau résultat de synthèse
164 Int64UniqueArray nb_data_min(all_data_count,0);
165 Int64UniqueArray nb_data_max(all_data_count,0);
166 Int64UniqueArray nb_data_sum(all_data_count,0);
167 Int32UniqueArray min_data_rank(all_data_count,0);
168 Int32UniqueArray max_data_rank(all_data_count,0);
169 m_parallel_mng->computeMinMaxSum(nb_data,
170 nb_data_min,
171 nb_data_max,
172 nb_data_sum,
173 min_data_rank,
174 max_data_rank);
175
176 // On vérifie s'il y a qqchose à afficher
177 {
178 bool is_empty = true;
179 for( Integer istat=0; istat<nstat; ++istat){
180 const Integer first_type = istat * data_by_stat;
181 Int64ConstArrayView nb_global_type = nb_data_sum.subConstView(first_type,nb_type);
182 for(Integer i = 0; i < nb_global_type.size(); ++i)
183 if(nb_global_type[i] != 0) {
184 is_empty = false;
185 break;
186 }
187 if (!is_empty)
188 break;
189 }
190 if (is_empty)
191 return;
192 }
193
194 ItemTypeMng* type_mng = m_mesh->itemTypeMng();
195
196 // On produit maintenant l'affichage des stats
197 const Integer nb_rank = m_parallel_mng->commSize();
198 for( Integer istat=0; istat<nstat; ++istat){
199 // Construction des vues de travail décompactées
200 // (on reprend les noms des variables de la précédente version)
201 const Integer first_type = istat * data_by_stat;
202 const Integer first_kind = first_type + nb_type;
203
204 Int64ConstArrayView nb_local_type = nb_data.subConstView(first_type,nb_type);
205 Int64ConstArrayView nb_local_min_type = nb_data_min.subConstView(first_type,nb_type);
206 Int64ConstArrayView nb_local_max_type = nb_data_max.subConstView(first_type,nb_type);
207 Int64ConstArrayView nb_global_type = nb_data_sum.subConstView(first_type,nb_type);
208 Int32ConstArrayView min_rank_type = min_data_rank.subConstView(first_type,nb_type);
209 Int32ConstArrayView max_rank_type = max_data_rank.subConstView(first_type,nb_type);
210
211 Int64ConstArrayView nb_local_kind = nb_data.subConstView(first_kind,nb_kind);
212 Int64ConstArrayView nb_local_min_kind = nb_data_min.subConstView(first_kind,nb_kind);
213 Int64ConstArrayView nb_local_max_kind = nb_data_max.subConstView(first_kind,nb_kind);
214 Int64ConstArrayView nb_global_kind = nb_data_sum.subConstView(first_kind,nb_kind);
215 Int32ConstArrayView min_rank_kind = min_data_rank.subConstView(first_kind,nb_kind);
216 Int32ConstArrayView max_rank_kind = max_data_rank.subConstView(first_kind,nb_kind);
217
218 info() << " -------------------------------------------";
219 _statLabel<T>(name[istat]);
220
221 info() << Trace::Width(18) << "Item"
222 << Trace::Width(10) << "Myself"
223 << Trace::Width(10) << "Min"
224 << Trace::Width(8) << "Rank"
225 << Trace::Width(10) << "Max"
226 << Trace::Width(8) << "Rank"
227 << Trace::Width(10) << "Average"
228 << Trace::Width(10) << "Bal"
229 << Trace::Width(12) << "Total";
230
231 info() << " ";
232 for( Integer i=0, s=nb_kind; i<s; ++i ){
233 eItemKind kt = static_cast<eItemKind>(i);
234 _printInfo(itemKindName(kt),nb_local_kind[i],
235 nb_local_min_kind[i],min_rank_kind[i],
236 nb_local_max_kind[i],max_rank_kind[i],
237 nb_global_kind[i],nb_rank);
238 // if (nb_global_kind[i]!=0) {
239 // pinfo() << "MeshStats ("<< m_parallel_mng->commRank() <<") : " << name[istat]
240 // << " kind=" << itemKindName(kt)
241 // << " n=" << nb_local_kind[i];
242 // plog() << "MeshStats ("<< m_parallel_mng->commRank() <<") : " << name[istat]
243 // << " kind=" << itemKindName(kt)
244 // << " n=" << nb_local_kind[i];
245 // }
246 }
247 info() << " ";
248 for( Integer i=0, s=nb_type; i<s; ++i ) {
249 _printInfo(type_mng->typeName(i),nb_local_type[i],
250 nb_local_min_type[i],min_rank_type[i],
251 nb_local_max_type[i],max_rank_type[i],
252 nb_global_type[i],nb_rank);
253 // if (nb_global_type[i]!=0)
254 // pinfo() << "MeshStats ("<< m_parallel_mng->commRank() <<") : " << name[istat]
255 // << " type=" << Item::typeName(i)
256 // << " n=" << nb_local_type[i];
257
258 }
259 info() << " ";
260 info() << " -------------------------------------------";
261 }
262 // _computeNeighboorsComm();
263
264 _dumpCommunicatingRanks();
265 _dumpLegacyConnectivityMemoryUsage();
266 _dumpIncrementalConnectivityMemoryUsage();
267}

272void MeshStats::
273_dumpLegacyConnectivityMemoryUsage()
274{
275 // Cette liste de nom est issue de mesh/ItemFamily.cc
276 UniqueArray<String> var_names = { "FamilyItemsData", "FamilyItemsShared" };
277 IVariableMng* vm = m_mesh->variableMng();
278 Real total_memory = 0.0;
279 for( IItemFamily* family : m_mesh->itemFamilies() ){
280 String name = family->name();
281 Real family_memory = 0.0;
282 info(4) << "Family name=" << family->name();
283 for( String s : var_names ){
284 IVariable* v = vm->findMeshVariable(m_mesh,name+s);
285 if (v){
286 Real v_memory = v->allocatedMemory();
287 family_memory += v_memory;
288 info(4) << "Allocated Memory n=" << s << " v=" << v_memory;
289 }
290 }
291 info(4) << "Memory for family name=" << name << " mem=" << family_memory;
292 total_memory += family_memory;
293 }
294 info() << "Total memory for legacy connectivities mem=" << total_memory;
295}

300void MeshStats::
301_dumpIncrementalConnectivityMemoryUsage()
302{
303 IVariableMng* vm = m_mesh->variableMng();
304 Real total_memory = 0.0;
305 VariableCollection used_variables = vm->usedVariables();
306 const String tag_name = "ArcaneConnectivity";
307 for( VariableCollection::Enumerator iv(used_variables); ++iv; ){
308 IVariable* v = *iv;
309 if (!v->hasTag(tag_name))
310 continue;
311 if (v->meshHandle().meshOrNull()==m_mesh){
312 Real v_memory = v->allocatedMemory();
313 info(4) << "Allocated Memory n=" << v->name() << " v=" << v_memory;
314 total_memory += v_memory;
315 }
316 }
317 info() << "Total memory for incremental connectivities mem=" << total_memory;
318}

322
323void MeshStats::
324_printInfo(const String& name,Int64 nb_local,
325 Int64 nb_local_min,Integer min_rank,
326 Int64 nb_local_max,Integer max_rank,
327 Int64 nb_global,Integer nb_rank)
328{
329 if (nb_global==0 && nb_local==0)
330 return;
331
332 String tr_name = m_dictionary->find(String(name));
333 if (tr_name.empty()) tr_name = name;
334
335 Int64 average = nb_global;
336 if (nb_rank!=0)
337 average /= nb_rank;
338 Real bal1 = (Real)(nb_local_max - average);
339 if (average!=0)
340 bal1 = bal1 / (Real)average;
341 Int64 bal = (Int64)(bal1*1000);
342 info() << Trace::Width(18) << tr_name
343 << Trace::Width(10) << nb_local
344 << Trace::Width(10) << nb_local_min
345 << Trace::Width(8) << min_rank
346 << Trace::Width(10) << nb_local_max
347 << Trace::Width(8) << max_rank
348 << Trace::Width(10) << average
349 << Trace::Width(10) << bal
350 << Trace::Width(12) << nb_global;
351}

356void MeshStats::
357_computeElementsOnGroup(Int64ArrayView nb_type,Int64ArrayView nb_kind,
358 ItemGroup group,Integer istat)
359{
360 Integer ik = (Integer)group.itemKind();
361 ENUMERATE_ITEM(i,group){
362 Item item = *i;
363 int type = item.type();
364 if (istat==0 || (istat==1 && item.isOwn()) || (istat==2 && !item.isOwn())){
365 ++nb_kind[ik];
366 ++nb_type[type];
367 }
368 }

371void MeshStats::
372_computeNeighboorsComm()
373{
374 Integer nb_proc = m_parallel_mng->commSize();
375 Int64UniqueArray vol_comm_out(nb_proc);
376 StringBuilder out = "";
377 vol_comm_out.fill(0);
378
379 ENUMERATE_ITEM(i,m_mesh->allCells()){
380 Item item = *i;
381 if (item.isOwn())
382 continue;
383 vol_comm_out[item.owner()]++;
384 }
385
386 for (Integer i =0 ; i < nb_proc ; ++i) {
387 out += " ";
388 out += vol_comm_out[i];
389 }
390 pinfo() << "Comm Proc " << m_parallel_mng->commRank() << " : " << out;
391}

396void MeshStats::
397_dumpCommunicatingRanks()
398{
399 // Affiche les statistiques sur le nombre de sous-domaines avec
400 // lequels on communique. On prend les voisins sur la liste des mailles.
401 IItemFamily* cell_family = m_mesh->cellFamily();
402 IVariableSynchronizer* sync_info = cell_family->allItemsSynchronizer();
403 Int64 nb_comm_local = sync_info->communicatingRanks().size();
404 Int32 nb_comm_min_rank = 0;
405 Int32 nb_comm_max_rank = 0;
406 Int64 nb_comm_max = 0;
407 Int64 nb_comm_min = 0;
408 Int64 nb_comm_sum = 0;
409 IParallelMng* pm = m_parallel_mng;
410 pm->computeMinMaxSum(nb_comm_local,nb_comm_min,nb_comm_max,nb_comm_sum,
411 nb_comm_min_rank,nb_comm_max_rank);
412 Int64 average = nb_comm_sum / pm->commSize();
413 info() << String::format("CommunicatingSubDomains: local={0}, min={1}, max={2}"
414 " average={3} min_rank={4} max_rank={5}",
415 nb_comm_local,nb_comm_min,nb_comm_max,average,
416 nb_comm_min_rank,nb_comm_max_rank);
417}
