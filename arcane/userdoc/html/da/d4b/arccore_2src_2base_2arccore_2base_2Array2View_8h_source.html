36 template<class DataType>
37 class Array2View
38 {
39  friend class SmallSpan2<DataType>;
40  friend class SmallSpan2<const DataType>;
41  friend class Span2<DataType>;
42  friend class Span2<const DataType>;
43  public:
44  friend class ConstArray2View<DataType>;
45  public:
47  constexpr Array2View(DataType* ptr,Integer dim1_size,Integer dim2_size)
48  : m_ptr(ptr), m_dim1_size(dim1_size), m_dim2_size(dim2_size) {}
50  constexpr Array2View() : m_ptr(nullptr), m_dim1_size(0), m_dim2_size(0) {}
51  public:
53  constexpr Integer dim1Size() const { return m_dim1_size; }
55  constexpr Integer dim2Size() const { return m_dim2_size; }
57  constexpr Integer totalNbElement() const { return m_dim1_size*m_dim2_size; }
58  public:
59  constexpr ArrayView<DataType> operator[](Integer i)
60  {
61  ARCCORE_CHECK_AT(i,m_dim1_size);
62  return ArrayView<DataType>(m_dim2_size,m_ptr + (m_dim2_size*i));
63  }
64  constexpr ConstArrayView<DataType> operator[](Integer i) const
65  {
66  ARCCORE_CHECK_AT(i,m_dim1_size);
67  return ConstArrayView<DataType>(m_dim2_size,m_ptr + (m_dim2_size*i));
68  }
70  constexpr DataType item(Integer i,Integer j) const
71  {
72  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
73  return m_ptr[(m_dim2_size*i) + j];
74  }
76  constexpr DataType setItem(Integer i,Integer j,const DataType& value)
77  {
78  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
79  m_ptr[(m_dim2_size*i) + j] = value;
80  }
82  constexpr const DataType operator()(Integer i,Integer j) const
83  {
84  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
85  return m_ptr[(m_dim2_size*i) + j];
86  }
88  constexpr DataType& operator()(Integer i,Integer j)
89  {
90  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
91  return m_ptr[(m_dim2_size*i) + j];
92  }
93 #ifdef ARCCORE_HAS_MULTI_SUBSCRIPT
95  constexpr const DataType operator[](Integer i,Integer j) const
96  {
97  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
98  return m_ptr[(m_dim2_size*i) + j];
99  }
101  constexpr DataType& operator[](Integer i,Integer j)
102  {
103  ARCCORE_CHECK_AT2(i,j,m_dim1_size,m_dim2_size);
104  return m_ptr[(m_dim2_size*i) + j];
105  }
106 #endif
107 
108  public:
112  constexpr inline DataType* unguardedBasePointer()
113  { return m_ptr; }
117  constexpr inline DataType* data() { return m_ptr; }
118  private:
119  DataType* m_ptr;
120  Integer m_dim1_size;
121  Integer m_dim2_size;
122 };
