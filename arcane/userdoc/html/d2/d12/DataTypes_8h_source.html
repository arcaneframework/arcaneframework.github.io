109/*!
110 * \brief Type de politique d'initialisation possible pour une donnée.
111 *
112 * Par défaut, pour des raisons historiques, il s'agit de DIP_Legacy.
113 *
114 * La politique d'initialisation est prise en compte pour
115 * l'initialisation des variables Arcane. Cette valeur peut être
116 * positionnée/récupérée via les fonctions
117 * setGlobalDataInitialisationPolicy() et getGlobalDataInitialisationPolicy().
118 *
119 */
120enum eDataInitialisationPolicy
121{
122 //! Pas d'initialisation forcée
123 DIP_None =0,
124 /*!
125 * \brief Initialisation avec le constructeur par défaut.
126 *
127 * Pour les entiers, il s'agit de la valeurs 0. Pour les réels, il
128 * s'agit de la valeur 0.0.
129 */
130 DIP_InitWithDefault =1,
131 /*!
132 * \brief Initialisation avec des NaN (Not a Number)
133 *
134 * Ce mode permet d'initialiser les données de type Real et dérivés (Real2, Real3, ...)
135 * avec des valeurs NaN qui déclenchent une exception lorsqu'elles
136 * sont utilisées.
137 */
138 DIP_InitWithNan = 2,
139 /*!
140 * \brief Initialisation en mode historique.
141 *
142 * Ce mode est conservé pour compatibilité avec les versions d'Arcane inférieurs
143 * à la version 2.0 où c'était le mode par défaut.
144 * Dans ce mode, les variables sur les entités du maillage étaient
145 * toujours initialisées avec le constructeur par défaut lors de
146 * leur première allocation, quelle que soit la valeur de
147 * getGlobalDataInitialisationPolicy(). La politique d'initialisation n'était
148 * prise en compte que lors d'un changement du nomnbre d'éléments (resize())
149 * où pour les variables qui n'étaient pas des variables du maillage.
150 */
151 DIP_Legacy = 3,
152 /*!
153 * \brief Initialisation avec des NaN pour à la création et le constructeur
154 * par défaut ensuite.
155 *
156 * Ce mode est identique à DIP_InitWithNan pour la création des variables
157 * et à DIP_InitWithDefault lorsqu'on la taille de la variable évolue
158 * (en géneral via un appel à IVariable::resize() ou IVariable::resizeFromGroup()).
159 */
160 DIP_InitInitialWithNanResizeWithDefault = 4
161};
