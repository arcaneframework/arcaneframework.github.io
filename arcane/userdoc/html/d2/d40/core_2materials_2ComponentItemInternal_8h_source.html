39/*!
40 * \internal
41 * \brief Index d'une entité constituant dans la liste des entités constituants.
42 *
43 * L'index est propre à chaque type d'entité consituant (AllEnvCell, EnvCell, MatCell).
44 * La liste est gérée par ComponentIemtInternalData.
45 */
46class ARCANE_CORE_EXPORT ConstituentItemIndex
47{
48 public:
49
50 ConstituentItemIndex() = default;
51 explicit ARCCORE_HOST_DEVICE constexpr ConstituentItemIndex(Int32 id)
52 : m_id(id)
53 {}
54 ARCCORE_HOST_DEVICE constexpr Int32 localId() const { return m_id; }
55 ARCCORE_HOST_DEVICE friend constexpr bool operator==(ConstituentItemIndex a,
56 ConstituentItemIndex b)
57 {
58 return a.m_id == b.m_id;
59 }
60 ARCCORE_HOST_DEVICE friend constexpr bool operator!=(ConstituentItemIndex a,
61 ConstituentItemIndex b)
62 {
63 return a.m_id != b.m_id;
64 }
65 ARCANE_CORE_EXPORT friend std::ostream&
66 operator<<(std::ostream& o, const ConstituentItemIndex& id);
67
68 ARCCORE_HOST_DEVICE constexpr bool isNull() const { return m_id == (-1); }
69
70 private:
71
72 Int32 m_id = -1;
73};

113/*!
114 * \internal
115 * \brief Informations partagées sur les 'ComponentItem'.
116 *
117 * Il y a 3 instances de cette classe : une pour les AllEnvCell, une pour les
118 * EnvCell et une pour les MatCell. Ces instances sont gérées par la classe
119 * ComponentItemInternalData. Il est possible de conserver un pointeur sur
120 * les intances de cette classe car ils sont valides durant toute la vie
121 * d'un MeshMaterialMng.
122 */
123class ARCANE_CORE_EXPORT ComponentItemSharedInfo
124: private ComponentItemSharedInfoStorageView
125{
126 friend class ComponentItemInternalData;
127 friend class CellComponentCellEnumerator;
128 friend class ConstituentItemLocalIdList;
129 friend class ConstituentItemLocalIdListView;
130 friend matimpl::ConstituentItemBase;
131 friend class ComponentCell;
132 friend class CellToAllEnvCellConverter;
133 friend class AllEnvCellVectorView;
134
135 static const int MAT_INDEX_OFFSET = 10;
136
137 private:
138
139 //! Pour l'entité nulle
140 static ComponentItemSharedInfo null_shared_info;
141 static ComponentItemSharedInfo* null_shared_info_pointer;
142 static ComponentItemSharedInfo* _nullInstance() { return null_shared_info_pointer; }
143
144 private:
145
146 inline constexpr matimpl::ConstituentItemBase _item(ConstituentItemIndex id);
147 inline ARCCORE_HOST_DEVICE ConstituentItemIndex _firstSubConstituentLocalId(ConstituentItemIndex id) const
148 {
149 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
150 return m_first_sub_constituent_item_id_data[id.localId()];
151 }
152 inline ARCCORE_HOST_DEVICE void
153 _setFirstSubConstituentLocalId(ConstituentItemIndex id, ConstituentItemIndex first_id)
154 {
155 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
156 m_first_sub_constituent_item_id_data[id.localId()] = first_id;
157 }
158 inline ARCCORE_HOST_DEVICE Int16 _nbSubConstituent(ConstituentItemIndex id) const
159 {
160 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
161 return m_nb_sub_constituent_item_data[id.localId()];
162 }
163 ARCCORE_HOST_DEVICE inline void _setNbSubConstituent(ConstituentItemIndex id, Int16 n)
164 {
165 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
166 m_nb_sub_constituent_item_data[id.localId()] = n;
167 }
168 inline ARCCORE_HOST_DEVICE Int16 _componentId(ConstituentItemIndex id) const
169 {
170 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
171 return m_component_id_data[id.localId()];
172 }
173 ARCCORE_HOST_DEVICE inline void _setComponentId(ConstituentItemIndex id, Int16 component_id)
174 {
175 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
176 m_component_id_data[id.localId()] = component_id;
177 }
178 IMeshComponent* _component(ConstituentItemIndex id) const
179 {
180 return m_components[_componentId(id)];
181 }
182 impl::ItemBase _globalItemBase(ConstituentItemIndex id) const
183 {
184 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
185 return impl::ItemBase(m_global_item_local_id_data[id.localId()], m_item_shared_info);
186 }
187 ARCCORE_HOST_DEVICE void _setGlobalItem(ConstituentItemIndex id, ItemLocalId global_item_lid)
188 {
189 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
190 m_global_item_local_id_data[id.localId()] = global_item_lid.localId();
191 }
192 matimpl::ConstituentItemBase _superItemBase(ConstituentItemIndex id) const;
193
194 ARCCORE_HOST_DEVICE void _setSuperItem(ConstituentItemIndex id, ConstituentItemIndex super_id)
195 {
196 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
197 m_super_component_item_local_id_data[id.localId()] = super_id;
198 }
199 inline ARCCORE_HOST_DEVICE matimpl::ConstituentItemBase _subItemBase(ConstituentItemIndex id,Int32 sub_index) const;
200
201 ARCCORE_HOST_DEVICE MatVarIndex _varIndex(ConstituentItemIndex id)
202 {
203 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
204 return m_var_index_data[id.localId()];
205 }
206 ARCCORE_HOST_DEVICE void _setVarIndex(ConstituentItemIndex id, MatVarIndex mv_index)
207 {
208 ARCCORE_CHECK_RANGE(id.localId(), -1, m_storage_size);
209 m_var_index_data[id.localId()] = mv_index;
210 }
211
212 //! Numéro unique de l'entité component
213 Int64 _componentUniqueId(ConstituentItemIndex id) const
214 {
215 // TODO: Vérifier que arrayIndex() ne dépasse pas (1<<MAT_INDEX_OFFSET)
216 impl::ItemBase item_base(_globalItemBase(id));
217 return (Int64)m_var_index_data[id.localId()].arrayIndex() + ((Int64)item_base.uniqueId() << MAT_INDEX_OFFSET);
218 }
219
220 ARCCORE_HOST_DEVICE void _reset(ConstituentItemIndex id)
221 {
222 Int32 local_id = id.localId();
223 ARCCORE_CHECK_RANGE(local_id, -1, m_storage_size);
224
225 m_var_index_data[local_id].reset();
226 m_first_sub_constituent_item_id_data[local_id] = {};
227 m_nb_sub_constituent_item_data[local_id] = 0;
228 m_component_id_data[local_id] = -1;
229 m_global_item_local_id_data[local_id] = NULL_ITEM_LOCAL_ID;
230 m_super_component_item_local_id_data[local_id] = {};
231 }
232
233 private:
234
235 // NOTE : Cette classe est partagée avec le wrapper C#
236 // Toute modification de la structure interne doit être reportée
237 // dans la structure C# correspondante
238 ItemSharedInfo* m_item_shared_info = ItemSharedInfo::nullInstance();
239 Int16 m_level = (-1);
240 ConstArrayView<IMeshComponent*> m_components;
241 ComponentItemSharedInfo* m_super_component_item_shared_info = null_shared_info_pointer;
242 ComponentItemSharedInfo* m_sub_component_item_shared_info = null_shared_info_pointer;
243};

255/*!
256 * \internal
257 * \brief Informations générique sur une entité d'un constituant.
258 *
259 * Cette classe est le pendant de ItemInternal pour la gestion des matériaux
260 * et des milieux. Elle ne doit en principe pas être utilisée directement, sauf
261 * par les classes de Arcane. Il vaut mieux utiliser les
262 * classes ComponentCell, MatCell, EnvCell ou AllEnvCell.

264class ARCANE_CORE_EXPORT ConstituentItemBase

276 friend Arcane::Materials::ComponentItemInternalData;
277
278 private:
279
280 ARCCORE_HOST_DEVICE constexpr ConstituentItemBase(ComponentItemSharedInfo* shared_info, ConstituentItemIndex id)
281 : m_constituent_item_index(id)
282 , m_shared_info(shared_info)
283 {
284 }
285
286 public:
287
288 //! Indexeur dans les variables matériaux
289 ARCCORE_HOST_DEVICE MatVarIndex variableIndex() const
290 {
291 return m_shared_info->_varIndex(m_constituent_item_index);

307 {

309 }
310
311 /*!
312 * \brief Composant associé.
313 *
314 * Cet appel n'est valide que pour les mailles matériaux ou milieux. Si on souhaite
315 * un appel valide pour toutes les 'ComponentItem', il faut utiliser componentId().

319 return m_shared_info->_component(m_constituent_item_index);
320 }
321
322 //! Nombre de sous-composants.

324 {
325 return m_shared_info->_nbSubConstituent(m_constituent_item_index);
326 }
327
328 //! Entité globale correspondante.
329 inline impl::ItemBase globalItemBase() const
330 {
331 return m_shared_info->_globalItemBase(m_constituent_item_index);
332 }
333
334 inline ARCCORE_HOST_DEVICE constexpr Int32 level() const
335 {
336 return m_shared_info->m_level;
337 }
338
339 //! Numéro unique de l'entité component
340 inline Int64 componentUniqueId() const
341 {
342 return m_shared_info->_componentUniqueId(m_constituent_item_index);
343 }
344
345 public:
346
347 ARCCORE_HOST_DEVICE constexpr friend bool
348 operator==(const ConstituentItemBase& a, const ConstituentItemBase& b)
349 {
350 return ((a.m_constituent_item_index == b.m_constituent_item_index) && (a.m_shared_info == b.m_shared_info));
351 }
352 ARCCORE_HOST_DEVICE constexpr friend bool
353 operator!=(const ConstituentItemBase& a, const ConstituentItemBase& b)
354 {
355 return !(a == b);
356 }
357
358 private:
359
360 //! Positionne l'indexeur dans les variables matériaux.
361 ARCCORE_HOST_DEVICE inline void _setVariableIndex(MatVarIndex index)

363 m_shared_info->_setVarIndex(m_constituent_item_index, index);

365

370 {

372 m_shared_info->_setGlobalItem(m_constituent_item_index, ii);

374
375 ARCCORE_HOST_DEVICE inline void _setGlobalItem(ItemLocalId ii)
376 {
377 m_shared_info->_setGlobalItem(m_constituent_item_index, ii);
378 }
379
380 //! Première entité sous-composant.

382 {
383 return m_shared_info->_firstSubConstituentLocalId(m_constituent_item_index);
384 }
385
386 inline ARCCORE_HOST_DEVICE matimpl::ConstituentItemBase _subItemBase(Int32 i) const;
387
388 //! Positionne le nombre de sous-composants.
389 ARCCORE_HOST_DEVICE void _setNbSubItem(Int16 nb_sub_item)
390 {
391 m_shared_info->_setNbSubConstituent(m_constituent_item_index, nb_sub_item);
392 }
393
394 //! Positionne le premier sous-composant.
395 ARCCORE_HOST_DEVICE void _setFirstSubItem(ConstituentItemIndex first_sub_item)

397 m_shared_info->_setFirstSubConstituentLocalId(m_constituent_item_index, first_sub_item);
398 }
399
400 ARCCORE_HOST_DEVICE void _setComponent(Int16 component_id)
401 {
402 m_shared_info->_setComponentId(m_constituent_item_index, component_id);
403 }
404
405 private:
406
407 ConstituentItemIndex m_constituent_item_index;
408 ComponentItemSharedInfo* m_shared_info = ComponentItemSharedInfo::null_shared_info_pointer;
409};
