187  int returnValue() const { return m_return_value; }
188  bool isValid() const
189  {
190  if (m_type==T_Null)
191  return false;
192  // Si le type de la requête est différent du type
193  // de la requête nulle, alors la requête est considérée
194  // comme valide.
195  if (m_type!=null_request_type)
196  return true;
197  if (m_type==T_Int)
198  return m_request.i!=null_request.i;
199  if (m_type==T_Long)
200  return m_request.l!=null_request.l;
201  if (m_type==T_SizeT)
202  return m_request.st!=null_request.st;
203  if (m_type==T_Ptr)
204  return m_request.cv!=null_request.cv;
205  return false;
206  }
