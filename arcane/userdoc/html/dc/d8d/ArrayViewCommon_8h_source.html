31 template<typename ViewType> ARCCORE_HOST_DEVICE
32 auto subViewInterval(ViewType view,
33  typename ViewType::size_type index,
34  typename ViewType::size_type nb_interval) -> ViewType
35 {
36  using size_type = typename ViewType::size_type;
37  if (nb_interval<=0)
38  return ViewType();
39  if (index<0 || index>=nb_interval)
40  return ViewType();
41  size_type n = view.size();
42  size_type isize = n / nb_interval;
43  size_type ibegin = index * isize;
44  // Pour le dernier interval, prend les elements restants
45  if ((index+1)==nb_interval)
46  isize = n - ibegin;
47  return ViewType::create(view.data()+ibegin,isize);
48 }

59 template<typename ViewType> inline void
60 dumpArray(std::ostream& o,ViewType val,int max_print)
61 {
62  using size_type = typename ViewType::size_type;
63  size_type n = val.size();
64  if (max_print>0 && n>max_print){
65  // N'affiche que les (max_print/2) premiers et les (max_print/2) derniers
66  // sinon si le tableau est très grand cela peut générer des
67  // sorties listings énormes.
68  size_type z = (max_print/2);
69  size_type z2 = n - z;
70  o << "[0]=\"" << val[0] << '"';
71  for( size_type i=1; i<z; ++i )
72  o << " [" << i << "]=\"" << val[i] << '"';
73  o << " ... ... (skipping indexes " << z << " to " << z2 << " ) ... ... ";
74  for( size_type i=(z2+1); i<n; ++i )
75  o << " [" << i << "]=\"" << val[i] << '"';
76  }
77  else{
78  for( size_type i=0; i<n; ++i ){
79  if (i!=0)
80  o << ' ';
81  o << "[" << i << "]=\"" << val[i] << '"';
82  }
83  }
84 }

90 template<typename ViewType> inline bool
91 areEqual(ViewType rhs, ViewType lhs)
92 {
93  using size_type = typename ViewType::size_type;
94  if (rhs.size()!=lhs.size())
95  return false;
96  size_type s = rhs.size();
97  for( size_type i=0; i<s; ++i ){
98  if (rhs[i]!=lhs[i])
99  return false;
100  }
101  return true;
102 }
