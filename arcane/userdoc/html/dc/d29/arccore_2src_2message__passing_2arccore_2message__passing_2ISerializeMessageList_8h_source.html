33/*!
34 * \brief Interface d'une liste de messages de sérialisation.
35 *
36 * Les instances de cette classe sont en général créées via la
37 * méthode mpCreateSerializeMessageListRef().
38 *
39 * \code
40 * using namespace Arccore;
41 * IMessagePassingMng* mpm = ...;
42 * Ref<ISerializeMessagList> serialize_list(mpCreateSerializeMessageListRef(mpm));
43 * // Ajoute deux messages pour recevoir du rang 1 et 3.
44 * Ref<ISerializeMessage> message1(serialize_list->createAndAddMessage(MessageRank(1),MsgReceive));
45 * Ref<ISerializeMessage> message2(serialize_list->createAndAddMessage(MessageRank(3),MsgReceive));
46 *
47 * // Attend que les messages soient terminés
48 * // En mode WaitSome ou TestSome, il est possible de savoir si
49 * // un message est terminé en appelant la méthode ISerializeMessage::finished()
50 * serialize_list->wait(WaitAll);
51 *
52 * // Récupère les données de sérialisation du premier message
53 * ISerializer* sb1 = message1.serializer();
54 * sb1->setMode(ISerializer::ModeGet);
55 * UniqueArray<Int32> int32_array;
56 * sb1->getArray(int32_array);
57 * ...
58 * \endcode
59 */
60class ARCCORE_MESSAGEPASSING_EXPORT ISerializeMessageList
61{
62 public:
63
64 virtual ~ISerializeMessageList() {} //!< Libère les ressources.
65
66 public:
67
68 /*!
69 * \brief Ajoute un message à la liste.
70 *
71 * Le message n'est pas posté tant qu'aucun appel à processPendingMessages()
72 * n'a été effectué. L'utilisateur garde la propriété du message qui
73 * ne doit pas être détruit tant qu'il n'est pas terminé.
74 */
75 virtual void addMessage(ISerializeMessage* msg) =0;
76
77 /*!
78 * \brief Envoie les messages de la liste qui ne l'ont pas encore été.
79 *
80 * Cette méthode envoie les messages ajoutés via addMessage() qui ne l'ont
81 * pas encore été. Il n'est en général pas nécessaire d'appeler cette
82 * méthode car cele est fait automatiquement lors de l'appel à waitMessages().
83 */
84 virtual void processPendingMessages() =0;
85
86 /*!
87 * \brief Attend que les messages aient terminé leur exécution.
88 *
89 * Le type d'attente est spécifié par \a wt.
90 *
91 * Il est ensuite possible de tester si un message est terminé
92 * via la méthode ISerializeMessage::isFinished(). Cette classe ne garde
93 * aucune référence sur les messages terminés qui peuvent donc être
94 * détruits par l'utilisateur dès qu'il n'en a plus besoin.
95 *
96 * \return le nombre de messages complètement exécutés ou (-1) s'ils
97 * l'ont tous été.
98 */
99 virtual Integer waitMessages(eWaitType wt) =0;
100
101 /*!
102 * \brief Créé et ajoute un message de sérialisation.
103 *
104 * Le message peut être un message d'envoie ou de réception.
105 *
106 * Si le message est de réception (MsgReceive), il est possible de
107 * spécifier un rang nul pour indiquer qu'on souhaite recevoir de n'importe
108 * qui.
109 *
110 * Cette méthode appelle addMessage() pour ajouter automatiquement le
111 * message créé à la liste des messages. L'instance ne conserve pas de référence
112 * au message créé qui ne doit pas être détruit tant qu'il n'est
113 * pas terminé.
114 */
115 virtual Ref<ISerializeMessage>
116 createAndAddMessage(MessageRank destination,
117 ePointToPointMessageType type) =0;
118};
