604 inline bool
605 searchExtrema(ConstArrayView<Real2> array,Real& xmin,Real& xmax,
606  Real& ymin,Real& ymax,bool need_init)
607 {
608  if (need_init){
609  xmin = ymin = 0.;
610  xmax = ymax = 1.;
611  }
612 
613  Integer size = array.size();
614  if (size==0)
615  return false;
616 
617  if (need_init){
618  xmin = xmax = array[0].x;
619  ymin = ymax = array[0].y;
620  }
621 
622  for( Integer i=1; i<size; ++i ){
623  if (array[i].x < xmin)
624  xmin = array[i].x;
625  if (array[i].x > xmax)
626  xmax = array[i].x;
627 
628  if (array[i].y < ymin)
629  ymin = array[i].y;
630  if (array[i].y > ymax)
631  ymax = array[i].y;
632  }
633  return true;
634 }
