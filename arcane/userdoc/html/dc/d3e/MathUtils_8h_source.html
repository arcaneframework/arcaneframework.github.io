642 inline bool
643 searchExtrema(ConstArrayView<Real2> array,Real& xmin,Real& xmax,
644  Real& ymin,Real& ymax,bool need_init)
645 {
646  if (need_init){
647  xmin = ymin = 0.;
648  xmax = ymax = 1.;
649  }
650 
651  Integer size = array.size();
652  if (size==0)
653  return false;
654 
655  if (need_init){
656  xmin = xmax = array[0].x;
657  ymin = ymax = array[0].y;
658  }
659 
660  for( Integer i=1; i<size; ++i ){
661  if (array[i].x < xmin)
662  xmin = array[i].x;
663  if (array[i].x > xmax)
664  xmax = array[i].x;
665 
666  if (array[i].y < ymin)
667  ymin = array[i].y;
668  if (array[i].y > ymax)
669  ymax = array[i].y;
670  }
671  return true;
672 }
