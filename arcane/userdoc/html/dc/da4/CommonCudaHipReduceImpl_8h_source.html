111 // Les devices d'architecture inférieure à 6.0 ne supportent pas
112 // les atomicAdd sur les 'double'.
113 // Ce code est issu de la documentation NVIDIA (programming guide)
114 __device__ inline double
115 preArch60atomicAdd(double* address, double val)
116 {
117  unsigned long long int* address_as_ull = (unsigned long long int*)address;
118  unsigned long long int old = *address_as_ull, assumed;
119 
120  do {
121  assumed = old;
122  old = atomicCAS(address_as_ull, assumed,
123  __double_as_longlong(val +
124  __longlong_as_double(assumed)));
125  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
126  } while (assumed != old);
127 
128  return __longlong_as_double(old);
129 }
130 __device__ inline double
131 atomicMaxDouble(double* address, double val)
132 {
133  unsigned long long int* address_as_ull = (unsigned long long int*)address;
134  unsigned long long int old = *address_as_ull, assumed;
135 
136  do {
137  assumed = old;
138  double assumed_as_double = __longlong_as_double(assumed);
139  old = atomicCAS(address_as_ull, assumed,
140  __double_as_longlong(val > assumed_as_double ? val :assumed_as_double ));
141  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
142  } while (assumed != old);
143 
144  return __longlong_as_double(old);
145 }

303 // Cette implémentation est celle de RAJA
305 template <typename ReduceOperator,typename T>
306 ARCCORE_DEVICE inline T block_reduce(T val, T identity)
307 {
308  int numThreads = blockDim.x * blockDim.y * blockDim.z;
309 
310  int threadId = getThreadId();
311 
312  int warpId = threadId % WARP_SIZE;
313  int warpNum = threadId / WARP_SIZE;
314 
315  T temp = val;
316 
317  if (numThreads % WARP_SIZE == 0) {
318 
319  // reduce each warp
320  for (int i = 1; i < WARP_SIZE; i *= 2) {
321  T rhs = impl::shfl_xor_sync(temp, i);
322  ReduceOperator::apply(temp, rhs);
323  }
324 
325  } else {
326 
327  // reduce each warp
328  for (int i = 1; i < WARP_SIZE; i *= 2) {
329  int srcLane = threadId ^ i;
330  T rhs = impl::shfl_sync(temp, srcLane);
331  // only add from threads that exist (don't double count own value)
332  if (srcLane < numThreads) {
333  ReduceOperator::apply(temp, rhs);
334  }
335  }
336  }
337 
338  //printf("CONTINUE tid=%d wid=%d wnum=%d\n",threadId,warpId,warpNum);
339 
340  // reduce per warp values
341  if (numThreads > WARP_SIZE) {
342 
343  __shared__ T sd[MAX_WARPS];
344 
345  // write per warp values to shared memory
346  if (warpId == 0) {
347  sd[warpNum] = temp;
348  }
349 
350  __syncthreads();
351 
352  if (warpNum == 0) {
353 
354  // read per warp values
355  if (warpId * WARP_SIZE < numThreads) {
356  temp = sd[warpId];
357  } else {
358  temp = identity;
359  }
360  for (int i = 1; i < WARP_SIZE; i *= 2) {
361  T rhs = impl::shfl_xor_sync(temp, i);
362  ReduceOperator::apply(temp, rhs);
363  }
364  }
365 
366  __syncthreads();
367  }
368  return temp;
369 }

374 // returns true if put reduced value in val
375 template <typename ReduceOperator, typename T>
376 ARCCORE_DEVICE inline bool
377 grid_reduce(T& val,T identity,SmallSpan<T> device_mem,unsigned int* device_count)
378 {
379  int numBlocks = gridDim.x * gridDim.y * gridDim.z;
380  int numThreads = blockDim.x * blockDim.y * blockDim.z;
381  Int32 wrap_around = numBlocks - 1;
382 
383  int blockId = blockIdx.x + gridDim.x * blockIdx.y +
384  (gridDim.x * gridDim.y) * blockIdx.z;
385 
386  int threadId = threadIdx.x + blockDim.x * threadIdx.y +
387  (blockDim.x * blockDim.y) * threadIdx.z;
388 
389  T temp = block_reduce<ReduceOperator>(val, identity);
390 
391  // one thread per block writes to device_mem
392  bool lastBlock = false;
393  if (threadId == 0) {
394  device_mem[blockId] = temp;
395  // ensure write visible to all threadblocks
396  __threadfence();
397  // increment counter, (wraps back to zero if old count == wrap_around)
398  unsigned int old_count = ::atomicInc(device_count, wrap_around);
399  lastBlock = (old_count == wrap_around);
400  }
401 
402  // returns non-zero value if any thread passes in a non-zero value
403  lastBlock = __syncthreads_or(lastBlock);
404 
405  // last block accumulates values from device_mem
406  if (lastBlock) {
407  temp = identity;
408 
409  for (int i = threadId; i < numBlocks; i += numThreads) {
410  ReduceOperator::apply(temp, device_mem[i]);
411  }
412 
413  temp = block_reduce<ReduceOperator>(temp, identity);
414 
415  // one thread returns value
416  if (threadId == 0) {
417  val = temp;
418  }
419  }
420 
421  return lastBlock && threadId == 0;
422 }

427 template<typename DataType,typename ReduceOperator> ARCANE_INLINE_REDUCE ARCCORE_DEVICE
428 void _applyDeviceGeneric(const ReduceDeviceInfo<DataType>& dev_info)
429 {
430  SmallSpan<DataType> grid_buffer = dev_info.m_grid_buffer;
431  DataType identity = dev_info.m_identity;
432  unsigned int* device_count = dev_info.m_device_count;
433  DataType* ptr = dev_info.m_final_ptr;
434  DataType v = dev_info.m_current_value;
435  bool do_grid_reduce = dev_info.m_use_grid_reduce;
436 
437  //if (impl::getThreadId()==0){
438  //printf("BLOCK ID=%d %p s=%d ptr=%p %p\n",getBlockId(),grid_buffer.data(),grid_buffer.size(),ptr,(void*)device_count);
439  //}
440  if (do_grid_reduce){
441  bool is_done = grid_reduce<ReduceOperator>(v,identity,grid_buffer,device_count);
442  if (is_done){
443  *ptr = v;
444  // Il est important de remettre cette à zéro pour la prochaine utilisation d'un Reducer.
445  (*device_count) = 0;
446  }
447  }
448  else{
449  DataType rv = impl::block_reduce<ReduceOperator>(v,identity);
450  if (impl::getThreadId()==0){
451  CommonCudaHipAtomicMin<DataType>::apply(ptr,rv);
452  }
453  }
454 }
