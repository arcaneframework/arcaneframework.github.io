151// Cette implémentation est celle de RAJA
152//! reduce values in block into thread 0
153template <typename ReduceOperator,typename T>
154ARCCORE_DEVICE inline T block_reduce(T val, T identity)
155{
156 int numThreads = blockDim.x * blockDim.y * blockDim.z;
157
158 int threadId = getThreadId();
159
160 int warpId = threadId % WARP_SIZE;
161 int warpNum = threadId / WARP_SIZE;
162
163 T temp = val;
164
165 if (numThreads % WARP_SIZE == 0) {
166
167 // reduce each warp
168 for (int i = 1; i < WARP_SIZE; i *= 2) {
169 T rhs = impl::shfl_xor_sync(temp, i);
170 ReduceOperator::apply(temp, rhs);
171 }
172
173 } else {
174
175 // reduce each warp
176 for (int i = 1; i < WARP_SIZE; i *= 2) {
177 int srcLane = threadId ^ i;
178 T rhs = impl::shfl_sync(temp, srcLane);
179 // only add from threads that exist (don't double count own value)
180 if (srcLane < numThreads) {
181 ReduceOperator::apply(temp, rhs);
182 }
183 }
184 }
185
186 //printf("CONTINUE tid=%d wid=%d wnum=%d\n",threadId,warpId,warpNum);
187
188 // reduce per warp values
189 if (numThreads > WARP_SIZE) {
190
191 __shared__ T sd[MAX_WARPS];
192
193 // write per warp values to shared memory
194 if (warpId == 0) {
195 sd[warpNum] = temp;
196 }
197
198 __syncthreads();
199
200 if (warpNum == 0) {
201
202 // read per warp values
203 if (warpId * WARP_SIZE < numThreads) {
204 temp = sd[warpId];
205 } else {
206 temp = identity;
207 }
208 for (int i = 1; i < WARP_SIZE; i *= 2) {
209 T rhs = impl::shfl_xor_sync(temp, i);
210 ReduceOperator::apply(temp, rhs);
211 }
212 }
213
214 __syncthreads();
215 }
216 return temp;
217}

221//! reduce values in grid into thread 0 of last running block
222// returns true if put reduced value in val
223template <typename ReduceOperator, typename T>
224ARCCORE_DEVICE inline bool
225grid_reduce(T& val,T identity,SmallSpan<T> device_mem,unsigned int* device_count)
226{
227 int numBlocks = gridDim.x * gridDim.y * gridDim.z;
228 int numThreads = blockDim.x * blockDim.y * blockDim.z;
229 int wrap_around = numBlocks - 1;
230
231 int blockId = blockIdx.x + gridDim.x * blockIdx.y +
232 (gridDim.x * gridDim.y) * blockIdx.z;
233
234 int threadId = threadIdx.x + blockDim.x * threadIdx.y +
235 (blockDim.x * blockDim.y) * threadIdx.z;
236
237 T temp = block_reduce<ReduceOperator>(val, identity);
238
239 // one thread per block writes to device_mem
240 bool lastBlock = false;
241 if (threadId == 0) {
242 device_mem[blockId] = temp;
243 // ensure write visible to all threadblocks
244 __threadfence();
245 // increment counter, (wraps back to zero if old count == wrap_around)
246 // Attention: avec ROCm et un GPU sur bus PCI express si 'device_count'
247 // est alloué en mémoire unifiée le atomicAdd ne fonctionne pas.
248 // Dans ce cas on peut le remplacer par:
249 // unsigned int old_count = ::atomicAdd(device_count, 1) % (wrap_around+1);
250 unsigned int old_count = ::atomicInc(device_count, wrap_around);
251 lastBlock = ((int)old_count == wrap_around);
252 }
253
254 // returns non-zero value if any thread passes in a non-zero value
255 lastBlock = __syncthreads_or(lastBlock);
256
257 // last block accumulates values from device_mem
258 if (lastBlock) {
259 temp = identity;
260
261 for (int i = threadId; i < numBlocks; i += numThreads) {
262 ReduceOperator::apply(temp, device_mem[i]);
263 }
264
265 temp = block_reduce<ReduceOperator>(temp, identity);
266
267 // one thread returns value
268 if (threadId == 0) {
269 val = temp;
270 }
271 }
272
273 return lastBlock && threadId == 0;
274}

279template<typename DataType,typename ReduceOperator,typename AtomicReduceOperator>
280ARCANE_INLINE_REDUCE ARCCORE_DEVICE
281void _applyDeviceGeneric(const ReduceDeviceInfo<DataType>& dev_info)
282{
283 SmallSpan<DataType> grid_buffer = dev_info.m_grid_buffer;
284 DataType identity = dev_info.m_identity;
285 unsigned int* device_count = dev_info.m_device_count;
286 DataType* ptr = dev_info.m_device_final_ptr;
287 DataType v = dev_info.m_current_value;
288 bool do_grid_reduce = dev_info.m_use_grid_reduce;
289
290 //if (impl::getThreadId()==0){
291 // printf("BLOCK ID=%d %p s=%d ptr=%p %p use_grid_reduce=%d\n",
292 // getBlockId(),grid_buffer.data(),grid_buffer.size(),ptr,
293 // (void*)device_count,(do_grid_reduce)?1:0);
294 //}
295 if (do_grid_reduce){
296 bool is_done = grid_reduce<ReduceOperator>(v,identity,grid_buffer,device_count);
297 if (is_done){
298 *ptr = v;
299 // Il est important de remettre cette à zéro pour la prochaine utilisation d'un Reducer.
300 (*device_count) = 0;
301 }
302 }
303 else{
304 DataType rv = impl::block_reduce<ReduceOperator>(v,identity);
305 if (impl::getThreadId()==0){
306 AtomicReduceOperator::apply(ptr,rv);
307 }
308 }
309}
