109/*!
110 * \brief Classe gérant un vecteur de réel de dimension 3.
111
112 Le vecteur comprend trois composantes \a x, \a y et \a z qui sont du
113 type \b Real.
114
115 \code
116 Real3 value (1.0,2.3,4.5); // Créé un triplet (x=1.0, y=2.3, z=4.5)
117 cout << value.x; // Imprime le composant x
118 value.y += 3.2; // Ajoute 3.2 à la composante \b y
119 \endcode
120
121 ou de maniere equivalente
122
123 \code
124 Real3 value (1.0,2.3,4.5); // Créé un triplet (x=1.0, y=2.3, z=4.5)
125 cout << value[0]; // Imprime le composant x
126 value[1] += 3.2; // Ajoute 3.2 à la composante \b y
127 \endcode
128 */
129

355 /*!
356 * \brief Normalise le triplet.
357 *
358 * Si le triplet est non nul, divise chaque composante par la norme du triplet
359 * (abs()), de telle sorte qu'après l'appel à cette méthode, normL2() vaux \a 1.
360 * Si le triplet est nul, ne fait rien.
361 */
362 Real3& normalize()
363 {
364 Real d = normL2();

366 divSame(d);
367 return (*this);
368 }

387 /*!
388 * \brief Opérateur de comparaison.
389 *
390 * Cet opérateur permet de trier les Real3 pour les utiliser par exemple
391 * dans les std::set
392 */
393 friend constexpr ARCCORE_HOST_DEVICE bool operator<(Real3 v1, Real3 v2)
394 {
395 if (v1.x == v2.x) {
396 if (v1.y == v2.y)
397 return v1.z < v2.z;
398 else
399 return v1.y < v2.y;
400 }
401 return (v1.x < v2.x);
402 }

413 return t.assign(i);
414 }
415
416 /*!
417 * \brief Compare composant pas composante l'instance courante à \a b.
418 *
419 * \retval true si this.x==b.x et this.y==b.y et this.z==b.z.
420 * \retval false sinon.
421 */
422 constexpr ARCCORE_HOST_DEVICE bool operator==(Real3 b) const
423 {
424 return _eq(x, b.x) && _eq(y, b.y) && _eq(z, b.z);
425 }
426
427 /*!
428 * \brief Compare deux triplets.
429 * Pour la notion d'égalité, voir operator==()
430 * \retval true si les deux triplets sont différents,
431 * \retval false sinon.
432 */
433 constexpr ARCCORE_HOST_DEVICE bool operator!=(Real3 b) const { return !operator==(b); }
434

437 /*!

440 * \retval false sinon.
441 */
442 constexpr ARCCORE_HOST_DEVICE static bool _eq(Real a, Real b)
443 {
444 return math::isEqual(a, b);
445 }

448 {
449 return math::sqrt(a);
450 }
