111 /*!
112 * \brief Vue spécifique pour gérer les entités nulles.
113 *
114 * Pour l'entité nulle, l'index vaut NULL_ITEM_LOCAL_ID (soit (-1)) et il faut
115 * pouvoir accéder à `m_data` avec cet indice ce qui n'est pas possible

120 public:
121 Int32 operator[](Int32 index) const
122 {
123#ifdef ARCANE_CHECK
124 if (index==NULL_ITEM_LOCAL_ID){
125 // Pour l'entité nulle, la taille doit être 0.
126 if (m_size!=0)
127 arcaneRangeError(index,m_size);
128 }
129 else
130 ARCANE_CHECK_AT(index,m_size);
131#endif
132 return m_data[index];
133 }
134 void operator=(ConstArrayView<Int32> v)
135 {
136 m_data = v.data();
137 m_size = v.size();
138 }
139 // Il faut que data[NULL_ITEM_LOCAL_ID] soit valide.
140 // Il faut donc que (data-1) pointe vers une adresse valide
141 void setNull(const Int32* data)
142 {
143 m_data = data;
144 m_size = 0;
145 }
146 operator ConstArrayView<Int32>() const
147 {
148 return ConstArrayView<Int32>(m_size,m_data);
149 }
150 operator SmallSpan<const Int32>() const
151 {
152 return SmallSpan<const Int32>(m_data,m_size);
153 }
154 private:
155 Int32 m_size;
156 const Int32* m_data;
