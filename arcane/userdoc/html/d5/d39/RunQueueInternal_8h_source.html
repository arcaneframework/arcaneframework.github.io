48template <typename T>
49struct Privatizer
50{
51 using value_type = T;
52 using reference_type = value_type&;
53 value_type m_private_copy;
54
55 ARCCORE_HOST_DEVICE Privatizer(const T& o) : m_private_copy{o} {}
56 ARCCORE_HOST_DEVICE reference_type privateCopy() { return m_private_copy; }
57};
58
59template <typename T>
60ARCCORE_HOST_DEVICE auto privatize(const T& item) -> Privatizer<T>
61{
62 return Privatizer<T>{item};
63}
