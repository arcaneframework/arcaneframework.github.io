37 template <typename T>
38 struct Privatizer
39 {
40  using value_type = T;
41  using reference_type = value_type&;
42  value_type m_private_copy;
43 
44  ARCCORE_HOST_DEVICE Privatizer(const T& o) : m_private_copy{o} {}
45  ARCCORE_HOST_DEVICE reference_type privateCopy() { return m_private_copy; }
46 };

48 template <typename T>
49 ARCCORE_HOST_DEVICE auto privatize(const T& item) -> Privatizer<T>

51  return Privatizer<T>{item};
