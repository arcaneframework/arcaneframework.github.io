173//! Indique si \a exec_policy correspond à un accélérateur
174inline bool
175isAcceleratorPolicy(eExecutionPolicy exec_policy)
176{
177 return exec_policy == eExecutionPolicy::CUDA || exec_policy == eExecutionPolicy::HIP;
178}

182/*!
183 * \brief Accessibilité de l'adresse \a ptr pour une exécution sur la file \a queue.
184 *
185 * Si \a queue est nul, retourne ePointerAccessibility::Unknown.

211//! Accessibilité de l'adresse \a ptr pour une exécution sur \a queue_or_runner_or_policy.
212template <typename T> inline ePointerAccessibility
213getPointerAccessibility(T& queue_or_runner_or_policy, const void* ptr, PointerAttribute* ptr_attr = nullptr)
214{
215 return getPointerAccessibility(&queue_or_runner_or_policy, ptr, ptr_attr);
