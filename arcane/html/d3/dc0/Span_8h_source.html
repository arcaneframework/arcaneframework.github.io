263  template<class U> ARCCORE_HOST_DEVICE
264  inline void copy(const U& copy_array)
265  {
266  Int64 n = copy_array.size();
267  Int64 size_as_int64 = m_size;
268  arccoreCheckAt(n,size_as_int64+1);
269  const T* copy_begin = copy_array.data();
270  T* to_ptr = m_ptr;
271  // On est sur que \a n tient sur un 'SizeType' car il est plus petit
272  // que \a m_size
273  SizeType n_as_sizetype = static_cast<SizeType>(n);
274  for( SizeType i=0; i<n_as_sizetype; ++i )
275  to_ptr[i] = copy_begin[i];
276  }

282  {
283  for( SizeType i=0; i<m_size; ++i ){
284  if (m_ptr[i]==v)
285  return true;
286  }
287  return false;
288  }

515 template<typename T,typename SizeType> inline bool
516 operator==(SpanImpl<const T,SizeType> rhs, SpanImpl<const T,SizeType> lhs)
517 {
518  if (rhs.size()!=lhs.size())
519  return false;
520  SizeType s = rhs.size();
521  for( SizeType i=0; i<s; ++i ){
522  if (rhs[i]==lhs[i])
523  continue;
524  else
525  return false;
526  }
527  return true;
528 }

585 template<typename T,typename SizeType> inline void
586 dumpArray(std::ostream& o,SpanImpl<const T,SizeType> val,int max_print)
587 {
588  SizeType n = val.size();
589  if (max_print>0 && n>max_print){
590  // N'affiche que les (max_print/2) premiers et les (max_print/2) derniers
591  // sinon si le tableau est très grand cela peut générer des
592  // sorties listings énormes.
593  SizeType z = (max_print/2);
594  SizeType z2 = n - z;
595  o << "[0]=\"" << val[0] << '"';
596  for( SizeType i=1; i<z; ++i )
597  o << " [" << i << "]=\"" << val[i] << '"';
598  o << " ... ... (skipping indexes " << z << " to " << z2 << " ) ... ... ";
599  for( SizeType i=(z2+1); i<n; ++i )
600  o << " [" << i << "]=\"" << val[i] << '"';
601  }
602  else{
603  for( SizeType i=0; i<n; ++i ){
604  if (i!=0)
605  o << ' ';
606  o << "[" << i << "]=\"" << val[i] << '"';
607  }
608  }
609 }

621 template<typename DataType,typename IntegerType,typename SizeType> inline void
622 _sampleSpan(SpanImpl<const DataType,SizeType> values,
623  SpanImpl<const IntegerType,SizeType> indexes,
624  SpanImpl<DataType,SizeType> result)
625 {
626  const Int64 result_size = indexes.size();
627  [[maybe_unused]] const Int64 my_size = values.size();
628  const DataType* ptr = values.data();
629  for( Int64 i=0; i<result_size; ++i) {
630  IntegerType index = indexes[i];
631  ARCCORE_CHECK_AT(index,my_size);
632  result[i] = ptr[index];
633  }
634 }
