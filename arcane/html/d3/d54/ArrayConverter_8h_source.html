30 template<typename TypeA,typename TypeB>
31 class DefaultConverter
32 {
33  public:
34  void convertFromAToB(ConstArrayView<TypeA> input,ArrayView<TypeB> output)
35  {
36  for( Integer i=0, is=input.size(); i<is; ++i )
37  output[i] = (TypeB)input[i];
38  }
39  void convertFromBToA(ConstArrayView<TypeB> input,ArrayView<TypeA> output)
40  {
41  for( Integer i=0, is=input.size(); i<is; ++i )
42  output[i] = (TypeA)input[i];
43  }
44 };

51 template<typename InputType,typename OutputType,
52  typename Converter = DefaultConverter<InputType,OutputType> >
53 class ArrayConverter
54 {
55  public:
56  typedef UniqueArray<OutputType> OutputArrayType;
57  typedef typename OutputArrayType::iterator iterator;

59  typedef typename OutputArrayType::pointer pointer;
60  typedef typename OutputArrayType::const_pointer const_pointer;
61  public:
62 
63  ArrayConverter() {}
64  ArrayConverter(Converter & conv)
65  : m_converter(conv)
66  {
67  }
68

70  {
71  m_input_array = ArrayView<InputType>(nb,values);
72  _init();
73  }
74  ArrayConverter(ArrayView<InputType> values)
75  {
76  m_input_array = values;
77  _init();
78  }
79  ArrayConverter(ArrayView<InputType> values, Converter& conv)
80  :m_converter(conv)
81  {
82  m_input_array = values;
83  _init();
84  }
85  ~ArrayConverter()
86  {
87  m_converter.convertFromBToA(m_output_array,m_input_array);
88  }
89  void operator=(ArrayView<InputType> values)
90  {
91  m_input_array = values;
92  _init();
93  }
94  void notifyOutputChanged()
95  {
96  m_converter.convertFromBToA(m_output_array,m_input_array);
97  }
98  void notifyInputChanged()
99  {
100  m_converter.convertFromAToB(m_input_array,m_output_array);
101  }
102  public:
103 
107  ARCANE_DEPRECATED_280 iterator begin() { return m_output_array.begin(); }
111  ARCANE_DEPRECATED_280 const_iterator begin() const { return m_output_array.begin(); }
112  OutputArrayType& array() { return m_output_array; }
113  OutputArrayType& array() const { return m_output_array; }
114  pointer data() { return m_output_array.data(); }
115  const_pointer data() const { return m_output_array.data(); }
116 
117  private:
118 
119  void _init()
120  {
121  m_output_array.resize(m_input_array.size());
122  m_converter.convertFromAToB(m_input_array,m_output_array);
123  }
124 
125  ArrayView<InputType> m_input_array;
126  OutputArrayType m_output_array;
127  Converter m_converter;
128 };
