35 class ARCANE_CORE_EXPORT IndexedItemConnectivityViewBase
36 {
37  public:
39  ARCCORE_HOST_DEVICE Int32 nbItem(ItemLocalId lid) const { return m_nb_item[lid]; }
41  ARCCORE_HOST_DEVICE ItemLocalIdView<Item> items(ItemLocalId lid) const
42  {
43  const Int32* ptr = & m_list_data[m_indexes[lid]];
44  return { reinterpret_cast<const ItemLocalId*>(ptr), m_nb_item[lid] };
45  }
46  eItemKind sourceItemKind() const { return m_source_kind; }
47  eItemKind targetItemKind() const { return m_target_kind; }
48  public:

52  {
53  m_indexes = indexes;
54  m_nb_item = nb_item;
55  m_list_data = list_data;
56  m_source_kind = source_kind;
57  m_target_kind = target_kind;
58  }

60  void set(IndexedItemConnectivityViewBase view)
61  {
62  m_indexes = view.m_indexes;
63  m_nb_item = view.m_nb_item;
64  m_list_data = view.m_list_data;
65  m_source_kind = view.m_source_kind;
66  m_target_kind = view.m_target_kind;
67  }
68  protected:
69  SmallSpan<const Int32> m_nb_item;
70  SmallSpan<const Int32> m_indexes;
71  SmallSpan<const Int32> m_list_data;
72  eItemKind m_source_kind = IK_Unknown;
73  eItemKind m_target_kind = IK_Unknown;
74  protected:
75  [[noreturn]] void _badConversion(eItemKind k1,eItemKind k2) const;
76  inline void _checkValid(eItemKind k1,eItemKind k2) const
77  {
78  if (k1!=m_source_kind || k2!=m_target_kind)
79  _badConversion(k1,k2);
80  }

89 template<typename ItemType1,typename ItemType2>
90 class IndexedItemConnectivityGenericView
91 : public IndexedItemConnectivityViewBase
92 {
93  public:
94  using ItemType1Type = ItemType1;
95  using ItemType2Type = ItemType2;
96  using ItemLocalId1 = typename ItemType1::LocalIdType;
97  using ItemLocalId2 = typename ItemType2::LocalIdType;
98  using ItemLocalIdViewType = ItemLocalIdView<ItemType2>;
99  public:
100  IndexedItemConnectivityGenericView(IndexedItemConnectivityViewBase view)
101  : IndexedItemConnectivityViewBase(view)
102  {
103 #ifdef ARCANE_CHECK
104  eItemKind k1 = ItemTraitsT<ItemType1>::kind();
105  eItemKind k2 = ItemTraitsT<ItemType2>::kind();
106  _checkValid(k1,k2);
107 #endif
108  }
109  IndexedItemConnectivityGenericView() = default;
110  public:
112  ARCCORE_HOST_DEVICE ItemLocalIdViewType items(ItemLocalId1 lid) const
113  {
114  const Int32* ptr = & m_list_data[m_indexes[lid]];
115  return { reinterpret_cast<const ItemLocalId2*>(ptr), m_nb_item[lid] };
116  }
117 };
