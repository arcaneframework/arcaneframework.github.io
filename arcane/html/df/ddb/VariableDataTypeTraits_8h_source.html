102  static bool verifDifferent(Byte v1,Byte v2,Byte& diff)
103  {
104  if (v1!=v2){
105  if (math::isZero(v1))
106  diff = (Byte)(v1-v2);
107  else
108  diff = (Byte)((v1-v2)/v1);
109  return true;
110  }
111  return false;
112  }
113

115  {
116  return (Byte)math::abs(v);

157  { builtInDumpValue(s,v); }
163  static bool getValue(Type& v,const String& s)
164  { return builtInGetValue(v,s); }
165 
166  static bool verifDifferent(Real v1,Real v2,Real& diff)
167  {
168  // Vérifie avant de les comparer que les deux nombres sont valides
169  // pour éviter une exception flottante sur certaines plates-formes
170  if (platform::isDenormalized(v1) || platform::isDenormalized(v2)){
171  diff = 1.0;
172  return true;
173  }
174  if (v1!=v2){
175  if (math::abs(v1) < 1.e-100) // TH: plantait pour v1 tres petit(math::isZero(v1))
176  diff = v1-v2;
177  else
178  diff = (v1-v2)/v1;
179  return true;
180  }
181  return false;
182  }
183 
184  static Real normeMax(Real v)
185  {
186  return math::abs(v);
187  }
188 };
