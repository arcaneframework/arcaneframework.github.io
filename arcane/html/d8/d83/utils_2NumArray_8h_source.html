56 template<typename DataType,int RankValue>
57 class MDSpanBase
58 {
59  using UnqualifiedValueType = std::remove_cv_t<DataType>;
60  friend class NumArrayBase<UnqualifiedValueType,RankValue>;
61  // Pour que MDSpan<const T> ait accès à MDSpan<T>
62  friend class MDSpanBase<const UnqualifiedValueType,RankValue>;
63  public:
64  MDSpanBase() = default;
65  ARCCORE_HOST_DEVICE MDSpanBase(DataType* ptr,ArrayExtentsWithOffset<RankValue> extents)
66  : m_ptr(ptr), m_extents(extents)
67  {
68  }
69  // Constructeur MDSpan<const T> à partir d'un MDSpan<T>
70  template<typename X,typename = std::enable_if_t<std::is_same_v<X,UnqualifiedValueType>>>
71  ARCCORE_HOST_DEVICE MDSpanBase(const MDSpanBase<X,RankValue>& rhs)
72  : m_ptr(rhs.m_ptr), m_extents(rhs.m_extents){}
73  public:
74  ARCCORE_HOST_DEVICE DataType* _internalData() { return m_ptr; }
75  ARCCORE_HOST_DEVICE const DataType* _internalData() const { return m_ptr; }
76  public:
77  ArrayExtents<RankValue> extents() const
78  {
79  return m_extents.extents();
80  }
81  ArrayExtentsWithOffset<RankValue> extentsWithOffset() const
82  {
83  return m_extents;
84  }
85  Int64 extent(int i) const { return m_extents(i); }
86  public:
87  ARCCORE_HOST_DEVICE Int64 offset(ArrayBoundsIndex<RankValue> idx) const
88  {
89  return m_extents.offset(idx);
90  }
92  ARCCORE_HOST_DEVICE DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
93  {
94  return m_ptr[offset(idx)];
95  }
97  ARCCORE_HOST_DEVICE DataType* ptrAt(ArrayBoundsIndex<RankValue> idx) const
98  {
99  return m_ptr+offset(idx);
100  }
101  public:
102  MDSpanBase<const DataType,RankValue> constSpan() const
103  { return MDSpanBase<const DataType,RankValue>(m_ptr,m_extents); }
104  Span<DataType> to1DSpan() { return { m_ptr, m_extents.totalNbElement() }; }
105  Span<const DataType> to1DSpan() const { return { m_ptr, m_extents.totalNbElement() }; }
106  private:
107  // Utilisé uniquement par NumArrayBase pour la copie
108  Span2<const DataType> _internalTo2DSpan() const
109  {
110  Int64 dim1_size = m_extents(0);
111  Int64 dim2_size = 1;
112  for (int i=1; i<RankValue; ++i )
113  dim2_size *= m_extents(i);
114  return { m_ptr, dim1_size, dim2_size };
115  }
116  protected:
117  DataType* m_ptr = nullptr;
118  ArrayExtentsWithOffset<RankValue> m_extents;
119 };

392 template<typename DataType,int RankValue>
393 class NumArrayBase
394 {
395  // On utilise pour l'instant un UniqueArray2 pour conserver les valeurs.
396  // La première dimension du UniqueArray2 correspond à extent(0) et la
397  // deuxième dimension au dimensions restantes. Par exemple pour un NumArray<Int32,3>
398  // ayant comme nombre d'éléments dans chaque dimension (5,9,3), cela
399  // correspond à un 'UniqueArray2' dont le nombre d'éléments est (5,9*3).
400  public:
401  using ConstSpanType = MDSpan<const DataType,RankValue>;
402  using SpanType = MDSpan<DataType,RankValue>;
403 
404  public:
406  Int64 totalNbElement() const { return m_total_nb_element; }
408  Int64 extent(int i) const { return m_span.extent(i); }
413  void resize(ArrayExtents<RankValue> extents)
414  {
415  m_span.m_extents.setSize(extents);
416  _resize();
417  }
418  protected:
419  NumArrayBase() : m_data(platform::getDefaultDataAllocator()){}
420  explicit NumArrayBase(ArrayExtents<RankValue> extents)
421  : m_data(platform::getDefaultDataAllocator())
422  {
423  resize(extents);
424  }
425  private:
426  void _resize()
427  {
428  Int64 dim1_size = extent(0);
429  Int64 dim2_size = 1;
430  // TODO: vérifier débordement.
431  for (int i=1; i<RankValue; ++i )
432  dim2_size *= extent(i);
433  m_total_nb_element = dim1_size * dim2_size;
434  m_data.resize(dim1_size,dim2_size);
435  m_span.m_ptr = m_data.to1DSpan().data();
436  }
437  public:
438  void fill(const DataType& v) { m_data.fill(v); }
439  Int32 nbDimension() const { return RankValue; }
440  ArrayExtents<RankValue> extents() const { return m_span.extents(); }
441  ArrayExtentsWithOffset<RankValue> extentsWithOffset() const
442  {
443  return m_span.extentsWithOffset();
444  }
445  public:
446  SpanType span() { return m_span; }
447  ConstSpanType span() const { return m_span.constSpan(); }
448  ConstSpanType constSpan() const { return m_span.constSpan(); }
449  public:
450  Span<const DataType> to1DSpan() const { return m_data.to1DSpan(); }
451  Span<DataType> to1DSpan() { return m_data.to1DSpan(); }
452  void copy(ConstSpanType rhs) { m_data.copy(rhs._internalTo2DSpan()); }
453  const DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
454  {
455  return m_span(idx);
456  }
457  DataType& s(ArrayBoundsIndex<RankValue> idx)
458  {
459  return m_span(idx);
460  }
461  void swap(NumArrayBase<DataType,RankValue>& rhs)
462  {
463  m_data.swap(rhs.m_data);
464  std::swap(m_span,rhs.m_span);
465  std::swap(m_total_nb_element,rhs.m_total_nb_element);
466  }
467  Int64 capacity() const { return m_data.capacity(); }
468  public:
470  DataType* _internalData() { return m_span._internalData(); }
471  protected:
472  MDSpan<DataType,RankValue> m_span;
473  UniqueArray2<DataType> m_data;
474  Int64 m_total_nb_element = 0;
475 };
