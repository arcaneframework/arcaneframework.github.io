54 template<class T>
55 class ReferenceCounter
56 : public CheckedPointer<T>
57 {
58  public:
59 
61  typedef CheckedPointer<T> BaseClass;
62 
63  using BaseClass::m_value;
64 
65  public:
66 
68  ReferenceCounter() : BaseClass(nullptr) {}
70  explicit ReferenceCounter(T* t) : BaseClass(nullptr) { _changeValue(t); }
72  ReferenceCounter(const ReferenceCounter<T>& from)
73  : BaseClass(nullptr) { _changeValue(from.m_value); }
74 
76  ReferenceCounter<T>& operator=(const ReferenceCounter<T>& from)
77  {
78  _changeValue(from.m_value);
79  return (*this);
80  }
81 
83  ReferenceCounter<T>& operator=(T* new_value)
84  {
85  _changeValue(new_value);
86  return (*this);
87  }
88 
90  ~ReferenceCounter() { _removeRef(); }
91 
92  private:
93 
95  void _removeRef()
96  {
97  if (m_value)
98  ReferenceCounterAccessor<T>::removeReference(m_value);
99  }
101  void _changeValue(T* new_value)
102  {
103  if (m_value==new_value)
104  return;
105  // Toujours ajouter avant pour le cas où la nouvelle valeur
106  // et l'ancienne seraient issues de la même instance.
107  if (new_value)
108  ReferenceCounterAccessor<T>::addReference(new_value);
109  _removeRef();
110  m_value = new_value;
111  }
112 };
