85 template <int N>
86 class ComplexLoopRanges
87 {
88  public:
89  typedef typename ArrayBounds<N>::IndexType IndexType;
90  public:
91  ComplexLoopRanges(ArrayBounds<N> lower,ArrayBounds<N> extents)
92  : m_lower_bounds(lower.asStdArray()), m_extents(extents){}
93  public:
94  constexpr Int64 lowerBound(int i) const { return m_lower_bounds[i]; }
95  constexpr Int64 upperBound(int i) const { return m_lower_bounds[i]+m_extents.extent(i); }
96  constexpr Int64 extent(int i) const { return m_extents.extent(i); }
97  constexpr Int64 nbElement() const { return m_extents.nbElement(); }
98  constexpr ArrayBoundsIndex<N> getIndices(Int64 i) const
99  {
100  auto x = m_extents.getIndices(i);
101  x.add(m_lower_bounds);
102  return x;
103  }
104  private:
105  ArrayBoundsIndex<N> m_lower_bounds;
106  ArrayBounds<N> m_extents;
107 };
