294  void _resize(Int64 new_size,InitBehaviour rb)
295  {
296  Int64 old_size = m_p->dim1_size;
297  if (new_size==old_size)
298  return;
299  _resize2(new_size,m_p->dim2_size,rb);

304  void _resize(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
305  {
306  if (new_size2==m_p->dim2_size){
307  _resize(new_size1,rb);
308  }
309  else if (totalNbElement()==0){
310  _resizeFromEmpty(new_size1,new_size2,rb);
311  }
312  else if (new_size2<m_p->dim2_size){
313  _resizeSameDim1ReduceDim2(new_size2,rb);
314  _resize(new_size1,rb);
315  }
316  else if (new_size2>m_p->dim2_size){
317  _resizeSameDim1IncreaseDim2(new_size2,rb);
318  _resize(new_size1,rb);
319  }
320  else
321  throw NotImplementedException("Array2::resize","already sized");
322  }
323  void _resizeFromEmpty(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
324  {
325  _resize2(new_size1,new_size2,rb);
326  m_p->dim1_size = new_size1;
327  m_p->dim2_size = new_size2;
328  _arccoreCheckSharedNull();
329  }
330  void _resizeSameDim1ReduceDim2(Int64 new_size2,InitBehaviour rb)
331  {
332  ARCCORE_ASSERT((new_size2<m_p->dim2_size),("Bad Size"));
333  Int64 n = m_p->dim1_size;
334  Int64 n2 = m_p->dim2_size;
335  for( Int64 i=0; i<n; ++i ){
336  for( Int64 j=0; j<new_size2; ++j )
337  m_p->ptr[(i*new_size2)+j] = m_p->ptr[(i*n2)+j];
338  }
339  _resize2(n,new_size2,rb);
340  m_p->dim2_size = new_size2;
341  _arccoreCheckSharedNull();
342  }
343  void _resizeSameDim1IncreaseDim2(Int64 new_size2,InitBehaviour rb)
344  {
345  ARCCORE_ASSERT((new_size2>m_p->dim2_size),("Bad Size"));
346  Int64 n = m_p->dim1_size;
347  Int64 n2 = m_p->dim2_size;
348  _resize2(n,new_size2,rb);
349  // Recopie en partant de la fin pour éviter tout écrasement
350  for( Int64 i=(n-1); i>=0; --i ){
351  for( Int64 j=(n2-1); j>=0; --j )
352  m_p->ptr[(i*new_size2)+j] = m_p->ptr[(i*n2)+j];
353  }
354  m_p->dim2_size = new_size2;
355  _arccoreCheckSharedNull();
356  }
357  void _resize2(Int64 d1,Int64 d2,InitBehaviour rb)
358  {
359  Int64 new_size = d1*d2;
360  // Si la nouvelle taille est nulle, il faut tout de meme faire une allocation
361  // pour stocker les valeurs dim1_size et dim2_size (sinon, elle seraient
362  // dans TrueImpl::shared_null
363  if (new_size==0)
364  this->_reserve(4);
365  if (rb==IB_InitWithDefault)
366  Base::_resize(new_size,DataType());
367  else if (rb==IB_NoInit)
368  Base::_resize(new_size);
369  else
370  throw NotSupportedException("Array2::_resize2","invalid value InitBehaviour");
371  }
372  void _move(Array2<DataType>& rhs)
373  {
374  Base::_move(rhs);
375  }

377  {
378  Base::_swap(rhs);
379  }

382  {
383  if (m_p==ArrayImplBase::shared_null)
384  ArrayImplBase::throwBadSharedNull();
385  }

497  {
498  Integer nb_ref = 1;
499  for( ThatClassType* i = m_prev; i; i = i->m_prev )
500  ++nb_ref;
501  for( ThatClassType* i = m_next; i; i = i->m_next )
502  ++nb_ref;
503  return nb_ref;
504  }
511  void _addReference(const ThatClassType* new_ref)
512  {
513  ThatClassType* nf = const_cast<ThatClassType*>(new_ref);
514  ThatClassType* prev = nf->m_prev;
515  nf->m_prev = this;
516  m_prev = prev;
517  m_next = nf;
518  if (prev)
519  prev->m_next = this;
520  }
522  void _removeReference()
523  {
524  if (m_prev)
525  m_prev->m_next = m_next;
526  if (m_next)
527  m_next->m_prev = m_prev;
528  }
530  void _checkFreeMemory()
531  {
532  if (m_p->nb_ref==0){
533  this->_destroy();
534  this->_internalDeallocate();
535  }
536  }
537  void _operatorEqual(const ThatClassType& rhs)
538  {
539  if (&rhs!=this){
540  _removeReference();
541  _addReference(&rhs);
542  ++rhs.m_p->nb_ref;
543  --m_p->nb_ref;
544  _checkFreeMemory();
545  this->_setMP(rhs.m_p);
546  }
547  }
548  private:
549  ThatClassType* m_next;
550  ThatClassType* m_prev;
551  private:
553  void operator=(const Array2<T>& rhs);
