417  template<typename StringContainer> void
418  split(StringContainer& str_array,char c) const
419  {
420  const String& str = *this;
421  //TODO: passer par String::bytes().
422  const char* str_str = str.localstr();
423  Int64 offset = 0;
424  Int64 len = str.length();
425  for( Int64 i=0; i<len; ++i ){
426  // GG: remet temporairement l'ancienne sémantique (équivalente à strtok())
427  // et supprime la modif IFPEN car cela cause trop d'incompatibilités avec
428  // le code existant. A noter que l'implémentation de l'ancienne sémantique
429  // a plusieurs bugs:
430  // 1. dans le cas où on répète 3 fois ou plus consécutivement le
431  // délimiteur. Par exemple 'X:::Y' retourne {'X',':','Y'} au lieu de
432  // {'X','Y'}
433  // 2. Si on commence par le délimiteur, ce dernier est retourné:
434  // Avec ':X:Y', on retourne {':X','Y'} au lieu de {'X','Y'}
435  //if (str_str[i]==c){
436  if (str_str[i]==c && i!=offset){
437  str_array.push_back(std::string_view(str_str+offset,i-offset));
438  offset = i+1;
439  }
440  }
441  if (len!=offset)
442  str_array.push_back(std::string_view(str_str+offset,len-offset));
443  }
