459  virtual void _internalRealloc(Int64 new_capacity,bool compute_capacity)
460  {
461  if (m_p==TrueImpl::shared_null){
462  if (new_capacity!=0)
463  _internalAllocate(new_capacity);
464  return;
465  }
466 
467  Int64 acapacity = new_capacity;
468  if (compute_capacity){
469  acapacity = m_p->capacity;
470  //std::cout << " REALLOC: want=" << wanted_size << " current_capacity=" << capacity << '\n';
471  while (new_capacity>acapacity)
472  acapacity = (acapacity==0) ? 4 : (acapacity + 1 + acapacity / 2);
473  //std::cout << " REALLOC: want=" << wanted_size << " new_capacity=" << capacity << '\n';
474  }
475  // Si la nouvelle capacité est inférieure à la courante,ne fait rien.
476  if (acapacity<m_p->capacity)
477  return;
478  _internalReallocate(acapacity,IsPODType());
479  }

482  virtual void _internalReallocate(Int64 new_capacity,TrueType)
483  {

485  Int64 old_capacity = m_p->capacity;
486  _directReAllocate(new_capacity);
487  bool update = (new_capacity < old_capacity) || (m_p != old_p);
488  if (update){
489  _updateReferences();
490  }
491  }

494  virtual void _internalReallocate(Int64 new_capacity,FalseType)
495  {
496  TrueImpl* old_p = m_p;
497  Int64 old_size = m_p->size;
498  _directAllocate(new_capacity);
499  if (m_p!=old_p){
500  for( Int64 i=0; i<old_size; ++i ){
501  new (m_p->ptr+i) T(old_p->ptr[i]);
502  old_p->ptr[i].~T();
503  }

505  ArrayImplBase::deallocate(old_p);
506  _updateReferences();
507  }
508  }

514  }
