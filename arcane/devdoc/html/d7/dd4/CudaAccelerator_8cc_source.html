58class CudaMemoryAllocatorBase
59: public Arccore::AlignedMemoryAllocator3
60{
61 public:
62
63 CudaMemoryAllocatorBase()
64 : AlignedMemoryAllocator3(128)
65 {}
66
67 bool hasRealloc(MemoryAllocationArgs) const final { return true; }
68 AllocatedMemoryInfo allocate(MemoryAllocationArgs args, Int64 new_size) final
69 {
70 void* out = nullptr;
71 ARCANE_CHECK_CUDA(_allocate(&out, new_size, args));
72 Int64 a = reinterpret_cast<Int64>(out);
73 if ((a % 128) != 0)
74 ARCANE_FATAL("Bad alignment for CUDA allocator: offset={0}", (a % 128));
75 return { out, new_size };
76 }
77 AllocatedMemoryInfo reallocate(MemoryAllocationArgs args, AllocatedMemoryInfo current_ptr, Int64 new_size) final
78 {
79 AllocatedMemoryInfo a = allocate(args, new_size);
80 ARCANE_CHECK_CUDA(cudaMemcpy(a.baseAddress(), current_ptr.baseAddress(), current_ptr.size(), cudaMemcpyDefault));
81 deallocate(args, current_ptr);
82 return a;
83 }
84 void deallocate(MemoryAllocationArgs args, AllocatedMemoryInfo mem_info) final
85 {
86 ARCANE_CHECK_CUDA_NOTHROW(_deallocate(mem_info, args));
87 }
88
89 protected:
90
91 virtual cudaError_t _allocate(void** ptr, size_t new_size, MemoryAllocationArgs args) = 0;
92 virtual cudaError_t _deallocate(AllocatedMemoryInfo mem_info, MemoryAllocationArgs args) = 0;
93};

98class UnifiedMemoryCudaMemoryAllocator
99: public CudaMemoryAllocatorBase
100{
101 public:
102
103 Int64 page_size = platform::getPageSize();
104
105 void initialize()
106 {
107 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_CUDA_UM_PAGE_ALLOC", true))
108 m_page_allocate_level = v.value();
109 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_CUDA_MALLOC_TRACE", true))
110 m_trace_level = v.value();
111 }
112
113 public:
114
115 void notifyMemoryArgsChanged([[maybe_unused]] MemoryAllocationArgs old_args,
116 MemoryAllocationArgs new_args, AllocatedMemoryInfo ptr) override
117 {
118 void* p = ptr.baseAddress();
119 Int64 s = ptr.capacity();
120 if (p && s > 0)
121 _applyHint(ptr.baseAddress(), ptr.size(), new_args);
122 }
123
124 Int64 adjustedCapacity(MemoryAllocationArgs args, Int64 wanted_capacity, Int64 element_size) const override
125 {
126 wanted_capacity = AlignedMemoryAllocator3::adjustedCapacity(args, wanted_capacity, element_size);
127 const bool do_page = m_page_allocate_level > 0;
128 if (!do_page)
129 return wanted_capacity;
130 // Alloue un multiple de la taille d'une page
131 // Comme les transfers de la mémoire unifiée se font par page,
132 // cela permet de détecter qu'elle allocation provoque le transfert
133 // TODO: vérifier que le début de l'allocation est bien un multiple
134 // de la taille de page.
135 Int64 orig_capacity = wanted_capacity;
136 Int64 new_size = orig_capacity * element_size;
137 size_t n = new_size / page_size;
138 if ((new_size % page_size) != 0)
139 ++n;
140 new_size = (n + 1) * page_size;
141 wanted_capacity = new_size / element_size;
142 if (wanted_capacity < orig_capacity)
143 wanted_capacity = orig_capacity;
144 return wanted_capacity;
145 }
146
147 protected:
148
149 cudaError_t _deallocate(AllocatedMemoryInfo mem_info, MemoryAllocationArgs args) override
150 {
151 void* ptr = mem_info.baseAddress();
152 const bool do_trace = m_trace_level > 0;
153 if (do_trace) {
154 // Utilise un flux spécifique pour être sur que les affichages ne seront pas mélangés
155 // en cas de multi-threading
156 std::ostringstream ostr;
157 if (m_trace_level >= 2)
158 ostr << "FREE_MANAGED=" << ptr << " size=" << mem_info.capacity() << " name=" << args.arrayName();
159 String s;
160 if (m_trace_level >= 3) {
161 s = platform::getStackTrace();
162 if (m_trace_level >= 4) {
163 ostr << " stack=" << s;
164 }
165 }
166 impl::MemoryTracer::notifyMemoryFree(ptr, args.arrayName(), s, 0);
167 if (m_trace_level >= 2) {
168 ostr << "\n";
169 std::cout << ostr.str();
170 }
171 }
172
173 return ::cudaFree(ptr);
174 }
175
176 cudaError_t _allocate(void** ptr, size_t new_size, MemoryAllocationArgs args) override
177 {
178 auto r = ::cudaMallocManaged(ptr, new_size, cudaMemAttachGlobal);
179 void* p = *ptr;
180 if (r != cudaSuccess)
181 return r;
182
183 const bool do_trace = m_trace_level > 0;
184 if (do_trace) {
185 // Utilise un flux spécifique pour être sur que les affichages ne seront pas mélangés
186 // en cas de multi-threading
187 std::ostringstream ostr;
188 if (m_trace_level >= 2)
189 ostr << "MALLOC_MANAGED=" << p << " size=" << new_size << " name=" << args.arrayName();
190 String s;
191 if (m_trace_level >= 3) {
192 s = platform::getStackTrace();
193 if (m_trace_level >= 4) {
194 ostr << " stack=" << s;
195 }
196 }
197 Span<const std::byte> bytes(reinterpret_cast<std::byte*>(p), new_size);
198 impl::MemoryTracer::notifyMemoryAllocation(bytes, args.arrayName(), s, 0);
199 if (m_trace_level >= 2) {
200 ostr << "\n";
201 std::cout << ostr.str();
202 }
203 }
204
205 _applyHint(*ptr, new_size, args);
206
207 return r;
208 }
209
210 void _applyHint(void* p, size_t new_size, MemoryAllocationArgs args)
211 {
212 // TODO: regarder comment utiliser une autre device que le device 0.
213 // (Peut-être prendre cudaGetDevice ?)
214 eMemoryLocationHint hint = args.memoryLocationHint();
215 //std::cout << "SET_MEMORY_HINT name=" << args.arrayName() << " size=" << new_size << " hint=" << (int)hint << "\n";
216 if (hint == eMemoryLocationHint::MainlyDevice || hint == eMemoryLocationHint::HostAndDeviceMostlyRead) {
217 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetPreferredLocation, 0));
218 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetAccessedBy, cudaCpuDeviceId));
219 }
220 if (hint == eMemoryLocationHint::MainlyHost) {
221 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetPreferredLocation, cudaCpuDeviceId));
222 //ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetAccessedBy, 0));
223 }
224 if (hint == eMemoryLocationHint::HostAndDeviceMostlyRead) {
225 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetReadMostly, 0));
226 }
227 }
228
229 private:
230
232 Int32 m_page_allocate_level = 0;
233
234 Int32 m_trace_level = 0;
235};
