78template <typename ConcreteAllocatorType>
79class UnderlyingAllocator
80: public AcceleratorMemoryAllocatorBase::IUnderlyingAllocator
81{
82 public:
83
84 UnderlyingAllocator() = default;
85
86 public:
87
88 void* allocateMemory(size_t size) final
89 {
90 void* out = nullptr;
91 ARCANE_CHECK_CUDA(m_concrete_allocator._allocate(&out, size));
92 return out;
93 }
94 void freeMemory(void* ptr, [[maybe_unused]] size_t size) final
95 {
96 ARCANE_CHECK_CUDA_NOTHROW(m_concrete_allocator._deallocate(ptr));
97 }
98
99 void doMemoryCopy(void* destination, const void* source, Int64 size) final
100 {
101 ARCANE_CHECK_CUDA(cudaMemcpy(destination, source, size, cudaMemcpyDefault));
102 }
103
104 eMemoryResource memoryResource() const final
105 {
106 return m_concrete_allocator.memoryResource();
107 }
108
109 public:
110
111 ConcreteAllocatorType m_concrete_allocator;
112};

117class UnifiedMemoryConcreteAllocator
118: public ConcreteAllocator
119{
120 public:
121
122 UnifiedMemoryConcreteAllocator()
123 {
124 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_CUDA_USE_ALLOC_ATS", true))
125 m_use_ats = v.value();
126 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_CUDA_MEMORY_HINT_ON_DEVICE", true))
127 m_use_hint_as_mainly_device = (v.value() != 0);
128 }
129
130 cudaError_t _deallocate(void* ptr) final
131 {
132 if (m_use_ats) {
133 ::free(ptr);
134 return cudaSuccess;
135 }
136 //std::cout << "CUDA_MANAGED_FREE ptr=" << ptr << "\n";
137 return ::cudaFree(ptr);
138 }
139
140 cudaError_t _allocate(void** ptr, size_t new_size) final
141 {
142 if (m_use_ats) {
143 *ptr = ::aligned_alloc(128, new_size);
144 }
145 else {
146 auto r = ::cudaMallocManaged(ptr, new_size, cudaMemAttachGlobal);
147 //std::cout << "CUDA_MANAGED_MALLOC ptr=" << (*ptr) << " size=" << new_size << "\n";
148 //if (new_size < 4000)
149 //std::cout << "STACK=" << platform::getStackTrace() << "\n";
150
151 if (r != cudaSuccess)
152 return r;
153
154 // Si demandé, indique qu'on préfère allouer sur le GPU.
155 // NOTE: Dans ce cas, on récupère le device actuel pour positionner la localisation
156 // préférée. Dans le cas où on utilise MemoryPool, cette allocation ne sera effectuée
157 // qu'une seule fois. Si le device par défaut pour un thread change au cours du calcul
158 // il y aura une incohérence. Pour éviter cela, on pourrait faire un cudaMemAdvise()
159 // pour chaque allocation (via _applyHint()) mais ces opérations sont assez couteuses
160 // et s'il y a beaucoup d'allocation il peut en résulter une perte de performance.
161 if (m_use_hint_as_mainly_device) {
162 int device_id = 0;
163 void* p = *ptr;
164 cudaGetDevice(&device_id);
165 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetPreferredLocation, _getMemoryLocation(device_id)));
166 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetAccessedBy, _getMemoryLocation(cudaCpuDeviceId)));
167 }
168 }
169
170 return cudaSuccess;
171 }
172
173 constexpr eMemoryResource memoryResource() const { return eMemoryResource::UnifiedMemory; }
174
175 public:
176
177 bool m_use_ats = false;
179 bool m_use_hint_as_mainly_device = false;
180};

191class UnifiedMemoryCudaMemoryAllocator
192: public AcceleratorMemoryAllocatorBase
193{
194 public:
195 public:
196
197 UnifiedMemoryCudaMemoryAllocator()
198 : AcceleratorMemoryAllocatorBase("UnifiedMemoryCudaMemory", new UnderlyingAllocator<UnifiedMemoryConcreteAllocator>())
199 {
200 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_CUDA_MALLOC_TRACE", true))
201 _setTraceLevel(v.value());
202 }
203
204 void initialize()
205 {
206 _doInitializeUVM();
207 }
208
209 public:
210
211 void notifyMemoryArgsChanged([[maybe_unused]] MemoryAllocationArgs old_args,
212 MemoryAllocationArgs new_args, AllocatedMemoryInfo ptr) final
213 {
214 void* p = ptr.baseAddress();
215 Int64 s = ptr.capacity();
216 if (p && s > 0)
217 _applyHint(ptr.baseAddress(), ptr.size(), new_args);
218 }
219
220 protected:
221
222 void _applyHint(void* p, size_t new_size, MemoryAllocationArgs args)
223 {
224 eMemoryLocationHint hint = args.memoryLocationHint();
225 // Utilise le device actif pour positionner le GPU par défaut
226 // On ne le fait que si le \a hint le nécessite pour éviter d'appeler
227 // cudaGetDevice() à chaque fois.
228 int device_id = 0;
229 if (hint == eMemoryLocationHint::MainlyDevice || hint == eMemoryLocationHint::HostAndDeviceMostlyRead) {
230 cudaGetDevice(&device_id);
231 }
232 auto device_memory_location = _getMemoryLocation(device_id);
233 auto cpu_memory_location = _getMemoryLocation(cudaCpuDeviceId);
234
235 //std::cout << "SET_MEMORY_HINT name=" << args.arrayName() << " size=" << new_size << " hint=" << (int)hint << "\n";
236 if (hint == eMemoryLocationHint::MainlyDevice || hint == eMemoryLocationHint::HostAndDeviceMostlyRead) {
237 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetPreferredLocation, device_memory_location));
238 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetAccessedBy, cpu_memory_location));
239 }
240 if (hint == eMemoryLocationHint::MainlyHost) {
241 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetPreferredLocation, cpu_memory_location));
242 //ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetAccessedBy, 0));
243 }
244 if (hint == eMemoryLocationHint::HostAndDeviceMostlyRead) {
245 ARCANE_CHECK_CUDA(cudaMemAdvise(p, new_size, cudaMemAdviseSetReadMostly, device_memory_location));
246 }
247 }
248 void _removeHint(void* p, size_t size, MemoryAllocationArgs args)
249 {
250 eMemoryLocationHint hint = args.memoryLocationHint();
251 if (hint == eMemoryLocationHint::None)
252 return;
253 int device_id = 0;
254 ARCANE_CHECK_CUDA(cudaMemAdvise(p, size, cudaMemAdviseUnsetReadMostly, _getMemoryLocation(device_id)));
255 }
256
257 private:
258
259 bool m_use_ats = false;
260};

306class DeviceConcreteAllocator
307: public ConcreteAllocator
308{
309 public:
310
311 DeviceConcreteAllocator()
312 {
313 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_CUDA_USE_ALLOC_ATS", true))
314 m_use_ats = v.value();
315 }
316
317 cudaError_t _allocate(void** ptr, size_t new_size) final
318 {
319 if (m_use_ats) {
320 // FIXME: it does not work on WIN32
321 *ptr = std::aligned_alloc(128, new_size);
322 if (*ptr)
323 return cudaSuccess;
324 return cudaErrorMemoryAllocation;
325 }
326 cudaError_t r = ::cudaMalloc(ptr, new_size);
327 //std::cout << "ALLOCATE_DEVICE ptr=" << (*ptr) << " size=" << new_size << " r=" << (int)r << "\n";
328 return r;
329 }
330 cudaError_t _deallocate(void* ptr) final
331 {
332 if (m_use_ats) {
333 std::free(ptr);
334 return cudaSuccess;
335 }
336 //std::cout << "FREE_DEVICE ptr=" << ptr << "\n";
337 return ::cudaFree(ptr);
338 }
339
340 constexpr eMemoryResource memoryResource() const { return eMemoryResource::Device; }
341
342 private:
343
344 bool m_use_ats = false;
345};
