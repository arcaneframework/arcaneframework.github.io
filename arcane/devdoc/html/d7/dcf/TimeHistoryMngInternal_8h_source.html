178class TimeHistoryValueT

181 /*
182 * ATTENTION CE QUI EST DECRIT DANS CE COMMENTAIRE N'EST PAS ENCORE OPERATIONNEL
183 * Lorsqu'il y a beaucoup de courbes et que le nombre d'itérations
184 * devient important, le stockage consomme de la mémoire. Pour éviter
185 * cela, il est possible de compresser le tableau des itérations.
186 * Si c'est le cas et que les itérations sont consécutives, on
187 * conserve uniquement la première et la dernière. Dans ce cas,
188 * m_iterations à 3 valeurs: [0] = COMPRESSED_TAG, [1] = première
189 * et [2] = dernière.
190 */
191 public:
192
193 typedef VariableRefArrayT<DataType> ValueList;
194 typedef VariableRefArrayT<Int32> IterationList;
195 static const Integer COMPRESSED_TAG = -15;
196
197 public:

213 : TimeHistoryValue(thpi, DataTypeTraitsT<DataType>::type(), index, nb_element)
214 , m_values(VariableBuildInfo(vm, String("TimeHistoryMngValues") + index, VAR_BUILD_FLAGS))
215 , m_iterations(VariableBuildInfo(vm, String("TimeHistoryMngIterations") + index, VAR_BUILD_FLAGS))

239
240 void fromOldToNewVariables(IVariableMng* vm, IMesh* default_mesh) override
241 {
242 IVariable* ptr_old_values = vm->findMeshVariable(default_mesh, String("TimeHistory_Values_") + index());
243 IVariable* ptr_old_iterations = vm->findMeshVariable(default_mesh, String("TimeHistory_Iterations_") + index());
244 if (ptr_old_values == nullptr || ptr_old_iterations == nullptr)
245 ARCANE_FATAL("Unknown old variable");
246
247 ValueList old_values(ptr_old_values);
248 IterationList old_iterations(ptr_old_iterations);
249
250 m_values.resize(old_values.size());
251 m_values.copy(old_values);
252
253 m_iterations.resize(old_iterations.size());
254 m_iterations.copy(old_iterations);
255
256 old_values.resize(0);
257 old_iterations.resize(0);
258 }
259
260 Integer size() const override
261 {
262 return m_iterations.size();
263 }
264
271 void addValue(ConstArrayView<DataType> values, Integer iteration)
272 {
273 Integer nb_iteration = m_iterations.size();
274 Integer nb_value = m_values.size();
275 Integer sub_size = values.size();
276 if (nb_iteration != 0)
277 if (m_iterations[nb_iteration - 1] == iteration) {
278 // Remplace la valeur
279 for (Integer i = 0; i < sub_size; ++i)
280 m_values[nb_value - sub_size + i] = values[i];
281 return;
282 }
283 Integer add_nb_iter = math::max(128, nb_iteration / 20);
284 Integer add_nb_value = math::max(1024, nb_value / 20);
285 m_iterations.resizeWithReserve(nb_iteration + 1, add_nb_iter);
286 m_values.resizeWithReserve(nb_value + sub_size, add_nb_value);
287 m_iterations[nb_iteration] = iteration;
288 for (Integer i = 0; i < sub_size; ++i)
289 m_values[nb_value + i] = values[i];
290 }
291
292 void removeAfterIteration(Integer last_iteration) override
293 {
294 Integer size = m_iterations.size();
295 Integer last_elem = size;
296 for (Integer i = 0; i < size; ++i)
297 if (m_iterations[i] >= last_iteration) {
298 last_elem = i;
299 break;
300 }
301 if (last_elem != size) {
302 m_iterations.resize(last_elem);
303 m_values.resize(last_elem * subSize());
304 }
305 }

307 // Ecriture d'une courbe pour les écrivains version 2.
308 void dumpValues(ITraceMng* msg,
309 ITimeHistoryCurveWriter2* writer,
310 const TimeHistoryCurveWriterInfo& infos) const override
311 {
312 ARCANE_UNUSED(msg);
313
314 // Pour l'instant, on ne fait rien
315 if (m_shrink_history)
316 return;
317
318 UniqueArray<Real> values_to_write;
319 UniqueArray<Int32> iterations_to_write;
320
321 arrayToWrite(iterations_to_write, values_to_write, infos);
322
323 Integer sd = localProcId();
324 if (!meshHandle().isNull()) {
325 TimeHistoryCurveInfo curve_info(name(), meshHandle().meshName(), iterations_to_write, values_to_write, subSize(), sd);
326 writer->writeCurve(curve_info);
327 }
328 else {
329 TimeHistoryCurveInfo curve_info(name(), iterations_to_write, values_to_write, subSize(), sd);
330 writer->writeCurve(curve_info);
331 }
332 }
333
334 void applyTransformation(ITraceMng* msg, ITimeHistoryTransformer* v) override
335 {
336 ITimeHistoryTransformer::CommonInfo ci;
337 ci.name = name();
338 SharedArray<Int32> iterations(m_iterations.asArray());
339 ci.iterations = iterations;
340 Integer sub_size = subSize();
341 ci.sub_size = subSize();
342
343 SharedArray<DataType> values(m_values.asArray());
344
345 v->transform(ci, values);
346
347 Integer nb_iteration = iterations.size();
348 Integer nb_value = values.size();
349 if (nb_iteration * sub_size != nb_value) {
350 msg->warning() << "Bad size after history transformation";
351 return;
352 }
353
354 m_iterations.resize(nb_iteration);
355 for (Integer i = 0; i < nb_iteration; ++i)
356 m_iterations[i] = iterations[i];
357
358 m_values.resize(nb_value);
359 for (Integer i = 0; i < nb_value; ++i)
360 m_values[i] = values[i];
361 }
362
363 void arrayToWrite(UniqueArray<Int32>& iterations, UniqueArray<Real>& values, const TimeHistoryCurveWriterInfo& infos) const override
364 {
365 // Pour vérifier qu'on ne sauve pas plus d'itérations qu'il y en
366 // a actuellement (ce qui peut arriver en cas de retour arrière).
367 Integer max_iter = infos.times().size();
368 Integer nb_iteration = m_iterations.size();
369 Integer sub_size = subSize();
370 iterations.clear();
371 iterations.reserve(nb_iteration);
372 values.clear();
373 values.reserve(nb_iteration * sub_size);
374 for (Integer i = 0, is = nb_iteration; i < is; ++i) {
375 Integer iter = m_iterations[i];
376 if (iter < max_iter) {
377 for (Integer z = 0; z < sub_size; ++z) {
378 values.add(Convert::toReal(m_values[(i * sub_size) + z]));
379 }
380 iterations.add(iter);
381 }
382 }
383 }
384
385 const ValueList& values() const { return m_values; }
386 const IterationList& iterations() const { return m_iterations; }
387
388 private:
389
390 ValueList m_values;
391 IterationList m_iterations;
392 bool m_use_compression;
393 bool m_shrink_history;
394};
