58 // deterministic polar method, uses trigonometric functions
59 template<class UniformRandomNumberGenerator>
60 class NormalDistribution
61 {
62  public:
63 
64  typedef UniformRandomNumberGenerator base_type;
65 
66  explicit NormalDistribution(base_type & rng,Real mean = 0.,Real sigma = 1.)
67  : _rng(rng), _mean(mean), _sigma(sigma), _valid(false)
68  {
69  }
70 
71  // compiler-generated copy constructor is NOT fine, need to purge cache
72  NormalDistribution(const NormalDistribution& other)
73  : _rng(other._rng), _mean(other._mean), _sigma(other._sigma), _valid(false)
74  {
75  }
76  // uniform_01 cannot be assigned, neither can this class
77 
78  Real operator()()
79  {
80  if(!_valid) {
81  _r1 = _rng();
82  _r2 = _rng();
83  _cached_rho = math::sqrt(-2 * math::log(1.0-_r2));
84  _valid = true;
85  } else {
86  _valid = false;
87  }
88  // Can we have a boost::mathconst please?
89  const double pi = 3.14159265358979323846;
90 
91  return _cached_rho * (_valid ? cos(2*pi*_r1) : sin(2*pi*_r1)) * _sigma +
92  _mean;
93  }
94  private:
95  Uniform01<base_type> _rng;
96  const Real _mean, _sigma;
97  Real _r1, _r2, _cached_rho;
98  bool _valid;
99 };
