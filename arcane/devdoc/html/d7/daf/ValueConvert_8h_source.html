67template <class T> inline bool
68builtInGetValueGeneric(T& v, StringView s)
69{
70 T read_val = T();
71 impl::StringViewInputStream svis(s);
72 std::istream& sbuf = svis.stream();
73 sbuf >> read_val;
74 if (sbuf.fail() || sbuf.bad())
75 return true;
76 if (!sbuf.eof())
77 return true;
78 v = read_val;
79 return false;
80}

189inline bool
190builtInGetValue(bool& v, StringView s)
191{
192 if (s.empty())
193 return true;
194 if (s == "false" || s == "faux" || s == "0") {
195 v = false;
196 return false;
197 }
198
199 if (s == "true" || s == "vrai" || s == "1") {
200 v = true;
201 return false;
202 }
203 return true;
204}
