145class FaceUniqueIdBuilder2::NarrowCellFaceInfo
146{
147 public:
148 static const Int64 BITS_CELL_UID = 39;
149 static const Int64 BITS_RANK = 20;
150 static const Int64 BITS_INDEX = 5;
151 static const Int64 ONE_INT64 = 1;
152 static const Int64 MASK_CELL_UID = (ONE_INT64 << BITS_CELL_UID) - 1;
153 static const Int64 MASK_RANK = ((ONE_INT64 << BITS_RANK) - 1) << BITS_CELL_UID;
154 static const Int64 MASK_INDEX = ((ONE_INT64 << BITS_INDEX) - 1) << (BITS_CELL_UID+BITS_RANK);
155
156 public:
157 NarrowCellFaceInfo()
158 {
159 setValue(NULL_ITEM_UNIQUE_ID,-1,-1);
160 }
161 public:
162
163 bool isMaxValue() const
164 {
165 Int64 max_id = (MASK_CELL_UID - 1);
166 return cellUid()==max_id;
167 }
168
169 void setMaxValue()
170 {
171 Int64 max_id = (MASK_CELL_UID - 1);
172 setValue(max_id,-1,-1);
173 }
174
175 void setValue(Int64 cell_uid,Int32 _rank,Int32 face_local_index)
176 {
177 Int64 v_fli = face_local_index+1;
178 Int64 v_rank = _rank+1;
179 Int64 v_uid = cell_uid+1;
180 m_value = v_fli << (BITS_CELL_UID+BITS_RANK);
181 m_value += v_rank << (BITS_CELL_UID);
182 m_value += v_uid;
183 if (cellUid()!=cell_uid)
184 ARCANE_FATAL("Bad uid expected='{0}' computed='{1}' v={2}",cell_uid,cellUid(),m_value);
185 if (rank()!=_rank)
186 ARCANE_FATAL("Bad rank expected='{0}' computed='{1}'",_rank,rank());
187 if (faceLocalIndex()!=face_local_index)
188 ARCANE_FATAL("Bad local_index expected='{0}' computed='{1}'",face_local_index,faceLocalIndex());
189 }
190 Int64 cellUid() const { return (m_value & MASK_CELL_UID) - 1; }
191 Int32 rank() const { return CheckedConvert::toInt32( ((m_value & MASK_RANK) >> BITS_CELL_UID) - 1 ); }
192 Int32 faceLocalIndex() const { return CheckedConvert::toInt32( ((m_value & MASK_INDEX) >> (BITS_CELL_UID+BITS_RANK)) - 1 ); }
193
194 bool isValid() const { return cellUid()!=NULL_ITEM_UNIQUE_ID; }
195
196 private:
197
198 Int64 m_value;
199};

512void FaceUniqueIdBuilder2::
513_computeSequential()
514{
515 info() << "Compute FacesUniqueId() Sequential V3";
516
517 //TODO: permettre de ne pas commencer a zero.
518 Int64 face_unique_id_counter = 0;
519
520 ItemInternalMap& cells_map = m_mesh->cellsMap();
521 Integer nb_cell = cells_map.count();
522 UniqueArray<ItemInternal*> cells;
523 cells.reserve(nb_cell);
524 // D'abord, il faut trier les mailles par leur uniqueId()
525 // en ordre croissant
526 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,cells_map){
527 cells.add(iid->value());
528 }
529 std::sort(std::begin(cells),std::end(cells),UniqueIdSorter());
530
531 // Invalide les uid pour être certain qu'ils seront tous positionnés.
532 _unsetFacesUniqueId();
533
534 for( Integer i=0; i<nb_cell; ++i ){
535 Cell cell = cells[i];
536 for( Face face : cell.faces()){
537 if (face.uniqueId()==NULL_ITEM_UNIQUE_ID){
538 face.mutableItemBase().setUniqueId(face_unique_id_counter);
539 ++face_unique_id_counter;
540 }
541 }
542 }

550void FaceUniqueIdBuilder2::
551_computeParallel()
552{
553 IParallelMng* pm = m_parallel_mng;
554 Int32 my_rank = pm->commRank();
555
556 bool is_verbose = m_is_verbose;
557
558 ItemInternalMap& cells_map = m_mesh->cellsMap();
559
560 info() << "Compute FacesUniqueId() V3 using parallel sort";
561
562 // Calcule et trie pour les faces frontières
563 UniqueArray<BoundaryFaceInfo> boundary_faces_info;
564 _computeAndSortBoundaryFaces(boundary_faces_info);
565
566 // Ici, les faces de bord sont triées en fonction de leur noeuds.
567 // Normalement, dans cette liste, 2 éléments consécutifs BoundaryFaceInfo qui
568 // ont les même noeuds représentent la même face. Dans ce cas, on génère
569 // un AnyFaceInfo avec les infos des deux mailles issus de ces deux éléments de
570 // la liste en faisant bien attention de mettre en premier la maille
571 // de plus petit uniqueId().
572 // Si deux éléments consécutifs de la liste n'ont pas les mêmes noeuds, cela
573 // signifie que la face est au bord du domaine global.
574 // Il faut tout de même traiter le cas des deux éléments consécutifs de la liste
575 // qui se trouvent sur des processeurs différents. Pour gérer ce cas, chaque
576 // processeur envoie au suivant le dernier élément de sa liste s'il ne peut pas
577 // être fusionné avec l'avant dernier, dans l'espoir qu'il pourra l'être avec le
578 // premier de la liste du processeur suivant.
579 UniqueArray<AnyFaceInfo> all_face_list;
580 {
581 ConstArrayView<BoundaryFaceInfo> all_fsi = boundary_faces_info;
582 Integer n = all_fsi.size();
583 bool is_last_already_done = false;
584 for( Integer i=0; i<n; ++i ){
585 const BoundaryFaceInfo& fsi = all_fsi[i];
586 Int64 cell_uid0 = fsi.m_cell_uid;
587 bool is_inside = false;
588 //TODO: traiter le cas si la maille d'avant est sur un autre proc
589 // Pour cela, il faut recupérer la dernière valeur du proc précédent
590 // et regarder si elle correspond à notre première valeur
591 is_inside = ((i+1)!=n && fsi.hasSameNodes(all_fsi[i+1]));
592 if (is_last_already_done){
593 is_last_already_done = false;
594 }
595 else{
596 AnyFaceInfo csi;
597 if (is_inside){
598 const BoundaryFaceInfo& next_fsi = all_fsi[i+1];
599 Int64 cell_uid1 = next_fsi.m_cell_uid;
600 if (cell_uid0<cell_uid1){
601 csi.setCell0(cell_uid0,fsi.m_rank,fsi.m_face_local_index);
602 csi.setCell1(cell_uid1,next_fsi.m_rank,next_fsi.m_face_local_index);
603 }
604 else{
605 csi.setCell0(cell_uid1,next_fsi.m_rank,next_fsi.m_face_local_index);
606 csi.setCell1(cell_uid0,fsi.m_rank,fsi.m_face_local_index);
607 }
608 is_last_already_done = true;
609 }
610 else{
611 csi.setCell0(cell_uid0,fsi.m_rank,fsi.m_face_local_index);
612 }
613 all_face_list.add(csi);
614 }
615 if (is_verbose)
616 info() << "FACES_KEY i=" << i
617 << " n0=" << fsi.m_node0_uid
618 << " n1=" << fsi.m_node1_uid
619 << " n2=" << fsi.m_node2_uid
620 << " cell=" << fsi.m_cell_uid
621 << " rank=" << fsi.m_rank
622 << " li=" << fsi.m_face_local_index
623 << " in=" << is_inside;
624 }
625 }
626
627 // Ajoute les faces propres a notre sous-domaine.
628 // Il s'agit de toutes les faces qui ont 2 mailles connectées.
629 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,cells_map){
630 Cell cell(iid->value());
631 Integer cell_nb_face = cell.nbFace();
632 Int64 cell_uid = cell.uniqueId();
633 for( Integer z=0; z<cell_nb_face; ++z ){
634 Face face = cell.face(z);
635 if (face.nbCell()!=2)
636 continue;
637 Cell cell0 = face.cell(0);
638 Cell cell1 = face.cell(1);
639 Cell next_cell = (cell0==cell) ? cell1 : cell0;
640 Int64 next_cell_uid = next_cell.uniqueId();
641 // N'enregistre que si je suis la maille de plus petit uid
642 if (cell_uid<next_cell_uid){
643 AnyFaceInfo csi;
644 csi.m_cell0.setValue(cell_uid,my_rank,z);
645 // Le face_local_index de la maille 1 ne sera pas utilisé
646 csi.m_cell1.setValue(next_cell_uid,my_rank,-1);
647 all_face_list.add(csi);
648 }
649 }
650 }
651
652 if (is_verbose){
653 Integer n = all_face_list.size();
654 for( Integer i=0; i<n; ++i ){
655 const AnyFaceInfo& csi = all_face_list[i];
656 info() << "CELL_TO_SORT i=" << i
657 << " cell0=" << csi.m_cell0.cellUid()
658 << " lidx0=" << csi.m_cell0.faceLocalIndex()
659 << " cell1=" << csi.m_cell1.cellUid();
660 }
661 }
662
663 info() << "ALL_FACE_LIST memorysize=" << sizeof(AnyFaceInfo)*all_face_list.size();
664 Parallel::BitonicSort<AnyFaceInfo,AnyFaceBitonicSortTraits> all_face_sorter(pm);
665 all_face_sorter.setNeedIndexAndRank(false);
666 Real sort_begin_time = platform::getRealTime();
667 all_face_sorter.sort(all_face_list);
668 Real sort_end_time = platform::getRealTime();
669 info() << "END_ALL_FACE_SORTER time=" << (Real)(sort_end_time - sort_begin_time);
670
671 _resendCellsAndComputeFacesUniqueId(all_face_sorter.keys());
672}

680void FaceUniqueIdBuilder2::
681_computeAndSortBoundaryFaces(Array<BoundaryFaceInfo>& boundary_faces_info)
682{
683 IParallelMng* pm = m_parallel_mng;
684 Int32 my_rank = pm->commRank();
685 bool is_verbose = m_is_verbose;
686 ItemInternalMap& faces_map = m_mesh->facesMap();
687
688 Parallel::BitonicSort<BoundaryFaceInfo,BoundaryFaceBitonicSortTraits> boundary_face_sorter(pm);
689 boundary_face_sorter.setNeedIndexAndRank(false);
690
691 //UniqueArray<BoundaryFaceInfo> boundary_face_list;
692 boundary_faces_info.clear();
693 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,faces_map){
694 Face face(iid->value());
695 BoundaryFaceInfo fsi;
696 Integer nb_cell = face.nbCell();
697 if (nb_cell==2)
698 continue;
699
700 fsi.m_rank = my_rank;
701 fsi.setNodes(face);
702 Cell cell = face.cell(0);
703 fsi.m_cell_uid = cell.uniqueId();
704 Integer face_local_index = 0;
705 for( Integer z=0, zs=cell.nbFace(); z<zs; ++z )
706 if (cell.face(z)==face){
707 face_local_index = z;
708 break;
709 }
710 fsi.m_face_local_index = face_local_index;
711 boundary_faces_info.add(fsi);
712 }
713
714 if (is_verbose){
715 ConstArrayView<BoundaryFaceInfo> all_fsi = boundary_faces_info;
716 Integer n = all_fsi.size();
717 for( Integer i=0; i<n; ++i ){
718 const BoundaryFaceInfo& fsi = all_fsi[i];
719 info() << "KEY i=" << i
720 << " n0=" << fsi.m_node0_uid
721 << " n1=" << fsi.m_node1_uid
722 << " n2=" << fsi.m_node2_uid
723 << " cell=" << fsi.m_cell_uid
724 << " rank=" << fsi.m_rank
725 << " li=" << fsi.m_face_local_index;
726 }
727 }
728
729 Real sort_begin_time = platform::getRealTime();
730 boundary_face_sorter.sort(boundary_faces_info);
731 Real sort_end_time = platform::getRealTime();
732 info() << "END_BOUNDARY_FACE_SORT time=" << (Real)(sort_end_time - sort_begin_time);
733
734 {
735 ConstArrayView<BoundaryFaceInfo> all_bfi = boundary_face_sorter.keys();
736 Integer n = all_bfi.size();
737 if (is_verbose){
738 for( Integer i=0; i<n; ++i ){
739 const BoundaryFaceInfo& bfi = all_bfi[i];
740 info() << " AFTER KEY i=" << i
741 << " n0=" << bfi.m_node0_uid
742 << " n1=" << bfi.m_node1_uid
743 << " n2=" << bfi.m_node2_uid
744 << " cell=" << bfi.m_cell_uid
745 << " rank=" << bfi.m_rank
746 << " li=" << bfi.m_face_local_index;
747 }
748 }
749
750 // Comme un même noeud peut être présent dans la liste du proc précédent, chaque PE
751 // (sauf le 0) envoie au proc précédent le début sa liste qui contient les même noeuds.
752
753 // TODO: fusionner ce code avec celui de GhostLayerBuilder2
754 UniqueArray<BoundaryFaceInfo> end_face_list;
755 Integer begin_own_list_index = 0;
756 if (n!=0 && my_rank!=0){
757 if (BoundaryFaceBitonicSortTraits::isValid(all_bfi[0])){
758 Int64 node0_uid = all_bfi[0].m_node0_uid;
759 for( Integer i=0; i<n; ++i ){
760 if (all_bfi[i].m_node0_uid!=node0_uid){
761 begin_own_list_index = i;
762 break;
763 }
764 else
765 end_face_list.add(all_bfi[i]);
766 }
767 }
768 }
769 info() << "BEGIN_OWN_LIST_INDEX=" << begin_own_list_index;
770 if (is_verbose){
771 for( Integer k=0, kn=end_face_list.size(); k<kn; ++k )
772 info() << " SEND n0=" << end_face_list[k].m_node0_uid
773 << " n1=" << end_face_list[k].m_node1_uid
774 << " n2=" << end_face_list[k].m_node2_uid;
775 }
776
777 UniqueArray<BoundaryFaceInfo> end_face_list_recv;
778
779 UniqueArray<Parallel::Request> requests;
780 Integer recv_message_size = 0;
781 Integer send_message_size = BoundaryFaceBitonicSortTraits::messageSize(end_face_list);
782
783 Int32 nb_rank = pm->commSize();
784
785 // Envoie et réceptionne d'abord les tailles.
786 if (my_rank!=(nb_rank-1)){
787 requests.add(pm->recv(IntegerArrayView(1,&recv_message_size),my_rank+1,false));
788 }
789 if (my_rank!=0){
790 requests.add(pm->send(IntegerConstArrayView(1,&send_message_size),my_rank-1,false));
791 }
792
793 pm->waitAllRequests(requests);
794 requests.clear();
795
796 if (recv_message_size!=0){
797 Integer message_size = CheckedConvert::toInteger(recv_message_size/sizeof(BoundaryFaceInfo));
798 end_face_list_recv.resize(message_size);
799 requests.add(BoundaryFaceBitonicSortTraits::recv(pm,my_rank+1,end_face_list_recv));
800 }
801 if (send_message_size!=0)
802 requests.add(BoundaryFaceBitonicSortTraits::send(pm,my_rank-1,end_face_list));
803
804 pm->waitAllRequests(requests);
805
806 boundary_faces_info.clear();
807 boundary_faces_info.addRange(all_bfi.subConstView(begin_own_list_index,n-begin_own_list_index));
808 boundary_faces_info.addRange(end_face_list_recv);
809 }

816void FaceUniqueIdBuilder2::
817_resendCellsAndComputeFacesUniqueId(ConstArrayView<AnyFaceInfo> all_csi)
818{
819 IParallelMng* pm = m_parallel_mng;
820 Int32 nb_rank = pm->commSize();
821 Int32 my_rank = pm->commRank();
822 bool is_verbose = m_is_verbose;
823
824 ItemInternalMap& cells_map = m_mesh->cellsMap();
825
826 Int64 nb_computed_face = all_csi.size();
827
828 if (is_verbose){
829 for( Integer i=0; i<nb_computed_face; ++i ){
830 const AnyFaceInfo& csi = all_csi[i];
831 info() << "CELLS_KEY i=" << i
832 << " cell0=" << csi.m_cell0.cellUid()
833 << " lidx0=" << csi.m_cell0.faceLocalIndex()
834 << " cell1=" << csi.m_cell1.cellUid()
835 << " lidx1=" << csi.m_cell1.faceLocalIndex()
836 << " rank0=" << csi.m_cell0.rank()
837 << " rank1=" << csi.m_cell1.rank();
838 }
839 }
840
841 // Calcul pour chaque rang le nombre de valeurs à envoyer
842 // et le stocke dans nb_info_to_send;
843 IntegerUniqueArray nb_info_to_send(nb_rank,0);
844 {
845 for( Integer i=0; i<nb_computed_face; ++i ){
846 const AnyFaceInfo& csi = all_csi[i];
847 Int32 rank0 = csi.m_cell0.rank();
848 Int32 rank1 = csi.m_cell1.rank();
849
850 ++nb_info_to_send[rank0];
851
852 // Il ne faut l'envoyer que si le rang est différent de m_rank0
853 if (csi.m_cell1.isValid() && rank1!=rank0)
854 ++nb_info_to_send[rank1];
855 }
856 }
857
858 // Tableau pour chaque proc indiquant le uniqueId() de la première
859 // face de ce proc.
860 Int64UniqueArray all_first_face_uid(nb_rank);
861 {
862 // Chaque proc récupère le nombre de mailles dans la liste.
863 // Comme cette liste sera triée, cela correspond aux uid de la première
864 // face de ce proc.
865 Int64 nb_cell_to_sort = all_csi.size();
866 pm->allGather(Int64ConstArrayView(1,&nb_cell_to_sort),all_first_face_uid);
867
868 Int64 to_add = 0;
869 for( Integer i=0; i<nb_rank; ++i ){
870 Int64 next = all_first_face_uid[i];
871 all_first_face_uid[i] = to_add;
872 to_add += next;
873 }
874 }
875
876 Integer total_nb_to_send = 0;
877 IntegerUniqueArray nb_info_to_send_indexes(nb_rank,0);
878 for( Integer i=0; i<nb_rank; ++i ){
879 nb_info_to_send_indexes[i] = total_nb_to_send;
880 total_nb_to_send += nb_info_to_send[i];
881 }
882 info() << "TOTAL_NB_TO_SEND=" << total_nb_to_send;
883
884 UniqueArray<ResendCellInfo> resend_infos(total_nb_to_send);
885 {
886 for( Integer i=0; i<nb_computed_face; ++i ){
887 const AnyFaceInfo& csi = all_csi[i];
888 Int32 rank0 = csi.m_cell0.rank();
889 Int32 rank1 = csi.m_cell1.rank();
890
891 ResendCellInfo& rci0 = resend_infos[nb_info_to_send_indexes[rank0]];
892 rci0.m_cell_uid = csi.m_cell0.cellUid();
893 rci0.m_face_local_index_and_owner_rank = (csi.m_cell0.faceLocalIndex() * nb_rank) + rank0;
894 rci0.m_index_in_rank_list = i;
895 ++nb_info_to_send_indexes[rank0];
896
897 if (csi.m_cell1.isValid() && rank1!=rank0){
898 ResendCellInfo& rci1 = resend_infos[nb_info_to_send_indexes[rank1]];
899 rci1.m_cell_uid = csi.m_cell1.cellUid();
900 // Meme si je suis la maille 1, le proprietaire de la face sera la maille 0.
901 rci1.m_face_local_index_and_owner_rank = (csi.m_cell1.faceLocalIndex() * nb_rank) + rank0;
902 rci1.m_index_in_rank_list = i;
903 ++nb_info_to_send_indexes[rank1];
904 }
905
906 }
907 }
908
909 // Faire un seul reduce
910 Int64 total_nb_computed_face = pm->reduce(Parallel::ReduceSum,nb_computed_face);
911 info() << "TOTAL_NB_COMPUTED_FACE=" << total_nb_computed_face;
912
913 // Indique a chaque PE combien d'infos je vais lui envoyer
914 if (is_verbose)
915 for( Integer i=0; i<nb_rank; ++i )
916 info() << "NB_TO_SEND: I=" << i << " n=" << nb_info_to_send[i];
917
918 IntegerUniqueArray nb_info_to_recv(nb_rank,0);
919 {
920 Timer::SimplePrinter sp(traceMng(),"Sending size with AllToAll");
921 pm->allToAll(nb_info_to_send,nb_info_to_recv,1);
922 }
923
924 if (is_verbose)
925 for( Integer i=0; i<nb_rank; ++i )
926 info() << "NB_TO_RECV: I=" << i << " n=" << nb_info_to_recv[i];
927
928 Integer total_nb_to_recv = 0;
929 for( Integer i=0; i<nb_rank; ++i )
930 total_nb_to_recv += nb_info_to_recv[i];
931
932 // Il y a de fortes chances que cela ne marche pas si le tableau est trop grand,
933 // il faut proceder avec des tableaux qui ne depassent pas 2Go a cause des
934 // Int32 de MPI.
935 // TODO: Faire le AllToAll en plusieurs fois si besoin.
936 UniqueArray<ResendCellInfo> recv_infos;
937 {
938 Int32 vsize = sizeof(ResendCellInfo) / sizeof(Int64);
939 Int32UniqueArray send_counts(nb_rank);
940 Int32UniqueArray send_indexes(nb_rank);
941 Int32UniqueArray recv_counts(nb_rank);
942 Int32UniqueArray recv_indexes(nb_rank);
943 Int32 total_send = 0;
944 Int32 total_recv = 0;
945 for( Integer i=0; i<nb_rank; ++i ){
946 send_counts[i] = (Int32)(nb_info_to_send[i] * vsize);
947 recv_counts[i] = (Int32)(nb_info_to_recv[i] * vsize);
948 send_indexes[i] = total_send;
949 recv_indexes[i] = total_recv;
950 total_send += send_counts[i];
951 total_recv += recv_counts[i];
952 }
953 recv_infos.resize(total_nb_to_recv);
954
955 Int64ConstArrayView send_buf(total_nb_to_send*vsize,(Int64*)resend_infos.data());
956 Int64ArrayView recv_buf(total_nb_to_recv*vsize,(Int64*)recv_infos.data());
957
958 info() << "BUF_SIZES: send=" << send_buf.size() << " recv=" << recv_buf.size();
959 {
960 Timer::SimplePrinter sp(traceMng(),"Send values with AllToAll");
961 pm->allToAllVariable(send_buf,send_counts,send_indexes,recv_buf,recv_counts,recv_indexes);
962 }
963 }
964
965 // Invalide les uid pour être certain qu'ils seront tous positionnés.
966 _unsetFacesUniqueId();
967
968 if (is_verbose){
969 Integer index = 0;
970 for( Int32 rank=0; rank<nb_rank; ++rank ){
971 for( Integer z=0, zs=nb_info_to_recv[rank]; z<zs; ++z ){
972 const ResendCellInfo& rci = recv_infos[index];
973 ++index;
974
975 Int64 cell_uid = rci.m_cell_uid;
976 Int32 full_local_index = rci.m_face_local_index_and_owner_rank;
977 Int32 face_local_index = full_local_index / nb_rank;
978 Int32 owner_rank = full_local_index % nb_rank;
979 Int64 face_uid = all_first_face_uid[rank] + rci.m_index_in_rank_list;
980 info() << "RECV index=" << index << " uid=" << cell_uid
981 << " local_idx=" << full_local_index
982 << " face_local_idx=" << face_local_index
983 << " owner_rank=" << owner_rank
984 << " rank_idx=" << rci.m_index_in_rank_list
985 << " rank="<< rank
986 << " first_face_uid=" << all_first_face_uid[rank]
987 << " computed_uid=" << face_uid;
988 }
989 }
990 }
991
992 // Positionne les uniqueId() et les owner() des faces.
993 {
994 Integer index = 0;
995 for( Int32 i=0; i<nb_rank; ++i ){
996 Int32 rank = i;
997 for( Integer z=0, zs=nb_info_to_recv[i]; z<zs; ++z ){
998 const ResendCellInfo& rci = recv_infos[index];
999 ++index;
1000
1001 Int64 cell_uid = rci.m_cell_uid;
1002 Int32 full_local_index = rci.m_face_local_index_and_owner_rank;
1003 Int32 face_local_index = full_local_index / nb_rank;
1004 Int32 owner_rank = full_local_index % nb_rank;
1005
1006 ItemInternalMapData* cell_data = cells_map.lookup(cell_uid);
1007 if (!cell_data)
1008 ARCANE_FATAL("Can not find cell data for '{0}'",cell_uid);
1009 Cell cell(cell_data->value());
1010 Face face = cell.face(face_local_index);
1011 Int64 face_uid = all_first_face_uid[rank] + rci.m_index_in_rank_list;
1012 face.mutableItemBase().setUniqueId(face_uid);
1013 face.mutableItemBase().setOwner(owner_rank,my_rank);
1014 }
1015 }
1016 }
1017
1018 // Vérifie que toutes les faces ont un uid valide
1019 _checkFacesUniqueId();
1020}
