225void AlephTopologyPETSc::
226_checkInitPETSc()
227{
228 // Ne fait rien si PETSc a déjà été initialisé.
229 PetscBool is_petsc_initialized = PETSC_FALSE;
230 PetscInitialized(&is_petsc_initialized);
231 if (is_petsc_initialized==PETSC_TRUE)
232 return;
233
234 AlephKernelSolverInitializeArguments& init_args = m_kernel->solverInitializeArgs();
235 bool has_args = init_args.hasValues();
236 debug() << Trace::Color::cyan() << "[AlephTopologyPETSc] Initializing PETSc. UseArg=" << has_args;
237 if (has_args){
238 const CommandLineArguments& args = init_args.commandLineArguments();
239 PetscInitialize(args.commandLineArgc(),args.commandLineArgv(),nullptr,nullptr);
240 }
241 else{
242 PetscInitializeNoArguments();
243 }
244}

263// ****************************************************************************
264// * AlephVectorCreate
265// ****************************************************************************
266void AlephVectorPETSc::
267AlephVectorCreate(void)
268{
269 for( int iCpu=0;iCpu<m_kernel->size();++iCpu){
270 if (m_kernel->rank()!=m_kernel->solverRanks(m_index)[iCpu])
271 continue;
272 if (jLower==-1)
273 jLower=m_kernel->topology()->gathered_nb_row(iCpu);
274 jUpper=m_kernel->topology()->gathered_nb_row(iCpu+1);
275 }
276 // Mise à jour de la taille locale du buffer pour le calcul plus tard de la norme max, par exemple
277 jSize=jUpper-jLower;
278 VecCreateMPI(MPI_COMM_SUB,
279 jSize, // n
280 m_kernel->topology()->nb_row_size(), // N
281 &m_petsc_vector);
282 debug()<<"\t\t[AlephVectorPETSc::AlephVectorCreate] PETSC VectorCreate"
283 <<" of local size=" <<jSize<<"/"<<m_kernel->topology()->nb_row_size()
284 <<", done";
285}

360// ****************************************************************************
361// * AlephMatrixCreate
362// ****************************************************************************
363void AlephMatrixPETSc::
364AlephMatrixCreate(void)
365{
366 Integer ilower=-1;
367 Integer iupper=0;
368
369 for( int iCpu=0;iCpu<m_kernel->size();++iCpu ){
370 if (m_kernel->rank()!=m_kernel->solverRanks(m_index)[iCpu]) continue;
371 if (ilower==-1) ilower=m_kernel->topology()->gathered_nb_row(iCpu);
372 iupper=m_kernel->topology()->gathered_nb_row(iCpu+1);
373 }
374 Integer size = iupper-ilower;
375 Integer jlower=ilower;
376 Integer jupper=iupper;
377
378#if PETSC_VERSION_GE(3,3,0)
379 #define PETSC_VERSION_MatCreate MatCreateAIJ
380#elif PETSC_VERSION_(3,0,0)
381 #define PETSC_VERSION_MatCreate MatCreateMPIAIJ
382#endif
383 PETSC_VERSION_MatCreate(MPI_COMM_SUB,
384 iupper-ilower, // m = number of rows
385 jupper-jlower, // n = number of columns
386 m_kernel->topology()->nb_row_size(), // M = number of global rows
387 m_kernel->topology()->nb_row_size(), // N = number of global columns
388 0, // ignored (number of nonzeros per row in DIAGONAL portion of local submatrix)
389 // array containing the number of nonzeros in the various rows of the DIAGONAL portion of local submatrix
390 m_kernel->topology()->gathered_nb_row_elements().subView(ilower,size).unguardedBasePointer(),
391 0, // ignored (number of nonzeros per row in the OFF-DIAGONAL portion of local submatrix)
392 // array containing the number of nonzeros in the various rows of the OFF-DIAGONAL portion of local submatrix
393 m_kernel->topology()->gathered_nb_row_elements().subView(ilower,size).unguardedBasePointer(),
394 &m_petsc_matrix);
395 MatSetOption(m_petsc_matrix, MAT_NEW_NONZERO_LOCATION_ERR, PETSC_FALSE);
396 MatSetUp(m_petsc_matrix);
397 debug()<<"\t\t[AlephMatrixPetsc::AlephMatrixCreate] PETSC MatrixCreate idx:"<<m_index
398 <<", ("<<ilower<<"->"<<(iupper-1)<<")";
399}

475// ****************************************************************************
476// * AlephMatrixSolve
477// ****************************************************************************
478int AlephMatrixPETSc::
479AlephMatrixSolve(AlephVector* x,AlephVector* b,AlephVector* t,
480 Integer& nb_iteration,Real* residual_norm,
481 AlephParams* solver_param)
482{
483 ARCANE_UNUSED(t);
484
485 PC prec;
486 PetscInt its;
487 PetscReal norm;
488 KSPConvergedReason reason;
489
490 AlephVectorPETSc* x_petsc = dynamic_cast<AlephVectorPETSc*> (x->implementation());
491 AlephVectorPETSc* b_petsc = dynamic_cast<AlephVectorPETSc*> (b->implementation());
492
493 ARCANE_CHECK_POINTER(x_petsc);
494 ARCANE_CHECK_POINTER(b_petsc);
495 const bool is_parallel = m_kernel->subParallelMng(m_index)->isParallel();
496
497 Vec solution = x_petsc->m_petsc_vector;
498 Vec RHS = b_petsc->m_petsc_vector;
499
500 debug()<<"[AlephMatrixSolve]";
501 KSPCreate(MPI_COMM_SUB,&m_ksp_solver);
502#if PETSC_VERSION_GE(3,6,1)
503 KSPSetOperators(m_ksp_solver,m_petsc_matrix,m_petsc_matrix);//SAME_PRECONDITIONER);
504#else
505 KSPSetOperators(m_ksp_solver,m_petsc_matrix,m_petsc_matrix,SAME_NONZERO_PATTERN);//SAME_PRECONDITIONER);
506#endif
507 // Builds KSP for a particular solver
508 switch(solver_param->method()){
509 // Preconditioned conjugate gradient (PCG) iterative method
510 case TypesSolver::PCG : KSPSetType(m_ksp_solver,KSPCG); break;
511 // BiCGStab (Stabilized version of BiConjugate Gradient Squared) method
512 case TypesSolver::BiCGStab : KSPSetType(m_ksp_solver,KSPBCGS); break;
513 // IBiCGStab (Improved Stabilized version of BiConjugate Gradient Squared) method
514 // in an alternative form to have only a single global reduction operation instead of the usual 3 (or 4)
515 case TypesSolver::BiCGStab2: KSPSetType(m_ksp_solver,KSPIBCGS); break;
516 // Generalized Minimal Residual method. (Saad and Schultz, 1986) with restart
517 case TypesSolver::GMRES : KSPSetType(m_ksp_solver,KSPGMRES); break;
518 default : throw ArgumentException("AlephMatrixPETSc::AlephMatrixSolve", "Unknown solver method");
519 }
520
521 KSPGetPC(m_ksp_solver,&prec);
522
523 switch(solver_param->precond()){
524 case TypesSolver::NONE : PCSetType(prec,PCNONE); break;
525 // Jacobi (i.e. diagonal scaling preconditioning)
526 case TypesSolver::DIAGONAL : PCSetType(prec,PCJACOBI); break;
527 // Incomplete factorization preconditioners.
528 case TypesSolver::ILU:
529 // ILU ne fonctionne pas en parallèle avec PETSc (sauf si compilé avec Hypre)
530 // ILU can be set up in parallel via a sub-preconditioner to Block-Jacobi preconditioner
531 if (is_parallel)
532 PCSetType(prec,PCBJACOBI);
533 else
534 PCSetType(prec,PCILU);
535 break;
536 // Incomplete Cholesky factorization preconditioners.
537 case TypesSolver::IC:
538 // IC can be set up in parallel via a sub-preconditioner to Block-Jacobi preconditioner
539 if (is_parallel) {
540 PCSetType(prec,PCBJACOBI); // set PC to Block Jacobi
541 PetscInt nlocal, first;
542 KSP *subksp; // KSP context for subdomain
543 PC subpc; // PC context for subdomain
544 KSPSetUp(m_ksp_solver);
545 PCBJacobiGetSubKSP(prec, &nlocal, &first, &subksp); // Extract the KSP context array for the local blocks
546 for (int i = 0; i < nlocal; i++) {
547 KSPGetPC(subksp[i], &subpc);
548 PCSetType(subpc, PCICC); // set subdomain PC to IC
549 }
550 }
551 else
552 PCSetType(prec,PCICC);
553 break;
554 // Sparse Approximate Inverse method of Grote and Barnard as a preconditioner (SIAM J. Sci. Comput.; vol 18, nr 3)
555 case TypesSolver::SPAIstat : PCSetType(prec,PCSPAI); break;
556 // Use multigrid preconditioning.
557 // This preconditioner requires you provide additional information
558 // about the coarser grid matrices and restriction/interpolation operators.
559 case TypesSolver::AMG:
560 // By default PCMG uses GMRES on the fine grid smoother so this should be used with KSPFGMRES
561 // or the smoother changed to not use GMRES
562 // Implements the Flexible Generalized Minimal Residual method. developed by Saad with restart
563 KSPSetType(m_ksp_solver,KSPFGMRES);
564 PCSetType(prec,PCMG);
565 //Sets the number of levels to use with MG. Must be called before any other MG routine.
566 PCMGSetLevels(prec,1,
567 (MPI_Comm*)(m_kernel->subParallelMng(m_index)->getMPICommunicator()));
568 // Determines the form of multigrid to use: multiplicative, additive, full, or the Kaskade algorithm.
569 PCMGSetType(prec,PC_MG_MULTIPLICATIVE); // PC_MG_MULTIPLICATIVE,PC_MG_ADDITIVE,PC_MG_FULL,PC_MG_KASKADE
570 // Sets the type cycles to use. Use PCMGSetCycleTypeOnLevel() for more complicated cycling.
571 PCMGSetCycleType(prec,PC_MG_CYCLE_V); // PC_MG_CYCLE_V or PC_MG_CYCLE_W
572 //Sets the number of pre-smoothing steps to use on all levels
573#if PETSC_VERSION_GE(3,10,2)
574 PCMGSetNumberSmooth(prec, 1);
575#else
576 PCMGSetNumberSmoothDown(prec, 1);
577 PCMGSetNumberSmoothUp(prec, 1);
578#endif
579 // PCMGSetResidual: Sets the function to be used to calculate the residual on the lth level
580 // PCMGSetInterpolation: Sets the function to be used to calculate the interpolation from l-1 to the lth level
581 // PCMGSetRestriction: Sets the function to be used to restrict vector from level l to l-1
582 // PCMGSetRhs: Sets the vector space to be used to store the right-hand side on a particular level
583 // PCMGSetX: Sets the vector space to be used to store the solution on a particular level
584 // PCMGSetR: Sets the vector space to be used to store the residual on a particular level
585 //PCMGSetRestriction, PCMGSetRhs, PCMGSetX, PCMGSetR, etc.
586 //PCMGSetLevels
587 break;
588 case TypesSolver::AINV : throw ArgumentException("AlephMatrixPETSc", "preconditionnement AINV indisponible");
589 case TypesSolver::SPAIdyn : throw ArgumentException("AlephMatrixPETSc", "preconditionnement SPAIdyn indisponible");
590 case TypesSolver::ILUp : throw ArgumentException("AlephMatrixPETSc", "preconditionnement ILUp indisponible");
591 case TypesSolver::POLY : throw ArgumentException("AlephMatrixPETSc", "preconditionnement POLY indisponible");
592 default : throw ArgumentException("AlephMatrixPETSc", "preconditionnement inconnu");
593 }
594
595 if (solver_param->xoUser())
596 KSPSetInitialGuessNonzero(m_ksp_solver,PETSC_TRUE);
597
598 KSPSetTolerances(m_ksp_solver,
599 solver_param->epsilon(),
600 PETSC_DEFAULT,
601 PETSC_DEFAULT,
602 solver_param->maxIter());
603
604 // override KSP options from commandline if present//
605 KSPSetFromOptions(m_ksp_solver);
606
607 // override PC options from commandline if present//
608 PCSetFromOptions(prec) ;
609
610 // setup KSP solver and precondtioner //
611 KSPSetUp(m_ksp_solver);
612 PCSetUp(prec);
613
614 // print info which KSP solver and PC used //
615 KSPType kt;
616 KSPGetType(m_ksp_solver,&kt);
617 PCType pt;
618 PCGetType(prec,&pt);
619 info(4) << "[AlephMatrixSolve] Will use KSP type :" << kt << " PC type : "<< pt
620 << " is_parallel=" << is_parallel;
621
622 // solve the linear system //
623 debug()<<"[AlephMatrixSolve] All set up, now solving";
624 petscCheck(KSPSolve(m_ksp_solver,RHS,solution));
625 debug()<<"[AlephMatrixSolve] solved";
626 petscCheck(KSPGetConvergedReason(m_ksp_solver,&reason));
627
628 switch(reason){
629#if !PETSC_VERSION_(3,0,0)
630 case(KSP_CONVERGED_RTOL_NORMAL):{break;}
631 case(KSP_CONVERGED_ATOL_NORMAL):{break;}
632#endif
633 case(KSP_CONVERGED_RTOL):{break;}
634 case(KSP_CONVERGED_ATOL):{break;}
635 case(KSP_CONVERGED_ITS):{break;}
636#if !PETSC_VERSION_GE(3,19,0)
637 case(KSP_CONVERGED_CG_NEG_CURVE):{break;}
638 case(KSP_CONVERGED_CG_CONSTRAINED):{break;}
639#endif
640 case(KSP_CONVERGED_STEP_LENGTH):{break;}
641 case(KSP_CONVERGED_HAPPY_BREAKDOWN):{break;}
642 /* diverged */
643 case(KSP_DIVERGED_NULL):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_NULL");}
644 case(KSP_DIVERGED_ITS):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_ITS");}
645 case(KSP_DIVERGED_DTOL):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_DTOL");}
