39 template <typename InputStream, typename Encoding = UTF8<> >
40 class CursorStreamWrapper : public GenericStreamWrapper<InputStream, Encoding> {
41 public:
42  typedef typename Encoding::Ch Ch;
43 
44  CursorStreamWrapper(InputStream& is):
45  GenericStreamWrapper<InputStream, Encoding>(is), line_(1), col_(0) {}
46 
47  // counting line and column number
48  Ch Take() {
49  Ch ch = this->is_.Take();
50  if(ch == '\n') {
51  line_ ++;
52  col_ = 0;
53  } else {
54  col_ ++;
55  }
56  return ch;
57  }
58 
60  size_t GetLine() const { return line_; }
62  size_t GetColumn() const { return col_; }
63 
64 private:
