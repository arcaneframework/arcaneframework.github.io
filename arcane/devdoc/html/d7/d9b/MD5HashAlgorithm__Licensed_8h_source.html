77 /* Initialize structure containing state of computation.
78  (RFC 1321, 3.3: Step 3) */
79 static void
80 _md5_init_ctx(md5_ctx* ctx)
81 {
82  ctx->A = 0x67452301;
83  ctx->B = 0xefcdab89;
84  ctx->C = 0x98badcfe;

91 /* Put result from CTX in first 16 bytes following RESBUF. The result
92  must be in little endian byte order.
93 
94  IMPORTANT: On some systems it is required that RESBUF is correctly
95  aligned for a 32 bits value. */
96 void*
97 _md5_read_ctx(const md5_ctx *ctx,void *resbuf)
98 {
99  ((md5_uint32 *) resbuf)[0] = SWAP (ctx->A);
100  ((md5_uint32 *) resbuf)[1] = SWAP (ctx->B);
101  ((md5_uint32 *) resbuf)[2] = SWAP (ctx->C);
102  ((md5_uint32 *) resbuf)[3] = SWAP (ctx->D);
103 
104  return resbuf;
105 }
106 
107 /* Process the remaining bytes in the internal buffer and the usual
108  prolog according to the standard and write the result to RESBUF.
109 
110  IMPORTANT: On some systems it is required that RESBUF is correctly
111  aligned for a 32 bits value. */
112 static void*
113 _md5_finish_ctx(md5_ctx *ctx,void *resbuf)
114 {
115  /* Take yet unprocessed bytes into account. */
116  md5_uint32 bytes = ctx->buflen;
117  size_t pad;
118 
119  /* Now count remaining bytes. */
120  ctx->total[0] += bytes;
121  if (ctx->total[0] < bytes)
122  ++ctx->total[1];
123 
124  pad = bytes >= 56 ? 64 + 56 - bytes : 56 - bytes;
125  memcpy (&ctx->buffer[bytes], fillbuf, pad);
126 
127  /* Put the 64-bit file length in *bits* at the end of the buffer. */
128  *(md5_uint32 *) &ctx->buffer[bytes + pad] = SWAP (ctx->total[0] << 3);
129  *(md5_uint32 *) &ctx->buffer[bytes + pad + 4] = SWAP ((ctx->total[1] << 3) |
130  (ctx->total[0] >> 29));
131 
132  /* Process last bytes. */
133  _md5_process_block (ctx->buffer, bytes + pad + 8, ctx);
134 
135  return _md5_read_ctx (ctx, resbuf);
136 }

139 /* Compute MD5 message digest for LEN bytes beginning at BUFFER. The
140  result is always in little endian byte order, so that a byte-wise
141  output yields to the wanted ASCII representation of the message
142  digest. */
143 static void*
144 _md5_buffer (const char *buffer,size_t len,void *resblock)
145 {
146  struct md5_ctx ctx;
147 
148  /* Initialize the computation context. */
149  _md5_init_ctx (&ctx);
150 
151  /* Process whole buffer but last len % 64 bytes. */
152  _md5_process_bytes (buffer, len, &ctx);
153 
154  /* Put result in desired memory area. */
155  return _md5_finish_ctx (&ctx, resblock);
156 }

159 static void
160 _md5_process_bytes(const void *buffer,size_t len,md5_ctx *ctx)
161 {
162  /* When we already have some bits in our internal buffer concatenate
163  both inputs first. */
164  if (ctx->buflen != 0)
165  {
166  size_t left_over = ctx->buflen;
167  size_t add = 128 - left_over > len ? len : 128 - left_over;
168 
169  memcpy (&ctx->buffer[left_over], buffer, add);
170  ctx->buflen += add;
171 
172  if (ctx->buflen > 64)
173  {
174  _md5_process_block (ctx->buffer, ctx->buflen & ~63, ctx);
175 
176  ctx->buflen &= 63;
177  /* The regions in the following copy operation cannot overlap. */
178  memcpy (ctx->buffer, &ctx->buffer[(left_over + add) & ~63],
179  ctx->buflen);
180  }
181 
182  buffer = (const char *) buffer + add;
183  len -= add;
184  }
185 
186  /* Process available complete blocks. */
187  if (len >= 64)
188  {
189 #if !_STRING_ARCH_unaligned
190  /* To check alignment gcc has an appropriate operator. Other
191  compilers don't. */
192 # if __GNUC__ >= 2
193 # define UNALIGNED_P(p) (((md5_uintptr) p) % __alignof__ (md5_uint32) != 0)
194 # else
195 # define UNALIGNED_P(p) (((md5_uintptr) p) % sizeof (md5_uint32) != 0)
196 # endif
197  if (UNALIGNED_P (buffer))
198  while (len > 64)
199  {
200  _md5_process_block (memcpy (ctx->buffer, buffer, 64), 64, ctx);
201  buffer = (const char *) buffer + 64;
202  len -= 64;
203  }
204  else
205 #endif
206  {
207  _md5_process_block (buffer, len & ~63, ctx);
208  buffer = (const char *) buffer + (len & ~63);
209  len &= 63;
210  }
211  }
212 
213  /* Move remaining bytes in internal buffer. */
214  if (len > 0)
215  {
216  size_t left_over = ctx->buflen;
217 
218  memcpy (&ctx->buffer[left_over], buffer, len);
219  left_over += len;
220  if (left_over >= 64)
221  {
222  _md5_process_block (ctx->buffer, 64, ctx);
223  left_over -= 64;
224  memcpy (ctx->buffer, &ctx->buffer[64], left_over);
225  }
226  ctx->buflen = left_over;
227  }
228 }

231 /* These are the four functions used in the four steps of the MD5 algorithm
232  and defined in the RFC 1321. The first function is a little bit optimized
233  (as found in Colin Plumbs public domain implementation). */
234 /* #define FF(b, c, d) ((b & c) | (~b & d)) */
235 #define FF(b, c, d) (d ^ (b & (c ^ d)))
236 #define FG(b, c, d) FF (d, b, c)
237 #define FH(b, c, d) (b ^ c ^ d)
238 #define FI(b, c, d) (c ^ (b | ~d))

241  It is assumed that LEN % 64 == 0. */

243 void
244 _md5_process_block(const void* buffer,size_t len,md5_ctx* ctx)
245 {
246  md5_uint32 correct_words[16];
247  const md5_uint32 *words = (const md5_uint32*)buffer;
248  size_t nwords = len / sizeof (md5_uint32);
249  const md5_uint32 *endp = words + nwords;
250  md5_uint32 A = ctx->A;
251  md5_uint32 B = ctx->B;
252  md5_uint32 C = ctx->C;
253  md5_uint32 D = ctx->D;
254 
255  /* First increment the byte count. RFC 1321 specifies the possible
256  length of the file up to 2^64 bits. Here we only compute the
257  number of bytes. Do a double word increment. */
258  ctx->total[0] += len;
259  if (ctx->total[0] < len)
260  ++ctx->total[1];
261 
262  /* Process all bytes in the buffer with 64 bytes in each round of
263  the loop. */
264  while (words < endp)
265  {
266  md5_uint32 *cwp = correct_words;
267  md5_uint32 A_save = A;
268  md5_uint32 B_save = B;
269  md5_uint32 C_save = C;
270  md5_uint32 D_save = D;
271 
272  /* First round: using the given function, the context and a constant
273  the next context is computed. Because the algorithms processing
274  unit is a 32-bit word and it is determined to work on words in
275  little endian byte order we perhaps have to change the byte order
276  before the computation. To reduce the work for the next steps
277  we store the swapped words in the array CORRECT_WORDS. */
278 
279 #define OP(a, b, c, d, s, T) \
280  do \
281  { \
282  a += FF (b, c, d) + (*cwp++ = SWAP (*words)) + T; \
283  ++words; \
284  CYCLIC (a, s); \
285  a += b; \
286  } \
287  while (0)
288 
289  /* It is unfortunate that C does not provide an operator for
290  cyclic rotation. Hope the C compiler is smart enough. */
291 #define CYCLIC(w, s) (w = (w << s) | (w >> (32 - s)))
292 
293  /* Before we start, one word to the strange constants.
294  They are defined in RFC 1321 as
295 
296  T[i] = (int) (4294967296.0 * fabs (sin (i))), i=1..64
297  */
298
