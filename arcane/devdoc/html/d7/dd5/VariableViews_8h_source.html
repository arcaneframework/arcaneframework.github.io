63 template<typename DataType>
64 class View1DSetter
65 {
66  // Pour accéder à m_data;
67  friend class View1DGetterSetter<DataType>;
68  public:
69  using ValueType = DataType;
70  using DataTypeReturnReference = View1DSetter<DataType>;
71  explicit ARCCORE_HOST_DEVICE View1DSetter(SmallSpan<DataType> data) : m_data(data){}
72  ARCCORE_HOST_DEVICE DataViewSetter<DataType> operator[](Int32 index) const
73  {
74  return DataViewSetter<DataType>(m_data.ptrAt(index));
75  }
76  ARCCORE_HOST_DEVICE DataViewSetter<DataType> operator()(Int32 index) const
77  {
78  return DataViewSetter<DataType>(m_data.ptrAt(index));
79  }
80  DataTypeReturnReference& operator=(const View1DSetter<DataType>& rhs) = delete;
81  public:
82  ARCCORE_HOST_DEVICE void copy(SmallSpan<const DataType> rhs)
83  {
84  m_data.copy(rhs);
85  }
86  private:
87  SmallSpan<DataType> m_data;
88 };

95 template<typename DataType>
96 class View1DGetterSetter
97 : public View1DSetter<DataType>
98 {
99  using View1DSetter<DataType>::m_data;
100  public:
101  using ValueType = DataType;
102  using DataTypeReturnReference = View1DGetterSetter<DataType>;
103  public:
104  explicit ARCCORE_HOST_DEVICE View1DGetterSetter(SmallSpan<DataType> data) : View1DSetter<DataType>(data){}
105  DataTypeReturnReference& operator=(const View1DGetterSetter<DataType>& rhs) = delete;
106  public:
107  ARCCORE_HOST_DEVICE DataViewGetterSetter<DataType> operator[](Int64 index) const
108  {
109  return DataViewGetterSetter<DataType>(m_data.ptrAt(index));
110  }
111  public:
112  ARCCORE_HOST_DEVICE SmallSpan<DataType> value() const { return m_data; }
113  ARCCORE_HOST_DEVICE operator SmallSpan<DataType>() { return m_data; }
114  ARCCORE_HOST_DEVICE operator SmallSpan<const DataType>() const { return m_data; }
115  ARCCORE_HOST_DEVICE operator Span<DataType>() { return { m_data.data(), m_data.size() }; }
116  ARCCORE_HOST_DEVICE operator Span<const DataType>() const { return { m_data.data(), m_data.size() }; }
117 };

124 template<typename ItemType, typename Accessor>
125 class ItemVariableScalarOutViewT
126 : public VariableViewBase
127 {
128  public:
129 
130  using DataType = typename Accessor::ValueType;
131  using DataTypeReturnReference = DataType&;
132  using ItemIndexType = typename ItemTraitsT<ItemType>::LocalIdType;
133 
134  public:
135 
136  ItemVariableScalarOutViewT(RunCommand& command,IVariable* var,SmallSpan<DataType> v)
137  : VariableViewBase(command,var), m_values(v.data()),
138  m_size(v.size()){}
139 
141  SimdSetter<DataType> operator[](SimdItemIndexT<ItemType> simd_item) const
142  {
143  return SimdSetter<DataType>(m_values,simd_item.simdLocalIds());
144  }
145 
147  SimdDirectSetter<DataType> operator[](SimdItemDirectIndexT<ItemType> simd_item) const
148  {
149  return SimdDirectSetter<DataType>(m_values+simd_item.baseLocalId());
150  }
151 
153  ARCCORE_HOST_DEVICE Accessor operator[](ItemIndexType item) const
154  {
155  ARCANE_CHECK_AT(item.localId(),m_size);
156  return Accessor(this->m_values + item.localId());
157  }
158 
160  ARCCORE_HOST_DEVICE Accessor operator()(ItemIndexType item) const
161  {
162  ARCANE_CHECK_AT(item.localId(),m_size);
163  return Accessor(this->m_values + item.localId());
164  }
165 
167  ARCCORE_HOST_DEVICE Accessor value(ItemIndexType item) const
168  {
169  ARCANE_CHECK_AT(item.localId(),m_size);
170  return Accessor(this->m_values + item.localId());
171  }
172 
174  ARCCORE_HOST_DEVICE void setValue(ItemIndexType item,const DataType& v) const
175  {
176  ARCANE_CHECK_AT(item.localId(),m_size);
177  this->m_values[item.localId()] = v;
178  }
179 
180  private:
181  DataType* m_values;
182  Int32 m_size;
183 };
