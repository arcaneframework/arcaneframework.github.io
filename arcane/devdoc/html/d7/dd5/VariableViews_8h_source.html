55template<typename DataType>
56class View1DSetter
57{
58 // Pour accéder à m_data;
59 friend class View1DGetterSetter<DataType>;
60 public:
61 using ValueType = DataType;
62 using DataTypeReturnReference = View1DSetter<DataType>;
63 explicit ARCCORE_HOST_DEVICE View1DSetter(SmallSpan<DataType> data) : m_data(data){}
64 ARCCORE_HOST_DEVICE DataViewSetter<DataType> operator[](Int32 index) const
65 {
66 return DataViewSetter<DataType>(m_data.ptrAt(index));
67 }
68 ARCCORE_HOST_DEVICE DataViewSetter<DataType> operator()(Int32 index) const
69 {
70 return DataViewSetter<DataType>(m_data.ptrAt(index));
71 }
72 DataTypeReturnReference& operator=(const View1DSetter<DataType>& rhs) = delete;
73 public:
74 ARCCORE_HOST_DEVICE void copy(SmallSpan<const DataType> rhs)
75 {
76 m_data.copy(rhs);
77 }
78 private:
79 SmallSpan<DataType> m_data;
80};

116template<typename ItemType, typename Accessor>
117class ItemVariableScalarOutViewT
118: public VariableViewBase
119{
120 public:
121
122 using DataType = typename Accessor::ValueType;
123 using DataTypeReturnReference = DataType&;
124 using ItemIndexType = typename ItemTraitsT<ItemType>::LocalIdType;
125
126 public:
127
128 ItemVariableScalarOutViewT(RunCommand& command,IVariable* var,SmallSpan<DataType> v)
129 : VariableViewBase(command,var), m_values(v.data()),
130 m_size(v.size()){}
131
133 SimdSetter<DataType> operator[](SimdItemIndexT<ItemType> simd_item) const
134 {
135 return SimdSetter<DataType>(m_values,simd_item.simdLocalIds());
136 }
137
139 SimdDirectSetter<DataType> operator[](SimdItemDirectIndexT<ItemType> simd_item) const
140 {
141 return SimdDirectSetter<DataType>(m_values+simd_item.baseLocalId());
142 }
143
145 ARCCORE_HOST_DEVICE Accessor operator[](ItemIndexType item) const
146 {
147 ARCANE_CHECK_AT(item.localId(),m_size);
148 return Accessor(this->m_values + item.localId());
149 }
150
152 ARCCORE_HOST_DEVICE Accessor operator()(ItemIndexType item) const
153 {
154 ARCANE_CHECK_AT(item.localId(),m_size);
155 return Accessor(this->m_values + item.localId());
156 }
157
159 ARCCORE_HOST_DEVICE Accessor value(ItemIndexType item) const
160 {
161 ARCANE_CHECK_AT(item.localId(),m_size);
162 return Accessor(this->m_values + item.localId());
163 }
164
166 ARCCORE_HOST_DEVICE void setValue(ItemIndexType item,const DataType& v) const
167 {
168 ARCANE_CHECK_AT(item.localId(),m_size);
169 this->m_values[item.localId()] = v;
170 }
171
172 private:
173 DataType* m_values;
174 Int32 m_size;
175};
