31class ArcaneCaseMeshMasterService
32: public ArcaneArcaneCaseMeshMasterServiceObject
33{
34 public:
35 ArcaneCaseMeshMasterService(const ServiceBuildInfo& sbi)
36 : ArcaneArcaneCaseMeshMasterServiceObject(sbi), m_sub_domain(sbi.subDomain()){}
37 private:
38 void createMeshes() override
39 {
40 if (m_is_created)
41 ARCANE_FATAL("Meshes have already been created");
42
43 Integer nb_mesh = options()->mesh.size();
44 info() << "Creating meshes from 'ArcaneCaseMeshMasterService' nb_mesh=" << nb_mesh;
45 Integer index = 0;
46 for( ICaseMeshService* s : options()->mesh ){
47 String name = String("Mesh")+String::fromNumber(index);
48 s->createMesh(name);
49 ++index;
50 }
51 m_is_created = true;
52 }
53
54 void allocateMeshes() override
55 {
56 if (m_is_allocated)
57 ARCANE_FATAL("Meshes have already been allocated");
58 if (!m_is_created)
59 ARCANE_FATAL("You need to call createMeshes() before allocateMeshes()");
60
61 for( ICaseMeshService* s : options()->mesh )
62 s->allocateMeshItems();
63 m_is_allocated = true;
64 }
65
66 void partitionMeshes() override
67 {
68 if (!m_is_allocated)
69 ARCANE_FATAL("Meshes have do be allocated before partitioning. call allocateMeshes() before");
70 for( ICaseMeshService* s : options()->mesh )
71 s->partitionMesh();
72 }
73
74 ICaseOptions* _options() override
75 {
76 return options()->caseOptions();
77 }
78 private:
79 ISubDomain* m_sub_domain;
80 bool m_is_created = false;
81 bool m_is_allocated = false;
82 UniqueArray<IPrimaryMesh*> m_meshes;
83};
