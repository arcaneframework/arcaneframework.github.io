355// Support pour l'alignement.
356// le C++11 utilise le mot clé alignas pour spécifier l'alignement.
357// Cela fonctionne avec GCC 4.9+ et Visual Studio 2015. Cela ne fonctionne
358// pas avec Visual Studio 2013. Donc pour Visual Studio on utilise dans tous
359// les cas __declspec qui fonctionne toujours. Sous Linux, __attribute__ fonctionne
360// aussi toujours donc on utilise cela. A noter que les structures Simd ont besoin
361// de l'attribut 'packed' qui n'existe que avec GCC et Intel. Il ne semble pas y avoir
362// d'équivalent avec MSVC.
363#ifdef _MSC_VER
365# define ARCCORE_ALIGNAS(value) __declspec(align(value))
367# define ARCCORE_ALIGNAS_PACKED(value) __declspec(align(value))
368#else
370# define ARCCORE_ALIGNAS_PACKED(value) __attribute__ ((aligned (value),packed))
372# define ARCCORE_ALIGNAS(value) __attribute__ ((aligned (value)))
