61_getMaxUniqueId(const ItemGroup& group)
62{
63 Int64 max_offset = 0;
64 ENUMERATE_ (Item, iitem, group) {

100void CartesianMeshCoarsening2::
101_doDoubleGhostLayers()
102{
103 IMesh* mesh = m_cartesian_mesh->mesh();
104 IMeshModifier* mesh_modifier = mesh->modifier();
105 IGhostLayerMng* gm = mesh->ghostLayerMng();
106 // Il faut au moins utiliser la version 3 pour pouvoir supporter
107 // plusieurs couches de mailles fantômes
108 Int32 version = gm->builderVersion();
109 if (version < 3)
110 gm->setBuilderVersion(3);
111 Int32 nb_ghost_layer = gm->nbGhostLayer();
112 gm->setNbGhostLayer(nb_ghost_layer * 2);
113 mesh_modifier->setDynamic(true);
114 mesh_modifier->updateGhostLayers();
115 // Remet le nombre initial de couches de mailles fantômes
116 gm->setNbGhostLayer(nb_ghost_layer);
117
118 // Comme le nombre d'entités a été modifié, il faut recalculer les directions
119 m_cartesian_mesh->computeDirections();
120
121 // Écrit le nouveau maillage
122 _writeMeshSVG("double_ghost");
123}

128void CartesianMeshCoarsening2::
129createCoarseCells()
130{
131 const bool is_verbose = m_verbosity_level > 0;
132 IMesh* mesh = m_cartesian_mesh->mesh();
133 IParallelMng* pm = mesh->parallelMng();
134 Integer nb_patch = m_cartesian_mesh->nbPatch();
135 if (nb_patch != 1)
136 ARCANE_FATAL("This method is only valid for 1 patch (nb_patch={0})", nb_patch);
137
138 if (!mesh->isAmrActivated())
139 ARCANE_FATAL("AMR is not activated for this case");
140
141 Integer nb_dir = mesh->dimension();
142 if (nb_dir != 2 && nb_dir != 3)
143 ARCANE_FATAL("This method is only valid for 2D or 3D mesh (dim={0})", nb_dir);
144
145 info() << "CoarseCartesianMesh nb_direction=" << nb_dir;
146
147 for (Integer idir = 0; idir < nb_dir; ++idir) {
148 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
149 Int32 nb_own_cell = cdm.ownNbCell();
150 info() << "NB_OWN_CELL dir=" << idir << " n=" << nb_own_cell;
151 if ((nb_own_cell % 2) != 0)
152 ARCANE_FATAL("Invalid number of cells ({0}) for direction {1}. Should be a multiple of 2",
153 nb_own_cell, idir);
154 }
155
156 _writeMeshSVG("orig");
157
158 // Double la couche de mailles fantômes
159 _doDoubleGhostLayers();
160
161 if (is_verbose) {
162 SmallArray<Int64, 8> uids;
163 ENUMERATE_ (Cell, icell, mesh->allCells()) {
164 Cell cell = *icell;
165 info() << "Orig cell=" << ItemPrinter(cell) << " Face0=" << cell.face(0).uniqueId()
166 << " Face1=" << cell.face(1).uniqueId() << " level=" << cell.level();
167 Int32 nb_node = cell.nbNode();
168 uids.resize(nb_node);
169 for (Int32 i = 0; i < nb_node; ++i)
170 uids[i] = cell.node(i).uniqueId();
171 info() << "Orig cell_uid=" << cell.uniqueId() << " Nodes=" << uids;
172 Int32 nb_face = cell.nbFace();
173 uids.resize(nb_face);
174 for (Int32 i = 0; i < nb_face; ++i)
175 uids[i] = cell.face(i).uniqueId();
176 info() << "Orig cell_uid=" << cell.uniqueId() << " Faces=" << uids;
177 }
178 }
179
180 // Calcul l'offset pour la création des uniqueId().
181 // On prend comme offset le max des uniqueId() des faces et des mailles.
182 // A terme avec la numérotation cartésienne partout, on pourra déterminer
183 // directement cette valeur
184 Int64 max_cell_uid = _getMaxUniqueId(mesh->allCells());
185 Int64 max_face_uid = _getMaxUniqueId(mesh->allFaces());
186 const Int64 coarse_grid_cell_offset = 1 + pm->reduce(Parallel::ReduceMax, math::max(max_cell_uid, max_face_uid));
187 m_first_own_cell_unique_id_offset = coarse_grid_cell_offset;
188 info() << "FirstCellUniqueIdOffset=" << m_first_own_cell_unique_id_offset;
189 if (nb_dir == 3)
190 _createCoarseCells3D();
191 else if (nb_dir == 2)
192 _createCoarseCells2D();
193 else
194 ARCANE_FATAL("Invalid dimension '{0}'", nb_dir);
195
196 mesh->modifier()->endUpdate();
197
198 if (is_verbose) {
199 ENUMERATE_ (Cell, icell, mesh->allCells()) {
200 Cell cell = *icell;
201 info() << "Final cell=" << ItemPrinter(cell) << " Face0=" << cell.face(0).uniqueId()
202 << " Face1=" << cell.face(1).uniqueId() << " level=" << cell.level();
203 }
204 }
205
206 _recomputeMeshGenerationInfo();
207
208 // Affiche les statistiques du nouveau maillage
209 {
210 MeshStats ms(traceMng(), mesh, mesh->parallelMng());
211 ms.dumpStats();
212 }
213
215 {
216 CellGroup parent_cells = mesh->allLevelCells(0);
217 m_cartesian_mesh->_internalApi()->addPatchFromExistingChildren(parent_cells.view().localIds());
218 }
219
220 // Recalcule les informations de synchronisation
221 // Cela n'est pas nécessaire pour l'AMR car ces informations seront recalculées
222 // lors du raffinement mais comme on ne sais pas si on va faire du raffinement
223 // après il est préférable de calculer ces informations dans tous les cas.
224 mesh->computeSynchronizeInfos();
225
226 // Il faut recalculer les nouvelles directions après les modifications
227 // et l'ajout de patch.
228 m_cartesian_mesh->computeDirections();
229
230 _writeMeshSVG("coarse");
231}

236void CartesianMeshCoarsening2::
237_createCoarseCells2D()
238{
239 const bool is_verbose = m_verbosity_level > 0;
240 IMesh* mesh = m_cartesian_mesh->mesh();
241 IParallelMng* pm = mesh->parallelMng();
242 const Int32 my_rank = pm->commRank();
243
244 CellDirectionMng cdm_x(m_cartesian_mesh->cellDirection(0));
245 CellDirectionMng cdm_y(m_cartesian_mesh->cellDirection(1));
246
247 const Int64 global_nb_cell_x = cdm_x.globalNbCell();
248 const Int64 global_nb_cell_y = cdm_y.globalNbCell();
249 CartesianGridDimension refined_grid_dim(global_nb_cell_x, global_nb_cell_y);
250 CartesianGridDimension coarse_grid_dim(global_nb_cell_x / 2, global_nb_cell_y / 2);
251 CartesianGridDimension::CellUniqueIdComputer2D refined_cell_uid_computer(refined_grid_dim.getCellComputer2D(0));
252 CartesianGridDimension::NodeUniqueIdComputer2D refined_node_uid_computer(refined_grid_dim.getNodeComputer2D(0));
253 CartesianGridDimension::CellUniqueIdComputer2D coarse_cell_uid_computer(coarse_grid_dim.getCellComputer2D(m_first_own_cell_unique_id_offset));
254 CartesianGridDimension::FaceUniqueIdComputer2D coarse_face_uid_computer(coarse_grid_dim.getFaceComputer2D(m_first_own_cell_unique_id_offset));
255
256 // Pour les mailles et faces grossières, les noeuds existent déjà
257 // On ne peut donc pas utiliser la connectivité cartésienne de la grille grossière
258 // pour eux (on pourra le faire lorsque l'AMR par patch avec duplication sera active)
259 // En attendant on utilise la numérotation de la grille raffinée.
260
261 // TODO: Calculer en avance le nombre de faces et de mailles et allouer en conséquence.
262 UniqueArray<Int64> faces_infos;
263 UniqueArray<Int64> cells_infos;
264 Int32 nb_coarse_face = 0;
265 Int32 nb_coarse_cell = 0;
267 UniqueArray<Cell> first_child_cells;
268
269 UniqueArray<Int64> refined_cells_lids;
270 UniqueArray<Int64> coarse_cells_uids;
271 UniqueArray<Int32> coarse_cells_owner;
272 UniqueArray<Int32> coarse_faces_owner;
273
274 ENUMERATE_ (Cell, icell, mesh->allCells()) {
275 Cell cell = *icell;
276 Int64 cell_uid = cell.uniqueId();
277 Int64x3 cell_xy = refined_cell_uid_computer.compute(cell_uid);
278 const Int64 cell_x = cell_xy.x;
279 const Int64 cell_y = cell_xy.y;
280 // Nécessaire pour le recalcul des mailles fantômes. On considère ces
281 // mailles comme si elles venaient juste d'être raffinées.
282 cell.mutableItemBase().addFlags(ItemFlags::II_JustRefined);
283 // Comme on déraffine par 2, ne prend que les mailles dont les coordoonnées
284 // topologiques sont paires
285 if ((cell_x % 2) != 0 || (cell_y % 2) != 0)
286 continue;
287 if (is_verbose)
288 info() << "CellToCoarse refined_uid=" << cell_uid << " x=" << cell_x << " y=" << cell_y;
289 coarse_cells_owner.add(cell.owner());
290 const Int64 coarse_cell_x = cell_x / 2;
291 const Int64 coarse_cell_y = cell_y / 2;
292 std::array<Int64, 4> node_uids_container;
293 ArrayView<Int64> node_uids(node_uids_container);
294 node_uids[0] = refined_node_uid_computer.compute(cell_x + 0, cell_y + 0);
295 node_uids[1] = refined_node_uid_computer.compute(cell_x + 2, cell_y + 0);
296 node_uids[2] = refined_node_uid_computer.compute(cell_x + 2, cell_y + 2);
297 node_uids[3] = refined_node_uid_computer.compute(cell_x + 0, cell_y + 2);
298 if (is_verbose)
299 info() << "CELLNodes uid=" << node_uids;
300 std::array<Int64, 4> coarse_face_uids = coarse_face_uid_computer.computeForCell(coarse_cell_x, coarse_cell_y);
301 const ItemTypeInfo* cell_type = cell.typeInfo();
302 // Ajoute les 4 faces
303 for (Int32 z = 0; z < 4; ++z) {
304 ItemTypeInfo::LocalFace lface = cell_type->localFace(z);
305 faces_infos.add(IT_Line2);
306 faces_infos.add(coarse_face_uids[z]);
307 Int64 node_uid0 = node_uids[lface.node(0)];
308 Int64 node_uid1 = node_uids[lface.node(1)];
309
310 if (node_uid0 > node_uid1)
311 std::swap(node_uid0, node_uid1);
312 if (is_verbose)
313 info() << "ADD_FACE coarse_uid=" << coarse_face_uids[z] << " n0=" << node_uid0 << " n1=" << node_uid1;
314 faces_infos.add(node_uid0);
315 faces_infos.add(node_uid1);
316 ++nb_coarse_face;
317 }
318 // Ajoute la maille
319 {
320 cells_infos.add(IT_Quad4);
321 Int64 coarse_cell_uid = coarse_cell_uid_computer.compute(coarse_cell_x, coarse_cell_y);
322 cells_infos.add(coarse_cell_uid);
323 if (is_verbose)
324 info() << "CoarseCellUid=" << coarse_cell_uid;
325 coarse_cells_uids.add(coarse_cell_uid);
326 for (Int32 z = 0; z < 4; ++z)
327 cells_infos.add(node_uids[z]);
328 ++nb_coarse_cell;
329 first_child_cells.add(cell);
330 }
331 // A partir de la première sous-maille, on peut connaitre les 3 autres
332 // car elles sont respectivement à droite, en haut à droite et en haut.
333 {
334 std::array<Int32, 4> sub_cell_lids_container;
335 ArrayView<Int32> sub_lids(sub_cell_lids_container);
336 Cell cell1 = cdm_x[cell].next();
337 // Vérifie la validité des sous-mailles.
338 // Normalement il ne devrait pas y avoir de problèmes sauf si le
339 // nombre de mailles dans chaque direction du sous-domaine
340 // n'est pas un nombre pair.
341 if (cell1.null())
342 ARCANE_FATAL("Bad right cell for cell {0}", ItemPrinter(cell));
343 Cell cell2 = cdm_y[cell1].next();
344 if (cell2.null())
345 ARCANE_FATAL("Bad upper right cell for cell {0}", ItemPrinter(cell));
346 Cell cell3 = cdm_y[cell].next();
347 if (cell3.null())
348 ARCANE_FATAL("Bad upper cell for cell {0}", ItemPrinter(cell));
349 sub_lids[0] = cell.localId();
350 sub_lids[1] = cell1.localId();
351 sub_lids[2] = cell2.localId();
352 sub_lids[3] = cell3.localId();
353 // Il faudra donner un propriétaire aux faces.
354 // Ces nouvelles faces auront le même propriétaire que les faces raffinées
355 // auquelles elles correspondent
356 //info() << "CELL_NB_FACE=" << cell.nbFace() << " " << cell1.nbFace() << " " << cell2.nbFace() << " " << cell3.nbFace();
357 coarse_faces_owner.add(cell.face(0).owner());
358 coarse_faces_owner.add(cell1.face(1).owner());
359 coarse_faces_owner.add(cell2.face(2).owner());
360 coarse_faces_owner.add(cell3.face(3).owner());
361 for (Int32 i = 0; i < 4; ++i)
362 refined_cells_lids.add(sub_lids[i]);
363 }
364 }
365
366 // Construit les faces
367 UniqueArray<Int32> faces_local_ids(nb_coarse_face);
368 mesh->modifier()->addFaces(nb_coarse_face, faces_infos, faces_local_ids);
369
370 // Construit les mailles
371 // Indique qu'on n'a pas le droit de construire à la volée les faces.
372 // Normalement elles ont toutes été ajoutées via addFaces();
373 UniqueArray<Int32> cells_local_ids(nb_coarse_cell);
374 MeshModifierAddCellsArgs add_cells_args(nb_coarse_cell, cells_infos, cells_local_ids);
375 add_cells_args.setAllowBuildFaces(false);
376 mesh->modifier()->addCells(add_cells_args);
377
378 IItemFamily* cell_family = mesh->cellFamily();
379
380 // Maintenant que les mailles grossières sont créées, il faut indiquer
381 // qu'elles sont parentes.
382 using mesh::CellFamily;
383 CellInfoListView cells(mesh->cellFamily());
384 CellFamily* true_cell_family = ARCANE_CHECK_POINTER(dynamic_cast<CellFamily*>(cell_family));
385 std::array<Int32, 4> sub_cell_lids_container;
386 ArrayView<Int32> sub_cell_lids(sub_cell_lids_container);
387 for (Int32 i = 0; i < nb_coarse_cell; ++i) {
388 Int32 coarse_cell_lid = cells_local_ids[i];
389 Cell coarse_cell = cells[coarse_cell_lid];
390 Cell first_child_cell = first_child_cells[i];
391 // A partir de la première sous-maille, on peut connaitre les 3 autres
392 // car elles sont respectivement à droite, en haut à droite et en haut.
393 sub_cell_lids[0] = first_child_cell.localId();
394 sub_cell_lids[1] = cdm_x[first_child_cell].next().localId();
395 sub_cell_lids[2] = cdm_y[CellLocalId(sub_cell_lids[1])].next().localId();
396 sub_cell_lids[3] = cdm_y[first_child_cell].next().localId();
397 if (is_verbose)
398 info() << "AddChildForCoarseCell i=" << i << " coarse=" << ItemPrinter(coarse_cell)
399 << " children_lid=" << sub_cell_lids;
400 for (Int32 z = 0; z < 4; ++z) {
401 Cell child_cell = cells[sub_cell_lids[z]];
402 if (is_verbose)
403 info() << " AddParentCellToCell: z=" << z << " child=" << ItemPrinter(child_cell);
404 true_cell_family->_addParentCellToCell(child_cell, coarse_cell);
405 }
406 true_cell_family->_addChildrenCellsToCell(coarse_cell, sub_cell_lids);
407 }
408
409 // Positionne les propriétaires des nouvelles mailles et faces
410 {
411 IItemFamily* face_family = mesh->faceFamily();
412 Int32 index = 0;
413 ENUMERATE_ (Cell, icell, cell_family->view(cells_local_ids)) {
414 Cell cell = *icell;
415 Int32 owner = coarse_cells_owner[index];
416 cell.mutableItemBase().setOwner(owner, my_rank);
417 const Int64 sub_cell_index = index * 4;
418 for (Int32 z = 0; z < 4; ++z) {
419 cell.face(z).mutableItemBase().setOwner(coarse_faces_owner[sub_cell_index + z], my_rank);
420 }
421 ++index;
422 }
423 cell_family->notifyItemsOwnerChanged();
424 face_family->notifyItemsOwnerChanged();
425 }
426}

431void CartesianMeshCoarsening2::
432_createCoarseCells3D()
433{
434 const bool is_verbose = m_verbosity_level > 0;
435 IMesh* mesh = m_cartesian_mesh->mesh();
436 IParallelMng* pm = mesh->parallelMng();
437 const Int32 my_rank = pm->commRank();
438
439 CellDirectionMng cdm_x(m_cartesian_mesh->cellDirection(0));
440 CellDirectionMng cdm_y(m_cartesian_mesh->cellDirection(1));
441 CellDirectionMng cdm_z(m_cartesian_mesh->cellDirection(2));
442
443 const Int64 global_nb_cell_x = cdm_x.globalNbCell();
444 const Int64 global_nb_cell_y = cdm_y.globalNbCell();
445 const Int64 global_nb_cell_z = cdm_z.globalNbCell();
446 CartesianGridDimension refined_grid_dim(global_nb_cell_x, global_nb_cell_y, global_nb_cell_z);
447 CartesianGridDimension coarse_grid_dim(global_nb_cell_x / 2, global_nb_cell_y / 2, global_nb_cell_z / 2);
448 CartesianGridDimension::CellUniqueIdComputer3D refined_cell_uid_computer(refined_grid_dim.getCellComputer3D(0));
449 CartesianGridDimension::NodeUniqueIdComputer3D refined_node_uid_computer(refined_grid_dim.getNodeComputer3D(0));
450 CartesianGridDimension::CellUniqueIdComputer3D coarse_cell_uid_computer(coarse_grid_dim.getCellComputer3D(m_first_own_cell_unique_id_offset));
451 CartesianGridDimension::FaceUniqueIdComputer3D coarse_face_uid_computer(coarse_grid_dim.getFaceComputer3D(m_first_own_cell_unique_id_offset));
452
453 // Pour les mailles et faces grossières, les noeuds existent déjà
454 // On ne peut donc pas utiliser la connectivité cartésienne de la grille grossière
455 // pour eux (on pourra le faire lorsque l'AMR par patch avec duplication sera active)
456 // En attendant on utilise la numérotation de la grille raffinée.
457
458 // TODO: Calculer en avance le nombre de faces et de mailles et allouer en conséquence.
459 UniqueArray<Int64> faces_infos;
460 UniqueArray<Int64> cells_infos;
461 Int32 nb_coarse_face = 0;
462 Int32 nb_coarse_cell = 0;
464 UniqueArray<Cell> first_child_cells;
465
466 UniqueArray<Int64> refined_cells_lids;
467 UniqueArray<Int64> coarse_cells_uids;
468 UniqueArray<Int32> coarse_cells_owner;
469 UniqueArray<Int32> coarse_faces_owner;
470
471 static constexpr Int32 const_cell_nb_node = 8;
472 static constexpr Int32 const_cell_nb_face = 6;
473 static constexpr Int32 const_cell_nb_sub_cell = 8;
474 static constexpr Int32 const_face_nb_node = 4;
475
476 // Liste des uniqueId() des noeuds des faces créées
477 SmallArray<Int64, const_face_nb_node> face_node_uids(const_face_nb_node);
478 // Liste ordonnée des noeuds des faces créées
479 SmallArray<Int64, const_face_nb_node> face_sorted_node_uids(const_face_nb_node);
480 ENUMERATE_ (Cell, icell, mesh->allCells()) {
481 Cell cell = *icell;
482 Int64 cell_uid = cell.uniqueId();
483 Int64x3 cell_xyz = refined_cell_uid_computer.compute(cell_uid);
484 const Int64 cell_x = cell_xyz.x;
485 const Int64 cell_y = cell_xyz.y;
486 const Int64 cell_z = cell_xyz.z;
487 // Nécessaire pour le recalcul des mailles fantômes. On considère ces
488 // mailles comme si elles venaient juste d'être raffinées.
489 cell.mutableItemBase().addFlags(ItemFlags::II_JustRefined);
490 // Comme on déraffine par 2, ne prend que les mailles dont les coordoonnées
491 // topologiques sont paires
492 if ((cell_x % 2) != 0 || (cell_y % 2) != 0 || (cell_z % 2) != 0)
493 continue;
494 if (is_verbose)
495 info() << "CellToCoarse refined_uid=" << cell_uid << " x=" << cell_x << " y=" << cell_y << " z=" << cell_z;
496 coarse_cells_owner.add(cell.owner());
497 const Int64 coarse_cell_x = cell_x / 2;
498 const Int64 coarse_cell_y = cell_y / 2;
499 const Int64 coarse_cell_z = cell_z / 2;
500 std::array<Int64, const_cell_nb_node> node_uids_container;
501 ArrayView<Int64> node_uids(node_uids_container);
502 node_uids[0] = refined_node_uid_computer.compute(cell_x + 0, cell_y + 0, cell_z + 0);
503 node_uids[1] = refined_node_uid_computer.compute(cell_x + 2, cell_y + 0, cell_z + 0);
504 node_uids[2] = refined_node_uid_computer.compute(cell_x + 2, cell_y + 2, cell_z + 0);
505 node_uids[3] = refined_node_uid_computer.compute(cell_x + 0, cell_y + 2, cell_z + 0);
506 node_uids[4] = refined_node_uid_computer.compute(cell_x + 0, cell_y + 0, cell_z + 2);
507 node_uids[5] = refined_node_uid_computer.compute(cell_x + 2, cell_y + 0, cell_z + 2);
508 node_uids[6] = refined_node_uid_computer.compute(cell_x + 2, cell_y + 2, cell_z + 2);
509 node_uids[7] = refined_node_uid_computer.compute(cell_x + 0, cell_y + 2, cell_z + 2);
510 if (is_verbose)
511 info() << "CELLNodes uid=" << node_uids;
512 std::array<Int64, const_cell_nb_face> coarse_face_uids = coarse_face_uid_computer.computeForCell(coarse_cell_x, coarse_cell_y, coarse_cell_z);
513 const ItemTypeInfo* cell_type = cell.typeInfo();
514
515 // Ajoute les 6 faces
516 for (Int32 z = 0; z < const_cell_nb_face; ++z) {
517 ItemTypeInfo::LocalFace lface = cell_type->localFace(z);
518 faces_infos.add(IT_Quad4);
519 faces_infos.add(coarse_face_uids[z]);
520 for (Int32 knode = 0; knode < const_face_nb_node; ++knode)
521 face_node_uids[knode] = node_uids[lface.node(knode)];
522 MeshUtils::reorderNodesOfFace(face_node_uids, face_sorted_node_uids);
523 if (is_verbose)
524 info() << "ADD_FACE coarse_uid=" << coarse_face_uids[z] << " n=" << face_sorted_node_uids;
525 faces_infos.addRange(face_sorted_node_uids);
526 ++nb_coarse_face;
527 }
528
529 // Ajoute la maille
530 {
531 cells_infos.add(IT_Hexaedron8);
532 Int64 coarse_cell_uid = coarse_cell_uid_computer.compute(coarse_cell_x, coarse_cell_y, coarse_cell_z);
533 cells_infos.add(coarse_cell_uid);
534 if (is_verbose)
535 info() << "CoarseCellUid=" << coarse_cell_uid;
536 coarse_cells_uids.add(coarse_cell_uid);
537 for (Int32 z = 0; z < const_cell_nb_node; ++z)
538 cells_infos.add(node_uids[z]);
539 ++nb_coarse_cell;
540 first_child_cells.add(cell);
541 }
542
543 // A partir de la première sous-maille, on peut connaitre les 7 autres
544 // car elles sont respectivement à droite, en haut à droite et en haut,
545 // au dessus, au dessus à droit, au dessus en haut à droit et .
546 {
547 std::array<Int32, const_cell_nb_sub_cell> sub_cell_lids_container;
548 ArrayView<Int32> sub_lids(sub_cell_lids_container);
549 Cell cell1 = cdm_x[cell].next();
550 // Vérifie la validité des sous-mailles.
551 // Normalement il ne devrait pas y avoir de problèmes sauf si le
552 // nombre de mailles dans chaque direction du sous-domaine
553 // n'est pas un nombre pair.
554 if (cell1.null())
555 ARCANE_FATAL("Bad right cell for cell {0}", ItemPrinter(cell));
556 Cell cell2 = cdm_y[cell1].next();
557 if (cell2.null())
558 ARCANE_FATAL("Bad upper right cell for cell {0}", ItemPrinter(cell));
559 Cell cell3 = cdm_y[cell].next();
560 if (cell3.null())
561 ARCANE_FATAL("Bad upper cell for cell {0}", ItemPrinter(cell));
562
563 Cell cell4 = cdm_z[cell].next();
564 if (cell4.null())
565 ARCANE_FATAL("Bad top cell for cell {0}", ItemPrinter(cell));
566
567 Cell cell5 = cdm_x[cell4].next();
568 if (cell5.null())
569 ARCANE_FATAL("Bad top right cell for cell {0}", ItemPrinter(cell));
570 Cell cell6 = cdm_y[cell5].next();
571 if (cell6.null())
572 ARCANE_FATAL("Bad top upper right cell for cell {0}", ItemPrinter(cell));
573 Cell cell7 = cdm_y[cell4].next();
574 if (cell7.null())
575 ARCANE_FATAL("Bad top upper cell for cell {0}", ItemPrinter(cell));
576
577 sub_lids[0] = cell.localId();
578 sub_lids[1] = cell1.localId();
579 sub_lids[2] = cell2.localId();
580 sub_lids[3] = cell3.localId();
581 sub_lids[4] = cell4.localId();
582 sub_lids[5] = cell5.localId();
583 sub_lids[6] = cell6.localId();
584 sub_lids[7] = cell7.localId();
585 // Il faudra donner un propriétaire aux faces.
586 // Ces nouvelles faces auront le même propriétaire que les faces raffinées
587 // auquelles elles correspondent
588 //info() << "CELL_NB_FACE=" << cell.nbFace() << " " << cell1.nbFace() << " " << cell2.nbFace() << " " << cell3.nbFace();
589 coarse_faces_owner.add(cell.face(0).owner());
590 coarse_faces_owner.add(cell1.face(1).owner());
591 coarse_faces_owner.add(cell2.face(2).owner());
592 coarse_faces_owner.add(cell3.face(3).owner());
593
594 coarse_faces_owner.add(cell4.face(4).owner());
595 coarse_faces_owner.add(cell5.face(5).owner());
596
597 for (Int32 i = 0; i < const_cell_nb_sub_cell; ++i)
598 refined_cells_lids.add(sub_lids[i]);
599 }
600 }
601
602 // Construit les faces
603 UniqueArray<Int32> faces_local_ids(nb_coarse_face);
604 mesh->modifier()->addFaces(nb_coarse_face, faces_infos, faces_local_ids);
605
606 // Construit les mailles
607 // Indique qu'on n'a pas le droit de construire à la volée les faces.
608 // Normalement elles ont toutes été ajoutées via addFaces();
609 UniqueArray<Int32> cells_local_ids(nb_coarse_cell);
610 MeshModifierAddCellsArgs add_cells_args(nb_coarse_cell, cells_infos, cells_local_ids);
611 add_cells_args.setAllowBuildFaces(false);
612 mesh->modifier()->addCells(add_cells_args);
613
614 IItemFamily* cell_family = mesh->cellFamily();
615
616 // Maintenant que les mailles grossières sont créées, il faut indiquer
617 // qu'elles sont parentes.
618 using mesh::CellFamily;
619 CellInfoListView cells(mesh->cellFamily());
620 CellFamily* true_cell_family = ARCANE_CHECK_POINTER(dynamic_cast<CellFamily*>(cell_family));
621 std::array<Int32, const_cell_nb_sub_cell> sub_cell_lids_container;
622 ArrayView<Int32> sub_cell_lids(sub_cell_lids_container);
623 for (Int32 i = 0; i < nb_coarse_cell; ++i) {
624 Int32 coarse_cell_lid = cells_local_ids[i];
625 Cell coarse_cell = cells[coarse_cell_lid];
626 Cell first_child_cell = first_child_cells[i];
627 // A partir de la première sous-maille, on peut connaitre les 3 autres
628 // car elles sont respectivement à droite, en haut à droite et en haut.
629 sub_cell_lids[0] = first_child_cell.localId();
630 sub_cell_lids[1] = cdm_x[first_child_cell].next().localId();
631 sub_cell_lids[2] = cdm_y[CellLocalId(sub_cell_lids[1])].next().localId();
632 sub_cell_lids[3] = cdm_y[first_child_cell].next().localId();
633
634 Cell top_first_child_cell = cdm_z[first_child_cell].next();
635 sub_cell_lids[4] = top_first_child_cell.localId();
636 sub_cell_lids[5] = cdm_x[top_first_child_cell].next().localId();
637 sub_cell_lids[6] = cdm_y[CellLocalId(sub_cell_lids[5])].next().localId();
638 sub_cell_lids[7] = cdm_y[top_first_child_cell].next().localId();
639
640 if (is_verbose)
641 info() << "AddChildForCoarseCell i=" << i << " coarse=" << ItemPrinter(coarse_cell)
642 << " children_lid=" << sub_cell_lids;
643 for (Int32 z = 0; z < const_cell_nb_sub_cell; ++z) {
644 Cell child_cell = cells[sub_cell_lids[z]];
645 if (is_verbose)
646 info() << " AddParentCellToCell: z=" << z << " child=" << ItemPrinter(child_cell);
647 true_cell_family->_addParentCellToCell(child_cell, coarse_cell);
648 }
649 true_cell_family->_addChildrenCellsToCell(coarse_cell, sub_cell_lids);
650 }
651
652 // Positionne les propriétaires des nouvelles mailles et faces
653 {
654 IItemFamily* face_family = mesh->faceFamily();
655 Int32 index = 0;
656 ENUMERATE_ (Cell, icell, cell_family->view(cells_local_ids)) {
657 Cell cell = *icell;
658 Int32 owner = coarse_cells_owner[index];
659 cell.mutableItemBase().setOwner(owner, my_rank);
660 const Int64 sub_cell_index = index * const_cell_nb_face;
661 for (Int32 z = 0; z < const_cell_nb_face; ++z) {
662 cell.face(z).mutableItemBase().setOwner(coarse_faces_owner[sub_cell_index + z], my_rank);
663 }
664 ++index;
665 }
666 cell_family->notifyItemsOwnerChanged();
667 face_family->notifyItemsOwnerChanged();
668 }
669}
