92template <typename LoopBoundType, typename Lambda, typename... RemainingArgs> __global__ void
93doDirectGPULambdaArrayBounds2(LoopBoundType bounds, Lambda func, RemainingArgs... remaining_args)
94{
95 using namespace Arcane::Accelerator::Impl;
96
97 // TODO: a supprimer quand il n'y aura plus les anciennes réductions
98 auto privatizer = privatize(func);
99 auto& body = privatizer.privateCopy();
100
101 Int32 i = blockDim.x * blockIdx.x + threadIdx.x;
102
103 CudaHipKernelRemainingArgsHelper::applyAtBegin(i, remaining_args...);
104 if constexpr (requires { bounds.nbStride(); }) {
105 // Test expérimental pour utiliser un pas de la taille
106 // de la grille. Le nombre de pas est donné par bounds.nbStride().
107 Int32 nb_grid_stride = bounds.nbStride();
108 Int32 offset = blockDim.x * gridDim.x;
109 for (Int32 k = 0; k < nb_grid_stride; ++k) {
110 Int32 true_i = i + (offset * k);
111 if (true_i < bounds.nbOriginalElement()) {
112 body(arcaneGetLoopIndexCudaHip(bounds, true_i), remaining_args...);
113 }
114 }
115 }
116 else {
117 if (i < bounds.nbElement()) {
118 body(arcaneGetLoopIndexCudaHip(bounds, i), remaining_args...);
119 }
120 }
121 CudaHipKernelRemainingArgsHelper::applyAtEnd(i, remaining_args...);
122}
123
124#endif

191template <typename LoopBoundType, typename Lambda, typename... RemainingArgs> void
192_applyGenericLoop(RunCommand& command, LoopBoundType bounds,
193 const Lambda& func, const RemainingArgs&... other_args)
194{
195 Int64 vsize = bounds.nbElement();
196 if (vsize == 0)
197 return;
198#if defined(ARCCORE_EXPERIMENTAL_GRID_STRIDE) && defined(ARCCORE_COMPILING_CUDA_OR_HIP)
199 using TrueLoopBoundType = Impl::StridedLoopRanges<LoopBoundType>;
200 TrueLoopBoundType bounds2(command.nbStride(), bounds);
201 Impl::RunCommandLaunchInfo launch_info(command, bounds2.strideValue());
202#else
203 using TrueLoopBoundType = LoopBoundType;
204 [[maybe_unused]] const TrueLoopBoundType& bounds2 = bounds;
205 Impl::RunCommandLaunchInfo launch_info(command, vsize);
206#endif
207 launch_info.beginExecute();
208 const eExecutionPolicy exec_policy = launch_info.executionPolicy();
209 switch (exec_policy) {
210 case eExecutionPolicy::CUDA:
211 ARCCORE_KERNEL_CUDA_FUNC((Impl::doDirectGPULambdaArrayBounds2<TrueLoopBoundType, Lambda, RemainingArgs...>),
212 launch_info, func, bounds2, other_args...);
213 break;
214 case eExecutionPolicy::HIP:
215 ARCCORE_KERNEL_HIP_FUNC((Impl::doDirectGPULambdaArrayBounds2<TrueLoopBoundType, Lambda, RemainingArgs...>),
216 launch_info, func, bounds2, other_args...);
217 break;
218 case eExecutionPolicy::SYCL:
219 ARCCORE_KERNEL_SYCL_FUNC((Impl::DoDirectSYCLLambdaArrayBounds<LoopBoundType, Lambda, RemainingArgs...>{}),
220 launch_info, func, bounds, other_args...);
221 break;
222 case eExecutionPolicy::Sequential:
223 arccoreSequentialFor(bounds, func, other_args...);
224 break;
225 case eExecutionPolicy::Thread:
226 arccoreParallelFor(bounds, launch_info.loopRunInfo(), func, other_args...);
227 break;
228 default:
229 ARCCORE_FATAL("Invalid execution policy '{0}'", exec_policy);
230 }
231 launch_info.endExecute();
232}
