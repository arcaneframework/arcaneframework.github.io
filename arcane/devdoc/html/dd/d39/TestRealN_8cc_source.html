17TEST(TestRealN,Real2)
18{
19 {
20 Real2 v1;

22 ASSERT_EQ(v1.y,0.0);
23 }
24 {
25 double value = 0.2;

30 ASSERT_EQ(v2.x,value);
31 ASSERT_EQ(v2.y,value);
32 v2.x = 3.5;
33 v2.y = 1.2;
34 v1 = v2;
35 ASSERT_EQ(v1.x,3.5);
36 ASSERT_EQ(v1.y,1.2);
37 ASSERT_EQ(v1,v2);
38 }
39 {
40 double value = 0.3;
41 Real2 v2;
42 v2 = value;
43 ASSERT_EQ(v2.x,value);
44 ASSERT_EQ(v2.x,value);
45 }
46 // Operator + and -
47 {
48 Real2 v2(1.2,4.5);
49 Real2 v1(1.3,2.3);
50 Real2 v3 = v1 + v2;
51 ASSERT_EQ(v3.x,2.5);
52 ASSERT_EQ(v3.y,6.8);
53 Real2 v4 = v3 - v2;
54 ASSERT_EQ(v4,v1);
55 }
56 // Operator * and /
57 {
58 Real2 v2(1.2,4.5);
59 Real2 v1(1.3,2.3);
60 Real2 v3 = v1 * v2;
61 ASSERT_EQ(v3.x,1.2*1.3);
62 ASSERT_EQ(v3.y,4.5*2.3);
63 Real2 v4 = v3 * 2.3;
64 ASSERT_EQ(v4.x,v3.x*2.3);
65 ASSERT_EQ(v4.y,v3.y*2.3);
66 Real2 v5 = v4 / 2.4;
67 ASSERT_EQ(v5.x,v4.x/2.4);
68 ASSERT_EQ(v5.y,v4.y/2.4);
69 }
70 {
71 Real3 v3(1.2,2.3,4.5);
72 Real2 v2(v3);

74 ASSERT_EQ(v2.y,2.3);
