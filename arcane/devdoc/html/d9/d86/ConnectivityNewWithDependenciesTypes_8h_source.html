101  class FaceToCellConnectivity
102  {
103  public:
104  FaceToCellConnectivity(IItemFamily* source_family,IItemFamily* target_family)
105  : m_face_family(nullptr)
106  , m_cell_family(nullptr) {
107  if (source_family->itemKind() != IK_Face || target_family->itemKind()!= IK_Cell)
108  throw FatalErrorException("FaceToCellIncrementalConnectivity must be created with face family as source and cell family as target. Exiting.");
109  m_face_family = dynamic_cast<FaceFamily*>(source_family);
110  m_cell_family = dynamic_cast<CellFamily*>(target_family);
111  // todo supprimer le cast, le type concret n'est plus nécessaire
112  }
113 
114  protected:
115 bool isFrontCell(ItemLocalId source_item,ItemLocalId target_item)
116 {
117  ItemInternal* face = m_face_family->itemsInternal()[source_item.localId()];
118  ItemInternal* cell = m_cell_family->itemsInternal()[target_item.localId()];
119  // Find if cell is front or back for given face
120  // get face and cell nodes
121  IMesh* mesh = m_face_family->mesh();
122  ConnectivityItemVector face_nodes_connectivity(mesh->itemFamilyNetwork()->getConnectivity(m_face_family,mesh->nodeFamily(), connectivityName(m_face_family,mesh->nodeFamily())));
123  ItemVectorView face_nodes = face_nodes_connectivity.connectedItems(Face(face));
124  ConnectivityItemVector cell_nodes_connectivity(mesh->itemFamilyNetwork()->getConnectivity(m_cell_family,mesh->nodeFamily(), connectivityName(m_cell_family,mesh->nodeFamily())));
125  ItemVectorView cell_nodes = cell_nodes_connectivity.connectedItems(Cell(cell));
126  ItemTypeInfo* cell_type_info = cell->typeInfo();
127  // Find which local_face
128  Integer face_index = -1;
129  Int64UniqueArray face_node_uids;
130  for (auto node : face_nodes) face_node_uids.add(node.uniqueId().asInt64());
131  std::set<Int64> face_nodes_set(face_node_uids.begin(),face_node_uids.end());
132  Int64UniqueArray local_face_node_uids;
133  for (Integer local_face_index = 0; local_face_index < cell_type_info->nbLocalFace() && face_index == -1; ++local_face_index)
134  {
135  local_face_node_uids.clear();
136  ItemTypeInfo::LocalFace local_face = cell_type_info->localFace(local_face_index);
137  if (local_face.nbNode() != face_nodes.size()) continue;
138  for (Integer local_node_index = 0; local_node_index < local_face.nbNode() ; ++local_node_index) local_face_node_uids.add(cell_nodes[local_face.node(local_node_index)].uniqueId().asInt64());
139  std::set<Int64> local_face_nodes_set(local_face_node_uids.begin(),local_face_node_uids.end());
140  if (local_face_nodes_set == face_nodes_set) face_index = local_face_index; // we found the face in the cell
141  }
142  if (face_index == -1) throw FatalErrorException(String::format("Face {0} does not belong to Cell {1}. Cannot connect. Exiting."));
143  if (mesh->dimension() == 1) // 1d mesh : back cell is local face 0, front is local face 1
144  return (face_index == 1);
145  else
146  {
147  UniqueArray<Int64> ordered_face_node_ids(face_node_uids.size());
148  if (mesh_utils::reorderNodesOfFace(local_face_node_uids,ordered_face_node_ids)) return true;
149  else return false;
150  }
151 }

211  void _addFrontCellToFace(ItemInternal* face, ItemInternal* cell){
212  _checkValidSourceTargetItems(face,cell);
213 
214  Integer nb_cell = nbConnectedItem(ItemLocalId(face));
215 
216  // SDP: les tests suivants sont imcompatibles avec le raffinement
217  // par couches
218  // SDC: dans l'etat le check ne peut etre fait car les flags sont positionnes par la connectivite legacy.
219  // On peut donc avoir une front_cell qui n'est pas dans cette connectivite. C'est donc legal et de plus
220  // le check va planter puisque connectedItemLocalId(ItemLocalId(face),1) va retourner -1...
221  // Dans le nouveau mode les check devront etre faits via des proprietes.
222  bool check_orientation = false; // SDC cf. ci-dessus
223  if (check_orientation){
224  if (face->flags() & ItemInternal::II_HasFrontCell){
225  ItemInternal* current_cell = m_cell_family->itemsInternal()[connectedItemLocalId(ItemLocalId(face),1)]; // FrontCell is the second connected cell
226  ARCANE_FATAL("Face already having a front cell."
227  " This is most probably due to the fact that the face"
228  " is connected to a reverse cell with a negative volume."
229  " Face={0}. new_cell={1} current_cell={2}",face->uniqueId().asInt64(),
230  cell->uniqueId().asInt64(),current_cell->uniqueId().asInt64()); // FullItemPrinter cannot be used here, the connectivities are not entirely set
231  }
232  }
233 
234  if (nb_cell>=2)
235  ARCANE_FATAL("face '{0}' already has two cells",face->uniqueId().asInt64());// FullItemPrinter cannot be used here, the connectivities are not entirely set
236 
237  // Si on a déjà une maille, il s'agit de la back cell.
238  Int32 iback_cell_lid = (nb_cell==1) ? face->cellLocalId(0) : NULL_ITEM_LOCAL_ID;
239  ItemLocalId back_cell_lid(iback_cell_lid);
240  ItemLocalId front_cell_lid(cell->localId());
241  _setBackAndFrontCells(face,back_cell_lid,front_cell_lid);
242 
243 
244  }

246  void _addBackCellToFace(ItemInternal* face, ItemInternal* cell){
247  _checkValidSourceTargetItems(face,cell);
248 
249  Integer nb_cell = nbConnectedItem(ItemLocalId(face));
250 
251  // SDP: les tests suivants sont imcompatibles avec le raffinement
252  // par couches
253  // SDC: dans l'etat le check ne peut etre fait car les flags sont positionnes par la connectivite legacy.
254  // On peut donc avoir une back_cell qui n'est pas dans cette connectivite. C'est donc legal et de plus
255  // le check va planter puisque connectedItemLocalId(ItemLocalId(face),0) va retourner -1.
256  // Dans le nouveau mode les check devront etre faits via des proprietes.
257  bool check_orientation = false; // SDC cf. ci-dessus
258  if (check_orientation){
259  if (face->flags() & ItemInternal::II_HasBackCell){
260  ItemInternal* current_cell = m_cell_family->itemsInternal()[connectedItemLocalId(ItemLocalId(face),0)]; // BackCell is the first connected cell
261  ARCANE_FATAL("Face already having a back cell."
262  " This is most probably due to the fact that the face"
263  " is connected to a reverse cell with a negative volume."
264  " Face={0}. new_cell={1} current_cell={2}",face->uniqueId().asInt64(),
265  cell->uniqueId().asInt64(),current_cell->uniqueId().asInt64()); // FullItemPrinter cannot be used here, the connectivities are not entirely set
266  }
267  }
268 
269  if (nb_cell>=2)
270  ARCANE_FATAL("face '{0}' already has two cells",face->uniqueId().asInt64());// FullItemPrinter cannot be used here, the connectivities are not entirely set
271  // Si on a déjà une maille, il s'agit de la front cell.
272  Int32 ifront_cell_lid = (nb_cell==1) ? face->cellLocalId(0) : NULL_ITEM_LOCAL_ID;
273 
274  ItemLocalId back_cell_lid(cell->localId());
275  ItemLocalId front_cell_lid(ifront_cell_lid);
276  _setBackAndFrontCells(face,back_cell_lid,front_cell_lid);
277 
278  }

282  void _setBackAndFrontCells(ItemInternal* face,ItemLocalId back_cell_lid,ItemLocalId front_cell_lid){
283  ItemLocalId face_lid(face->localId());
284  // Supprime toutes les mailles connectées => la méthode est mutualisée pour les ajouts ou suppressions
285  // TODO: optimiser en ne supprimant pas s'il n'y a pas besoin pour éviter
286  // des réallocations.
287  removeConnectedItems(face_lid);
288  Int32 mod_flags = 0;
289  if (front_cell_lid==NULL_ITEM_LOCAL_ID){
290  if (back_cell_lid!=NULL_ITEM_LOCAL_ID){
291  // Reste uniquement la back_cell
292  IncrementalItemConnectivity::addConnectedItem(face_lid,back_cell_lid); // add the class name for the case of Face to Cell connectivity. The class is overridden to handle family dependencies but the base method must be called here.
293  // add flags
294  mod_flags = (ItemInternal::II_Boundary | ItemInternal::II_HasBackCell | ItemInternal::II_BackCellIsFirst);
295  }
296  // Ici reste aucune maille mais comme on a tout supprimé il n'y a rien
297  // à faire
298  }
299  else if (back_cell_lid==NULL_ITEM_LOCAL_ID){
300  // Reste uniquement la front cell
301  IncrementalItemConnectivity::addConnectedItem(face_lid,front_cell_lid);
302  // add flags
303  mod_flags = (ItemInternal::II_Boundary | ItemInternal::II_HasFrontCell | ItemInternal::II_FrontCellIsFirst);
304  }
305  else{
306  // Il y a deux mailles connectées. La back_cell est toujours la première.
307  IncrementalItemConnectivity::addConnectedItem(face_lid,back_cell_lid);
308  IncrementalItemConnectivity::addConnectedItem(face_lid,front_cell_lid);
309  // add flags
310  mod_flags = (ItemInternal::II_HasFrontCell | ItemInternal::II_HasBackCell | ItemInternal::II_BackCellIsFirst);
311  }
312  Int32 face_flags = face->flags();
313  face_flags &= ~ItemInternal::II_InterfaceFlags;
314  face_flags |= mod_flags;
315  face->setFlags(face_flags);
316  }

318  void _removeConnectedItem(ItemInternal* face,ItemLocalId cell_to_remove_lid)
319  {
320  // Code taken from FaceFamily::removeCellFromFace where new and legacy are separated.
321  // Here only for new connectivity.
322  // This duplication allows to reach the specificity of Cell/Face connectivity through the unique interface removeConnectedItem (instead of removeCellFromFace)
323  _checkValidSourceTargetItems(face,m_cell_family->itemsInternal()[cell_to_remove_lid]);
324 
325  Integer nb_cell = nbConnectedItem(ItemLocalId(face));
326 
327 #ifdef ARCANE_CHECK
328  if (face->isSuppressed())
329  ARCANE_FATAL("Can not remove cell from destroyed face={0}",ItemPrinter(face));
330  if (nb_cell==0)
331  ARCANE_FATAL("Can not remove cell lid={0} from face uid={1} with no cell connected",
332  cell_to_remove_lid, face->uniqueId());
333 #endif /* ARCANE_CHECK */
334 
335  Integer nb_cell_after = nb_cell-1;
336  const Int32 null_cell_lid = NULL_ITEM_LOCAL_ID;
337 
339 
340  // OFF AMR
341  if (nb_cell_after!=0){
342  Int32 cell0 = face->cellLocalId(0);
343  Int32 cell1 = face->cellLocalId(1);
344  // On avait obligatoirement deux mailles connectées avant,
345  // donc la back_cell est la maille 0, la front cell la maille 1
346  if (cell0==cell_to_remove_lid){
347  // Reste la front cell
348  _setBackAndFrontCells(face,ItemLocalId(null_cell_lid),ItemLocalId(cell1));
349  }
350  else{
351  // Reste la back cell
352  _setBackAndFrontCells(face,ItemLocalId(cell0),ItemLocalId(null_cell_lid));
353  }
354  }
355  else{
356  _setBackAndFrontCells(face,ItemLocalId(null_cell_lid),ItemLocalId(null_cell_lid));
357  }
358  }
