140 ThreadPrivate(ThreadPrivateStorage* key)
141 : m_storage(key,this)
142 {
143 }
144
145 ~ThreadPrivate()
146 {
147 for( T* item : m_allocated_items )
148 delete item;
149 }
150
151 public:
153 T* item()
154 {

156 }
157
158 private:
159
160 void* createInstance() override
161 {
162 T* new_ptr = new T();
163 new_ptr->build();
164 m_allocated_items.push_back(new_ptr);
165 return new_ptr;
166 }
167
168 private:
169 std::vector<T*> m_allocated_items;
170 ThreadPrivateBase m_storage;
