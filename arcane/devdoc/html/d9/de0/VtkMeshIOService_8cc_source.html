262bool VtkFile::
263isEmptyNextLine()
264{
265 m_is_init = true;
266
267 // On veut que getNextLine lise une nouvelle ligne.
268 // (on met à false dans le cas où cette méthode serai
269 // appelée plusieurs fois à la suite).
270 m_need_reread_current_line = false;
271
272 // Si l'on est arrivé à la fin du fichier lors du précédent appel de cette méthode ou
273 // de getNextLine, on envoie une erreur.
274 if (m_is_eof) {
275 throw IOException("VtkFile::isEmptyNextLine()", "Unexpected EndOfFile");
276 }
277
278 if (m_stream->good()) {
279 // Le getline s'arrete (par défaut) au char '\n' et ne l'inclus pas dans le buf
280 // mais le remplace par '\0'.
281 m_stream->getline(m_buf, sizeof(m_buf) - 1);
282
283 // Si on arrive au bout du fichier, on return true (pour dire oui, il y a une ligne vide,
284 // à l'appelant de gérer ça).
285 if (m_stream->eof()) {
286 m_is_eof = true;
287 return true;
288 }
289
290 // Sous Windows, une ligne vide commence par \r.
291 // getline remplace \n par \0, que ce soit sous Windows ou Linux.
292 if (m_buf[0] == '\r' || m_buf[0] == '\0') {
293 getNextLine();
294
295 // On demande à ce que le prochain appel à getNextLine renvoie la ligne
296 // qui vient tout juste d'être bufferisée.
297 m_need_reread_current_line = true;
298 return true;
299 }
300 else {
301 bool is_comment = true;
302
303 // On retire le commentaire, s'il y en a un, en remplaçant '#' par '\0'.
304 for (int i = 0; i < BUFSIZE && m_buf[i] != '\0'; ++i) {
305 if (!isspace(m_buf[i]) && m_buf[i] != '#' && is_comment) {
306 is_comment = false;
307 }
308 if (m_buf[i] == '#') {
309 m_buf[i] = '\0';
310 break;
311 }
312 }
313
314 // Si ce n'est pas un commentaire, on supprime juste le '\r' final (si windows).
315 if (!is_comment) {
316 // Supprime le '\r' final
317 for (int i = 0; i < BUFSIZE && m_buf[i] != '\0'; ++i) {
318 if (m_buf[i] == '\r') {
319 m_buf[i] = '\0';
320 break;
321 }
322 }
323 }
324
325 // Si c'était un commentaire, on recherche la prochaine ligne "valide"
326 // en appelant getNextLine.
327 else {
328 getNextLine();
329 }
330 }
331 m_need_reread_current_line = true;
332 return false;
333 }
334 throw IOException("VtkFile::isEmptyNextLine()", "Not Good");
335}

345const char* VtkFile::
346getNextLine()
347{
348 m_is_init = true;
349
350 // On return le buffer actuel, si celui-ci n'a pas été utilisé.
351 if (m_need_reread_current_line) {
352 m_need_reread_current_line = false;
353 return getCurrentLine();
354 }
355
356 // Si l'on est arrivé à la fin du fichier lors du précédent appel de cette méthode ou
357 // de isEmptyNextLine, on envoie une erreur.
358 if (m_is_eof) {
359 throw IOException("VtkFile::isEmptyNextLine()", "Unexpected EndOfFile");
360 }
361
362 while (m_stream->good()) {
363 // Le getline s'arrete (par défaut) au char '\n' et ne l'inclus pas dans le buf mais le remplace par '\0'.
364 m_stream->getline(m_buf, sizeof(m_buf) - 1);
365
366 // Si on arrive au bout du fichier, on return le buffer avec \0 au début (c'est à l'appelant d'appeler
367 // isEof() pour savoir si le fichier est fini ou non).
368 if (m_stream->eof()) {
369 m_is_eof = true;
370 m_buf[0] = '\0';
371 return m_buf;
372 }
373
374 bool is_comment = true;
375
376 // Sous Windows, une ligne vide commence par \r.
377 // getline remplace \n par \0, que ce soit sous Windows ou Linux.
378 if (m_buf[0] == '\0' || m_buf[0] == '\r')
379 continue;
380
381 // On retire le commentaire, s'il y en a un, en remplaçant '#' par '\0'.
382 for (int i = 0; i < BUFSIZE && m_buf[i] != '\0'; ++i) {
383 if (!isspace(m_buf[i]) && m_buf[i] != '#' && is_comment) {
384 is_comment = false;
385 }
386 if (m_buf[i] == '#') {
387 m_buf[i] = '\0';
388 break;
389 }
390 }
391
392 // Si ce n'est pas un commentaire, on supprime juste le '\r' final (si windows).
393 if (!is_comment) {
394 // Supprime le '\r' final
395 for (int i = 0; i < BUFSIZE && m_buf[i] != '\0'; ++i) {
396 if (m_buf[i] == '\r') {
397 m_buf[i] = '\0';
398 break;
399 }
400 }
401 return m_buf;
402 }
403 }
404 throw IOException("VtkFile::getNextLine()", "Not good");
405}

487template <class T>
488void VtkFile::
489getBinary(T& value)
490{
491 constexpr size_t sizeofT = sizeof(T);
492
493 // Le fichier VTK est en big endian et les CPU actuels sont en little endian.
494 Byte big_endian[sizeofT];
495 Byte little_endian[sizeofT];
496
497 // On lit les 'sizeofT' prochains octets que l'on met dans big_endian.
498 m_stream->read((char*)big_endian, sizeofT);
499
500 // On transforme le big_endian en little_endian.
501 for (size_t i = 0; i < sizeofT; i++) {
502 little_endian[sizeofT - 1 - i] = big_endian[i];
503 }
504
505 // On 'cast' la liste d'octet en type 'T'.
506 T* conv = new (little_endian) T;
507 value = *conv;
508}

610bool VtkMeshIOService::
611readMesh(IPrimaryMesh* mesh, const String& file_name, const String& dir_name, bool use_internal_partition)
612{
613 std::ifstream ifile(file_name.localstr(), std::ifstream::binary);
614
615 if (!ifile) {
616 error() << "Unable to read file '" << file_name << "'";
617 return true;
618 }
619
620 debug() << "Fichier ouvert : " << file_name.localstr();
621
622 VtkFile vtk_file(&ifile);
623 const char* buf = 0;
624
625 // Lecture de la description
626 // Lecture title.
627 String title = vtk_file.getNextLine();
628
629 info() << "Titre du fichier VTK : " << title.localstr();
630
631 // Lecture format.
632 String format = vtk_file.getNextLine();
633
634 debug() << "Format du fichier VTK : " << format.localstr();
635
636 if (VtkFile::isEqualString(format, "BINARY")) {
637 vtk_file.setIsBinaryFile(true);
638 }
639
640 eMeshType mesh_type = VTK_MT_Unknown;
641
642 // Lecture du type de maillage
643 // TODO: en parallèle, avec use_internal_partition vrai, seul le processeur 0
644 // lit les données. Dans ce cas, inutile que les autres ouvrent le fichier.
645 {
646 buf = vtk_file.getNextLine();
647
648 std::istringstream mesh_type_line(buf);
649 std::string dataset_str;
650 std::string mesh_type_str;
651
652 mesh_type_line >> ws >> dataset_str >> ws >> mesh_type_str;
653
654 vtk_file.checkString(dataset_str, "DATASET");
655
656 if (VtkFile::isEqualString(mesh_type_str, "STRUCTURED_GRID")) {
657 mesh_type = VTK_MT_StructuredGrid;
658 }
659
660 if (VtkFile::isEqualString(mesh_type_str, "UNSTRUCTURED_GRID")) {
661 mesh_type = VTK_MT_UnstructuredGrid;
662 }
663
664 if (mesh_type == VTK_MT_Unknown) {
665 error() << "Support exists only for 'STRUCTURED_GRID' and 'UNSTRUCTURED_GRID' formats (format=" << mesh_type_str << "')";
666 return true;
667 }
668 }
669 debug() << "Lecture en-tête OK";
670
671 bool ret = true;
672 switch (mesh_type) {
673 case VTK_MT_StructuredGrid:
674 ret = _readStructuredGrid(mesh, vtk_file, use_internal_partition);
675 break;
676
677 case VTK_MT_UnstructuredGrid:
678 ret = _readUnstructuredGrid(mesh, vtk_file, use_internal_partition);
679 debug() << "Lecture _readUnstructuredGrid OK";
680 if (!ret) {
681 // Tente de lire le fichier des faces s'il existe
682 _readFacesMesh(mesh, file_name + "faces.vtk", dir_name, use_internal_partition);
683 debug() << "Lecture _readFacesMesh OK";
684 }
685 break;
686
687 case VTK_MT_Unknown:
688 break;
689 }
690 /*while ( (buf=vtk_file.getNextLine()) != 0 ){
691 info() << " STR " << buf;
692 }*/
693
694 ifile.close();
695 return ret;
696}

709bool VtkMeshIOService::
710_readStructuredGrid(IPrimaryMesh* mesh, VtkFile& vtk_file, bool use_internal_partition)
711{
712 // Lecture du nombre de points: DIMENSIONS nx ny nz
713 const char* buf = nullptr;
714 Integer nb_node_x = 0;
715 Integer nb_node_y = 0;
716 Integer nb_node_z = 0;
717 {
718 buf = vtk_file.getNextLine();
719 std::istringstream iline(buf);
720 std::string dimension_str;
721 iline >> ws >> dimension_str >> ws >> nb_node_x >> ws >> nb_node_y >> ws >> nb_node_z;
722
723 if (!iline) {
724 error() << "Syntax error while reading grid dimensions";
725 return true;
726 }
727
728 vtk_file.checkString(dimension_str, "DIMENSIONS");
729 if (nb_node_x <= 1 || nb_node_y <= 1 || nb_node_z <= 1) {
730 error() << "Invalid dimensions: x=" << nb_node_x << " y=" << nb_node_y << " z=" << nb_node_z;
731 return true;
732 }
733 }
734 info() << " Infos: " << nb_node_x << " " << nb_node_y << " " << nb_node_z;
735 Integer nb_node = nb_node_x * nb_node_y * nb_node_z;
736
737 // Lecture du nombre de points: POINTS nb float
738 std::string float_str;
739 {
740 buf = vtk_file.getNextLine();
741 std::istringstream iline(buf);
742 std::string points_str;
743 Integer nb_node_read = 0;
744 iline >> ws >> points_str >> ws >> nb_node_read >> ws >> float_str;
745 if (!iline) {
746 error() << "Syntax error while reading grid dimensions";
747 return true;
748 }
749 vtk_file.checkString(points_str, "POINTS");
750 if (nb_node_read != nb_node) {
751 error() << "Number of invalid nodes: expected=" << nb_node << " found=" << nb_node_read;
752 return true;
753 }
754 }
755
756 Int32 rank = mesh->parallelMng()->commRank();
757
758 Integer nb_cell_x = nb_node_x - 1;
759 Integer nb_cell_y = nb_node_y - 1;
760 Integer nb_cell_z = nb_node_z - 1;
761
762 if (use_internal_partition && rank != 0) {
763 nb_node_x = 0;
764 nb_node_y = 0;
765 nb_node_z = 0;
766 nb_cell_x = 0;
767 nb_cell_y = 0;
768 nb_cell_z = 0;
769 }
770
771 const Integer nb_node_yz = nb_node_y * nb_node_z;
772 const Integer nb_node_xy = nb_node_x * nb_node_y;
773
774 Integer nb_cell = nb_cell_x * nb_cell_y * nb_cell_z;
775 UniqueArray<Int32> cells_local_id(nb_cell);
776
777 // Creation du maillage
778 {
779 UniqueArray<Integer> nodes_unique_id(nb_node);
780
781 info() << " NODE YZ = " << nb_node_yz;
782 // Création des noeuds
783 //Integer nb_node_local_id = 0;
784 {
785 Integer node_local_id = 0;
786 for (Integer x = 0; x < nb_node_x; ++x) {
787 for (Integer z = 0; z < nb_node_z; ++z) {
788 for (Integer y = 0; y < nb_node_y; ++y) {
789
790 Integer node_unique_id = y + (z)*nb_node_y + x * nb_node_y * nb_node_z;
791
792 nodes_unique_id[node_local_id] = node_unique_id;
793
794 ++node_local_id;
795 }
796 }
797 }
798 //nb_node_local_id = node_local_id;
799 //warning() << " NB NODE LOCAL ID=" << node_local_id;
800 }
801
802 // Création des mailles
803
804 // Infos pour la création des mailles
805 // par maille: 1 pour son unique id,
806 // 1 pour son type,
807 // 8 pour chaque noeud
808 UniqueArray<Int64> cells_infos(nb_cell * 10);
809
810 {
811 Integer cell_local_id = 0;
812 Integer cells_infos_index = 0;
813
814 // Normalement ne doit pas arriver car les valeurs de nb_node_x et
815 // nb_node_y sont testées lors de la lecture.
816 if (nb_node_xy == 0)
817 ARCANE_FATAL("Null value for nb_node_xy");
818
819 //Integer index = 0;
820 for (Integer z = 0; z < nb_cell_z; ++z) {
821 for (Integer y = 0; y < nb_cell_y; ++y) {
822 for (Integer x = 0; x < nb_cell_x; ++x) {
823 Integer current_cell_nb_node = 8;
824
825 //Integer cell_unique_id = y + (z)*nb_cell_y + x*nb_cell_y*nb_cell_z;
826 Int64 cell_unique_id = x + y * nb_cell_x + z * nb_cell_x * nb_cell_y;
827
828 cells_infos[cells_infos_index] = IT_Hexaedron8;
829 ++cells_infos_index;
830
831 cells_infos[cells_infos_index] = cell_unique_id;
832 ++cells_infos_index;
833
834 //Integer base_id = y + z*nb_node_y + x*nb_node_yz;
835 Integer base_id = x + y * nb_node_x + z * nb_node_xy;
836 cells_infos[cells_infos_index + 0] = nodes_unique_id[base_id];
837 cells_infos[cells_infos_index + 1] = nodes_unique_id[base_id + 1];
838 cells_infos[cells_infos_index + 2] = nodes_unique_id[base_id + nb_node_x + 1];
839 cells_infos[cells_infos_index + 3] = nodes_unique_id[base_id + nb_node_x + 0];
840 cells_infos[cells_infos_index + 4] = nodes_unique_id[base_id + nb_node_xy];
841 cells_infos[cells_infos_index + 5] = nodes_unique_id[base_id + nb_node_xy + 1];
842 cells_infos[cells_infos_index + 6] = nodes_unique_id[base_id + nb_node_xy + nb_node_x + 1];
843 cells_infos[cells_infos_index + 7] = nodes_unique_id[base_id + nb_node_xy + nb_node_x + 0];
844 cells_infos_index += current_cell_nb_node;
845 cells_local_id[cell_local_id] = cell_local_id;

854 mesh->endAllocate();

856 // Positionne les coordonnées
857 {
858 UniqueArray<Real3> coords(nb_node);
859
860 if (vtk_file.isEqualString(float_str, "int")) {
861 for (Integer z = 0; z < nb_node_z; ++z) {
862 for (Integer y = 0; y < nb_node_y; ++y) {
863 for (Integer x = 0; x < nb_node_x; ++x) {
864 Real nx = vtk_file.getInt();
865 Real ny = vtk_file.getInt();
866 Real nz = vtk_file.getInt();
867 Integer node_unique_id = x + y * nb_node_x + z * nb_node_xy;
868 coords[node_unique_id] = Real3(nx, ny, nz);
869 }
870 }
871 }
872 }
873 else if (vtk_file.isEqualString(float_str, "float")) {
874 for (Integer z = 0; z < nb_node_z; ++z) {
875 for (Integer y = 0; y < nb_node_y; ++y) {
876 for (Integer x = 0; x < nb_node_x; ++x) {
877 Real nx = vtk_file.getFloat();
878 Real ny = vtk_file.getFloat();
879 Real nz = vtk_file.getFloat();
880 Integer node_unique_id = x + y * nb_node_x + z * nb_node_xy;
881 coords[node_unique_id] = Real3(nx, ny, nz);
882 }
883 }
884 }
885 }
886 else if (vtk_file.isEqualString(float_str, "double")) {
887 for (Integer z = 0; z < nb_node_z; ++z) {
888 for (Integer y = 0; y < nb_node_y; ++y) {
889 for (Integer x = 0; x < nb_node_x; ++x) {
890 Real nx = vtk_file.getDouble();
891 Real ny = vtk_file.getDouble();
892 Real nz = vtk_file.getDouble();
893 Integer node_unique_id = x + y * nb_node_x + z * nb_node_xy;
894 coords[node_unique_id] = Real3(nx, ny, nz);
895 }
896 }
897 }
898 }
899 else {
900 throw IOException("_readStructuredGrid", "Invalid type name");
901 }
902
903 VariableNodeReal3& nodes_coord_var(mesh->nodesCoordinates());
904 ENUMERATE_NODE (inode, mesh->allNodes()) {
905 Node node = *inode;
906 nodes_coord_var[inode] = coords[node.uniqueId().asInt32()];
907 }
908 }
909 }

911 // Créé les groupes de faces des côtés du parallélépipède
912 {
913 Int32UniqueArray xmin_surface_lid;
914 Int32UniqueArray xmax_surface_lid;
915 Int32UniqueArray ymin_surface_lid;
916 Int32UniqueArray ymax_surface_lid;
917 Int32UniqueArray zmin_surface_lid;
918 Int32UniqueArray zmax_surface_lid;
919
920 ENUMERATE_FACE (iface, mesh->allFaces()) {
921 Face face = *iface;
922 Integer face_local_id = face.localId();
923 bool is_xmin = true;
924 bool is_xmax = true;
925 bool is_ymin = true;
926 bool is_ymax = true;
927 bool is_zmin = true;
928 bool is_zmax = true;
929 for (Node node : face.nodes() ) {
930 Int64 node_unique_id = node.uniqueId().asInt64();
931 Int64 node_z = node_unique_id / nb_node_xy;
932 Int64 node_y = (node_unique_id - node_z * nb_node_xy) / nb_node_x;
933 Int64 node_x = node_unique_id - node_z * nb_node_xy - node_y * nb_node_x;
934 if (node_x != 0)
935 is_xmin = false;
936 if (node_x != (nb_node_x - 1))
937 is_xmax = false;
938 if (node_y != 0)
939 is_ymin = false;
940 if (node_y != (nb_node_y - 1))
941 is_ymax = false;
942 if (node_z != 0)
943 is_zmin = false;
944 if (node_z != (nb_node_z - 1))
945 is_zmax = false;
946 }
947 if (is_xmin)
948 xmin_surface_lid.add(face_local_id);
949 if (is_xmax)
950 xmax_surface_lid.add(face_local_id);
951 if (is_ymin)
952 ymin_surface_lid.add(face_local_id);
953 if (is_ymax)
954 ymax_surface_lid.add(face_local_id);
955 if (is_zmin)
956 zmin_surface_lid.add(face_local_id);
957 if (is_zmax)
958 zmax_surface_lid.add(face_local_id);
959 }
960 _createFaceGroup(mesh, "XMIN", xmin_surface_lid);
961 _createFaceGroup(mesh, "XMAX", xmax_surface_lid);
962 _createFaceGroup(mesh, "YMIN", ymin_surface_lid);
963 _createFaceGroup(mesh, "YMAX", ymax_surface_lid);
964 _createFaceGroup(mesh, "ZMIN", zmin_surface_lid);
965 _createFaceGroup(mesh, "ZMAX", zmax_surface_lid);
966 }
967
968 _readMetadata(mesh, vtk_file);
969
970 // Maintenant, regarde s'il existe des données associées aux fichier
971 bool r = _readData(mesh, vtk_file, use_internal_partition, IK_Cell, cells_local_id, nb_node);
972 if (r)
973 return r;
974
975 return false;
976}

1908void VtkLegacyMeshWriter::
1909_writeMeshToFile(IMesh* mesh, const String& file_name, eItemKind cell_kind)
1910{
1911 std::ofstream ofile(file_name.localstr());
1912 ofile.precision(FloatInfo<Real>::maxDigit());
1913 if (!ofile)
1914 throw IOException("VtkMeshIOService::writeMeshToFile(): Unable to open file");
1915 ofile << "# vtk DataFile Version 2.0\n";
1916 ofile << "Maillage Arcane\n";
1917 ofile << "ASCII\n";
1918 ofile << "DATASET UNSTRUCTURED_GRID\n";
1919 UniqueArray<Integer> nodes_local_id_to_current(mesh->itemFamily(IK_Node)->maxLocalId());
1920 nodes_local_id_to_current.fill(NULL_ITEM_ID);
1921
1922 Integer nb_node = mesh->nbNode();
1923 IItemFamily* cell_kind_family = mesh->itemFamily(cell_kind);
1924 Integer nb_cell_kind = cell_kind_family->nbItem();
1925
1926 // Sauve les nœuds
1927 {
1928 ofile << "POINTS " << nb_node << " double\n";
1929 VariableNodeReal3& coords(mesh->toPrimaryMesh()->nodesCoordinates());
1930 Integer node_index = 0;
1931 ENUMERATE_NODE (inode, mesh->allNodes()) {
1932 const Node& node = *inode;
1933 nodes_local_id_to_current[node.localId()] = node_index;
1934 Real3 xyz = coords[inode];
1935 ofile << xyz.x << ' ' << xyz.y << ' ' << xyz.z << '\n';
1936 ++node_index;
1937 }
1938 }
1939
1940 // Sauve les mailles ou faces
1941 {
1942 Integer nb_node_cell_kind = nb_cell_kind;
1943 ENUMERATE_ITEMWITHNODES(iitem, cell_kind_family->allItems())
1944 {
1945 nb_node_cell_kind += (*iitem).nbNode();
1946 }
1947 ofile << "CELLS " << nb_cell_kind << ' ' << nb_node_cell_kind << "\n";
1948 ENUMERATE_ITEMWITHNODES(iitem, cell_kind_family->allItems())
1949 {
1950 const ItemWithNodes& item = *iitem;
1951 Integer item_nb_node = item.nbNode();
1952 ofile << item_nb_node;
1953 for (NodeLocalId inode_lid : item.nodes() ) {
1954 ofile << ' ' << nodes_local_id_to_current[inode_lid];
1955 }
1956 ofile << '\n';
1957 }
1958 // Le type doit être coherent avec celui de vtkCellType.h
1959 ofile << "CELL_TYPES " << nb_cell_kind << "\n";
1960 ENUMERATE_ (ItemWithNodes, iitem, cell_kind_family->allItems()) {
1961 Int16 arcane_type = (*iitem).type();
1962 int type = arcaneToVtkCellType(arcane_type);
1963 ofile << type << '\n';
1964 }
1965 }
1966
1967 // Si on est dans le maillage des mailles, sauve les groupes de noeuds.
1968 if (cell_kind == IK_Cell) {
1969 ofile << "POINT_DATA " << nb_node << "\n";
1970 _saveGroups(mesh->itemFamily(IK_Node), ofile);
1971 }
1972
1973 // Sauve les groupes de mailles
1974 ofile << "CELL_DATA " << nb_cell_kind << "\n";
1975 _saveGroups(mesh->itemFamily(cell_kind), ofile);
1976}
