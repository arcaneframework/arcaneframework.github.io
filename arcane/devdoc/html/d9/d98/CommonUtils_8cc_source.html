42toNativeStream(const NativeStream& v)
43{
44 cudaStream_t* s = reinterpret_cast<cudaStream_t*>(v.m_native_pointer);

50cudaStream_t CudaUtils::
51toNativeStream(const RunQueue* queue)
52{
53 eExecutionPolicy p = eExecutionPolicy::None;
54 if (queue)
55 p = queue->executionPolicy();
56 if (p != eExecutionPolicy::CUDA)
57 ARCCORE_FATAL("RunQueue is not a CUDA queue");
58 return toNativeStream(queue->_internalNativeStream());
59}

138void DeviceStorageBase::
139_copyToAsync(Span<std::byte> destination, Span<const std::byte> source, const RunQueue& queue)
140{
141#if defined(ARCCORE_COMPILING_CUDA)
142 cudaStream_t stream = Impl::CudaUtils::toNativeStream(queue);
143 ARCANE_CHECK_CUDA(::cudaMemcpyAsync(destination.data(), source.data(), source.size(), cudaMemcpyDeviceToHost, stream));
144#elif defined(ARCCORE_COMPILING_HIP)
145 hipStream_t stream = Impl::HipUtils::toNativeStream(queue);
146 ARCANE_CHECK_HIP(::hipMemcpyAsync(destination.data(), source.data(), source.size(), hipMemcpyDefault, stream));
147#else
148 ARCCORE_UNUSED(destination);
149 ARCCORE_UNUSED(source);
150 ARCCORE_UNUSED(queue);
151 ARCCORE_FATAL("No valid implementation for copy");
152#endif
153}
