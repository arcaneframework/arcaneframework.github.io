140class CartesianMeshPartitionerService
141: public BasicService
142, public IMeshPartitionerBase
143{
144 public:
145 explicit CartesianMeshPartitionerService(const ServiceBuildInfo& sbi)
146 : BasicService(sbi){}
147 public:
148 void build() override {}
149 void notifyEndPartition() override {}
150 IPrimaryMesh* primaryMesh() override { return mesh()->toPrimaryMesh(); }
151 void partitionMesh(bool initial_partition) override
152 {
153 if (!initial_partition)
154 return;
155 IMesh* mesh = this->mesh();
156 IParallelMng* pm = mesh->parallelMng();
157 Int32 nb_rank = pm->commSize();
158 IItemFamily* cell_family = mesh->cellFamily();
159 VariableItemInt32& cells_new_owner = cell_family->itemsNewOwner();
160 ItemGroup own_cells = cell_family->allItems().own();
161 Int64 nb_cell = own_cells.size();
162 Int64 nb_bloc = nb_rank * 3;
163 Int64 cell_index = 0;
164 info() << "Partitioning with 'CartesianMeshPartitionerService' nb_rank=" << nb_rank;
165 ENUMERATE_CELL(icell,mesh->ownCells()){
166 Cell cell = *icell;
167 // Utilise des Int64 plutôt que des Int32 pour être sur de ne pas déborder.
168 Int64 new_owner = ((cell_index * nb_bloc) / nb_cell) % nb_rank;
169 cells_new_owner[cell] = CheckedConvert::toInt32(new_owner);
170 ++cell_index;
171 }
172 cells_new_owner.synchronize();
173 mesh->utilities()->changeOwnersFromCells();
174 }
175};

334void CartesianMeshTesterModule::
335init()
336{
337 IMesh* mesh = defaultMesh();
338
339 IItemFamily* cell_family = mesh->cellFamily();
340 Int32UniqueArray ids(1);
341 ids[0] = 0;
342 cell_family->createGroup("CELL0",ids,true);
343 ids[0] = 1;
344 cell_family->createGroup("CELL1",ids,true);
345 ids[0] = 2;
346 cell_family->createGroup("CELL2",ids,true);
347 IItemFamily* face_family = defaultMesh()->faceFamily();
348 ids[0] = 0;
349 face_family->createGroup("FACE0",ids,true);
350 ids[0] = 1;
351 face_family->createGroup("FACE1",ids,true);
352 ids[0] = 2;
353 face_family->createGroup("FACE2",ids,true);
354 ids[0] = 3;
355 face_family->createGroup("FACE3",ids,true);
356 ids[0] = 4;
357 face_family->createGroup("FACE4",ids,true);
358 ids[0] = 5;
359 face_family->createGroup("FACE5",ids,true);
360
361 // Calcule le centre des mailles
362 {
363 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
364 ENUMERATE_CELL(icell,allCells()){
365 Cell cell = *icell;
366 Real3 center;
367 for( NodeLocalId inode : cell.nodeIds() )
368 center += nodes_coord[inode];
369 center /= cell.nbNode();
370 m_cell_center[icell] = center;
371 }
372 }
373
374 // Calcule le centre des faces
375 {
376 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
377 ENUMERATE_FACE(iface,allFaces()){
378 Face face = *iface;
379 Real3 center;
380 for( NodeLocalId inode : face.nodeIds() )
381 center += nodes_coord[inode];
382 center /= face.nbNode();
383 m_face_center[iface] = center;
384 }
385 }
386
387 m_cartesian_mesh = ICartesianMesh::getReference(mesh);
388 m_cartesian_mesh->computeDirections();
389
390 m_utils = makeRef(new CartesianMeshTestUtils(m_cartesian_mesh,acceleratorMng()));
391 m_utils_v2 = makeRef(new CartesianMeshV2TestUtils(m_cartesian_mesh));
392
393 // Initialise la densité.
394 // On met une densité de 1.0 à l'intérieur
395 // et on ajoute une densité de 5.0 pour chaque direction dans les
396 // mailles de bord.
397 m_density.fill(1.0);
398 Integer nb_dir = defaultMesh()->dimension();
399 for( Integer idir=0; idir<nb_dir; ++idir){
400 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
401 Integer nb_boundary1 = 0;
402 Integer nb_boundary2 = 0;
403 ENUMERATE_CELL(icell,cdm.innerCells()){
404 DirCell cc(cdm.cell(*icell));
405 Cell next = cc.next();
406 Cell prev = cc.previous();
407 if (next.null() || prev.null()){
408 // Maille au bord. J'ajoute de la densité.
409 // Ne devrait pas arriver car on est sur les innerCells()
410 ++nb_boundary1;
411 m_density[icell] += 5.0;
412 }
413 }
414 // Parcours les mailles frontières pour la direction
415 ENUMERATE_CELL(icell,cdm.outerCells()){
416 DirCell cc(cdm[icell]);
417 if (icell.index()<5)
418 info() << "CELL: cell=" << ItemPrinter(*icell)
419 << " next=" << ItemPrinter(cc.next())
420 << " previous=" << ItemPrinter(cc.previous());
421 // Maille au bord. J'ajoute de la densité.
422 ++nb_boundary2;
423 m_density[icell] += 5.0;
424 }
425
426 info() << "NB_BOUNDARY1=" << nb_boundary1 << " NB_BOUNDARY2=" << nb_boundary2;
427 }
428
429 m_utils->testAll(false);
430 m_utils_v2->testAll();
431 _checkFaceUniqueIdsAreContiguous();
432 _testXmlInfos();
433 _testGridPartitioning();
434 _printCartesianMeshInfos();
435 _checkSpecificApplyOperator();
436}

502void CartesianMeshTesterModule::
503_compute1()
504{
505 // Pour test, on parcours les N directions
506 // et pour chaque maille, on modifie sa densité
507 // par la formule new_density = (density+density_next+density_prev) / 3.0.
508
509 // Effectue l'operation en deux fois. Une premiere sur les
510 // mailles internes, et une deuxieme sur les mailles externes.
511 // Du coup, il faut passer par une variable intermediaire (m_old_density)
512 // mais on evite un test dans la boucle principale
513 Integer nb_dir = defaultMesh()->dimension();
514 for( Integer idir=0; idir<nb_dir; ++idir){
515 m_old_density.copy(m_density);
516 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
517 // Travail sur les mailles internes
518 ENUMERATE_CELL(icell,cdm.innerCells()){
519 DirCell cc(cdm.cell(*icell));
520 Cell next = cc.next();
521 Cell prev = cc.previous();
522 Real d = m_old_density[icell] + m_old_density[next] + m_old_density[prev];
523 m_density[icell] = d / 3.0;
524 }
525 // Travail sur les mailles externes
526 // Test si la maille avant ou apres est nulle.
527 ENUMERATE_CELL(icell,cdm.outerCells()){
528 DirCell cc(cdm[icell]);
529 Cell next = cc.next();
530 Cell prev = cc.previous();
531 Real d = m_old_density[icell];
532 Integer n = 1;
533 if (!next.null()){
534 d += m_old_density[next];
535 ++n;
536 }
537 if (!prev.null()){
538 d += m_old_density[prev];
539 ++n;
540 }
541 m_density[icell] = d / n;
542 }
543 }
544
545 {
546 Int64 to_add = m_global_iteration();
547 ENUMERATE_(Face,iface,ownFaces()){
548 Int64 uid = iface->uniqueId();
549 m_faces_uid[iface] = uid + to_add;
550 }
551 m_faces_uid.synchronize();
552 ENUMERATE_(Face,iface,allFaces()){
553 Face face(*iface);
554 Int64 uid = face.uniqueId();
555 Int64 expected_value = uid + to_add;
556 if (expected_value!=m_faces_uid[iface])
557 ARCANE_FATAL("Bad FaceUid face={0} expected={1} value={2}",ItemPrinter(face),expected_value,m_faces_uid[iface]);
558 }
559 }
560}

564
565void CartesianMeshTesterModule::
566_compute2()
567{
568 // Pour test, on parcours les N directions
569 // et pour chaque maille, on modifie sa densité
570 // par la formule new_density = (density+density_next+density_prev) / 3.0.
571
572 // A noter que cette methode ne donne pas le meme comportement que
573 // _compute1() car les mailles de bord et internes sont mises à jour
574 // dans un ordre différent.
575 Integer nb_dir = defaultMesh()->dimension();
576 for( Integer idir=0; idir<nb_dir; ++idir){
577 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
578 // Travail sur toutes les mailles
579 ENUMERATE_CELL(icell,cdm.allCells()){
580 DirCell cc(cdm[icell]);
581 Cell next = cc.next();
582 Cell prev = cc.previous();
583 Real d = m_density[icell];
584 Integer n = 1;
585 if (!next.null()){
586 d += m_density[next];
587 ++n;
588 }
589 if (!prev.null()){
590 d += m_density[prev];
591 ++n;
592 }
593 m_density[icell] = d / n;
594 }
595 }

624 if (!options()->checkContiguousFaceUniqueIds())

630 total_nb_face = parallelMng()->reduce(Parallel::ReduceSum,total_nb_face);
631 info() << "TotalNbFace=" << total_nb_face;
632 ENUMERATE_(Face,iface,allFaces()){
633 Face face = *iface;
634 if (face.uniqueId()>=total_nb_face)
635 ARCANE_FATAL("FaceUniqueId is too big: uid={0} total_nb_face={1}",
636 face.uniqueId(),total_nb_face);
637 }

692 info() << "UnstructuredMeshFileName=" << file_name;
693
694 Arcane::ISubDomain* sd = subDomain();
695 Arcane::ICartesianMesh* cartesian_mesh = m_cartesian_mesh;
696 Arcane::IMesh* current_mesh = cartesian_mesh->mesh();
697 Arcane::IParallelMng* pm = current_mesh->parallelMng();
698
699 Arcane::MeshReaderMng reader_mng(sd);
700 Arcane::IMesh* new_mesh = reader_mng.readMesh("UnstructuredMesh2",file_name,pm);
701 info() << "MESH=" << new_mesh;
702
703 // Création du service de partitionnement
704 Arcane::ServiceBuilder<Arcane::IGridMeshPartitioner> sbuilder(sd);
705 auto partitioner_ref = sbuilder.createReference("SimpleGridMeshPartitioner",new_mesh);
706 Arcane::IGridMeshPartitioner* partitioner = partitioner_ref.get();
707
708 // Positionne les coordonnées de notre sous-domaine dans la grille
709 Int32 sd_x = cartesian_mesh->cellDirection(MD_DirX).subDomainOffset();
710 Int32 sd_y = cartesian_mesh->cellDirection(MD_DirY).subDomainOffset();
711 Int32 sd_z = cartesian_mesh->cellDirection(MD_DirZ).subDomainOffset();
712 partitioner->setPartIndex(sd_x,sd_y,sd_z);
713
714 // Positionne la bounding box de notre sous-domaine.
715 // Pour cela, parcours uniquement nos noeuds et prend les coordonnées min et max
716 Real max_value = FloatInfo<Real>::maxValue();
717 Real min_value = -max_value;
718 Arcane::Real3 min_box(max_value,max_value,max_value);
719 Arcane::Real3 max_box(min_value,min_value,min_value);
720 VariableNodeReal3& nodes_coord = current_mesh->nodesCoordinates();
721 ENUMERATE_(Cell,icell,current_mesh->ownCells()){
722 Cell cell{*icell};
723 for( Node node : cell.nodes() ){
724 Real3 coord = nodes_coord[node];
725 min_box = math::min(min_box,coord);
726 max_box = math::max(max_box,coord);
727 }
728 }
729 partitioner->setBoundingBox(min_box,max_box);
730
731 // Applique le partitionnement
732 partitioner->applyMeshPartitioning(new_mesh);
734
735 // Maintenant, écrit le fichier du maillage non structuré et de notre partie
736 // cartésienne.
737 const bool is_debug = false;
738 if (is_debug){
739 ServiceBuilder<IMeshWriter> sbuilder2(sd);
740 auto mesh_writer = sbuilder2.createReference("VtkLegacyMeshWriter",SB_Collective);
741 {
742 StringBuilder fname = "cut_mesh_";
743 fname += pm->commRank();
744 fname += ".vtk";
745 mesh_writer->writeMeshToFile(new_mesh,fname);
746 }
747 {
748 StringBuilder fname = "my_mesh_";
749 fname += pm->commRank();
750 fname += ".vtk";
751 mesh_writer->writeMeshToFile(current_mesh,fname);
752 }
753 }
754}
