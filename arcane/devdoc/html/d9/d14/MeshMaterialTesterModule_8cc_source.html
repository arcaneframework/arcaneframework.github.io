268void MeshMaterialTesterModule::
269buildInit()
270{
271 // La création des milieux et des matériaux doit se faire dans un point
272 // d'entrée de type 'build' pour que la liste des variables créés par les
273 // milieux et les matériaux soit accessible dans le post-traitement.
274 info() << "MESH_MATERIAL_TESTER :: buildInit()";
275
276 Materials::IMeshMaterialMng* mm = IMeshMaterialMng::getReference(defaultMesh());
277
278 m_material_mng->setModificationFlags(options()->modificationFlags());
279
280 m_material_mng->setMeshModificationNotified(true);
281
282 // En parallèle, test la création de variables milieux aussi sur les matériaux
283 if (parallelMng()->isParallel())
284 m_material_mng->setAllocateScalarEnvironmentVariableAsMaterial(true);
285
286 if (subDomain()->isContinue() && options()->recreateFromDump()){
287 mm->recreateFromDump();
288 }
289 else{
290 UniqueArray<MeshMaterialInfo*> materials_info;
291 // Lit les infos des matériaux du JDD et les enregistre dans le gestionnaire
292 for( Integer i=0,n=options()->material().size(); i<n; ++i ){
293 String mat_name = options()->material[i].name;
294 info() << "Found material name=" << mat_name;
295 materials_info.add(mm->registerMaterialInfo(mat_name));
296 }
297
298 MeshBlockBuildInfo mbbi("BLOCK1",allCells());
299 UniqueArray<IMeshEnvironment*> saved_envs;
300
301 // Créé les milieux
302 for( Integer i=0,n=options()->environment().size(); i<n; ++i ){
303 String env_name = options()->environment[i].name;
304 info() << "Found environment name=" << env_name;
305 Materials::MeshEnvironmentBuildInfo env_build(env_name);
306 for( Integer k=0,kn=options()->environment[i].material.size(); k<kn; ++k ){
307 String mat_name = options()->environment[i].material[k];
308 info() << "Add material " << mat_name << " for environment " << env_name;
309 env_build.addMaterial(mat_name);
310 }
311 IMeshEnvironment* env = mm->createEnvironment(env_build);
312 saved_envs.add(env);
313 // Le bloc ne contient que 2 milieux
314 if (i<2){
315 info() << "Add environment " << env_name << " to block1";
316 mbbi.addEnvironment(env);
317 }
318 }
319
320 // Création du bloc BLOCK1 sur le groupe de toutes les mailles
321 // et contenant les milieux ENV1 et ENV2
322 m_block1 = mm->createBlock(mbbi);
323
324 {
325 // Création d'un deuxième bloc de manière incrémentalle.
326 Integer nb_env = saved_envs.size();
327 if (nb_env>=2){
328 MeshBlockBuildInfo mbbi2("BLOCK2",allCells());
329 mbbi2.addEnvironment(saved_envs[0]);
330 IMeshBlock* block2 = mm->createBlock(mbbi2);
331 Integer nb_env1 = block2->nbEnvironment();
332 mm->addEnvironmentToBlock(block2,saved_envs[1]);
333 info() << "Finished incremental creation of block";
334 Integer nb_env2 = block2->nbEnvironment();
335 if (nb_env2!=(nb_env1+1))
336 ARCANE_FATAL("Bad number of environment");
337 if (block2->environments()[nb_env1]!=saved_envs[1])
338 ARCANE_FATAL("Bad last environment");
339 // Supprime le premier milieu du bloc
340 IMeshEnvironment* first_env = block2->environments()[0];
341 IMeshEnvironment* second_env = block2->environments()[1];
342 mm->removeEnvironmentToBlock(block2,first_env);
343 nb_env2 = block2->nbEnvironment();
344 if (nb_env2!=nb_env1)
345 ARCANE_FATAL("Bad number of environment after remove");
346 if (block2->environments()[0]!=second_env)
347 ARCANE_FATAL("Bad first environment");
348 }
349 }
350
351 mm->endCreate(subDomain()->isContinue());
352
353 info() << "List of materials:";
354 for( MeshMaterialInfo* m : materials_info ){
355 info() << "MAT=" << m->name();
356 for( String s : m->environmentsName() )
357 info() << " In ENV=" << s;
358 }
359 }
360
361 // Récupère deux matériaux de deux milieux différents pour test.
362 ConstArrayView<Materials::IMeshEnvironment*> envs = mm->environments();
363 Integer nb_env = envs.size();
364 m_mat1 = mm->environments()[0]->materials()[0];
365 if (nb_env>1)
366 m_mat2 = mm->environments()[1]->materials()[0];
367
368 m_global_deltat.assign(1.0);
369
370 for( Integer i=0, n=m_material_mng->materials().size(); i<n; ++i ){
371 IMeshMaterial* mat = m_material_mng->materials()[i];
372 VariableCellReal* var = new VariableCellReal(VariableBuildInfo(defaultMesh(),String("Density_")+mat->name()));
373 m_density_post_processing.add(var);
374 }
375
376 m_mesh_partitioner = options()->loadBalanceService();
377 if (m_mesh_partitioner)
378 info() << "Activating load balance test";
379}

395void MeshMaterialTesterModule::
396startInit()
397{
398 m_mat_not_used_real.globalVariable().setUsed(false);
399
400 info() << "MESH_MATERIAL_TESTER :: startInit()";
401 m_material_mng->forceRecompute();
402
403 ValueChecker vc(A_FUNCINFO);
404
405 m_nb_starting_cell.assign(parallelMng()->reduce(Parallel::ReduceSum,ownCells().size()));
406
407 Int32UniqueArray env1_indexes;
408 Int32UniqueArray mat2_indexes;
409 Int32UniqueArray sub_group_indexes;
410 Integer nb_cell = ownCells().size();
411 IParallelMng* pm = parallelMng();
412 Int64 total_nb_cell = pm->reduce(Parallel::ReduceMax,nb_cell);
413 ENUMERATE_CELL(icell,allCells()){
414 Cell cell = *icell;
415 Int64 cell_index = cell.uniqueId();
416 if (cell_index>(total_nb_cell/2))
417 break;
418 if (cell_index<(total_nb_cell/5)){
419 env1_indexes.add(icell.itemLocalId());
420 }
421 if (cell_index<(total_nb_cell/3) && cell_index>(total_nb_cell/6)){
422 mat2_indexes.add(icell.itemLocalId());
423 }
424 if ((cell_index%2)==0)
425 sub_group_indexes.add(icell.itemLocalId());
426 }
427
428 // Ajoute les mailles du milieu 1
429 {
430 Materials::MeshMaterialModifier modifier(m_material_mng);
431 Materials::IMeshEnvironment* env = m_mat1->environment();
432 // Ajoute les mailles du milieu
433 //modifier.addCells(env,env1_indexes);
434 Int32UniqueArray mat1_indexes;
435 Int32UniqueArray mat2_indexes;
436 Integer nb_cell = env1_indexes.size();
437 for( Integer z=0; z<nb_cell; ++z ){
438 bool add_to_mat1 = (z<(nb_cell/2) && z>(nb_cell/4));
439 bool add_to_mat2 = (z>=(nb_cell/2) || z<(nb_cell/3));
440 if (add_to_mat1){
441 mat1_indexes.add(env1_indexes[z]);
442 }
443 if (add_to_mat2)
444 mat2_indexes.add(env1_indexes[z]);
445 }
446 // Ajoute les mailles du matériau 1
447 modifier.addCells(m_mat1,mat1_indexes);
448 Integer nb_mat = env->nbMaterial();
449 if (nb_mat>1)
450 // Ajoute les mailles du matériau 2
451 modifier.addCells(env->materials()[1],mat2_indexes);
452 }
453 CellGroup test_group = defaultMesh()->cellFamily()->createGroup("SUB_GROUP_TEST",sub_group_indexes);
454 // Ajoute les mailles du milieu 2
455 if (m_mat2){
456 Materials::MeshMaterialModifier modifier(m_material_mng);
457 //modifier.addCells(m_mat2->environment(),mat2_indexes);
458 modifier.addCells(m_mat2,mat2_indexes);
459 }
460
461 const Integer spectral_size = 5;
462 m_mat_spectral1.resize(spectral_size);
463 m_mat_spectral2.resize(spectral_size*2);
464 m_env_spectral1.resize(spectral_size*3);
465 m_env_spectral2.resize(spectral_size*4);
466 // NOTE: m_env_empty_int64array ne doit pas avoir de resize
467 // pour pouvoir tester la gestion des variables avec une dim2Size() nulle.
468
469 // TODO tester que les valeurs partielles sont correctes
470 m_mat_density.fillPartialValues(3.0);
471 m_env_int32.fillPartialValues(5);
472 m_mat_int32.fillPartialValues(8);
473
474 info() << "Liste des mailles de test_group";
475 ENUMERATE_CELL(icell,test_group){
476 info(6) << "Cell=" << ItemPrinter(*icell);
477 }
478
479 ENUMERATE_CELL(icell,allCells()){
480 m_density[icell] = 1.0;
481 m_mat_density[icell] = 2.0;
482 Integer idx2 = icell.itemLocalId() % spectral_size;
483 m_mat_spectral1[icell][idx2] = 3.0 + (Real)(icell.itemLocalId() * spectral_size);
484 }
485
486 CellGroup mat1_cells = m_mat1->cells();
487 ENUMERATE_CELL(icell,mat1_cells){
488 m_density[icell] = 2.0;
489 }
490 {
491 VariableCellReal& gvar = m_mat_density.globalVariable();
492 info() << "GVAR_NAME = " << gvar.name();
493 ENUMERATE_CELL(icell,mat1_cells){
494 if (gvar[icell]!=2.0)
495 fatal() << "Bad value for global variable v=" << gvar[icell];
496 }
497 }
498
499 if (m_mat2){
500 CellGroup mat2_cells = m_mat2->cells();
501 ENUMERATE_CELL(icell,mat2_cells){
502 m_density[icell] = 1.5;
503 }
504 }
505
506 _checkTemporaryVectors(test_group);
507 _checkSubViews(test_group);
508
509 {
510 OStringStream oss;
511 m_material_mng->dumpInfos(oss());
512 info() << oss.str();
513 }
514
515 // A supprimer
516 //m_mat_density.updateFromInternal();
517
518 m_present_material.fill(0);
519 m_mat_density.fill(0.0);
520 m_mat_nodump_real.fill(0.0);
521
522 constexpr IMeshMaterial* null_mat = nullptr;
523 constexpr IMeshEnvironment* null_env = nullptr;
524 // Itération sur tous les milieux puis tous les matériaux
525 // et toutes les mailles de ce matériau
526 ENUMERATE_ENV(ienv,m_material_mng){
527 IMeshEnvironment* env = *ienv;
528 info() << "ENV name=" << env->name();
529 vc.areEqual(env->isEnvironment(),true,"IsEnvEnvOK");
530 vc.areEqual(env->isMaterial(),false,"IsEnvMatOK");
531 vc.areEqual(env->asEnvironment(),env,"ToEnvEnvOK");
532 vc.areEqual(env->asMaterial(),null_mat,"ToEnvMatOK");
533 ENUMERATE_MAT(imat,env){
534 Materials::IMeshMaterial* mat = *imat;
535 info() << "MAT name=" << mat->name();
536 vc.areEqual(mat->isEnvironment(),false,"IsMatEnvOK");
537 vc.areEqual(mat->isMaterial(),true,"IsMatMatOK");
538 vc.areEqual(mat->asEnvironment(),null_env,"ToMatEnvOK");
539 vc.areEqual(mat->asMaterial(),mat,"ToMatMatOK");
540 ENUMERATE_MATCELL(icell,mat){
541 MatCell mmcell = *icell;
542 //info() << "Cell name=" << mmcell._varIndex();
543 m_mat_density[mmcell] = 200.0;
544 m_mat_nodump_real[mmcell] = 1.2;
545 ComponentCell x1(mmcell);
546 if (x1._varIndex()!=mmcell._varIndex())
547 ARCANE_FATAL("Bad convertsion MatCell -> ComponentCell");
548 MatCell x2(x1);
549 if (x1._varIndex()!=x2._varIndex())
550 ARCANE_FATAL("Bad convertsion ComponentCell -> MatCell");
551 }
552 }
553 }
554
555 // Idem mais à partir d'un bloc.
556 ENUMERATE_ENV(ienv,m_block1){
557 IMeshEnvironment* env = *ienv;
558 info() << "BLOCK_ENV name=" << env->name();
559 ENUMERATE_MAT(imat,env){
560 Materials::IMeshMaterial* mat = *imat;
561 info() << "BLOCK_MAT name=" << mat->name();
562 }
563 }
564
565 // Idem mais itération sur des milieux sous forme de composants
566 ENUMERATE_COMPONENT(icmp,m_material_mng->environmentsAsComponents()){
567 IMeshComponent* cmp = *icmp;
568 info() << "ENV COMPONENT name=" << cmp->name();
569 }
570
571 // Itération sur les matériaux sous forme de composant.
572 ENUMERATE_COMPONENT(icmp,m_material_mng->materialsAsComponents()){
573 IMeshComponent* cmp = *icmp;
574 info() << "MAT COMPONENT name=" << cmp->name();
575 }
576
577 // Itération directement avec tous les matériaux du gestionnaire
578 ENUMERATE_MAT(imat,m_material_mng){
579 Materials::IMeshMaterial* mat = *imat;
580 info() << "MAT name=" << mat->name()
581 << " density_var_name=" << m_mat_density.materialVariable()->materialVariable(mat)->name();
582 ENUMERATE_MATCELL(icell,mat){
583 MatCell mmcell = *icell;
584 m_mat_density[mmcell] = 200.0;
585 }
586 }
587
588
589 if (0){
590 ENUMERATE_ENV(ienv,m_material_mng){
591 IMeshEnvironment* env = *ienv;
592 info() << "Env name=" << env->name();
593 ENUMERATE_ENVCELL(ienvcell,env){
594 EnvCell ev = *ienvcell;
595 info() << "EnvCell nb_mat=" << ev.nbMaterial() << " cell_uid=" << ItemPrinter(ev.globalCell())
596 << " component_uid="<< ev.componentUniqueId()
597 << " var_index=" << ev._varIndex();
598 }
599 }
600 }
601
602 if (1){
603 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
604 ENUMERATE_FACE(iface,allFaces()){
605 Face face = *iface;
606 Cell back_cell = face.backCell();
607 if (!back_cell.null()){
608 AllEnvCell all_env_back_cell = all_env_cell_converter[back_cell];
609 info() << "NB_ENV=" << all_env_back_cell.nbEnvironment();
610 ComponentCell x1 = all_env_back_cell;
611 AllEnvCell x2(x1);
612 if (x1._varIndex()!=all_env_back_cell._varIndex())
613 ARCANE_FATAL("Bad convertsion AllEnvCell -> ComponentCell");
614 if (x1._varIndex()!=x2._varIndex())
615 ARCANE_FATAL("Bad convertsion ComponentCell -> EnvCell");
616 }
617 }
618 }
619
620 // Itération sur tous les milieux et tous les matériaux d'une maille.
621 ENUMERATE_ALLENVCELL(iallenvcell,m_material_mng,allCells()){
