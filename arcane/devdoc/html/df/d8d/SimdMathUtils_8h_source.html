31inline SimdReal3
32cross(const SimdReal3& u,const SimdReal3& v)
33{
34 return SimdReal3(
35 u.y * v.z - u.z * v.y,
36 u.z * v.x - u.x * v.z,
37 u.x * v.y - u.y * v.x
38 );
39}
