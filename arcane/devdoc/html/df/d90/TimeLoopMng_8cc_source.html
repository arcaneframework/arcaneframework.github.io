398void TimeLoopMng::
399build()
400{
401 // Créé en enregistre une boucle par défaut.
402 ITimeLoop * tm = createTimeLoop(String("ArcaneEmptyLoop"));
403 m_default_time_loop = tm;
404 registerTimeLoop(tm);
405
406 {
407 String verif_env = platform::getEnvironmentVariable("STDENV_VERIF");
408 if (verif_env=="READ"){
409 m_verif_type = VerifRead;
410 info() << "Checking in read mode";
411 }
412 if (verif_env=="WRITE"){
413 m_verif_type = VerifWrite;
414 info() << "Checking in write mode";
415 }
416 if (verif_env=="CHECKSYNC"){
417 m_verif_type = VerifSync;
418 info() << "Checking synchronizations";
419 }
420 if (verif_env=="CHECKREPLICA"){
421 m_verif_type = VerifSameReplica;
422 info() << "Checking variables values between replica";
423 }
424 if (m_verif_type!=VerifNone)
425 m_verification_active = true;
426 }
427 {
428 String s = platform::getEnvironmentVariable("STDENV_VERIF_ENTRYPOINT");
429 if (!s.null()){
430 m_verification_at_entry_point = true;
431 info() << "Do verification at each entry point";
432 }
433 }
434 {
435 String s = platform::getEnvironmentVariable("STDENV_VERIF_ONLY_AT_EXIT");
436 if (s=="1" || s=="true" || s=="TRUE"){
437 m_verification_only_at_exit = true;
438 info() << "Do verification only at exit";
439 }
440 }
441 // Regarde si on n'exécute qu'un seul point d'entrée au lieu de la boucle
442 // en temps. Cela est utilisé uniquement pour des tests
443 {
444 String s = platform::getEnvironmentVariable("ARCANE_CALL_SPECIFIC_ENTRY_POINT");
445 if (!s.null()){
446 m_specific_entry_point_name = s;
447 info() << "Use specific entry point: " << s;
448 }
449 }
450
451 //
452 if (m_verification_active){
453 m_verif_path = platform::getEnvironmentVariable("STDENV_VERIF_PATH");
454 if (m_verif_path.null()){
455 String user_name = subDomain()->application()->userName();
456 m_verif_path = "/tmp/" + user_name + "/verif";
457 //m_verif_path += user_name;
458 //m_verif_path += "/verif";
459 }
460 }
461 if (m_verif_type==VerifWrite){
462 if (subDomain()->parallelMng()->isMasterIO()){
463 info() << "Creating directory '"<< m_verif_path << "' for checking";
464 platform::recursiveCreateDirectory(m_verif_path);
465 }
466 }
467
468 // Creation du service de "message passing profiling" le cas echeant
469 {
470 String msg_pass_prof_str = platform::getEnvironmentVariable("ARCANE_MESSAGE_PASSING_PROFILING");
471 if (!msg_pass_prof_str.null()) {
472 String service_name;
473 // TODO: ne pas faire de if mais utiliser directement le nom spécifié par la
474 // variable d'environnement.
475 if (msg_pass_prof_str == "JSON") {
476 service_name = "JsonMessagePassingProfiling";
477 } else if (msg_pass_prof_str == "OTF2") {
478 service_name = "Otf2MessagePassingProfiling";
479 }
480 ServiceBuilder<IMessagePassingProfilingService> srv(this->subDomain());
481 m_msg_pass_prof_srv = srv.createReference(service_name ,SB_AllowNull);
482 }
483 }
484}

657
658void TimeLoopMng::
659_checkVerif(const String& entry_point_name,Integer index,bool do_verif)
660{
661 IParallelMng * pm = subDomain()->parallelMng();
662
663 ISubDomain* sd = subDomain();
664 IApplication* app = sd->application();
665
666
667 if ((m_verif_type==VerifRead || m_verif_type==VerifWrite) && !m_verifier_service.get()){
668 String service_name1 = platform::getEnvironmentVariable("STDENV_VERIF_SERVICE");
669 if (service_name1.empty())
670 service_name1 = "ArcaneBasicVerifier2";
671
672 ServiceFinder2T<IVerifierService,ISubDomain> sf(app,sd);
673 m_verifier_service = sf.createReference(service_name1);
674 if (!m_verifier_service.get()){
675 warning() << "No verification service is available."
676 << " No verification will be performed";
677 m_verif_type = VerifNone;
678 }
679 if (m_verifier_service.get()){
680 info() << "Use the service <" << service_name1
681 << "> for verification";
682 }
683 }
684
685 if (pm->isParallel()){
686 if (m_verif_type == VerifSync){
687 Integer nb_error = 0;
688 VariableCollection variables(subDomain()->variableMng()->usedVariables());
689 for( VariableCollection::Enumerator i(variables); ++i; ){
690 IVariable* var = *i;
691 if (var->property() & IVariable::PNoNeedSync)
692 continue;
693 if (var->isPartial())
694 continue;
695 nb_error += var->checkIfSync(5);
696 }
697 if (nb_error!=0)
698 info() << "Error in synchronization nb_error=" << nb_error
699 << " entry_point=" << entry_point_name;
700 }
701 }
702
703 if (m_verif_type == VerifSameReplica && pm->replication()->hasReplication()){
704 _checkVerifSameOnAllReplica(entry_point_name);
705 }
706
707 if ((m_verif_type == VerifRead || m_verif_type == VerifWrite) && do_verif){
708 Integer current_iter = subDomain()->commonVariables().globalIteration();
709 if (current_iter>=0){
710 {
711 StringBuilder path = Directory(m_verif_path).file("verif_file");
712 if (m_verif_same_parallel){
713 path += "_";
714 path += pm->commRank();
715 }
716
717 StringBuilder sub_dir;
718 sub_dir += "iter";
719 sub_dir += current_iter;
720 sub_dir += "/";
721 sub_dir += entry_point_name;
722 sub_dir += index;
723
724 m_verifier_service->setFileName(path.toString());
725 m_verifier_service->setSubDir(sub_dir.toString());
726 }
727 bool parallel_sequential = pm->isParallel();
728 if (m_verif_same_parallel)
729 parallel_sequential = false;
730 if (m_verif_type==VerifRead){
731
732 // Activer cette partie si on souhaite sauver les valeurs actuelles
733#if 0
734 {

845
846int TimeLoopMng::
847doOneIteration()
848{
849 ITraceMng * msg = traceMng();
850 ISubDomain* sd = subDomain();
851 Trace::Setter mci(msg,m_message_class_name);
852
853 if (!m_used_time_loop)
854 ARCANE_FATAL("No time loop");
855
856 Integer current_iteration = sd->commonVariables().globalIteration();
857 // Le numéro d'itération 0 correspond à l'initialisation. La première
858 // itération de la boucle en temps porte le numéro 1.
859 if (current_iteration==0){
860 VariableScalarInteger global_iteration(sd->commonVariables().m_global_iteration);
861 global_iteration = 1;
862 current_iteration = 1;
863 }
864
865 _resetTimer();
866
867 if (m_stop_time_loop){
868 if (m_stop_has_error)
869 return (-1);
870 return (+1);
871 }
872
873 m_backward_mng->beginAction();
874
875 // Action de restauration
876 if (m_backward_mng->checkAndApplyRestore()) {
877
878 execRestoreEntryPoints();
879
880 // Repartitionnement inutile si retour-arrière
881 m_mesh_partitioner = nullptr;
882 }
883
884 // Repartionnement demandé
885 bool mesh_partition_done = false;
886 if (m_mesh_partitioner){
887 _doMeshPartition();
888 mesh_partition_done = true;
889 }
890
891 // Action de sauvegarde
892 m_backward_mng->checkAndApplySave(mesh_partition_done);
893
894 m_backward_mng->endAction();
895
896 {
897 // Regarde les modules inactifs au cours du temps
898 Real global_time = sd->commonVariables().globalTime();
899 CaseOptionsCollection blocks(sd->caseMng()->blocks());
900 for( CaseOptionsCollection::Enumerator i(blocks); ++i; ){
901 ICaseOptions * opt = * i;
902 IModule * mod = opt->caseModule();
903 ICaseFunction * f = opt->activateFunction();
904 if (mod && f){
905 bool is_active = true;
906 f->value(global_time, is_active);
907 bool mod_disabled = mod->disabled();
908 bool mod_new_disabled = !is_active;
909 if (mod_new_disabled != mod_disabled){
910 if (mod_new_disabled)
911 info() << "The module " << mod->name() << " is desactivated";
912 else
913 info() << "The module " << mod->name() << " is activated";
914 mod->setDisabled(mod_new_disabled);
915 }
916 }
917 }
918 }
919
920 m_observables[eTimeLoopEventType::BeginIteration]->notifyAllObservers();
921
922 // Exécute chaque point d'entrée de l'itération
923 {
924 Integer index =0;
925 sd->timeStats()->notifyNewIterationLoop();
926 Timer::Action ts_action(sd,"LoopEntryPoints");
927 for( EntryPointList::Enumerator i(m_loop_entry_points); ++i; ++index ){
928 IEntryPoint* ep = *i;
929 IModule* mod = ep->module();
930 if (mod && mod->disabled()){
931 continue;
932 //warning() << "MODULE " << mod->name() << " is disabled";
933 }
934 try{
935 _execOneEntryPoint(*i, index, true);
936 } catch(const GoBackwardException&){
937 m_backward_mng->goBackward();
938 } catch(...){ // On remonte toute autre exception
939 throw;
940 }
941 if (m_backward_mng->isBackwardEnabled()){
942 break;
943 }
944 }
945 if (!m_verification_at_entry_point && !m_verification_only_at_exit)
946 _checkVerif("_EndLoop",0,true);
947 }
948 m_observables[eTimeLoopEventType::EndIteration]->notifyAllObservers();
949
950 {
951 bool force_prepare_dump = false;
952 if (m_verification_active && m_verif_type==VerifWrite){
953 if (!m_verif_same_parallel)
954 force_prepare_dump = true;
955 }
956 else{
957 String force_prepare_dump_str = platform::getEnvironmentVariable("ARCANE_FORCE_PREPARE_DUMP");
958 if (force_prepare_dump_str=="TRUE" || force_prepare_dump_str=="1" || force_prepare_dump_str=="true")
959 force_prepare_dump = true;
960 }
961 if (force_prepare_dump){
962 info() << "TimeLoopMng::doOneIteration(): Force prepareDump()";
963 // TODO: vérifier si nécessaire et si oui le faire pour tous les sous-domaines.
964 sd->defaultMesh()->prepareForDump();
965 }
966 }
967
968 if (m_stop_time_loop){
969 if (m_stop_has_error)
970 return (-1);
971 return (+1);
972 }
973 return 0;
974}

1119 // enregistrés dans les modules
1120 // Les points d'entrée non auto-load non référencés ne sont donc pas pris en charge.

1124 EntryPointList timeloop_entry_points;
1125 _processEntryPoints(timeloop_entry_points,
1126 m_used_time_loop->entryPoints(ITimeLoop::WBuild),

1169 _addExecuteEntryPoint(ic);
1170 }
1171 }

1342 << "\" can't be declared mandatory in the time loop"
1343 << " while being inactive in the input data."

1764/*---------------------------------------------------------------------------*/
1765
1766int TimeLoopMng::
1767doComputeLoop(Integer max_loop)
1768{
1769 m_nb_loop = 0;
1770 bool is_end = false;
1771 bool is_end_by_max_loop = false;
1772 int ret_val = 0;
1773 //m_alarm_timer_value = 0;
1774 //pwarning() << "Force le alarm_timer_value a zéro";
1775 if (m_alarm_timer_value>0){
1776 info() << "Set the timeout before alarm at " << m_alarm_timer_value << " seconds";
1777 }
1778
1779 // Allocation d'un gestionnaire de retour-arrière par défaut
1780 if (!m_backward_mng)
1781 _createOwnDefaultBackwardMng();
1782
1783 IProfilingService* ps = platform::getProfilingService();
1784 bool want_specific_profiling = false;
1785 // Regarde si on demande un profiling spécifique. Dans ce cas,
1786 // les modules et services gèrent eux même le profiling et donc
1787 // on ne démarre pas automatiquement le profiling au début de la
1788 // boucle de calcul
1789 if (!platform::getEnvironmentVariable("ARCANE_SPECIFIC_PROFILING").null()){
1790 info() << "Specific profiling activated";
1791 want_specific_profiling = true;
1792 }
1793
1794 {
1795 // NOTE: arcaneGlobalMemoryInfo() peut changer au cours du calcul
1796 // donc il faut le récupérer à chaque fois qu'on en a besoin.
1797 IMemoryInfo* mem_info = arcaneGlobalMemoryInfo();
1798 String s = platform::getEnvironmentVariable("ARCANE_CHECK_MEMORY_BLOCK_SIZE_ITERATION");
1799 if (!s.null()){
1800 Int64 block_size = 0;
1801 bool is_bad = builtInGetValue(block_size,s);
1802 if (!is_bad && block_size>2){
1803 info() << "Set Memory StackTraceMinAllocSize to " << block_size;
1804 mem_info->setStackTraceMinAllocSize(block_size);
1805 }
1806 }
1807 }
1808
1809 // Demarrage du profiling de message passing le cas echeant
1810 if (m_msg_pass_prof_srv.get())
1811 m_msg_pass_prof_srv->startProfiling();
1812
1813 try{
1814 if (ps){
1815 if (!ps->isInitialized())
1816 ps->initialize();
1817 ps->reset();
1818 }
1819 Item::resetStats();
1820 // Désactive le profiling si demande spécifique
1821 if (want_specific_profiling)
1822 ps = nullptr;
1823 ProfilingSentryWithInitialize ps_sentry(ps);
1824 if (!m_specific_entry_point_name.null()){
1825 _callSpecificEntryPoint();
1826 is_end = true;
1827 }
1828 while (!is_end){
1829 if (max_loop!=0 && m_nb_loop>=max_loop){
1830 info()<<"===================================================";
1831 info()<<"====== MAXIMUM NUMBER OF ITERATION REACHED =======";
1832 info()<<"===================================================";
1833 is_end_by_max_loop = true;
1834 m_stop_reason = eTimeLoopStopReason::MaxIterationReached;
1835 stopComputeLoop(false,false);
1836 break;
1837 }
1838 // Indique qu'on va s'arrêter par nombre max d'itération
1839 if (max_loop!=0 && (1+m_nb_loop)>=max_loop){
1840 m_stop_reason = eTimeLoopStopReason::MaxIterationReached;
1841 }
1842 ret_val = doOneIteration();
1843 IMemoryInfo* mem_info = arcaneGlobalMemoryInfo();
1844 if (mem_info && mem_info->isCollecting()){
1845 mem_info->endCollect();
1846 Integer iteration = subDomain()->commonVariables().globalIteration();
1847 mem_info->setIteration(iteration);
1848
1849 std::ostringstream ostr;
1850 if (iteration>0)
1851 mem_info->printAllocatedMemory(ostr,iteration-1);
1852 if (iteration>1)
1853 mem_info->printAllocatedMemory(ostr,iteration-2);
1854 if (iteration>4)
1855 mem_info->printAllocatedMemory(ostr,iteration-5);
1856 info() << "ITERATION_MemoryInfo: " << ostr.str();
1857 mem_info->beginCollect();
1858 }
1859 subDomain()->variableMng()->synchronizerMng()->flushPendingStats();
1860 if (ret_val!=0)
1861 is_end = true;
1862 ++m_nb_loop;
1863 }
1864 }
1865 catch (const TimeoutException& ex){
1866 if (m_msg_pass_prof_srv.get())
1867 m_msg_pass_prof_srv->stopProfiling();
1868 IParallelMng * pm = subDomain()->parallelMng();
1869 pinfo() << "TIMEOUT " << Trace::Width(8) << pm->commRank() << "_RANK Infos:"<<ex.additionalInfo();
1870 traceMng()->flush();
1871 // Attend que tous les processeurs envoient le signal et affichent le
1872 // message précédent
1873 platform::sleep(40);
1874 throw;
1875 }
1876 if (m_verification_only_at_exit){
1877 info() << "Doing verification at exit";
1878 doVerification("AtExit");
1879 }
1880 {
1881 IProfilingService* ps2 = platform::getProfilingService();
1882 if (ps2)
1883 ps2->printInfos(true);
1884 }
1885 if (m_msg_pass_prof_srv.get())
1886 m_msg_pass_prof_srv->stopProfiling();
1887 if (IItemEnumeratorTracer::singleton())
1888 IItemEnumeratorTracer::singleton()->dumpStats();
1889 info() << "End of compute loop: reason=" << (int)m_stop_reason;
1890 if (is_end_by_max_loop)
1891 return 2;
1892 if (m_final_time_reached)
1893 return 1;
1894 if (ret_val<0){
1895 return ret_val;
1896 }
1897 return 0;
1898}
