119template<>
120class ReduceAtomicSum<double>
121{
122 public:
123 static double apply(std::atomic<double>* ptr,double v)
124 {
125 double old = ptr->load(std::memory_order_consume);
126 double wanted = old + v;
127 while (!ptr->compare_exchange_weak(old,wanted,std::memory_order_release,std::memory_order_consume))
128 wanted = old + v;
129 return wanted;
130 }
131};
132template<>
133class ReduceAtomicSum<Int64>
134{
135 public:
136 static Int64 apply(std::atomic<Int64>* ptr,Int64 v)
137 {
138 Int64 x = std::atomic_fetch_add(ptr,v);
139 return x+v;
140 }
141};
142template<>
143class ReduceAtomicSum<Int32>
144{
145 public:
146 static Int32 apply(std::atomic<Int32>* ptr,Int32 v)
147 {
148 Int32 x = std::atomic_fetch_add(ptr,v);
149 return x+v;
150 }
151};

153template<typename DataType>
154class ReduceFunctorSum
155{
156 public:
157 static ARCCORE_DEVICE

159 {
160 _applyDevice(dev_info);
161 return *(dev_info.m_device_final_ptr);
162 }

164 {
165 return ReduceAtomicSum<DataType>::apply(ptr,v);

175template<typename DataType>
176class ReduceFunctorMax
177{
178 public:
179 static ARCCORE_DEVICE
180 DataType applyDevice(const ReduceDeviceInfo<DataType>& dev_info)
181 {
182 _applyDevice(dev_info);
183 return *(dev_info.m_device_final_ptr);
184 }
185 static DataType apply(std::atomic<DataType>* ptr,DataType v)
186 {
187 DataType prev_value = *ptr;
188 while(prev_value < v && !ptr->compare_exchange_weak(prev_value, v)) {}
189 return *ptr;
190 }
191 public:
192 ARCCORE_HOST_DEVICE static constexpr
193 DataType identity() { return impl::ReduceIdentity<DataType>::maxValue(); }
194 private:
195 static ARCCORE_DEVICE
196 void _applyDevice(const ReduceDeviceInfo<DataType>& dev_info);
197};
