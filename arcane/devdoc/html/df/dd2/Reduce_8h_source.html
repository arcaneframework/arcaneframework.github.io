69 template<typename DataType>
70 class ReduceFunctorSum
71 {
72  public:
73  static ARCCORE_DEVICE
74  DataType applyDevice(DataType* ptr,DataType v,DataType identity)
75  {
76  _applyDevice(ptr,v,identity);
77  return *ptr;
78  }
79  static DataType apply(std::atomic<DataType>* ptr,DataType v)
80  {
81  *ptr = *ptr + v;
82  return *ptr;
83  }
84  public:
85  ARCCORE_HOST_DEVICE static constexpr
86  DataType identity() { return impl::ReduceIdentity<DataType>::sumValue(); }
87  private:
88  static ARCCORE_DEVICE
89  void _applyDevice(DataType* ptr,DataType v,DataType identity);
90 };

92 template<typename DataType>
93 class ReduceFunctorMax
94 {
95  public:
96  static ARCCORE_DEVICE
97  DataType applyDevice(DataType* ptr,DataType v,DataType identity)
98  {
99  _applyDevice(ptr,v,identity);
100  return *ptr;
101  }
102  static DataType apply(std::atomic<DataType>* ptr,DataType v)
103  {
104  DataType prev_value = *ptr;
105  while(prev_value < v && !ptr->compare_exchange_weak(prev_value, v)) {}
106  return *ptr;
107  }
108  public:
109  ARCCORE_HOST_DEVICE static constexpr
110  DataType identity() { return impl::ReduceIdentity<DataType>::maxValue(); }
111  private:
112  static ARCCORE_DEVICE
113  void _applyDevice(DataType* ptr,DataType v,DataType identity);
114 };

171 template<typename DataType,typename ReduceFunctor>
172 class Reducer
173 {
174  public:
175  // TODO: obliger à construire une instance avec un runtime associé qui se chargera d'allouer
176  // 'm_managed_memory_value' si besoin.
177  Reducer(RunCommand& command)
178  : m_managed_memory_value(&m_local_value), m_command(&command)
179  {
180  m_is_master_instance = true;
181  m_identity = ReduceFunctor::identity();
182  m_local_value = m_identity;
183  m_atomic_value = m_identity;
184  m_parent_value = &m_atomic_value;
185  //printf("Create null host parent_value=%p this=%p\n",(void*)m_parent_value,(void*)this);
186  m_memory_impl = impl::internalGetOrCreateReduceMemoryImpl(&command);
187  if (m_memory_impl){
188  m_managed_memory_value = impl::allocateReduceMemory<DataType>(m_memory_impl);
189  *m_managed_memory_value = m_identity;
190  }
191  }
192  ARCCORE_HOST_DEVICE Reducer(const Reducer& rhs)
193  : m_managed_memory_value(rhs.m_managed_memory_value), m_local_value(rhs.m_local_value), m_identity(rhs.m_identity)
194  {
195 #ifdef ARCCORE_DEVICE_CODE
196  //int threadId = threadIdx.x + blockDim.x * threadIdx.y + (blockDim.x * blockDim.y) * threadIdx.z;
197  //if ((threadId%16)==0)
198  // printf("Create ref device Id=%d parent=%p\n",threadId,&rhs);
199 #else
200  m_parent_value = rhs.m_parent_value;
201  //printf("Create ref host parent_value=%p this=%p rhs=%p\n",(void*)m_parent_value,(void*)this,(void*)&rhs);
202 #endif
203  }
204  ARCCORE_HOST_DEVICE Reducer(Reducer&& rhs)
205  : m_managed_memory_value(rhs.m_managed_memory_value), m_local_value(rhs.m_local_value), m_identity(rhs.m_identity)
206  {
207 #ifdef ARCCORE_DEVICE_CODE
208  int threadId = threadIdx.x + blockDim.x * threadIdx.y + (blockDim.x * blockDim.y) * threadIdx.z;
209  printf("Create && device Id=%d\n",threadId);
210 #else
211  //printf("Create && host = %p\n",((void*)this));
212  //#if TEST_PARENT
213  m_parent_value = rhs.m_parent_value;
214  //#endif
215 #endif
216  }
217 
218  ARCCORE_HOST_DEVICE ~Reducer()
219  {
220 #ifdef ARCCORE_DEVICE_CODE
221  //int threadId = threadIdx.x + blockDim.x * threadIdx.y + (blockDim.x * blockDim.y) * threadIdx.z;
222  //if ((threadId%16)==0)
223  //printf("Destroy device Id=%d\n",threadId);
224  ReduceFunctor::applyDevice(m_managed_memory_value,m_local_value,m_identity);
225 #else
226  // printf("Destroy host parent_value=%p this=%p\n",(void*)m_parent_value,(void*)this);
227  // Code hôte
228  if (!m_is_master_instance)
229  ReduceFunctor::apply(m_parent_value,m_local_value);
230 
231  //printf("Destroy host %p %p\n",m_managed_memory_value,this);
232  if (m_memory_impl)
233  m_memory_impl->release();
234 #endif
235  }
236  public:
237  ARCCORE_HOST_DEVICE void setValue(DataType v)
238  {
239  m_local_value = v;
240  }
241  ARCCORE_HOST_DEVICE DataType localValue() const
242  {
243  return m_local_value;
244  }
245  // Effectue la réduction et récupère la valeur. ATTENTION: ne faire qu'une seule fois.
246  DataType reduce()
247  {
248  // TODO: faire un fatal si 'm_is_master_instance' vaut 'false'.
249  // Effectue la réduction sur m_managed_memory_value
250  // One ne peut pas le faire avant car comme en CUDA elle est gérée
251  // par la mémoire unifiée, il n'est pas possible d'y accéder tant que
252  // le CPU est en train de le faire.
253  if (m_parent_value){
254  //std::cout << "Reduce host p=" << this << " local_value=" << m_local_value << " v=" << *m_parent_value << "\n";
255  ReduceFunctor::apply(m_parent_value,*m_managed_memory_value);
256  *m_managed_memory_value = *m_parent_value;
257  }
258  return *m_managed_memory_value;
259  }
260  protected:
261  impl::IReduceMemoryImpl* m_memory_impl = nullptr;
262  DataType* m_managed_memory_value;
263  private:
264  RunCommand* m_command;
265  protected:
266  mutable DataType m_local_value;
267  std::atomic<DataType>* m_parent_value = nullptr;
268  mutable std::atomic<DataType> m_atomic_value;
269  private:
270  DataType m_identity;
271  private:
272  bool m_is_allocated = false;
273  bool m_is_master_instance = false;
274 };
