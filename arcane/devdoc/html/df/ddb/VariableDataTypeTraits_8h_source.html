37template <typename DataType> inline void
38builtInDumpValue(String& s, const DataType& v)
39{
40 std::ostringstream sbuf;
41 sbuf << v << '\0';
42 s = sbuf.str();
43}

171 static bool getValue(Type& v, const String& s) { return builtInGetValue(v, s); }
172
173 static bool verifDifferent(Real v1, Real v2, Real& diff, bool is_nan_equal = false)
174 {
175 if (is_nan_equal) {
176 if (std::isnan(v1) && std::isnan(v2))
177 return false;
178 }
179 // Vérifie avant de les comparer que les deux nombres sont valides
180 // pour éviter une exception flottante sur certaines plates-formes
181 if (platform::isDenormalized(v1) || platform::isDenormalized(v2)) {
182 diff = 1.0;
183 return true;
184 }
185 if (v1 != v2) {
186 if (math::abs(v1) < 1.e-100) // TH: plantait pour v1 tres petit(math::isZero(v1))
187 diff = v1 - v2;
188 else
189 diff = (v1 - v2) / v1;
190 return true;
191 }
192 return false;
193 }
194
195 static Real normeMax(Real v)
196 {
197 return math::abs(v);
198 }
199};
