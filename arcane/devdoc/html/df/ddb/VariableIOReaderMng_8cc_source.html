64// Extraction des infos de type à partir d'une chaîne de caractères
65// NOTE L'extraction doit être cohérente avec la construction qui est
66// dans Variable.cc
67class VariableIOReaderMng::VariableDataTypeInfo
68{
69 public:
70
71 explicit VariableDataTypeInfo(const String& full_type)
72 : m_is_partial(false)
73 {
74 std::vector<String> split_strs;
75 full_type.split(split_strs, '.');
76 size_t nb_split = split_strs.size();
77 if (nb_split == 5) {
78 if (split_strs[4] != "Partial")
79 ARCANE_FATAL("Invalid value for partial full_type '{0}'", full_type);
80 m_is_partial = true;
81 --nb_split;
82 }
83 if (nb_split != 4)
84 ARCANE_FATAL("Invalid value for full_type '{0}'", full_type);
85 m_data_type_name = split_strs[0];
86 m_item_kind_name = split_strs[1];
87 m_dimension = split_strs[2];
88 m_multi_tag = split_strs[3];
89 }
90
91 public:
92
93 const String& dataTypeName() const { return m_data_type_name; }
94 const String& itemKindName() const { return m_item_kind_name; }
95 const String& dimension() const { return m_dimension; }
96 const String& multiTag() const { return m_multi_tag; }
97 bool isPartial() const { return m_is_partial; }
98
99 private:
100
101 String m_data_type_name;
102 String m_item_kind_name;
103 String m_dimension;
104 String m_multi_tag;
105 bool m_is_partial;
106};

301
302void VariableReaderMng::
303_buildVariablesToRead(IVariableMng* vm)
304{
305 m_vars_to_read.clear();
306 m_var_read_info_list.clear();
307 for (const auto& x : m_vmd_list) {
308 const String& full_name = x.first;
309 IVariable* var = vm->findVariableFullyQualified(full_name);
310 if (!var)
311 ARCANE_FATAL("Var {0} not in VariableMng", full_name);
312 m_vars_to_read.add(var);
313 m_var_read_info_list.add(VarReadInfo(var, var->data(), x.second));
314 }
315}

318/*---------------------------------------------------------------------------*/
319
320void VariableReaderMng::
321readVariablesData(IVariableMng* vm, VariableIOReaderMng::IDataReaderWrapper* reader)
322{
323 _buildVariablesToRead(vm);
324 reader->beginRead(m_vars_to_read);
325 for (const auto& ivar : m_var_read_info_list) {
326 // NOTE: var peut-être nul
327 IVariable* var = ivar.m_variable;
328 IData* data = ivar.m_data;
329 VariableMetaData* vmd = ivar.m_meta_data;
330 String exception_message;
331 bool has_error = false;
332 try {
333 reader->read(vmd, var, data);
334 if (var)
335 var->notifyEndRead();
336 }
337 catch (const Exception& ex) {
338 OStringStream ostr;
339 ostr() << ex;
340 exception_message = ostr.str();
341 has_error = true;
342 }
343 catch (const std::exception& ex) {
344 exception_message = ex.what();
345 has_error = true;
346 }
347 if (has_error) {
348 OStringStream ostr;
349 String var_full_name = vmd->fullName();
350 ostr() << "Variable = " << var_full_name;
351 if (var) {
352 for (VarRefEnumerator ivar(var); ivar.hasNext(); ++ivar) {
353 VariableRef* ref = *ivar;
354 String s = ref->assignmentStackTrace();
355 if (!s.null())
356 ostr() << "Stack assignement: " << s;
357 }
358 }
359
360 ARCANE_FATAL("Can not read variable variable={0} exception={1} infos={2}",
361 var_full_name, exception_message, ostr.str());
362 }
363 }
364 reader->endRead();
365}

388
389void VariableIOReaderMng::
390readCheckpoint(ICheckpointReader* service)
391{
392 Trace::Setter mci(traceMng(), _msgClassName());
393
394 if (!service)
395 ARCANE_FATAL("No protection service specified");
396
397 service->notifyBeginRead();
398
399 IDataReader* data_reader = service->dataReader();
400 if (!data_reader)
401 ARCANE_FATAL("no dataReader()");
402 String meta_data = data_reader->metaData();
403
404 if (meta_data.null())
405 ARCANE_FATAL("No meta-data in checkpoint.");
406
407 OldDataReaderWrapper wrapper(data_reader);
408
409 info(6) << "METADATA (ICheckpointReader): FromCheckpoint: " << meta_data;
410
411 VariableReaderMng var_read_mng(traceMng());
412 VariableMetaDataList& vmd_list = var_read_mng.variableMetaDataList();
413 {
414 _readMetaData(vmd_list, meta_data.bytes());
415 _createVariablesFromMetaData(vmd_list);
416 _readVariablesData(var_read_mng, &wrapper);
417 }
418
419 service->notifyEndRead();
420 _checkHashFunction(vmd_list);
421}

423/*---------------------------------------------------------------------------*/
424/*---------------------------------------------------------------------------*/
425
426void VariableIOReaderMng::
427readCheckpoint(const CheckpointReadInfo& infos)
428{
429 Trace::Setter mci(traceMng(), _msgClassName());
430
431 ICheckpointReader2* service = infos.reader();
432 ARCANE_CHECK_POINTER2(service, "No checkpoint service specified");
433
434 IParallelMng* pm = infos.parallelMng();
435 ARCANE_CHECK_POINTER2(pm, "no parallelMng()");
436
437 service->notifyBeginRead(infos);
438
439 IDataReader2* data_reader = service->dataReader();
440 ARCANE_CHECK_POINTER2(data_reader, "No dataReader()");
441
442 UniqueArray<Byte> meta_data_bytes;
443 data_reader->fillMetaData(meta_data_bytes);
444 if (meta_data_bytes.empty())
445 ARCANE_FATAL("No meta-data in checkpoint.");
446
447 DataReaderWrapper wrapper(data_reader);
448
449 info(6) << "METADATA (ICheckpointReader2): FromCheckpoint: " << String(meta_data_bytes);
450
451 VariableReaderMng var_read_mng(traceMng());
452 VariableMetaDataList& vmd_list = var_read_mng.variableMetaDataList();
453 {
454 _readMetaData(vmd_list, meta_data_bytes);
455 _createVariablesFromMetaData(vmd_list);
456 _readVariablesData(var_read_mng, &wrapper);
457 }
458
459 service->notifyEndRead();
460 _checkHashFunction(vmd_list);
461}

489void VariableIOReaderMng::
490_readMetaData(VariableMetaDataList& vmd_list, Span<const Byte> bytes)
491{
492 ScopedPtrT<IXmlDocumentHolder> doc(IXmlDocumentHolder::loadFromBuffer(bytes, "meta_data", traceMng()));
493 if (!doc.get())
494 ARCANE_FATAL("The meta-data are invalid");
495 String hash_service_name;
496 JSONDocument json_reader;
497
498 XmlNode root_node = doc->documentNode().documentElement();
499 XmlNode json_node = root_node.child("json");
500
501 // A partir de la version 3.11 de Arcane (juillet 2023), les
502 // méta-données sont aussi disponibles au format JSON. On les utilise
503 // si 'm_is_use_json_metadata' est vrai.
504 JSONValue json_variables;
505 JSONValue json_meshes;
506 if (!json_node.null()) {
507 String json_meta_data = json_node.value();
508 info(6) << "READER_JSON=" << json_meta_data;
509 json_reader.parse(json_meta_data.bytes());
510
511 JSONValue json_meta_data_object = json_reader.root().expectedChild("arcane-checkpoint-metadata");
512
513 // Lit toujours le nom de l'algorithme même si on n'utilise pas les meta-données
514 // car on s'en sert pour les comparaisons de la valeur du hash.
515 String hash_algo_name = json_meta_data_object.child("hash-algorithm-name").value();
516 vmd_list.setHashAlgorithmName(hash_algo_name);
517
518 if (m_is_use_json_metadata) {
519 JSONValue json_version = json_meta_data_object.expectedChild("version");
520 Int32 v = json_version.valueAsInt32();
521 if (v != 1)
522 ARCANE_FATAL("Bad version for JSON Meta Data (v={0}). Only version '1' is supported", v);
523 json_variables = json_meta_data_object.expectedChild("variables");
524 json_meshes = json_meta_data_object.expectedChild("meshes");
525 }
526 }
527 XmlNode variables_node = root_node.child("variables");
528 _readVariablesMetaData(vmd_list, json_variables, variables_node);
529 XmlNode meshes_node = root_node.child("meshes");
530 _readMeshesMetaData(json_meshes, meshes_node);
531}

544void VariableIOReaderMng::
545_checkHashFunction(const VariableMetaDataList& vmd_list)
546{
547 ByteUniqueArray hash_values;
548 MD5HashAlgorithm md5_hash_algorithm;
549 SHA1HashAlgorithm sha1_hash_algorithm;
550 // Par défaut si rien n'est spécifié, il s'agit d'une protection issue
551 // d'une version antérieure à la 3.12 de Arcane. Dans ce cas l'algorithme
552 // utilisé est 'MD5'.
553 IHashAlgorithm* hash_algo = &md5_hash_algorithm;
554 Ref<IHashAlgorithmContext> hash_context;
555 String hash_service_name = vmd_list.hashAlgorithmName();
556 if (!hash_service_name.empty()) {
557 if (hash_service_name == "MD5")
558 hash_algo = &md5_hash_algorithm;
559 else if (hash_service_name == "SHA1") {
560 hash_algo = &sha1_hash_algorithm;
561 hash_context = sha1_hash_algorithm.createContext();
562 }
563 else
564 ARCANE_FATAL("Not supported hash algorithm '{0}'. Valid values are 'SHA1' or 'MD5'");
565 }
566 Integer nb_error = 0;
567 IParallelMng* pm = m_variable_mng->m_parallel_mng;
568 Int32 sid = pm->commRank();
569 Directory listing_dir = m_variable_mng->subDomain()->listingDirectory();
570 for (const auto& i : vmd_list) {
571 const VariableMetaData* vmd = i.second;
572 Int32 hash_version = vmd->hashVersion();
573 String reference_hash = (hash_version > 0) ? vmd->hash2() : vmd->hash();
574 // Teste si la valeur de hashage est présente. C'est normalement
575 // toujours le cas, sauf si la protection vient d'une ancienne
576 // version de Arcane qui ne sauvait pas cette information.
577 // Ce test pourra être supprimé plus tard.
578 if (reference_hash.null())
579 continue;
580 const String& full_name = i.first;
581 IVariable* var = m_variable_mng->findVariableFullyQualified(full_name);
582 if (!var)
583 // Ne devrait pas arriver
584 continue;
585 hash_values.clear();
586 IData* data = var->data();
587 String hash_str;
588 bool do_compare = true;
589 if (hash_version > 0) {
590 ARCANE_CHECK_POINTER(hash_context.get());
591 hash_context->reset();
592 DataHashInfo hash_info(hash_context.get());
593 data->_commonInternal()->computeHash(hash_info);
594 HashAlgorithmValue hash_value;
595 hash_context->computeHashValue(hash_value);
596 hash_str = Convert::toHexaString(asBytes(hash_value.bytes()));
597 // Ne compare si les versions de hash associées à la variable différent
598 if (hash_version != hash_info.version())
599 do_compare = false;
600 }
601 else {
602 data->computeHash(hash_algo, hash_values);
603 hash_str = Convert::toHexaString(hash_values);
604 }
605 if (do_compare && (hash_str != reference_hash)) {
606 ++nb_error;
607 error() << "Hash values are different. Corrumpted values."
608 << " name=" << var->fullName()
609 << " ref=" << reference_hash
610 << " current=" << hash_str;
611 Ref<ISerializedData> sdata(data->createSerializedDataRef(false));
612 Span<const Byte> buf(sdata->constBytes());
613 String fname = listing_dir.file(String::format("dump-{0}-sid_{1}", var->fullName(), sid));
614 std::ofstream ofile(fname.localstr());
615 ofile.write(reinterpret_cast<const char*>(buf.data()), buf.size());
616 }
617 }
618 Integer total_nb_error = pm->reduce(Parallel::ReduceSum, nb_error);
619 if (total_nb_error != 0) {
620 bool allow_bad = !platform::getEnvironmentVariable("ARCANE_ALLOW_DIFFERENT_CHECKPOINT_HASH").null();
621 if (!allow_bad)
622 throw ParallelFatalErrorException(A_FUNCINFO, "hash functions differs");
623 }
624}

629void VariableIOReaderMng::
630_createVariablesFromMetaData(const VariableMetaDataList& vmd_list)
631{
632 ISubDomain* sd = m_variable_mng->subDomain();
633 // Récupère ou construit les variables qui n'existent pas encore.
634 for (const auto& xvmd : vmd_list) {
635 auto& vmd = *(xvmd.second);
636 const String& full_name = vmd.fullName();
637 IVariable* var = m_variable_mng->findVariableFullyQualified(full_name);
638 if (var)
639 continue;
640 const String& base_name = vmd.baseName();
641 Integer property = vmd.property();
642 const String& mesh_name = vmd.meshName();
643 const String& group_name = vmd.itemGroupName();
644 const String& family_name = vmd.itemFamilyName();
645 VariableBuildInfo vbi(sd, base_name, property);
646 if (!mesh_name.null()) {
647 if (vmd.isPartial())
648 vbi = VariableBuildInfo(sd, base_name, mesh_name, family_name, group_name, property);
649 else
650 vbi = VariableBuildInfo(sd, base_name, mesh_name, family_name, property);
651 }
652 info(5) << "Create variable TYPE=" << full_name;
653 m_variable_mng->_createVariableFromType(vmd.fullType(), vbi);
654 }
655}

659
660void VariableIOReaderMng::
661_readVariablesMetaData(VariableMetaDataList& vmd_list, JSONValue variables_json,
662 const XmlNode& variables_node)
663{
664 String ustr_base_name("base-name");
665 String ustr_family_name("item-family-name");
666 String ustr_group_name("item-group-name");
667 String ustr_mesh_name("mesh-name");
668 String ustr_full_type("full-type");
669 String ustr_data_type("data-type");
670 String ustr_hash("hash");
671 String ustr_hash2("hash2");
672 String ustr_hash_version("hash-version");
673 String ustr_property("property");
674 String ustr_multitag("multi-tag");
675 vmd_list.clear();
676
677 struct VariableReadInfo
678 {
679 String full_type;
680 String base_name;
681 String mesh_name;
682 String family_name;
683 String group_name;
684 String hash_value;
685 String hash2_value;
686 Int32 hash_version = 0;
687 String multi_tag;
688 String data_type;
689 Int32 property = 0;
690 };
691 UniqueArray<VariableReadInfo> variables_info;
692
693 // Lit les informations des variables à partir des données JSON
694 // si ces dernières existent.
695 if (!variables_json.null()) {
696 // Lecture via JSON
697 // Déclare la liste ici pour éviter de retourner un temporaire dans 'for-range'
698 JSONValueList vars = variables_json.valueAsArray();
699 for (const JSONValue& var : vars) {
700 VariableReadInfo r;
701 r.full_type = var.expectedChild(ustr_full_type).value();
702 r.base_name = var.expectedChild(ustr_base_name).value();
703 r.data_type = var.child(ustr_data_type).value();
704 r.mesh_name = var.child(ustr_mesh_name).value();
705 r.family_name = var.child(ustr_family_name).value();
706 r.group_name = var.child(ustr_group_name).value();
707 r.hash_value = var.child(ustr_hash).value();
708 r.hash2_value = var.child(ustr_hash2).value();
709 r.hash_version = var.child(ustr_hash_version).valueAsInt32();
710 r.multi_tag = var.child(ustr_multitag).value();
711 r.property = var.child(ustr_property).valueAsInt32();
712 variables_info.add(r);
713 }
714 }
715 else {
716 // Lecture via les données XML
717 XmlNodeList vars = variables_node.children("variable");
718 for (const auto& var : vars) {
719 VariableReadInfo r;
720 r.full_type = var.attrValue(ustr_full_type);
721 r.data_type = var.attrValue(ustr_data_type);
722 r.base_name = var.attrValue(ustr_base_name);
723 r.mesh_name = var.attrValue(ustr_mesh_name);
724 r.group_name = var.attrValue(ustr_group_name);
725 r.family_name = var.attrValue(ustr_family_name);
726 r.hash_value = var.attrValue(ustr_hash);
727 r.multi_tag = var.attrValue(ustr_multitag);
728 r.property = var.attr(ustr_property).valueAsInteger();
729 variables_info.add(r);
730 }
731 }
732
733 for (const VariableReadInfo& r : variables_info) {
734 String full_type = r.full_type;
735 VariableDataTypeInfo vdti(full_type);
736
737 // Vérifie que 'data-type' est cohérent avec la valeur dans 'full_type'
738 if (vdti.dataTypeName() != r.data_type)
739 ARCANE_FATAL("Incoherent value for 'data-type' name v='{0}' expected='{1}'", r.data_type, vdti.dataTypeName());
740
741 String family_name = r.family_name;
742 // Actuellement, si la variable n'est pas partielle alors son groupe
743 // n'est pas sauvé dans les meta-données. Il faut donc le générer.
744 String group_name = r.group_name;
745 bool is_partial = vdti.isPartial();
746 if (!is_partial) {
747 // NOTE: Cette construction doit être cohérente avec celle de
748 // DynamicMeshKindInfos. A terme il faudra toujours sauver le nom du groupe
749 // dans les meta-données.
750 group_name = "All" + family_name + "s";
751 }
752 auto vmd = vmd_list.add(r.base_name, r.mesh_name, r.family_name, group_name, is_partial);
753
754 vmd->setFullType(full_type);
755 vmd->setHash(r.hash_value);
756 vmd->setHash2(r.hash2_value);
757 vmd->setHashVersion(r.hash_version);
758 vmd->setMultiTag(r.multi_tag);
759 vmd->setProperty(r.property);
760
761 info(5) << "CHECK VAR: "
762 << " base-name=" << vmd->baseName()
763 << " mesh-name=" << vmd->meshName()
764 << " family-name=" << vmd->itemFamilyName()
765 << " full-type=" << vmd->fullType()
766 << " name=" << vmd->fullName()
767 << " multitag=" << vmd->multiTag()
768 << " property=" << vmd->property()
769 << " hash=" << vmd->hash();
770 }
771}

775
776void VariableIOReaderMng::
777_readMeshesMetaData(JSONValue meshes_json, const XmlNode& meshes_node)
778{
779 ISubDomain* sd = m_variable_mng->subDomain();
780 IMeshMng* mesh_mng = sd->meshMng();
781 IMeshFactoryMng* mesh_factory_mng = mesh_mng->meshFactoryMng();
782
783 struct MeshInfo
784 {
785 String name;
786 String factory_name;
787 bool is_sequential;
788 };
789 UniqueArray<MeshInfo> meshes_info;
790
791 // Lit les informations des maillages à partir des données JSON
792 // si ces dernières existent.
793 if (!meshes_json.null()) {
794 // Déclare la liste ici pour éviter de retourner un temporaire dans 'for-range'
795 JSONValueList vars = meshes_json.valueAsArray();
796 for (const JSONValue& var : vars) {
797 String mesh_name = var.expectedChild("name").value();
798 String mesh_factory_name = var.child("factory-name").value();
799 bool is_sequential = false;
800 JSONValue v = var.child("sequential");
801 if (!v.null())
802 is_sequential = v.valueAsBool();
803 meshes_info.add({ mesh_name, mesh_factory_name, is_sequential });
804 }
805 }
806 else {
807 XmlNodeList meshes = meshes_node.children("mesh");
808 for (XmlNode var : meshes) {
809 String mesh_name = var.attrValue("name");
810 String mesh_factory_name = var.attrValue("factory-name");
811 bool is_sequential = var.attr("sequential", false).valueAsBoolean();
812 meshes_info.add({ mesh_name, mesh_factory_name, is_sequential });
813 }
814 }
815
816 for (const MeshInfo& mesh_info : meshes_info) {
817 String mesh_name = mesh_info.name;
818 String mesh_factory_name = mesh_info.factory_name;
819 MeshHandle* mesh_handle = mesh_mng->findMeshHandle(mesh_name, false);
820 IMesh* mesh = (mesh_handle) ? mesh_handle->mesh() : nullptr;
821 if (mesh)
822 continue;
823 bool is_sequential = mesh_info.is_sequential;
824 info() << "Creating from checkpoint mesh='" << mesh_name
825 << "' sequential?=" << is_sequential
826 << " factory=" << mesh_factory_name;
827 // Depuis avril 2020, l'attribut 'factory-name' doit être présent
828 // et sa valeur non nulle.
829 if (mesh_factory_name.null())
830 ARCANE_FATAL("No attribute 'factory-name' for mesh");
831
832 {
833 MeshBuildInfo mbi(mesh_name);
834 mbi.addFactoryName(mesh_factory_name);
835 IParallelMng* mesh_pm = m_variable_mng->m_parallel_mng;
836 if (is_sequential)
837 mesh_pm = mesh_pm->sequentialParallelMng();
838 mbi.addParallelMng(Arccore::makeRef(mesh_pm));
839 mesh_factory_mng->createMesh(mbi);
840 }
841 }
842}
