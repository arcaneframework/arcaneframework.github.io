55 template<typename DataType,A_MDRANK_TYPE(RankValue),typename LayoutType>
56 class MDSpanBase
57 {
58  using UnqualifiedValueType = std::remove_cv_t<DataType>;
59  friend class NumArrayBase<UnqualifiedValueType,RankValue,LayoutType>;
60  // Pour que MDSpan<const T> ait accès à MDSpan<T>
61  friend class MDSpanBase<const UnqualifiedValueType,RankValue,LayoutType>;
62  public:
63  using ArrayBoundsIndexType = ArrayBoundsIndex<A_MDRANK_RANK_VALUE(RankValue)>;
64  public:
65  MDSpanBase() = default;
66  ARCCORE_HOST_DEVICE MDSpanBase(DataType* ptr,ArrayExtentsWithOffset<RankValue,LayoutType> extents)
67  : m_ptr(ptr), m_extents(extents)
68  {
69  }
70  // Constructeur MDSpan<const T> à partir d'un MDSpan<T>
71  template<typename X,typename = std::enable_if_t<std::is_same_v<X,UnqualifiedValueType>>>
72  ARCCORE_HOST_DEVICE MDSpanBase(const MDSpanBase<X,RankValue>& rhs)
73  : m_ptr(rhs.m_ptr), m_extents(rhs.m_extents){}
74  public:
75  ARCCORE_HOST_DEVICE DataType* _internalData() { return m_ptr; }
76  ARCCORE_HOST_DEVICE const DataType* _internalData() const { return m_ptr; }
77  public:
78  ArrayExtents<RankValue> extents() const
79  {
80  return m_extents.extents();
81  }
82  ArrayExtentsWithOffset<RankValue,LayoutType> extentsWithOffset() const
83  {
84  return m_extents;
85  }
86  Int32 extent(int i) const { return m_extents(i); }
87  public:
88  ARCCORE_HOST_DEVICE Int64 offset(ArrayBoundsIndexType idx) const
89  {
90  return m_extents.offset(idx);
91  }
93  ARCCORE_HOST_DEVICE DataType& operator()(ArrayBoundsIndexType idx) const
94  {
95  return m_ptr[offset(idx)];
96  }
98  ARCCORE_HOST_DEVICE DataType* ptrAt(ArrayBoundsIndexType idx) const
99  {
100  return m_ptr+offset(idx);
101  }
102  public:
103  ARCCORE_HOST_DEVICE MDSpanBase<const DataType,RankValue> constSpan() const
104  { return MDSpanBase<const DataType,RankValue>(m_ptr,m_extents); }
105  ARCCORE_HOST_DEVICE Span<DataType> to1DSpan() const { return { m_ptr, m_extents.totalNbElement() }; }
106  private:
107  // Utilisé uniquement par NumArrayBase pour la copie
108  Span2<const DataType> _internalTo2DSpan() const
109  {
110  Int64 dim1_size = m_extents(0);
111  Int64 dim2_size = 1;
112  for (int i=1; i<A_MDRANK_RANK_VALUE(RankValue); ++i )
113  dim2_size *= m_extents(i);
114  return { m_ptr, dim1_size, dim2_size };
115  }
116  protected:
117  DataType* m_ptr = nullptr;
118  ArrayExtentsWithOffset<RankValue,LayoutType> m_extents;
119 };

129 template<class DataType,typename LayoutType>
130 class MDSpan<DataType,MDDim1,LayoutType>
131 : public MDSpanBase<DataType,MDDim1,LayoutType>
132 {
133  using UnqualifiedValueType = std::remove_cv_t<DataType>;
134  friend class NumArrayBase<UnqualifiedValueType,MDDim1,LayoutType>;
135  using BaseClass = MDSpanBase<DataType,MDDim1,LayoutType>;
136  using BaseClass::m_extents;
137  using BaseClass::m_ptr;
138  public:
139  using BaseClass::offset;
140  using BaseClass::ptrAt;
141  using BaseClass::operator();
142  public:
144  MDSpan() = default;
146  ARCCORE_HOST_DEVICE MDSpan(DataType* ptr,Int32 dim1_size)
147  {
148  m_extents.setSize(dim1_size);
149  m_ptr = ptr;
150  }
151  ARCCORE_HOST_DEVICE MDSpan(DataType* ptr,ArrayExtentsWithOffset<MDDim1,LayoutType> extents_and_offset)
152  : BaseClass(ptr,extents_and_offset) {}
153 
154  public:
156  ARCCORE_HOST_DEVICE Int64 dim1Size() const { return m_extents(0); }
157  public:
158  ARCCORE_HOST_DEVICE Int64 offset(Int32 i) const { return m_extents.offset(i); }
160  ARCCORE_HOST_DEVICE DataType& operator()(Int32 i) const { return m_ptr[offset(i)]; }
162  ARCCORE_HOST_DEVICE DataType* ptrAt(Int32 i) const { return m_ptr+offset(i); }
164  ARCCORE_HOST_DEVICE DataType operator[](Int32 i) const { return m_ptr[offset(i)]; }
166  template<typename X = DataType,typename SubType = typename NumericTraitsT<X>::SubscriptType >
167  ARCCORE_HOST_DEVICE SubType operator()(Int32 i,Int32 a) const { return m_ptr[offset(i)][a]; }
169  template<typename X = DataType,typename Sub2Type = typename NumericTraitsT<X>::Subscript2Type >
170  ARCCORE_HOST_DEVICE Sub2Type operator()(Int32 i,Int32 a,Int32 b) const { return m_ptr[offset(i)][a][b]; }
171  public:
172  ARCCORE_HOST_DEVICE MDSpan<const DataType,MDDim1,LayoutType> constSpan() const
173  { return MDSpan<const DataType,MDDim1,LayoutType>(m_ptr,m_extents); }
174 };
