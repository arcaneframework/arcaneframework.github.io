54 template<typename DataType,int RankValue,typename LayoutType>
55 class MDSpanBase
56 {
57  using UnqualifiedValueType = std::remove_cv_t<DataType>;
58  friend class NumArrayBase<UnqualifiedValueType,RankValue,LayoutType>;
59  // Pour que MDSpan<const T> ait accès à MDSpan<T>
60  friend class MDSpanBase<const UnqualifiedValueType,RankValue,LayoutType>;
61  public:
62  MDSpanBase() = default;
63  ARCCORE_HOST_DEVICE MDSpanBase(DataType* ptr,ArrayExtentsWithOffset<RankValue,LayoutType> extents)
64  : m_ptr(ptr), m_extents(extents)
65  {
66  }
67  // Constructeur MDSpan<const T> à partir d'un MDSpan<T>
68  template<typename X,typename = std::enable_if_t<std::is_same_v<X,UnqualifiedValueType>>>
69  ARCCORE_HOST_DEVICE MDSpanBase(const MDSpanBase<X,RankValue>& rhs)
70  : m_ptr(rhs.m_ptr), m_extents(rhs.m_extents){}
71  public:
72  ARCCORE_HOST_DEVICE DataType* _internalData() { return m_ptr; }
73  ARCCORE_HOST_DEVICE const DataType* _internalData() const { return m_ptr; }
74  public:
75  ArrayExtents<RankValue> extents() const
76  {
77  return m_extents.extents();
78  }
79  ArrayExtentsWithOffset<RankValue,LayoutType> extentsWithOffset() const
80  {
81  return m_extents;
82  }
83  Int32 extent(int i) const { return m_extents(i); }
84  public:
85  ARCCORE_HOST_DEVICE Int64 offset(ArrayBoundsIndex<RankValue> idx) const
86  {
87  return m_extents.offset(idx);
88  }
90  ARCCORE_HOST_DEVICE DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
91  {
92  return m_ptr[offset(idx)];
93  }
95  ARCCORE_HOST_DEVICE DataType* ptrAt(ArrayBoundsIndex<RankValue> idx) const
96  {
97  return m_ptr+offset(idx);
98  }
99  public:
100  MDSpanBase<const DataType,RankValue> constSpan() const
101  { return MDSpanBase<const DataType,RankValue>(m_ptr,m_extents); }
102  Span<DataType> to1DSpan() { return { m_ptr, m_extents.totalNbElement() }; }
103  Span<const DataType> to1DSpan() const { return { m_ptr, m_extents.totalNbElement() }; }
104  private:
105  // Utilisé uniquement par NumArrayBase pour la copie
106  Span2<const DataType> _internalTo2DSpan() const
107  {
108  Int64 dim1_size = m_extents(0);
109  Int64 dim2_size = 1;
110  for (int i=1; i<RankValue; ++i )
111  dim2_size *= m_extents(i);
112  return { m_ptr, dim1_size, dim2_size };
113  }
114  protected:
115  DataType* m_ptr = nullptr;
116  ArrayExtentsWithOffset<RankValue,LayoutType> m_extents;
117 };
