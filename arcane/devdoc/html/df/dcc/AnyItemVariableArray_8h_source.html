59 template<typename DataType>

67  class VariableAdder
68  {
69  public:
70 
71  VariableAdder(VariableArray<DataType>& variable, const ItemGroup& group)
72  : m_variable(variable)
73  , m_group(group)
74  , m_used(false) {}
75 
76  ~VariableAdder()
77  {
78  ARCANE_ASSERT((m_used == true),("VariableAdder never used"));
79  }
80 
82  template<typename K, typename T>
83  inline void operator<<(MeshVariableArrayRefT<K,T>& v)
84  {
85  ARCANE_ASSERT((m_used == false),("VariableAdder already used"));
86  m_variable._insertVariable(m_group,v.asArray());
87  m_variable._insertInternalVariable(m_group,v.variable());
88  m_used = true;
89  }
90 
92  template<typename K, typename T>
93  inline void operator<<(MeshPartialVariableArrayRefT<K,T>& v)
94  {
95  ARCANE_ASSERT((m_used == false),("VariableAdder already used"));
96  m_variable._insertPartialVariable(m_group,v.asArray());
97  m_variable._insertInternalVariable(m_group,v.variable());
98  m_used = true;
99  }
100  private:
101 
103  VariableArray<DataType>& m_variable;
104 
106  const ItemGroup& m_group;
107 
109  bool m_used;
110  };
111 
112 public:
113 
114  VariableArray(const Family& family)
115  : m_family(family)
116  , m_values(m_family.groupSize())
117  , m_variables(m_family.groupSize())
118  {
119  // la famille enregistre les variables portées
120  m_family.registerObserver(*this);
121  for(Integer i = 0; i < m_family.groupSize(); ++i)
122  m_variables[i] = NULL;
123  }
124 
125  VariableArray(const VariableArray& v)
126  : m_family(v.m_family)
127  , m_values(v.m_values)
128  , m_variables(v.m_variables)
129  {
130  // la famille enregistre les variables portées
131  m_family.registerObserver(*this);
132  }
133 
134  ~VariableArray()
135  {
136  // la famille désenregistre la variable
137  m_family.removeObserver(*this);
138  }

194  inline void notifyFamilyIsInvalidate() {
195  // Si la famille change, on invalide les variables et on retaille
196  m_values.resize(m_family.groupSize());
197  m_variables.resize(m_family.groupSize());
198  for(Integer i = 0; i < m_family.groupSize(); ++i)
199  m_variables[i] = NULL;
200  }
201 
203  inline void notifyFamilyIsIncreased() {
204  // Si la famille est agrandie, on retaille simplement
205  const Integer old_size = m_values.size();
206  ARCANE_ASSERT((old_size < m_family.groupSize()),("Old size greater than new size!"));
207  m_values.resize(m_family.groupSize());
208  m_variables.resize(m_family.groupSize());
209  for(Integer i = old_size; i < m_family.groupSize(); ++i)
210  m_variables[i] = NULL;
211  }
212 
213 private:
214 
215  inline void _insertVariable(ItemGroup group, Array2View<DataType> v)
216  {
217  if(m_family.isPartial(group))
218  throw FatalErrorException(String::format("Group '{0}' defined partial",group.name()));
219  m_values[m_family.groupIndex(group)] = v;
220  }
221 
222  inline void _insertPartialVariable(ItemGroup group, Array2View<DataType> v)
223  {
224  if(not m_family.isPartial(group))
225  throw FatalErrorException(String::format("Group '{0}' not defined partial",group.name()));
226  m_values[m_family.groupIndex(group)] = v;
227  }
228 
229  inline void _insertInternalVariable(ItemGroup group, IVariable* v)
230  {
231  m_variables[m_family.groupIndex(group)] = v;
232  }
233 
234 private:
235 
237  const Family m_family;
238 
240  Arcane::UniqueArray< Array2View<DataType> > m_values;
241 
243  Arcane::UniqueArray< IVariable* > m_variables;
244 };
