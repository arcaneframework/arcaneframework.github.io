70void MeshComponentPartData::
71_setFromMatVarIndexes(ConstArrayView<MatVarIndex> matvar_indexes, RunQueue& queue)
72{
73 Int32 nb_index = matvar_indexes.size();
74
75 Int32Array& pure_indexes = m_value_indexes[(Int32)eMatPart::Pure];
76 Int32Array& impure_indexes = m_value_indexes[(Int32)eMatPart::Impure];
77
78 Int32Array& pure_internal_indexes = m_items_internal_indexes[(Int32)eMatPart::Pure];
79 Int32Array& impure_internal_indexes = m_items_internal_indexes[(Int32)eMatPart::Impure];
80
81 pure_indexes.resize(nb_index);
82 pure_internal_indexes.resize(nb_index);
83
84 // TODO: Faire une première passe pour calculer le nombre de valeurs pures
85 // et ainsi allouer directement à la bonne taille.
86 info(4) << "BEGIN_BUILD_PART_DATA_FOR_COMPONENT c=" << m_component->name();
87 Accelerator::GenericFilterer filterer(&queue);
88
89 Int32 nb_impure = 0;
90 {
91 SmallSpan<Int32> pure_indexes_view(pure_indexes.view());
92 SmallSpan<Int32> pure_internal_indexes_view(pure_internal_indexes.view());
93
94 auto is_pure_lambda = [=] ARCCORE_HOST_DEVICE(Int32 i) -> bool {
95 return matvar_indexes[i].arrayIndex() == 0;
96 };
97 auto setter_lambda = [=] ARCCORE_HOST_DEVICE(Int32 i, Int32 output_index) {
98 pure_indexes_view[output_index] = matvar_indexes[i].valueIndex();
99 pure_internal_indexes_view[output_index] = i;
100 };
101 filterer.applyWithIndex(nb_index, is_pure_lambda, setter_lambda, A_FUNCINFO);
102 Int32 nb_out = filterer.nbOutputElement();
103 pure_indexes.resize(nb_out);
104 pure_internal_indexes.resize(nb_out);
105 nb_impure = nb_index - nb_out;
106 }
107
108 impure_indexes.resize(nb_impure);
109 impure_internal_indexes.resize(nb_impure);
110
111 {
112 SmallSpan<Int32> impure_indexes_view(impure_indexes.view());
113 SmallSpan<Int32> impure_internal_indexes_view(impure_internal_indexes.view());
114 auto is_impure_lambda = [=] ARCCORE_HOST_DEVICE(Int32 i) -> bool {
115 return matvar_indexes[i].arrayIndex() != 0;
116 };
117 auto setter_lambda = [=] ARCCORE_HOST_DEVICE(Int32 i, Int32 output_index) {
118 impure_indexes_view[output_index] = matvar_indexes[i].valueIndex();
119 impure_internal_indexes_view[output_index] = i;
120 };
121 filterer.applyWithIndex(nb_index, is_impure_lambda, setter_lambda, A_FUNCINFO);
122 filterer.nbOutputElement();
123 }
124
125 info(4) << "BUILD_PART_DATA_FOR_COMPONENT c=" << m_component->name()
126 << " nb_pure=" << pure_indexes.size()
127 << " nb_impure=" << impure_indexes.size();
128
129 _notifyValueIndexesChanged();
130}

135void MeshComponentPartData::
136checkValid() const
137{
138 info(4) << "CHECK_VALID_COMPONENT_PART_DATA c=" << m_component->name();
139 ValueChecker vc(A_FUNCINFO);
140 Integer nb_error = 0;
141 for (Integer i = 0; i < 2; ++i) {
142 Int32 var_idx = (i == 0) ? 0 : m_impure_var_idx;
143 Int32ConstArrayView indexes = m_value_indexes[i];
144 Int32ConstArrayView item_indexes = m_items_internal_indexes[i];
145 Integer nb_item = indexes.size();
146 vc.areEqual(nb_item, item_indexes.size(), "Indexes size");
147 for (Integer k = 0; k < nb_item; ++k) {
148 MatVarIndex mvi(var_idx, indexes[k]);
149 MatVarIndex component_mvi = m_constituent_list_view._matVarIndex(item_indexes[k]);
150 if (mvi != component_mvi) {
151 info() << "Bad MatVarIndex i=" << i << " k=" << k
152 << " mvi=" << mvi << " component_mvi=" << component_mvi;
153 ++nb_error;
154 }
155 }
156 }
157 if (nb_error != 0)
158 ARCANE_FATAL("Bad component part data nb_error={0}", nb_error);
159}
