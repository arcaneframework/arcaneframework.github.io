42

55 template<Int32 I,Int32 J,Int32 K>
56 class ArrayLayout3
57 {
58  public:
59 
60  static constexpr Int64 LastExtent = K;
61 
62  static ARCCORE_HOST_DEVICE constexpr Int64
63  offset(ArrayBoundsIndex<3> idx,Int64 extent1,Int64 extent2)
64  {
65  return (extent2 * idx[I]) + (extent1*idx[J]) + idx.asInt64(K);
66  }
67 
68  static ARCCORE_HOST_DEVICE constexpr Int64
69  computeOffsetIndexes(std::array<Int32,3> extents)
70  {
71  return extents[J] * extents[K];
72  }
73 
74  static std::array<Int32,3> layoutInfo() { return { I, J, K }; }
75 };
