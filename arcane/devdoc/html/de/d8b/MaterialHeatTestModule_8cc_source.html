225void MaterialHeatTestModule::
226buildInit()
227{
228 m_sequential_runner.initialize(Accelerator::eExecutionPolicy::Sequential);
229 //m_queue = makeQueue(m_sequential_runner);
230 m_queue = *acceleratorMng()->defaultQueue();
231 ProfilingRegistry::setProfilingLevel(2);
232
233 // La création des milieux et des matériaux doit se faire dans un point
234 // d'entrée de type 'build' pour que la liste des variables créés par les
235 // milieux et les matériaux soit accessibles dans le post-traitement.
236 info() << "MaterialHeatTestModule::buildInit()";
237
238 IMesh* mesh = defaultMesh();
239 Materials::IMeshMaterialMng* mm = IMeshMaterialMng::getReference(mesh);
240
241 int flags = options()->modificationFlags();
242
243 info() << "MaterialHeatTestModule modification_flags=" << flags;
244
245 m_material_mng->setModificationFlags(flags);
246 m_material_mng->setMeshModificationNotified(true);
247 m_material_mng->setUseMaterialValueWhenRemovingPartialValue(true);
248 if (subDomain()->isContinue()) {
249 mm->recreateFromDump();
250 }
251 else {
252 UniqueArray<MeshMaterialInfo*> materials_info;
253 // Lit les infos des matériaux du JDD et les enregistre dans le gestionnaire
254 for (Integer i = 0, n = options()->material().size(); i < n; ++i) {
255 String mat_name = options()->material[i].name;
256 info() << "Found material name=" << mat_name;
257 materials_info.add(mm->registerMaterialInfo(mat_name));
258 }
259
260 UniqueArray<IMeshEnvironment*> saved_envs;
261
262 // Créé les milieux
263 for (Integer i = 0, n = options()->environment().size(); i < n; ++i) {
264 String env_name = options()->environment[i].name;
265 info() << "Found environment name=" << env_name;
266 Materials::MeshEnvironmentBuildInfo env_build(env_name);
267 for (Integer k = 0, kn = options()->environment[i].material.size(); k < kn; ++k) {
268 String mat_name = options()->environment[i].material[k];
269 info() << "Add material " << mat_name << " for environment " << env_name;
270 env_build.addMaterial(mat_name);
271 }
272 IMeshEnvironment* env = mm->createEnvironment(env_build);
273 saved_envs.add(env);
274 }
275
276 mm->endCreate(false);
277
278 info() << "List of materials:";
279 for (MeshMaterialInfo* m : materials_info) {
280 info() << "MAT=" << m->name();
281 for (String s : m->environmentsName())
282 info() << " In ENV=" << s;
283 }
284 }
285
286 // Créé les éventuelles variables additionnelles.
287 {
288 Int32 nb_var_to_add = options()->nbAdditionalVariable();
289 info() << "NbVariableToAdd = " << nb_var_to_add;
290 for (Int32 i = 0; i < nb_var_to_add; ++i) {
291 String var_name = "MaterialAdditionalVar" + String::fromNumber(i);
292 auto* v = new MaterialVariableCellInt32(VariableBuildInfo(mesh, var_name));
293 m_additional_variables.add(v);
294 v->fill(i + 2);
295 }
296 }
297}

302void MaterialHeatTestModule::
303startInit()
304{
305 _buildHeatObjects();
306 m_global_deltat.assign(1.0);
307 m_mat_temperature.globalVariable().fill(0.0);
308 m_material_mng->forceRecompute();
309 _computeCellsCenter();
310 MeshUtils::markMeshConnectivitiesAsMostlyReadOnly(defaultMesh(), &m_queue, true);
311 VariableUtils::markVariableAsMostlyReadOnly(m_cell_center);
312 VariableUtils::markVariableAsMostlyReadOnly(defaultMesh()->nodesCoordinates());
313
314 eMemoryRessource mem_ressource = eMemoryRessource::UnifiedMemory;
315 if (m_queue.isAcceleratorPolicy() && m_material_mng->_internalApi()->runQueue().isAcceleratorPolicy())
316 mem_ressource = eMemoryRessource::Device;
317
318 const bool do_change_allocator = true;
319 if (do_change_allocator) {
320 info() << "Changing allocator to use device memory to '" << mem_ressource << "'";
321 VariableUtils::experimentalChangeAllocator(m_mat_device_temperature.materialVariable(), mem_ressource);
322 }
323 {
324 eMemoryRessource group_mem_ressource = mem_ressource;
325 // En mode check il ne faut pas utiliser la mémoire du device car il y a des tests
326 // effectués uniquement sur CPU
327 if (arcaneIsCheck())
328 group_mem_ressource = eMemoryRessource::UnifiedMemory;
329 ENUMERATE_MAT (imat, m_material_mng) {
330 IMeshMaterial* mat = *imat;
331 mat->cells()._internalApi()->setMemoryRessourceForItemLocalId(group_mem_ressource);
332 }
333 ENUMERATE_ENV (ienv, m_material_mng) {
334 IMeshEnvironment* env = *ienv;
335 env->cells()._internalApi()->setMemoryRessourceForItemLocalId(group_mem_ressource);
336 }
337 }

389/*---------------------------------------------------------------------------*/
390/*---------------------------------------------------------------------------*/
391
392void MaterialHeatTestModule::
393_compute()
394{
395 RunQueue* queue = acceleratorMng()->defaultQueue();
396 eMemoryRessource mem = eMemoryRessource::UnifiedMemory;
397 if (queue->isAcceleratorPolicy())
398 mem = eMemoryRessource::Device;
399
400 const Int32 nb_heat = m_heat_objects.size();
401 UniqueArray<MaterialWorkArray> work_arrays;
402 work_arrays.reserve(nb_heat);
403 for (Int32 i = 0; i < nb_heat; ++i)
404 work_arrays.add(MaterialWorkArray(mem));
405
406 // Ajoute de la chaleur à chaque matériau
407 for (const HeatObject& ho : m_heat_objects)
408 _addHeat(ho);
409
410 // Calcule les mailles à ajouter et supprimer
411 for (const HeatObject& ho : m_heat_objects) {
412 MaterialWorkArray& wa = work_arrays[ho.index];
413 _computeCellsToRemove(ho, wa);
414 _computeCellsToAdd(ho, wa);
415 }
416
417 // Effectue les modifications des matériaux
418 {
419 ProfilingSentryWithInitialize ps_sentry(m_profiling_service);
420 MeshMaterialModifier modifier(m_material_mng);
421 for (const HeatObject& ho : m_heat_objects) {
422 MaterialWorkArray& wa = work_arrays[ho.index];
423 IMeshMaterial* mat = ho.material;
424
425 // Supprime les mailles matériaux nécessaires
426 ConstArrayView<Int32> remove_ids = wa.mat_cells_to_remove;
427 if (!remove_ids.empty()) {
428 info() << "MAT_MODIF: Remove n=" << remove_ids.size() << " cells to material=" << mat->name();
429 modifier.removeCells(mat, remove_ids);
430 }
431
432 // Ajoute les mailles matériaux nécessaires
433 ConstArrayView<Int32> add_ids(wa.mat_cells_to_add.constView());
434 if (!add_ids.empty()) {
435 info() << "MAT_MODIF: Add n=" << add_ids.size() << " cells to material=" << mat->name();
436 modifier.addCells(mat, add_ids);
437 }
438 }
439 }
440
441 // Affiche les valeurs pour les mailles modifiées
442 if (options()->verbosityLevel() > 0) {
443 for (const HeatObject& ho : m_heat_objects) {
444 _printCellsTemperature(work_arrays[ho.index].mat_cells_to_add.constView());
445 _printCellsTemperature(work_arrays[ho.index].mat_cells_to_remove.constView());
446 }
447 }
448
449 // Initialise les nouvelles valeurs partielles
450 for (const HeatObject& ho : m_heat_objects)
451 _initNewCells(ho, work_arrays[ho.index]);
452
453 // Refroidit chaque matériau
454 for (const HeatObject& ho : m_heat_objects)
455 _addCold(ho);
456
457 for (const HeatObject& ho : m_heat_objects) {
458 _copyToGlobal(ho);
459 }
460
461 _computeGlobalTemperature();
462}

520void MaterialHeatTestModule::
521_addCold(const HeatObject& heat_object)
522{
523 IMeshMaterial* current_mat = heat_object.material;
524 const Real cold_value = heat_object.cold_value;
525
526 RunQueue* queue = this->acceleratorMng()->defaultQueue();

544 }

551void MaterialHeatTestModule::
552_addHeat(const HeatObject& heat_object)
553{
554 Real3 heat_center = heat_object.center;
555 heat_center += heat_object.velocity * m_global_time();
556 const Real heat_value = heat_object.heat_value;
557 const Real heat_radius_norm = heat_object.radius * heat_object.radius;
558
559 IMeshMaterial* current_mat = heat_object.material;
560 //RunQueue* queue = this->acceleratorMng()->defaultQueue();
561 //auto queue = makeQueue(m_sequential_runner);
562 auto command = makeCommand(m_queue);
563
564 auto in_cell_center = viewIn(command, m_cell_center);
565 auto inout_mat_temperature = viewInOut(command, m_mat_temperature);
566 auto out_mat_device_temperature = viewInOut(command, m_mat_device_temperature);
567
569 command << RUNCOMMAND_MAT_ENUMERATE(MatAndGlobalCell, iter, current_mat)
570 {
571 auto [matcell, cell] = iter();
572 Real3 center = in_cell_center[cell];
573 Real distance2 = (center - heat_center).squareNormL2();
574 if (distance2 < heat_radius_norm) {
575 Real to_add = heat_value / (1.0 + distance2);
576 inout_mat_temperature[matcell] += to_add;
577 out_mat_device_temperature[matcell] = inout_mat_temperature[matcell];
578 }
579 };
580}

585void MaterialHeatTestModule::
586_computeCellsToAdd(const HeatObject& heat_object, MaterialWorkArray& wa)
587{
588 Real3 heat_center = heat_object.center;
589 heat_center += heat_object.velocity * m_global_time();
590 const Real heat_value = heat_object.heat_value;
591 Real heat_radius = heat_object.radius;
592 const Real heat_radius_norm = heat_radius * heat_radius;
593 IMeshMaterial* current_mat = heat_object.material;
594 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
595
596 const Int32 mat_id = current_mat->id();
597
598 // Détermine les nouvelles à ajouter au matériau
599 {
600 CellVectorView all_cells(allCells().view());
601
602 const Int32 nb_item = all_cells.size();
603 wa.resizeNbAdd(nb_item);
604 Accelerator::GenericFilterer filterer(&m_queue);
605 auto in_cell_center = viewIn(m_queue, m_cell_center);
606 auto cells_ids = viewIn(m_queue, all_cells.localIds());
607
608 auto select_functor = [=] ARCCORE_HOST_DEVICE(Int32 index) -> bool {
609 CellLocalId cellid(cells_ids[index]);
610 AllEnvCell all_env_cell = all_env_cell_converter[cellid];
611 Real3 center = in_cell_center[cellid];
612 Real distance2 = (center - heat_center).squareNormL2();
613 if (distance2 < heat_radius_norm) {
614 MatCell mc = _getMatCell(all_env_cell, mat_id);
615 // Si 'mc' est nul cela signifie que ce matériau n'est
616 // pas présent dans la maille. Il faudra donc l'ajouter.
617 // On conserve la valeur à ajouter pour ne pas la recalculer
618 return mc.null();
619 }
620 return false;
621 };
622
623 auto cells_to_add_view = viewOut(m_queue, wa.mat_cells_to_add);
624 auto cells_to_add_value_view = viewOut(m_queue, wa.mat_cells_to_add_value);
625 auto setter_functor = [=] ARCCORE_HOST_DEVICE(Int32 input_index, Int32 output_index) {
626 CellLocalId cellid(cells_ids[input_index]);
627 Real3 center = in_cell_center[cellid];
628 Real distance2 = (center - heat_center).squareNormL2();
629 Real to_add = heat_value / (1.0 + distance2);
630 cells_to_add_view[output_index] = cellid;
631 cells_to_add_value_view[output_index] = to_add;
632 };
633 filterer.applyWithIndex(nb_item, select_functor, setter_functor, A_FUNCINFO);
634 Int32 nb_out = filterer.nbOutputElement();
635 wa.resizeNbAdd(nb_out);
636 }
637}

641
642void MaterialHeatTestModule::
643_computeCellsToRemove(const HeatObject& heat_object, MaterialWorkArray& wa)
644{
645 IMeshMaterial* current_mat = heat_object.material;
646 const bool is_verbose = false;
647 const Real cold_value = 300.0;
648
649 // Refroidit chaque maille du matériau d'une quantité fixe.
650 // Si la températeure devient inférieure à zéro on supprime la maille
651 // de ce matériau.
652 if (is_verbose)
653 info() << "MAT_BEFORE: " << current_mat->matView()._internalLocalIds();
654 const Int32 nb_cell = current_mat->cells().size();
655 wa.resizeNbRemove(nb_cell);
656
657 {
658 auto command = makeCommand(m_queue);
659 auto out_cells_remove_filter = viewOut(command, wa.mat_cells_remove_filter);
660 auto out_cells_local_id = viewOut(command, wa.mat_cells_to_remove);
661 auto in_mat_temperature = viewIn(command, m_mat_temperature);
662 command << RUNCOMMAND_MAT_ENUMERATE(MatAndGlobalCell, iter, current_mat)
663 {
664 auto [mvi, cid] = iter();
665 Int32 index = iter.index();
666 out_cells_remove_filter[index] = in_mat_temperature[mvi] < cold_value;
667 out_cells_local_id[index] = cid;
668 };
669 }
670
671 {
672 Accelerator::GenericFilterer filterer(&m_queue);
673 SmallSpan<const Int32> in_remove_view = wa.mat_cells_to_remove.view();
674 SmallSpan<Int32> out_remove_view = wa.mat_cells_to_remove.view();
675 SmallSpan<const bool> filter_view = wa.mat_cells_remove_filter.to1DSmallSpan();
676 filterer.apply(in_remove_view, out_remove_view, filter_view);
677 Int32 nb_out = filterer.nbOutputElement();
678 wa.resizeNbRemove(nb_out);
679 }
680
681 if (is_verbose)
682 info() << "MAT_AFTER: " << current_mat->matView()._internalLocalIds();
683}

691 IMeshMaterial* mat = heat_object.material;

710void MaterialHeatTestModule::
711_computeTotalTemperature(const HeatObject& heat_object, bool do_check)
712{
713 IMeshMaterial* mat = heat_object.material;
714 Real total_mat_temperature = 0.0;
715
716 {
717 auto command = makeCommand(m_queue);
718 auto in_mat_temperature = viewIn(command, m_mat_temperature);
719 Accelerator::ReducerSum2<double> total_temperature_reducer(command);
720 CellInfoListView cells_info(defaultMesh()->cellFamily());
721 command << RUNCOMMAND_MAT_ENUMERATE(MatAndGlobalCell, iter, mat, total_temperature_reducer)
722 {
723 auto [mvi, cid] = iter();
724 if (cells_info.isOwn(cid)) {
725 Real t = in_mat_temperature[mvi];
726 total_temperature_reducer.combine(t);
727 }
728 };
729 total_mat_temperature = total_temperature_reducer.reducedValue();
730 }
731
732 total_mat_temperature = parallelMng()->reduce(Parallel::ReduceSum, total_mat_temperature);
733 info() << "TotalMatTemperature mat=" << mat->name() << " T=" << total_mat_temperature;
734 if (do_check) {
735 Real ref_value = heat_object.expected_final_temperature;
736 Real current_value = total_mat_temperature;
737 Real epsilon = 1.0e-12;
738 if (!math::isNearlyEqualWithEpsilon(current_value, ref_value, epsilon)) {
739 Real relative_diff = math::abs(ref_value - current_value);
740 if (ref_value != 0.0)
741 relative_diff /= ref_value;
742 ARCANE_FATAL("Bad value for mat '{0}' ref={1} v={2} diff={3}",
743 mat->name(), ref_value, current_value, relative_diff);
744 }
745 }
746}

812 // Calcule dans 'Temperature' la somme des températures des milieux et matériaux
813 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
814 auto command = makeCommand(m_queue);
815 auto inout_mat_temperature = viewInOut(command, m_mat_temperature);
816 command << RUNCOMMAND_ENUMERATE (Cell, cellid, allCells())
817 {
818 //Cell cell = *icell;
819 AllEnvCell all_env_cell = all_env_cell_converter[cellid];
820 Real global_temperature = 0.0;
821 for (EnvCell env_cell : all_env_cell.subEnvItems()) {
822 Real env_temperature = 0.0;
823 for (MatCell mc : env_cell.subMatItems()) {
824 env_temperature += inout_mat_temperature[mc];
825 }
826 inout_mat_temperature[env_cell] = env_temperature;
827 global_temperature += env_temperature;
828 }
829 inout_mat_temperature[cellid] = global_temperature;
830 };
