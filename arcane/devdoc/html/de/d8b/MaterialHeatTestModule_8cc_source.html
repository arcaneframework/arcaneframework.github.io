177void MaterialHeatTestModule::
178buildInit()
179{
180 m_queue = acceleratorMng()->defaultQueue();
181 ProfilingRegistry::setProfilingLevel(2);
182
183 // La création des milieux et des matériaux doit se faire dans un point
184 // d'entrée de type 'build' pour que la liste des variables créés par les
185 // milieux et les matériaux soit accessibles dans le post-traitement.
186 info() << "MaterialHeatTestModule::buildInit()";
187
188 Materials::IMeshMaterialMng* mm = IMeshMaterialMng::getReference(defaultMesh());
189
190 int flags = options()->modificationFlags();
191
192 info() << "MaterialHeatTestModule modification_flags=" << flags;
193
194 m_material_mng->setModificationFlags(flags);
195 m_material_mng->setMeshModificationNotified(true);
196 m_material_mng->setUseMaterialValueWhenRemovingPartialValue(true);
197 if (subDomain()->isContinue()) {
198 mm->recreateFromDump();
199 }
200 else {
201 UniqueArray<MeshMaterialInfo*> materials_info;
202 // Lit les infos des matériaux du JDD et les enregistre dans le gestionnaire
203 for (Integer i = 0, n = options()->material().size(); i < n; ++i) {
204 String mat_name = options()->material[i].name;
205 info() << "Found material name=" << mat_name;
206 materials_info.add(mm->registerMaterialInfo(mat_name));
207 }
208
209 UniqueArray<IMeshEnvironment*> saved_envs;
210
211 // Créé les milieux
212 for (Integer i = 0, n = options()->environment().size(); i < n; ++i) {
213 String env_name = options()->environment[i].name;
214 info() << "Found environment name=" << env_name;
215 Materials::MeshEnvironmentBuildInfo env_build(env_name);
216 for (Integer k = 0, kn = options()->environment[i].material.size(); k < kn; ++k) {
217 String mat_name = options()->environment[i].material[k];
218 info() << "Add material " << mat_name << " for environment " << env_name;
219 env_build.addMaterial(mat_name);
220 }
221 IMeshEnvironment* env = mm->createEnvironment(env_build);
222 saved_envs.add(env);
223 }
224
225 mm->endCreate(false);
226
227 info() << "List of materials:";
228 for (MeshMaterialInfo* m : materials_info) {
229 info() << "MAT=" << m->name();
230 for (String s : m->environmentsName())
231 info() << " In ENV=" << s;
232 }
233 }
234}

259 for (VariableRef* vref : mat_var->variableReferenceList()) {

295/*---------------------------------------------------------------------------*/
296/*---------------------------------------------------------------------------*/
297
298void MaterialHeatTestModule::
299_compute()
300{
301 UniqueArray<MaterialWorkArray> work_arrays(m_heat_objects.size());
302
303 // Ajoute de la chaleur à chaque matériau
304 for (const HeatObject& ho : m_heat_objects)
305 _addHeat(ho);
306
307 // Calcule les mailles à ajouter et supprimer
308 for (const HeatObject& ho : m_heat_objects) {
309 MaterialWorkArray& wa = work_arrays[ho.index];
310 _computeCellsToRemove(ho, wa);
311 _computeCellsToAdd(ho, wa);
312 }
313
314 // Effectue les modifications des matériaux
315 {
316 ProfilingSentryWithInitialize ps_sentry(m_profiling_service);
317 MeshMaterialModifier modifier(m_material_mng);
318 for (const HeatObject& ho : m_heat_objects) {
319 MaterialWorkArray& wa = work_arrays[ho.index];
320 IMeshMaterial* mat = ho.material;
321
322 // Supprime les mailles matériaux nécessaires
323 ConstArrayView<Int32> remove_ids = wa.mat_cells_to_remove;
324 if (!remove_ids.empty()) {
325 info() << "MAT_MODIF: Remove n=" << remove_ids.size() << " cells to material=" << mat->name();
326 modifier.removeCells(mat, remove_ids);
327 }
328
329 // Ajoute les mailles matériaux nécessaires
330 ConstArrayView<Int32> add_ids(wa.mat_cells_to_add.constView());
331 if (!add_ids.empty()) {
332 info() << "MAT_MODIF: Add n=" << add_ids.size() << " cells to material=" << mat->name();
333 modifier.addCells(mat, add_ids);
334 }
335 }
336 }
337
338 // Affiche les valeurs pour les mailles modifiées
339 if (options()->verbosityLevel() > 0) {
340 for (const HeatObject& ho : m_heat_objects) {
341 _printCellsTemperature(work_arrays[ho.index].mat_cells_to_add.constView());
342 _printCellsTemperature(work_arrays[ho.index].mat_cells_to_remove.constView());
343 }
344 }
345
346 // Initialise les nouvelles valeurs partielles
347 for (const HeatObject& ho : m_heat_objects)
348 _initNewCells(ho, work_arrays[ho.index]);
349
350 // Refroidit chaque matériau
351 for (const HeatObject& ho : m_heat_objects)
352 _addCold(ho);
353
354 // Remplit la variable composante associée de la variable 'AllTemperature'
355 // pour chaque matériau. Cela permet de voir leur valeur facilement dans
356 // les outils de dépouillement
357 ENUMERATE_ (Cell, icell, allCells()) {
358 m_all_temperature[icell].fill(0.0);
359 }
360
361 for (const HeatObject& ho : m_heat_objects) {
362 _copyToGlobal(ho);
363 }
364
365 _computeGlobalTemperature();
366}

371void MaterialHeatTestModule::
372_initNewCells(const HeatObject& heat_object, MaterialWorkArray& wa)
373{
374 // Initialise les nouvelles valeurs partielles
375 IMeshMaterial* current_mat = heat_object.material;
376 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
377 ConstArrayView<Int32> ids(wa.mat_cells_to_add.constView());
378
379 for (Int32 i = 0, n = ids.size(); i < n; ++i) {
380 AllEnvCell all_env_cell = all_env_cell_converter[CellLocalId(ids[i])];
381 MatCell mc = current_mat->findMatCell(all_env_cell);
382 // Teste que la maille n'est pas nulle.
383 // Ne devrait pas arriver car on l'a ajouté juste avant
384 if (mc.null())
385 ARCANE_FATAL("Internal invalid null mat cell");
386 m_mat_temperature[mc] = wa.mat_cells_to_add_value[i];
387 }
388}

393void MaterialHeatTestModule::
394_addCold(const HeatObject& heat_object)
395{
396 IMeshMaterial* current_mat = heat_object.material;
397 const Real cold_value = heat_object.cold_value;
398
399 RunQueue* queue = this->acceleratorMng()->defaultQueue();
400
401 {
402 auto command = makeCommand(queue);
403 auto inout_mat_temperature = viewInOut(command, m_mat_temperature);
404 command << RUNCOMMAND_MAT_ENUMERATE(MatCell, matcell, current_mat)
405 {
406 Real t = inout_mat_temperature[matcell];
407 t -= cold_value;
408 inout_mat_temperature[matcell] = t;
409 };
410 }
411
412 if (arcaneIsCheck()) {
413 ENUMERATE_MATCELL (imatcell, current_mat) {
414 Real t = m_mat_temperature[imatcell];
415 if (t <= 0)
416 ARCANE_FATAL("Invalid negative temperature '{0}' cell_lid={1}", t, (*imatcell).globalCell().localId());
417 }
418 }
419}

423
424void MaterialHeatTestModule::
425_addHeat(const HeatObject& heat_object)
426{
427 Real3 heat_center = heat_object.center;
428 heat_center += heat_object.velocity * m_global_time();
429 const Real heat_value = heat_object.heat_value;
430 const Real heat_radius_norm = heat_object.radius * heat_object.radius;
431
432 IMeshMaterial* current_mat = heat_object.material;
433 RunQueue* queue = this->acceleratorMng()->defaultQueue();
434 auto command = makeCommand(queue);
435
436 auto in_cell_center = viewIn(command, m_cell_center);
437 auto inout_mat_temperature = viewInOut(command, m_mat_temperature);
438 auto out_mat_device_temperature = viewInOut(command, m_mat_device_temperature);
439
441 command << RUNCOMMAND_MAT_ENUMERATE(MatAndGlobalCell, iter, current_mat)
442 {
443 auto [matcell, cell] = iter();
444 Real3 center = in_cell_center[cell];
445 Real distance2 = (center - heat_center).squareNormL2();
446 if (distance2 < heat_radius_norm) {
447 Real to_add = heat_value / (1.0 + distance2);
448 inout_mat_temperature[matcell] += to_add;
449 out_mat_device_temperature[matcell] = inout_mat_temperature[matcell];
450 }
451 };
452}

457void MaterialHeatTestModule::
458_computeCellsToAdd(const HeatObject& heat_object, MaterialWorkArray& wa)
459{
460 Real3 heat_center = heat_object.center;
461 heat_center += heat_object.velocity * m_global_time();
462 const Real heat_value = heat_object.heat_value;
463 Real heat_radius = heat_object.radius;
464 const Real heat_radius_norm = heat_radius * heat_radius;
465 IMeshMaterial* current_mat = heat_object.material;
466 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
467
468 const bool is_verbose = false;
469
470 // Détermine les nouvelles à ajouter au matériau
471 ENUMERATE_ (Cell, icell, allCells()) {
472 Cell cell = *icell;
473 AllEnvCell all_env_cell = all_env_cell_converter[cell];
474 Real3 center = m_cell_center[icell];
475 Real distance2 = (center - heat_center).squareNormL2();
476 if (distance2 < heat_radius_norm) {
477 Real to_add = heat_value / (1.0 + distance2);
478 MatCell mc = current_mat->findMatCell(all_env_cell);
479 if (mc.null()) {
480 // Si 'mc' est nul cela signifie que ce matériau n'est
481 // pas présent dans la maille. Il faudra donc l'ajouter.
482 // On conserve la valeur à ajouter pour ne pas la recalculer
483 wa.mat_cells_to_add_value.add(to_add);
484 wa.mat_cells_to_add.add(cell.localId());
485 if (is_verbose)
486 info() << "Add LID=" << cell.localId() << " T=" << to_add;
487 }
488 }
489 }
490}

495void MaterialHeatTestModule::
496_computeCellsToRemove(const HeatObject& heat_object, MaterialWorkArray& wa)
497{
498 IMeshMaterial* current_mat = heat_object.material;
499 const bool is_verbose = false;
500 const Real cold_value = 300.0;
501
502 // Refroidit chaque maille du matériau d'une quantité fixe.
503 // Si la températeure devient inférieure à zéro on supprime la maille
504 // de ce matériau.
505 if (is_verbose)
506 info() << "MAT_BEFORE: " << current_mat->matView()._internalLocalIds();
507 {
508 ENUMERATE_MATCELL (imatcell, current_mat) {
509 MatCell mc = *imatcell;
510 Real t = m_mat_temperature[mc];
511 if (t < cold_value) {
512 wa.mat_cells_to_remove.add(mc.globalCell().localId());
513 if (is_verbose)
514 info() << "Remove LID=" << mc.globalCell().localId() << " T=" << t;
515 }
516 }
517 }
518 if (is_verbose)
519 info() << "MAT_AFTER: " << current_mat->matView()._internalLocalIds();
520}

539
540void MaterialHeatTestModule::
541_computeTotalTemperature(const HeatObject& heat_object, bool do_check)
542{
543 IMeshMaterial* mat = heat_object.material;
544 Real total_mat_temperature = 0.0;
545 ENUMERATE_MATCELL (imatcell, mat) {
546 MatCell mc = *imatcell;
547 Cell cell = mc.globalCell();
548 if (cell.isOwn())
549 total_mat_temperature += m_mat_temperature[mc];
550 }
551 total_mat_temperature = parallelMng()->reduce(Parallel::ReduceSum, total_mat_temperature);
552 info() << "TotalMatTemperature mat=" << mat->name() << " T=" << total_mat_temperature;
553 if (do_check) {
554 Real ref_value = heat_object.expected_final_temperature;
555 Real current_value = total_mat_temperature;
556 Real epsilon = 1.0e-12;
557 if (!math::isNearlyEqualWithEpsilon(current_value, ref_value, epsilon)) {
558 Real relative_diff = math::abs(ref_value - current_value);
559 if (ref_value != 0.0)
560 relative_diff /= ref_value;
561 ARCANE_FATAL("Bad value for mat '{0}' ref={1} v={2} diff={3}",
562 mat->name(), ref_value, current_value, relative_diff);
563 }
564 }

569
570void MaterialHeatTestModule::
571_computeCellsCenter()
572{
573 // Calcule le centre des mailles
574 VariableNodeReal3& node_coord = defaultMesh()->nodesCoordinates();
575 ENUMERATE_ (Cell, icell, allCells()) {
576 Cell cell = *icell;
577 Real3 center;
578 auto cell_nodes = cell.nodeIds();
579 for (NodeLocalId nodeid : cell_nodes) {
580 center += node_coord[nodeid];
581 }
582 center /= cell_nodes.size();
583 m_cell_center[icell] = center;
584 }
585}

590void MaterialHeatTestModule::
591_buildHeatObjects()
592{
593 info() << "MaterialHeatTestModule::_buildHeatObjects()";
594
595 {
596 Int32 index = 0;
597 for (const auto& opt : options()->heatObject()) {
598 HeatObject ho;
599 ho.center = opt->center;
600 ho.velocity = opt->velocity;
601 ho.radius = opt->radius;
602 ho.material = _findMaterial(opt->material);
603 ho.index = index;
604 ho.expected_final_temperature = opt->expectedFinalTemperature;
605 m_heat_objects.add(ho);
606 ++index;
607 }
608 }
609
610 m_all_temperature.resize(m_heat_objects.size());
611}

627
628void MaterialHeatTestModule::
629_computeGlobalTemperature()
630{
631 // Calcule dans 'Temperature' la somme des températures des milieux et matériaux
632 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
633 auto command = makeCommand(m_queue);
634 auto inout_mat_temperature = viewInOut(command, m_mat_temperature);
635 command << RUNCOMMAND_ENUMERATE (Cell, cellid, allCells())
636 {
637 //Cell cell = *icell;
638 AllEnvCell all_env_cell = all_env_cell_converter[cellid];
639 Real global_temperature = 0.0;
640 ENUMERATE_CELL_ENVCELL (ienvcell, all_env_cell) {
641 EnvCell env_cell = *ienvcell;
642 Real env_temperature = 0.0;
643 ENUMERATE_CELL_MATCELL (imatcell, env_cell) {
644 MatCell mc = *imatcell;
645 env_temperature += inout_mat_temperature[mc];
646 }
647 inout_mat_temperature[env_cell] = env_temperature;
648 global_temperature += env_temperature;
649 }
650 inout_mat_temperature[cellid] = global_temperature;
651 };
652}

657void MaterialHeatTestModule::
658_printCellsTemperature(Int32ConstArrayView ids)
659{
660 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
661 for (Int32 lid : ids) {
662 CellLocalId cell_id(lid);
663 AllEnvCell all_env_cell = all_env_cell_converter[cell_id];
664 Cell global_cell = all_env_cell.globalCell();
665 info() << "Cell=" << global_cell.uniqueId() << " v=" << m_mat_temperature[global_cell];
666 ENUMERATE_CELL_ENVCELL (ienvcell, all_env_cell) {
667 EnvCell ec = *ienvcell;
668 info() << " EnvCell " << m_mat_temperature[ec]
669 << " mv=" << ec._varIndex()
670 << " env=" << ec.component()->name();
671 ENUMERATE_CELL_MATCELL (imatcell, (*ienvcell)) {
672 MatCell mc = *imatcell;
673 info() << " MatCell " << m_mat_temperature[mc]
674 << " mv=" << mc._varIndex()
675 << " mat=" << mc.component()->name();
676 }
677 }
678 }
