45  public:
46  constexpr Array3View(DataType* ptr,Integer dim1_size,Integer dim2_size,Integer dim3_size)
47  : m_ptr(ptr), m_dim1_size(dim1_size), m_dim2_size(dim2_size), m_dim3_size(dim3_size),
48  m_dim23_size(dim2_size*dim3_size)
49  {
50  }
51  constexpr Array3View()
52  : m_ptr(0), m_dim1_size(0), m_dim2_size(0), m_dim3_size(0), m_dim23_size(0)
53  {
54  }
55  public:
56  constexpr Integer dim1Size() const { return m_dim1_size; }
57  constexpr Integer dim2Size() const { return m_dim2_size; }
58  constexpr Integer dim3Size() const { return m_dim3_size; }
59  constexpr Integer totalNbElement() const { return m_dim1_size*m_dim23_size; }
60  public:
61  constexpr Array2View<DataType> operator[](Integer i)
62  {
63  ARCCORE_CHECK_AT(i,m_dim1_size);
64  return Array2View<DataType>(m_ptr + (m_dim23_size*i),m_dim2_size,m_dim3_size);
65  }

68  ARCCORE_CHECK_AT(i,m_dim1_size);
69  return ConstArray2View<DataType>(m_ptr + (m_dim23_size*i),m_dim2_size,m_dim3_size);
70  }

73  ARCCORE_CHECK_AT(i,m_dim1_size);
74  ARCCORE_CHECK_AT(j,m_dim2_size);
75  ARCCORE_CHECK_AT(k,m_dim3_size);
