52 public:
53  // item with one dof
54  static Int64 uid(Int64 connected_item_uid) {return connected_item_uid;} // Trivial.
55 
56  // item with several dof

58  Int64 max_connected_item_family_uid,
59  Int64 connected_item_uid,
60  Int32 dof_index_in_item)
61  { return connected_item_uid+(max_connected_item_family_uid+1)*dof_index_in_item + max_dof_family_uid+1;} // very temporary solution...
62 
63  // utilities
64  static Int64 getMaxItemUid(IItemFamily* family)
65  {
66  Int64 max_uid = 0;
67  // This method can be used within IItemFamily::endUpdate when new items have been created but the groups are not yet updated.
68  // Therefore we use internal map enumeration instead of group enumeration
69  ItemFamily* item_family = static_cast<ItemFamily*>(family);
70  ENUMERATE_ITEM_INTERNAL_MAP_DATA(item,item_family->itemsMap())
71  {
72  if (max_uid < item->value()->uniqueId().asInt64()) max_uid = item->value()->uniqueId().asInt64();
73  }
74  Int64 pmax_uid = family->mesh()->parallelMng()->reduce(Parallel::ReduceMax,max_uid);
75  return pmax_uid;
76  }
77 };
