53void EnvCellVector::
54_build(CellVectorView view)
55{
56 FixedArray<UniqueArray<ConstituentItemIndex>, 2> item_indexes;
57 IMeshComponent* my_component = _component();
58
59 ENUMERATE_ALLENVCELL(iallenvcell,_materialMng()->view(view)){
60 AllEnvCell all_env_cell = *iallenvcell;
61 ENUMERATE_CELL_ENVCELL(ienvcell,all_env_cell){
62 EnvCell ec = *ienvcell;
63 if (ec.component()==my_component){
64 MatVarIndex idx = ec._varIndex();
65 ConstituentItemIndex cii = ec._constituentItemIndex();
66 Int32 array_index = (idx.arrayIndex() == 0) ? 0 : 1;
67 item_indexes[array_index].add(cii);
68 }
69 }
70 }
71 this->_setItems(item_indexes[0], item_indexes[1]);
72}
