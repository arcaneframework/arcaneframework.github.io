53template <typename T, Int32 Extent>
54class LocalMemory
55{
56 friend Impl::LocalMemoryKernelRemainingArg;
57
58 public:
59
60 static_assert(std::is_trivially_copyable_v<T>, "type T is not trivially copiable");
61
62 public:
63
64 using SpanType = SmallSpan<T, Extent>;
65 using RemainingArgHandlerType = Impl::LocalMemoryKernelRemainingArg;
66
67 public:
68
69 LocalMemory(RunCommand& command, Int32 size)
70 : m_size(size)
71 {
72 _addShareMemory(command);
73 }
74
75 explicit LocalMemory(RunCommand& command) requires(Extent != DynExtent)
76 {
77 _addShareMemory(command);
78 }
79
80 constexpr ARCCORE_HOST_DEVICE SmallSpan<T, Extent> span()
81 {
82 return { m_ptr, m_size.size() };
83 }
84
85 private:
86
87 T* m_ptr = nullptr;
88 // TODO: l'offset n'est utilisé on pourrait supprimer l'offset en le passant
90 Int32 m_offset = 0;
92 [[no_unique_address]] ::Arcane::Impl::ExtentStorage<Int32, Extent> m_size;
93
94 protected:
95
96 void _addShareMemory(RunCommand& command)
97 {
98 m_offset = command._addSharedMemory(static_cast<Int32>(sizeof(T) * m_size.size()));
99 }
100};
101
102} // namespace Arcane::Accelerator
