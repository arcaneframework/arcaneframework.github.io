149 // Calcul de la consommation mémoire pour chaque type d'entité.
150 // Les mailles bénéficient ensuite des contributions des autres entités adjacentes.
151 void computeMemory(IVariableMng* varMng)
152 {
153 Int32 length = m_family_names.size();
154 m_overall_memory.resize(length);
155 m_resident_memory.resize(length);
156 m_buffer.id = -1;
157 m_overall_memory.fill(0);
158 m_resident_memory.fill(0);
159
160 // For each variable, compute the size for one object.
161 for (VariableCollectionEnumerator vc(varMng->usedVariables()); ++vc;) {
162 const IVariable* var = *vc;
163 Integer memory = 0;
164 try {
165 if (var->dataType() != DT_String)
166 memory = dataTypeSize(var->dataType());
167 }
168 catch (const ArgumentException&) {
169 memory = 0; // Cannot know memory used for that ...
170 continue;
171 }
172 Int32 family_index = -1;
173 Integer kind = var->itemKind();
174 if (kind == IK_Particle) { // Not the same counter for all items
175 family_index = _findEntity(var->itemFamilyName());
176 if (family_index >= 0) {
177 m_overall_memory[family_index] += memory;
178 }
179 }
180 m_overall_memory[kind] += memory;
181
182 int properties = var->property();
183 if ((properties & IVariable::PNoExchange) ||
184 (properties & IVariable::PNoNeedSync) ||
185 (properties & IVariable::PTemporary) ||
186 (properties & IVariable::PSubDomainPrivate)) {
187 continue;
188 }
189 m_resident_memory[kind] += memory;
190 if (family_index >= 0) {
191 m_resident_memory[family_index] += memory;
192 }
193 }
194 }

255 // Pour les mailles, on calcule la contribution mémoire des noeuds, aretes et faces.
256 void _computeMemCell(Cell cell)
257 {
258 Real contrib;
259 if (cell.localId() == m_buffer.id) // already computed
260 return;
261 m_buffer.id = cell.localId();
262 m_buffer.overall_memory = m_overall_memory[IK_Cell];
263 m_buffer.resident_memory = m_resident_memory[IK_Cell];
264
265 contrib = _computeMemContrib<Node>(cell.nodes());
266 m_buffer.overall_memory += contrib * m_overall_memory[IK_Node];
267 m_buffer.resident_memory += contrib * m_resident_memory[IK_Node];
268
269 contrib = _computeMemContrib<Face>(cell.faces());
270 m_buffer.overall_memory += contrib * m_overall_memory[IK_Face];
271 m_buffer.resident_memory += contrib * m_resident_memory[IK_Face];
272
273 contrib = _computeMemContrib<Edge>(cell.edges());
274 m_buffer.overall_memory += contrib * m_overall_memory[IK_Edge];
275 m_buffer.resident_memory += contrib * m_resident_memory[IK_Edge];
276 }
277
279 template <typename ItemKind>
280 Real _computeMemContrib(ItemConnectedListViewTypeT<ItemKind> list)
281 {
282 Real contrib = 0.0;
283 //ItemEnumeratorT<ItemKind> iterator = list.enumerator();
284 for (const auto& item : list) {
285 contrib += 1.0 / (Real)(item.nbCell());
286 }
287 return contrib;
288 }

504 void _computeOverallMass()
505 {
506 VariableCellReal& mass_over_weigth = *m_mass_over_weigth;
507 ENUMERATE_CELL (icell, m_mesh->ownCells()) {
508 mass_over_weigth[icell] = m_criteria->getOverallMemory(*icell);
509 }
510 }
511
512 void _computeComm()
513 {
514 VariableFaceReal& comm_costs = *m_comm_costs;
515 VariableCellReal& mass_res_weight = *m_mass_res_weight;
516
517 Integer penalty = 2; // How many times we do synchronization ?
518
519 if (!m_comm_vars.empty())
520 comm_costs.fill(0);
521
522 for (auto& commvar : m_comm_vars) {
523 ENUMERATE_FACE (iface, m_mesh->ownFaces()) {
524 comm_costs[iface] += commvar[iface] * m_criteria->getResidentMemory(commvar.getPos());
525 }
526 }
527 if (m_cell_comm) {
528 ENUMERATE_CELL (icell, m_mesh->ownCells()) {
529 Real mem = mass_res_weight[icell];
530 for (Face face : icell->faces()) {
531 comm_costs[face] += mem * penalty;
532 }
533 }
534 }
535
536 IVariable* ivar;
537 // Make sure that ghosts contribution is used
538 ivar = m_comm_costs->variable();
539 ivar->itemFamily()->reduceFromGhostItems(ivar, Parallel::ReduceSum);
540 m_comm_costs->synchronize();
541 }
542
543 void _computeResidentMass()
544 {
545 VariableCellReal& mass_res_weight = *m_mass_res_weight;
546 ENUMERATE_CELL (icell, m_mesh->ownCells()) {
547 mass_res_weight[icell] = m_criteria->getResidentMemory(*icell);
548 }
549 }
550
551 void _computeEvents()
552 {
553 m_event_weights->resize(nbCriteria());
554
555 ArrayView<StoreIProxyItemVariable> event_vars = criteria();
556
557 VariableCellArrayReal& event_weights = *(m_event_weights);
558
559 for (Integer i = 0; i < event_vars.size(); ++i) {
560 ENUMERATE_CELL (icell, m_mesh->ownCells()) {
561 Integer count = i;
562 if (m_nb_cells_as_criterion) {
563 count += 1;
564 event_weights(icell, 0) = 1;
565 }
566 event_weights(icell, count) = event_vars[i][icell];
567 }
568 }
569 }
