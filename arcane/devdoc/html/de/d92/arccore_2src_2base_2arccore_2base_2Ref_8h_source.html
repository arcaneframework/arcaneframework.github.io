126 template<typename InstanceType,int ImplTagId>
127 class Ref
128 : public RefBase
129 {
130  public:
131  typedef Ref<InstanceType,ImplTagId> ThatClass;
132  typedef typename RefTraitsTagId<InstanceType,ImplTagId>::ImplType ImplType;
133  private:
134 
135  template<typename... _Args>
136  using _IsRefConstructible = typename std::enable_if<std::is_constructible<ImplType, _Args...>::value>::type;
137 
138  using RefBase::DeleterBase;
139  class Deleter : DeleterBase
140  {
141  public:
142  Deleter(Internal::ExternalRef h) : m_handle(h), m_no_destroy(false){}
143  Deleter(Internal::ExternalRef h,bool no_destroy)
144  : m_handle(h), m_no_destroy(no_destroy){}
145  void operator()(InstanceType* tt)
146  {
147  if (m_no_destroy)
148  return;
149  bool is_destroyed = this->_destroyHandle(tt,m_handle);
150  if (!is_destroyed)
151  delete tt;
152  }
153  void setNoDestroy(bool x) { m_no_destroy = x; }
154  private:
155  Internal::ExternalRef m_handle;
156  bool m_no_destroy;
157  };
158  public:
159  static constexpr int RefType = RefTraitsTagId<InstanceType,ImplTagId>::RefType;
160  private:
161  explicit Ref(InstanceType* t) : m_instance(t,Deleter(nullptr)){}
162  Ref(InstanceType* t,Internal::ExternalRef handle) : m_instance(t,Deleter(handle)){}
163  Ref(InstanceType* t,bool no_destroy) : m_instance(t,Deleter(nullptr,no_destroy)){}
164  private:
165  Ref(ImplType&& t) : m_instance(t) {}
166  public:
173  template<typename T, typename = _IsRefConstructible<typename Ref<T>::ImplType> >
174  Ref(const Ref<T>& rhs) noexcept
175  : m_instance(rhs._internalInstance()){}
176  Ref() = default;
177  Ref(const ThatClass& rhs) = default;
178  ThatClass& operator=(const ThatClass& rhs) = default;
179  ~Ref() = default;
180 
181  public:
191  static ThatClass create(InstanceType* t)
192  {
193  return ThatClass(t);
194  }
195 
196  template<typename PointerType, typename... Args>
197  static inline Ref<InstanceType>
198  createRef(Args&&... args)
199  {
200  PointerType* pt = new PointerType(std::forward<Args>(args)...);
201  return Ref<InstanceType>(pt);
202  }
203 
209  static ThatClass createWithHandle(InstanceType* t,Internal::ExternalRef handle)
210  {
211  return ThatClass(t,handle);
212  }
213 
214  static ThatClass _createNoDestroy(InstanceType* t)
215  {
216  return ThatClass(t,true);
217  }
218 
219  public:
221  InstanceType* get() const { return m_instance.get(); }
223  bool isNull() const { return m_instance.get()==nullptr; }
224  InstanceType* operator->() const { return m_instance.get(); }
226  void reset() { m_instance.reset(); }
232  InstanceType* _release()
233  {
234  // Relâche l'instance. Pour cela, on indique au destructeur
235  // de ne pas détruire l'instance de 'm_instance' et on
236  // retourne cette dernière.
237  Deleter* r = std::get_deleter<Deleter>(m_instance);
238  if (r)
239  r->setNoDestroy(true);
240  InstanceType* t = m_instance.get();
241  m_instance.reset();
242  return t;
243  }
244  const ImplType& _internalInstance() const { return m_instance; }
245  private:
246  ImplType m_instance;
247 };
