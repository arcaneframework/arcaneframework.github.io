56Int32 MaterialModifierOperation::
57_checkMaterialPresence(MaterialModifierOperation* operation)
58{
59 IMeshMaterial* mat = operation->material();
60 Int32ConstArrayView ids = operation->ids();
61
62 const MeshMaterialVariableIndexer* indexer = mat->_internalApi()->variableIndexer();
63 IItemFamily* item_family = mat->cells().itemFamily();
64 ItemInfoListView items_internal(item_family);
65 Integer max_local_id = item_family->maxLocalId();
66 UniqueArray<bool> presence_flags(max_local_id,false);
67 Int32ConstArrayView mat_local_ids = indexer->localIds();
68 Integer nb_error = 0;
69 String name = mat->name();
70 ITraceMng* tm = mat->traceMng();
71
72 for( Int32 lid : ids ){
73 if (presence_flags[lid]){
74 tm->info() << "ERROR: item " << ItemPrinter(items_internal[lid])
75 << " is present several times in add/remove list for material mat=" << name;
76 ++nb_error;
77 }
78 presence_flags[lid] = true;
79 }
80
81 if (operation->isAdd()){
82 for( Int32 lid : mat_local_ids ){
83 if (presence_flags[lid]){
84 tm->info() << "ERROR: item " << ItemPrinter(items_internal[lid])
85 << " is already in material mat=" << name;
86 ++nb_error;
87 }
88 }
89 }
90 else{
91 for( Int32 lid : mat_local_ids ){
92 presence_flags[lid]= false;
93 }
94
95 for( Int32 lid : ids ){
96 if (presence_flags[lid]){
97 tm->info() << "ERROR: item " << ItemPrinter(items_internal[lid])
98 << " is not in material mat=" << name;
99 ++nb_error;
100 }
101 }
102 }
103
104 return nb_error;
105}

119void MaterialModifierOperation::
120_filterValidIds(MaterialModifierOperation* operation, Int32Array& valid_ids)
121{
122 IMeshMaterial* mat = operation->material();
123 const bool do_add = operation->isAdd();
124 Int32ConstArrayView ids = operation->ids();
125 const MeshMaterialVariableIndexer* indexer = mat->_internalApi()->variableIndexer();
126 const IItemFamily* item_family = mat->cells().itemFamily();
127 Integer max_local_id = item_family->maxLocalId();
128 UniqueArray<bool> presence_flags(max_local_id, false);
129 Int32ConstArrayView mat_local_ids = indexer->localIds();
130 ITraceMng* tm = mat->traceMng();
131
132 UniqueArray<Int32> unique_occurence_lids;
133 unique_occurence_lids.reserve(ids.size());
134
135 for (Int32 lid : ids) {
136 if (!presence_flags[lid]) {
137 unique_occurence_lids.add(lid);
138 presence_flags[lid] = true;
139 }
140 }
141
142 valid_ids.clear();
143
144 if (do_add) {
145 for (Int32 lid : mat_local_ids) {
146 if (presence_flags[lid]) {
147 ;
148 }
149 else
150 valid_ids.add(lid);
151 }
152 }
153 else {
154 for (Int32 lid : mat_local_ids)
155 presence_flags[lid] = false;
156
157 for (Int32 lid : unique_occurence_lids) {
158 if (presence_flags[lid]) {
159 ;
160 }
161 else
162 valid_ids.add(lid);
163 }
164 }
165 tm->info(4) << "FILTERED_IDS n=" << valid_ids.size() << " ids=" << valid_ids;
166}

171void MaterialModifierOperation::
172filterIds()
173{
174 // TODO: changer le défaut à 'false' et tester les deux configurations
175 const bool filter_invalid = true;
176 Integer nb_error = _checkMaterialPresence(this);
177 if (nb_error != 0) {
178 if (filter_invalid) {
179 UniqueArray<Int32> filtered_ids;
180 _filterValidIds(this, filtered_ids);
181 m_ids.swap(filtered_ids);
182 }
183 else
184 ARCANE_FATAL("Invalid values for adding items in material name={0} nb_error={1}",
185 m_mat->name(), nb_error);
186 }
187}
