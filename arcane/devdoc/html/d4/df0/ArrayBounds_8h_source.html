46 template<int RankValue>
47 class ArrayBoundsBase
48 : private ArrayExtents<RankValue>
49 {
50  public:
51  using ArrayExtents<RankValue>::extent;
52  public:
53  constexpr ArrayBoundsBase() : m_nb_element(0) {}
54  public:
55  ARCCORE_HOST_DEVICE constexpr Int64 nbElement() const { return m_nb_element; }
56  ARCCORE_HOST_DEVICE constexpr std::array<Int32,RankValue> asStdArray() const { return ArrayExtents<RankValue>::asStdArray(); }
57  protected:
58  constexpr void _computeNbElement()
59  {
60  m_nb_element = this->totalNbElement();
61  }
62  protected:
63  using ArrayExtents<RankValue>::m_extents;
64  Int64 m_nb_element;
65 };

110 template<>
111 class ArrayBounds<3>
112 : public ArrayBoundsBase<3>
113 {
114  public:
115  using IndexType = ArrayBoundsIndex<3>;
116  using ArrayBoundsBase<3>::m_extents;
117  constexpr ArrayBounds(Int32 dim1,Int32 dim2,Int32 dim3)
118  : ArrayBoundsBase<3>()
119  {
120  m_extents[0] = dim1;
121  m_extents[1] = dim2;
122  m_extents[2] = dim3;
123  _computeNbElement();
124  }
125  ARCCORE_HOST_DEVICE constexpr IndexType getIndices(Int32 i) const
126  {
127  Int32 i2 = impl::fastmod(i,m_extents[2]);
128  Int32 fac = m_extents[2];
129  Int32 i1 = impl::fastmod(i / fac,m_extents[1]);
130  fac *= m_extents[1];
131  Int32 i0 = i / fac;
132  return { i0, i1, i2 };
133  }
134 };

150  _computeNbElement();
151  }
152  ARCCORE_HOST_DEVICE constexpr IndexType getIndices(Int32 i) const
153  {
154  // Compute base indices
155  Int32 i3 = impl::fastmod(i,m_extents[3]);
156  Int32 fac = m_extents[3];
157  Int32 i2 = impl::fastmod(i/fac,m_extents[2]);
158  fac *= m_extents[2];
159  Int32 i1 = impl::fastmod(i/fac,m_extents[1]);
160  fac *= m_extents[1];
161  Int32 i0 = i /fac;
162  return { i0, i1, i2, i3 };
163  }
