238void Exception::
239write(std::ostream& o) const
240{
241 o << "Exception!\n\nThrown in: '" << m_where
242 << "'\nType: '" << m_name << "'\n\n";
243 if (!m_message.null())
244 o << "Message: " << m_message << '\n';
245 this->explain(o);
246 String st = m_stack_trace.toString();
247 if (!st.null()){
248 o << "\nCall stack:\n";
249 o << st << '\n';
250 }
251}

292
293void Exception::
294_checkExplainAndPause()
295{
296 if (global_explain_in_constructor){
297 std::cerr << "** Exception:" << (*this) << "\n";
298 }
299
300 if (global_pause_in_constructor){
301 std::cerr << "** Exception: Debug mode activated. Execution paused.\n";
302 std::cerr << "** Exception: To find the location of the exception, start the debugger\n";
303 // Utilise format pour être sur que le message ne sera pas affiché en plusieurs
304 // morceaux
305 std::cerr << String::format("** Exception: using process number {0} on host '{1}'\n",
306 Platform::getProcessId(),Platform::getHostName());
307
308#ifndef ARCCORE_OS_WIN32
309 ::pause();
310#endif
311 }
