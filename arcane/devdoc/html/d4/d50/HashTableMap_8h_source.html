468  inline Data* _baseRemoveBucket(Integer bucket,KeyTypeConstRef id)
469  {
470  Data* i = m_buckets[bucket];
471  if (i){
472  if (i->m_key==id){
473  m_buckets[bucket] = i->next();
474  --m_count;
475  return i;
476  }
477  for( ; i->next(); i=i->next() ){
478  if (i->next()->key()==id){
479  Data* r = i->next();
480  i->setNext(i->next()->next());
481  --m_count;
482  return r;
483  }
484  }
485  }
486  this->_throwNotFound(id,Printable());
487  return 0;
488  }

604  HashTableMapEnumeratorT(const HashType& rhs)
605  : m_buckets(rhs.buckets()), m_current_data(0), m_current_bucket(-1) {}
606 
607  public:
608 
609  bool operator++()
610  {
611  if (m_current_data)
612  m_current_data = m_current_data->next();
613  if (!m_current_data){
614  while (m_current_data==0 && (m_current_bucket+1)<m_buckets.size()){
615  ++m_current_bucket;
616  m_current_data = m_buckets[m_current_bucket];
617  }
618  }
619  return m_current_data!=0;
620  }
621  ValueType& operator*() { return m_current_data->value(); }
622  const ValueType& operator*() const { return m_current_data->value(); }
623  Data* data() { return m_current_data; }
624  const Data* data() const { return m_current_data; }
625 
626  public:
627 
628  ConstArrayView<Data*> m_buckets;
629  Data* m_current_data;
630  Integer m_current_bucket;
