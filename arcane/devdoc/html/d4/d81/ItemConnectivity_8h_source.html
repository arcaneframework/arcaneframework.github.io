94 class ARCANE_MESH_EXPORT ItemConnectivity
95 : public AbstractConnectivity
96 {
97 public:
98  typedef ItemScalarProperty<Int32> ItemPropertyType;
99 
100  public:
101  ItemConnectivity(IItemFamily* source_family, IItemFamily* target_family,const String& aname)
102  : AbstractConnectivity(source_family,target_family,aname)
103  {
104  compute();
105  }
106 
107  ItemConnectivity(IItemFamily* source_family, IItemFamily* target_family,const ItemPropertyType& item_property, const String& aname)
108  : AbstractConnectivity(source_family,target_family,aname) // IFPEN : voir le paramètre own de GE...
109  {
110  m_item_property.copy(item_property);
111  }
112 
113  public:
114 
115  virtual ItemVectorView _connectedItems(ItemLocalId item,ConnectivityItemVector& con_items) const
116  {
117  ARCANE_ASSERT((con_items.accessor()==this),("Bad connectivity"));
118  return con_items.setItem(m_item_property[item]);
119  }
120 
121  virtual ConnectivityItemVectorCatalyst _connectedItems(ItemLocalId item) const
122  {
123  auto set = [this](ConnectivityItemVector& civ)mutable{civ = ConnectivityItemVector((ItemConnectivity*)this);};
124  auto apply = [this,item](ConnectivityItemVector& civ){this->_connectedItems(item,civ);};
125  return {set,apply};
126  }
127 
128  virtual void updateConnectivity(Int32ConstArrayView from_items, Int32ConstArrayView to_items);
129 
130  public:
131 
132  const Item operator() (ItemLocalId item) const
133  {
134  /*
135  ARCANE_ASSERT((m_item_property[item] != NULL_ITEM_LOCAL_ID),
136  ("Item must be connected to one item in ItemConnectivity."));
137  //TODO: conserver ItemInternalList pour des raisons de performance.
138  return Item(_targetFamily()->itemsInternal()[m_item_property[item]]);
139  */
140  // Needed for IFPEN applicative test: eventually returns a null item (reasonable for perf ?)
141  if (m_item_property[item] != NULL_ITEM_LOCAL_ID)
142  return Item(_targetFamily()->itemsInternal()[m_item_property[item]]);
143  else return Item();
144  }
145 
146  ItemScalarProperty<Int32>& itemProperty() {return m_item_property;}
147 
148  void updateItemProperty(const ItemScalarProperty<Int32>& item_property) {m_item_property.copy(item_property);}
149 
150  virtual Integer nbConnectedItem(ItemLocalId lid) const
151  {
152  ARCANE_UNUSED(lid);
153  return 1;
154  }
155 
156  virtual Int32 connectedItemLocalId(ItemLocalId lid,Integer index) const
157  {
158  ARCANE_ASSERT((index==0),("Invalid value for index"))
159  return m_item_property[lid];
160  }
161 
163  virtual void notifySourceFamilyLocalIdChanged(Int32ConstArrayView new_to_old_ids)
164  {
165  m_item_property.updateSupport(new_to_old_ids);
166  }
167 
169  virtual void notifyTargetFamilyLocalIdChanged(Int32ConstArrayView old_to_new_ids);
170 
171  private:
172 
173  ItemScalarProperty<Int32> m_item_property;
174  SharedArray<ItemInternal*> m_item_internals;
175 
176  private:
177 
178  void compute();
179 };

273  return m_item_property[lid][index];
