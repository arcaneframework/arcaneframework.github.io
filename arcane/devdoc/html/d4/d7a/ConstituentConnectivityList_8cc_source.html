38 // Cette méthode est la même que MeshUtils::removeItemAndKeepOrder().
39 // Il faudrait pouvoir fusionner les deux.
40 // NOTE: Avec le C++20, on devrait pouvoir supprimer cette méthode et
41 // utiliser std::erase()
42
43 template <typename DataType>
44 ARCCORE_HOST_DEVICE void _removeValueAndKeepOrder(ArrayView<DataType> values, DataType value_to_remove)
45 {
46 Integer n = values.size();
47#ifndef ARCCORE_DEVICE_CODE
48 if (n <= 0)
49 ARCANE_FATAL("Can not remove item lid={0} because list is empty", value_to_remove);
50#endif
51 --n;
52 if (n == 0) {
53 if (values[0] == value_to_remove)
54 return;
55 }
56 else {
57 // Si l'élément est le dernier, ne fait rien.
58 if (values[n] == value_to_remove)
59 return;
60 for (Integer i = 0; i < n; ++i) {
61 if (values[i] == value_to_remove) {
62 for (Integer z = i; z < n; ++z)
63 values[z] = values[z + 1];
64 return;
65 }
66 }
67 }
68#if defined(ARCCORE_DEVICE_CODE)
69 // Avec Intel DPC++ 2024.1 l'utilisation de cette fonction avec le
70 // backend CUDA provoque une erreur dans l'assembleur généré.
71# if !defined(__INTEL_LLVM_COMPILER)
72 assert(false);
73# endif
74#else
75 ARCANE_FATAL("No value to remove '{0}' found in list {1}", value_to_remove, values);
76#endif
77 }

298void ConstituentConnectivityList::
299endCreate(bool is_continue)
300{
301 // Maintenant (février 2024) on construit toujours les connectivités incrémentales
302 const bool always_build_connectivity = true;
303
304 // S'enregistre auprès la famille pour être notifié des évolutions
305 // mais uniquement si on a demandé le support des modifications incrémentales
306 // pour éviter de consommer inutilement de la mémoire.
307 // A terme on le fera tout le temps
308 m_cell_family = m_material_mng->mesh()->cellFamily();
309 {
310 int opt_flag_value = m_material_mng->modificationFlags();
311 bool use_incremental = (opt_flag_value & (int)eModificationFlags::IncrementalRecompute) != 0;
312 if (use_incremental || always_build_connectivity) {
313 m_cell_family->_internalApi()->addSourceConnectivity(this);
314 m_is_active = true;
315 info() << "Activating incremental material connectivities";
316 }
317 }
318 if (!is_continue) {
319 Int32 max_local_id = m_cell_family->maxLocalId();
320 m_container->checkResize(max_local_id + 1);
321 m_container->m_environment.endCreate(is_continue);
322 m_container->m_material.endCreate(is_continue);
323 }
324
325 // Remplit un tableau indiquant pour chaque index de matériau l'index
326 // du milieu correspondant.
327 {
328 ConstArrayView<MeshMaterial*> materials = m_material_mng->trueMaterials();
329 const Int32 nb_mat = materials.size();
330 auto environment_for_materials = m_environment_for_materials.hostModifier();
331 environment_for_materials.resize(nb_mat);
332 auto local_view = environment_for_materials.view();
333 for (Int32 i = 0; i < nb_mat; ++i)
334 local_view[i] = materials[i]->trueEnvironment()->componentId();
335 }
336}

341void ConstituentConnectivityList::
342_addCells(Int16 component_id, SmallSpan<const Int32> cells_local_id,
343 ConstituentContainer& component, RunQueue& queue)
344{
345 const Int32 nb_item = cells_local_id.size();
346 if (nb_item == 0)
347 return;
348 Array<Int16>& nb_component = component.m_nb_component_as_array;
349 Array<Int32>& component_index = component.m_component_index_as_array;
350 Array<Int16>& component_list = component.m_component_list_as_array;
351
352 SmallSpan<Int16> nb_component_view = component.m_nb_component_as_array.view();
353
354 NumArray<Int32, MDDim1> new_indexes(nb_item);
355 const bool is_device = queue.isAcceleratorPolicy();
356 // Pour recopier le nombre d'éléments à ajouter du device vers le CPU
357 NumArray<Int32, MDDim1> new_indexes_to_add(1, eMemoryRessource::HostPinned);
358
359 // Calcul l'index des nouveaux éléments
360 {
361 Accelerator::GenericScanner scanner(queue);
362 SmallSpan<Int32> new_indexes_view = new_indexes;
363 SmallSpan<Int32> new_indexes_to_add_view = new_indexes_to_add;
364 auto getter = [=] ARCCORE_HOST_DEVICE(Int32 index) -> Int32 {
365 return 1 + nb_component_view[cells_local_id[index]];
366 };
367 auto setter = [=] ARCCORE_HOST_DEVICE(Int32 index, Int32 value) {
368 new_indexes_view[index] = value;
369 if (index == (nb_item - 1))
370 new_indexes_to_add_view[0] = new_indexes_view[index] + nb_component_view[cells_local_id[index]] + 1;
371 };
372 Accelerator::ScannerSumOperator<Int32> op;
373 scanner.applyWithIndexExclusive(nb_item, 0, getter, setter, op, A_FUNCINFO);
374 }
375 queue.barrier();
376
377 const Int32 nb_indexes_to_add = new_indexes_to_add[0];
378 const Int32 current_list_index = component_list.size();
379
380 MemoryUtils::checkResizeArrayWithCapacity(component_list, current_list_index + nb_indexes_to_add, false);
381
382 {
383 auto command = makeCommand(queue);
384 SmallSpan<Int16> nb_component_view = nb_component.view();
385 SmallSpan<Int32> component_index_view = component_index.view();
386 SmallSpan<Int16> component_list_view = component_list.view();
387 SmallSpan<const Int32> new_indexes_view = new_indexes;
388 command << RUNCOMMAND_LOOP1(iter, nb_item)
389 {
390 auto [i] = iter();
391 Int32 cell_id = cells_local_id[i];
392 const Int16 n = nb_component_view[cell_id];
393 Int32 new_pos = current_list_index + new_indexes_view[i];
394 if (n != 0) {
395 // Recopie les anciennes valeurs.
396 // TODO: cela laisse des trous dans la liste qu'il faudra supprimer
397 // via un compactage.
398 Int32 current_pos = component_index_view[cell_id];
399 SmallSpan<const Int16> current_values(&component_list_view[current_pos], n);
400 SmallSpan<Int16> new_values(&component_list_view[new_pos], n);
401 new_values.copy(current_values);
402 }
403 component_index_view[cell_id] = new_pos;
404 component_list_view[new_pos + n] = component_id;
405 ++nb_component_view[cell_id];
406 };
407 }
408
409 component.notifyUpdateConnectivityList();
410}

415void ConstituentConnectivityList::
416_removeCells(Int16 component_id, SmallSpan<const Int32> cells_local_id,
417 ConstituentContainer& component, RunQueue& queue)
418{
419 SmallSpan<Int16> nb_component = component.m_nb_component_as_array.view();
420 SmallSpan<Int32> component_index = component.m_component_index_as_array.view();
421 SmallSpan<Int16> component_list = component.m_component_list_as_array.view();
422
423 const Int32 n = cells_local_id.size();
424 auto command = makeCommand(queue);
425 command << RUNCOMMAND_LOOP1(iter, n)
426 {
427 auto [i] = iter();
428 Int32 id = cells_local_id[i];
429 //for (Int32 id : cell_ids) {
430 CellLocalId cell_id(id);
431 const Int32 current_pos = component_index[cell_id];
432 const Int32 n = nb_component[cell_id];
433 ArrayView<Int16> current_values(n, &component_list[current_pos]);
434 // Enlève de la liste le milieu supprimé
435 _removeValueAndKeepOrder(current_values, component_id);
436 // Met une valeur invalide pour indiquer que l'emplacement est libre
437 current_values[n - 1] = (-1);
438 --nb_component[cell_id];
439 };
440}

545
546void ConstituentConnectivityList::
547fillCellsToTransform(SmallSpan<const Int32> cells_local_id, Int16 env_id,
548 SmallSpan<bool> cells_do_transform, bool is_add, RunQueue& queue)
549{
550 ConstituentContainer::View materials_container_view(m_container->m_material);
551 bool is_device = queue.isAcceleratorPolicy();
552 auto environment_for_materials = m_environment_for_materials.view(is_device);
553
554 NumberOfMaterialComputer nb_mat_computer(materials_container_view, environment_for_materials);
555
556 SmallSpan<const Int16> cells_nb_env = cellsNbEnvironment();
557 const Int32 n = cells_local_id.size();
558 auto command = makeCommand(queue);
559 command << RUNCOMMAND_LOOP1(iter, n)
560 {
561 auto [i] = iter();
562 Int32 local_id = cells_local_id[i];
563 bool do_transform = false;
564 // En cas d'ajout on passe de pure à partiel s'il y a plusieurs milieux ou
565 // plusieurs matériaux dans le milieu.
566 // En cas de supression, on passe de partiel à pure si on est le seul matériau
567 // et le seul milieu.
568 const Int16 nb_env = cells_nb_env[local_id];
569 if (is_add) {
570 do_transform = (nb_env > 1);
571 if (!do_transform)
572 do_transform = nb_mat_computer.cellNbMaterial(local_id, env_id) > 1;
573 }
574 else {
575 do_transform = (nb_env == 1);
576 if (do_transform)
577 do_transform = nb_mat_computer.cellNbMaterial(local_id, env_id) == 1;
578 }
579 cells_do_transform[local_id] = do_transform;
580 };
581}

586void ConstituentConnectivityList::

588 SmallSpan<bool> cells_is_partial, RunQueue& queue)
589{
590 ConstituentContainer::View materials_container_view(m_container->m_material);

592 auto environment_for_materials = m_environment_for_materials.view(is_device);
593 NumberOfMaterialComputer nb_mat_computer(materials_container_view, environment_for_materials);
594 SmallSpan<const Int16> cells_nb_env = cellsNbEnvironment();
595 const Int32 n = cells_local_id.size();
596 auto command = makeCommand(queue);
597
598 command << RUNCOMMAND_LOOP1(iter, n)
599 {
600 auto [i] = iter();
601 Int32 local_id = cells_local_id[i];
602 // On ne prend l'indice global que si on est le seul matériau et le seul
603 // milieu de la maille. Sinon, on prend un indice multiple
604 bool is_partial = (cells_nb_env[local_id] > 1 || nb_mat_computer.cellNbMaterial(local_id, env_id) > 1);
605 cells_is_partial[i] = is_partial;
606 };
