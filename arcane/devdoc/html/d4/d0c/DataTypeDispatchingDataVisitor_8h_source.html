213  template<template<typename DataType> class TrueDispatcherType,typename BuildArgType> static ThatClass*
214  create(BuildArgType arg)
215  {
216  typedef typename IDispatcherType::HasStringDispatch HasStringDispatch;
217  return _create<TrueDispatcherType,BuildArgType>(arg,HasStringDispatch());
218  }
219 
220  private:

225  return _create2(arg,a_string,a_string);
226  }
227  template<template<typename DataType> class TrueDispatcherType,typename BuildArgType> static ThatClass*
228  _create(BuildArgType arg,FalseType)

230  return _create2<TrueDispatcherType,BuildArgType>(arg,0,0);
231  }
232  template<template<typename DataType> class TrueDispatcherType,typename BuildArgType> static ThatClass*
233  _create2(BuildArgType arg,IDataTypeDataDispatcherT<String>* a_string,IDispatcherType* i_string)
234  {
235  TrueDispatcherType<Byte>* a_byte = new TrueDispatcherType<Byte>(arg);
236  TrueDispatcherType<Real>* a_real = new TrueDispatcherType<Real>(arg);
237  TrueDispatcherType<Int16>* a_int16 = new TrueDispatcherType<Int16>(arg);
238  TrueDispatcherType<Int32>* a_int32 = new TrueDispatcherType<Int32>(arg);
239  TrueDispatcherType<Int64>* a_int64 = new TrueDispatcherType<Int64>(arg);
240  TrueDispatcherType<Real2>* a_real2 = new TrueDispatcherType<Real2>(arg);
241  TrueDispatcherType<Real3>* a_real3 = new TrueDispatcherType<Real3>(arg);
242  TrueDispatcherType<Real2x2>* a_real2x2 = new TrueDispatcherType<Real2x2>(arg);
243  TrueDispatcherType<Real3x3>* a_real3x3 = new TrueDispatcherType<Real3x3>(arg);
244  ThatClass* p = new ThatClass(a_byte,a_real,a_int16,a_int32,a_int64,a_real2,a_real3,
245  a_real2x2,a_real3x3,a_string);
246  p->setDispatchers(a_byte,a_real,a_int16,a_int32,a_int64,a_real2,a_real3,
247  a_real2x2,a_real3x3,i_string);
