216 template<template<typename DataType> class TrueDispatcherType,typename BuildArgType> static ThatClass*
217 create(BuildArgType arg)
218 {
219 typedef typename IDispatcherType::HasStringDispatch HasStringDispatch;
220 return _create<TrueDispatcherType,BuildArgType>(arg,HasStringDispatch());

224 template<template<typename DataType> class TrueDispatcherType,typename BuildArgType> static ThatClass*
225 _create(BuildArgType arg,TrueType)

228 return _create2(arg,a_string,a_string);
229 }
230 template<template<typename DataType> class TrueDispatcherType,typename BuildArgType> static ThatClass*
231 _create(BuildArgType arg,FalseType)

233 return _create2<TrueDispatcherType,BuildArgType>(arg,0,0);
234 }
235 template<template<typename DataType> class TrueDispatcherType,typename BuildArgType> static ThatClass*
236 _create2(BuildArgType arg,IDataTypeDataDispatcherT<String>* a_string,IDispatcherType* i_string)
237 {
238 TrueDispatcherType<Byte>* a_byte = new TrueDispatcherType<Byte>(arg);
239 TrueDispatcherType<Real>* a_real = new TrueDispatcherType<Real>(arg);
240 TrueDispatcherType<Int16>* a_int16 = new TrueDispatcherType<Int16>(arg);
241 TrueDispatcherType<Int32>* a_int32 = new TrueDispatcherType<Int32>(arg);
242 TrueDispatcherType<Int64>* a_int64 = new TrueDispatcherType<Int64>(arg);
243 TrueDispatcherType<Real2>* a_real2 = new TrueDispatcherType<Real2>(arg);
244 TrueDispatcherType<Real3>* a_real3 = new TrueDispatcherType<Real3>(arg);
245 TrueDispatcherType<Real2x2>* a_real2x2 = new TrueDispatcherType<Real2x2>(arg);
246 TrueDispatcherType<Real3x3>* a_real3x3 = new TrueDispatcherType<Real3x3>(arg);
247 ThatClass* p = new ThatClass(a_byte,a_real,a_int16,a_int32,a_int64,a_real2,a_real3,
248 a_real2x2,a_real3x3,a_string);
249 p->setDispatchers(a_byte,a_real,a_int16,a_int32,a_int64,a_real2,a_real3,
250 a_real2x2,a_real3x3,i_string);
251 return p;
252 }
