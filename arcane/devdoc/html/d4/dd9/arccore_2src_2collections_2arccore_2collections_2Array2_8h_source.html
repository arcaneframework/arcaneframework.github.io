377 void _resize(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
378 {
379 if (new_size2==m_md->dim2_size){
380 _resize(new_size1,rb);
381 }
382 else if (totalNbElement()==0){
383 _resizeFromEmpty(new_size1,new_size2,rb);
384 }
385 else if (new_size2<m_md->dim2_size){
386 _resizeSameDim1ReduceDim2(new_size2,rb);
387 _resize(new_size1,rb);
388 }
389 else if (new_size2>m_md->dim2_size){
390 _resizeSameDim1IncreaseDim2(new_size2,rb);
391 _resize(new_size1,rb);
392 }
393 else
394 throw NotImplementedException("Array2::resize","already sized");
395 }

405 void _resizeSameDim1ReduceDim2(Int64 new_size2,InitBehaviour rb)
406 {
407 ARCCORE_ASSERT((new_size2<m_md->dim2_size),("Bad Size"));
408 Int64 n = m_md->dim1_size;
409 Int64 n2 = m_md->dim2_size;
410 for( Int64 i=0; i<n; ++i ){
411 for( Int64 j=0; j<new_size2; ++j )
412 m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
413 }
414 _resize2(n,new_size2,rb);
415 m_md->dim2_size = new_size2;
416 _arccoreCheckSharedNull();
417 }
418
419 void _resizeSameDim1IncreaseDim2(Int64 new_size2,InitBehaviour rb)
420 {
421 ARCCORE_ASSERT((new_size2>m_md->dim2_size),("Bad Size"));
422 Int64 n = m_md->dim1_size;
423 Int64 n2 = m_md->dim2_size;
424 _resize2(n,new_size2,rb);
425 // Recopie en partant de la fin pour éviter tout écrasement
426 for( Int64 i=(n-1); i>=0; --i ){
427 for( Int64 j=(n2-1); j>=0; --j )
428 m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
429 }
430 m_md->dim2_size = new_size2;
431 _arccoreCheckSharedNull();
432 if (rb==IB_InitWithDefault){
433 // Remet les valeurs par défaut pour les nouveaux éléments
434 for( Int64 i=0; i<n; ++i ){
435 for( Int64 j=n2; j<new_size2; ++j ){
436 m_ptr[(i*new_size2)+j] = DataType{};
437 }
438 }
439
440 }
441 }

443 void _resize2(Int64 d1,Int64 d2,InitBehaviour rb)
444 {
445 Int64 new_size = d1*d2;
446 // Si la nouvelle taille est nulle, il faut tout de meme faire une allocation
447 // pour stocker les valeurs dim1_size et dim2_size (sinon, elle seraient
448 // dans TrueImpl::shared_null
449 if (new_size==0)
450 this->_reserve(4);
451 if (rb==IB_InitWithDefault)
452 Base::_resize(new_size,DataType());
453 else if (rb==IB_NoInit)
454 Base::_resize(new_size);
455 else
456 throw NotSupportedException("Array2::_resize2","invalid value InitBehaviour");
457 }

484 void _resizeAndCopyView(Span2<const DataType> rhs)
485 {
486 Int64 total = rhs.totalNbElement();
487 if (total==0){
488 // Si la taille est nulle, il faut tout de meme faire une allocation
489 // pour stocker les valeurs dim1_size et dim2_size (sinon, elle seraient
490 // dans TrueImpl::shared_null)
491 this->_reserve(4);
492 }
493 Span<const DataType> aview(rhs.data(),total);
494 Base::_resizeAndCopyView(aview);
495 m_md->dim1_size = rhs.dim1Size();
496 m_md->dim2_size = rhs.dim2Size();
497 _arccoreCheckSharedNull();
498 }

661 ThatClassType* prev = nf->m_prev;
662 nf->m_prev = this;
663 m_prev = prev;
664 m_next = nf;
665 if (prev)
666 prev->m_next = this;
667 }
