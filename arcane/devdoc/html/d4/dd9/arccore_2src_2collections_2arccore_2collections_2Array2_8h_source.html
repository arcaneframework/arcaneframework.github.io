373 void _resize(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
374 {
375 if (new_size2==m_md->dim2_size){
376 _resize(new_size1,rb);
377 }
378 else if (totalNbElement()==0){
379 _resizeFromEmpty(new_size1,new_size2,rb);
380 }
381 else if (new_size2<m_md->dim2_size){
382 _resizeSameDim1ReduceDim2(new_size2,rb);
383 _resize(new_size1,rb);
384 }
385 else if (new_size2>m_md->dim2_size){
386 _resizeSameDim1IncreaseDim2(new_size2,rb);
387 _resize(new_size1,rb);
388 }
389 else
390 throw NotImplementedException("Array2::resize","already sized");
391 }

401 void _resizeSameDim1ReduceDim2(Int64 new_size2,InitBehaviour rb)
402 {
403 ARCCORE_ASSERT((new_size2<m_md->dim2_size),("Bad Size"));
404 Int64 n = m_md->dim1_size;
405 Int64 n2 = m_md->dim2_size;
406 for( Int64 i=0; i<n; ++i ){
407 for( Int64 j=0; j<new_size2; ++j )
408 m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
409 }
410 _resize2(n,new_size2,rb);
411 m_md->dim2_size = new_size2;
412 _arccoreCheckSharedNull();
413 }
414
415 void _resizeSameDim1IncreaseDim2(Int64 new_size2,InitBehaviour rb)
416 {
417 ARCCORE_ASSERT((new_size2>m_md->dim2_size),("Bad Size"));
418 Int64 n = m_md->dim1_size;
419 Int64 n2 = m_md->dim2_size;
420 _resize2(n,new_size2,rb);
421 // Recopie en partant de la fin pour éviter tout écrasement
422 for( Int64 i=(n-1); i>=0; --i ){
423 for( Int64 j=(n2-1); j>=0; --j )
424 m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
425 }
426 m_md->dim2_size = new_size2;
427 _arccoreCheckSharedNull();
428 if (rb==IB_InitWithDefault){
429 // Remet les valeurs par défaut pour les nouveaux éléments
430 for( Int64 i=0; i<n; ++i ){
431 for( Int64 j=n2; j<new_size2; ++j ){
432 m_ptr[(i*new_size2)+j] = DataType{};
433 }
434 }
435
436 }
437 }

439 void _resize2(Int64 d1,Int64 d2,InitBehaviour rb)
440 {
441 Int64 new_size = d1*d2;
442 // Si la nouvelle taille est nulle, il faut tout de meme faire une allocation
443 // pour stocker les valeurs dim1_size et dim2_size (sinon, elle seraient
444 // dans TrueImpl::shared_null
445 if (new_size==0)
446 this->_reserve(4);
447 if (rb==IB_InitWithDefault)
448 Base::_resize(new_size,DataType());
449 else if (rb==IB_NoInit)
450 Base::_resize(new_size);
451 else
452 throw NotSupportedException("Array2::_resize2","invalid value InitBehaviour");
453 }

480 void _resizeAndCopyView(Span2<const DataType> rhs)
481 {
482 Int64 total = rhs.totalNbElement();
483 if (total==0){
484 // Si la taille est nulle, il faut tout de meme faire une allocation
485 // pour stocker les valeurs dim1_size et dim2_size (sinon, elle seraient
486 // dans TrueImpl::shared_null)
487 this->_reserve(4);
488 }
489 Span<const DataType> aview(rhs.data(),total);
490 Base::_resizeAndCopyView(aview);
491 m_md->dim1_size = rhs.dim1Size();
492 m_md->dim2_size = rhs.dim2Size();
493 _arccoreCheckSharedNull();
494 }

656 ThatClassType* nf = const_cast<ThatClassType*>(new_ref);
657 ThatClassType* prev = nf->m_prev;
658 nf->m_prev = this;
659 m_prev = prev;
660 m_next = nf;
661 if (prev)
662 prev->m_next = this;
663 }

668 m_prev->m_next = m_next;
669 if (m_next)
670 m_next->m_prev = m_prev;

684 _addReference(&rhs);
685 ++rhs.m_md->nb_ref;
686 --m_md->nb_ref;
687 _checkFreeMemory();
688 this->_setMP2(rhs.m_ptr,rhs.m_md);
689 }
