298  void _resize(Int64 new_size,InitBehaviour rb)
299  {

308  void _resize(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
309  {
310  if (new_size2==m_md->dim2_size){
311  _resize(new_size1,rb);
312  }
313  else if (totalNbElement()==0){
314  _resizeFromEmpty(new_size1,new_size2,rb);
315  }
316  else if (new_size2<m_md->dim2_size){
317  _resizeSameDim1ReduceDim2(new_size2,rb);
318  _resize(new_size1,rb);
319  }
320  else if (new_size2>m_md->dim2_size){
321  _resizeSameDim1IncreaseDim2(new_size2,rb);
322  _resize(new_size1,rb);
323  }
324  else
325  throw NotImplementedException("Array2::resize","already sized");
326  }
327  void _resizeFromEmpty(Int64 new_size1,Int64 new_size2,InitBehaviour rb)
328  {
329  _resize2(new_size1,new_size2,rb);
330  m_md->dim1_size = new_size1;
331  m_md->dim2_size = new_size2;
332  _arccoreCheckSharedNull();
333  }
334  void _resizeSameDim1ReduceDim2(Int64 new_size2,InitBehaviour rb)
335  {
336  ARCCORE_ASSERT((new_size2<m_md->dim2_size),("Bad Size"));
337  Int64 n = m_md->dim1_size;
338  Int64 n2 = m_md->dim2_size;
339  for( Int64 i=0; i<n; ++i ){
340  for( Int64 j=0; j<new_size2; ++j )
341  m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
342  }
343  _resize2(n,new_size2,rb);
344  m_md->dim2_size = new_size2;
345  _arccoreCheckSharedNull();
346  }
347  void _resizeSameDim1IncreaseDim2(Int64 new_size2,InitBehaviour rb)
348  {
349  ARCCORE_ASSERT((new_size2>m_md->dim2_size),("Bad Size"));
350  Int64 n = m_md->dim1_size;
351  Int64 n2 = m_md->dim2_size;
352  _resize2(n,new_size2,rb);
353  // Recopie en partant de la fin pour éviter tout écrasement
354  for( Int64 i=(n-1); i>=0; --i ){
355  for( Int64 j=(n2-1); j>=0; --j )
356  m_ptr[(i*new_size2)+j] = m_ptr[(i*n2)+j];
357  }
358  m_md->dim2_size = new_size2;
359  _arccoreCheckSharedNull();
360  }
361  void _resize2(Int64 d1,Int64 d2,InitBehaviour rb)
362  {
363  Int64 new_size = d1*d2;
364  // Si la nouvelle taille est nulle, il faut tout de meme faire une allocation
365  // pour stocker les valeurs dim1_size et dim2_size (sinon, elle seraient
366  // dans TrueImpl::shared_null
367  if (new_size==0)
368  this->_reserve(4);
369  if (rb==IB_InitWithDefault)
370  Base::_resize(new_size,DataType());
371  else if (rb==IB_NoInit)
372  Base::_resize(new_size);
373  else
374  throw NotSupportedException("Array2::_resize2","invalid value InitBehaviour");
375  }

531  ThatClassType* prev = nf->m_prev;
532  nf->m_prev = this;
533  m_prev = prev;
534  m_next = nf;
535  if (prev)
536  prev->m_next = this;
537  }

542  m_prev->m_next = m_next;
543  if (m_next)

561  _checkFreeMemory();
562  this->_setMP2(rhs.m_ptr,rhs.m_md);
563  }
564  }
