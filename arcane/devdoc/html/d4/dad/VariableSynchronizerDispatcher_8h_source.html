184  Span<std::byte> globalReceiveBuffer() override { return _toBytes(m_buffer->ghostBuffer()); }

190  void copyReceive(Int32 index) override { m_buffer->copyReceive(index); }
191  Int64 totalSendSize() const { return m_buffer->totalShareSize(); }
192  Int64 totalReceiveSize() const { return m_buffer->totalGhostSize(); }
193  private:
194  SyncBuffer* m_buffer;
195  Span<std::byte> _toBytes(ArrayView<SimpleType> view)
196  {
197  Int64 s = static_cast<Int64>(view.size()) * sizeof(SimpleType);
198  return { reinterpret_cast<std::byte*>(view.data()), s };
199  }
200  };
201  public:

214  ArrayView<SimpleType> ghostBuffer() { return m_ghost_buffer; }
215  ArrayView<SimpleType> shareBuffer() { return m_share_buffer; }
216  ConstArrayView<SimpleType> ghostBuffer() const { return m_ghost_buffer; }
217  ConstArrayView<SimpleType> shareBuffer() const { return m_share_buffer; }
218  void setDataView(ArrayView<SimpleType> v) { m_data_view = v; }
219  ArrayView<SimpleType> dataView() { return m_data_view; }
220  void copyReceive(Integer index);
221  void copySend(Integer index);
222  Int64 totalGhostSize() const { return asBytes(m_ghost_buffer.constSpan()).size(); }
223  Int64 totalShareSize() const { return asBytes(m_share_buffer.constSpan()).size(); }
