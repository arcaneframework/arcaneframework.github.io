86bool MeshMaterialSynchronizer::
87synchronizeMaterialsInCells()
88{
89 /*
90 L'algorithme utilisé est le suivant:
91
92 On utilise une variable aux mailles qui utilise un bit pour chaque
93 matériau pour indiquer sa présence: si ce bit est positionné, le matériau
94 est présent, sinon il est absent. La variable utilisée est donc de type
95 ArrayByte aux mailles. Les méthodes _hasBit() et _setBit() permettent
96 de positionner le bit d'un matériau donné.
97
98 1. Le sous-domaine remplit cette variables pour ces mailles.
99 2. La variable est synchronisée.
100 3. Le sous-domaine compare pour chacune de ses mailles fantômes
101 ce tableau de présence des matériaux et ajoute/supprime les matériaux en fonction
102 de ce tableau.
103 */
104 IMesh* mesh = m_material_mng->mesh();
105 if (!mesh->parallelMng()->isParallel())
106 return false;
107
108 ConstArrayView<IMeshMaterial*> materials = m_material_mng->materials();
109 Integer nb_mat = materials.size();
110 VariableCellArrayByte mat_presence(VariableBuildInfo(mesh,"ArcaneMaterialSyncPresence"));
111 Integer dim2_size = nb_mat / 8;
112 if ((nb_mat % 8)!=0)
113 ++dim2_size;
114 mat_presence.resize(dim2_size);
115 info(4) << "Resize presence variable nb_mat=" << nb_mat << " dim2=" << dim2_size;
116 CellToAllEnvCellConverter cell_converter = m_material_mng->cellToAllEnvCellConverter();
117 ENUMERATE_CELL(icell,mesh->ownCells()){
118 ByteArrayView presence = mat_presence[icell];
119 presence.fill(0);
120 AllEnvCell all_env_cell = cell_converter[*icell];
121 _fillPresence(all_env_cell,presence);
122 }
123
124 bool has_changed = false;
125
126 mat_presence.synchronize();
127 {
128 ByteUniqueArray before_presence(dim2_size);
129 UniqueArray< UniqueArray<Int32> > to_add(nb_mat);
130 UniqueArray< UniqueArray<Int32> > to_remove(nb_mat);
131 ENUMERATE_CELL(icell,mesh->allCells()){
132 Cell cell = *icell;
133 // Ne traite que les mailles fantomes.
134 if (cell.isOwn())
135 continue;
136 Int32 cell_lid = cell.localId();
137 AllEnvCell all_env_cell = cell_converter[cell];
138 before_presence.fill(0);
139 _fillPresence(all_env_cell,before_presence);
140 ByteConstArrayView after_presence = mat_presence[cell];
141 // Ajoute/Supprime cette maille des matériaux si besoin.
142 for( Integer imat=0; imat<nb_mat; ++imat ){
143 bool has_before = _hasBit(before_presence,imat);
144 bool has_after = _hasBit(after_presence,imat);
145 if (has_before && !has_after){
146 to_remove[imat].add(cell_lid);
147 }
148 else if (has_after && !has_before)
149 to_add[imat].add(cell_lid);
150 }
151 }
152
153 MeshMaterialModifier modifier(m_material_mng);
154 for( Integer i=0; i<nb_mat; ++i ){
155 if (!to_add[i].empty()){
156 modifier.addCells(materials[i],to_add[i]);
157 has_changed = true;
158 }
159 if (!to_remove[i].empty()){
160 modifier.removeCells(materials[i],to_remove[i]);
161 has_changed = true;
162 }
163 }
164 }
165
166 return has_changed;
167}

176void MeshMaterialSynchronizer::
177checkMaterialsInCells(Integer max_print)
178{
179 /*
180 Pour cela, on utilise une variable aux mailles et on applique
181 l'algorithme suivant pour chaque matériau:
182 - le sous-domaine propriétaire remplit cette variable
183 avec l'indice du matériau
184 - la variable est synchronisée.
185 - chaque sous-domaine vérifie ensuite pour chaque maille
186 que si la variable a pour valeur l'indice du matériau, alors
187 ce matériau est présent.
188 */
189
190 IMesh* mesh = m_material_mng->mesh();
191 if (!mesh->parallelMng()->isParallel())
192 return;
193 m_material_mng->checkValid();
194 info(4) << "CheckMaterialsInCells";
195 VariableCellInt32 indexes(VariableBuildInfo(mesh,"ArcaneMaterialPresenceIndexes"));
196 _checkComponents(indexes,m_material_mng->materialsAsComponents(),max_print);
197 _checkComponents(indexes,m_material_mng->environmentsAsComponents(),max_print);
198}

203void MeshMaterialSynchronizer::
204_checkComponents(VariableCellInt32& indexes,
205 ConstArrayView<IMeshComponent*> components,
206 Integer max_print)
207{
208 IMesh* mesh = m_material_mng->mesh();
209 Integer nb_component = components.size();
210 Integer nb_error = 0;
211
212 info() << "Checking components nb=" << nb_component;
213
214 CellToAllEnvCellConverter all_env_cell_converter(m_material_mng);
215
216 for( Integer i=0; i<nb_component; ++i ){
217 indexes.fill(-1);
218 IMeshComponent* c = components[i];
219 ENUMERATE_COMPONENTCELL(iccell,c){
220 ComponentCell cc = *iccell;
221 indexes[cc.globalCell()] = i;
222 }
223
224 indexes.synchronize();
225
226 ENUMERATE_ALLENVCELL(iallenvcell,m_material_mng,mesh->allCells()){
227 AllEnvCell all_env_cell = *iallenvcell;
228 Cell cell = all_env_cell.globalCell();
229 bool has_sync_mat = (indexes[cell]==i);
230 ComponentCell cc = c->findComponentCell(all_env_cell);
231 bool has_component = !cc.null();
232 if (has_sync_mat!=has_component){
233 ++nb_error;
234 if (max_print<0 || nb_error<max_print)
235 error() << "Bad component synchronisation for i=" << i
236 << " name=" << c->name()
237 << " cell_uid=" << cell.uniqueId()
238 << " sync_mat=" << has_sync_mat
239 << " has_component=" << has_component;
240 }
241 }
242 }
243 if (nb_error!=0)
244 ARCANE_FATAL("Bad synchronisation");
245}
