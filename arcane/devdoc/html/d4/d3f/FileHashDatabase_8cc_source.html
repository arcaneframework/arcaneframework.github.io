114 void readValues(const HashDatabaseReadArgs& args) override
115 {
116 const String& hash_value = args.hashValueAsString();
117 Span<std::byte> bytes = args.values();
118 DirFileInfo dirfile_info = _getDirFileInfo(hash_value);
119 {
120 String full_filename = dirfile_info.full_filename;
121 //info() << "READ_VALUE hash_value=" << hash_value << " name=" << args.key() << " file=" << full_filename;
122 std::ifstream ifile(full_filename.localstr(), ios::binary);
123 if (!ifile.good())
124 ARCANE_FATAL("Can not open file '{0}'", full_filename);
125 binaryRead(ifile, bytes);
126 ++m_nb_read;
127 if (!ifile.good())
128 ARCANE_FATAL("Can not read file '{0}'", full_filename);
129 }
130 }
131
132 private:
133
134 DirFileInfo _getDirFileInfo(const String& hash_value)
135 {
136 char name1 = hash_value.bytes()[0];
137 char name2 = hash_value.bytes()[1];
138 char name3 = hash_value.bytes()[2];
139 StringBuilder base_name_builder = m_directory;
140 base_name_builder += '/';
141 base_name_builder += name1;
142 base_name_builder += '/';
143 base_name_builder += name2;
144 base_name_builder += name3;
145 String base_name = base_name_builder.toString();
146 String full_filename = String::format("{0}/{1}", base_name, hash_value);
147 return DirFileInfo{ base_name, hash_value, full_filename };
148 }
149

154 Int64 m_nb_write = 0;
155 Int64 m_nb_read = 0;
