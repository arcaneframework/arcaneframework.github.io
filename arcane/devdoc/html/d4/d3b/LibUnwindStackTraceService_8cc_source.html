119LibUnwindStackTraceService::ProcInfo LibUnwindStackTraceService::
120_getFuncInfo(unw_word_t ip,unw_cursor_t* cursor)
121{
122 ProcInfoMap::const_iterator v = m_proc_name_map.find(ip);
123 if (v!=m_proc_name_map.end())
124 return v->second;
125
126 unw_word_t offset;
127 char func_name_buf[10000];
128 char demangled_func_name_buf[10000];
129 unw_get_proc_name(cursor,func_name_buf,10000,&offset);
130 int dstatus = 0;
131 size_t len = 10000;
132 char* buf = abi::__cxa_demangle (func_name_buf,demangled_func_name_buf,&len,&dstatus);
133 ProcInfo pi;
134 pi.m_base_ip = offset;
135 {
136 Dl_info dl_info;
137 void* addr = (void*)ip;
138 int r2 = dladdr(addr,&dl_info);
139 if (r2!=0){
140 const char* dli_fname = dl_info.dli_fname;
141 // Adresse de base de chargement du fichier.
142 void* dli_fbase = dl_info.dli_fbase;
143 pi.m_file_name = dli_fname;
144 pi.m_file_loaded_address = (unw_word_t)dli_fbase;
145 }
146 }
147
148 if (buf)
149 pi.m_name = std::string_view(buf);
150 else
151 pi.m_name = std::string_view(func_name_buf);
152 m_proc_name_map.insert(ProcInfoMap::value_type(ip,pi));
153
154 return pi;
155}

163LibUnwindStackTraceService::ProcInfo LibUnwindStackTraceService::
164_getFuncInfo(void* addr)
165{
166 ProcInfoMap::const_iterator v = m_proc_name_map.find((unw_word_t)addr);
167 if (v!=m_proc_name_map.end()){
168 return v->second;
169 }
170 const size_t buf_size = 10000;
171 char demangled_func_name_buf[buf_size];
172 Dl_info dl_info;
173 int r = dladdr(addr,&dl_info);
174 if (r==0){
175 // Erreur dans dladdr.
176 std::cout << "ERROR in dladdr\n";
177 return ProcInfo("Unknown");
178 }
179 const char* dli_sname = dl_info.dli_sname;
180 if (!dli_sname)
181 dli_sname = "";
182 int dstatus = 0;
183 size_t len = buf_size;
184 char* buf = abi::__cxa_demangle (dli_sname,demangled_func_name_buf,&len,&dstatus);
185 //char* buf = (char*)dli_sname;
186 ProcInfo pi;
187 if (buf)
188 pi.m_name = std::string_view(buf);
189 else
190 pi.m_name = std::string_view(dli_sname);
191 m_proc_name_map.insert(ProcInfoMap::value_type((unw_word_t)addr,pi));
192 return pi;
193}

198String LibUnwindStackTraceService::
199_getGDBStack()
200{
201 void *array [256];
202 char **names;
203 const size_t cmd_size = 4096;
204 char cmd [cmd_size+1];
205 int i, size;
206
207 fprintf (stderr, "\nNative stacktrace:\n\n");
208
209 size = backtrace (array, 256);
210 names = backtrace_symbols (array, size);
211 for (i =0; i < size; ++i) {
212 fprintf (stderr, "\t%s\n", names [i]);
213 }
214
215 fflush (stderr);
216
217 //sprintf (cmd, "gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt'", (long)getpid ());
218 //sprintf (cmd, "gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt' --batch", (long)getpid ());
219 char filename[4096];
220 long pid = (long)getpid();
221 sprintf(filename,"errlog.%ld",pid);
222 snprintf (cmd,cmd_size,"gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt full' --batch",pid);
223 int ret_value = system(cmd);
224
225 long unsigned int file_length = 0;
226 if (ret_value==0)
227 file_length = platform::getFileLength(filename);
228 if (file_length==0)
229 return String();
230
231 std::ifstream ifile;
232 ifile.open(filename,std::ios::binary);
233 ByteUniqueArray bytes(arcaneCheckArraySize(file_length));
234 ifile.read((char*)bytes.data(),file_length);
235 return String(bytes);
236}

242StackTrace LibUnwindStackTraceService::
243stackTrace(int first_function)
244{
245 unw_cursor_t cursor;
246 unw_context_t uc;
247
248 String last_str;
249 if (m_want_gdb_info)
250 last_str = _getGDBStack();
251 FixedStackFrameArray backtrace_stack_frames = _backtraceStackFrame(first_function);
252 if (m_use_backtrace)
253 return _backtraceStackTrace(backtrace_stack_frames);
254
255 const size_t hexa_buf_size = 100;
256 char hexa[hexa_buf_size+1];
257
258 unw_getcontext(&uc);
259 unw_init_local(&cursor, &uc);
260 int current_func = 0;
261 StringBuilder message;
262
263 FixedStackFrameArray stack_frames;
264 while (unw_step(&cursor) > 0) {
265 unw_word_t ip;
266 unw_get_reg(&cursor, UNW_REG_IP, &ip);
267 if (current_func>=first_function){
268 ProcInfo pi = _getFuncInfo(ip,&cursor);
269 String func_name = pi.m_name;
270 message += " ";
271 snprintf(hexa,hexa_buf_size,"%14llx",(long long)ip);
272 message += hexa;
273 message += " ";
274 message += func_name;
275 message += "\n";
276
277 stack_frames.addFrame(StackFrame((intptr_t)ip));
278 }
279 ++current_func;
280 }
281 ISymbolizerService* ss = platform::getSymbolizerService();
282
283 if (ss){
284 // Il faut mieux lire la pile d'appel de _backtrace() car
285 // la libunwind retourne l'adresse de retour pour chaque fonction
286 // ce qui provoque un décalage dans les infos des lignes du code
287 // source (on pointe vers la ligne après celle qu'on est en train d'exécuter).
288 last_str = ss->stackTrace(backtrace_stack_frames.view());
289 }
290 else{
291 message += "\nFileAndOffsetStack:{{\n";
292 message += _generateFileAndOffset(backtrace_stack_frames);
293 message += "}}\n";
294 }
295 message += last_str;
296 return StackTrace(stack_frames,message);
297}

306StackTrace LibUnwindStackTraceService::
307stackTraceFunction(int function_index)
308{
309 unw_cursor_t cursor;
310 unw_context_t uc;
311 unw_word_t ip;
312 //unw_word_t offset;
313
314 String last_str;
315 if (m_want_gdb_info)
316 last_str = _getGDBStack();
317
318 unw_getcontext(&uc);
319 unw_init_local(&cursor, &uc);
320 int current_func = 0;
321 StringBuilder message;
322
323 while (unw_step(&cursor) > 0) {
324 unw_get_reg(&cursor, UNW_REG_IP, &ip);
325 if (current_func==function_index){
326 ProcInfo pi = _getFuncInfo(ip,&cursor);
327 String func_name = pi.m_name;
328 message += func_name;
329 break;
330 }
331 ++current_func;
332 }
333 return StackTrace(message.toString()+last_str);
334}

357StackTrace LibUnwindStackTraceService::
358_backtraceStackTrace(const FixedStackFrameArray& stack_frames)
359{
360 const size_t buf_size = 100;
361 char hexa[buf_size+1];
362
363 StringBuilder message;
364 ConstArrayView<StackFrame> frames_view = stack_frames.view();
365 for( StackFrame f : frames_view ){
366 intptr_t ip = f.address();
367 ProcInfo pinfo = _getFuncInfo((void*)ip);
368 String func_name = pinfo.name();
369 message += " ";
370 snprintf(hexa,buf_size,"%10llx",(long long)ip);
371 message += hexa;
372 message += " ";
373 message += func_name;
374 message += "\n";
375 }
376 return StackTrace(stack_frames,message);

391String LibUnwindStackTraceService::
392_generateFileAndOffset(const FixedStackFrameArray& stack_frames)
393{
394 const size_t buf_size = 100;
395 char hexa[buf_size+1];
396
397 StringBuilder message;
398 ConstArrayView<StackFrame> frames_view = stack_frames.view();
399 for( StackFrame f : frames_view ){
400 intptr_t ip = f.address();
401 ProcInfo pinfo = _getFuncInfo((void*)ip);
402 message += (pinfo.fileName() ? pinfo.fileName() : "()");
403 message += " ";
404 intptr_t file_base_address = pinfo.m_file_loaded_address;
405 // Sous Linux (CentOS 6 et 7), l'adresse 0x400000 correspond à celle
406 // de chargement de l'exécutable mais si on retranche cette adresse de celle
407 // de la fonction alors le symboliser ne fonctionne pas (que ce soit
408 // llvm-symbolize ou addr2line)
409 if (file_base_address==0x400000)
410 file_base_address = 0;
411 intptr_t offset_ip = (ip - file_base_address);
412 snprintf(hexa,buf_size,"%llx",(long long)offset_ip);
413 message += "0x";
414 message += hexa;
415 message += "\n";
416 }
417 return message;
418}

465String LLVMSymbolizerService::
466stackTrace(ConstArrayView<StackFrame> frames)
467{
468 _checkValid();
469 if (!m_is_valid)
470 return String();
471 std::stringstream ostr;
472 // NOTE: le code ci dessous est similaire à
473 // LibUnwindStackTraceService::_generateFileAndOffset(). Il faudrait
474 // fusionner les deux.
475 for( Integer i=0, n=frames.size(); i<n; ++i ){
476 Dl_info dl_info;
477 intptr_t addr = frames[i].address();
478 int r2 = dladdr((void*)addr,&dl_info);
479 const char* dli_fname = nullptr;
480 intptr_t base_address = 0;
481 if (r2!=0){
482 dli_fname = dl_info.dli_fname;
483 // Adresse de base de chargement du fichier.
484 void* dli_fbase = dl_info.dli_fbase;
485 intptr_t true_base = (intptr_t)dli_fbase;
486 // Sous Linux (CentOS 6 et 7), l'adresse 0x400000 correspond à celle
487 // de chargement de l'exécutable mais si on retranche cette adresse de celle
488 // de la fonction alors le symboliser ne fonctionne pas (que ce soit
489 // llvm-symbolize ou addr2line)
490 if (true_base==0x400000)
491 true_base = 0;
492 base_address = addr - true_base;
493 }
494 // TODO: écrire base address en hexa pour pouvoir le relire avec addr2line
495 ostr << (dli_fname ? dli_fname : "??") << " " << base_address << '\n';
496 }
497
498 std::string input_str(ostr.str());
499 String output_str;
500 ProcessExecArgs args;
501 args.setCommand(m_llvm_symbolizer_path);
502 Integer input_size = arcaneCheckArraySize(input_str.length());
503 ByteConstArrayView input_bytes(input_size,(const Byte*)input_str.c_str());
504 args.setInputBytes(input_bytes);
505 args.addArguments("--demangle");
506 args.addArguments("--pretty-print");
507 //args.addArguments("--print-source-context-lines=5");
508 Process::execute(args);
509 return String(args.outputBytes());
510}
