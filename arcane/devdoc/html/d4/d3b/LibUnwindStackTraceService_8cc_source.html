122LibUnwindStackTraceService::ProcInfo LibUnwindStackTraceService::
123_getFuncInfo(unw_word_t ip,unw_cursor_t* cursor)
124{
125 {
126 std::lock_guard<std::mutex> lk(m_proc_name_map_mutex);
127 auto v = m_proc_name_map.find(ip);
128 if (v!=m_proc_name_map.end())
129 return v->second;
130 }
131
132 unw_word_t offset;
133 char func_name_buf[10000];
134 char demangled_func_name_buf[10000];
135 unw_get_proc_name(cursor,func_name_buf,10000,&offset);
136 int dstatus = 0;
137 size_t len = 10000;
138 char* buf = abi::__cxa_demangle (func_name_buf,demangled_func_name_buf,&len,&dstatus);
139 ProcInfo pi;
140 pi.m_base_ip = offset;
141 {
142 Dl_info dl_info;
143 void* addr = (void*)ip;
144 int r2 = dladdr(addr,&dl_info);
145 if (r2!=0){
146 const char* dli_fname = dl_info.dli_fname;
147 // Adresse de base de chargement du fichier.
148 void* dli_fbase = dl_info.dli_fbase;
149 pi.m_file_name = dli_fname;
150 pi.m_file_loaded_address = (unw_word_t)dli_fbase;
151 }
152 }
153
154 if (buf)
155 pi.m_name = std::string_view(buf);
156 else
157 pi.m_name = std::string_view(func_name_buf);
158 {
159 std::lock_guard<std::mutex> lk(m_proc_name_map_mutex);
160 m_proc_name_map.insert(ProcInfoMap::value_type(ip,pi));
161 }
162 return pi;
163}

171LibUnwindStackTraceService::ProcInfo LibUnwindStackTraceService::
172_getFuncInfo(void* addr)
173{
174 {
175 std::lock_guard<std::mutex> lk(m_proc_name_map_mutex);
176 auto v = m_proc_name_map.find((unw_word_t)addr);
177 if (v!=m_proc_name_map.end()){
178 return v->second;
179 }
180 }
181 const size_t buf_size = 10000;
182 char demangled_func_name_buf[buf_size];
183 Dl_info dl_info;
184 int r = dladdr(addr,&dl_info);
185 if (r==0){
186 // Erreur dans dladdr.
187 std::cout << "ERROR in dladdr\n";
188 return ProcInfo("Unknown");
189 }
190 const char* dli_sname = dl_info.dli_sname;
191 if (!dli_sname)
192 dli_sname = "";
193 int dstatus = 0;
194 size_t len = buf_size;
195 char* buf = abi::__cxa_demangle (dli_sname,demangled_func_name_buf,&len,&dstatus);
196 //char* buf = (char*)dli_sname;
197 ProcInfo pi;
198 if (buf)
199 pi.m_name = std::string_view(buf);
200 else
201 pi.m_name = std::string_view(dli_sname);
202 {
203 std::lock_guard<std::mutex> lk(m_proc_name_map_mutex);
204 m_proc_name_map.insert(ProcInfoMap::value_type((unw_word_t)addr,pi));
205 }
206 return pi;
207}

212String LibUnwindStackTraceService::
213_getGDBStack()
214{
215 void *array [256];
216 char **names;
217 const size_t cmd_size = 4096;
218 char cmd [cmd_size+1];
219 int i, size;
220
221 fprintf (stderr, "\nNative stacktrace:\n\n");
222
223 size = backtrace (array, 256);
224 names = backtrace_symbols (array, size);
225 for (i =0; i < size; ++i) {
226 fprintf (stderr, "\t%s\n", names [i]);
227 }
228
229 fflush (stderr);
230
231 //sprintf (cmd, "gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt'", (long)getpid ());
232 //sprintf (cmd, "gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt' --batch", (long)getpid ());
233 char filename[4096];
234 long pid = (long)getpid();
235 sprintf(filename,"errlog.%ld",pid);
236 snprintf (cmd,cmd_size,"gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt full' --batch",pid);
237 int ret_value = system(cmd);
238
239 long unsigned int file_length = 0;
240 if (ret_value==0)
241 file_length = platform::getFileLength(filename);
242 if (file_length==0)
243 return String();
244
245 std::ifstream ifile;
246 ifile.open(filename,std::ios::binary);
247 ByteUniqueArray bytes(arcaneCheckArraySize(file_length));
248 ifile.read((char*)bytes.data(),file_length);
249 return String(bytes);
250}

256StackTrace LibUnwindStackTraceService::
257stackTrace(int first_function)
258{
259 unw_cursor_t cursor;
260 unw_context_t uc;
261
262 String last_str;
263 if (m_want_gdb_info)
264 last_str = _getGDBStack();
265 FixedStackFrameArray backtrace_stack_frames = _backtraceStackFrame(first_function);
266 if (m_use_backtrace)
267 return _backtraceStackTrace(backtrace_stack_frames);
268
269 const size_t hexa_buf_size = 100;
270 char hexa[hexa_buf_size+1];
271
272 unw_getcontext(&uc);
273 unw_init_local(&cursor, &uc);
274 int current_func = 0;
275 StringBuilder message;
276
277 FixedStackFrameArray stack_frames;
278 while (unw_step(&cursor) > 0) {
279 unw_word_t ip;
280 unw_get_reg(&cursor, UNW_REG_IP, &ip);
281 if (current_func>=first_function){
282 ProcInfo pi = _getFuncInfo(ip,&cursor);
283 String func_name = pi.m_name;
284 message += " ";
285 snprintf(hexa,hexa_buf_size,"%14llx",(long long)ip);
286 message += hexa;
287 message += " ";
288 message += func_name;
289 message += "\n";
290
291 stack_frames.addFrame(StackFrame((intptr_t)ip));
292 }
293 ++current_func;
294 }
295 ISymbolizerService* ss = platform::getSymbolizerService();
296
297 if (ss){
298 // Il faut mieux lire la pile d'appel de _backtrace() car
299 // la libunwind retourne l'adresse de retour pour chaque fonction
300 // ce qui provoque un décalage dans les infos des lignes du code
301 // source (on pointe vers la ligne après celle qu'on est en train d'exécuter).
302 last_str = ss->stackTrace(backtrace_stack_frames.view());
303 }
304 else{
305 message += "\nFileAndOffsetStack:{{\n";
306 message += _generateFileAndOffset(backtrace_stack_frames);
307 message += "}}\n";
308 }
309 message += last_str;
310 return StackTrace(stack_frames,message);
311}

320StackTrace LibUnwindStackTraceService::
321stackTraceFunction(int function_index)
322{
323 unw_cursor_t cursor;
324 unw_context_t uc;
325 unw_word_t ip;
326 //unw_word_t offset;
327
328 String last_str;
329 if (m_want_gdb_info)
330 last_str = _getGDBStack();
331
332 unw_getcontext(&uc);
333 unw_init_local(&cursor, &uc);
334 int current_func = 0;
335 StringBuilder message;
336
337 while (unw_step(&cursor) > 0) {
338 unw_get_reg(&cursor, UNW_REG_IP, &ip);
339 if (current_func==function_index){
340 ProcInfo pi = _getFuncInfo(ip,&cursor);
341 String func_name = pi.m_name;
342 message += func_name;
343 break;
344 }
345 ++current_func;
346 }
347 return StackTrace(message.toString()+last_str);
348}

405String LibUnwindStackTraceService::
406_generateFileAndOffset(const FixedStackFrameArray& stack_frames)
407{
408 const size_t buf_size = 100;
409 char hexa[buf_size+1];
410
411 StringBuilder message;
412 ConstArrayView<StackFrame> frames_view = stack_frames.view();
413 for( StackFrame f : frames_view ){
414 intptr_t ip = f.address();
415 ProcInfo pinfo = _getFuncInfo((void*)ip);
416 message += (pinfo.fileName() ? pinfo.fileName() : "()");
417 message += " ";
418 intptr_t file_base_address = pinfo.m_file_loaded_address;
419 // Sous Linux (CentOS 6 et 7), l'adresse 0x400000 correspond à celle
420 // de chargement de l'exécutable mais si on retranche cette adresse de celle
421 // de la fonction alors le symboliser ne fonctionne pas (que ce soit
422 // llvm-symbolize ou addr2line)
423 if (file_base_address==0x400000)
424 file_base_address = 0;
425 intptr_t offset_ip = (ip - file_base_address);
426 snprintf(hexa,buf_size,"%llx",(long long)offset_ip);
427 message += "0x";
428 message += hexa;
429 message += "\n";
430 }
431 return message;
432}

479String LLVMSymbolizerService::
480stackTrace(ConstArrayView<StackFrame> frames)
481{
482 _checkValid();
483 if (!m_is_valid)
484 return String();
485 std::stringstream ostr;
486 // NOTE: le code ci dessous est similaire à
487 // LibUnwindStackTraceService::_generateFileAndOffset(). Il faudrait
488 // fusionner les deux.
489 for( Integer i=0, n=frames.size(); i<n; ++i ){
490 Dl_info dl_info;
491 intptr_t addr = frames[i].address();
492 int r2 = dladdr((void*)addr,&dl_info);
493 const char* dli_fname = nullptr;
494 intptr_t base_address = 0;
495 if (r2!=0){
496 dli_fname = dl_info.dli_fname;
497 // Adresse de base de chargement du fichier.
498 void* dli_fbase = dl_info.dli_fbase;
499 intptr_t true_base = (intptr_t)dli_fbase;
500 // Sous Linux (CentOS 6 et 7), l'adresse 0x400000 correspond à celle
501 // de chargement de l'exécutable mais si on retranche cette adresse de celle
502 // de la fonction alors le symboliser ne fonctionne pas (que ce soit
503 // llvm-symbolize ou addr2line)
504 if (true_base==0x400000)
505 true_base = 0;
506 base_address = addr - true_base;
507 }
508 // TODO: écrire base address en hexa pour pouvoir le relire avec addr2line
509 ostr << (dli_fname ? dli_fname : "??") << " " << base_address << '\n';
510 }
511
512 std::string input_str(ostr.str());
513 String output_str;
514 ProcessExecArgs args;
515 args.setCommand(m_llvm_symbolizer_path);
516 Integer input_size = arcaneCheckArraySize(input_str.length());
517 ByteConstArrayView input_bytes(input_size,(const Byte*)input_str.c_str());
518 args.setInputBytes(input_bytes);
519 args.addArguments("--demangle");
520 args.addArguments("--pretty-print");
521 //args.addArguments("--print-source-context-lines=5");
522 Process::execute(args);
523 return String(args.outputBytes());
524}
