622ItemGroupImpl* ItemGroupImpl::
623createSubGroup(const String& suffix, IItemFamily* family, ItemGroupComputeFunctor * functor)
624{
625 String sub_name = name() + "_" + suffix;
626 auto finder = m_p->m_sub_groups.find(sub_name);
627 if (finder != m_p->m_sub_groups.end()) {
628 ARCANE_FATAL("Cannot create already existing sub-group ({0}) in group ({1})",
629 suffix, name());
630 }
631 ItemGroup ig = family->createGroup(sub_name,ItemGroup(this));
632 ItemGroupImpl* ii = ig.internal();
633 ii->setComputeFunctor(functor);
634 functor->setGroup(ii);
635 // Observer par défaut : le sous groupe n'est pas intéressé par les infos détaillées de transition
636 attachObserver(ii,newItemGroupObserverT(ii,
637 &ItemGroupImpl::_executeInvalidate));
638 m_p->m_sub_groups[sub_name] = ii;
639 ii->invalidate(false);
640 return ii;
641}

768
769void ItemGroupImpl::
770addItems(Int32ConstArrayView items_local_id,bool check_if_present)
771{
772 ARCANE_ASSERT(( (!m_p->m_need_recompute && !isAllItems()) || (m_p->m_transaction_mode && isAllItems()) ),
773 ("Operation on invalid group"));
774 if (m_p->m_compute_functor && !m_p->m_transaction_mode)
775 ARCANE_FATAL("Cannot add items on computed group ({0})", name());
776 IMesh* amesh = mesh();
777 if (!amesh)
778 throw ArgumentException(A_FUNCINFO,"null group");
779 ITraceMng* trace = amesh->traceMng();
780
781 Integer nb_item_to_add = items_local_id.size();
782 if (nb_item_to_add==0)
783 return;
784
785 Int32Array& items_lid = m_p->mutableItemsLocalId();
786 const Integer current_size = items_lid.size();
787 Integer nb_added = 0;
788
789 if(isAllItems()) {
790 // Ajoute les nouveaux items à la fin
791 Integer nb_items_id = current_size;
792 m_p->m_items_index_in_all_group.resize(m_p->maxLocalId());
793 for( Integer i=0, is=nb_item_to_add; i<is; ++i ){
794 Int32 local_id = items_local_id[i];
795 items_lid.add(local_id);
796 m_p->m_items_index_in_all_group[local_id] = nb_items_id+i;
797 }
798 nb_added = nb_item_to_add;
799 }
800 else if (check_if_present) {
801 // Vérifie que les entités à ajouter ne sont pas déjà présentes
802 UniqueArray<bool> presence_checks(m_p->maxLocalId());
803 presence_checks.fill(false);
804 for( Integer i=0, is=items_lid.size(); i<is; ++i ){
805 presence_checks[items_lid[i]] = true;
806 }
807
808 for( Integer i=0; i<nb_item_to_add; ++i ) {
809 const Integer lid = items_local_id[i];
810 if (!presence_checks[lid]){
811 items_lid.add(lid);
812 // Met à \a true comme cela si l'entité est présente plusieurs fois
813 // dans \a items_local_id cela fonctionne quand même.
814 presence_checks[lid] = true;
815 ++nb_added;
816 }
817 }
818 }
819 else {
820 nb_added = nb_item_to_add;
821 items_lid.resize(current_size+nb_added);
822 Int32ArrayView ptr = items_lid;
823 memcpy(&ptr[current_size],items_local_id.data(),sizeof(Int32)*nb_added);
824 }
825
826 if (arcaneIsCheck()){
827 trace->info(5) << "ItemGroupImpl::addItems() group <" << name() << "> "
828 << " checkpresent=" << check_if_present
829 << " nb_current=" << current_size
830 << " want_to_add=" << nb_item_to_add
831 << " effective_added=" << nb_added;
832 checkValid();
833 }
834
835 if (nb_added!=0) {
836 m_p->updateTimestamp();

845void ItemGroupImpl::
846removeItems(Int32ConstArrayView items_local_id,bool check_if_present)
847{
848 // TODO: check_if_present n'est pas pris en compte
849 ARCANE_UNUSED(check_if_present);
850
851 ARCANE_ASSERT(( (!m_p->m_need_recompute && !isAllItems()) || (m_p->m_transaction_mode && isAllItems()) ),
852 ("Operation on invalid group"));
853 if (m_p->m_compute_functor && !m_p->m_transaction_mode)
854 ARCANE_FATAL("Cannot remove items on computed group ({0})", name());
855 IMesh* amesh = mesh();
856 if (!amesh)
857 throw ArgumentException(A_FUNCINFO,"null group");
858 ITraceMng* trace = amesh->traceMng();
859 if (isOwn() && amesh->meshPartInfo().nbPart()!=1){
860 ARCANE_THROW(NotSupportedException,"Cannot remove items if isOwn() is true");
861 }
862
863 Integer nb_item_to_remove = items_local_id.size();
864 Int32UniqueArray removed_ids, removed_lids;
865
866 if (nb_item_to_remove!=0) { // on ne peut tout de même pas faire de retour anticipé à cause des observers
867
868 Int32Array& items_lid = m_p->mutableItemsLocalId();
869 const Integer old_size = items_lid.size();
870 bool has_removed = false;
871
872 if(isAllItems()) {
873 // Algorithme anciennement dans DynamicMeshKindInfo
874 // Supprime des items du groupe all_items par commutation avec les
875 // éléments de fin de ce groupe
876 // ie memoire persistante O(size groupe), algo O(remove items)
877 has_removed = true;
878 Integer nb_item = old_size;
879 for( Integer i=0, is=nb_item_to_remove; i<is; ++i ){
880 Int32 removed_local_id = items_local_id[i];
881 Int32 index = m_p->m_items_index_in_all_group[removed_local_id];
882 --nb_item;
883 Int32 moved_local_id = items_lid[nb_item];
884 items_lid[index] = moved_local_id;
885 m_p->m_items_index_in_all_group[moved_local_id] = index;
886 }
887 items_lid.resize(nb_item);
888 }
889 else {
890 // Algorithme pour les autres groupes
891 // Décalage de tableau
892 // ie mémoire locale O(size groupe), algo O(size groupe)
893 // Marque un tableau indiquant les entités à supprimer
894 BoolUniqueArray remove_flags(m_p->maxLocalId(),false);
895 for( Integer i=0; i<nb_item_to_remove; ++i )
896 remove_flags[items_local_id[i]] = true;
897
898 {
899 Integer next_index = 0;
900 for( Integer i=0; i<old_size; ++i ){
901 Int32 lid = items_lid[i];
902 if (remove_flags[lid]) {
903 removed_ids.add(i);
904 removed_lids.add(lid);
905 continue;
906 }
907 items_lid[next_index] = lid;
908 ++next_index;
909 }
910 if (next_index!=old_size) {
911 has_removed = true;
912 items_lid.resize(next_index);
913 }
914 }
915 }
916
917 m_p->updateTimestamp();
918 if(arcaneIsCheck()){
919 trace->info(5) << "ItemGroupImpl::removeItems() group <" << name() << "> "
920 << " old_size=" << old_size
921 << " new_size=" << size()
922 << " removed?=" << has_removed;
923 checkValid();
924 }
925 }
926
927 Int32ConstArrayView observation_info(removed_lids.size(), removed_lids.unguardedBasePointer());
928 m_p->notifyReduceObservers(&observation_info);
929}

1182void ItemGroupImpl::
1183checkValid()
1184{
1185 ITraceMng* msg = m_p->mesh()->traceMng();
1186 if (m_p->m_need_recompute && m_p->m_compute_functor) {
1187 msg->debug(Trace::High) << "ItemGroupImpl::checkValid on " << name() << " : skip group to recompute";
1188 return;
1189 }
1190
1191 // Les points suivants sont vérifiés:
1192 // - une entité n'est présente qu'une fois dans le groupe
1193 // - les entités du groupe ne sont pas détruites
1194 UniqueArray<bool> presence_checks(m_p->maxLocalId());
1195 presence_checks.fill(0);
1196 Integer nb_error = 0;
1197
1198 ItemInternalList items(m_p->items());
1199 Integer items_size = items.size();
1200 Int32ConstArrayView items_lid(m_p->itemsLocalId());
1201
1202 for( Integer i=0, is=items_lid.size(); i<is; ++i ){
1203 Integer lid = items_lid[i];
1204 if (lid>=items_size){
1205 if (nb_error<10){
1206 msg->error() << "Wrong local index lid=" << lid << " max=" << items_size
1207 << " var_max_size=" << m_p->maxLocalId();
1208 }
1209 ++nb_error;
1210 continue;
1211 }
1212 ItemInternal* item = items[lid];
1213 if (item->isSuppressed()){
1214 if (nb_error<10){
1215 msg->error() << "Item " << ItemPrinter(item) << " in group "
1216 << name() << " does not exist anymore";
1217 }
1218 ++nb_error;
1219 }
1220 if (presence_checks[lid]){
1221 if (nb_error<10){
1222 msg->error() << "Item " << ItemPrinter(item) << " in group "
1223 << name() << " was found twice or more";
1224 }
1225 ++nb_error;
1226 }
1227 presence_checks[lid] = true;
1228 }
1229 if (isAllItems()) {
1230 for( Integer i=0, n=items_lid.size(); i<n; ++i ){
1231 Int32 local_id = items_lid[i];
1232 Int32 index_in_all_group = m_p->m_items_index_in_all_group[local_id];
1233 if (index_in_all_group!=i){
1234 if (nb_error<10){
1235 msg->error() << A_FUNCINFO
1236 << ": " << itemKindName(m_p->m_kind)
1237 << ": incoherence between 'local_id' and index in the group 'All' "
1238 << " i=" << i
1239 << " local_id=" << local_id
1240 << " index=" << index_in_all_group;
1241 }
1242 ++nb_error;
1243 }
1244 }
1245 }
1246 if (nb_error!=0) {
1247 String parent_name = "none";
1248 if (parent())
1249 parent_name = parent()->name();
1250 ARCANE_FATAL("Error in group name='{0}' parent='{1}' nb_error={2}",
1251 name(),parent_name,nb_error);
1252 }
1253}
