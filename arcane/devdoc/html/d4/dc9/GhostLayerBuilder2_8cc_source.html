145class GhostLayerBuilder2::BoundaryNodeBitonicSortTraits
146{
147 public:
148 static bool compareLess(const BoundaryNodeInfo& k1,const BoundaryNodeInfo& k2)
149 {
150 Int64 k1_node_uid = k1.node_uid;
151 Int64 k2_node_uid = k2.node_uid;
152 if (k1_node_uid<k2_node_uid)
153 return true;
154 if (k1_node_uid>k2_node_uid)
155 return false;
156
157 Int64 k1_cell_uid = k1.cell_uid;
158 Int64 k2_cell_uid = k2.cell_uid;
159 if (k1_cell_uid<k2_cell_uid)
160 return true;
161 if (k1_cell_uid>k2_cell_uid)
162 return false;
163
164 return (k1.cell_owner<k2.cell_owner);
165 }
166
167 static Parallel::Request send(IParallelMng* pm,Int32 rank,ConstArrayView<BoundaryNodeInfo> values)
168 {
169 const BoundaryNodeInfo* fsi_base = values.data();
170 return pm->send(ByteConstArrayView(messageSize(values),(const Byte*)fsi_base),rank,false);
171 }
172
173 static Parallel::Request recv(IParallelMng* pm,Int32 rank,ArrayView<BoundaryNodeInfo> values)
174 {
175 BoundaryNodeInfo* fsi_base = values.data();
176 return pm->recv(ByteArrayView(messageSize(values),(Byte*)fsi_base),rank,false);
177 }
178
179 static Integer messageSize(ConstArrayView<BoundaryNodeInfo> values)
180 {
181 return CheckedConvert::toInteger(values.size()*sizeof(BoundaryNodeInfo));
182 }
183
184 static BoundaryNodeInfo maxValue()
185 {
186 BoundaryNodeInfo bni;
187 bni.node_uid = INT64_MAX;
188 bni.cell_uid = INT64_MAX;
189 bni.cell_owner = -1;
190 return bni;
191 }
192
193 static bool isValid(const BoundaryNodeInfo& bni)
194 {
195 return bni.node_uid!=INT64_MAX;
196 }

233void GhostLayerBuilder2::
234addGhostLayers()
235{
236 IParallelMng* pm = m_parallel_mng;
237 if (!pm->isParallel())
238 return;
239 Integer nb_ghost_layer = m_mesh->ghostLayerMng()->nbGhostLayer();
240 info() << "** GHOST LAYER BUILDER V" << m_version << " with sort (nb_ghost_layer=" << nb_ghost_layer << ")";
241 if (nb_ghost_layer==0)
242 return;
243 const Int32 my_rank = pm->commRank();
244
245 ItemInternalMap& cells_map = m_mesh->cellsMap();
246 ItemInternalMap& nodes_map = m_mesh->nodesMap();
247
248 // Marque les noeuds frontières
249 _markBoundaryItems();
250
251 Integer boundary_nodes_uid_count = 0;
252
253 // Vérifie qu'il n'y a pas de mailles fantômes avec la version 3.
254 // Si c'est le cas, affiche un avertissement et indique de passer à la version 4.
255 if (m_version==3){
256 Integer nb_ghost = 0;
257 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,cells_map){
258 ItemInternal* cell = iid->value();
259 if (!cell->isOwn())
260 ++nb_ghost;
261 }
262 if (nb_ghost!=0)
263 warning() << "Invalid call to addGhostLayers() with version 3 because mesh "
264 << " already has '" << nb_ghost << "' ghost cells. The computed ghost cells"
265 << " may be wrong. Use version 4 of ghost layer builder if you want to handle this case";
266 }
267
268 // Couche fantôme à laquelle appartient le noeud.
269 UniqueArray<Integer> node_layer(m_mesh->nodeFamily()->maxLocalId(),-1);
270
271 // Couche fantôme à laquelle appartient la maille.
272 UniqueArray<Integer> cell_layer(m_mesh->cellFamily()->maxLocalId(),-1);
273
274 if (m_version>=4){
275 _markBoundaryNodes(node_layer);
276 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,nodes_map){
277 ItemInternal* node = iid->value();
278 if (node_layer[node->localId()]==1)
279 ++boundary_nodes_uid_count;
280 }
281 }
282 else{
283 // Parcours les noeuds et calcule le nombre de noeud frontières
284 // et marque la première couche
285 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,nodes_map){
286 ItemInternal* node = iid->value();
287 Int32 f = node->flags();
288 if (f & ItemFlags::II_Shared){
289 node_layer[node->localId()] = 1;
290 ++boundary_nodes_uid_count;
291 }
292 }
293 }
294
295 info() << "NB BOUNDARY NODE=" << boundary_nodes_uid_count;
296
297 for(Integer current_layer=1; current_layer<=nb_ghost_layer; ++current_layer){
298 //Integer current_layer = 1;
299 info() << "Processing layer " << current_layer;
300 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,cells_map){
301 Cell cell = iid->value();
302 // Ne traite pas les mailles qui ne m'appartiennent pas
303 if (m_version>=4 && cell.owner()!=my_rank)
304 continue;
305 //Int64 cell_uid = cell->uniqueId();
306 Int32 cell_lid = cell.localId();
307 if (cell_layer[cell_lid]!=(-1))
308 continue;
309 bool is_current_layer = false;
310 for( Int32 inode_local_id : cell.nodeIds() ){
311 Integer layer = node_layer[inode_local_id];
312 //info() << "NODE_LAYER lid=" << i_node->localId() << " layer=" << layer;
313 if (layer==current_layer){
314 is_current_layer = true;
315 break;
316 }
317 }
318 if (is_current_layer){
319 cell_layer[cell_lid] = current_layer;
320 //info() << "Current layer celluid=" << cell_uid;
321 // Si non marqué, initialise à la couche courante + 1.
322 for( Int32 inode_local_id : cell.nodeIds() ){
323 Integer layer = node_layer[inode_local_id];
324 if (layer==(-1)){
325 //info() << "Marque node uid=" << i_node->uniqueId();
326 node_layer[inode_local_id] = current_layer + 1;
327 }
328 }
329 }
330 }
331 }
332
333 for( Integer i=1; i<=nb_ghost_layer; ++i )
334 _addGhostLayer(i,node_layer);
335}

351void GhostLayerBuilder2::
352_markBoundaryNodes(ArrayView<Int32> node_layer)
353{
354 IParallelMng* pm = m_mesh->parallelMng();
355 const Int32 my_rank = pm->commRank();
356 ItemInternalMap& faces_map = m_mesh->facesMap();
357 // TODO: regarder s'il est correcte de modifier ItemFlags::II_SubDomainBoundary
358 const int shared_and_boundary_flags = ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary;
359 // Parcours les faces et marque les noeuds, arêtes et faces frontieres
360 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,faces_map){
361 ItemInternal* face_internal = iid->value();
362 Face face = iid->value();
363 Int32 nb_own = 0;
364 for( Integer i=0, n=face.nbCell(); i<n; ++i )
365 if (face.cell(i).owner()==my_rank)
366 ++nb_own;
367 if (nb_own==1){
368 face_internal->addFlags(shared_and_boundary_flags);
369 //++nb_sub_domain_boundary_face;
370 for( Item inode : face.nodes() ){
371 inode.mutableItemBase().addFlags(shared_and_boundary_flags);
372 node_layer[inode.localId()] = 1;
373 }
374 for( Item iedge : face.edges() )
375 iedge.mutableItemBase().addFlags(shared_and_boundary_flags);
376 }
377 }
378}

383void GhostLayerBuilder2::
384_addGhostLayer(Integer current_layer,Int32ConstArrayView node_layer)
385{
386 info() << "Processing ghost layer " << current_layer;
387
388 SharedArray<BoundaryNodeInfo> boundary_node_list;
389 //boundary_node_list.reserve(boundary_nodes_uid_count);
390
391 IParallelMng* pm = m_parallel_mng;
392 Int32 my_rank = pm->commRank();
393 Int32 nb_rank = pm->commSize();
394
395 bool is_verbose = m_is_verbose;
396
397 ItemInternalMap& cells_map = m_mesh->cellsMap();
398 ItemInternalMap& nodes_map = m_mesh->nodesMap();
399
400 // On doit envoyer tous les noeuds dont le numéro de couche est différent de (-1).
401 // NOTE: pour la couche au dessus de 1, il ne faut envoyer qu'une seule valeur.
402 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,cells_map){
403 Cell cell = iid->value();
404 // Ne traite pas les mailles qui ne m'appartiennent pas
405 if (m_version>=4 && cell.owner()!=my_rank)
406 continue;
407 Int64 cell_uid = cell.uniqueId();
408 for( Node node : cell.nodes() ){
409 Int32 node_lid = node.localId();
410 bool do_it = false;
411 //if (node_lid>=node_layer.size())
412 //do_it = true;
413 if (cell.owner()!=my_rank){
414 do_it = true;
415 }
416 else{
417 Integer layer = node_layer[node_lid];
418 do_it = layer<=current_layer;
419 }
420 if (do_it){
421 Int64 node_uid = node.uniqueId();
422 BoundaryNodeInfo nci;
423 nci.node_uid = node_uid;
424 nci.cell_uid = cell_uid;
425 nci.cell_owner = my_rank;
426 boundary_node_list.add(nci);
427 }
428 }
429 }
430 info() << "NB BOUNDARY NODE LIST=" << boundary_node_list.size();
431
432 _sortBoundaryNodeList(boundary_node_list);
433 SharedArray<BoundaryNodeInfo> all_boundary_node_info = boundary_node_list;
434
435 UniqueArray<BoundaryNodeToSendInfo> node_list_to_send;
436 {
437 ConstArrayView<BoundaryNodeInfo> all_bni = all_boundary_node_info;
438 Integer bi_n = all_bni.size();
439 for( Integer i=0; i<bi_n; ++i ){
440 const BoundaryNodeInfo& bni = all_bni[i];
441 // Recherche tous les éléments de all_bni qui ont le même noeud.
442 // Cela représente toutes les mailles connectées à ce noeud.
443 Int64 node_uid = bni.node_uid;
444 Integer last_i = i;
445 for( ; last_i<bi_n; ++last_i )
446 if (all_bni[last_i].node_uid!=node_uid)
447 break;
448 Integer nb_same_node = (last_i - i);
449 if (is_verbose)
450 info() << "NB_SAME_NODE uid=" << node_uid << " n=" << nb_same_node << " last_i=" << last_i;
451 // Maintenant, regarde si les mailles connectées à ce noeud ont le même propriétaire.
452 // Si c'est le cas, il s'agit d'un vrai noeud frontière et il n'y a donc rien à faire.
453 // Sinon, il faudra envoyer la liste des mailles à tous les PE dont les rangs apparaissent dans cette liste
454 Int32 owner = bni.cell_owner;
455 bool has_ghost = false;
456 for( Integer z=0; z<nb_same_node; ++z )
457 if (all_bni[i+z].cell_owner!=owner){
458 has_ghost = true;
459 break;
460 }
461 if (has_ghost){
462 BoundaryNodeToSendInfo si;
463 si.m_index = i;
464 si.m_nb_cell = nb_same_node;
465 node_list_to_send.add(si);
466 }
467 i = last_i-1;
468 }
469 }
470
471 IntegerUniqueArray nb_info_to_send(nb_rank,0);
472 {
473 ConstArrayView<BoundaryNodeInfo> all_bni = all_boundary_node_info;
474 Integer nb_node_to_send = node_list_to_send.size();
475 std::set<Int32> ranks_done;
476 for( Integer i=0; i<nb_node_to_send; ++i ){
477 Integer index = node_list_to_send[i].m_index;
478 Integer nb_cell = node_list_to_send[i].m_nb_cell;
479
480 ranks_done.clear();
481
482 for( Integer kz=0; kz<nb_cell; ++kz ){
483 Int32 krank = all_bni[index+kz].cell_owner;
484 if (ranks_done.find(krank)==ranks_done.end()){
485 ranks_done.insert(krank);
486 // Pour chacun, il faudra envoyer
487 // - le nombre de mailles (1*Int64)
488 // - le uid du noeud (1*Int64)
489 // - le uid et le rank de chaque maille (2*Int64*nb_cell)
490 //TODO: il est possible de stocker les rangs sur Int32
491 nb_info_to_send[krank] += (nb_cell*2) + 2;
492 }
493 }
494 }
495 }
496
497 if (is_verbose){
498 for( Integer i=0; i<nb_rank; ++i ){
499 Integer nb_to_send = nb_info_to_send[i];
500 if (nb_to_send!=0)
501 info() << "NB_TO_SEND rank=" << i << " n=" << nb_to_send;
502 }
503 }
504
505 Integer total_nb_to_send = 0;
506 IntegerUniqueArray nb_info_to_send_indexes(nb_rank,0);
507 for( Integer i=0; i<nb_rank; ++i ){
508 nb_info_to_send_indexes[i] = total_nb_to_send;
509 total_nb_to_send += nb_info_to_send[i];
510 }
511 info() << "TOTAL_NB_TO_SEND=" << total_nb_to_send;
512
513 UniqueArray<Int64> resend_infos(total_nb_to_send);
514 {
515 ConstArrayView<BoundaryNodeInfo> all_bni = all_boundary_node_info;
516 Integer nb_node_to_send = node_list_to_send.size();
517 std::set<Int32> ranks_done;
518 for( Integer i=0; i<nb_node_to_send; ++i ){
519 Integer node_index = node_list_to_send[i].m_index;
520 Integer nb_cell = node_list_to_send[i].m_nb_cell;
521 Int64 node_uid = all_bni[node_index].node_uid;
522
523 ranks_done.clear();
524
525 for( Integer kz=0; kz<nb_cell; ++kz ){
526 Int32 krank = all_bni[node_index+kz].cell_owner;
527 if (ranks_done.find(krank)==ranks_done.end()){
528 ranks_done.insert(krank);
529 Integer send_index = nb_info_to_send_indexes[krank];
530 resend_infos[send_index] = node_uid;
531 ++send_index;
532 resend_infos[send_index] = nb_cell;
533 ++send_index;
534 for( Integer zz=0; zz<nb_cell; ++zz ){
535 resend_infos[send_index] = all_bni[node_index+zz].cell_uid;
536 ++send_index;
537 resend_infos[send_index] = all_bni[node_index+zz].cell_owner;
538 ++send_index;
539 }
540 nb_info_to_send_indexes[krank] = send_index;
541 }
542 }
543 }
544 }
545
546 IntegerUniqueArray nb_info_to_recv(nb_rank,0);
547 {
548 Timer::SimplePrinter sp(traceMng(),"Sending size with AllToAll");
549 pm->allToAll(nb_info_to_send,nb_info_to_recv,1);
550 }
551
552 if (is_verbose)
553 for( Integer i=0; i<nb_rank; ++i )
554 info() << "NB_TO_RECV: I=" << i << " n=" << nb_info_to_recv[i];
555
556 Integer total_nb_to_recv = 0;
557 for( Integer i=0; i<nb_rank; ++i )
558 total_nb_to_recv += nb_info_to_recv[i];
559
560 // Il y a de fortes chances que cela ne marche pas si le tableau est trop grand,
561 // il faut proceder avec des tableaux qui ne depassent pas 2Go a cause des
562 // Int32 de MPI.
563 // TODO: Faire le AllToAll en plusieurs fois si besoin.
564 // TOOD: Fusionner ce code avec celui de FaceUniqueIdBuilder2.
565 UniqueArray<Int64> recv_infos;
566 {
567 Int32 vsize = sizeof(Int64) / sizeof(Int64);
568 Int32UniqueArray send_counts(nb_rank);
569 Int32UniqueArray send_indexes(nb_rank);
570 Int32UniqueArray recv_counts(nb_rank);
571 Int32UniqueArray recv_indexes(nb_rank);
572 Int32 total_send = 0;
573 Int32 total_recv = 0;
574 for( Integer i=0; i<nb_rank; ++i ){
575 send_counts[i] = (Int32)(nb_info_to_send[i] * vsize);
576 recv_counts[i] = (Int32)(nb_info_to_recv[i] * vsize);
577 send_indexes[i] = total_send;
578 recv_indexes[i] = total_recv;
579 total_send += send_counts[i];
580 total_recv += recv_counts[i];
581 }
582 recv_infos.resize(total_nb_to_recv);
583
584 Int64ConstArrayView send_buf(total_nb_to_send*vsize,(Int64*)resend_infos.data());
585 Int64ArrayView recv_buf(total_nb_to_recv*vsize,(Int64*)recv_infos.data());
586
587 info() << "BUF_SIZES: send=" << send_buf.size() << " recv=" << recv_buf.size();
588 {
589 Timer::SimplePrinter sp(traceMng(),"Send values with AllToAll");
590 pm->allToAllVariable(send_buf,send_counts,send_indexes,recv_buf,recv_counts,recv_indexes);
591 }
592 }
593
594 SubDomainItemMap cells_to_send(50,true);
595
596 // TODO: il n'y a a priori pas besoin d'avoir les mailles ici mais
597 // seulement la liste des procs a qui il faut envoyer. Ensuite,
598 // si le proc connait a qui il doit envoyer, il peut envoyer les mailles
599 // à ce moment la. Cela permet d'envoyer moins d'infos dans le AllToAll précédent
600
601 {
602 Integer index = 0;
603 Int32UniqueArray my_cells;
604 SharedArray<Int32> ranks_to_send;
605 std::set<Int32> ranks_done;
606 while (index<total_nb_to_recv){
607 Int64 node_uid = recv_infos[index];
608 ++index;
609 Int64 nb_cell = recv_infos[index];
610 ++index;
611 if (is_verbose)
612 info() << "NODE uid=" << node_uid << " nb_cell=" << nb_cell << " idx=" << (index-2);
613 my_cells.clear();
614 ranks_to_send.clear();
615 ranks_done.clear();
616 for( Integer kk=0; kk<nb_cell; ++kk ){
617 Int64 cell_uid = recv_infos[index];
618 ++index;
619 Int32 cell_owner = CheckedConvert::toInt32(recv_infos[index]);
620 ++index;
621 if (kk==0 && current_layer==1 && m_is_allocate)
622 // Je suis la maille de plus petit uid et donc je
623 // positionne le propriétaire du noeud.
624 // TODO: ne pas faire cela ici, mais le faire dans une routine à part.
625 nodes_map[node_uid]->setOwner(cell_owner,my_rank);
626 if (is_verbose)
627 info() << " CELL=" << cell_uid << " owner=" << cell_owner;
628 if (cell_owner==my_rank){
629 ItemInternalMap::Data* dcell = cells_map.lookup(cell_uid);
630 if (!dcell)
631 throw FatalErrorException(A_FUNCINFO,"Internal error: cell not in our mesh");
632 my_cells.add(dcell->value()->localId());
633 }
634 else{
635 if (ranks_done.find(cell_owner)==ranks_done.end()){
636 ranks_to_send.add(cell_owner);
637 ranks_done.insert(cell_owner);
638 }
639 }
640 }
641
642 if (is_verbose)
643 info() << "CELLS TO SEND: node_uid=" << node_uid
644 << " nb_rank=" << ranks_to_send.size()
645 << " nb_cell=" << my_cells.size();
646
647 for( Integer zrank=0, zn=ranks_to_send.size(); zrank<zn; ++zrank ){
648 SubDomainItemMap::Data* d = cells_to_send.lookupAdd(ranks_to_send[zrank]);
649 Int32Array& c = d->value();
650 for( Integer zid=0, zid_size=my_cells.size(); zid<zid_size; ++zid ){
651 // TODO: regarder si maille pas déjà présente et ne pas l'ajouter si ce n'est pas nécessaire.
652 c.add(my_cells[zid]);
653 }
654 }
655 }
656 }
657
658 info() << "GHOST V3 SERIALIZE CELLS";
659 _sendAndReceiveCells(cells_to_send);
660}

671void GhostLayerBuilder2::
672_sortBoundaryNodeList(Array<BoundaryNodeInfo>& boundary_node_list)
673{
674 IParallelMng* pm = m_parallel_mng;
675 Int32 my_rank = pm->commRank();
676 Int32 nb_rank = pm->commSize();
677 bool is_verbose = m_is_verbose;
678
679 Parallel::BitonicSort<BoundaryNodeInfo,BoundaryNodeBitonicSortTraits> boundary_node_sorter(pm);
680 boundary_node_sorter.setNeedIndexAndRank(false);
681
682 {
683 Timer::SimplePrinter sp(traceMng(),"Sorting boundary nodes");
684 boundary_node_sorter.sort(boundary_node_list);
685 }
686
687 if (is_verbose){
688 ConstArrayView<BoundaryNodeInfo> all_bni = boundary_node_sorter.keys();
689 Integer n = all_bni.size();
690 for( Integer i=0; i<n; ++i ){
691 const BoundaryNodeInfo& bni = all_bni[i];
692 info() << "NODES_KEY i=" << i
693 << " node=" << bni.node_uid
694 << " cell=" << bni.cell_uid
695 << " rank=" << bni.cell_owner;
696 }
697 }
698
699 // TODO: il n'y a pas besoin d'envoyer toutes les mailles.
700 // pour déterminer le propriétaire d'un noeud, il suffit
701 // que chaque PE envoie sa maille de plus petit UID.
702 // Ensuite, chaque noeud a besoin de savoir la liste
703 // des sous-domaines connectés pour renvoyer l'info. Chaque
704 // sous-domaine en sachant cela saura a qui il doit envoyer
705 // les mailles fantomes.
706
707 {
708 ConstArrayView<BoundaryNodeInfo> all_bni = boundary_node_sorter.keys();
709 Integer n = all_bni.size();
710 // Comme un même noeud peut être présent dans la liste du proc précédent, chaque PE
711 // (sauf le 0) envoie au proc précédent le début sa liste qui contient les même noeuds.
712
713 UniqueArray<BoundaryNodeInfo> end_node_list;
714 Integer begin_own_list_index = 0;
715 if (n!=0 && my_rank!=0){
716 if (BoundaryNodeBitonicSortTraits::isValid(all_bni[0])){
717 Int64 node_uid = all_bni[0].node_uid;
718 for( Integer i=0; i<n; ++i ){
719 if (all_bni[i].node_uid!=node_uid){
720 begin_own_list_index = i;
721 break;
722 }
723 else
724 end_node_list.add(all_bni[i]);
725 }
726 }
727 }
728 info() << "BEGIN_OWN_LIST_INDEX=" << begin_own_list_index;
729 if (is_verbose){
730 for( Integer k=0, kn=end_node_list.size(); k<kn; ++k )
731 info() << " SEND node_uid=" << end_node_list[k].node_uid
732 << " cell_uid=" << end_node_list[k].cell_uid;
733 }
734
735 UniqueArray<BoundaryNodeInfo> end_node_list_recv;
736
737 UniqueArray<Parallel::Request> requests;
738 Integer recv_message_size = 0;
739 Integer send_message_size = BoundaryNodeBitonicSortTraits::messageSize(end_node_list);
740
741 // Envoie et réceptionne d'abord les tailles.
742 if (my_rank!=(nb_rank-1)){
743 requests.add(pm->recv(IntegerArrayView(1,&recv_message_size),my_rank+1,false));
744 }
745 if (my_rank!=0){
746 requests.add(pm->send(IntegerConstArrayView(1,&send_message_size),my_rank-1,false));
747 }
748
749 pm->waitAllRequests(requests);
750 requests.clear();
751
752 if (recv_message_size!=0){
753 Integer message_size = CheckedConvert::toInteger(recv_message_size/sizeof(BoundaryNodeInfo));
754 end_node_list_recv.resize(message_size);
755 requests.add(BoundaryNodeBitonicSortTraits::recv(pm,my_rank+1,end_node_list_recv));
756 }
757 if (send_message_size!=0)
758 requests.add(BoundaryNodeBitonicSortTraits::send(pm,my_rank-1,end_node_list));
759
760 pm->waitAllRequests(requests);
761
762 boundary_node_list.clear();
763 boundary_node_list.addRange(all_bni.subConstView(begin_own_list_index,n-begin_own_list_index));
764 boundary_node_list.addRange(end_node_list_recv);
765 }
766}

771void GhostLayerBuilder2::
772_sendAndReceiveCells(SubDomainItemMap& cells_to_send)
773{
774 auto exchanger { ParallelMngUtils::createExchangerRef(m_parallel_mng) };
775
776 const bool is_verbose = m_is_verbose;
777
778 // Envoie et réceptionne les mailles fantômes
779 for( SubDomainItemMap::Enumerator i_map(cells_to_send); ++i_map; ){
780 Int32 sd = i_map.data()->key();
781 Int32Array& items = i_map.data()->value();
782
783 // Comme la liste par sous-domaine peut contenir plusieurs
784 // fois la même maille, on trie la liste et on supprime les
785 // doublons
786 std::sort(std::begin(items),std::end(items));
787 auto new_end = std::unique(std::begin(items),std::end(items));
788 items.resize(CheckedConvert::toInteger(new_end-std::begin(items)));
789 if (is_verbose)
790 info(4) << "CELLS TO SEND SD=" << sd << " Items=" << items;
791 else
792 info(4) << "CELLS TO SEND SD=" << sd << " nb=" << items.size();
793 exchanger->addSender(sd);
794 }
795 exchanger->initializeCommunicationsMessages();
796 for( Integer i=0, ns=exchanger->nbSender(); i<ns; ++i ){
797 ISerializeMessage* sm = exchanger->messageToSend(i);
798 Int32 rank = sm->destination().value();
799 ISerializer* s = sm->serializer();
800 Int32ConstArrayView items_to_send = cells_to_send[rank];
801 m_mesh->serializeCells(s,items_to_send);
802 }
803 exchanger->processExchange();
804 info(4) << "END EXCHANGE CELLS";
805 for( Integer i=0, ns=exchanger->nbReceiver(); i<ns; ++i ){
806 ISerializeMessage* sm = exchanger->messageToReceive(i);
807 ISerializer* s = sm->serializer();
808 m_mesh->addCells(s);
809 }
810 m_mesh_builder->printStats();
811}

835 if (face.itemBase().flags() & ItemFlags::II_Boundary){
836 is_sub_domain_boundary_face = true;
837 }
838 else{
839 if (face.nbCell()==2 && (face.cell(0).owner()!=my_rank || face.cell(1).owner()!=my_rank))
840 is_sub_domain_boundary_face = true;

845 inode.mutableItemBase().addFlags(shared_and_boundary_flags);
846 for( Item iedge : face.edges() )
847 iedge.mutableItemBase().addFlags(shared_and_boundary_flags);
848 }
