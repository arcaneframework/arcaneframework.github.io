42 class ItemInternalVectorViewConstIterator
43 {
44  friend class ItemInternalVectorView;
45  typedef ItemInternal* ItemInternalPtr;
46  private:
47  ItemInternalVectorViewConstIterator(const ItemInternalPtr* items,
48  const Int32* ARCANE_RESTRICT local_ids,
49  Integer index)
50  : m_items(items), m_local_ids(local_ids), m_index(index){}
51  public:
52  typedef ItemInternalVectorViewConstIterator ThatClass;
53  public:
54  typedef std::random_access_iterator_tag iterator_category;
56  typedef const ItemInternalPtr* pointer;
58  typedef const ItemInternalPtr& reference;
60  typedef ItemInternalPtr value_type;
62  typedef Integer size_type;
64  typedef Integer difference_type;
65  public:
66  value_type operator*() const { return m_items[ m_local_ids[m_index] ]; }
67  value_type operator->() const { return m_items[ m_local_ids[m_index] ]; }
68  ThatClass& operator++() { ++m_index; return (*this); }
69  ThatClass& operator--() { --m_index; return (*this); }
70  void operator+=(difference_type v) { m_index += v; }
71  void operator-=(difference_type v) { m_index -= v; }
72  friend Integer operator-(const ThatClass& a,const ThatClass& b)
73  {
74  return a.m_index - b.m_index;
75  }
76  friend ThatClass operator-(const ThatClass& a,difference_type v)
77  {
78  Integer index = a.m_index - v;
79  return ThatClass(a.m_items,a.m_local_ids,index);
80  }
81  friend ThatClass operator+(const ThatClass& a,difference_type v)
82  {
83  Integer index = a.m_index + v;
84  return ThatClass(a.m_items,a.m_local_ids,index);
85  }
86  friend bool operator<(const ThatClass& lhs,const ThatClass& rhs)
87  {
88  return lhs.m_index<=rhs.m_index;
89  }
90  friend bool operator==(const ThatClass& lhs,const ThatClass& rhs)
91  {
92  // TODO: regarder si cela ne pose pas de problemes de performance
93  // de faire ces trois comparaions. Si c'est le cas on peut
94  // faire uniquement la dernière.
95  return lhs.m_items==rhs.m_items && lhs.m_local_ids==rhs.m_local_ids && lhs.m_index==rhs.m_index;
96  }
97  friend bool operator!=(const ThatClass& lhs,const ThatClass& rhs)
98  {
99  return !(lhs==rhs);
100  }
101 
102  private:
103 
104  const ItemInternalPtr* m_items;
105  const Int32* ARCANE_RESTRICT m_local_ids;
106  Integer m_index;
107 };
