86void FaceReorienter::
87checkAndChangeOrientation(Face face)
88{
89 if (!m_face_family)
90 m_face_family = ARCANE_CHECK_POINTER(dynamic_cast<FaceFamily*>(face.itemFamily()));
91 Integer face_nb_node = face.nbNode();
92
93 m_nodes_unique_id.resize(face_nb_node);
94 m_nodes_local_id.resize(face_nb_node);
95
96 for (Integer i_node=0; i_node<face_nb_node; ++i_node) {
97 m_nodes_unique_id[i_node] = face.node(i_node).uniqueId();
98 m_nodes_local_id[i_node] = face.node(i_node).localId();
99 }
100
101 m_face_nodes_index.resize(face_nb_node);
102 mesh_utils::reorderNodesOfFace2(m_nodes_unique_id,m_face_nodes_index);
103
104 for (Integer i_node=0; i_node<face_nb_node; ++i_node) {
105 ItemLocalId node_lid(m_nodes_local_id[m_face_nodes_index[i_node]]);
106 m_face_family->replaceNode(face,i_node, node_lid);
107 }
108
109 // On cherche le plus petit uid de la face
110 std::pair<Int64,Int64> face_smallest_node_uids = std::make_pair(face.node(0).uniqueId(),
111 face.node(1).uniqueId());
112
113 Cell cell = face.cell(0);
114 Int32 cell0_lid = cell.localId();
115 Integer local_face_number = -1;
116 for (Integer i_face=0; i_face<cell.nbFace(); ++i_face) {
117 if (cell.face(i_face) == face) {
118 // On a trouvé la bonne face
119 local_face_number = i_face;
120 break;
121 }
122 }
123
124 if (local_face_number==(-1))
125 ARCANE_FATAL("Incoherent connectivity: Face {0} not connected to cell {1}",
126 face.uniqueId(),cell.uniqueId());
127
128 const ItemTypeInfo::LocalFace& local_face = cell.typeInfo()->localFace(local_face_number);
129 bool cell0_is_back_cell = false;
130
131 if (face_nb_node == 2) {
132 cell0_is_back_cell = (cell.node(local_face.node(0)).uniqueId() == face_smallest_node_uids.first);
133 }
134 else {
135 for (Integer i_node=0; i_node<local_face.nbNode(); ++i_node) {
136 if (cell.node(local_face.node(i_node)).uniqueId() == face_smallest_node_uids.first) {
137 if (cell.node(local_face.node((i_node+1)%local_face.nbNode())).uniqueId() == face_smallest_node_uids.second) {
138 cell0_is_back_cell = true;
139 break;
140 }
141 else {
142 cell0_is_back_cell = false;
143 break;
144 }
145 }
146 }
147 }
148
149 Int32 cell1_lid = (face.nbCell()==2) ? face.cell(1).localId() : NULL_ITEM_LOCAL_ID;
150 // Paire contenant la back_cell et front_cell de la face.
151 std::pair<Int32,Int32> face_cells(cell1_lid, cell0_lid);
152 if (cell0_is_back_cell) {
153 // Si on arrive ici c'est que la maille 0 est la back_cell
154 std::swap(face_cells.first,face_cells.second);
155 }
156 m_face_family->setBackAndFrontCells(face,face_cells.first,face_cells.second);
157}

162void FaceReorienter::
163checkAndChangeOrientationAMR(Face face)
164{
165 if (!m_face_family)
166 m_face_family = ARCANE_CHECK_POINTER(dynamic_cast<FaceFamily*>(face.itemFamily()));
167
168 m_nodes_unique_id.resize(face.nbNode());
169 m_nodes_local_id.resize(face.nbNode());
170
171 for (Integer i_node=0; i_node<face.nbNode(); ++i_node) {
172 m_nodes_unique_id[i_node] = face.node(i_node).uniqueId();
173 m_nodes_local_id[i_node] = face.node(i_node).localId();
174 }
175
176 m_face_nodes_index.resize(face.nbNode());
177 mesh_utils::reorderNodesOfFace2(m_nodes_unique_id,m_face_nodes_index);
178
179 for (Integer i_node=0; i_node<face.nbNode(); ++i_node) {
180 ItemLocalId node_lid(m_nodes_local_id[m_face_nodes_index[i_node]]);
181 m_face_family->replaceNode(face,i_node, node_lid);
182 }
183
184 // On cherche le plus petit uid de la face
185 std::pair<Int64,Int64> face_smallest_node_uids = std::make_pair(face.node(0).uniqueId(),
186 face.node(1).uniqueId());
187
188 //ItemInternal* cell = face->cell(0);
189 bool cell_0 = false;
190 bool cell_1 = false;
191 Cell cell;
192
193 if(face.nbCell() == 2){
194 if(face.cell(0).level() >= face.cell(1).level()){
195 cell = face.cell(0);
196 cell_0 = true;
197 }
198 else {
199 cell = face.cell(1);
200 cell_1 = true;
201 }
202 }
203 else {
204 cell = face.cell(0);
205 cell_0 = true;
206 }
207 Integer local_face_number = -1;
208 for (Integer i_face=0; i_face<cell.nbFace(); ++i_face) {
209 if (cell.face(i_face) == face) {
210 // On a trouvé la bonne face
211 local_face_number = i_face;
212 break;
213 }
214 }
215
216 if (local_face_number==(-1))
217 ARCANE_FATAL("Incoherent connectivity: Face {0} not connected to cell {1}",
218 face.uniqueId(),cell.uniqueId());
219
220 const ItemTypeInfo::LocalFace& local_face = cell.typeInfo()->localFace(local_face_number);
221 bool cell_is_back_cell = false;
222
223 if (face.nbNode() == 2) {
224 cell_is_back_cell = (cell.node(local_face.node(0)).uniqueId() == face_smallest_node_uids.first);
225 }
226 else {
227 for (Integer i_node=0; i_node<local_face.nbNode(); ++i_node) {
228 if (cell.node(local_face.node(i_node)).uniqueId() == face_smallest_node_uids.first) {
229 if (cell.node(local_face.node((i_node+1)%local_face.nbNode())).uniqueId() == face_smallest_node_uids.second) {
230 cell_is_back_cell = true;
231 break;
232 }
233 else {
234 cell_is_back_cell = false;
235 break;
236 }
237 }
238 }
239 }
240
241 // Paire contenant la back_cell et front_cell de la face.
242 std::pair<Int32,Int32> face_cells(NULL_ITEM_LOCAL_ID,NULL_ITEM_LOCAL_ID);
243 bool face_has_two_cell = (face.nbCell()==2);
244
245 if(cell_0){
246 Int32 cell1_lid = (face_has_two_cell) ? face.cell(1).localId() : NULL_ITEM_LOCAL_ID;
247 if (cell_is_back_cell) {
248 // Si on arrive ici c'est que la maille 0 est la back_cell
249 // La front cell est toujours cell1_lid (qui peut être nulle).
250 face_cells = { face.cell(0).localId(), cell1_lid };
251 }
252 else {
253 // Si on arrive ici c'est que la maille 0 est la front_cell
254 // La back cell est toujours cell1_lid (qui peut être nulle)
255 face_cells.first = cell1_lid;
256 face_cells.second = (face_has_two_cell) ? cell.localId() : face.cell(0).localId();
257 }
258 }
259 else if(cell_1){
260 if (cell_is_back_cell) {
261 // Si on arrive ici c'est que la maille 0 est la front_cell
262 // On met à jour les infos d'orientation
263 face_cells.second = face.cell(0).localId();
264 // GG Attention, si ici, cela signifie qu'il faut échanger la front cell
265 // et la back cell car la back cell doit toujours être la première
266 face_cells.first = (face_has_two_cell) ? cell.localId() : NULL_ITEM_LOCAL_ID;
267 }
268 else {
269 // Si on arrive ici c'est que la maille 0 est la back_cell
270 // On met à jour les infos d'orientation
271 face_cells.first = face.cell(0).localId();
272 face_cells.second = (face_has_two_cell) ? face.cell(1).localId() : NULL_ITEM_LOCAL_ID;
273 }
274 }
275 m_face_family->setBackAndFrontCells(face,face_cells.first,face_cells.second);
276}
