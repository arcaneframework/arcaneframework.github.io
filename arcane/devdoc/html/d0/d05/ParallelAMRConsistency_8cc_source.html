89void ParallelAMRConsistency::
90init()
91{
92 CHECKPERF( m_perf_counter.start(PerfCounter::INIT) )
93 // Marque les noeuds sur la frontière
94 Integer nb_active_face = static_cast<Integer> (m_mesh->nbFace() * 0.2); // 20% de faces shared (sur estimé)
95 m_shared_face_uids.reserve(nb_active_face) ;
96 m_shared_face_uids.clear() ;
97 m_connected_shared_face_uids.reserve(nb_active_face) ;
98 m_connected_shared_face_uids.clear() ;
99 Integer sid = m_mesh->parallelMng()->commRank();
100 ENUMERATE_FACE(iface,m_mesh->allFaces()){
101 Face face = *iface;
102 ItemUniqueId face_uid = face.uniqueId();
103 int face_flags = face.itemBase().flags();
104 if (face.nbCell()==2 && (face.cell(0).level()==0 && face.cell(1).level()==0)){
105 if ( (face.cell(0).owner()!=sid || face.cell(1).owner()!= sid) ||
106 (face_flags & ItemFlags::II_Shared) ||
107 (face_flags & ItemFlags::II_SubDomainBoundary)){
108 m_shared_face_uids.add(face_uid);
109 }
110 else if (_hasSharedNodes(face)){
111 m_connected_shared_face_uids.add(face_uid);
112 }
113 }
114 else if (face.nbCell()==1 && face.cell(0).level() == 0){
115 if (_hasSharedNodes(face)){
116 m_connected_shared_face_uids.add(face_uid);
117 }
118 }
119 }
120 m_is_updated = true ;
121
122 CHECKPERF( m_perf_counter.stop(PerfCounter::INIT) )
123}

129void ParallelAMRConsistency::
130_addFaceToList(Face face, FaceInfoMap& face_map)
131{
132 Integer nb_node = face.nbNode();
133 Real3 center(0., 0., 0.);
134 Integer data_index = m_face_info_mng.size();
135 for (Node node : face.nodes()){
136 Real3 node_coord = m_nodes_coord[node];
137 ItemUniqueId uid = node.uniqueId();
138 m_face_info_mng.add(uid);
139 center += node_coord;
140 }
141 center /= nb_node;
142 //info() << "ADD FACE uid=" << face.uniqueId() << " nb_node="
143 //<< nb_node << " center=" << center;
144 FaceInfo fi(face.uniqueId(), face.cell(0).uniqueId(), nb_node, face.owner(), data_index, &m_face_info_mng);
145 fi.setCenter(center);
146 face_map.add(face.uniqueId(), fi);
147}

159 }
160 return false;

172makeNewItemsConsistent(NodeMapCoordToUid& node_finder, FaceMapCoordToUid& face_finder)
173{
174 CHECKPERF( m_perf_counter.start(PerfCounter::COMPUTE) )
175 //Integer nb_sub_domain_boundary_face = 0;
176 // Marque les noeuds sur la frontière
177 Integer nb_active_face = static_cast<Integer> (m_mesh->nbFace() * 0.2); // 20% de faces shared (sur estimé)
178
179 m_nodes_info.resize((nb_active_face * 2) + 5);
180 m_nodes_info.clear() ;
181 ENUMERATE_NODE(i_item,m_mesh->allNodes()){
182 Real3 node_coord = m_nodes_coord[i_item];
183 ItemUniqueId uid = i_item->uniqueId();
184 NodeInfo node_info(uid, i_item->owner());
185 node_info.setCoord(node_coord);
186 m_nodes_info.add(uid, node_info);
187 }
188
189
190 m_active_faces.resize((nb_active_face * 2) + 5);
191 m_active_nodes.resize((nb_active_face * 2) + 5);
192 m_active_faces.clear() ;
193 m_face_info_mng.m_nodes_unique_id.clear() ;
194 m_face_info_mng.m_nodes_unique_id.reserve(nb_active_face) ;
195
196
197 FaceFamily* true_face_family = dynamic_cast<FaceFamily*> (m_mesh->faceFamily());
198 if (!true_face_family)
199 ARCANE_FATAL("can not obtain FaceFamily");
200
201 Int32 sid = m_mesh->parallelMng()->commRank();
202 //UniqueArray<ItemInternal*> active_faces;
203 ItemMap active_faces ;
204 UniqueArray < ItemUniqueId > active_faces_to_send;
205 typedef std::set<ItemInternal*> Set;
206 Set active_nodes_set;
207 ItemMap active_nodes ;
208 // Parcours les faces et marque les noeuds frontieres actives
209 DynamicMesh* mesh = dynamic_cast<DynamicMesh*> (m_mesh);
210 if (!mesh)
211 throw FatalErrorException(A_FUNCINFO, "can not obtain DynamicMesh");
212 ItemInternalMap& faces_map = mesh->facesMap();
213
214 for(Integer iface=0;iface<m_shared_face_uids.size();++iface){
215 Int64 face_uid = m_shared_face_uids[iface];
216 Face face = faces_map[face_uid] ;
217 {
218 UniqueArray<ItemInternal*> subfaces;
219 true_face_family->allSubFaces(face, subfaces);
220 if (subfaces.size() != 1){
221 for (Integer s = 0; s < subfaces.size(); s++){
222 Face face2 = subfaces[s];
223 face2.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
224 Int64 uid = face2.uniqueId() ;
225 bool face_to_send = face_finder.isNewUid(uid);
226 if(face_to_send){
227 _addFaceToList(face2, m_active_faces);
228
229 //active_faces.add(face2);
230 active_faces.insert(ItemMapValue(uid,face2));
231 active_faces_to_send.add(ItemUniqueId(uid));
232 for ( Node node : face2.nodes() ){
233 node.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
234 active_nodes.insert(ItemMapValue(node.uniqueId(),node));
235 }
236 for ( Edge edge : face2.edges() ){
237 edge.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
238 }
239 }
240 }
241 }
242 }
243 }
244 for(Integer iface=0;iface<m_connected_shared_face_uids.size();++iface){
245 Int64 face_uid = m_connected_shared_face_uids[iface];
246 Face face(faces_map[face_uid]);
247
248 {
249 Integer nb_node = face.nbNode() ;
250 typedef std::pair<Real3,Real3> Edge ;
251 UniqueArray<Edge> edges ;
252 edges.reserve(nb_node) ;
253 for(Integer i=0;i<nb_node;++i){
254 Integer next = i==nb_node-1?0:i+1;
255 Node node1 = face.node(i) ;
256 Node node2 = face.node(next) ;
257 if( (node1.itemBase().flags() & (ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary)) &&
258 (node2.itemBase().flags() & (ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary) ) ){
259 edges.add(Edge(m_nodes_coord[node1],m_nodes_coord[node2]-m_nodes_coord[node1])) ;
260 }
261 }
262 UniqueArray<ItemInternal*> subfaces;
263 true_face_family->allSubFaces(face, subfaces);
264 for (Integer s = 0; s < subfaces.size(); s++){
265 Face face2(subfaces[s]);
266 Int64 uid = face2.uniqueId() ;
267 bool face_to_send = face_finder.isNewUid(uid) ;
268 if (face_to_send){
269 Integer nb_node2 = face2.nbNode() ;
270 for(Integer i=0;i<nb_node2;++i){
271 Node node_i = face2.node(i) ;
272 Real3 Xi = m_nodes_coord[node_i] ;
273 for(Integer j=0;j<edges.size();++j){
274 Real3 n = Xi-edges[j].first ;
275 Real sinteta = math::cross(edges[j].second,n).squareNormL2() ;
276 if (math::isZero(sinteta)){
277 node_i.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
278 //active_nodes_set.insert(node_i);
279 active_nodes.insert(ItemMapValue(node_i.uniqueId(),node_i));
280 _addNodeToList(node_i, m_active_nodes);
281 }
282 }
283 }
284 }
285 }
286 }
287 }
288
289 UniqueArray<ItemUniqueId> active_nodes_to_send(arcaneCheckArraySize(active_nodes.size()));
290 ItemMap::const_iterator nit(active_nodes.begin()), nend(active_nodes.end());
291 for (int i=0; nit != nend; ++nit,++i){
292 Item node = nit->second;
293 active_nodes_to_send[i]= node.uniqueId();
294 //active_nodes.insert(ItemMapValue(uid,node)) ;
295 }
296 CHECKPERF( m_perf_counter.stop(PerfCounter::COMPUTE) )
297
298 //CHECKPERF( m_perf_counter.start(PerfCounter::GATHERFACE) )
299 ItemUidSet update_face_uids ;
300 ItemUidSet update_node_uids ;
301 _gatherFaces(active_faces_to_send,active_nodes_to_send, m_active_faces, node_finder, face_finder,update_face_uids,update_node_uids);
302 //CHECKPERF( m_perf_counter.stop(PerfCounter::GATHERFACE) )
303
304
305
306 CHECKPERF( m_perf_counter.start(PerfCounter::UPDATE) )
307 //UPDATE FACES
308 //for (Integer index = 0; index < active_faces.size(); index++)
309 for(ItemUidSet::iterator iter = update_face_uids.begin();iter!=update_face_uids.end();++iter){
310 Int64 face_uid = *iter ;
311 Item face = active_faces[face_uid];
312 //const Int64 current_uid = face->uniqueId();
313 FaceInfo& fi = m_active_faces[face.uniqueId()];
314 Int64 new_uid = fi.uniqueId() ;
315 faces_map.remove(face_uid) ;
316 //if (current_uid != fi.uniqueId())
317 ARCANE_ASSERT((face_uid != new_uid),("AMR CONSISTENCY UPDATE FACE ERROR")) ;
318 face.mutableItemBase().setUniqueId(fi.uniqueId());
319 face.mutableItemBase().setOwner(fi.owner(), sid);
320 //debug() << "[\t ParallelAMRConsistency] NEW FACE BEFORE " << face->uniqueId()<<" new uid "<<fi.uniqueId() << " " << fi.owner();
321 faces_map.add(new_uid,ItemCompatibility::_itemInternal(face));
322 }
323
324 //UPDATE NODES
325 //ENUMERATE_NODE(i_item,m_mesh->allNodes())
326 ItemInternalMap& nodes_map = mesh->nodesMap();
327 for(ItemUidSet::iterator iter = update_node_uids.begin();iter!=update_node_uids.end();++iter){
328 //ItemInternal* node = i_item->internal() ;
329 Int64 node_uid = *iter ;
330 Item node = active_nodes[node_uid];
331 if (node.null())
332 ARCANE_FATAL("AMR CONSISTENCY NULL NODE ERROR");
333 //const Int64 node_uid = node->uniqueId();
334 NodeInfo ni = m_nodes_info[node.uniqueId()];
335 Int64 new_uid = ni.uniqueId() ;
336 //if (node_uid != ni.uniqueId())
337 ARCANE_ASSERT((node_uid != new_uid),("AMR CONSISTENCY UPDATE NODE ERROR")) ;
338 //debug() << "[\t ParallelAMRConsistency] OLD NEW NODE " << node_uid << " " << ni.uniqueId()<<" owner "<<ni.owner();
339 nodes_map.remove(node_uid) ;
340 node.mutableItemBase().setUniqueId(ni.uniqueId());
341 node.mutableItemBase().setOwner(ni.owner(), sid);
342 node.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
343 nodes_map.add(new_uid,ItemCompatibility::_itemInternal(node));
344 }
345 CHECKPERF( m_perf_counter.stop(PerfCounter::UPDATE) )
346
347 CHECKPERF( m_perf_counter.start(PerfCounter::ENDUPDATE) )
348 m_mesh->nodeFamily()->partialEndUpdate();
349 m_mesh->faceFamily()->partialEndUpdate();
350 CHECKPERF( m_perf_counter.stop(PerfCounter::ENDUPDATE) )
351}

365void ParallelAMRConsistency::
366_gatherFaces(ConstArrayView<ItemUniqueId> faces_to_send,
367 ConstArrayView<ItemUniqueId> nodes_to_send,
368 FaceInfoMap& face_map,
369 MapCoordToUid& node_finder,
370 MapCoordToUid& face_finder,
371 ItemUidSet& updated_face_uids,
372 ItemUidSet& updated_node_uids)
373{
374 CHECKPERF( m_perf_counter.start(PerfCounter::GATHERFACE) )
375
376 IParallelMng* pm = m_mesh->parallelMng();
377 Integer sub_domain_id = pm->commRank();
378 Integer nb_sub_domain = pm->commSize();
379
380 const Real tol = 10e-12;
381
382 SerializeBuffer sbuf;
383 sbuf.setMode(ISerializer::ModeReserve);
384 Integer nb_to_send = faces_to_send.size();
385 Integer nb_node_to_send = nodes_to_send.size();
386
387 Int64UniqueArray unique_ids(nb_to_send);
388 Int64UniqueArray cells_unique_ids(nb_to_send);
389 Int64UniqueArray nodes_unique_id;
390 nodes_unique_id.reserve(nb_node_to_send);
391 RealUniqueArray coords;
392 coords.reserve(3 * nb_to_send);
393 RealUniqueArray nodes_coords;
394 nodes_coords.reserve(3 *nb_node_to_send );
395 for (Integer i = 0; i < nb_to_send; ++i){
396 const FaceInfo& fi = face_map[faces_to_send[i]];
397 unique_ids[i] = fi.uniqueId().asInt64();
398 cells_unique_ids[i] = fi.cellUniqueId().asInt64();
399 coords.add(fi.center().x);
400 coords.add(fi.center().y);
401 coords.add(fi.center().z);
402 }
403 debug()<<"SEND NODE : "<<nb_node_to_send;
404 for (Integer i = 0; i < nb_node_to_send; ++i){
405 //debug()<<"SEND NODE : "<<nodes_to_send[i];
406 ItemUniqueId nuid(nodes_to_send[i]);
407 nodes_unique_id.add(nuid.asInt64());
408 NodeInfo ni = m_nodes_info[nuid];
409 Real3 c = ni.getCoord();
410 nodes_coords.add(c.x);
411 nodes_coords.add(c.y);
412 nodes_coords.add(c.z);
413 }
414 sbuf.reserveInteger(1); // pour le nombre de faces
415 sbuf.reserveInteger(1); // pour le numéro du sous-domaine
416 sbuf.reserveInteger(1); // pour le nombre de noeuds dans la liste
417 sbuf.reserveArray(unique_ids); // pour le unique id des faces
418 sbuf.reserveArray(cells_unique_ids); // pour le unique id des mailles des faces
419 sbuf.reserveArray(nodes_unique_id); // pour la liste des noeuds
420 sbuf.reserveArray(coords); // pour les coordonnées du centre
421 sbuf.reserveArray(nodes_coords); // pour les coordonnées des noeuds
422
423 sbuf.allocateBuffer();
424 sbuf.setMode(ISerializer::ModePut);
425
426 sbuf.putInteger(nb_to_send);
427 sbuf.putInteger(sub_domain_id);
428 sbuf.putInteger(nodes_unique_id.size());
429 sbuf.putArray(unique_ids);
430 sbuf.putArray(cells_unique_ids);
431 sbuf.putArray(nodes_unique_id);
432 sbuf.putArray(coords);
433 sbuf.putArray(nodes_coords);
434
435 SerializeBuffer recv_buf;
436 pm->allGather(&sbuf, &recv_buf);
437 recv_buf.setMode(ISerializer::ModeGet);
438
439 for (Integer i = 0; i < nb_sub_domain; ++i){
440 Integer nb_face = recv_buf.getInteger();
441 Integer sid = recv_buf.getInteger();
442 Integer nb_node_unique_id = recv_buf.getInteger();
443 info() << " [\t ParallelAMRConsistency::_gatherFaces] READ nface=" << nb_face << " FROM sid=" << sid<<" "<<m_face_info_mng.size();
444 recv_buf.getArray(unique_ids);
445 recv_buf.getArray(cells_unique_ids);
446 recv_buf.getArray(nodes_unique_id);
447 recv_buf.getArray(coords);
448 recv_buf.getArray(nodes_coords);
449
450 // Parcours toutes les faces reçues si certaines sont absentes,
451 // on les ignore.
452 for (Integer z = 0; z < nb_face; ++z){
453 ItemUniqueId new_uid(unique_ids[z]);
454 ItemUniqueId cell_uid(cells_unique_ids[z]);
455
456 Real3 center;
457 center.x = coords[z * 3];
458 center.y = coords[z * 3 + 1];
459 center.z = coords[z * 3 + 2];
460
461 const Int64 current_uid = face_finder.find(center, tol);
462 if ((current_uid != NULL_ITEM_ID) && (new_uid < current_uid)){
463 if (!face_map.hasKey(ItemUniqueId(current_uid))){
464 error() << "face uid not found \n";
465 }
466 //UPDATE FACE_FINDER
467 face_finder.insert(center,new_uid,tol) ;
468 updated_face_uids.insert(current_uid) ;
469
470 FaceInfo& fi_old = face_map[ItemUniqueId(current_uid)];
471 Integer data_index = fi_old.getDataIndex();
472 Integer nb_node = fi_old.nbNode() ;
473 FaceInfo fi(new_uid, cell_uid, nb_node, sid, data_index, &m_face_info_mng);
474 fi.setCenter(center);
475 face_map[ItemUniqueId(current_uid)] = fi;
476 }
477 }
478
479 for (Integer z = 0; z < nb_node_unique_id; ++z){
480 ItemUniqueId nuid(nodes_unique_id[z]);
481 Real3 node_coord;
482 node_coord.x = nodes_coords[z * 3];
483 node_coord.y = nodes_coords[z * 3 + 1];
484 node_coord.z = nodes_coords[z * 3 + 2];
485 Int64 current_node_uid = node_finder.find(node_coord, tol);
486 if ((current_node_uid != NULL_ITEM_ID) && (nuid < current_node_uid)){
487 ItemUniqueId c_nuid(current_node_uid);
488 if(nuid<m_nodes_info[c_nuid].uniqueId()){
489 NodeInfo ni(nuid, sid);
490 ni.setCoord(node_coord);
491 m_nodes_info[c_nuid] = ni;
492
493 // UPDATE NODE FINDER
494 updated_node_uids.insert(current_node_uid) ;
495 }
496 }
497 }
498 }
499 CHECKPERF( m_perf_counter.stop(PerfCounter::GATHERFACE) )
500
501 CHECKPERF( m_perf_counter.start(PerfCounter::UPDATE) )
502 //update node_finder
503 for(ItemUidSet::iterator iter = updated_node_uids.begin();iter!=updated_node_uids.end();++iter){
504 ItemUniqueId uid(*iter);
505 NodeInfo& ni = m_nodes_info[uid] ;
506 node_finder.insert(ni.getCoord(),ni.uniqueId(),tol) ;
507 }
508
509 //update m_face_info_mng
510 _update(m_face_info_mng.m_nodes_unique_id,m_nodes_info) ;
511
512 CHECKPERF( m_perf_counter.stop(PerfCounter::UPDATE) )
513
514 //pm->barrier();
515}

580void ParallelAMRConsistency::
581makeNewItemsConsistent2(MapCoordToUid& node_finder, MapCoordToUid& face_finder)
582{
583 // Marque les noeuds sur la frontière
584 Integer nb_active_face = static_cast<Integer> (m_mesh->nbFace() * 0.2); // 20% de faces shared (sur estimé)
585 m_active_nodes.resize((nb_active_face * 2) + 5);
586 m_active_faces.resize((nb_active_face * 2) + 5);
587
588 FaceFamily* true_face_family = dynamic_cast<FaceFamily*> (m_mesh->itemFamily(IK_Face));
589 if (!true_face_family)
590 throw FatalErrorException(A_FUNCINFO, "can not obtain FaceFamily");
591
592 Int32 sid = m_mesh->parallelMng()->commRank();
593
594
595 typedef std::set<Item> Set;
596 Set active_nodes_set, active_faces_set;
597 // Parcours les faces et marque les noeuds frontieres actives
598 DynamicMesh* mesh = dynamic_cast<DynamicMesh*> (m_mesh);
599 if (!mesh)
600 throw FatalErrorException(A_FUNCINFO, "can not obtain DynamicMesh");
601 ItemInternalMap& faces_map = mesh->facesMap();
602
603 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,faces_map){
604 Face face(nbid->value());
605 bool is_sub_domain_boundary_face = false;
606 if (face.itemBase().flags() & ItemFlags::II_SubDomainBoundary){
607 is_sub_domain_boundary_face = true; // true is not needed
608 }
609 else{
610 if (face.cell(0).level() == 0 && face.cell(1).level() == 0){
611 if ((face.cell(0).owner() != sid || face.cell(1).owner() != sid))
612 is_sub_domain_boundary_face = true;
613 }
614 }
615
616 if (is_sub_domain_boundary_face){
617 UniqueArray<ItemInternal*> subfaces;
618 true_face_family->allSubFaces(face, subfaces);
619 for (Integer s = 0; s < subfaces.size(); s++){
620 Face face2 = subfaces[s];
621 face2.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
622 _addFaceToList2(face2, m_active_faces2);
623 active_faces_set.insert(face2);
624 for ( Node node : face2.nodes() ){
625 node.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
626 active_nodes_set.insert(node);
627 _addNodeToList(node, m_active_nodes);
628 }
629 for ( Edge edge : face2.edges() )
630 edge.mutableItemBase().addFlags(ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary);
631 }
632 }
633 }
634
635 UniqueArray<ItemUniqueId> active_faces_to_send(arcaneCheckArraySize(active_faces_set.size()));
636 UniqueArray<ItemUniqueId> active_nodes_to_send(arcaneCheckArraySize(active_nodes_set.size()));
637
638 UniqueArray<Item> active_faces(arcaneCheckArraySize(active_faces_set.size()));
639 UniqueArray<Item> active_nodes(arcaneCheckArraySize(active_nodes_set.size()));
640
641 Set::const_iterator fit(active_faces_set.begin()), fend(active_faces_set.end());
642 Integer i=0;
643 for (; fit != fend; ++fit){
644 Item face = *fit;
645 active_faces_to_send[i]=(face.uniqueId());
646 active_faces[i]=face;
647 i++;
648 }
649 Set::const_iterator nit(active_nodes_set.begin()), nend(active_nodes_set.end());
650 i=0;
651 for (; nit != nend; ++nit){
652 Item node = *nit;
653 active_nodes_to_send[i]= (node.uniqueId());
654 //debug() << "ACTIVE NODE TO SEND " << node->uniqueId() << " " << active_nodes_to_send[i];
655 active_nodes[i]=node;
656 i++;
657 }
658
659 _gatherItems(active_nodes_to_send, active_faces_to_send, m_active_nodes, m_active_faces2, node_finder, face_finder);
660
661 for (Integer index = 0; index < active_faces.size(); index++){
662 Item face = active_faces[index];
663 const Int64 current_uid = face.uniqueId();
664 FaceInfo2& fi = m_active_faces2[face.uniqueId()];
665 if (current_uid != fi.uniqueId()){
666 face.mutableItemBase().setUniqueId(fi.uniqueId());
667 face.mutableItemBase().setOwner(fi.owner(), sid);
668 //debug() << "[\t ParallelAMRConsistency] NEW FACE BEFORE " << fi.uniqueId() << " " << fi.owner();
669 }
670 }
671 for (Integer index = 0; index < active_nodes.size(); index++){
672 Item node = active_nodes[index];
673 const Int64 current_uid = node.uniqueId();
674 NodeInfo& ni = m_active_nodes[node.uniqueId()];
675 if (current_uid != ni.uniqueId()){
676 node.mutableItemBase().setUniqueId(ni.uniqueId());
677 node.mutableItemBase().setOwner(ni.owner(), sid);
678 //debug() << "[\t ParallelAMRConsistency] NEW FACE BEFORE " << fi.uniqueId() << " " << fi.owner();
679 }
680 }
681
682 // Il faut ranger à nouveau #m_faces_map car les uniqueId() des
683 // faces ont été modifiés
684 faces_map.notifyUniqueIdsChanged();
685 // idem pour les noeuds
686 mesh->nodesMap().notifyUniqueIdsChanged();
687}

701void ParallelAMRConsistency::
702_gatherItems(ConstArrayView<ItemUniqueId> nodes_to_send,
703 ConstArrayView<ItemUniqueId> faces_to_send,
704 NodeInfoList& node_map,
705 FaceInfoMap2& face_map,
706 MapCoordToUid& node_finder,
707 MapCoordToUid& face_finder)
708{
709 IParallelMng* pm = m_mesh->parallelMng();
710 Integer sub_domain_id = pm->commRank();
711 Integer nb_sub_domain = pm->commSize();
712
713 SerializeBuffer sbuf;
714 sbuf.setMode(ISerializer::ModeReserve);
715 Integer nb_to_send = faces_to_send.size();
716 Integer nb_node_to_send = nodes_to_send.size();
717 Int64UniqueArray unique_ids(nb_to_send);
718 Int64UniqueArray node_unique_ids(nb_node_to_send);
719 RealUniqueArray coords;
720 coords.reserve(3 * nb_to_send);
721 RealUniqueArray nodes_coords;
722 nodes_coords.reserve(3 * nb_node_to_send);
723
724 for (Integer i = 0; i < nb_node_to_send; ++i){
725 const NodeInfo& ni = node_map[nodes_to_send[i]];
726 node_unique_ids[i] = ni.uniqueId().asInt64();
727 const Real3 c = ni.getCoord();
728 nodes_coords.add(c.x);
729 nodes_coords.add(c.y);
730 nodes_coords.add(c.z);
731 }
732
733 for (Integer i = 0; i < nb_to_send; ++i){
734 const FaceInfo2& fi = face_map[faces_to_send[i]];
735 unique_ids[i] = fi.uniqueId().asInt64();
736 coords.add(fi.center().x);
737 coords.add(fi.center().y);
738 coords.add(fi.center().z);
739 }
740
741 sbuf.reserveInteger(1); // pour le nombre de faces
742 sbuf.reserveInteger(1); // pour le numéro du sous-domaine
743 sbuf.reserveInteger(1); // pour le nombre de noeuds dans la liste
744 sbuf.reserveArray(unique_ids); // pour le unique id des faces
745 sbuf.reserveArray(node_unique_ids); // pour la liste des noeuds
746 sbuf.reserveArray(coords); // pour les coordonnées du centre
747 sbuf.reserveArray(nodes_coords); // pour les coordonnées des noeuds
748
749 sbuf.allocateBuffer();
750 sbuf.setMode(ISerializer::ModePut);
751
752 sbuf.putInteger(nb_to_send);
753 sbuf.putInteger(sub_domain_id);
754 sbuf.putInteger(node_unique_ids.size());
755 sbuf.putArray(unique_ids);
756 sbuf.putArray(node_unique_ids);
757 sbuf.putArray(coords);
758 sbuf.putArray(nodes_coords);
759
760 SerializeBuffer recv_buf;
761 pm->allGather(&sbuf, &recv_buf);
762 recv_buf.setMode(ISerializer::ModeGet);
763
764 for (Integer i = 0; i < nb_sub_domain; ++i){
765 Integer nb_face = recv_buf.getInteger();
766 Integer sid = recv_buf.getInteger();
767 Integer nb_node_unique_id = recv_buf.getInteger();
768 //info() << " [\t ParallelAMRConsistency::_gatherFaces] READ nface=" << nb_face << " FROM sid=" << sid;
769
770 recv_buf.getArray(unique_ids);
771 recv_buf.getArray(node_unique_ids);
772 recv_buf.getArray(coords);
773 recv_buf.getArray(nodes_coords);
774
775 // Parcours toutes les faces reçues si certaines sont absentes,
776 // on les ignore.
777 const Real tol = 10e-6;
778 for (Integer z = 0; z < nb_face; ++z){
779 ItemUniqueId new_uid(unique_ids[z]);
780
781 Real3 center;
782 center.x = coords[z * 3];
783 center.y = coords[z * 3 + 1];
784 center.z = coords[z * 3 + 2];
785
786 const Int64 current_uid = face_finder.find(center, tol);
787 if ((current_uid != NULL_ITEM_ID) && (new_uid < current_uid)){
788 if (!face_map.hasKey(ItemUniqueId(current_uid))){
789 error() << "face uid not found \n";
790 }
791 FaceInfo2 fi(new_uid, sid);
792 fi.setCenter(center);
793 face_map[ItemUniqueId(current_uid)] = fi;
794 }
795 }
796
797 for (Integer z = 0; z < nb_node_unique_id; ++z){
798 ItemUniqueId nuid(node_unique_ids[z]);
799 Real3 node_coord;
800 node_coord.x = nodes_coords[z * 3];
801 node_coord.y = nodes_coords[z * 3 + 1];
802 node_coord.z = nodes_coords[z * 3 + 2];
803 Int64 current_node_uid = node_finder.find(node_coord, tol);
804 if ((current_node_uid != NULL_ITEM_ID) && (nuid < current_node_uid)){
805 NodeInfo ni(nuid, sid);
806 ni.setCoord(node_coord);
807 ItemUniqueId c_nuid(current_node_uid);
808 node_map[c_nuid] = ni;
809 }
810 }
811 }
812 pm->barrier();
813}
