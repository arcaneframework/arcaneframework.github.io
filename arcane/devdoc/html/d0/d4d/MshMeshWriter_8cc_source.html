94Integer MshMeshWriter::
95_convertToMshType(Int32 arcane_type)
96{
97 switch (arcane_type) {
98 // case (IT_NullType): return MSH_LIN_2; //case (0) is not used
99 case IT_Vertex:
100 return MSH_PNT; //printf("1-node point");
101 case IT_Cell3D_Line2:
102 case IT_Line2:
103 return MSH_LIN_2; //printf("2-node line");
104 case IT_Cell3D_Triangle3:
105 case IT_Triangle3:
106 return MSH_TRI_3; //printf("3-node triangle");
107 case IT_Cell3D_Quad4:
108 case IT_Quad4:
109 return MSH_QUA_4; //printf("4-node quadrangle");
110 case IT_Tetraedron4:
111 return MSH_TET_4; //printf("4-node tetrahedron");
112 case IT_Hexaedron8:
113 return MSH_HEX_8; //printf("8-node hexahedron");
114 case IT_Pentaedron6:
115 return MSH_PRI_6; //printf("6-node prism");
116 case IT_Pyramid5:
117 return MSH_PYR_5; //printf("5-node pyramid");
118 // Beneath, are some meshes that have been tried to match gmsh's ones
119 // Other 5-nodes
120 case IT_Pentagon5:
121 return MSH_PYR_5; // Could use a tag to encode these
122 case IT_HemiHexa5:
123 return MSH_PYR_5;
124 case IT_DiTetra5:
125 return MSH_PYR_5;
126 // Other 6-nodes
127 case IT_Hexagon6:
128 return MSH_PRI_6;
129 case IT_HemiHexa6:
130 return MSH_PRI_6;
131 case IT_AntiWedgeLeft6:
132 return MSH_PRI_6;
133 case IT_AntiWedgeRight6:
134 return MSH_PRI_6;
135 // Other 10-nodes
136 case IT_Heptaedron10:
137 return MSH_TRI_10;
138 // Other 12-nodes
139 case IT_Octaedron12:
140 return MSH_TRI_12;
141 // Others ar still considered as default, rising an exception
142 default:
143 break;
144 }
145 ARCANE_THROW(NotSupportedException, "Arcane type '{0}'", m_item_type_mng->typeFromId(arcane_type));
146}

167bool MshMeshWriter::
168_writeMeshToFileV4(IMesh* mesh, const String& file_name)
169{
170 m_item_type_mng = mesh->itemTypeMng();
171 String mshFileName(file_name + ".msh");
172 std::ofstream ofile(mshFileName.localstr());
173 ofile.precision(20);
174 if (!ofile)
175 throw IOException("VtkMeshIOService::writeMeshToFile(): Unable to open file");
176
177 info() << "[writNodes=" << mesh->nbNode() << " nCells=" << mesh->nbCell();
178
179 ofile << "$MeshFormat\n";
180 // 4.1 pour le format
181 // 0 pour ASCII (1 pour binaire)
182 // 8 pour sizeof(size_t)
183 ofile << "4.1 0 " << sizeof(size_t) << "\n";
184 ofile << "$EndMeshFormat\n";
185
186 IItemFamily* cell_family = mesh->cellFamily();
187 CellGroup all_cells = mesh->allCells();
188 const Int32 mesh_nb_node = mesh->nbNode();
189 IItemFamily* node_family = mesh->nodeFamily();
190 ItemGroup all_nodes = mesh->allNodes();
191
192 // Pour GMSH, il faut commencer par les 'Entities'.
193 // Il faut une entité par type de maille.
194 // On commence donc par calculer les types de mailles.
195 // Pour les maillages non-manifold les mailles peuvent être de dimension
196 // différentes
197 FixedArray<EntityInfo, NB_BASIC_ITEM_TYPE> entities_by_type;
198
199 // Pour GMSH, il faut trier les mailles par leur type (Triangle, Quadrangle, ...)
200 // On fait une entity par type de maille.
201 FixedArray<UniqueArray<Int32>, NB_BASIC_ITEM_TYPE> cells_by_type;
202 ENUMERATE_ (Cell, icell, all_cells) {
203 Cell cell = *icell;
204 Int16 cell_type = cell.type();
205 if (cell_type >= NB_BASIC_ITEM_TYPE || cell_type <= 0)
206 ARCANE_FATAL("Only pre-defined cell type are supported (current cell type is '{0}')",
207 m_item_type_mng->typeFromId(cell_type));
208 cells_by_type[cell_type].add(cell.localId());
209 }
210
211 FixedArray<Int32, 4> nb_entities_by_dim;
212
213 // Conserve les types pré-définis qui ont des éléments
214 UniqueArray<Int32> existing_cells_type;
215 Int64 total_nb_cell = 0;
216 for (Int32 i = 0; i < NB_BASIC_ITEM_TYPE; ++i) {
217 Int64 nb_type = cells_by_type[i].size();
218 if (nb_type > 0)
219 existing_cells_type.add(i);
220 total_nb_cell += nb_type;
221 }
222 Int32 nb_existing_type = existing_cells_type.size();
223 info() << "NbExistingType=" << nb_existing_type;
224 for (Int32 type_index = 0; type_index < nb_existing_type; ++type_index) {
225 Int32 cell_type = existing_cells_type[type_index];
226 ItemTypeInfo* item_type_info = m_item_type_mng->typeFromId(cell_type);
227 Int32 type_dimension = item_type_info->dimension();
228 entities_by_type[cell_type] = EntityInfo{ type_dimension, cell_type, (type_index + 1) * 100 };
229 ++nb_entities_by_dim[type_dimension];
230 }
231
232 // Calcule la bounding box des noeuds
233 VariableNodeReal3 nodes_coords = mesh->nodesCoordinates();
234 Real3 node_min_bounding_box;
235 Real3 node_max_bounding_box;
236 {
237 Real max_value = FloatInfo<Real>::maxValue();
238 Real min_value = -max_value;
239 Real3 min_box(max_value, max_value, max_value);
240 Real3 max_box(min_value, min_value, min_value);
241 ENUMERATE_ (Node, inode, all_nodes) {
242 Real3 pos = nodes_coords[inode];
243 min_box = math::min(min_box, pos);
244 max_box = math::max(max_box, pos);
245 }
246 node_min_bounding_box = min_box;
247 node_max_bounding_box = max_box;
248 }
249
250 // $Entities
251 // numPoints(size_t) numCurves(size_t)
252 // numSurfaces(size_t) numVolumes(size_t)
253 // pointTag(int) X(double) Y(double) Z(double)
254 // numPhysicalTags(size_t) physicalTag(int) ...
255 // ...
256 // curveTag(int) minX(double) minY(double) minZ(double)
257 // maxX(double) maxY(double) maxZ(double)
258 // numPhysicalTags(size_t) physicalTag(int) ...
259 // numBoundingPoints(size_t) pointTag(int; sign encodes orientation) ...
260 // ...
261 // surfaceTag(int) minX(double) minY(double) minZ(double)
262 // maxX(double) maxY(double) maxZ(double)
263 // numPhysicalTags(size_t) physicalTag(int) ...
264 // numBoundingCurves(size_t) curveTag(int; sign encodes orientation) ...
265 // ...
266 // volumeTag(int) minX(double) minY(double) minZ(double)
267 // maxX(double) maxY(double) maxZ(double)
268 // numPhysicalTags(size_t) physicalTag(int) ...
269 // numBoundngSurfaces(size_t) surfaceTag(int; sign encodes orientation) ...
270 // ...
271 // $EndEntities
272 {
273 ofile << "$Entities\n";
274 nb_entities_by_dim[0] = 0;
275 ofile << nb_entities_by_dim[0] << " " << nb_entities_by_dim[1]
276 << " " << nb_entities_by_dim[2] << " " << nb_entities_by_dim[3] << "\n";
277 for (Int32 idim = 1; idim < 4; ++idim) {
278 for (Int32 k = 0; k < NB_BASIC_ITEM_TYPE; ++k) {
279 const EntityInfo& entity_info = entities_by_type[k];
280 if (entity_info.m_dim != idim)
281 continue;
282 ofile << entity_info.m_entity_tag << " " << node_min_bounding_box.x << " " << node_min_bounding_box.y << " " << node_min_bounding_box.z
283 << " " << node_max_bounding_box.x << " " << node_max_bounding_box.y << " " << node_max_bounding_box.z;
284 // Pas de tag pour l'instant
285 ofile << " 0";
286 // Pas de bounding pour l'instant
287 ofile << " 0";
288 ofile << "\n";
289 }
290 }
291 ofile << "$EndEntities\n";
292 }
293
294 // $Nodes
295 // numEntityBlocks(size_t) numNodes(size_t)
296 // minNodeTag(size_t) maxNodeTag(size_t)
297 // entityDim(int) entityTag(int) parametric(int; 0 or 1)
298 // numNodesInBlock(size_t)
299 // nodeTag(size_t)
300 // ...
301 // x(double) y(double) z(double)
302 // < u(double; if parametric and entityDim >= 1) >
303 // < v(double; if parametric and entityDim >= 2) >
304 // < w(double; if parametric and entityDim == 3) >
305 // ...
306 // ...
307 // $EndNodes
308
309 // Bloc contenant les noeuds
310 ofile << "$Nodes\n";
311
312 auto [node_min_uid, node_max_uid] = _getFamilyMinMaxUniqueId(node_family);
313
314 ofile << "1 " << mesh_nb_node << " " << node_min_uid << " " << node_max_uid << "\n";
315 // entityDim(int) entityTag(int) parametric(int; 0 or 1) numNodesInBlock(size_t)
316 ofile << "0 " << "100 " << "0 " << mesh_nb_node << "\n";
317
318 // Sauve les uniqueId()
319 ENUMERATE_ (Node, inode, all_nodes) {
320 Int64 uid = inode->uniqueId();
321 ofile << uid << "\n";
322 }
323
324 // Sauve les coordonnées
325 ENUMERATE_ (Node, inode, all_nodes) {
326 Real3 coord = nodes_coords[inode];
327 ofile << coord.x << " " << coord.y << " " << coord.z << "\n";
328 }
329
330 ofile << "$EndNodes\n";
331
332 auto [cell_min_uid, cell_max_uid] = _getFamilyMinMaxUniqueId(cell_family);
333
334 // $Elements
335 // numEntityBlocks(size_t) numElements(size_t)
336 // minElementTag(size_t) maxElementTag(size_t)
337 // entityDim(int) entityTag(int) elementType(int; see below)
338 // numElementsInBlock(size_t)
339 // elementTag(size_t) nodeTag(size_t) ...
340 // ...
341 // ...
342 // $EndElements
343
344 // Bloc contenant les mailles
345 ofile << "$Elements\n";
346
347 ofile << nb_existing_type << " " << total_nb_cell << " " << cell_min_uid << " " << cell_max_uid << "\n";
348 for (Int32 type_index = 0; type_index < nb_existing_type; ++type_index) {
349 Int32 cell_type = existing_cells_type[type_index];
350 const EntityInfo& entity_info = entities_by_type[cell_type];
351 ConstArrayView<Int32> cells_of_current_type = cells_by_type[cell_type];
352 ItemTypeInfo* item_type_info = m_item_type_mng->typeFromId(cell_type);
353 Int32 type_dimension = item_type_info->dimension();
354 ofile << "\n";
355 ofile << type_dimension << " " << entity_info.m_entity_tag << " " << _convertToMshType(cell_type)
356 << " " << cells_of_current_type.size() << "\n";
357 info() << "Writing cells of type=" << item_type_info->typeName()
358 << " n=" << cells_of_current_type.size()
359 << " dimension=" << type_dimension;
360 Int32 nb_node_for_type = item_type_info->nbLocalNode();
361 ENUMERATE_ (Cell, icell, cell_family->view(cells_of_current_type)) {
362 Cell cell = *icell;
363 ofile << cell.uniqueId();
364 for (Int32 i = 0; i < nb_node_for_type; ++i)
365 ofile << " " << cell.node(i).uniqueId();
366 ofile << "\n";
367 }
368 }
369 ofile << "$EndElements\n";
370
371 return false;
372}

377std::pair<Int64, Int64> MshMeshWriter::
378_getFamilyMinMaxUniqueId(IItemFamily* family)
379{
380 Int64 min_uid = INT64_MAX;
381 Int64 max_uid = -1;
382 ENUMERATE_ (Item, iitem, family->allItems()) {
383 Item item = *iitem;
384 Int64 uid = item.uniqueId();
385 if (uid < min_uid)
386 min_uid = uid;
387 if (uid > max_uid)
388 max_uid = uid;
389 }
390 return { min_uid, max_uid };
391}
