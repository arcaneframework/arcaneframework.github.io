92void ArcaneCaseMeshService::
93createMesh(const String& default_name)
94{
95 if (m_mesh)
96 ARCANE_FATAL("Mesh is already created");
97
98 info() << "Creating mesh from 'ArcaneCaseMeshService'";
99 MeshBuildInfo build_info(default_name);
100 ISubDomain* sd = m_sub_domain;
101 String filename = options()->filename();
102 bool has_filename = options()->filename.isPresent();
103 bool has_generator = options()->generator.isPresent();
104
105 // Il faut spécifier soit le nom du fichier, soit le générateur mais pas les deux
106 if ((has_filename && has_generator) || (!has_filename && !has_generator))
107 ARCANE_FATAL("In '{0}': one and one only of <{1}> or <{2}> has to be specified",
108 options()->configList()->xpathFullName(),
109 options()->generator.rootTagName(),
110 options()->filename.name());
111 if (has_filename){
112 m_mesh_file_name = StringVariableReplace::replaceWithCmdLineArgs(m_sub_domain->applicationInfo().commandLineArguments().parameters(), options()->filename);
113 if (m_mesh_file_name.empty())
114 ARCANE_FATAL("Invalid filename '{0}' in option '{1}'",
115 m_mesh_file_name,options()->filename.xpathFullName());
116 CaseMeshReaderReadInfo read_info;
117 _fillReadInfo(read_info);
118 auto& specific_reader = options()->specificReader;
119 if (specific_reader.isPresent()){
120 m_mesh_builder_ref = specific_reader()->createBuilder(read_info);
121 if (m_mesh_builder_ref.isNull())
122 ARCANE_FATAL("No 'IMeshBuilder' created by specific reader");
123 }
124 else{
125 m_mesh_builder_ref = _createBuilderFromFile(read_info);
126 }
127 m_mesh_builder = m_mesh_builder_ref.get();
128 }
129 else if (has_generator)
130 m_mesh_builder = options()->generator();
131 else
132 ARCANE_FATAL("Invalid operation");
133
134 ARCANE_CHECK_POINTER(m_mesh_builder);
135
136 m_mesh_builder->fillMeshBuildInfo(build_info);
137 // Le générateur peut forcer l'utilisation du partitionnement
138 if (build_info.isNeedPartitioning())
139 m_partitioner_name = options()->partitioner();
140
141 // Positionne avec des valeurs par défaut les champs non remplit.
142 if (build_info.factoryName().empty())
143 build_info.addFactoryName("ArcaneDynamicMeshFactory");
144 if (build_info.parallelMngRef().isNull())
145 build_info.addParallelMng(makeRef(sd->parallelMng()));
146 IPrimaryMesh* pm = sd->meshMng()->meshFactoryMng()->createMesh(build_info);
147 m_mesh = pm;
148}

153void ArcaneCaseMeshService::
154allocateMeshItems()
155{

172{
173 if (!m_mesh)
174 ARCANE_FATAL("Mesh is not created. You should call createMesh() before");
175 if (!m_mesh->isAllocated())
176 ARCANE_FATAL("Mesh is not allocated. You should call initializeMesh() before");
177

179 if (!m_partitioner_name.empty())
180 _doInitialPartition();
181}

186void ArcaneCaseMeshService::
187_fillReadInfo(CaseMeshReaderReadInfo& read_info)
188{
189 read_info.setFileName(m_mesh_file_name);
190
191 {
192 String extension;
193 // Cherche l'extension du fichier et la conserve dans \a case_ext
194 std::string_view fview = m_mesh_file_name.toStdStringView();
195 std::size_t extension_pos = fview.find_last_of('.');
196 if (extension_pos!=std::string_view::npos){
197 fview.remove_prefix(extension_pos+1);
198 extension = fview;
199 }
200 read_info.setFormat(extension);
201 }
202
203 String partitioner_name = options()->partitioner();
204 bool use_internal_partitioner = partitioner_name != "External";
205 if (use_internal_partitioner)
206 m_partitioner_name = partitioner_name;
207
208 info() << "Mesh filename=" << m_mesh_file_name
209 << " extension=" << read_info.format() << " partitioner=" << partitioner_name;
210
211 read_info.setParallelRead(use_internal_partitioner);
212}

217Ref<IMeshBuilder> ArcaneCaseMeshService::
218_createBuilderFromFile(const CaseMeshReaderReadInfo& read_info)
219{
220 // Construit les services potentiels de lecture de maillage. Ce sont
221 // ceux qui implémentent ICaseMeshReader. Construit une instance
222 // de chaque service et appelle la méthode createBuilder(). Dès que l'une
223 // de ces méthodes renvoie une référence non nulle, on l'utilise
224 // pour générer le maillage.
225 ServiceBuilder<ICaseMeshReader> builder(m_sub_domain);
226 UniqueArray<Ref<ICaseMeshReader>> mesh_readers(builder.createAllInstances());
227
228 for( auto& mesh_reader_ref : mesh_readers ){
229 ICaseMeshReader* mesh_reader = mesh_reader_ref.get();
230 Ref<IMeshBuilder> builder = mesh_reader->createBuilder(read_info);
231 if (!builder.isNull())
232 return builder;
233 }
234
235 // Pas de service trouvé pour ce format de fichier.
236 // Affiche la liste des services disponibles et fait un fatal.
237 StringUniqueArray valid_names;
238 builder.getServicesNames(valid_names);
239 String available_readers = String::join(", ",valid_names);
240 ARCANE_FATAL("The mesh reader required for format '{0}' is not available."
241 "The following reader services are available: {1}",
242 read_info.format(),available_readers);
243}

248void ArcaneCaseMeshService::
249_doInitialPartition()
250{
251 // N'utilise plus le service de partitionnement de test pour garantir
252 // avec ParMetis qu'on n'a pas de partitions vides car cela est maintenant
253 // normalement supporté.
254 const bool use_partitioner_tester = false;
255 String test_service = "MeshPartitionerTester";
256 if (use_partitioner_tester) {
257 Int64 nb_cell = m_mesh->nbCell();
258 Int64 min_nb_cell = m_mesh->parallelMng()->reduce(Parallel::ReduceMin,nb_cell);
259 info() << "Min nb cell=" << min_nb_cell;
260 if (min_nb_cell==0)
261 _doInitialPartition2(test_service);
262 else
263 info() << "Mesh name=" << m_mesh->name() << " have cells. Do not use " << test_service;
264 }
265 else
266 info() << "No basic partition first needed";
267 _doInitialPartition2(m_partitioner_name);
268}

273void ArcaneCaseMeshService::
274_doInitialPartition2(const String& partitioner_name)
275{
276 info() << "Doing initial partitioning service=" << partitioner_name;
277 // NOTE: Ce service n'utilise que les partionneurs qui implémentent
278 // IMeshPartitionerBase et pas ceux (historiques) qui n'implémentent que
279 // IMeshPartitioner.
280 ServiceBuilder<IMeshPartitionerBase> sbuilder(m_sub_domain);
281 auto mesh_partitioner = sbuilder.createReference(partitioner_name,m_mesh);
282
283 IMesh* mesh = m_mesh;
284 bool is_dynamic = mesh->isDynamic();
285 mesh->modifier()->setDynamic(true);
286 mesh->utilities()->partitionAndExchangeMeshWithReplication(mesh_partitioner.get(),true);
287 mesh->modifier()->setDynamic(is_dynamic);
288}

293void ArcaneCaseMeshService::
294_initializeVariables()
295{
296 IVariableMng* vm = m_sub_domain->variableMng();
297 const auto& vars_opt = options()->initialization().variable;
298 UniqueArray<String> errors;
299 for( Integer i=0, n=vars_opt.size(); i<n; ++i ){
300 const auto& o = vars_opt[i];
301 String var_name = o.name;
302 String group_name = o.group;
303 String value = o.value;
304 info() << "Initialize variable=" << var_name << " group=" << group_name << " value=" << value;
305 IVariable* var = vm->findMeshVariable(m_mesh,var_name);
306 if (!var){
307 errors.add(String::format("No variable named '{0}' exists",var_name));
308 continue;
309 }
310
311 // Alloue la variable si besoin.
312 if (!var->isUsed())
313 var->setUsed(true);
314 IItemFamily* var_family = var->itemFamily();
315 if (!var_family){
316 errors.add(String::format("Variable '{0}' has no family",var->fullName()));
317 continue;
318 }
319
320 ItemGroup group = var_family->findGroup(group_name);
321 if (group.null()){
322 errors.add(String::format("No group named '{0}' exists in family '{1}'",
323 group_name,var_family->name()));
324 continue;
325 }
326
327 bool ret = var->initialize(group,value);
328 if (ret){
329 errors.add(String::format("Bad value '{0}' for initializing variable '{1}'",
330 value,var->fullName()));
331 continue;
332 }
333 }
334 if (!errors.empty()){
335 for( String s : errors )
336 pinfo() << "ERROR: " << s;
337 ARCANE_FATAL("Variable initialization failed for option '{0}'",
338 vars_opt.xpathFullName());
339 }
340}

345void ArcaneCaseMeshService::
346_setGhostLayerInfos()
347{
348 IGhostLayerMng* gm = m_mesh->ghostLayerMng();
349 if (!gm)
350 return;
351
352 // Positionne les infos sur les mailles fantômes.
353 // TODO Cela est fait pour rester compatible avec le mode historique mais
354 // il faudrait pouvoir gérer cela autrement (via un service par exemple)
355 Integer nb_ghost_layer = options()->nbGhostLayer();
356 if (nb_ghost_layer>=0){
357 info() << "Set number of ghost layers to '" << nb_ghost_layer << "' from caseoption";
358 gm->setNbGhostLayer(nb_ghost_layer);
359 }
360
361 Integer builder_version = options()->ghostLayerBuilderVersion();
362 if (builder_version>=0){
363 info() << "Set ghostlayer builder version to '" << builder_version << "' from caseoption";
364 gm->setBuilderVersion(builder_version);
365 }
366}
