87void DynamicMeshIncrementalBuilder::
88_printCellFaceInfos(ItemInternal* icell,const String& str)
89{
90 Cell cell(icell);
91 Integer nb_num_back_face = 0;
92 Integer nb_num_front_face = 0;
93 Integer index = 0;
94 for( Face face : cell.faces() ){
95 String add_msg;
96 if (face.backCell()==cell){
97 ++nb_num_back_face;
98 add_msg = "is a back face";
99 }
100 else{
101 ++nb_num_front_face;
102 add_msg = "is a front face";
103 }
104 info() << str << ": celluid=" << cell.uniqueId()
105 << " faceuid=" << face.uniqueId()
106 << " faceindex=" << index
107 << add_msg;
108 ++index;
109 }
110 info() << str << ": celluid=" << cell.uniqueId()
111 << " nbback=" << nb_num_back_face
112 << " nbfront=" << nb_num_front_face;
113}

126addCells(Integer nb_cell,Int64ConstArrayView cells_infos,
127 Integer sub_domain_id,Int32ArrayView cells,
128 bool allow_build_face)
129{
130 ItemTypeMng* itm = m_item_type_mng;
131
132 debug() << "[addCells] ADD CELLS mesh=" << m_mesh->name() << " nb=" << nb_cell;
133 Integer cells_infos_index = 0;
134 bool add_to_cells = cells.size()!=0;
135 if (add_to_cells && nb_cell!=cells.size())
136 ARCANE_THROW(ArgumentException,"return array 'cells' has to have same size as number of cells");
137 for( Integer i_cell=0; i_cell<nb_cell; ++i_cell ){
138 ItemTypeId item_type_id { (Int16)cells_infos[cells_infos_index] };
139 ++cells_infos_index;
140 Int64 cell_unique_id = cells_infos[cells_infos_index];

145 Int64ConstArrayView current_cell_nodes_uid(current_cell_nb_node,&cells_infos[cells_infos_index]);

148 allow_build_face);
149
150 if (add_to_cells)
151 cells[i_cell] = cell->localId();
152 cells_infos_index += current_cell_nb_node;
153 }

169void DynamicMeshIncrementalBuilder::
170addCells3(Integer nb_cell,Int64ConstArrayView cells_infos,
171 Integer sub_domain_id,Int32ArrayView cells,
172 bool allow_build_face)
173{
174 bool add_to_cells = cells.size()!=0;
175 if (add_to_cells && nb_cell!=cells.size())
176 throw ArgumentException(A_FUNCINFO,
177 "return array 'cells' has to have same size as number of cells");
178 ItemDataList item_data_list; // store item ids and dependency connectivities
179 ItemDataList item_relation_data_list; // store relation connectivities
180 ItemData& cell_data = item_data_list.itemData(Integer(m_mesh->cellFamily()->itemKind()),
181 nb_cell,0,cells,&mesh()->trueCellFamily(),&mesh()->trueCellFamily(),sub_domain_id);
182 // item info_size cannot be guessed
183 Int64UniqueArray faces_infos, edges_infos, nodes_infos, node_uids;
184 Int32UniqueArray face_lids, edge_lids, node_lids;
185 Integer nb_face = 0 , nb_edge = 0, nb_node = 0;
186 std::map<Int64, Int64SharedArray> cell_to_face_connectivity_info;
187 std::map<std::pair<Int64,Int64>, Int64> edge_uid_map;
188
189 // Fill classical info
190 _fillFaceInfo(nb_face, nb_cell, faces_infos, cells_infos, cell_to_face_connectivity_info);
191 _fillEdgeInfo(nb_edge, nb_face, edges_infos, faces_infos, edge_uid_map);
192 _fillNodeInfo(nb_node, nb_face, nodes_infos, faces_infos);
193
194 // Fill Cell info 2
195 _fillCellNewInfoNew(nb_cell,cells_infos,cell_data.itemInfos(),cell_to_face_connectivity_info, edge_uid_map);
196 // Fill Face info 2
197 if (allow_build_face) {
198 ItemData& face_data = item_data_list.itemData(Integer(m_mesh->faceFamily()->itemKind()),nb_face,0,face_lids,&mesh()->trueFaceFamily(),&mesh()->trueFaceFamily(),sub_domain_id);
199 face_lids.resize(item_data_list[m_mesh->faceFamily()->itemKind()].nbItems());
200 _fillFaceNewInfoNew(nb_face,faces_infos,face_data.itemInfos(),edge_uid_map);// todo voir appel depuis add Faces
201 ItemData& face_relation_data = item_relation_data_list.itemData(Integer(m_mesh->faceFamily()->itemKind()),nb_face,0,face_lids,&mesh()->trueFaceFamily(),&mesh()->trueFaceFamily(),sub_domain_id);
202 _initFaceRelationInfo(face_relation_data,cell_data, faces_infos); // face-cell relations
203 }
204 // Fill Edge info 2
205 if (hasEdge()) {
206 Integer edge_info_size = 1 + nb_edge*6;//New info = Nb_connected_family + for each edge : type, uid, connected_family_kind, nb_node_connected, first_node_uid, second_node_uid
207 ItemData& edge_data = item_data_list.itemData(Integer(m_mesh->edgeFamily()->itemKind()),nb_edge,edge_info_size,edge_lids,&mesh()->trueEdgeFamily(),&mesh()->trueEdgeFamily(),sub_domain_id);
208 edge_lids.resize(item_data_list[m_mesh->edgeFamily()->itemKind()].nbItems());
209 _fillEdgeNewInfoNew(nb_edge,edges_infos,edge_data.itemInfos()); // reprendre fillEdgeInfo et voir l'appel depuis add Edges + attention ordre des noeuds,
210 ItemData& edge_relation_data = item_relation_data_list.itemData(Integer(m_mesh->edgeFamily()->itemKind()),nb_edge,edge_info_size,edge_lids,&mesh()->trueEdgeFamily(),&mesh()->trueEdgeFamily(),sub_domain_id);
211 _initEdgeRelationInfo(edge_relation_data, cell_data, edges_infos); // edge-cell relations
212 if (allow_build_face) {
213 _appendEdgeRelationInfo(edge_relation_data,item_data_list[IK_Face], edges_infos); // edge-face relations
214 _appendFaceRelationInfo(item_relation_data_list[IK_Face],edge_data,faces_infos); // face-edge relations
215 }
216 }
217 // Fill Node info 2
218 Integer node_info_size = 1 + nb_node*2;//New info = Nb_connected_family + for each node : type, uid (no connected family)
219 ItemData& node_data = item_data_list.itemData(Integer(m_mesh->nodeFamily()->itemKind()),nb_node,node_info_size,node_lids,&mesh()->trueNodeFamily(),&mesh()->trueNodeFamily(),sub_domain_id);
220 node_lids.resize(item_data_list[m_mesh->nodeFamily()->itemKind()].nbItems());
221 _fillNodeNewInfo(nodes_infos,node_data.itemInfos());
222 ItemData& node_relation_data = item_relation_data_list.itemData(Integer(m_mesh->nodeFamily()->itemKind()),nb_node,node_info_size,node_lids,&mesh()->trueNodeFamily(),&mesh()->trueNodeFamily(),sub_domain_id);
223 _initNodeRelationInfo(node_relation_data, cell_data, nodes_infos); // node-cell relations
224 if (allow_build_face)_appendNodeRelationInfo(node_relation_data, item_data_list[IK_Face], nodes_infos); // node-face relations
225 if (hasEdge())_appendNodeRelationInfo(node_relation_data, item_data_list[IK_Edge], nodes_infos); // node-edge relations
226
227 // Launch node, face and edges (not mandatory) and cell creation
228 addItems(item_data_list, item_relation_data_list);
229
230}

234void DynamicMeshIncrementalBuilder::
235addCells2(Integer nb_cell,Int64ConstArrayView cells_infos,
236 Integer sub_domain_id,Int32ArrayView cells,
237 bool allow_build_face)
238{
239 ItemDataList item_data_list;
240 ItemData& cell_data = item_data_list.itemData(Integer(m_mesh->cellFamily()->itemKind()),
241 nb_cell,0,cells,&mesh()->trueCellFamily(),&mesh()->trueCellFamily(),sub_domain_id);
242 // item info_size cannot be guessed
243 Int64Array& cell_infos2 = cell_data.itemInfos();
244 Int64UniqueArray faces_infos;
245 Int64UniqueArray node_uids;
246 Int32UniqueArray face_lids;
247 Integer nb_face;
248
249
250 // Fill cell info 2
251 _fillCellInfo2(nb_cell,cells_infos,cell_infos2,nb_face, faces_infos,node_uids,allow_build_face);
252
253 // Fill face info 2
254 if (allow_build_face)
255 {
256 Integer face_infos2_size = faces_infos.size()+ 1 + 2 * nb_face; // additive info nb_connected_family + nb_face * (family_id + nb_connected_element)
257 face_lids.resize(nb_face);
258 ItemData& face_data = item_data_list.itemData(Integer(m_mesh->faceFamily()->itemKind()),
259 nb_face,face_infos2_size,face_lids,&mesh()->trueFaceFamily(),&mesh()->trueFaceFamily(),sub_domain_id);
260 Int64ArrayView face_infos2 = face_data.itemInfos().view();
261 _fillFaceInfo2(nb_face,faces_infos,face_infos2,node_uids); // if allow_build_face false, node_uids built in fillCellInfo3
262 }
263
264 // Fill node info 2
265 Integer nb_node_to_add = node_uids.size();
266 Int32UniqueArray node_lids(nb_node_to_add);
267 ItemData& node_data = item_data_list.itemData(Integer(m_mesh->nodeFamily()->itemKind()),
268 nb_node_to_add,2*nb_node_to_add+1,node_lids,&mesh()->trueNodeFamily(),&mesh()->trueNodeFamily(),sub_domain_id);
269 Int64Array& node_infos2 = node_data.itemInfos();
270 _fillNodeNewInfo(node_uids,node_infos2);
271
272 // Launch node and face and cell creation
273 addItems(item_data_list);
274
275}

280void DynamicMeshIncrementalBuilder::
281_fillFaceInfo(Integer& nb_face, Integer nb_cell,Int64Array& faces_infos, Int64ConstArrayView cells_infos, std::map<Int64,Int64SharedArray>& cell_to_face_connectivity_info)
282{
283 faces_infos.reserve(2*cells_infos.size());
284 Int64UniqueArray work_face_sorted_nodes;
285 Int64UniqueArray work_face_orig_nodes;
286 ItemTypeMng* itm = m_item_type_mng;
287 Integer cells_infos_index = 0;
288 nb_face = 0;
289 NodeInFaceSet face_nodes_set;
290 for (Integer i_cell = 0; i_cell < nb_cell; ++i_cell) {
291 Integer item_type_id = (Integer)cells_infos[cells_infos_index++];
292 Int64 current_cell_unique_id = cells_infos[cells_infos_index++];
293 ItemTypeInfo* it = itm->typeFromId(item_type_id);
294 Integer current_cell_nb_node = it->nbLocalNode();
295 Int64ConstArrayView current_cell_node_uids(current_cell_nb_node,&cells_infos[cells_infos_index]);
296 Integer current_cell_nb_face = it->nbLocalFace();
297 for (Integer face_in_cell_index = 0; face_in_cell_index < current_cell_nb_face; ++face_in_cell_index)
298 {
299 const ItemTypeInfo::LocalFace& current_face = it->localFace(face_in_cell_index);
300 Integer current_face_nb_node = current_face.nbNode();
301 work_face_orig_nodes.resize(current_face_nb_node);
302 work_face_sorted_nodes.resize(current_face_nb_node);
303 Integer i = 0;
304 for (Integer node_in_face_index = 0; node_in_face_index < current_face_nb_node; ++node_in_face_index)
305 {
306 work_face_orig_nodes[i++] = current_cell_node_uids[current_face.node(node_in_face_index)];
307 }
308 mesh_utils::reorderNodesOfFace(work_face_orig_nodes,work_face_sorted_nodes);
309 // get face uid: we shouldn't have to guess, it should be given in arguments...
310 Int64 current_face_uid = _findFaceUniqueId(work_face_sorted_nodes,face_nodes_set);
311 if (current_face_uid == NULL_ITEM_ID) // face not already known add to face info
312 {
313 current_face_uid = m_face_uid_pool++;
314 faces_infos.add(current_face.typeId()); // face info type
315 faces_infos.add(current_face_uid); // face info uid
316 faces_infos.addRange(work_face_sorted_nodes); // face info node uids
317 _addFaceNodes(face_nodes_set,work_face_sorted_nodes,current_face_uid);
318 ++nb_face;
319 }
320 cell_to_face_connectivity_info[current_cell_unique_id].add(current_face_uid);
321 }
322 cells_infos_index += current_cell_nb_node;
323 }
324}

328void DynamicMeshIncrementalBuilder::
329_fillEdgeInfo(Integer& nb_edge, Integer nb_face,Int64Array& edges_infos,
330 Int64ConstArrayView faces_infos,std::map<std::pair<Int64,Int64>, Int64>& edge_uid_map)
331{
332 edges_infos.reserve(2*faces_infos.size());
333 ItemTypeMng* itm = m_item_type_mng;
334 Integer faces_infos_index = 0;
335 nb_edge = 0;
336 for (Integer i_face = 0; i_face < nb_face; ++i_face) {
337 Integer item_type_id = (Integer)faces_infos[faces_infos_index++]; // face type
338 ++faces_infos_index;// face uid (unused)
339 ItemTypeInfo* it = itm->typeFromId(item_type_id);
340 Integer current_face_nb_node = it->nbLocalNode();
341 Int64ConstArrayView current_face_node_uids(current_face_nb_node,&faces_infos[faces_infos_index]);
342 Integer current_face_nb_edge = it->nbLocalEdge();
343 for (Integer edge_in_face_index = 0; edge_in_face_index < current_face_nb_edge; ++edge_in_face_index)
344 {
345 const ItemTypeInfo::LocalEdge& current_edge = it->localEdge(edge_in_face_index);
346 Int64 first_node = current_face_node_uids[current_edge.beginNode()];
347 Int64 second_node = current_face_node_uids[current_edge.endNode()];
348 if (first_node > second_node) std::swap(first_node,second_node);
349 auto edge_it = edge_uid_map.insert(std::make_pair(std::make_pair(first_node,second_node), m_edge_uid_pool));
350 if (edge_it.second)
351 {
352 edges_infos.add(m_edge_uid_pool); // edge uid
353 edges_infos.add(first_node); // first node uid
354 edges_infos.add(second_node); // second node uid
355 ++nb_edge;
356 ++m_edge_uid_pool;
357 }
358 }
359 faces_infos_index += current_face_nb_node;
360 }
361}

407void DynamicMeshIncrementalBuilder::
408_fillCellNewInfoNew(Integer nb_cell,Int64ConstArrayView cells_infos,Int64Array& cell_infos2,
409 const std::map<Int64,Int64SharedArray>& cell_to_face_connectivity_info, const std::map<std::pair<Int64,Int64>, Int64>& edge_uid_map)
410{
411 Integer cell_infos2_size_approx = 1 + 3 * (cells_infos.size()+ 2*nb_cell); // supposes as many faces as nodes as edges...
412 cell_infos2.reserve(cell_infos2_size_approx);
413 ItemTypeMng* itm = m_item_type_mng;
414 Integer cells_infos_index = 0;
415 Integer nb_connected_families = hasEdge() ? 3 : 2;
416 cell_infos2.add(nb_connected_families); // nb_connected_families (node and face +/- edge)
417 for (Integer i_cell = 0; i_cell < nb_cell; ++i_cell) {
418 Integer item_type_id = (Integer)cells_infos[cells_infos_index];
419 cell_infos2.add(cells_infos[cells_infos_index++]); // cell type
420 Int64 current_cell_uid = cells_infos[cells_infos_index];
421 cell_infos2.add(cells_infos[cells_infos_index++]); // cell uid
422 ItemTypeInfo* it = itm->typeFromId(item_type_id);
423 //--- Cell node connectivity
424 cell_infos2.add(mesh()->nodeFamily()->itemKind()); // connected_family_id: node family
425 Integer current_cell_nb_node = it->nbLocalNode();
426 cell_infos2.add(current_cell_nb_node); // nb_connected_nodes
427 Int64ConstArrayView current_cell_node_uids(current_cell_nb_node,&cells_infos[cells_infos_index]);
428 cell_infos2.addRange(current_cell_node_uids); // node ids
429 //--- Cell face connectivity
430 cell_infos2.add(mesh()->faceFamily()->itemKind()); // connected_family_id: face family
431 Integer current_cell_nb_face = it->nbLocalFace();
432 cell_infos2.add(current_cell_nb_face); // nb_connected_faces
433 Int64ArrayView current_cell_faces = cell_to_face_connectivity_info.find(current_cell_uid)->second.view();
434 if (current_cell_nb_face != current_cell_faces.size())
435 ARCANE_FATAL("Incoherent number of faces for cell {0}. Expected {1} found {2}",
436 current_cell_uid,current_cell_nb_face,current_cell_faces.size());
437 cell_infos2.addRange(current_cell_faces); // face ids
438 //-- Cell edge connectivity
439 if (hasEdge())
440 {
441 cell_infos2.add(mesh()->edgeFamily()->itemKind()); // connected_family_id: edge family
442 Integer current_cell_nb_edge = it->nbLocalEdge();
443 cell_infos2.add(current_cell_nb_edge); // nb_connected_edges
444 for (int edge_index = 0; edge_index < current_cell_nb_edge; ++edge_index) {
445 Int64 first_node = current_cell_node_uids[it->localEdge(edge_index).beginNode()];
446 Int64 second_node = current_cell_node_uids[it->localEdge(edge_index).endNode()];
447 if (first_node > second_node) std::swap(first_node,second_node);// edge may be oriented negatively in the cell
448 auto edge_it = edge_uid_map.find(std::make_pair(first_node,second_node));
449 if (edge_it == edge_uid_map.end())
450 ARCANE_FATAL("Do not find edge with nodes {0}-{1} in edge uid map. Exiting",
451 current_cell_node_uids[it->localEdge(edge_index).beginNode()],
452 current_cell_node_uids[it->localEdge(edge_index).endNode()]);
453 cell_infos2.add(edge_it->second);
454 }
455 }
456 cells_infos_index += current_cell_nb_node;
457 }
458}

464void DynamicMeshIncrementalBuilder::
465_fillCellInfo2(Integer nb_cell,Int64ConstArrayView cells_infos,Int64Array& cell_infos2,
466 Integer& nb_face, Int64Array& faces_infos, Int64Array& node_uids, bool allow_build_face)
467{
468 Integer cell_infos2_size_approx = 1 + 2 * (cells_infos.size()+ 2*nb_cell); // supposes as many faces as nodes...
469 cell_infos2.reserve(cell_infos2_size_approx);
470 // Fill infos
471 ItemTypeMng* itm = m_item_type_mng;
472 Integer cells_infos_index = 0;
473 cell_infos2.add(2); // nb_connected_families (node and face)
474
475 nb_face = 0;
476 faces_infos.reserve(2*cells_infos.size());
477 Int64UniqueArray work_face_sorted_nodes;
478 Int64UniqueArray work_face_orig_nodes;
479 NodeInFaceSet face_nodes_set;
480 std::set<Int64> node_uid_set;
481 // Fill face_infos2 from faces_infos
482 for (Integer i_cell = 0; i_cell < nb_cell; ++i_cell) {
483
484 Integer item_type_id = (Integer)cells_infos[cells_infos_index];
485 cell_infos2.add(cells_infos[cells_infos_index++]); // cell type
486 cell_infos2.add(cells_infos[cells_infos_index++]); // cell uid
487 cell_infos2.add(mesh()->nodeFamily()->itemKind()); // connected_family_id: node family
488
489 ItemTypeInfo* it = itm->typeFromId(item_type_id);
490 // Cell node connectivity
491 Integer current_cell_nb_node = it->nbLocalNode();
492 cell_infos2.add(current_cell_nb_node); // nb_connected_nodes
493 Int64ConstArrayView current_cell_node_uids(current_cell_nb_node,&cells_infos[cells_infos_index]);
494 cell_infos2.addRange(current_cell_node_uids); // node ids
495 if (!allow_build_face) node_uid_set.insert(current_cell_node_uids.begin(),current_cell_node_uids.end()); // otherwise node ids will be computed in face info build
496 // cell face connectivity
497 cell_infos2.add(mesh()->faceFamily()->itemKind()); // connected_family_id: face family
498 Integer current_cell_nb_face = it->nbLocalFace();
499 cell_infos2.add(current_cell_nb_face); // nb_connected_faces
500 for (Integer face_in_cell_index = 0; face_in_cell_index < current_cell_nb_face; ++face_in_cell_index)
501 {
502 ItemTypeInfo::LocalFace current_face = it->localFace(face_in_cell_index);
503 Integer current_face_nb_node = current_face.nbNode();
504 work_face_orig_nodes.resize(current_face_nb_node);
505 work_face_sorted_nodes.resize(current_face_nb_node);
506 Integer i = 0;
507 for (Integer node_in_face_index = 0; node_in_face_index < current_face_nb_node; ++node_in_face_index)
508 {
509 work_face_orig_nodes[i++] = current_cell_node_uids[current_face.node(node_in_face_index)];
510 }
511 mesh_utils::reorderNodesOfFace(work_face_orig_nodes,work_face_sorted_nodes);
512 // get face uid: we shouldn't have to guess, it should be given in arguments...
513 Int64 current_face_uid = _findFaceUniqueId(work_face_sorted_nodes,face_nodes_set);
514 if (current_face_uid == NULL_ITEM_ID) // face not already known add to face info
515 {
516 current_face_uid = m_face_uid_pool++;
517 faces_infos.add(current_face.typeId()); // face info type
518 faces_infos.add(current_face_uid); // face info uid
519 faces_infos.addRange(work_face_sorted_nodes); // face info node uids
520 _addFaceNodes(face_nodes_set,work_face_sorted_nodes,current_face_uid);
521 ++nb_face;
522 }
523 cell_infos2.add(current_face_uid); // face ids
524 }
525 cells_infos_index += current_cell_nb_node;
526 }
527 if (! allow_build_face)
528 {
529 node_uids.resize(node_uid_set.size());
530 Integer index = 0;
531 for (Int64 node_uid : node_uid_set) node_uids[index++] = node_uid;
532 }

641void DynamicMeshIncrementalBuilder::
642_fillItemRelationInfo(ItemData& source_item_relation_data, const ItemData& target_item_dependencies_data,
643 Int64ConstArrayView source_item_uids_and_types,
644 bool is_source_item_relation_data_empty)
645{
646 ARCANE_ASSERT((source_item_relation_data.nbItems()*2 == source_item_uids_and_types.size()),
647 ("source item number incoherent between source_item_relation_data and source_item_uids_and_types. Exiting."));
648 // Fill an ItemData containing source to target relations (target to source relations are included in target_item_data)
649 auto & source_relation_info = source_item_relation_data.itemInfos();
650 const auto & target_dependencies_info = target_item_dependencies_data.itemInfos();
651 auto source_family = source_item_relation_data.itemFamily();
652 auto target_family = target_item_dependencies_data.itemFamily();
653 if (! source_family || !target_family) return;
654 std::map<Int64, Int64SharedArray> source_to_target_uids;
655 Integer nb_families_connected_to_target = CheckedConvert::toInteger(target_dependencies_info[0]);
656 Integer target_info_index = 1; // 0 is nb_connected_families
657
658 // Fill map source to target traversing target_item_dependencies_data
659 for (; target_info_index < target_dependencies_info.size();) {
660 target_info_index++; // current target item_type
661 Int64 target_item_uid = target_dependencies_info[target_info_index++];// current target item uid
662 for (Integer family_connected_to_target = 0; family_connected_to_target < nb_families_connected_to_target; ++family_connected_to_target) {
663 Int64 family_connected_to_target_kind = target_dependencies_info[target_info_index++];// current target item connected family kind
664 if (family_connected_to_target_kind != source_family->itemKind()) {//this connection info does not concern source family. Skip
665 Integer nb_non_read_values = CheckedConvert::toInteger(target_dependencies_info[target_info_index++]); // nb_connected_item on this other family (/= to source family)
666 target_info_index+= nb_non_read_values;
667 continue;
668 }
669 else {
670 Int32 nb_source_item_connected_to_target_item = CheckedConvert::toInt32(target_dependencies_info[target_info_index++]);
671 for (Integer source_item_index = 0; source_item_index < nb_source_item_connected_to_target_item; ++source_item_index) {
672 source_to_target_uids[target_dependencies_info[source_item_index+target_info_index]].add(target_item_uid);
673 }
674 target_info_index += nb_source_item_connected_to_target_item;
675 }
676 }
677 }
678 // Fill or append source_item_relation_data traversing the map source_to_target_connectivity
679 if (is_source_item_relation_data_empty) _initEmptyRelationInfo(source_relation_info, source_to_target_uids, source_item_uids_and_types,
680 target_dependencies_info.size(), target_family);
681 else _appendInitializedRelationInfo(source_relation_info, source_to_target_uids, source_item_uids_and_types,
682 target_dependencies_info.size(), target_family);
683
684}

710void DynamicMeshIncrementalBuilder::
711_appendInitializedRelationInfo(Int64Array& source_relation_info, std::map<Int64, Int64SharedArray>& source_to_target_uids, Int64ConstArrayView source_item_uids_and_types,
712 Integer approx_relation_size,
713 IItemFamily const * target_family)
714{
715 // Create a working array to build the consolidated info. This array will be copied in to source_relation_info at the end.
716 Int64UniqueArray source_relation_info_wrk_copy;
717 source_relation_info_wrk_copy.reserve(source_relation_info.size()+approx_relation_size);
718 std::set<Int64> treated_items;// To detect eventual source_items not already present in source_relation_info
719 Integer source_relation_info_index = 0;
720 Integer nb_connected_family = CheckedConvert::toInteger(source_relation_info[source_relation_info_index++]);
721 source_relation_info_wrk_copy.add(nb_connected_family+1); // adding a new connected family
722 for(; source_relation_info_index < source_relation_info.size() ;){
723 source_relation_info_wrk_copy.add(source_relation_info[source_relation_info_index++]); // item type
724 Int64 source_item_uid = source_relation_info[source_relation_info_index++]; // get item uid
725 source_relation_info_wrk_copy.add(source_item_uid);
726 treated_items.insert(source_item_uid);
727 for (Integer connected_family_index = 0; connected_family_index < nb_connected_family; ++connected_family_index) {
728 source_relation_info_wrk_copy.add(source_relation_info[source_relation_info_index++]); // family kind
729 Integer nb_connected_elements = CheckedConvert::toInteger(source_relation_info[source_relation_info_index++]); // nb connected elements
730 source_relation_info_wrk_copy.add(nb_connected_elements);
731 source_relation_info_wrk_copy.addRange(source_relation_info.subConstView(source_relation_info_index,nb_connected_elements));
732 source_relation_info_index += nb_connected_elements;
733 }
734 // Add new connection info
735 source_relation_info_wrk_copy.add(target_family->itemKind()); // family kind
736 const auto& connected_items = source_to_target_uids[source_item_uid]; // get connected items
737 source_relation_info_wrk_copy.add(connected_items.size()); // nb connected elements
738 source_relation_info_wrk_copy.addRange(connected_items);
739 }
740 // Find eventual source_items not already present in source_relation_info : todo check this can happen
741 for (Integer source_item_uids_and_types_index = 0; source_item_uids_and_types_index < source_item_uids_and_types.size(); source_item_uids_and_types_index+=2) {
742 Int64 source_item_uid = source_item_uids_and_types[source_item_uids_and_types_index];
743 if (treated_items.find(source_item_uid) == treated_items.end()){
744 source_relation_info_wrk_copy.add(source_item_uids_and_types[source_item_uids_and_types_index+1]); // item type
745 source_relation_info_wrk_copy.add(source_item_uid); // item uid
746 source_relation_info_wrk_copy.add(target_family->itemKind()); // family kind
747 const auto& connected_items = source_to_target_uids[source_item_uid]; // get connected items
748 source_relation_info_wrk_copy.add(connected_items.size()); // nb connected elements
749 source_relation_info_wrk_copy.addRange(connected_items);
750 // Padd this item info with family info : we added only one family, add void values for the other nb_connected_family families
751 for (Integer connected_family_index = 0; connected_family_index < nb_connected_family; ++connected_family_index) {
752 source_relation_info_wrk_copy.add(0); // family id (does not matter)
753 source_relation_info_wrk_copy.add(0); // nb_connected_elements for this family
754 }
755 }
756 }
757 // Copy working array into destination array
758 source_relation_info.resize(source_relation_info_wrk_copy.size());
759 std::copy(source_relation_info_wrk_copy.begin(), source_relation_info_wrk_copy.end(), source_relation_info.begin());
760}

1679void DynamicMeshIncrementalBuilder::
1680removeNeedRemoveMarkedItems()
1681{
1682 if(!m_mesh->useMeshItemFamilyDependencies())
1683 {
1684
1685 IItemFamily* links_family = m_mesh->findItemFamily(IK_DoF, GraphDoFs::linkFamilyName(), false, false);
1686 if(links_family)
1687 {
1688 info() << "Remove Items for family : "<<links_family->name() ;
1689 links_family->removeNeedRemoveMarkedItems() ;
1690 }
1691
1692 for( IItemFamily* family : m_mesh->itemFamilies() )
1693 {
1694 if (family->itemKind()!=IK_DoF || family->name()==GraphDoFs::linkFamilyName())
1695 continue;
1696 info() << "Remove Items for family : "<<family->name() ;
1697 family->removeNeedRemoveMarkedItems() ;
1698 }
1699 // Supression des particules de la famille Particle
1700
1701 for( IItemFamilyCollection::Enumerator i(m_mesh->itemFamilies()); ++i; ){
1702 IItemFamily* family = *i;
1703 if (family->itemKind()==IK_Particle){
1704 ParticleFamily* particle_family = dynamic_cast<ParticleFamily*>(family) ;
1705 if(particle_family && particle_family->getEnableGhostItems()==true){
1706 UniqueArray<Integer> lids_to_remove ;
1707 lids_to_remove.reserve(1000);
1708
1709 ItemInternalMap& particle_map = particle_family->itemsMap();
1710 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,particle_map){
1711 ItemInternal* item = nbid->value();
1712 Integer f = item->flags();
1713 if (f & ItemFlags::II_NeedRemove){
1714 f &= ~ItemFlags::II_NeedRemove;
1715 item->setFlags(f);
1716 lids_to_remove.add(item->localId());
1717 }
1718 }
1719
1720 info() << "Number of particles of family "<<family->name()<<" to remove: " << lids_to_remove.size();
1721 if(lids_to_remove.size()>0)
1722 particle_family->removeParticles(lids_to_remove) ;
1723 }
1724 }
1725 }
1726
1727 // Suppression des mailles
1728 CellFamily& cell_family = m_mesh->trueCellFamily();
1729
1730 SharedArray<ItemInternal*> cells_to_remove = _removeNeedRemoveMarkedItems(cell_family.itemsMap());
1731
1732 info() << "Number of cells to remove: " << cells_to_remove.size();
1733
1734#ifdef ARCANE_DEBUG_DYNAMIC_MESH
1735 for( Integer i=0, is=cells_to_remove.size(); i<is; ++i )
1736 info() << "remove cell with uid=" << cells_to_remove[i]->uniqueId()
1737 << ",lid=" << cells_to_remove[i]->localId()
1738 << ",owner=" << cells_to_remove[i]->owner();
1739#endif /* ARCANE_DEBUG_DYNAMIC_MESH */
1740
1741 for( Integer i=0, is=cells_to_remove.size(); i<is; ++i )
1742 cell_family.removeCell(cells_to_remove[i]);
1743 }
1744 // With ItemFamilyNetwork
1745 else
1746 {
1747 // handle families already in the network
1748 m_mesh->itemFamilyNetwork()->schedule([](IItemFamily* family){
1749 family->removeNeedRemoveMarkedItems();
1750 // Remove items from family graph leaves to root
1751 }, IItemFamilyNetwork::InverseTopologicalOrder);
1752 for (auto family : m_mesh->itemFamilies()) {
1753 // handle remaining families (particles, links, dual nodes & dof).
1754 // Will have no effect on the families already handled in the graph (since NeedRemove flag is reset)
1755 family->removeNeedRemoveMarkedItems();
1756 }
1757 }
1758}
