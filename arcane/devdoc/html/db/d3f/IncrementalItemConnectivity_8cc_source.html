103class IncrementalItemConnectivityContainer
104{
105 public:
106 struct Idx
107 {
109 Int32 nb;
111 Int32 index;
112
113 Idx(Int32 n,Int32 i) : nb(n), index(i){}
114 };
115 public:
116 IncrementalItemConnectivityContainer(IMesh* mesh,const String& var_name)
117 : m_var_name(var_name),
118 m_connectivity_nb_item_variable(VariableBuildInfo(mesh,var_name+"Nb",IVariable::PPrivate)),
119 m_connectivity_index_variable(VariableBuildInfo(mesh,var_name+"Index",IVariable::PPrivate)),
120 m_connectivity_list_variable(VariableBuildInfo(mesh,var_name+"List",IVariable::PPrivate)),
121 m_connectivity_nb_item_array(m_connectivity_nb_item_variable._internalTrueData()->_internalDeprecatedValue()),
122 m_connectivity_index_array(m_connectivity_index_variable._internalTrueData()->_internalDeprecatedValue()),
123 m_connectivity_list_array(m_connectivity_list_variable._internalTrueData()->_internalDeprecatedValue())
124 {
125 // Ajoute un tag pour indiquer que ce sont des variables associées à la connectivité.
126 // Pour l'instant cela n'est utilisé que pour les statistiques d'affichage.
127
128 String tag_name = "ArcaneConnectivity";
129 m_connectivity_nb_item_variable.addTag(tag_name,"1");
130 m_connectivity_index_variable.addTag(tag_name,"1");
131 m_connectivity_list_variable.addTag(tag_name,"1");
132 }
133
134 String m_var_name;
135
136 VariableArrayInt32 m_connectivity_nb_item_variable;
137 VariableArrayInt32 m_connectivity_index_variable;
138 VariableArrayInt32 m_connectivity_list_variable;
139
140 VariableArrayInt32::ContainerType& m_connectivity_nb_item_array;
141 VariableArrayInt32::ContainerType& m_connectivity_index_array;
142 VariableArrayInt32::ContainerType& m_connectivity_list_array;
143
144 ObserverPool m_observers;
145
153 Int32 m_max_nb_item = 0;
154
155 public:
156
157 Integer size() const { return m_connectivity_nb_item_array.size(); }
158
159 bool isAllocated() const { return size()>0; }
160
161 void _checkResize(Int32 lid)
162 {
163 //TODO: réutiliser le code de ItemFamily::_setUniqueId().
164 Integer size = m_connectivity_nb_item_array.size();
165 Integer wanted_size = lid + 1;
166 if (wanted_size<size)
167 return;
168 Integer capacity = m_connectivity_nb_item_array.capacity();
169 if (wanted_size<capacity){
170 // Pas besoin d'augmenter la capacité.
171 }
172 else{
173 Integer reserve_size = 1000;
174 while (lid>reserve_size){
175 reserve_size *= 2;
176 }
177 m_connectivity_nb_item_array.reserve(reserve_size);
178 m_connectivity_index_array.reserve(reserve_size);
179 }
180 m_connectivity_nb_item_array.resize(wanted_size);
181 m_connectivity_index_array.resize(wanted_size);
182 }
183
184 void reserveForItems(Int32 capacity)
185 {
186 m_connectivity_nb_item_array.reserve(capacity);
187 m_connectivity_index_array.reserve(capacity);
188 }
189
190 public:
191
192};

197IncrementalItemConnectivityBase::
198IncrementalItemConnectivityBase(IItemFamily* source_family,IItemFamily* target_family,
199 const String& aname)
200: AbstractIncrementalItemConnectivity(source_family,target_family,aname)
201, m_item_connectivity_list(nullptr)
202, m_item_connectivity_index(-1)
203{
204 StringBuilder var_name("Connectivity");
205 var_name += aname;
206 var_name += source_family->name();
207 var_name += target_family->name();
208
209 IMesh* mesh = source_family->mesh();
210 m_p = new IncrementalItemConnectivityContainer(mesh,var_name);
211
212 typedef IncrementalItemConnectivityBase ThatClass;
213 // Récupère les évènements de lecture pour indiquer qu'il faut mettre
214 // à jour les vues.
215 m_p->m_observers.addObserver(this,&ThatClass::_notifyConnectivityNbItemChangedFromObservable,
216 m_p->m_connectivity_nb_item_variable.variable()->readObservable());
217
218 m_p->m_observers.addObserver(this,&ThatClass::_notifyConnectivityIndexChanged,
219 m_p->m_connectivity_index_variable.variable()->readObservable());
220
221 m_p->m_observers.addObserver(this,&ThatClass::_notifyConnectivityListChanged,
222 m_p->m_connectivity_list_variable.variable()->readObservable());
223
224 // Met à jour les vues à partir des tableaux associées.
225 // Il faut le faire dès que la taille d'un tableau change car alors
226 // il peut être réalloué et donc la vue associée devenir invalide.
227 _notifyConnectivityListChanged();
228 _notifyConnectivityIndexChanged();
229 _notifyConnectivityNbItemChangedFromObservable();
230}

244void IncrementalItemConnectivityBase::
245reserveMemoryForNbSourceItems(Int32 n, bool pre_alloc_connectivity)
246{
247 if (n<=0)
248 return;
249
250 m_p->reserveForItems(n);
251 _notifyConnectivityIndexChanged();
252 _notifyConnectivityNbItemChanged();
253
254 if (pre_alloc_connectivity){
255 Int32 pre_alloc_size = preAllocatedSize();
256 if (pre_alloc_size>0){
257 m_p->m_connectivity_list_array.reserve(n * pre_alloc_size);
258 _notifyConnectivityListChanged();
259 }
260 }

522
523void IncrementalItemConnectivity::
524_increaseIndexList(Int32 lid,Integer size,Int32 target_lid)
525{
526 Integer added_range = (m_pre_allocated_size>0) ? m_pre_allocated_size : 1;
527 ++m_nb_memcopy;
528 Integer pos_in_index = m_connectivity_index[lid];
529 Integer new_pos_in_list = _increaseConnectivityList(NULL_ITEM_LOCAL_ID,size+added_range);
530 ArrayView<Int32> current_list(size,&(m_connectivity_list[pos_in_index]));
531 ArrayView<Int32> new_list(size+1,&(m_connectivity_list[new_pos_in_list]));
532 new_list.copy(current_list);
533 // Ajoute la nouvelle entité à la fin de la liste des connectivités
534 // TODO: regarder pour le tri dans l'ordre des uid() croissant.
535 new_list[size] = target_lid;
536 m_connectivity_index[lid] = new_pos_in_list;
537}

541
542void IncrementalItemConnectivity::
543addConnectedItem(ItemLocalId source_item,ItemLocalId target_item)
544{
545 ++m_nb_add;
546 const Int32 lid = source_item.localId();
547 const Int32 target_lid = target_item.localId();
548 Integer size = m_connectivity_nb_item[lid];
549 // Ajoute une entité connectée.
550 // Pour l'instant, le fonctionnement est basique.
551 // On ajoute toujours les entités à la fin de m_p->m_connectivity_list.
552 // S'il n'y en a pas, il suffit d'ajouter à la fin.
553 // S'il y en a déjà, il faut allouer à la fin de
554 // m_p->m_connectivity_list assez \a size+1 éléments et
555 // on recopie la précédente connectivité dans le nouvel emplacement.
556 // Forcément, avec le temps la liste va toujours grossir
557 // car les trous ne sont pas réutilisés.
558 if (m_pre_allocated_size!=0){
559 // En cas de préallocation, on alloue par bloc de taille 'm_pre_allocated_size'.
560 // Il faut donc réallouer si la taille est un multiple de m_pre_allocated_size
561 if (size==0){
562 Integer new_pos_in_list = _increaseConnectivityList(NULL_ITEM_LOCAL_ID,m_pre_allocated_size);
563 m_connectivity_index[lid] = new_pos_in_list;
564 m_connectivity_list[new_pos_in_list] = target_lid;
565 }
566 else{
567 if (size<m_pre_allocated_size || (size%m_pre_allocated_size)!=0){
568 Integer index = m_connectivity_index[lid];
569 m_connectivity_list[index+size] = target_lid;
570 }
571 else{
572 _increaseIndexList(lid,size,target_lid);
573 }
574 }
575 }
576 else{
577 if (size==0){
578 Integer new_pos_in_list = _increaseConnectivityList(target_lid);
579 m_connectivity_index[lid] = new_pos_in_list;
580 }
581 else{
582 _increaseIndexList(lid,size,target_lid);
583 }
584 }
585 ++(m_connectivity_nb_item[lid]);
586 _setNewMaxNbConnectedItems(m_connectivity_nb_item[lid]);
587}

589/*---------------------------------------------------------------------------*/
590/*---------------------------------------------------------------------------*/
591
592Integer IncrementalItemConnectivity::
593_computeAllocSize(Integer nb_item)
594{
595 if (m_pre_allocated_size!=0){
596 // Alloue un multiple de \a m_pre_allocated_size
597 Integer alloc_size = nb_item / m_pre_allocated_size;
598 if (alloc_size==0)
599 return m_pre_allocated_size;
600 if ((nb_item%m_pre_allocated_size)==0)
601 return nb_item;
602 return m_pre_allocated_size * (alloc_size + 1);
603 }
604 return nb_item;
605}

612{
613 const Int32 lid = source_item.localId();
614 Integer size = m_connectivity_nb_item[lid];
615 if (size!=0)
616 ARCANE_FATAL("source_item already have connected items");
617 Integer alloc_size = _computeAllocSize(nb_item);
618 Integer new_pos_in_list = _increaseConnectivityList(NULL_ITEM_LOCAL_ID,alloc_size);
619 m_connectivity_index[lid] = new_pos_in_list;
620 m_connectivity_nb_item[lid] += nb_item;
621 _setNewMaxNbConnectedItems(m_connectivity_nb_item[lid]);
622}

717/*---------------------------------------------------------------------------*/
718
719void IncrementalItemConnectivity::
720setPreAllocatedSize(Integer prealloc_size)
721{
722 if (m_pre_allocated_size<0)
723 throw ArgumentException(A_FUNCINFO,
724 String::format("Invalid prealloc_size v={0}",
725 prealloc_size));
726
727 // Ne fait rien rien si on a déjà alloué des entités sinon cela rendrait
728 // incohérent les allocations.
729 // NOTE: on pourrait autoriser cela mais cela nécessiterait de reconstruire
730 // les indices des connectivités. A priori un appel à compactConnectivityList()
731 // suffirait.

771{

787/*---------------------------------------------------------------------------*/
799void IncrementalItemConnectivity::
800compactConnectivityList()
801{
802 info(4) << "Begin Compacting IncrementalItemConnectivity name=" << name()
803 << " new_size=" << m_connectivity_list.size()
804 << " prealloc_size=" << m_pre_allocated_size;
805 // TODO: essayer de trouver un moyen de ne faire le compactage que si
806 // cela est nécessaire. Une facon serait de compter le nombre d'appel à
807 // _increaseIndexList() depuis le dernier compactage.
808 UniqueArray<Int32> old_connectivity_list(m_connectivity_list);
809 Integer old_size = old_connectivity_list.size();
810 Integer nb_item = m_connectivity_nb_item.size();
811 m_p->m_connectivity_list_array.clear();
812 _notifyConnectivityListChanged();
813 _checkAddNullItem();
814 Integer new_pos_in_list = m_p->m_connectivity_list_array.size();
815 Int32 pre_allocated_size = m_pre_allocated_size;
816 for( Integer i=0; i<nb_item; ++i ){
817 Int32 lid = i;
818 Int32 nb = m_connectivity_nb_item[lid];
819 if (nb==0){
820 m_connectivity_index[lid] = 0;
821 continue;
822 }
823 Int32 index = m_connectivity_index[lid];
824 Int32ConstArrayView con_list(nb,old_connectivity_list.data()+index);
825 Integer alloc_size = _computeAllocSize(nb);
826 m_connectivity_index[lid] = new_pos_in_list;
827 new_pos_in_list += alloc_size;
828 //info() << "NEW_POS_IN_LIST=" << new_pos_in_list << " nb=" << nb << " alloc_size=" << alloc_size;
829 // Vérifie que la position est bien un multiple de pre_allocated_size.
830 if (pre_allocated_size!=0){
831 Int32 pos_modulo = new_pos_in_list % pre_allocated_size;
832 if (pos_modulo!=0)
833 ARCANE_FATAL("Bad position i={0} pos={1} pre_alloc_size={2} modulo={3}",
834 i,new_pos_in_list,pre_allocated_size,pos_modulo);
835 }
836 m_p->m_connectivity_list_array.addRange(con_list);
837 // Si préallocation, complète le reste des éléments avec l'entité nulle..
838 if (alloc_size!=nb)
839 m_p->m_connectivity_list_array.addRange(NULL_ITEM_LOCAL_ID,alloc_size-nb);
840 if (m_pre_allocated_size==0 && nb==0)
841 m_connectivity_index[lid] = 0;
842 }
843 _notifyConnectivityListChanged();
844 _computeMaxNbConnectedItem();
845 info(4) << "Compacting IncrementalItemConnectivity name=" << name()
846 << " nb_item=" << nb_item << " old_size=" << old_size
847 << " new_size=" << m_connectivity_list.size()
848 << " prealloc_size=" << m_pre_allocated_size;
849}
