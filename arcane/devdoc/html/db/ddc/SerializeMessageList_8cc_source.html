56void SerializeMessageList::
57processPendingMessages()
58{
59 for( BasicSerializeMessage* sm : m_messages_to_process ){
60 PointToPointMessageInfo message_info(buildMessageInfo(sm));
61 if (sm->destination().isNull() && !m_allow_any_rank_receive){
62 // Il faudra faire un probe pour ce message
63 m_messages_to_probe.add({sm,message_info});
64 }
65 else
66 _addMessage(sm,message_info);
67 sm->setIsProcessed(true);
68 }
69 m_messages_to_process.clear();
70}

75Integer SerializeMessageList::
76waitMessages(eWaitType wait_type)
77{
78 processPendingMessages();
79 // NOTE: il faudrait peut-être faire aussi faire des probe() dans l'appel
80 // à _waitMessages() car il est possible que tous les messages n'aient pas
81 // été posté. Dans ce cas, il faudrait passer en mode non bloquant tant
82 // qu'il y a des probe à faire
83 while(!m_messages_to_probe.empty())
84 _doProbe();
85
86 return _waitMessages(wait_type);
87}

92void SerializeMessageList::
93_doProbe()
94{
95 // Il faut tester avec probe() si des messages sont disponibles
96 for( ProbeInfo& p : m_messages_to_probe ){
97 //tm->info() << "CHECK PROBE msg=" << p.m_message_info << " is_done?=" << p.m_is_probe_done;
98 // Ne devrait pas être 'vrai' mais par sécurité on fait le test.
99 if (p.m_is_probe_done)
100 continue;
101 MessageId message_id = mpProbe(m_message_passing_mng,p.m_message_info);
102 if (message_id.isValid()){
103 //tm->info() << "FOUND PROBE message_id=" << message_id;
104 PointToPointMessageInfo message_info(message_id,NonBlocking);
105 _addMessage(p.m_serialize_message,message_info);
106 p.m_is_probe_done = true;
107 }
108 }
109
110 // Supprime les probes qui sont terminés.
111 auto k = std::remove_if(m_messages_to_probe.begin(),m_messages_to_probe.end(),
112 [](const ProbeInfo& p) { return p.m_is_probe_done; });
113 m_messages_to_probe.resize(k-m_messages_to_probe.begin());
114}

119Integer SerializeMessageList::
120_waitMessages(eWaitType wait_type)
121{
122 TimeMetricSentry tphase(m_message_passing_phase);
123
124 if (wait_type==WaitAll){
125 m_request_list->wait(WaitAll);
126 // Indique que les messages sont bien terminés
127 for( ISerializeMessage* sm : m_messages_serialize )
128 sm->setFinished(true);
129 m_request_list->clear();
130 m_messages_serialize.clear();
131 return (-1);
132 }
133
134 if (wait_type==WaitSome || wait_type==TestSome){
135 Integer nb_request = m_request_list->size();
136 m_request_list->wait(wait_type);
137 m_remaining_serialize_messages.clear();
138 Integer nb_done = 0;
139 for( Integer i=0; i<nb_request; ++i ){
140 BasicSerializeMessage* sm = m_messages_serialize[i];
141 if (m_request_list->isRequestDone(i)){
142 ++nb_done;
143 sm->setFinished(true);
144 }
145 else{
146 m_remaining_serialize_messages.add(sm);
147 }
148 }
149 m_request_list->removeDoneRequests();
150 m_messages_serialize = m_remaining_serialize_messages;
151 if (nb_done==nb_request)
152 return (-1);
153 return nb_done;
154 }
155
156 ARCCORE_THROW(NotImplementedException,"waitMessage with wait_type=={0}",(int)wait_type);
157}

162void SerializeMessageList::
163_addMessage(BasicSerializeMessage* sm,const PointToPointMessageInfo& message_info)
164{
165 Request r;
166 ISerializer* s = sm->serializer();
167 if (sm->isSend())
168 r = mpSend(m_message_passing_mng,s,message_info);
169 else
170 r = mpReceive(m_message_passing_mng,s,message_info);
171 m_request_list->add(r);
172 m_messages_serialize.add(sm);
173}

178PointToPointMessageInfo SerializeMessageList::
179buildMessageInfo(ISerializeMessage* sm)
180{
181 MessageId message_id(sm->_internalMessageId());
182 if (message_id.isValid()){
183 PointToPointMessageInfo message_info(message_id,NonBlocking);
184 message_info.setEmiterRank(sm->source());
185 return message_info;
186 }
187 return { sm->source(), sm->destination(), sm->internalTag(), NonBlocking };
188}
