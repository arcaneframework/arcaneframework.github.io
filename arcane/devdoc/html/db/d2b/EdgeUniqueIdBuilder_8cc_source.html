62void EdgeUniqueIdBuilder::
63computeEdgesUniqueIds()
64{
65 double begin_time = platform::getRealTime();
66
67 Int32 edge_version = m_mesh->meshUniqueIdMng()->edgeBuilderVersion();
68
69 info() << "Using version=" << edge_version << " to compute edges unique ids"
70 << " mesh=" << m_mesh->name();
71
72 if (edge_version==1)
73 _computeEdgesUniqueIdsParallel3();
74 else if (edge_version==2)
75 _computeEdgesUniqueIdsParallelV2();
76 else if (edge_version==0)
77 info() << "No renumbering for edges";
78 else
79 ARCANE_FATAL("Invalid valid version '{0}'. Valid values are 0, 1 or 2");
80
81 double end_time = platform::getRealTime();
82 Real diff = (Real)(end_time - begin_time);
83 info() << "TIME to compute edge unique ids=" << diff;
84
85 ItemInternalMap& edges_map = m_mesh->edgesMap();
86
87 // Il faut ranger à nouveau #m_edges_map car les uniqueId() des
88 // edges ont été modifiés
89 edges_map.notifyUniqueIdsChanged();
90
91 if (m_mesh_builder->isVerbose()){
92 info() << "NEW EDGES_MAP after re-indexing";
93 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,edges_map){
94 ItemInternal* edge = nbid->value();
95 info() << "Edge uid=" << edge->uniqueId() << " lid=" << edge->localId();
96 }
97 }
98}

144void EdgeUniqueIdBuilder::
145_exchangeData(IParallelExchanger* exchanger,BoundaryInfosMap& boundary_infos_to_send)
146{
147 for( BoundaryInfosMapEnumerator i_map(boundary_infos_to_send); ++i_map; ){
148 Int32 sd = i_map.data()->key();
149 exchanger->addSender(sd);
150 }
151 exchanger->initializeCommunicationsMessages();
152 {
153 for( Integer i=0, ns=exchanger->nbSender(); i<ns; ++i ){
154 ISerializeMessage* sm = exchanger->messageToSend(i);
155 Int32 rank = sm->destination().value();
156 ISerializer* s = sm->serializer();
157 Int64ConstArrayView infos = boundary_infos_to_send[rank];
158 Integer nb_info = infos.size();
159 s->setMode(ISerializer::ModeReserve);
160 s->reserve(DT_Int64,1); // Pour le nombre d'elements
161 s->reserveSpan(DT_Int64,nb_info); // Pour les elements
162 s->allocateBuffer();
163 s->setMode(ISerializer::ModePut);
164 //info() << " SEND1 rank=" << rank << " nb_info=" << nb_info;
165 s->putInt64(nb_info);
166 s->putSpan(infos);
167 }
168 }
169 exchanger->processExchange();
170 debug() << "END EXCHANGE";
171}

235 //Integer nb_local_cell = m_mesh_builder->nbCell();

241 IntegerUniqueArray edges_opposite_cell_owner(nb_local_edge);
242
243 // Pour vérification, s'assure que tous les éléments de ce tableau
244 // sont valides, ce qui signifie que toutes les edges ont bien été
245 // renumérotés
246 Int64UniqueArray edges_new_uid(nb_local_edge);
247 edges_new_uid.fill(NULL_ITEM_ID);
248
249 //UniqueArray<T_EdgeInfo> edges_local_infos(m_mesh_nb_edge);
250
251 //Integer nb_recv_sub_domain_boundary_edge = 0;
252
253 Int64UniqueArray edges_infos;
254 edges_infos.reserve(10000);
255 //ItemInternalMap& cells_map = m_mesh->cellsMap();
256 ItemInternalMap& edges_map = m_mesh->edgesMap();
257 ItemInternalMap& faces_map = m_mesh->facesMap(); // utilisé pour détecter le bord
258 ItemInternalMap& nodes_map = m_mesh->nodesMap();
259
260 // NOTE: ce tableau n'est pas utile sur toutes les mailles. Il
261 // suffit qu'il contienne les mailles dont on a besoin, c'est à dire
262 // les notres + celles connectées à une de nos edges.
263 HashTableMapT<Int32,Int32> cell_first_edge_uid(m_mesh_builder->oneMeshItemAdder()->nbCell()*2,true);
264 //OLD Int64UniqueArray cell_first_edge_uid(global_max_cell_uid+1);
265 //OLD cell_first_edge_uid.fill(0);
266
267 // Rassemble les données des autres processeurs dans recv_cells;
268 // Pour éviter que les tableaux ne soient trop gros, on procède en plusieurs
269 // étapes.
270 // Chaque sous-domaine construit sa liste de edges frontières, avec pour
271 // chaque edge:
272 // - son type
273 // - la liste de ses noeuds,
274 // - le numéro unique de sa maille
275 // - le propriétaire de sa maille
276 // - son indice dans sa maille
277 // Cette liste sera ensuite envoyée à tous les sous-domaines.
278 ItemTypeMng* itm = m_mesh->itemTypeMng();
279
280 // Détermine le unique id max des noeuds
281 Int64 my_max_node_uid = NULL_ITEM_UNIQUE_ID;
282 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,nodes_map){
283 ItemInternal* node = nbid->value();
284 Int64 node_uid = node->uniqueId();
285 if (node_uid>my_max_node_uid)
286 my_max_node_uid = node_uid;
287 }
288 Int64 global_max_node_uid = pm->reduce(Parallel::ReduceMax,my_max_node_uid);
289 debug() << "NODE_UID_INFO: MY_MAX_UID=" << my_max_node_uid
290 << " GLOBAL=" << global_max_node_uid;
291
292 //TODO: choisir bonne valeur pour initialiser la table
293 BoundaryInfosMap boundary_infos_to_send(nb_rank,true);
294 NodeUidToSubDomain uid_to_subdomain_converter(global_max_node_uid,nb_rank);
295
296 HashTableMapT<Int64,SharedArray<Int64> > nodes_info(100000,true);
297 IItemFamily* node_family = m_mesh->nodeFamily();
298 UniqueArray<bool> is_boundary_nodes(node_family->maxLocalId(),false);
299
300 // Marque tous les noeuds frontieres car ce sont ceux qu'il faudra envoyer
301 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,faces_map){
302 Face face = nbid->value();
303 Integer face_nb_cell = face.nbCell();
304 if (face_nb_cell==1){
305 for( Int32 ilid : face.nodeIds() )
306 is_boundary_nodes[ilid] = true;
307 }
308 }
309
310 // Détermine la liste des edges frontières
311 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,edges_map){
312 Edge edge = nbid->value();
313 Node first_node = edge.node(0);
314 Int64 first_node_uid = first_node.uniqueId();
315 SharedArray<Int64> v;
316 Int32 dest_rank = -1;
317 if (!is_boundary_nodes[first_node.localId()]){
318 v = nodes_info.lookupAdd(first_node_uid)->value();
319 }
320 else{
321 dest_rank = uid_to_subdomain_converter.uidToRank(first_node_uid);
322 v = boundary_infos_to_send.lookupAdd(dest_rank)->value();
323 }
324 v.add(first_node_uid); // 0
325 v.add(my_rank); // 1
326 v.add(edge.uniqueId()); // 2
327 v.add(edge.type()); // 3
328 v.add(NULL_ITEM_UNIQUE_ID); // 4 : only used for debug
329 v.add(NULL_ITEM_UNIQUE_ID); // 5 : only used for debug
330 if (is_verbose)
331 info() << "Edge uid=" << edge.uniqueId() << " n0,n1=" << edge.node(0).uniqueId() << "," << edge.node(1).uniqueId()
332 << " n0=" << ItemPrinter(edge.node(0)) << " n1=" << ItemPrinter(edge.node(1)) << " dest_rank=" << dest_rank;
333 for( Node edge_node : edge.nodes() )
334 v.add(edge_node.uniqueId());
335 }
336
337 // Positionne la liste des envoies
338 Ref<IParallelExchanger> exchanger{ParallelMngUtils::createExchangerRef(pm)};
339 _exchangeData(exchanger.get(),boundary_infos_to_send);
340
341 {
342 Integer nb_receiver = exchanger->nbReceiver();
343 debug() << "NB RECEIVER=" << nb_receiver;
344 SharedArray<Int64> received_infos;
345 for( Integer i=0; i<nb_receiver; ++i ){
346 ISerializeMessage* sm = exchanger->messageToReceive(i);
347 //Int32 orig_rank = sm->destSubDomain();
348 ISerializer* s = sm->serializer();
349 s->setMode(ISerializer::ModeGet);
350 Int64 nb_info = s->getInt64();
351 //info() << "RECEIVE NB_INFO=" << nb_info << " from=" << orig_rank;
352 received_infos.resize(nb_info);
353 s->getSpan(received_infos);
354 //if ((nb_info % 3)!=0)
355 //fatal() << "info size can not be divided by 3";
356 Integer z =0;
357 while(z<nb_info){
358 Int64 node_uid = received_infos[z+0];
359 //Int64 sender_rank = received_infos[z+1];
360 //Int64 edge_uid = received_infos[z+2];
361 Int32 edge_type = (Int32)received_infos[z+3];
362 // received_infos[z+4];
363 // received_infos[z+5];
364 ItemTypeInfo* itt = itm->typeFromId(edge_type);
365 Integer edge_nb_node = itt->nbLocalNode();
366 Int64Array& a = nodes_info.lookupAdd(node_uid)->value();
367 a.addRange(Int64ConstArrayView(6+edge_nb_node,&received_infos[z]));
368 z += 6;
369 z += edge_nb_node;
370 //info() << "NODE UID=" << node_uid << " sender=" << sender_rank
371 // << " edge_uid=" << edge_uid;
372 //node_cell_list.add(node_uid,cell_uid,cell_owner);
373 //HashTableMapT<Int64,Int32>::Data* v = nodes_nb_cell.lookupAdd(node_uid);
374 //++v->value();
375 }
376 }
377 Integer my_max_edge_node = 0;
378 for( HashTableMapT<Int64,SharedArray<Int64> >::Enumerator inode(nodes_info); ++inode; ){
379 //Int64 key = inode.data()->key();
380 Int64ConstArrayView a = *inode;
381 //info() << "A key=" << key << " size=" << a.size();
382 Integer nb_info = a.size();
383 Integer z = 0;
384 Integer node_nb_edge = 0;
385 while(z<nb_info){
386 ++node_nb_edge;
387 //Int64 node_uid = a[z+0];
388 //Int64 sender_rank = a[z+1];
389 //Int64 edge_uid = a[z+2];
390 Int32 edge_type = (Int32)a[z+3];
391 // a[z+4];
392 // a[z+5];
393 ItemTypeInfo* itt = itm->typeFromId(edge_type);
394 Integer edge_nb_node = itt->nbLocalNode();
395 /*info() << "NODE2 UID=" << node_uid << " sender=" << sender_rank
396 << " edge_uid=" << edge_uid */
397 //for( Integer y=0; y<edge_nb_node; ++y )
398 //info() << "Nodes = i="<< y << " " << a[z+6+y];
399 z += 6;
400 z += edge_nb_node;
401 }
402 my_max_edge_node = math::max(node_nb_edge,my_max_edge_node);
403 }
404 Integer global_max_edge_node = pm->reduce(Parallel::ReduceMax,my_max_edge_node);
405 debug() << "GLOBAL MAX EDGE NODE=" << global_max_edge_node;
406 // OK, maintenant donne comme uid de la edge (node_uid * global_max_edge_node + index)
407 IntegerUniqueArray indexes;
408 boundary_infos_to_send = BoundaryInfosMap(nb_rank,true);
409
410 for( HashTableMapT<Int64,SharedArray<Int64> >::Enumerator inode(nodes_info); ++inode; ){
411 //Int64 key = inode.data()->key();
412 Int64ConstArrayView a = *inode;
413 //info() << "A key=" << key << " size=" << a.size();
414 Integer nb_info = a.size();
415 Integer z = 0;
416 Integer node_nb_edge = 0;
417 indexes.clear();
418 while(z<nb_info){
419 Int64 node_uid = a[z+0];
420 Int32 sender_rank = (Int32)a[z+1];
421 Int64 edge_uid = a[z+2];
422 Int32 edge_type = (Int32)a[z+3];
423 // a[z+4];
424 // a[z+5];
425 ItemTypeInfo* itt = itm->typeFromId(edge_type);
426 Integer edge_nb_node = itt->nbLocalNode();
427
428 // Regarde si la edge est déjà dans la liste:
429 Integer edge_index = node_nb_edge;
430 Int32 edge_new_owner = sender_rank;
431 for( Integer y=0; y<node_nb_edge; ++y ){
432 if (memcmp(&a[indexes[y]+6],&a[z+6],sizeof(Int64)*edge_nb_node)==0){
433 edge_index = y;
434 edge_new_owner = (Int32)a[indexes[y]+1];
435 }
436 }
437 Int64 edge_new_uid = (node_uid * global_max_edge_node) + edge_index;
438 Int64Array& v = boundary_infos_to_send.lookupAdd(sender_rank)->value();
439 // Indique au propriétaire de cette edge son nouvel uid
440 v.add(edge_uid);
441 v.add(edge_new_uid);
442 v.add(edge_new_owner);
443 indexes.add(z);
444 z += 6;
445 z += edge_nb_node;
446 /*info() << "NODE3 UID=" << node_uid << " sender=" << sender_rank
447 << " edge_uid=" << edge_uid
448 << " edge_index=" << edge_index;*/
449 ++node_nb_edge;
450 }
451 my_max_edge_node = math::max(node_nb_edge,my_max_edge_node);
452 }
453 }
454 exchanger = ParallelMngUtils::createExchangerRef(pm);
455
456 _exchangeData(exchanger.get(),boundary_infos_to_send);
457 {
458 Integer nb_receiver = exchanger->nbReceiver();
459 debug() << "NB RECEIVER=" << nb_receiver;
460 Int64UniqueArray received_infos;
461 for( Integer i=0; i<nb_receiver; ++i ){
462 ISerializeMessage* sm = exchanger->messageToReceive(i);
463 auto orig_rank = sm->destination();
464 ISerializer* s = sm->serializer();
465 s->setMode(ISerializer::ModeGet);
466 Int64 nb_info = s->getInt64();
467 if (is_verbose)
468 info() << "RECEIVE NB_INFO=" << nb_info << " from=" << orig_rank;
469 received_infos.resize(nb_info);
470 s->getSpan(received_infos);
471 if ((nb_info % 3)!=0)
472 ARCANE_FATAL("info size can not be divided by 3 x={0}",nb_info);
473 Int64 nb_item = nb_info / 3;
474 for (Int64 z=0; z<nb_item; ++z ){
475 Int64 old_uid = received_infos[(z*3)];
476 Int64 new_uid = received_infos[(z*3)+1];
477 Int32 new_owner = (Int32)received_infos[(z*3)+2];
478 //info() << "EDGE old_uid=" << old_uid << " new_uid=" << new_uid;
479 ItemInternalMapData* edge_data = edges_map.lookup(old_uid);
480 if (!edge_data)
481 ARCANE_FATAL("Can not find own edge uid={0}",old_uid);
482 ItemInternal* iedge = edge_data->value();
483 Edge edge{iedge};
484 iedge->setUniqueId(new_uid);
485 iedge->setOwner(new_owner,my_rank);
486 if (is_verbose)
487 info() << "SetEdgeOwner uid=" << new_uid << " owner=" << new_owner
488 << " n0,n1=" << edge.node(0).uniqueId() << "," << edge.node(1).uniqueId()
489 << " n0=" << ItemPrinter(edge.node(0)) << " n1=" << ItemPrinter(edge.node(1));
490 }
491 }
492 }
493
494 traceMng()->flush();
495 pm->barrier();
496 info() << "END OF TEST NEW EDGE COMPUTE";
497 return;
498}

507void EdgeUniqueIdBuilder::
508_computeEdgesUniqueIdsSequential()
509{
510 bool is_verbose = m_mesh_builder->isVerbose();
511 Integer nb_cell = m_mesh_builder->oneMeshItemAdder()->nbCell();
512
513 ItemInternalMap& cells_map = m_mesh->cellsMap();
514
515 // En séquentiel, les uniqueId() des mailles ne peuvent dépasser la
516 // taille des Integers même en 32bits.
517 Int32 max_uid = 0;
518 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
519 ItemInternal* cell = nbid->value();
520 Int32 cell_uid = cell->uniqueId().asInt32();
521 if (cell_uid>max_uid)
522 max_uid = cell_uid;
523 }
524 info() << "Max uid=" << max_uid;
525 Int32UniqueArray cell_first_edge_uid(max_uid+1);
526 Int32UniqueArray cell_nb_num_back_edge(max_uid+1);
527 Int32UniqueArray cell_true_boundary_edge(max_uid+1);
528
529 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
530 Cell cell = nbid->value();
531 Int32 cell_uid = cell.uniqueId().asInt32();
532 Integer nb_num_back_edge = 0;
533 Integer nb_true_boundary_edge = 0;
534 for( Edge edge : cell.edges()){
535 if (edge.itemBase().backCell()==cell)
536 ++nb_num_back_edge;
537 else if (edge.nbCell()==1){
538 ++nb_true_boundary_edge;
539 }
540 }
541 cell_nb_num_back_edge[cell_uid] = nb_num_back_edge;
542 cell_true_boundary_edge[cell_uid] = nb_true_boundary_edge;
543 }
544
545 Integer current_edge_uid = 0;
546 for( Integer i=0; i<nb_cell; ++i ){
547 cell_first_edge_uid[i] = current_edge_uid;
548 current_edge_uid += cell_nb_num_back_edge[i] + cell_true_boundary_edge[i];
549 }
550
551 if (is_verbose){
552 for( Integer i=0; i<nb_cell; ++i ){
553 info() << "Recv: Cell EdgeInfo celluid=" << i
554 << " firstedgeuid=" << cell_first_edge_uid[i]
555 << " nbback=" << cell_nb_num_back_edge[i]
556 << " nbbound=" << cell_true_boundary_edge[i];
557 }
558 }
559
560 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
561 Cell cell = nbid->value();
562 Int32 cell_uid = cell.uniqueId().asInt32();
563 Integer nb_num_back_edge = 0;
564 Integer nb_true_boundary_edge = 0;
565 for( Edge edge : cell.edges() ){
566 Int64 edge_new_uid = NULL_ITEM_UNIQUE_ID;
567 //info() << "CHECK CELLUID=" << cell_uid << " EDGELID=" << edge->localId();
568 if (edge.itemBase().backCell()==cell){
569 edge_new_uid = cell_first_edge_uid[cell_uid] + nb_num_back_edge;
570 ++nb_num_back_edge;
571 }
572 else if (edge.nbCell()==1){
573 edge_new_uid = cell_first_edge_uid[cell_uid] + cell_nb_num_back_edge[cell_uid] + nb_true_boundary_edge;
574 ++nb_true_boundary_edge;
575 }
576 if (edge_new_uid!=NULL_ITEM_UNIQUE_ID){
577 //info() << "NEW EDGE UID: LID=" << edge->localId() << " OLDUID=" << edge->uniqueId()
578 //<< " NEWUID=" << edge_new_uid << " THIS=" << edge;
579 edge.mutableItemBase().setUniqueId(edge_new_uid);
580 }
581 }
582 }
583
584 if (is_verbose){
585 OStringStream ostr;
586 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
587 Cell cell = nbid->value();
588 Int32 cell_uid = cell.uniqueId().asInt32();
589 Integer index = 0;
590 for( Edge edge : cell.edges() ){
591 Int64 opposite_cell_uid = NULL_ITEM_UNIQUE_ID;
592 bool true_boundary = false;
593 bool internal_other = false;
594 if (edge.itemBase().backCell()==cell){
595 }
596 else if (edge.nbCell()==1){
597 true_boundary = true;
598 }
599 else{
600 internal_other = true;
601 opposite_cell_uid = edge.itemBase().backCell().uniqueId().asInt64();
602 }
603 ostr() << "NEW LOCAL ID FOR CELLEDGE " << cell_uid << ' '
604 << index << ' ' << edge.uniqueId() << " (";
605 for( Node node : edge.nodes() ){
606 ostr() << ' ' << node.uniqueId();
607 }
608 ostr() << ")";
609 if (internal_other)
610 ostr() << " internal-other";
611 if (true_boundary)
612 ostr() << " true-boundary";
613 if (opposite_cell_uid!=NULL_ITEM_ID)
614 ostr() << " opposite " << opposite_cell_uid;
615 ostr() << '\n';
616 ++index;
617 }
618 }
619 info() << ostr.str();
620 }

628{
629 // Positionne les uniqueId() des arêtes de manière très simple.
630 // Si le maximum des uniqueId() des noeuds est MAX_NODE_UID, alors
631 // le uniqueId() d'une arête est :
632 //
633 // node(0).uniqueId() * MAX_NODE_UID + node(1).uniqueId()
634 //
635 // Cela ne fonctionne que si MAX_NODE_UID est inférieur à 2^31.
636
637 IParallelMng* pm = m_mesh->parallelMng();
638
639 ItemInternalMap& nodes_map = m_mesh->nodesMap();
640 ItemInternalMap& edges_map = m_mesh->edgesMap();
641
642 Int64 max_uid = 0;
643 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,nodes_map){
644 ItemInternal* node = nbid->value();
645 if (node->uniqueId()>max_uid)
646 max_uid = node->uniqueId();
647 }
648 Int64 total_max_uid = pm->reduce(Parallel::ReduceMax,max_uid);
649 if (total_max_uid>INT32_MAX)
650 ARCANE_FATAL("Max uniqueId() for node is too big v={0} max_allowed={1}",total_max_uid,INT32_MAX);
651
652 ENUMERATE_ITEM_INTERNAL_MAP_DATA(ebid,edges_map){
653 Edge edge{ebid->value()};
654 Node node0{edge.node(0)};
655 Node node1{edge.node(1)};
656 Int64 new_uid = (node0.uniqueId().asInt64() * total_max_uid) + node1.uniqueId().asInt64();
657 edge.mutableItemBase().setUniqueId(new_uid);
658 }
659}
