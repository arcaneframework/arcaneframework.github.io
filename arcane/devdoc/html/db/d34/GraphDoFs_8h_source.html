46 : public IGraphConnectivity
47 {
48  public:
49  GraphIncrementalConnectivity(IItemFamily const* dualnode_family,
50  IItemFamily const* link_family,
51  Arcane::mesh::IncrementalItemConnectivity* link_connectivity,
52  UniqueArray<Arcane::mesh::IncrementalItemConnectivity*> const& dualitem_connectivities,
53  ItemScalarProperty<Integer> const& dualnode_to_connectivity)
54  : m_dualnode_family(dualnode_family)
55  , m_link_family(link_family)
56  , m_link_connectivity(link_connectivity)
57  , m_dualitem_connectivities(dualitem_connectivities)
58  , m_dualnode_to_connectivity_index(dualnode_to_connectivity)
59  , m_link_connectivity_accessor(link_connectivity->connectivityAccessor())
60  {
61  m_dualitem_connectivity_accessors.resize(m_dualitem_connectivities.size());
62  for (Integer i = 0; i < m_dualitem_connectivities.size(); ++i) {
63  if (m_dualitem_connectivities[i]) {
64  m_dualitem_connectivity_accessors[i] = m_dualitem_connectivities[i]->connectivityAccessor();
65  }
66  }
67  }

71  , m_link_family(rhs.m_link_family)
72  , m_link_connectivity(rhs.m_link_connectivity)
73  , m_dualitem_connectivities(rhs.m_dualitem_connectivities)
74  , m_dualnode_to_connectivity_index(rhs.m_dualnode_to_connectivity_index)
