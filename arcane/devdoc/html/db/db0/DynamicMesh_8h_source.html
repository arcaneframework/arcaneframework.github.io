621  // Add a dependency (downward adjacencies only) between two family: ie the source family

623  template <class SourceFamily, class TargetFamily>
624  void _addDependency(SourceFamily* source_family, TargetFamily* target_family)
625  {
626  typedef typename NewWithLegacyConnectivityType<SourceFamily,TargetFamily>::type CType;
627  String name = String::format("{0}{1}",source_family->name(),target_family->name());
