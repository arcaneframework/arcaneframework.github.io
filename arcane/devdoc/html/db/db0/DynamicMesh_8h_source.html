688 // Add a dependency (downward adjacencies only) between two family: ie the source family
689 // is built on the target family (ex a cell is build owns its nodes)
690 template <class SourceFamily, class TargetFamily>
691 void _addDependency(SourceFamily* source_family, TargetFamily* target_family)
692 {
693 typedef typename NewWithLegacyConnectivityType<SourceFamily,TargetFamily>::type CType;
694 String name = String::format("{0}{1}",source_family->name(),target_family->name());

696 m_item_family_network->addDependency(source_family,target_family,connectivity);
