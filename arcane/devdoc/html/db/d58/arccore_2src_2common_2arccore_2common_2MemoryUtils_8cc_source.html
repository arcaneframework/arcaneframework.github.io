99eMemoryResource MemoryUtils::
100getMemoryResourceFromName(const String& name)
101{
102 eMemoryResource v = eMemoryResource::Unknown;
103 if (name.null())
104 return v;
105 if (name == "Device")
106 v = eMemoryResource::Device;
107 else if (name == "Host")
108 v = eMemoryResource::Host;
109 else if (name == "HostPinned")
110 v = eMemoryResource::HostPinned;
111 else if (name == "UnifiedMemory")
112 v = eMemoryResource::UnifiedMemory;
113 else
114 ARCCORE_FATAL("Invalid name '{0}' for memory resource. Valid names are "
115 "'Device', 'Host', 'HostPinned' or 'UnifieMemory'.",
116 name);
117 return v;
118}

256void MemoryUtils::
257copyHost(MutableMemoryView destination, ConstMemoryView source)
258{
259 auto b_source = source.bytes();
260 auto b_destination = destination.bytes();
261 Int64 source_size = b_source.size();
262 if (source_size == 0)
263 return;
264 Int64 destination_size = b_destination.size();
265 if (source_size > destination_size)
266 ARCCORE_FATAL("Destination is too small source_size={0} destination_size={1}",
267 source_size, destination_size);
268 auto* destination_data = b_destination.data();
269 auto* source_data = b_source.data();
270 ARCCORE_CHECK_POINTER(destination_data);
271 ARCCORE_CHECK_POINTER(source_data);
272 std::memmove(destination_data, source_data, source_size);
273}

288void MemoryUtils::
289copyWithIndexedDestination(MutableMemoryView destination, ConstMemoryView source,
290 SmallSpan<const Int32> indexes, RunQueue* queue)
291{
292
293 Int32 one_data_size = _checkDataTypeSize(A_FUNCINFO, destination.datatypeSize(), source.datatypeSize());
294
295 Int64 nb_index = indexes.size();
296 if (nb_index == 0)
297 return;
298
299 auto b_source = source.bytes();
300 auto b_destination = destination.bytes();
301
302 _getDefaultCopyList(queue)->copyFrom(one_data_size, { indexes, b_source, b_destination, queue });
303}
