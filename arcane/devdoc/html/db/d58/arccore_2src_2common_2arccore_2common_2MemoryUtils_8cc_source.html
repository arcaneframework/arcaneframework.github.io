84eMemoryResource MemoryUtils::
85getMemoryResourceFromName(const String& name)
86{
87 eMemoryResource v = eMemoryResource::Unknown;
88 if (name.null())
89 return v;
90 if (name == "Device")
91 v = eMemoryResource::Device;
92 else if (name == "Host")
93 v = eMemoryResource::Host;
94 else if (name == "HostPinned")
95 v = eMemoryResource::HostPinned;
96 else if (name == "UnifiedMemory")
97 v = eMemoryResource::UnifiedMemory;
98 else
99 ARCCORE_FATAL("Invalid name '{0}' for memory resource. Valid names are "
100 "'Device', 'Host', 'HostPinned' or 'UnifieMemory'.",
101 name);
102 return v;
103}

239void MemoryUtils::
240copyHost(MutableMemoryView destination, ConstMemoryView source)
241{
242 auto b_source = source.bytes();
243 auto b_destination = destination.bytes();
244 Int64 source_size = b_source.size();
245 if (source_size == 0)
246 return;
247 Int64 destination_size = b_destination.size();
248 if (source_size > destination_size)
249 ARCCORE_FATAL("Destination is too small source_size={0} destination_size={1}",
250 source_size, destination_size);
251 auto* destination_data = b_destination.data();
252 auto* source_data = b_source.data();
253 ARCCORE_CHECK_POINTER(destination_data);
254 ARCCORE_CHECK_POINTER(source_data);
255 std::memmove(destination_data, source_data, source_size);
256}

273 SmallSpan<const Int32> indexes, RunQueue* queue)
274{
275
276 Int32 one_data_size = _checkDataTypeSize(A_FUNCINFO, destination.datatypeSize(), source.datatypeSize());

284
