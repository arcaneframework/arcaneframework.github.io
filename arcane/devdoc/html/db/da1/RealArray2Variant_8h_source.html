134  operator NumArray<Real,MDDim2,LayoutType>() const
135  {
136  NumArray<Real,MDDim2> v(m_nb_dim1,m_nb_dim2);
137  for( Integer i=0, m=m_nb_dim1; i<m; ++i )
138  for( Integer j=0, n=m_nb_dim2; j<n; ++j )
139  v(i,j) = m_value[i][j];
140  return v;
141  }
142 
143  private:

146  Int32 m_nb_dim1 = 0;

148 
149  private:
150 
151  void _setValue(const Real* v, Integer nb_dim1, Integer nb_dim2)
152  {
153  m_nb_dim1 = nb_dim1;
154  ARCANE_ASSERT(nb_dim1 <= MAX_DIM1_SIZE, ("Dim1 size too large"));
155  m_nb_dim2 = nb_dim2;
156  ARCANE_ASSERT(nb_dim2 <= MAX_DIM2_SIZE, ("Dim2 size too large"));
157  for (Integer i = 0 ; i < nb_dim1; ++i)
158  for (Integer j = 0 ; j < nb_dim2; ++j)
159  m_value[i][j] = v[i * nb_dim2 + j];
160  }
