31String VariableTypeInfo::
32_buildFullTypeName() const
33{
34 StringBuilder full_type_b;
35 full_type_b = dataTypeName(dataType());
36 full_type_b += ".";
37 full_type_b += itemKindName(itemKind());
38 full_type_b += ".";
39 full_type_b += dimension();
40 full_type_b += ".";
41 full_type_b += multiTag();
42 if (isPartial())

45}
