219void AMRCartesianMeshTesterModule::
220init()
221{
222 info() << "AMR Init";
223
224 IMesh* mesh = defaultMesh();
225
226 IItemFamily* cell_family = mesh->cellFamily();
227 Int32UniqueArray ids(1);
228 ids[0] = 0;
229 cell_family->createGroup("CELL0",ids,true);
230 ids[0] = 1;
231 cell_family->createGroup("CELL1",ids,true);
232 ids[0] = 2;
233 cell_family->createGroup("CELL2",ids,true);
234 IItemFamily* face_family = defaultMesh()->faceFamily();
235 ids[0] = 0;
236 face_family->createGroup("FACE0",ids,true);
237 ids[0] = 1;
238 face_family->createGroup("FACE1",ids,true);
239 ids[0] = 2;
240 face_family->createGroup("FACE2",ids,true);
241 ids[0] = 3;
242 face_family->createGroup("FACE3",ids,true);
243 ids[0] = 4;
244 face_family->createGroup("FACE4",ids,true);
245 ids[0] = 5;
246 face_family->createGroup("FACE5",ids,true);
247
248 m_cartesian_mesh = ICartesianMesh::getReference(mesh);
249 m_utils = makeRef(new CartesianMeshTestUtils(m_cartesian_mesh,acceleratorMng()));
250
251 if (!subDomain()->isContinue())
252 _initAMR();
253
254 _computeCenters();
255
256 const bool do_coarse_at_init = options()->coarseAtInit();
257
258 const Integer dimension = defaultMesh()->dimension();
259 if (dimension==2)
260 m_nb_expected_patch = 1 + options()->refinement2d().size();
261 else if (dimension==3)
262 m_nb_expected_patch = 1 + options()->refinement3d().size();
263
264 // Si on dé-raffine à l'init, on aura un patch de plus
265 if (do_coarse_at_init)
266 ++m_nb_expected_patch;
267
268 if (subDomain()->isContinue())
269 m_cartesian_mesh->recreateFromDump();
270 else{
271 m_cartesian_mesh->computeDirections();
272 CartesianMeshRenumberingInfo renumbering_info;
273 renumbering_info.setRenumberPatchMethod(options()->renumberPatchMethod());
274 renumbering_info.setSortAfterRenumbering(true);
275 if (options()->coarseAtInit())
276 renumbering_info.setParentPatch(m_cartesian_mesh->amrPatch(1));
277 m_cartesian_mesh->renumberItemsUniqueId(renumbering_info);
278 _checkUniqueIds();
279 _processPatches();
280 info() << "MaxUid for mesh=" << MeshUtils::getMaxItemUniqueIdCollective(m_cartesian_mesh->mesh());
281 }
282
283 // Initialise la densité.
284 // On met une densité de 1.0 à l'intérieur
285 // et on ajoute une densité de 5.0 pour chaque direction dans les
286 // mailles de bord.
287 m_density.fill(1.0);
288 for( Integer idir=0, nb_dir=dimension; idir<nb_dir; ++idir){
289 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
290 Integer nb_boundary1 = 0;
291 Integer nb_boundary2 = 0;
292 ENUMERATE_CELL(icell,cdm.innerCells()){
293 DirCell cc(cdm.cell(*icell));
294 Cell next = cc.next();
295 Cell prev = cc.previous();
296 if (next.null() || prev.null()){
297 // Maille au bord. J'ajoute de la densité.
298 // Ne devrait pas arriver car on est sur les innerCells()
299 ++nb_boundary1;
300 m_density[icell] += 5.0;
301 }
302 }
303 // Parcours les mailles frontières pour la direction
304 ENUMERATE_CELL(icell,cdm.outerCells()){
305 DirCell cc(cdm[icell]);
306 if (icell.index()<5)
307 info() << "CELL: cell=" << ItemPrinter(*icell)
308 << " next=" << ItemPrinter(cc.next())
309 << " previous=" << ItemPrinter(cc.previous());
310 // Maille au bord. J'ajoute de la densité.
311 ++nb_boundary2;
312 m_density[icell] += 5.0;
313 }
314
315 info() << "NB_BOUNDARY1=" << nb_boundary1 << " NB_BOUNDARY2=" << nb_boundary2;
316 }
317 bool is_amr = m_nb_expected_patch!=1;
318 if (options()->verbosityLevel()==0)
319 m_utils->setNbPrint(5);
320 m_utils->testAll(is_amr);
321 _writePostProcessing();
322 _testDirections();
323}

344
345void AMRCartesianMeshTesterModule::
346_processPatches()
347{
348 const bool do_check = true;
349 const bool is_verbose = options()->verbosityLevel()>=1;
350
351 const Int32 dimension = defaultMesh()->dimension();
352 // Vérifie qu'il y a autant de patchs que d'options raffinement dans
353 // le jeu de données (en comptant le patch 0 qui est le maillage cartésien).
354 // Cela permet de vérifier que les appels successifs
355 // à computeDirections() n'ajoutent pas de patchs.
356 Integer nb_expected_patch = m_nb_expected_patch;
357
358 Integer nb_patch = m_cartesian_mesh->nbPatch();
359 if (nb_expected_patch!=nb_patch)
360 ARCANE_FATAL("Bad number of patchs expected={0} value={1}",nb_expected_patch,nb_patch);
361
362 IParallelMng* pm = parallelMng();
363 Int32 comm_rank = pm->commRank();
364 Int32 comm_size = pm->commSize();
365
366 UniqueArray<Int32> nb_cells_expected(options()->expectedNumberOfCellsInPatchs);
367 if (nb_cells_expected.size()!=nb_patch)
368 ARCANE_FATAL("Bad size ({0}, expected={1}) for option '{2}'",
369 nb_cells_expected.size(),nb_patch,options()->expectedNumberOfCellsInPatchs.name());
370
371 // Nombre de mailles fantômes attendu. Utilisé uniquement en parallèle
372 bool has_expected_ghost_cells = options()->expectedNumberOfGhostCellsInPatchs.isPresent();
373 if (!pm->isParallel())
374 has_expected_ghost_cells = false;
375
376 UniqueArray<Int32> nb_ghost_cells_expected(options()->expectedNumberOfGhostCellsInPatchs);
377 if (has_expected_ghost_cells && (nb_ghost_cells_expected.size()!=nb_patch))
378 ARCANE_FATAL("Bad size ({0}, expected={1}) for option '{2}'",
379 nb_ghost_cells_expected.size(), nb_patch, options()->expectedNumberOfGhostCellsInPatchs.name());
380 // Affiche les informations sur les patchs
381 for( Integer i=0; i<nb_patch; ++i ){
382 ICartesianMeshPatch* p = m_cartesian_mesh->patch(i);
383 CellGroup patch_cells(p->cells());
384 info() << "Patch cell_group=" << patch_cells.name() << " nb_cell=" << patch_cells.size();
385 VariableCellReal* cellv = new VariableCellReal(VariableBuildInfo(defaultMesh(),String("CellPatch")+i));
386 m_cell_patch_variables.add(cellv);
387 cellv->fill(0.0);
388 ENUMERATE_CELL(icell,patch_cells){
389 (*cellv)[icell] = 2.0;
390 }
391
392 CellGroup patch_own_cell = patch_cells.own();
393 UniqueArray<Int64> own_cells_uid;
394 ENUMERATE_(Cell,icell,patch_own_cell){
395 Cell cell{*icell};
396 if (is_verbose)
397 info() << "Patch i=" << i << " cell=" << ItemPrinter(*icell);
398 own_cells_uid.add(cell.uniqueId());
399 }
400 // Affiche la liste globales des uniqueId() des mailles.
401 {
402 UniqueArray<Int64> global_cells_uid;
403 pm->allGatherVariable(own_cells_uid,global_cells_uid);
404 std::sort(global_cells_uid.begin(),global_cells_uid.end());
405 Integer nb_global_uid = global_cells_uid.size();
406 info() << "GlobalUids Patch=" << i << " NB=" << nb_global_uid
407 << " expected=" << nb_cells_expected[i];
408 // Vérifie que le nombre de mailles par patch est le bon.
409 if (do_check && nb_cells_expected[i]!=nb_global_uid)
410 ARCANE_FATAL("Bad number of cells for patch I={0} N={1} expected={2}",
411 i,nb_global_uid,nb_cells_expected[i]);
412 if (is_verbose)
413 for( Integer c=0; c<nb_global_uid; ++c )
414 info() << "GlobalUid Patch=" << i << " I=" << c << " cell_uid=" << global_cells_uid[c];
415 }
416 // Teste le nombre de mailles fantômes
417 if (has_expected_ghost_cells){
418 Int32 local_nb_ghost_cell = patch_cells.size() - patch_own_cell.size();
419 Int32 total = pm->reduce(Parallel::ReduceSum,local_nb_ghost_cell);
420 pinfo() << "NbGhostCells my_rank=" << comm_rank << " local=" << local_nb_ghost_cell << " total=" << total;
421 if (total!=nb_ghost_cells_expected[i])
422 ARCANE_FATAL("Bad number of ghost cells for patch I={0} N={1} expected={2}",
423 i,total,nb_ghost_cells_expected[i]);
424 }
425
426 // Exporte le patch au format SVG
427 if (dimension==2 && options()->dumpSvg()){
428 String filename = String::format("Patch{0}-{1}-{2}.svg",i,comm_rank,comm_size);
429 Directory directory = subDomain()->exportDirectory();
430 String full_filename = directory.file(filename);
431 std::ofstream ofile(full_filename.localstr());
432 SimpleSVGMeshExporter exporter(ofile);
433 exporter.write(patch_cells);
434 }
435 }
436}

441void AMRCartesianMeshTesterModule::
442_computeCenters()
443{
444 IMesh* mesh = defaultMesh();
445
446 // Calcule le centre des mailles
447 {
448 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
449 ENUMERATE_CELL(icell,allCells()){
450 Cell cell = *icell;
451 Real3 center;
452 for( NodeLocalId inode : cell.nodes() )
453 center += nodes_coord[inode];
454 center /= cell.nbNode();
455 m_cell_center[icell] = center;
456 }
457 }
458
459 // Calcule le centre des faces
460 {
461 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
462 ENUMERATE_FACE(iface,allFaces()){
463 Face face = *iface;
464 Real3 center;
465 for( NodeLocalId inode : face.nodes() )
466 center += nodes_coord[inode];
467 center /= face.nbNode();
468 m_face_center[iface] = center;
469 }
470 }
471}

476void AMRCartesianMeshTesterModule::
477_initAMR()
478{
479 // Regarde si on dé-raffine le maillage initial
480 if (options()->coarseAtInit()){
481 // Il faut que les directions aient été calculées avant d'appeler le dé-raffinement
482 m_cartesian_mesh->computeDirections();
483
484 info() << "Doint initial coarsening";
485
486 if (m_cartesian_mesh->mesh()->meshKind().meshAMRKind() == eMeshAMRKind::PatchCartesianMeshOnly) {
487 debug() << "Coarse with specific coarser (for cartesian mesh only)";
488 Ref<ICartesianMeshAMRPatchMng> coarser = CartesianMeshUtils::cartesianMeshAMRPatchMng(m_cartesian_mesh);
489 coarser->coarse();
490 }
491 else {
492 Ref<CartesianMeshCoarsening2> coarser = CartesianMeshUtils::createCartesianMeshCoarsening2(m_cartesian_mesh);
493 coarser->createCoarseCells();
494 }
495
496 CartesianMeshPatchListView patches = m_cartesian_mesh->patches();
497 Int32 nb_patch = patches.size();
498 {
499 Int32 index = 0;
500 info() << "NB_PATCH=" << nb_patch;
501 for( CartesianPatch p : patches){
502 info() << "Patch i=" << index << " nb_cell=" << p.cells().size();
503 ++index;
504 }
505 }
506 }
507 // Parcours les mailles actives et ajoute dans la liste des mailles
508 // à raffiner celles qui sont contenues dans le boîte englobante
509 // spécifiée dans le jeu de données.
510 Int32 dim = defaultMesh()->dimension();
511 if (dim==2){
512 for( auto& x : options()->refinement2d() ){
513 m_cartesian_mesh->refinePatch2D(x->position(),x->length());
514 m_cartesian_mesh->computeDirections();
515 }
516 }
517 if (dim==3){
518 for( auto& x : options()->refinement3d() ){
519 m_cartesian_mesh->refinePatch3D(x->position(),x->length());
520 m_cartesian_mesh->computeDirections();
521 }
522 }
523}

539void AMRCartesianMeshTesterModule::
540_computeSubCellDensity(Cell cell)
541{
542 Int32 nb_children = cell.nbHChildren();
543 if (nb_children==0)
544 return;
545 // Pour les mailles AMR, la densité est la moyenne des noeuds qui la compose.
546 for( Int32 j=0; j<nb_children; ++j ) {
547 Real sub_density = 0.0;
548 Cell sub_cell = cell.hChild(j);
549 Integer sub_cell_nb_node = sub_cell.nbNode();
550 for( Integer k=0; k<sub_cell_nb_node; ++k )
551 sub_density += m_node_density[sub_cell.node(k)];
552 sub_density /= (Real)sub_cell_nb_node;
553 m_density[sub_cell] =sub_density;
554 _computeSubCellDensity(sub_cell);
555 }
556}

560
561void AMRCartesianMeshTesterModule::
562_compute1()
563{
564 // Pour test, on parcours les N directions
565 // et pour chaque maille, on modifie sa densité
566 // par la formule new_density = (density+density_next+density_prev) / 3.0.
567
568 // Effectue l'operation en deux fois. Une premiere sur les
569 // mailles internes, et une deuxieme sur les mailles externes.
570 // Du coup, il faut passer par une variable intermediaire (m_old_density)
571 // mais on evite un test dans la boucle principale
572 IMesh* mesh = defaultMesh();
573 Integer nb_dir = mesh->dimension();
574 for( Integer idir=0; idir<nb_dir; ++idir){
575 m_old_density.copy(m_density);
576 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
577 // Travail sur les mailles internes
578 info() << "Direction=" << idir << " cells=" << cdm.innerCells().name()
579 << " n=" << cdm.innerCells().size();
580 ENUMERATE_CELL(icell,cdm.innerCells()){
581 Cell cell = *icell;
582 DirCell cc(cdm.cell(cell));
583 Cell next = cc.next();
584 Cell prev = cc.previous();
585 Real d = m_old_density[icell] + m_old_density[next] + m_old_density[prev];
586 m_density[icell] = d / 3.0;
587 _computeSubCellDensity(cell);
588 }
589 // Travail sur les mailles externes
590 // Test si la maille avant ou apres est nulle.
591 ENUMERATE_CELL(icell,cdm.outerCells()){
592 Cell cell = *icell;
593 DirCell cc(cdm[icell]);
594 Cell next = cc.next();
595 Cell prev = cc.previous();
596 Real d = m_old_density[icell];
597 Integer n = 1;
598 if (!next.null()){
599 d += m_old_density[next];
600 ++n;
601 }
602 if (!prev.null()){
603 d += m_old_density[prev];
604 ++n;
605 }
606 m_density[icell] = d / n;
607 _computeSubCellDensity(cell);
608 }
609 }
610 // Modifie la densité aux noeuds.
611 // Elle sera égale à la moyenne des densités des mailles entourant ce noeud
612 ENUMERATE_NODE(inode,mesh->allNodes()){
613 Node node = *inode;
614 Integer nb_cell = node.nbCell();
615 Real density = 0.0;
616 for( Integer i=0; i<nb_cell; ++i )
617 density += m_density[node.cell(i)];
618 density /= (Real)nb_cell;
619 m_node_density[inode] = density;
620 }
621}

625
626void AMRCartesianMeshTesterModule::
627_compute2()
628{
629 // Pour test, on parcours les N directions
630 // et pour chaque maille, on modifie sa densité
631 // par la formule new_density = (density+density_next+density_prev) / 3.0.
632
633 // A noter que cette methode ne donne pas le meme comportement que
634 // _compute1() car les mailles de bord et internes sont mises à jour
635 // dans un ordre différent.
636 Integer nb_dir = defaultMesh()->dimension();
637 for( Integer idir=0; idir<nb_dir; ++idir){
638 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
639 // Travail sur toutes les mailles
640 ENUMERATE_CELL(icell,cdm.allCells()){
641 DirCell cc(cdm[icell]);
642 Cell next = cc.next();
643 Cell prev = cc.previous();
644 Real d = m_density[icell];
645 Integer n = 1;
646 if (!next.null()){
647 d += m_density[next];
648 ++n;
649 }
650 if (!prev.null()){
651 d += m_density[prev];
652 ++n;
653 }
654 m_density[icell] = d / n;
