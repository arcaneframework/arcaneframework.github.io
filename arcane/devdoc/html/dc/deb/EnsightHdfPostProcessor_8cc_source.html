147void EnsightHdfDataWriter::
148beginWrite(const VariableCollection& vars)
149{
150 warning() << "L'implémentation du format 'EnsightHdf' n'est pas encore opérationnelle";
151
152 StringBuilder sb("hdf5_ensight_");
153 sb += m_time_index;
154 m_filename = sb.toString();
155
156 m_variable_index = 0;
157
158 info() << "ENSIGHT HDF BEGIN WRITE file=" << m_filename;
159
160 H5open();
161
162 m_file_id.openTruncate(m_filename);
163 HGroup top_group;
164 top_group.create(m_file_id,"model");
165
166 _addStringAttribute(top_group,"hdfrwRelease","0.9.0",20);
167 _addIntegerAttribute(top_group,"nParts",m_groups.count());
168 _addIntegerAttribute(top_group,"nVariables",vars.count());
169 _addIntegerAttribute(top_group,"nDomains",1);
170 _addIntegerAttribute(top_group,"hasNodeIDs",0);
171 _addIntegerAttribute(top_group,"hasElementIDs",0);
172 _addIntegerAttribute(top_group,"timeStep",1);
173 // La valeur 2 indique que la connectivité est décrite dans le fichier
174 //_addIntegerAttribute(top_group,"geomChangingFlag",2); // CONN_HDFRW
175 //_addIntegerAttribute(top_group,"writingConn",0);
176
177 _addIntegerAttribute(top_group,"geomChangingFlag",1); // COORDS_ONLY_HDFRW
178 _addIntegerAttribute(top_group,"writingConn",1);
179
180 _addRealAttribute(top_group,"timeValue",m_times[m_time_index-1]);
181 _addStringAttribute(top_group,"modelDescription1","MyModel1",80);
182 _addStringAttribute(top_group,"modelDescription2","MyModel2",80);
183 _addStringAttribute(top_group,"lastGeomFile","None",256);
184
185 m_geometry_top_group.create(top_group,"Geometry");
186 m_variable_top_group.create(top_group,"Variables");
187
188 {
189 Integer index = 1;
190 for( ItemGroupCollection::Enumerator igroup(m_groups); ++igroup; ++index ){
191 const ItemGroup& group = *igroup;
192
193 HGroup geometry_group;
194 String part_name(String("geom_part_")+index);
195 geometry_group.create(m_geometry_top_group,part_name.localstr());
196
197 HGroup domain_group;
198 String domain_name("geom_domain_1");
199 domain_group.create(geometry_group,domain_name.localstr());
200
201 // La valeur 0 indique qu'il s'agit d'un maillage non-structuré
202 _addIntegerAttribute(geometry_group,"structuredFlag",0); // UNSTRUCTURED_HDFRW
203 _addStringAttribute(geometry_group,"partName",group.name().localstr(),80);
204 _addStringAttribute(geometry_group,"partDescription","None",80);
205 _saveGroup(group,domain_group);
206 }
207 }

213void EnsightHdfDataWriter::
214_saveGroup(const ItemGroup& group,HGroup& domain_group)
215{
216 info() << "SAVE GROUP name=" << group.name();
217 //coord_dataset.open(m_domain_group,"coordinates");
218
219 VariableNodeReal3 nodes_coordinates(m_mesh->toPrimaryMesh()->nodesCoordinates());
220 Real3ConstArrayView nodes_coordinates_array = nodes_coordinates.asArray();
221
222 // Premièrement, détermine les noeuds utilisés pour ce groupe, et
223 // leur assigne un index local (qui doit commencer à un pour Ensight)
224 IItemFamily* node_family = m_mesh->nodeFamily();
225 Integer max_index = node_family->maxLocalId();
226 Int32UniqueArray node_local_indexes(max_index);
227 node_local_indexes.fill(NULL_ITEM_LOCAL_ID);
228 Real3UniqueArray node_local_coords;
229 Integer nb_local_node = 0;
230
231 {
232 ENUMERATE_ITEM(iitem,group){
233 const Item& _item = *iitem;
234 const ItemWithNodes& item = _item.toItemWithNodes();
235 //Integer index = 0;
236 for( NodeLocalId inode_lid : item.nodeIds() ){
237 node_local_indexes[inode_lid] = 0;
238 }
239 }
240 Integer current_index = 1;
241 for( Integer i=0; i<max_index; ++i ){
242 if (node_local_indexes[i]!=NULL_ITEM_LOCAL_ID){
243 node_local_indexes[i] = current_index;
244 ++current_index;
245 node_local_coords.add(nodes_coordinates_array[i]);
246 }
247 }
248 nb_local_node = current_index - 1;
249 }
250
251 info() << "TODO: save extents for geometry";
252
253 // Sauve les coordonnées des noeuds
254 {
255 HDataset coord_dataset;
256
257 Real3* ptr = node_local_coords.data();
258
259#if 0
260 ENUMERATE_NODE(inode,all_nodes){
261 const Node& node = *inode;
262 info() << " NODE uid=" << node.uniqueId() << " lid=" << node.localId();
263 }
264#endif
265
266 hsize_t dims[1];
267 dims[0] = nb_local_node*3;
268
269 HSpace hspace;
270 hspace.createSimple(1,dims);
271
272 coord_dataset.create(domain_group,"coordinates",H5T_NATIVE_FLOAT,hspace,H5P_DEFAULT);
273 coord_dataset.write(H5T_NATIVE_DOUBLE,(Real*)ptr);
274
275 _addIntegerAttribute(coord_dataset,"layout",1); // INTERLACE_ORDER_1D
276 _addIntegerAttribute(coord_dataset,"nNodes",nb_local_node);
277 }
278
279 // Sauve les éléments
280 // ATTENTION: suppose uniquement des hexa8
281 {
282 HDataset item_dataset;
283
284 ItemGroup all_items = group;
285 Integer nb_item = all_items.size();
286 Int32UniqueArray item_node_ids;
287 ENUMERATE_ITEM(iitem,all_items){
288 const Item& _item = *iitem;
289 const ItemWithNodes& item = _item.toItemWithNodes();
290 for( NodeLocalId inode_lid : item.nodeIds() ){
291 item_node_ids.add(node_local_indexes[inode_lid]);
292 }
293 }
294 hsize_t dims[1];
295 dims[0] = nb_item*8;
296
297 HSpace hspace;
298 hspace.createSimple(1,dims);
299
300 item_dataset.create(domain_group,"hexa8",H5T_NATIVE_INT,hspace,H5P_DEFAULT);
301 item_dataset.write(H5T_NATIVE_INT,item_node_ids.data());
302
303 _addIntegerAttribute(item_dataset,"layout",1); // INTERLACE_ORDER_1D
304 // NOTE: dans le pdf de hdf_rw, l'attribut s'appelle 'nElems' mais il
305 // faut mettre 'numof'
306 _addIntegerAttribute(item_dataset,"numof",nb_item);
307 }
308
309}

314void EnsightHdfDataWriter::
315_addIntegerAttribute(Hid& hid,const char* name,int value)
316{
317 hid_t aid = H5Screate(H5S_SCALAR);
318 hid_t attr = H5Acreate2(hid.id(), name, H5T_NATIVE_INT, aid, H5P_DEFAULT,H5P_DEFAULT);
319 if (attr<0)
320 throw FatalErrorException(A_FUNCINFO,String("Can not create attribute ")+name);
321 int ret = H5Awrite(attr, H5T_NATIVE_INT, &value);
322 ret = H5Sclose(aid);
323 ret = H5Aclose(attr);
324 if (ret<0)
325 throw FatalErrorException(A_FUNCINFO,String("Can not write attribute ")+name);
326}

367void EnsightHdfDataWriter::
368_addStringAttribute(Hid& hid,const char* name,const String& value,hsize_t len)
369{
370 UniqueArray<char> buf(CheckedConvert::toInteger(len));
371 buf.fill('\0');
372 const char* value_str = value.localstr();
373 strncpy(buf.data(),value_str,len-1);
374
375 hid_t aid = H5Screate_simple(1, &len, NULL);
376 hid_t attr = H5Acreate2(hid.id(),name, H5T_NATIVE_CHAR, aid, H5P_DEFAULT,H5P_DEFAULT);
377 if (attr<0)
378 throw FatalErrorException(String("Can not create attribute ")+name);
379 int ret = H5Awrite(attr, H5T_NATIVE_CHAR, buf.data());
380 ret = H5Sclose(aid);
381 ret = H5Aclose(attr);
382 if (ret<0)
383 throw FatalErrorException(A_FUNCINFO,String("Can not write attribute ")+name);

417void EnsightHdfDataWriter::
418write(IVariable* var,IData* data)
419{
420 eItemKind item_kind = var->itemKind();
421 if (item_kind!=IK_Cell){
422 pwarning() << "Can only export cell variable (name=" << var->name() << ")";
423 return;
424 }
425 if (var->dimension()!=1){
426 pwarning() << "Can only export 1-dimension variable (name=" << var->name() << ")";
427 return;
428 }
429
430 ++m_variable_index;
431
432 {
433
434 HGroup variable_group;
435 String variable_name(String("variable_")+m_variable_index);
436 variable_group.create(m_variable_top_group,variable_name.localstr());
437
438 _addStringAttribute(variable_group,"varName",var->name().localstr(),80);
439 _addIntegerAttribute(variable_group,"varType",2); // SCALAR_HDFRW,
440 _addStringAttribute(variable_group,"varDescription","None",80);
441 _addIntegerAttribute(variable_group,"mappingType",1); // PER_ELEM_HDFRW
442
443 RealUniqueArray global_min_values(9);
444 global_min_values.fill(0.0);
445 RealUniqueArray global_max_values(9);
446 global_max_values.fill(0.0);
447
448 Integer index = 1;
449 for( ItemGroupCollection::Enumerator igroup(m_groups); ++igroup; ++index ){
450 const ItemGroup& group = *igroup;
451
452 info() << " VARIABLE INDEX=" << m_variable_index << " INDEX=" << index << " name=" << var->name();
453
454
455 HGroup part_group;
456 String part_name(String("var_part_")+index);
457 info() << "CREATE PART GROUP " << part_name;
458 part_group.create(variable_group,part_name.localstr());
459
460 HGroup domain_group;
461 String domain_name("var_domain_1");
462 info() << "CREATE DOMAIN GROUP " << domain_name;
463 domain_group.create(part_group,domain_name.localstr());
464
465 info() << " DOMAIN GROUP " << variable_name << "/" << part_name << "/" << domain_name;
466
467 RealUniqueArray min_values(9);
468 min_values.fill(0.0);
469 RealUniqueArray max_values(9);
470 max_values.fill(0.0);
471 _saveVariableOnGroup(var,data,group,min_values,max_values,domain_group);
472 _addRealArrayAttribute(domain_group,"mins",min_values);
473 _addRealArrayAttribute(domain_group,"maxs",max_values);
474
475 _addRealArrayAttribute(part_group,"mins",min_values);
476 _addRealArrayAttribute(part_group,"maxs",max_values);
477
478 if (index==1){
479 global_min_values.copy(min_values);
480 global_max_values.copy(max_values);
481 }
482 else{
483 for( Integer i=0; i<9; ++i ){
484 if (min_values[i]<global_min_values[i])
485 global_min_values[i] = min_values[i];
486 if (max_values[i]>global_max_values[i])
487 global_max_values[i] = max_values[i];
488 }
489 }
490 }
491
492 _addRealArrayAttribute(variable_group,"mins",global_min_values);
493 _addRealArrayAttribute(variable_group,"maxs",global_max_values);
494
495 }
496
497}

502void EnsightHdfDataWriter::
503_saveVariableOnGroup(IVariable* var,IData* data,const ItemGroup& group,
504 RealArrayView min_values,
505 RealArrayView max_values,
506 HGroup& domain_group)
507{
508 Ref<ISerializedData> sdata(data->createSerializedDataRef(false));
509 info() << "SAVE VARIABLE var=" << var->fullName() << " on group=" << group.name();
510 if (sdata->baseDataType()!=DT_Real)
511 ARCANE_FATAL("Bad datatype (only DT_Real is allowed)");
512
513 Span<const Byte> sbuffer = sdata->constBytes();
514 const Real* ptr = reinterpret_cast<const Real*>(sbuffer.data());
515 Span<const Real> true_values(ptr,sdata->nbBaseElement());
516 RealUniqueArray values;
517
518 // Sauve les valeurs des éléments
519 // ATTENTION: suppose uniquement des hexa8
520 Real min_val = FloatInfo<Real>::maxValue();
521 Real max_val = -min_val;
522 {
523 HDataset item_dataset;
524
525 Int32UniqueArray item_node_ids;
526 ENUMERATE_ITEM(iitem,group){
527 Real value = true_values[iitem.itemLocalId()];
528 values.add(value);
529 if (value<min_val)
530 min_val = value;
531 if (value>max_val)
532 max_val = value;
533 }
534 info() << "MIN =" << min_val << " MAX=" << max_val;
535 min_values[0] = min_val;
536 max_values[0] = max_val;
537 Integer nb_value = values.size();
538 hsize_t dims[1];
539 dims[0] = nb_value;
540
541 HSpace hspace;
542 hspace.createSimple(1,dims);
543
544 item_dataset.create(domain_group,"hexa8",H5T_NATIVE_FLOAT,hspace,H5P_DEFAULT);
545 item_dataset.write(H5T_NATIVE_DOUBLE,values.data());
546
547 _addIntegerAttribute(item_dataset,"layout",1); // INTERLACE_ORDER_1D
548 _addIntegerAttribute(item_dataset,"dataType",0); // Indique une valeur flottante
549 }
550
551}
