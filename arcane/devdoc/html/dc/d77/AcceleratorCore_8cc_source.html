232ePointerAccessibility impl::RuntimeStaticInfo::
233getPointerAccessibility(eExecutionPolicy policy, const void* ptr, PointerAttribute* ptr_attr)
234{
235 // Regarde si le pointeur est accessible pour la politique d'exécution donnée.
236 // Le seul cas où on peut le savoir exactement est si on a un runtime
237 // accélérateur et que la valeur retournée par getPointeAttribute() est valide.
238 if (policy == eExecutionPolicy::None)
239 return ePointerAccessibility::Unknown;
240 IRunnerRuntime* r = getAcceleratorRunnerRuntime();
241 if (!r)
242 return ePointerAccessibility::Unknown;
243 PointerAttribute attr;
244 r->getPointerAttribute(attr, ptr);
245 if (ptr_attr) {
246 *ptr_attr = attr;
247 }
248 if (attr.isValid()) {
249 if (isAcceleratorPolicy(policy))
250 return attr.devicePointer() ? ePointerAccessibility::Yes : ePointerAccessibility::No;
251 else {
252 if (attr.memoryType() == ePointerMemoryType::Unregistered)
253 return ePointerAccessibility::Yes;
254 return attr.hostPointer() ? ePointerAccessibility::Yes : ePointerAccessibility::No;
255 }
256 }
257 return ePointerAccessibility::Unknown;
258}

263void impl::RuntimeStaticInfo::
264checkPointerIsAcccessible(eExecutionPolicy policy, const void* ptr,
265 const char* name, const TraceInfo& ti)
266{
267 // Le pointeur nul est toujours accessible.
268 if (!ptr)
269 return;
270 PointerAttribute ptr_attr;
271 ePointerAccessibility a = getPointerAccessibility(policy, ptr, &ptr_attr);
272 if (a == ePointerAccessibility::No) {
273 auto s = String::format("Pointer 'addr={0}' ({1}) is not accessible "
274 "for this execution policy ({2}).\n PointerInfo: {3}",
275 ptr, name, policy, ptr_attr);
276
277 throw FatalErrorException(ti, s);
278 }
279}
