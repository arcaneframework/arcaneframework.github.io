34 class ARCANE_CORE_EXPORT IndexedItemConnectivityViewBase
35 {
36  public:
38  ARCCORE_HOST_DEVICE Int32 nbItem(ItemLocalId lid) const { return m_nb_item[lid]; }
40  ARCCORE_HOST_DEVICE ItemLocalIdViewT<Item> items(ItemLocalId lid) const
41  {
42  const Int32* ptr = & m_list_data[m_indexes[lid]];
43  return { reinterpret_cast<const ItemLocalId*>(ptr), m_nb_item[lid] };
44  }
45  eItemKind sourceItemKind() const { return m_source_kind; }
46  eItemKind targetItemKind() const { return m_target_kind; }
47  public:
49  void init(SmallSpan<const Int32> nb_item,SmallSpan<const Int32> indexes,
50  SmallSpan<const Int32> list_data,eItemKind source_kind,eItemKind target_kind)
51  {
52  m_indexes = indexes;
53  m_nb_item = nb_item;
54  m_list_data = list_data;
55  m_source_kind = source_kind;
56  m_target_kind = target_kind;
57  }
58 
59  void set(IndexedItemConnectivityViewBase view)
60  {
61  m_indexes = view.m_indexes;
62  m_nb_item = view.m_nb_item;
63  m_list_data = view.m_list_data;
64  m_source_kind = view.m_source_kind;
65  m_target_kind = view.m_target_kind;
66  }
67  protected:
68  SmallSpan<const Int32> m_nb_item;
69  SmallSpan<const Int32> m_indexes;
70  SmallSpan<const Int32> m_list_data;
71  eItemKind m_source_kind = IK_Unknown;
72  eItemKind m_target_kind = IK_Unknown;
73  protected:
74  [[noreturn]] void _badConversion(eItemKind k1,eItemKind k2) const;
75  inline void _checkValid(eItemKind k1,eItemKind k2) const
76  {
77  if (k1!=m_source_kind || k2!=m_target_kind)
78  _badConversion(k1,k2);
79  }
80 };

87 template<typename ItemType1,typename ItemType2>
88 class IndexedItemConnectivityGenericViewT
89 : public IndexedItemConnectivityViewBase
90 {
91  public:
92  using ItemType1Type = ItemType1;
93  using ItemType2Type = ItemType2;
94  using ItemLocalId1 = typename ItemType1::LocalIdType;
95  using ItemLocalId2 = typename ItemType2::LocalIdType;
96  using ItemLocalIdViewType = ItemLocalIdViewT<ItemType2>;
97  public:
98  IndexedItemConnectivityGenericViewT(IndexedItemConnectivityViewBase view)
99  : IndexedItemConnectivityViewBase(view)
100  {
101 #ifdef ARCANE_CHECK
102  eItemKind k1 = ItemTraitsT<ItemType1>::kind();
103  eItemKind k2 = ItemTraitsT<ItemType2>::kind();
104  _checkValid(k1,k2);
105 #endif
106  }
107  IndexedItemConnectivityGenericViewT() = default;
108  public:
110  ARCCORE_HOST_DEVICE ItemLocalIdViewType items(ItemLocalId1 lid) const
111  {
112  const Int32* ptr = & m_list_data[m_indexes[lid]];
113  return { reinterpret_cast<const ItemLocalId2*>(ptr), m_nb_item[lid] };
114  }
116  ARCCORE_HOST_DEVICE ItemLocalIdViewType itemIds(ItemLocalId1 lid) const
117  {
118  const Int32* ptr = & m_list_data[m_indexes[lid]];
119  return { reinterpret_cast<const ItemLocalId2*>(ptr), m_nb_item[lid] };
120  }
122  ARCCORE_HOST_DEVICE ItemLocalId2 itemId(ItemLocalId1 lid,Int32 index) const
123  {
124  return ItemLocalId2(m_list_data[m_indexes[lid]+index]);
125  }
126 };
