64void MeshMaterial::
65build()
66{
67 IMesh* mesh = m_material_mng->mesh();
68 IItemFamily* cell_family = mesh->cellFamily();
69 String group_name = m_material_mng->name() + "_" + name();
70 CellGroup items = cell_family->findGroup(group_name, true);
71 m_data._setItems(items);
72}

86MatCell MeshMaterial::
87findMatCell(AllEnvCell c) const
88{
89 Int32 mat_id = m_data.componentId();
90 ENUMERATE_CELL_ENVCELL (ienvcell, c) {
91 ENUMERATE_CELL_MATCELL (imatcell, (*ienvcell)) {
92 MatCell mc = *imatcell;
93 Int32 mid = mc.materialId();
94 if (mid == mat_id)
95 return mc;
96 }
97 }
98 return MatCell();
99}
