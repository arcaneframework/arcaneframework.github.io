159class ReferenceCounterAccessor
160{
161 public:
162 static void addReference(T* t)
163 {
164 if constexpr(Arcane::impl::HasInternalAddReference<T>::value)
165 t->_internalAddReference();
166 else
167 t->addReference();
168 }
169 static void removeReference(T* t)
170 {
171 if constexpr(Arcane::impl::HasInternalRemoveReference<T>::value){
172 bool need_destroy = t->_internalRemoveReference();
173 if (need_destroy)
174 delete t;
175 }
176 else
177 t->removeReference();
178 }
