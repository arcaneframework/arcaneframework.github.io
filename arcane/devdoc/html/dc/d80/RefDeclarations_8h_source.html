146template<class T>
147class ReferenceCounterAccessor
148{
149 public:
150 static void addReference(T* t)
151 {
152 if constexpr(impl::HasInternalAddReference<T>::value)
153 t->_internalAddReference();
154 else
155 t->addReference();
156 }
157 static void removeReference(T* t)
158 {
159 if constexpr(impl::HasInternalRemoveReference<T>::value){
160 bool need_destroy = t->_internalRemoveReference();
161 if (need_destroy)
162 delete t;
163 }
164 else
165 t->removeReference();
166 }
167};
