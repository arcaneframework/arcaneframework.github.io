29isEqual(const char* s1,const char* s2)
30{
31 if (!s1 && !s2)
32 return true;
33 if (!s1 || !s2)
34 return false;
35 while (*s1==*s2){
36 if (*s1=='\0')
37 return true;
38 ++s1; ++s2;
39 }
40 return false;
41}
