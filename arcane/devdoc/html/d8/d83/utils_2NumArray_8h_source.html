49  // Wrapper de Arccore::Array pour la classe NumArray
50  template<typename DataType>
51  class NumArrayContainer
52  : private Arccore::Array<DataType>
53  {
54  private:
55  using BaseClass = Arccore::Array<DataType>;
56  public:
57  using BaseClass::capacity;
58  using BaseClass::fill;
59  public:
60  explicit NumArrayContainer(IMemoryAllocator* a)
61  {
62  this->_initFromAllocator(a,0);
63  }
64  NumArrayContainer(const NumArrayContainer<DataType>& rhs) : BaseClass()
65  {
66  this->_initFromSpan(rhs.to1DSpan());
67  }
68  NumArrayContainer<DataType>& operator=(const NumArrayContainer<DataType>& rhs)
69  {
70  if (this!=&rhs){
71  BaseClass::_copy(rhs.data());
72  }
73  return (*this);
74  }
75  public:
76  void resize(Int64 new_size) { BaseClass::_resize(new_size); }
77  Span<DataType> to1DSpan() { return BaseClass::span(); }
78  Span<const DataType> to1DSpan() const { return BaseClass::constSpan(); }
79  Span<std::byte> bytes() { return asWritableBytes(BaseClass::span()); }
80  Span<const std::byte> bytes() const { return asBytes(BaseClass::constSpan()); }
81  void swap(NumArrayContainer<DataType>& rhs) { BaseClass::_swap(rhs); }
82  void copy(Span<const DataType> rhs) { BaseClass::_copy(rhs.data()); }
83  void copyInitializerList(std::initializer_list<DataType> alist)
84  {
85  Span<DataType> s = to1DSpan();
86  Int64 s1 = s.size();
87  Int32 index = 0;
88  for( auto x : alist ){
89  s[index] = x;
90  ++index;
91  // S'assure qu'on ne déborde pas
92  if (index>=s1)
93  break;
94  }
95  }
96  };
97 }

125 template<typename DataType,A_MDRANK_TYPE(RankValue),typename LayoutType>
126 class NumArrayBase
127 : public NumArrayBaseCommon
128 {
129  public:
130 
131  using ConstSpanType = MDSpan<const DataType,RankValue,LayoutType>;
132  using SpanType = MDSpan<DataType,RankValue,LayoutType>;
133  using ArrayWrapper = impl::NumArrayContainer<DataType>;
134  using ArrayBoundsIndexType = typename SpanType::ArrayBoundsIndexType;
135 
136  public:
137 
139  Int64 totalNbElement() const { return m_total_nb_element; }
141  Int32 extent(int i) const { return m_span.extent(i); }
146  void resize(ArrayExtents<RankValue> extents)
147  {
148  m_span.m_extents.setSize(extents);
149  _resize();
150  }
151  protected:
152 
153  NumArrayBase() : m_data(_getDefaultAllocator()){}
154  explicit NumArrayBase(eMemoryRessource r) : m_data(_getDefaultAllocator(r)), m_memory_ressource(r){}
155  explicit NumArrayBase(ArrayExtents<RankValue> extents)
156  : m_data(_getDefaultAllocator()), m_memory_ressource(eMemoryRessource::UnifiedMemory)
157  {
158  resize(extents);
159  }
160  NumArrayBase(ArrayExtents<RankValue> extents,eMemoryRessource r)
161  : m_data(_getDefaultAllocator(r)), m_memory_ressource(r)
162  {
163  resize(extents);
164  }
165 
166  private:
167  void _resize()
168  {
169  Int32 dim1_size = extent(0);
170  Int32 dim2_size = 1;
171  // TODO: vérifier débordement.
172  for (int i=1; i< A_MDRANK_RANK_VALUE(RankValue) ; ++i )
173  dim2_size *= extent(i);
174  m_total_nb_element = dim1_size * dim2_size;
175  m_data.resize(m_total_nb_element);
176  m_span.m_ptr = m_data.to1DSpan().data();
177  }
178  public:
179  void fill(const DataType& v) { m_data.fill(v); }
180  constexpr Int32 nbDimension() const { return A_MDRANK_RANK_VALUE(RankValue); }
181  ArrayExtents<RankValue> extents() const { return m_span.extents(); }
182  ArrayExtentsWithOffset<RankValue,LayoutType> extentsWithOffset() const
183  {
184  return m_span.extentsWithOffset();
185  }
186  public:
187  SpanType span() { return m_span; }
188  ConstSpanType span() const { return m_span.constSpan(); }
189  ConstSpanType constSpan() const { return m_span.constSpan(); }
190  public:
191  Span<const DataType> to1DSpan() const { return m_span.to1DSpan(); }
192  Span<DataType> to1DSpan() { return m_span.to1DSpan(); }
193  void copy(ConstSpanType rhs)
194  {
195  _checkHost(m_memory_ressource);
196  m_data.copy(rhs.to1DSpan());
197  }
198  void copy(const NumArrayBase<DataType,RankValue,LayoutType>& rhs)
199  {
200  this->resize(rhs.extents());
201  _copy(asBytes(rhs.to1DSpan()),rhs.m_memory_ressource,
202  asWritableBytes(to1DSpan()),m_memory_ressource);
203  }
204  const DataType& operator()(ArrayBoundsIndexType idx) const
205  {
206  return m_span(idx);
207  }
208  DataType& operator()(ArrayBoundsIndexType idx)
209  {
210  return m_span(idx);
211  }
212  DataType& s(ArrayBoundsIndexType idx)
213  {
214  return m_span(idx);
215  }
216  void swap(NumArrayBase<DataType,RankValue>& rhs)
217  {
218  m_data.swap(rhs.m_data);
219  std::swap(m_span,rhs.m_span);
220  std::swap(m_total_nb_element,rhs.m_total_nb_element);
221  std::swap(m_memory_ressource,rhs.m_memory_ressource);
222  }
223  Int64 capacity() const { return m_data.capacity(); }
224  eMemoryRessource memoryRessource() const { return m_memory_ressource; }
225  Span<std::byte> bytes() { return asWritableBytes(to1DSpan()); }
226  Span<const std::byte> bytes() const { return asBytes(to1DSpan()); }
227  public:
229  DataType* _internalData() { return m_span._internalData(); }
230  protected:
231  SpanType m_span;
232  ArrayWrapper m_data;
233  Int64 m_total_nb_element = 0;
234  eMemoryRessource m_memory_ressource = eMemoryRessource::Unknown;
235 };
