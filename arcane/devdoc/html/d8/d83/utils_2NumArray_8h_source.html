105 template<typename DataType,int RankValue,typename LayoutType>
106 class NumArrayBase
107 : public NumArrayBaseCommon
108 {
109  public:
110 
111  using ConstSpanType = MDSpan<const DataType,RankValue,LayoutType>;
112  using SpanType = MDSpan<DataType,RankValue,LayoutType>;
113  using ArrayWrapper = impl::NumArrayContainer<DataType>;
114 
115  public:
116 
118  Int64 totalNbElement() const { return m_total_nb_element; }
120  Int32 extent(int i) const { return m_span.extent(i); }
125  void resize(ArrayExtents<RankValue> extents)
126  {
127  m_span.m_extents.setSize(extents);
128  _resize();
129  }
130  protected:
131 
132  NumArrayBase() : m_data(_getDefaultAllocator()){}
133  explicit NumArrayBase(eMemoryRessource r) : m_data(_getDefaultAllocator(r)), m_memory_ressource(r){}
134  explicit NumArrayBase(ArrayExtents<RankValue> extents)
135  : m_data(_getDefaultAllocator()), m_memory_ressource(eMemoryRessource::UnifiedMemory)
136  {
137  resize(extents);
138  }
139  NumArrayBase(ArrayExtents<RankValue> extents,eMemoryRessource r)
140  : m_data(_getDefaultAllocator(r)), m_memory_ressource(r)
141  {
142  resize(extents);
143  }
144 
145  private:
146  void _resize()
147  {
148  Int32 dim1_size = extent(0);
149  Int32 dim2_size = 1;
150  // TODO: vérifier débordement.
151  for (int i=1; i<RankValue; ++i )
152  dim2_size *= extent(i);
153  m_total_nb_element = dim1_size * dim2_size;
154  m_data.resize(m_total_nb_element);
155  m_span.m_ptr = m_data.to1DSpan().data();
156  }
157  public:
158  void fill(const DataType& v) { m_data.fill(v); }
159  Int32 nbDimension() const { return RankValue; }
160  ArrayExtents<RankValue> extents() const { return m_span.extents(); }
161  ArrayExtentsWithOffset<RankValue,LayoutType> extentsWithOffset() const
162  {
163  return m_span.extentsWithOffset();
164  }
165  public:
166  SpanType span() { return m_span; }
167  ConstSpanType span() const { return m_span.constSpan(); }
168  ConstSpanType constSpan() const { return m_span.constSpan(); }
169  public:
170  Span<const DataType> to1DSpan() const { return m_span.to1DSpan(); }
171  Span<DataType> to1DSpan() { return m_span.to1DSpan(); }
172  void copy(ConstSpanType rhs)
173  {
174  _checkHost(m_memory_ressource);
175  m_data.copy(rhs.to1DSpan());
176  }
177  void copy(const NumArrayBase<DataType,RankValue,LayoutType>& rhs)
178  {
179  this->resize(rhs.extents());
180  _copy(asBytes(rhs.to1DSpan()),rhs.m_memory_ressource,
181  asWritableBytes(to1DSpan()),m_memory_ressource);
182  }
183  const DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
184  {
185  return m_span(idx);
186  }
187  DataType& s(ArrayBoundsIndex<RankValue> idx)
188  {
189  return m_span(idx);
190  }
191  void swap(NumArrayBase<DataType,RankValue>& rhs)
192  {
193  m_data.swap(rhs.m_data);
194  std::swap(m_span,rhs.m_span);
195  std::swap(m_total_nb_element,rhs.m_total_nb_element);
196  std::swap(m_memory_ressource,rhs.m_memory_ressource);
197  }
198  Int64 capacity() const { return m_data.capacity(); }
199  eMemoryRessource memoryRessource() const { return m_memory_ressource; }
200  Span<std::byte> bytes() { return asWritableBytes(to1DSpan()); }
201  Span<const std::byte> bytes() const { return asBytes(to1DSpan()); }
202  public:
204  DataType* _internalData() { return m_span._internalData(); }
205  protected:
206  SpanType m_span;
207  ArrayWrapper m_data;
208  Int64 m_total_nb_element = 0;
209  eMemoryRessource m_memory_ressource = eMemoryRessource::Unknown;
210 };

237  private:
238  using BaseClass::m_span;
239  public:
241  NumArray() : NumArray(0){}
242  explicit NumArray(eMemoryRessource r) : BaseClass(r){}
244  explicit NumArray(Int32 dim1_size)
245  : BaseClass(ArrayExtents<1>(dim1_size)){}

247  : BaseClass(ArrayExtents<1>{dim1_size},r){}
248  public:
249  void resize(Int32 dim1_size)
250  {
251  this->resize(ArrayExtents<1>(dim1_size));
252  }
253  public:
255  constexpr Int32 dim1Size() const { return this->extent(0); }
256  public:
258  DataType operator()(Int32 i) const { return m_span(i); }
260  DataType& s(Int32 i) { return m_span(i); }
262  DataType& operator[](Int32 i) { return m_span(i); }
264  DataType operator[](Int32 i) const { return m_span(i); }
265  public:
266  constexpr operator SpanType () { return this->span(); }
267  constexpr operator ConstSpanType () const { return this->constSpan(); }
