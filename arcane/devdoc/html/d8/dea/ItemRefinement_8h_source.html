66  struct FaceSetCompare
67  {
68  typedef std::set<Int64> FaceSet;
69  bool operator()(const FaceSet& s1, const FaceSet& s2) const
70  {
71  //return (s1 == s2);
72  if (s1.size() != s2.size())
73  return false;
74  else {
75  FaceSet::const_iterator its1 = s1.begin(), its2 = s2.begin();
76  FaceSet::const_iterator its1_end = s1.end();
77  for(; its1 != its1_end; ++its1,++its2)
78  if (*its1 != *its2)
79  return false;
80  }
81  return true;
82  }
83  };
