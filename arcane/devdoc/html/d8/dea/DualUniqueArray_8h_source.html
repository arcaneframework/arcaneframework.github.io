264 void _checkUpdateDeviceView()
265 {
266 if (!m_is_valid_numarray) {
267 _checkCreateNumArray();
268 MDSpan<DataType, MDDim1> s(m_array.data(), ArrayIndex<1>(m_array.size()));
269 m_device_array->copy(s);
270 m_is_valid_numarray = true;
271 }
272 m_device_view = *(m_device_array.get());
273 }
274 void _checkUpdateHostView()
275 {
276 if (!m_is_valid_array) {
277 _checkCreateNumArray();
278 SmallSpan<const DataType> device_view = *(m_device_array.get());
279 m_array.resize(device_view.size());
280 _memoryCopy(asBytes(device_view), asWritableBytes(m_array.span()));
281 m_is_valid_array = true;
282 }
283 }
284 void _checkCreateNumArray()
285 {
286 if (!m_device_array) {
287 m_device_array = std::make_unique<NumArrayType>(eMemoryRessource::Device);
288 }
289 }
