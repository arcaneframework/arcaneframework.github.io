145void ParallelDataWriter::Impl::
146sort(Int32ConstArrayView local_ids,Int64ConstArrayView items_uid)
147{
148 IParallelMng* pm = m_parallel_mng;
149
150 Parallel::BitonicSort<Int64> uid_sorter(pm);
151 uid_sorter.sort(items_uid);
152
153 ConstArrayView<Int32> key_indexes = uid_sorter.keyIndexes();
154 ConstArrayView<Int32> key_ranks = uid_sorter.keyRanks();
155 ConstArrayView<Int64> keys = uid_sorter.keys();
156
157 UniqueArray<Int64> global_all_keys;
158 UniqueArray<Int32> global_all_key_indexes;
159 UniqueArray<Int32> global_all_key_ranks;
160
161 Int32 nb_item = keys.size();
162 const Int32 my_rank = pm->commRank();
163 const bool is_verbose = m_is_verbose;
164 if (is_verbose) {
165 for (Integer i = 0; i < math::min(nb_item, 20); ++i) {
166 info() << "ORIGINAL I=" << i << " UID=" << items_uid[i]
167 << " INDEX=" << key_indexes[i]
168 << " RANK=" << key_ranks[i]
169 << " KEY=" << keys[i];
170 }
171 }
172
173 if (m_gather_all){
174 // Le proc 0 récupère tout
175 pm->allGatherVariable(keys,global_all_keys);
176 pm->allGatherVariable(key_indexes,global_all_key_indexes);
177 pm->allGatherVariable(key_ranks,global_all_key_ranks);
178 Int32 gather_rank = 0;
179
180 if (pm->commRank()!=gather_rank){
181 global_all_key_ranks.clear();
182 global_all_key_indexes.clear();
183 global_all_keys.clear();
184 }
185 nb_item = global_all_keys.size();
186
187 key_ranks = global_all_key_ranks.view();
188 key_indexes = global_all_key_indexes.view();
189 keys = global_all_keys.view();
190 }
191
192 m_nb_item = nb_item;
193
194 m_sorted_unique_ids.resize(nb_item);
195 m_sorted_unique_ids.copy(keys);
196
197 //info() << "END SORT SIZE=" << nb_item << " KEY_SIZE=" << keys.size();
198 if (is_verbose) {
199 for (Integer i = 0; i < math::min(nb_item, 20); ++i) {
200 info() << "I=" << i << " KEY=" << keys[i]
201 << " INDEX=" << key_indexes[i]
202 << " RANK=" << key_ranks[i];
203 }
204 }
205 {
206 UniqueArray<UniqueArray<Int32>> indexes_list(pm->commSize());
207 UniqueArray<UniqueArray<Int32>> own_indexes_list(pm->commSize());
208 auto sd_exchange { ParallelMngUtils::createExchangerRef(pm) };
209
210 for( Integer i=0; i<nb_item; ++i ){
211 Int32 index = key_indexes[i];
212 Int32 rank = key_ranks[i];
213 if (rank != my_rank && indexes_list[rank].empty())
214 sd_exchange->addSender(rank);
215 indexes_list[rank].add(index);
216 own_indexes_list[rank].add(i);
217 }
218 m_local_indexes_to_recv = own_indexes_list[my_rank];
219 m_local_indexes_to_send.resize(indexes_list[my_rank].size());
220
221 sd_exchange->initializeCommunicationsMessages();
222 //info() << "NB SEND=" << sd_exchange->nbSender()
223 // << " NB_RECV=" << sd_exchange->nbReceiver();
224 Int32ConstArrayView send_sd = sd_exchange->senderRanks();
225 Integer nb_send = send_sd.size();
226 m_indexes_to_recv.resize(nb_send);
227 m_ranks_to_recv.resize(nb_send);
228 for( Integer i=0; i<nb_send; ++i ){
229 //info() << " SEND TO A: rank=" << send_sd[i];
230 ISerializeMessage* send_msg = sd_exchange->messageToSend(i);
231 Int32 dest_rank = send_sd[i];
232 ISerializer* serializer = send_msg->serializer();
233 m_indexes_to_recv[i] = own_indexes_list[dest_rank];
234 m_ranks_to_recv[i] = dest_rank;
235 serializer->setMode(ISerializer::ModeReserve);
236
237 serializer->reserveArray(indexes_list[dest_rank]);
238
239 serializer->allocateBuffer();
240 serializer->setMode(ISerializer::ModePut);
241
242 serializer->putArray(indexes_list[dest_rank]);
243 if (is_verbose) {
244 Integer nb_to_send = indexes_list[dest_rank].size();
245 for (Integer z = 0; z < nb_to_send; ++z) {
246 Integer index = indexes_list[dest_rank][z];
247 info() << " SEND Z=" << z << " RANK=" << dest_rank << " index=" << index;
248 }
249 }
250 }
251 sd_exchange->processExchange();
252
253 ConstArrayView<Int32> recv_sd = sd_exchange->receiverRanks();
254 const Int32 nb_recv = recv_sd.size();
255 m_indexes_to_send.resize(nb_recv);
256 m_ranks_to_send.resize(nb_recv);
257 for( Integer i=0; i<nb_recv; ++i ){
258 //info() << " RECEIVE FROM A: rank=" << recv_sd[i];
259 ISerializeMessage* recv_msg = sd_exchange->messageToReceive(i);
260 Int32 orig_rank = recv_sd[i];
261 ISerializer* serializer = recv_msg->serializer();
262 serializer->setMode(ISerializer::ModeGet);
263 Int32Array& recv_indexes = m_indexes_to_send[i];
264 m_ranks_to_send[i] = orig_rank;
265 serializer->getArray(recv_indexes);
266 const Int32 nb_to_recv = recv_indexes.size();
267 //info() << " RECEIVE FROM A: NB_TO_RECEIVE " << nb_to_recv << " S2=" << own_group_local_ids.size();
268 for( Integer z=0; z<nb_to_recv; ++z ){
269 Int32 index = recv_indexes[z];
270 //info() << " RECV Z=" << z << " RANK=" << orig_rank << " index=" << index
271 // << " index2=" << own_group_local_ids[index];
272 recv_indexes[z] = local_ids[index];
273 }
274 //info() << "END RECEIVE FROM A: NB_TO_RECEIVE " << nb_to_recv;
275 }
276
277 // Traite les entités locales
278 {
279 const Int32 nb_local = m_local_indexes_to_send.size();
280 for (Int32 z = 0; z < nb_local; ++z) {
281 Int32 index = indexes_list[my_rank][z];
282 //info() << " RECV Z=" << z << " RANK=" << orig_rank << " index=" << index
283 // << " index2=" << own_group_local_ids[index];
284 m_local_indexes_to_send[z] = local_ids[index];
285 }
286 }
287
288 }
289}

294Ref<IData> ParallelDataWriter::Impl::
295getSortedValues(IData* data)
296{
297 IParallelMng* pm = m_parallel_mng;
298 Ref<IData> sorted_data = data->cloneEmptyRef();
299
300 auto sd_exchange { ParallelMngUtils::createExchangerRef(pm) };
301 for (Int32 rank_to_send : m_ranks_to_send)
302 sd_exchange->addSender(rank_to_send);
303
304 UniqueArray<Int32> ranks_to_recv2;
305 for (Int32 rank_to_receive : m_ranks_to_recv)
306 ranks_to_recv2.add(rank_to_receive);
307
308 sd_exchange->initializeCommunicationsMessages(ranks_to_recv2);
309 Int32ConstArrayView send_sd = sd_exchange->senderRanks();
310 const Int32 nb_send = send_sd.size();
311 for( Integer i=0; i<nb_send; ++i ){
312 //info() << " SEND TO B: rank=" << send_sd[i];
313 ISerializeMessage* send_msg = sd_exchange->messageToSend(i);
314 //Int32 dest_rank = send_sd[i];
315 ISerializer* serializer = send_msg->serializer();
316 serializer->setMode(ISerializer::ModeReserve);
317 data->serialize(serializer,m_indexes_to_send[i],0);
318 serializer->allocateBuffer();
319 serializer->setMode(ISerializer::ModePut);
320 data->serialize(serializer,m_indexes_to_send[i],0);
321 }
322
323 sd_exchange->processExchange();
324
325 Int32ConstArrayView recv_sd = sd_exchange->receiverRanks();
326 const Int32 nb_recv = recv_sd.size();
327 sorted_data->resize(m_nb_item);
328 for( Integer i=0; i<nb_recv; ++i ){
329 //info() << " RECEIVE FROM B: rank=" << recv_sd[i];
330 ISerializeMessage* recv_msg = sd_exchange->messageToReceive(i);
331 //Int32 orig_rank = recv_sd[i];
332 ISerializer* serializer = recv_msg->serializer();
333 serializer->setMode(ISerializer::ModeGet);
334 sorted_data->serialize(serializer,m_indexes_to_recv[i],0);
335 }
336
337 // Traite les données qui sont déjà présentes sur ce processeur.
338 {
339 ConstArrayView<Int32> local_recv_indexes = m_local_indexes_to_recv;
340 const Int32 nb_local_index = local_recv_indexes.size();
341 if (nb_local_index>0){
342 SerializeBuffer sbuf;
343 sbuf.setMode(ISerializer::ModeReserve);
344 data->serialize(&sbuf, m_local_indexes_to_send, nullptr);
345 sbuf.allocateBuffer();
346 sbuf.setMode(ISerializer::ModePut);
347 data->serialize(&sbuf, m_local_indexes_to_send, nullptr);
348 sbuf.setMode(ISerializer::ModeGet);
349 sorted_data->serialize(&sbuf, local_recv_indexes, nullptr);
350 }
351 }
352 return sorted_data;
353}

361Ref<ParallelDataWriter> ParallelDataWriterList::
362getOrCreateWriter(const ItemGroup& group)
363{
364 auto i = m_data_writers.find(group);
365 if (i != m_data_writers.end())
366 return i->second;
367 IParallelMng* pm = group.itemFamily()->parallelMng();
368 Ref<ParallelDataWriter> writer = makeRef(new ParallelDataWriter(pm));
369 {
370 Int64UniqueArray items_uid;
371 ItemGroup own_group = group.own();
372 MeshUtils::fillUniqueIds(own_group.view(), items_uid);
373 Int32ConstArrayView local_ids = own_group.internal()->itemsLocalId();
374 writer->sort(local_ids, items_uid);
375 }
376 m_data_writers.try_emplace(group, writer);
377 return writer;
378}
