298void ItemFamily::
299build()
300{
301 m_item_type_mng = m_mesh->itemTypeMng();
302
303 // Créé le modificateur de topologie si cela n'a pas encore été fait par
304 // la classe dérivée.
305 if (!m_topology_modifier)
306 m_topology_modifier = new AbstractItemFamilyTopologyModifier(this);
307
308 m_full_name = m_mesh->name() + "_" + m_name;
309
310 m_local_connectivity_info = new ItemConnectivityInfo();
311 m_global_connectivity_info = new ItemConnectivityInfo();
312
313 IParallelMng* pm = m_mesh->parallelMng();
314 if (!pm->isParallel()){
315 m_default_sub_domain_owner = 0;
316 m_is_parallel = false;
317 }
318 else
319 m_is_parallel = true;
320
321 // D'abord initialiser les infos car cela créé les groupes d'entités
322 // et c'est indispensable avant de créer des variables dessus.
323 m_infos.build();
324
325 // Construit l'instance qui contiendra les variables
326 // NOTE: si on change les noms ici, il faut aussi les changer dans MeshStats.cc
327 // sinon les statistiques ne seront pas fiables.
328 {
329 String var_unique_ids_name(_variableName("FamilyUniqueIds"));
330 String var_owner_name(_variableName("FamilyOwner"));
331 String var_flags_name(_variableName("FamilyFlags"));
332 String var_typeid_name(_variableName("FamilyItemTypeId"));
333 String var_nb_parent_name(_variableName("FamilyItemNbParent"));
334 String var_count_name(_variableName("FamilyItemsShared"));
335 String var_groups_name(_variableName("FamilyGroupsName"));
336 String var_current_id_name(_variableName("FamilyCurrentId"));
337 String var_new_owner_name(_variableName("FamilyNewOwnerName"));
338 String var_parent_mesh_name(_variableName("ParentMeshName"));
339 String var_parent_family_name(_variableName("ParentFamilyName"));
340 String var_parent_family_depth_name(_variableName("ParentFamilyDepthName"));
341 String var_child_meshes_name(_variableName("ChildMeshesName"));
342 String var_child_families_name(_variableName("ChildFamiliesName"));
343 m_internal_variables = new Variables(m_mesh,name(),itemKind(),var_count_name,
344 var_unique_ids_name,var_owner_name,
345 var_flags_name,var_typeid_name,var_nb_parent_name,var_groups_name,
346 var_current_id_name,var_new_owner_name,
347 var_parent_mesh_name,var_parent_family_name,
348 var_parent_family_depth_name,
349 var_child_meshes_name,
350 var_child_families_name);
351 // Il ne faut pas utiliser ces tableaux pour accéder aux entités car il y a un décalage
352 // de 1 pour gérer l'entité nulle. Il faut passer par la vue associée
353 // qui est dans m_common_item_shared_info.
354 m_items_unique_id = &m_internal_variables->m_items_unique_id._internalTrueData()->_internalDeprecatedValue();
355 m_items_owner = &m_internal_variables->m_items_owner._internalTrueData()->_internalDeprecatedValue();
356 m_items_flags = &m_internal_variables->m_items_flags._internalTrueData()->_internalDeprecatedValue();
357 m_items_type_id = &m_internal_variables->m_items_type_id._internalTrueData()->_internalDeprecatedValue();
358 m_items_nb_parent = &m_internal_variables->m_items_nb_parent._internalTrueData()->_internalDeprecatedValue();
359 _updateItemViews();
360 }
361
362 m_variable_synchronizer = ParallelMngUtils::createSynchronizerRef(pm,this);
363
364 m_item_sort_function = _defaultItemSortFunction();
365
366 {
367 String s = platform::getEnvironmentVariable("ARCANE_USE_LEGACY_COMPACT_ITEMS");
368 if (s=="TRUE" || s=="1"){
369 info() << "WARNING: Using legacy 'compactItem()' without compactReference()'";
370 m_use_legacy_compact_item = true;
371 }
372 }

542
543bool ItemFamily::
544_partialEndUpdate()
545{
546 bool need_end_update = m_infos.changed();
547 info(4) << "ItemFamily::endUpdate() " << fullName() << " need_end_update?=" << need_end_update;
548 if (!need_end_update){
549 // Même si aucune entité n'est ajoutée ou supprimée, si \a m_need_prepare_dump
550 // est vrai cela signifie que quelque chose a changé dans la famille. Dans ce
551 // cas il est préférable d'incrémenter le current_id. Sans cela, si on appelle
552 // readFromDump() (par exemple suite à un retour-arrière), les données ne seront
553 // pas restaurées si entre temps current_id n'a pas changé.
554 if (m_need_prepare_dump){
555 _computeConnectivityInfo(m_local_connectivity_info);
556 ++m_current_id;
557 }
558 return true;
559 }
560 m_item_need_prepare_dump = true;
561 _computeConnectivityInfo(m_local_connectivity_info);
562 ++m_current_id;
563
564 // Update "external" connectivities
565 if (m_connectivity_mng)
566 m_connectivity_mng->setModifiedItems(this,m_infos.addedItems(), m_infos.removedItems());
567 //
568 m_infos.finalizeMeshChanged();
569
570 return false;
571}

1113 if (dump_version<0x0307)

1117 // remettre à jour. De même, si on passe d'un maillage séquentiel à un
1118 // maillage en plusieurs parties en reprise, il faut supprimer le isOwn()

1126 // entites sont consécutifs et croissants avec le localId() des entités

1129 // Cette condition est vrai si compactReferences() a été appelé.

1131 // ajouter une variable data_index sur les entités.

1133 if (!use_type_variable)
1134 nb_item = items_shared_data_index.size();
1135

1137 << " count=" << nb_item
1138 << " currentid=" << m_current_id

1160 _checkComputeSynchronizeInfos(0);
1161 _readGroups();
1162 _invalidateComputedGroups();

1164 }
1165
1166 m_current_id = m_internal_variables->m_current_id();
1167 // IMPORTANT: remise à zéro pour obligatoirement redimensionner les variables
1168 // au prochain ajout d'entités.

1170
1171 // Données de liaison de famille
1172 {
1173 IMeshMng* mesh_mng = m_mesh->meshMng();

1214 }
1215 }
1216
1217 // Supprime les entités du groupe total car elles vont être remises à jour
1218 // lors de l'appel à _endUpdate()

1225 // Recréation des groupes si nécessaire
1226 _readGroups();
1227
1228 // Invalide les groupes recalculés
1229 _invalidateComputedGroups();
1230
1231 _endUpdate(false);
1232
1233 _checkComputeSynchronizeInfos(1);

2022/*---------------------------------------------------------------------------*/
2023
2024void ItemFamily::
2025removeItems2(ItemDataList& item_data_list)
2026{
2027 if (!m_mesh->itemFamilyNetwork())
2028 ARCANE_FATAL("Family name='{0}': IMesh::itemFamilyNetwork() is null",name());
2029
2030 ItemData& item_data = item_data_list[itemKind()];
2031 // 1-Prepare : Get child connectivities and families
2032 SharedArray<IIncrementalItemConnectivity*> child_connectivities = m_mesh->itemFamilyNetwork()->getChildDependencies(this); // TODO : change return type ? (List to Array ?)
2033 UniqueArray<IItemFamily*> child_families;
2034 UniqueArray<IIncrementalItemConnectivity*> child_families_to_current_family;
2035 UniqueArray<ItemScalarProperty<bool>> child_families_has_extra_parent_properties; // To indicate wheter child items have other parents.
2036 Integer index = 0;
2037 for (auto child_connectivity : child_connectivities) {
2038 child_families.add(child_connectivity->targetFamily());
2039 child_families_to_current_family.add(m_mesh->itemFamilyNetwork()->getConnectivity(child_families.back(),this,connectivityName(child_families.back(),this)));
2040 if (child_families_to_current_family.back() == nullptr) fatal() << "removeItems2 needs reverse connectivity. Missing Connectivity " << connectivityName(child_families.back(),this);
2041 child_families_has_extra_parent_properties.add(ItemScalarProperty<bool>());
2042 child_families_has_extra_parent_properties.back().resize(child_families.back(),false);
2043 for (auto parent_connectivity : m_mesh->itemFamilyNetwork()->getParentDependencies(child_families.back())) { // exclure parent actuel
2044 if (parent_connectivity == child_connectivity) continue;
2045 ItemVector connected_items;
2046 _getConnectedItems(parent_connectivity,connected_items);
2047 _fillHasExtraParentProperty(child_families_has_extra_parent_properties[index],connected_items);
2048 }
2049 index++;
2050 }
2051 // 2-Propagates item removal to child families
2052 Int64ArrayView removed_item_lids = item_data.itemInfos().view(); // Todo change ItemData to store removed item lids in Int32
2053 for (auto removed_item_lid_int64 : removed_item_lids) {
2054 Int32 removed_item_lid = CheckedConvert::toInt32(removed_item_lid_int64);
2055 index = 0;
2056 for (auto child_connectivity : child_connectivities) {
2057 ConnectivityItemVector child_con_accessor(child_connectivity);
2058 ENUMERATE_ITEM(connected_item, child_con_accessor.connectedItems(ItemLocalId(removed_item_lid))) {
2059 if (!this->itemsInternal()[removed_item_lid]->isDetached()) {// test necessary when doing removeDetached (the relations are already deleted).
2060 child_families_to_current_family[index]->removeConnectedItem(ItemLocalId(connected_item),ItemLocalId(removed_item_lid));
2061 }
2062 // Check if connected item is to remove
2063 if (! child_families_has_extra_parent_properties[index][connected_item] && child_families_to_current_family[index]->nbConnectedItem(ItemLocalId(connected_item)) == 0) {
2064 item_data_list[child_connectivity->targetFamily()->itemKind()].itemInfos().add((Int64) connected_item.localId());
2065 }
2066 }
2067 index++;
2068 }
2069 }
2070 // => merge this loop with previous one ?
2071 // 3-1 Remove relations for child relations
2072 for (auto removed_item_lid_int64 : removed_item_lids) {
2073 Int32 removed_item_lid = CheckedConvert::toInt32(removed_item_lid_int64);
2074 for (auto child_relation : m_mesh->itemFamilyNetwork()->getChildRelations(this)) {
2075 ConnectivityItemVector connectivity_accessor(child_relation);
2076 ENUMERATE_ITEM(connected_item, connectivity_accessor.connectedItems(ItemLocalId(removed_item_lid))) {
2077 child_relation->removeConnectedItem(ItemLocalId(removed_item_lid),ItemLocalId(connected_item));
2078 }
2079 }
2080 }
2081 // 3-2 Remove relations for parent relations
2082 ItemScalarProperty<bool> is_removed_item;
2083 is_removed_item.resize(this,false);
2084 for (auto removed_item_lid_int64 : removed_item_lids) {
2085 Int32 removed_item_lid = CheckedConvert::toInt32(removed_item_lid_int64);
2086 is_removed_item[*(this->itemsInternal()[removed_item_lid])] = true;
2087 }
2088 for (auto parent_relation : m_mesh->itemFamilyNetwork()->getParentRelations(this)) {
2089 for (auto source_item : parent_relation->sourceFamily()->itemsInternal()) {
2090 if (source_item->isSuppressed()) continue;
2091 ConnectivityItemVector connectivity_accessor(parent_relation);
2092 ENUMERATE_ITEM(connected_item, connectivity_accessor.connectedItems(ItemLocalId(source_item))) {
2093 if (is_removed_item[connected_item])
2094 parent_relation->removeConnectedItem(ItemLocalId(source_item),connected_item);
2095 }
2096 }
2097 }
2098 // 4-Remove items. Child items will be removed by an automatic call of removeItems2 on their family...
2099 for (auto removed_item_lid_int64 : removed_item_lids) {
2100 Int32 removed_item_lid = CheckedConvert::toInt32(removed_item_lid_int64);
2101 ItemInternal* removed_item = m_infos.itemInternal(removed_item_lid);
2102 if (removed_item->isDetached()) {
2103 m_infos.removeDetachedOne(removed_item);
2104 }
2105 else {
2106 m_infos.removeOne(removed_item);
2107 }
2108 }
2109 this->endUpdate();// endUpdate is needed since we then go deeper in the dependency graph and will need to enumerate this changed family.
2110}
