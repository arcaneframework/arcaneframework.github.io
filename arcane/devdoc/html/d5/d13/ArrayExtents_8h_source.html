124  ARCCORE_HOST_DEVICE Int64 totalStride() const
125  {

181 template<int RankValue>
182 class ArrayExtentsBase
183 {
184  public:
185  ARCCORE_HOST_DEVICE constexpr ArrayExtentsBase()
186  : m_extents(detail::ArrayExtentsTraits<RankValue>::extendsInitHelper()) { }
187  protected:
188  explicit ARCCORE_HOST_DEVICE ArrayExtentsBase(SmallSpan<const Int32> extents)
189  {
190  Integer n = extents.size();
191  Integer vn = math::min(n,RankValue);
192  for( int i=0; i<vn; ++i )
193  m_extents[i] = extents[i];
194  for( int i=vn; i<RankValue; ++i )
195  m_extents[i] = 0;
196  }
197  public:
199  ARCCORE_HOST_DEVICE Int32 extent(int i) const { return m_extents[i]; }
201  ARCCORE_HOST_DEVICE void setExtent(int i,Int32 v) { m_extents[i] = v; }
202  ARCCORE_HOST_DEVICE Int32 operator()(int i) const { return m_extents[i]; }
203  ARCCORE_HOST_DEVICE SmallSpan<const Int32> asSpan() const { return { m_extents.data(), RankValue }; }
204  ARCCORE_HOST_DEVICE std::array<Int32,RankValue> asStdArray() const { return m_extents; }
206  ARCCORE_HOST_DEVICE constexpr Int64 totalNbElement() const
207  {
208  Int64 nb_element = 1;
209  for (int i=0; i<RankValue; i++)
210  nb_element *= m_extents[i];
211  return nb_element;
212  }
213  // Instance contenant les dimensions après la première
214  ARCCORE_HOST_DEVICE ArrayExtentsBase<RankValue-1> removeFirstExtent() const
215  {
216  return ArrayExtentsBase<RankValue-1>::fromSpan({m_extents.data()+1,RankValue-1});
217  }
221  ARCCORE_HOST_DEVICE static ArrayExtentsBase<RankValue> fromSpan(SmallSpan<const Int32> extents)
222  {
223  return ArrayExtentsBase<RankValue>(extents);
224  }
225  protected:
226  std::array<Int32,RankValue> m_extents;
227 };
