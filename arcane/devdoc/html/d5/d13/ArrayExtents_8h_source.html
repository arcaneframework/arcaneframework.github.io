66 template<int RankValue>
67 class ArrayStridesBase
68 {
69  public:
70  ARCCORE_HOST_DEVICE ArrayStridesBase()
71  {
72  for( int i=0; i<RankValue; ++i )
73  m_strides[i] = 0;
74  }
76  ARCCORE_HOST_DEVICE Int32 stride(int i) const { return m_strides[i]; }
77  ARCCORE_HOST_DEVICE Int32 operator()(int i) const { return m_strides[i]; }
78  ARCCORE_HOST_DEVICE SmallSpan<const Int32> asSpan() const { return { m_strides, RankValue }; }
80  ARCCORE_HOST_DEVICE Int64 totalStride() const
81  {
82  Int64 nb_element = 1;
83  for (int i=0; i<RankValue; i++)
84  nb_element *= m_strides[i];
85  return nb_element;
86  }
87  // Instance contenant les dimensions après la première
88  ARCCORE_HOST_DEVICE ArrayStridesBase<RankValue-1> removeFirstStride() const
89  {
90  return ArrayStridesBase<RankValue-1>::fromSpan({m_strides+1,RankValue-1});
91  }
96  ARCCORE_HOST_DEVICE static ArrayStridesBase<RankValue> fromSpan(Span<const Int32> strides)
97  {
98  ArrayStridesBase<RankValue> v;
99  // TODO: vérifier la taille
100  for( int i=0; i<RankValue; ++i )
101  v.m_strides[i] = strides[i];
102  return v;
103  }
104  protected:
105  Int32 m_strides[RankValue];
106 };
