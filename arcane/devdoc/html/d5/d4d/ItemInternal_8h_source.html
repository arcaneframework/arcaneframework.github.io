87  struct Int32View
88  {
89  public:
90  Int32 operator[](Int32 index) const
91  {
92 #ifdef ARCANE_CHECK
93  if (index==NULL_ITEM_LOCAL_ID){
94  // Pour l'entité nulle, la taille doit être 0.
95  if (m_size!=0)
96  arcaneRangeError(index,m_size);

98  else

100 #endif
101  return m_data[index];

104  {
105  m_data = v.data();
106  m_size = v.size();
107  }

109  void setNull(Int32* data)
110  {
111  m_data = data;
112  m_size = 0;
113  }
114  operator Int32ConstArrayView() const
115  {
116  return Int32ConstArrayView(m_size,m_data);
117  }
118  Int32ArrayView toMutableView() const
119  {
120  return Int32ArrayView(m_size,m_data);
121  }
122  private:
123  Int32 m_size;
124  Int32* m_data;

366  II_HasBackCell = 1 << 3,
367  II_FrontCellIsFirst = 1 << 4,
368  II_BackCellIsFirst = 1 << 5,
369  II_Own = 1 << 6,
370  II_Added = 1 << 7,
371  II_Suppressed = 1 << 8,
372  II_Shared = 1 << 9,
373  II_SubDomainBoundary = 1 << 10,
374  //II_JustRemoved = 1 << 11, //!< L'entité vient d'être supprimé
375  II_JustAdded = 1 << 12,
376  II_NeedRemove = 1 << 13,
377  II_SlaveFace = 1 << 14,
378  II_MasterFace = 1 << 15,
379  II_Detached = 1 << 16,
380  II_HasTrace = 1 << 17,
381 
382  II_Coarsen = 1 << 18,
383  II_DoNothing = 1 << 19,
384  II_Refine = 1 << 20,
385  II_JustRefined = 1 << 21,
386  II_JustCoarsened = 1 << 22,
387  II_Inactive = 1 << 23,
388  II_CoarsenInactive = 1 << 24,
389 
390  II_UserMark1 = 1 << 25,
391  II_UserMark2 = 1 << 26

411  ItemUniqueId uniqueId() const
412  {
413 #ifdef ARCANE_CHECK
414  if (m_local_id!=NULL_ITEM_LOCAL_ID)
415  arcaneCheckAt((Integer)m_local_id,m_shared_info->m_unique_ids->size());
416 #endif
417  // Ne pas utiliser l'accesseur normal car ce tableau peut etre utilise pour la maille
418  // nulle et dans ce cas m_local_id vaut NULL_ITEM_LOCAL_ID (qui est negatif)

420  return ItemUniqueId(m_shared_info->m_unique_ids->data()[m_local_id]);
