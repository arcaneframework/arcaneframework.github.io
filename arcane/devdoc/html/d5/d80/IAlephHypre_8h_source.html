114 class AlephVectorHypre
115 : public IAlephVector
116 {
117  public:
118  AlephVectorHypre(ITraceMng* tm, AlephKernel* kernel, Integer index)
119  : IAlephVector(tm, kernel, index)
120  , m_hypre_ijvector(0)
121  , m_hypre_parvector(0)
122  , jSize(0)
123  , jUpper(0)
124  , jLower(-1)
125  {
126  debug() << "[AlephVectorHypre::AlephVectorHypre] new SolverVectorHypre";
127  }
128 
129  public:
130  /******************************************************************************
131  * The Create() routine creates an empty vector object that lives on the comm communicator. This is
132  * a collective call, with each process passing its own index extents, jLower and jupper. The names
133  * of these extent parameters begin with a j because we typically think of matrix-vector multiplies
134  * as the fundamental operation involving both matrices and vectors. For matrix-vector multiplies,
135  * the vector partitioning should match the column partitioning of the matrix (which also uses the j
136  * notation). For linear system solves, these extents will typically match the row partitioning of the
137  * matrix as well.
138  *****************************************************************************/
139  void AlephVectorCreate(void)
140  {
141  debug() << "[AlephVectorHypre::AlephVectorCreate] HYPRE VectorCreate";
142  void* object;
143 
144  for (int iCpu = 0; iCpu < m_kernel->size(); ++iCpu) {
145  if (m_kernel->rank() != m_kernel->solverRanks(m_index)[iCpu])
146  continue;
147  debug() << "[AlephVectorHypre::AlephVectorCreate] adding contibution of core #" << iCpu;
148  if (jLower == -1)
149  jLower = m_kernel->topology()->gathered_nb_row(iCpu);
150  jUpper = m_kernel->topology()->gathered_nb_row(iCpu + 1) - 1;
151  }
152 
153  debug() << "[AlephVectorHypre::AlephVectorCreate] jLower=" << jLower << ", jupper=" << jUpper;
154  // Mise à jour de la taille locale du buffer pour le calcul plus tard de la norme max, par exemple
155  jSize = jUpper - jLower + 1;
156 
157  hypreCheck("IJVectorCreate",
158  HYPRE_IJVectorCreate(MPI_COMM_SUB,
159  jLower,
160  jUpper,
161  &m_hypre_ijvector));
162 
163  debug() << "[AlephVectorHypre::AlephVectorCreate] HYPRE IJVectorSetObjectType";
164  hypreCheck("IJVectorSetObjectType", HYPRE_IJVectorSetObjectType(m_hypre_ijvector, HYPRE_PARCSR));
165 
166  debug() << "[AlephVectorHypre::AlephVectorCreate] HYPRE IJVectorInitialize";
167  hypreCheck("HYPRE_IJVectorInitialize", HYPRE_IJVectorInitialize(m_hypre_ijvector));
168 
169  HYPRE_IJVectorGetObject(m_hypre_ijvector, &object);
170  m_hypre_parvector = (HYPRE_ParVector)object;
171  debug() << "[AlephVectorHypre::AlephVectorCreate] done";
172  }

199  /******************************************************************************
200  * norm_max
201  *****************************************************************************/
202  Real norm_max()
203  {
204  Real normInf = 0.0;
205  UniqueArray<HYPRE_Int> bfr_idx(jSize);
206  UniqueArray<double> bfr_val(jSize);
207 
208  for (HYPRE_Int i = 0; i < jSize; ++i)
209  bfr_idx[i] = jLower + i;
210 
211  hypreCheck("HYPRE_IJVectorGetValues", HYPRE_IJVectorGetValues(m_hypre_ijvector, jSize, bfr_idx.unguardedBasePointer(), bfr_val.unguardedBasePointer()));
212  for (HYPRE_Int i = 0; i < jSize; ++i) {
213  const Real abs_val = math::abs(bfr_val[i]);
214  if (abs_val > normInf)
215  normInf = abs_val;
216  }
217  normInf = m_kernel->subParallelMng(m_index)->reduce(Parallel::ReduceMax, normInf);
218  return normInf;
219  }

246  /******************************************************************************

249  AlephMatrixHypre(ITraceMng* tm, AlephKernel* kernel, Integer index)
250  : IAlephMatrix(tm, kernel, index)
251  , m_hypre_ijmatrix(0)
252  {
253  debug() << "[AlephMatrixHypre] new AlephMatrixHypre";
254  }
255 
256  ~AlephMatrixHypre()
257  {
258  debug() << "[~AlephMatrixHypre]";
259  }
260 
261  public:
262  /******************************************************************************
263  * Each submatrix Ap is "owned" by a single process and its first and last row numbers are
264  * given by the global indices ilower and iupper in the Create() call below.
265  *******************************************************************************
266  * The Create() routine creates an empty matrix object that lives on the comm communicator. This
267  * is a collective call (i.e., must be called on all processes from a common synchronization point),
268  * with each process passing its own row extents, ilower and iupper. The row partitioning must be
269  * contiguous, i.e., iupper for process i must equal ilower-1 for process i+1. Note that this allows
270  * matrices to have 0- or 1-based indexing. The parameters jlower and jupper define a column
271  * partitioning, and should match ilower and iupper when solving square linear systems.
272  *****************************************************************************/
273  void AlephMatrixCreate(void)
274  {
275  debug() << "[AlephMatrixHypre::AlephMatrixCreate] HYPRE MatrixCreate idx:" << m_index;
276  void* object;
277  Integer ilower = -1;
278  Integer iupper = 0;
279  for (int iCpu = 0; iCpu < m_kernel->size(); ++iCpu) {
280  if (m_kernel->rank() != m_kernel->solverRanks(m_index)[iCpu])
281  continue;
282  if (ilower == -1)
283  ilower = m_kernel->topology()->gathered_nb_row(iCpu);
284  iupper = m_kernel->topology()->gathered_nb_row(iCpu + 1) - 1;
285  }
286  debug() << "[AlephMatrixHypre::AlephMatrixCreate] ilower=" << ilower << ", iupper=" << iupper;
287 
288  Integer jlower = ilower; //0;
289  Integer jupper = iupper; //m_kernel->topology()->gathered_nb_row(m_kernel->size())-1;
290  debug() << "[AlephMatrixHypre::AlephMatrixCreate] jlower=" << jlower << ", jupper=" << jupper;
291 
292  hypreCheck("HYPRE_IJMatrixCreate",
293  HYPRE_IJMatrixCreate(MPI_COMM_SUB,
294  ilower, iupper,
295  jlower, jupper,
296  &m_hypre_ijmatrix));
297 
298  debug() << "[AlephMatrixHypre::AlephMatrixCreate] HYPRE IJMatrixSetObjectType";
299  HYPRE_IJMatrixSetObjectType(m_hypre_ijmatrix, HYPRE_PARCSR);
300  debug() << "[AlephMatrixHypre::AlephMatrixCreate] HYPRE IJMatrixSetRowSizes";
301  HYPRE_IJMatrixSetRowSizes(m_hypre_ijmatrix, m_kernel->topology()->gathered_nb_row_elements().data());
302  debug() << "[AlephMatrixHypre::AlephMatrixCreate] HYPRE IJMatrixInitialize";
303  HYPRE_IJMatrixInitialize(m_hypre_ijmatrix);
304  HYPRE_IJMatrixGetObject(m_hypre_ijmatrix, &object);
305  m_hypre_parmatrix = (HYPRE_ParCSRMatrix)object;
306  }

337  /******************************************************************************
338  * isAlreadySolved
339  *****************************************************************************/
340  bool isAlreadySolved(AlephVectorHypre* x,
341  AlephVectorHypre* b,
342  AlephVectorHypre* tmp,
343  Real* residual_norm,
344  AlephParams* params)
345  {
346  HYPRE_ClearAllErrors();
347  const bool convergence_analyse = params->convergenceAnalyse();
348 
349  // test le second membre du système linéaire
350  const Real res0 = b->norm_max();
351 
352  if (convergence_analyse)
353  info() << "analyse convergence : norme max du second membre res0 : " << res0;
354 
355  const Real considered_as_null = params->minRHSNorm();
356  if (res0 < considered_as_null) {
357  HYPRE_ParVectorSetConstantValues(x->m_hypre_parvector, 0.0);
358  residual_norm[0] = res0;
359  if (convergence_analyse)
360  info() << "analyse convergence : le second membre du système linéaire est inférieur à : " << considered_as_null;
361  return true;
362  }
363 
364  if (params->xoUser()) {
365  // on test si b est déjà solution du système à epsilon près
366  //matrix->vectorProduct(b, tmp_vector); tmp_vector->sub(x);
367  //M->vector_multiply(*tmp,*x); // tmp=A*x
368  //tmp->substract(*tmp,*b); // tmp=A*x-b
369 
370  // X= alpha* M.B + beta * X (lu dans les sources de HYPRE)
371  HYPRE_ParCSRMatrixMatvec(1.0, m_hypre_parmatrix, x->m_hypre_parvector, 0., tmp->m_hypre_parvector);
372  HYPRE_ParVectorAxpy(-1.0, b->m_hypre_parvector, tmp->m_hypre_parvector);
373 
374  const Real residu = tmp->norm_max();
375  //info() << "[IAlephHypre::isAlreadySolved] residu="<<residu;
376 
377  if (residu < considered_as_null) {
378  if (convergence_analyse) {
379  info() << "analyse convergence : |Ax0-b| est inférieur à " << considered_as_null;
380  info() << "analyse convergence : x0 est déjà solution du système.";
381  }
382  residual_norm[0] = residu;
383  return true;
384  }
385 
386  const Real relative_error = residu / res0;
387  if (convergence_analyse)
388  info() << "analyse convergence : résidu initial : " << residu
389  << " --- residu relatif initial (residu/res0) : " << residu / res0;
390 
391  if (relative_error < (params->epsilon())) {
392  if (convergence_analyse)
393  info() << "analyse convergence : X est déjà solution du système";
394  residual_norm[0] = residu;
395  return true;
396  }
397  }
398  return false;
399  }

732  if (gamma != -1)
733  max_levels = gamma; // utilisation de la valeur du jeu de donnees
734 
735  double strong_threshold = 0.1; // 0.25; // set AMG threshold Theta = val
736  const double alpha = solver_param->alpha();
737  if (alpha > 0.0)
738  strong_threshold = alpha; // utilisation de la valeur du jeu de donnees

757  grid_relax_points[0][2] = 1;

764  grid_relax_points[1][1] = 1;
765  grid_relax_points[1][2] = -2;
766  grid_relax_points[1][3] = -2;
767 
768  /* up cycle */
769  num_grid_sweeps[2] = 4;
770  grid_relax_type[2] = relax_default;
771  grid_relax_points[2] = _allocHypre<int>(4);
772  grid_relax_points[2][0] = -2;
773  grid_relax_points[2][1] = -2;

775  grid_relax_points[2][3] = -1;
