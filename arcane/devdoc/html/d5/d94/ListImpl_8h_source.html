179  virtual void add(ObjectRef elem)
180  {
181  this->onInsert();
182  Integer s = this->count();
183  if (s>=m_capacity){
184  _resize(_ptr(),s+1);

286  void _arrayCopy(const T* from_ptr,Integer from_size)
287  {
288  if (from_size==0){
289  this->clear();
290  }
291  else{
292  if (from_size<capacity()){
293  this->_unguardedCopy(_ptr(),from_ptr,from_size);
294  this->_setCount(from_size);
295  }
296  else{
297  T* new_ptr = _resize(from_ptr,from_size);
298  this->_setList(new_ptr,from_size);
299  }
300  }

337  void _unguardedCopy(T* to_ptr,const T* from_ptr,Integer s)
338  {
339  T* p = to_ptr;
340  _setPtr(to_ptr);
341  Integer z = s;
342  const T* optr = from_ptr;
343  while (z--)
344  *p++ = *optr++;
345  }

366  T* _resize(const T* from_ptr,Integer new_size)
367  {
368  Integer c = _capacity();
369  while(new_size>c)
370  c = (c==0) ? 4 : (c*2);
371  _setCapacity(c);
372  T* old_ptr = _ptr();
373  T* new_ptr = _allocate(c);
374  if (from_ptr){
375  this->_unguardedCopy(new_ptr,from_ptr,this->count());
376  this->_setCount(new_size);
377  if (from_ptr==old_ptr)
378  _deallocate(old_ptr);
379  }
380  else
381  _setList(new_ptr,new_size);
382  return new_ptr;
383  }
384 
388  void _deallocate(T* ptr)
389  { m_allocator->deallocate(ptr,m_capacity); }
390

392  { return m_allocator->allocate(new_capacity); }
