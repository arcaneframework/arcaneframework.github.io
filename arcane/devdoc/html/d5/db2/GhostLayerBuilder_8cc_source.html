89
90void GhostLayerBuilder::
91addGhostLayers(bool is_allocate)
92{
93 Real begin_time = platform::getRealTime();
94 Integer version = m_mesh->ghostLayerMng()->builderVersion();
95 if (version==1){
96 throw NotSupportedException(A_FUNCINFO,"Version 1 is no longer supported");
97 }
98 else if (version==2){
99 info() << "Use ghost layer builder version 2";
100 _addOneGhostLayerV2();
101 }
102 else if (version==3 || version==4){
103 info() << "Use GhostLayerBuilder with sort (version " << version << ")";
104 _buildGhostLayerNewVersion(m_mesh,is_allocate,version);
105 }
106 else
107 throw NotSupportedException(A_FUNCINFO,"Bad version number for addGhostLayer");
108
109 Real end_time = platform::getRealTime();
110 Real diff = (Real)(end_time - begin_time);
111 info() << "TIME to compute ghost layer=" << diff;
112}

155void GhostLayerBuilder::
156_exchangeData(IParallelExchanger* exchanger,BoundaryInfosMap& boundary_infos_to_send)
157{
158 for( BoundaryInfosMapEnumerator i_map(boundary_infos_to_send); ++i_map; ){
159 Int32 sd = i_map.data()->key();
160 exchanger->addSender(sd);
161 }
162 exchanger->initializeCommunicationsMessages();
163 {
164 for( Integer i=0, ns=exchanger->nbSender(); i<ns; ++i ){
165 ISerializeMessage* sm = exchanger->messageToSend(i);
166 Int32 rank = sm->destination().value();
167 ISerializer* s = sm->serializer();
168 Int64ConstArrayView infos = boundary_infos_to_send[rank];
169
170 s->setMode(ISerializer::ModeReserve);
171 s->reserveArray(infos); // Pour les elements
172
173 s->allocateBuffer();
174 s->setMode(ISerializer::ModePut);
175
176 s->putArray(infos);
177 }
178 }
179 exchanger->processExchange();
180 debug() << "END EXCHANGE";
181}

186void GhostLayerBuilder::
187_addOneGhostLayerV2()
188{
189 info() << "** NEW GHOST LAYER BUILDER V2";
190 if (m_mesh->ghostLayerMng()->nbGhostLayer()!=1)
191 ARCANE_THROW(NotImplementedException,"Only one layer of ghost cells is supported");
192
193 IParallelMng* pm = m_mesh->parallelMng();
194 Int32 my_rank = pm->commRank();
195 Int32 nb_rank = pm->commSize();
196 debug() << " RANK="<< pm->commRank() << " size=" << pm->commSize();
197 if (!pm->isParallel()){
198 debug() << "NOT PARALLEL";
199 return;
200 }
201#ifdef ARCANE_DEBUG_DYNAMIC_MESH
202 const bool is_verbose = true;
203#else
204 const bool is_verbose = false;
205#endif
206
207 OStringStream ostr;
208 if (is_verbose)
209 ostr() << "** FACES LIST\n";
210
211 Integer nb_sub_domain_boundary_face = 0;
212 // Marque les noeuds sur la frontière
213 ItemInternalMap& cells_map = m_mesh->cellsMap(); // Supporte les transferts du maillage
214 ItemInternalMap& faces_map = m_mesh->facesMap(); // Détermine les frontières avant transfert
215 // ItemInternalMap& edges_map = m_mesh->edgesMap(); // N'est pas utilisé directement par l'algo
216 ItemInternalMap& nodes_map = m_mesh->nodesMap(); // Localise les modifications
217
218 const int shared_and_boundary_flags = ItemFlags::II_Shared | ItemFlags::II_SubDomainBoundary;
219 // Parcours les faces et marque les noeuds, arêtes et faces frontieres
220 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,faces_map){
221 Face face = iid->value();
222 ItemInternal* face_internal = iid->value();
223 if (is_verbose){
224 _printItem(face_internal,ostr());
225 ostr() << '\n';
226 }
227 bool is_sub_domain_boundary_face = false;
228 if (face_internal->flags() & ItemFlags::II_Boundary){
229 is_sub_domain_boundary_face = true;
230 }
231 else{
232 if (face.nbCell()==2 && (face.cell(0).owner()!=my_rank || face.cell(1).owner()!=my_rank))
233 is_sub_domain_boundary_face = true;
234 }
235 if (is_sub_domain_boundary_face){
236 face_internal->addFlags(shared_and_boundary_flags);
237 ++nb_sub_domain_boundary_face;
238 for( Item inode : face.nodes() )
239 inode.mutableItemBase().addFlags(shared_and_boundary_flags);
240 for( Item iedge : face.edges() )
241 iedge.mutableItemBase().addFlags(shared_and_boundary_flags);
242 }
243 }
244
245 Integer boundary_nodes_uid_count = 0;
246
247 // Parcours les noeuds et ajoute les noeuds frontières
248 Int64 my_max_node_uid = NULL_ITEM_UNIQUE_ID;
249 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,nodes_map){
250 ItemInternal* node = iid->value();
251 Int32 f = node->flags();
252 if (f & ItemFlags::II_Shared){
253 Int64 node_uid = node->uniqueId();
254 if (node_uid>my_max_node_uid)
255 my_max_node_uid = node_uid;
256 //boundary_nodes_uid.add(node_uid);
257 ++boundary_nodes_uid_count;
258 }
259 }
260
261 Int64 global_max_node_uid = pm->reduce(Parallel::ReduceMax,my_max_node_uid);
262 debug() << "NB BOUNDARY NODE=" << boundary_nodes_uid_count
263 << " MY_MAX_UID=" << my_max_node_uid
264 << " GLOBAL=" << global_max_node_uid;
265
266 if (is_verbose){
267 ostr.reset();
268 ostr() << "List of shared cells:\n";
269 }
270
271
272 //TODO: choisir bonne valeur pour initialiser la table
273 BoundaryInfosMap boundary_infos_to_send(200,true);
274 NodeUidToSubDomain uid_to_subdomain_converter(global_max_node_uid,nb_rank);
275
276 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid,cells_map){
277 Cell cell = iid->value();
278 if (is_verbose){
279 ostr() << "Send cell " << ItemPrinter(cell) << '\n';
280 }
281 //info() << " CHECK cell uid=" << cell->uniqueId() << " owner=" << cell->owner();
282 //bool add_cell = false;
283 for( Node inode : cell.nodes() ){
284 //info() << "** CHECK NODE node=" << i_node->uniqueId() << " cell=" << cell->uniqueId();
285 if (inode.itemBase().flags() & ItemFlags::II_Shared){
286 Int64 node_uid = inode.uniqueId();
287 //info() << "** ADD BOUNDARY CELL node=" << node_uid << " cell=" << cell->uniqueId();
288 Int32 dest_rank = uid_to_subdomain_converter.uidToRank(node_uid);
289 SharedArray<Int64> v = boundary_infos_to_send.lookupAdd(dest_rank)->value();
290 v.add(node_uid);
291 v.add(cell.owner());
292 v.add(cell.uniqueId());
293 //TODO: supprimer les doublons ?
294 //add_cell = true;
295 //break;
296 }
297 }
298 }
299
300 if (is_verbose)
301 info() << ostr.str();
302
303 info() << "Number of shared faces: " << nb_sub_domain_boundary_face;
304
305 auto exchanger { ParallelMngUtils::createExchangerRef(pm) };
306
307 if (!platform::getEnvironmentVariable("ARCANE_COLLECTIVE_GHOST_LAYER").null())
308 exchanger->setExchangeMode(IParallelExchanger::EM_Collective);
309
310 _exchangeData(exchanger.get(),boundary_infos_to_send);
311
312 traceMng()->flush();
313 pm->barrier();
314 NodeCellList node_cell_list;
315 {
316 Integer nb_receiver = exchanger->nbReceiver();
317 debug() << "NB RECEIVER=" << nb_receiver;
318 Int64UniqueArray received_infos;
319 for( Integer i=0; i<nb_receiver; ++i ){
320 ISerializeMessage* sm = exchanger->messageToReceive(i);
321 //Int32 orig_rank = sm->destSubDomain();
322 ISerializer* s = sm->serializer();
323 s->setMode(ISerializer::ModeGet);
324 s->getArray(received_infos);
325 Int64 nb_info = received_infos.largeSize();
326 //info() << "RECEIVE NB_INFO=" << nb_info << " from=" << orig_rank;
327 if ((nb_info % 3)!=0)
328 ARCANE_FATAL("Inconsistent received data v={0}",nb_info);
329 Int64 nb_info_true = nb_info / 3;
330 for( Int64 z=0; z<nb_info_true; ++z ){
331 Int64 node_uid = received_infos[(z*3)+0];
332 Int64 cell_owner = received_infos[(z*3)+1];
333 Int64 cell_uid = received_infos[(z*3)+2];
334 node_cell_list.add(node_uid,cell_uid,cell_owner);
335 }
336 }
337 }
338
339 boundary_infos_to_send = BoundaryInfosMap(1000,true);
340
341 {
342 Int64ConstArrayView cell_indexes = node_cell_list.m_cell_indexes;
343 debug() << "NB_CELL_INDEXES = " << cell_indexes.size();
344 //for( Integer i=0, s=cell_indexes.size(); i<s; ++i )
345 //info() << "INDEX I=" << i << " V=" << cell_indexes[i];
346 Int32UniqueArray ranks;
347 Int64UniqueArray cells;
348 for( HashTableMapEnumeratorT<Int64,Int32> i_map(node_cell_list.m_cell_last_index); ++i_map; ){
349 HashTableMapT<Int64,Int32>::Data* d = i_map.data();
350 Int32 index = d->value();
351 Int64 node_uid = d->key();
352 //info() << "NODE UID=" << node_uid;
353 ranks.clear();
354 cells.clear();
355 // Comme on connait la liste des mailles connectées à ce noeud ainsi que leur
356 // propriétaire, en profite pour calculer le propriètaire du noeud en considérant
357 // qu'il s'agit du même propriétaire que celui de la maille de plus petit uniqueId()
358 // connecté à ce noeud.
359 Int32 node_new_owner = NULL_SUB_DOMAIN_ID;
360 Int64 smallest_cell_uid = NULL_ITEM_UNIQUE_ID;
361 //TODO ajouter securite en calculant le nombre max de valeurs
362 while(index!=(-1)){
363 Int64 cell_uid = cell_indexes[index];
364 Int32 cell_owner = CheckedConvert::toInt32(cell_indexes[index+1]);
365 index = CheckedConvert::toInteger(cell_indexes[index+2]);
366 //info() << " CELLS: uid=" << cell_uid << " owner=" << cell_owner;
367 ranks.add((Int32)cell_owner);
368 cells.add(cell_uid);
369 if (cell_uid<smallest_cell_uid || node_new_owner==NULL_SUB_DOMAIN_ID){
370 smallest_cell_uid = cell_uid;
371 node_new_owner = cell_owner;
372 }
373 }
374 // Tri les rangs puis supprime les doublons
375 std::sort(std::begin(ranks),std::end(ranks));
376 Integer new_size = CheckedConvert::toInteger(std::unique(std::begin(ranks),std::end(ranks)) - std::begin(ranks));
377 ranks.resize(new_size);
378 //info() << "NEW_SIZE=" << new_size;
379 //for( Integer z=0; z<new_size; ++z )
380 //info() << "NEW_RANK=" << ranks[z];
381
382 // Si le nombre de rang vaut 1, cela signifie que le noeud n'appartient qu'à un seul sous-domaine
383 // et donc il s'agit d'un vrai noeud frontière. Il n'y a pas besoin de transférer ses mailles.
384 if (new_size==1)
385 continue;
386 Integer nb_cell = cells.size();
387 for( Integer z=0; z<new_size; ++z ){
388 Int32 dest_rank = ranks[z];
389 //info() << "NEW_RANK=" << dest_rank;
390 Int64Array& v = boundary_infos_to_send.lookupAdd(dest_rank)->value();
391 v.add(node_uid);
392 v.add(node_new_owner);
393 v.add(new_size);
394 v.add(nb_cell);
395 for( Integer z2=0; z2<new_size; ++z2 )
396 v.add(ranks[z2]);
397 for( Integer z2=0; z2<nb_cell; ++z2 )
398 v.add(cells[z2]);
399 }
400 }
401 }
402
403 exchanger = ParallelMngUtils::createExchangerRef(pm);
404 _exchangeData(exchanger.get(),boundary_infos_to_send);
405 debug() << "END OF EXCHANGE";
406
407 typedef HashTableMapT<Int32,SharedArray<Int32> > SubDomainItemMap;
408 SubDomainItemMap cells_to_send(50,true);
409 {
410 Integer nb_receiver = exchanger->nbReceiver();
411 debug() << "NB RECEIVER 2 =" << nb_receiver;
412 Int64UniqueArray received_infos;
413 //HashTableMapT<Int64,Int32> nodes_nb_cell(1000,true);
414 for( Integer i=0; i<nb_receiver; ++i ){
415 ISerializeMessage* sm = exchanger->messageToReceive(i);
416 //Int32 orig_rank = sm->destSubDomain();
417 ISerializer* s = sm->serializer();
418 s->setMode(ISerializer::ModeGet);
419 //info() << "RECEIVE NB_INFO=" << nb_info << " from=" << orig_rank;
420 s->getArray(received_infos);
421 Int64 nb_info = received_infos.largeSize();
422 Int64 z = 0;
423 Int32UniqueArray ranks;
424 Int32UniqueArray cells;
425 while (z<nb_info){
426 Int64 node_uid = received_infos[z];
427 Int32 node_new_owner = CheckedConvert::toInt32(received_infos[z+1]);
428 Int32 nb_rank = CheckedConvert::toInt32(received_infos[z+2]);
429 Int32 nb_cell = CheckedConvert::toInt32(received_infos[z+3]);
430 //info() << "RECEIVE NODE uid="<< node_uid << " nb_rank=" << nb_rank << " nb_cell=" << nb_cell;
431 nodes_map[node_uid]->setOwner(node_new_owner,my_rank);
432 ranks.clear();
433 cells.clear();
434 z += 4;
435 for( Integer z2=0; z2<nb_rank; ++z2 ){
436 Int32 nrank = (Int32)received_infos[z+z2];
437 if (nrank!=my_rank)
438 ranks.add(nrank);
439 }
440 z += nb_rank;
441 for( Integer z2=0; z2<nb_cell; ++z2 ){
442 Int64 cell_uid = received_infos[z+z2];
443 ItemInternalMap::Data* dcell = cells_map.lookup(cell_uid);
444 if (dcell)
445 cells.add(dcell->value()->localId());
446 }
447 for( Integer z2=0,zs=ranks.size(); z2<zs; ++z2 ){
448 SubDomainItemMap::Data* d = cells_to_send.lookupAdd(ranks[z2]);
449 SharedArray<Int32> dv = d->value();
450 for( Integer z3=0, zs3=cells.size(); z3<zs3; ++z3 )
451 dv.add(cells[z3]);
452 }
453 z += nb_cell;
454 }
455 }
456 }
457
458 // Envoie et réceptionne les mailles fantômes
459 _exchangeCells(cells_to_send,false);
460 m_mesh_builder->printStats();
461}

503void GhostLayerBuilder::
504addGhostChildFromParent()
505{
506 info() << "** AMR GHOST CHILD FROM PARENT BUILDER V1";
507
508 IParallelMng* pm = m_mesh->parallelMng();
509 debug() << " RANK="<< pm->commRank() << " size=" << pm->commSize();
510 if (!pm->isParallel()){
511 debug() << "NOT PARALLEL";
512 return;
513 }
514 Integer sid = pm->commRank();
515
516 // Marque les noeuds sur la frontière
517 ItemInternalMap& cells_map = m_mesh->cellsMap();
518
519 FaceFamily& true_face_family = m_mesh->trueFaceFamily();
520
521 //TODO: choisir bonne valeur pour initialiser la table
522 BoundaryInfosMap boundary_infos_to_send(200,true);
523
524 ENUMERATE_ITEM_INTERNAL_MAP_DATA(ncd,cells_map){
525 ItemInternal* cell = ncd->value();
526 ARCANE_ASSERT((cell->owner() != -1),(""));
527 if(cell->level() == 0 && cell->owner() != sid){
528 ARCANE_ASSERT((cell->owner() != -1),(""));
529 Int64Array& v = boundary_infos_to_send.lookupAdd(cell->owner())->value();
530 v.add(sid);
531 v.add(cell->uniqueId());
532 }
533 }
534
535 // Positionne la liste des envoies
536 auto exchanger { ParallelMngUtils::createExchangerRef(pm) };
537 _exchangeData(exchanger.get(),boundary_infos_to_send);
538
539 traceMng()->flush();
540 pm->barrier();
541
542 typedef HashTableMapT<Int32,SharedArray<Int32> > SubDomainItemMap;
543 SubDomainItemMap cells_to_send(50,true);
544 {
545 Integer nb_receiver = exchanger->nbReceiver();
546 debug() << "NB RECEIVER=" << nb_receiver;
547 Int64UniqueArray received_infos;
548 for( Integer i=0; i<nb_receiver; ++i ){
549 ISerializeMessage* sm = exchanger->messageToReceive(i);
550 ISerializer* s = sm->serializer();
551 s->setMode(ISerializer::ModeGet);
552 s->getArray(received_infos);
553 Int64 nb_info = received_infos.size();
554 //Int32 orig_rank = sm->destSubDomain();
555 //info() << "RECEIVE NB_INFO=" << nb_info << " from=" << orig_rank;
556 if ((nb_info % 2)!=0)
557 ARCANE_FATAL("info size can not be divided by 2 v={0}",nb_info);
558 Int64 nb_info_true = nb_info / 2;
559 Integer nb_recv_child=0;
560 for( Int64 z=0; z<nb_info_true; ++z ){
561 Int32 cell_owner = CheckedConvert::toInt32(received_infos[(z*2)+0]);
562 Int64 cell_uid = received_infos[(z*2)+1];
563
564 ItemInternal* cell= cells_map[cell_uid];
565 ARCANE_ASSERT((cell->uniqueId() == cell_uid),(""));
566 if (!cell->hasHChildren())
567 continue;
568 UniqueArray<ItemInternal*> cell_family;
569 ARCANE_ASSERT((cell->level() == 0),(""));
570 ARCANE_ASSERT((cell->owner() != -1),("CELL"));
571 ARCANE_ASSERT((cell_owner != -1),("CELL"));
572 true_face_family.familyTree(cell_family,cell);
573 SubDomainItemMap::Data* d = cells_to_send.lookupAdd(cell_owner);
574 Int32Array& dv = d->value();
575 const Integer cs=cell_family.size();
576 nb_recv_child +=cs;
577 for(Integer c=1;c<cs;c++){
578 ItemInternal* child= cell_family[c];
579 ARCANE_ASSERT((child->owner() != -1),("CHILD"));
580 //debug() << child->topHParent()->uniqueId() << " " << cell->uniqueId() << " " << child->topHParent()->owner() << " " << cell->owner();
581 //ARCANE_ASSERT((child->topHParent() == cell),("CHILD"));
582 dv.add(child->localId());
583 }
584 cell_family.clear();
585 }
586 debug() << "nb_recv_child= " << nb_recv_child;
587 }
588 }
589 // Envoie et réceptionne les mailles fantômes
590 _exchangeCells(cells_to_send,true);
591 m_mesh_builder->printStats();
592}

665 ARCANE_ASSERT((cell.owner() != -1),("CELL"));

682 }
683
684 // Envoie et réceptionne les mailles fantômes
685 _exchangeCells(cells_to_send,true);
686 m_mesh_builder->printStats();
