103 template<typename ConnectivityIndexType,typename CustomType>
104 class ARCANE_MESH_EXPORT ItemConnectivitySelectorT
105 : public ItemConnectivitySelector
106 {
107  public:
108 
109  ItemConnectivitySelectorT(ItemFamily* source_family,IItemFamily* target_family,
110  const String& connectivity_name)
111  : ItemConnectivitySelector(source_family,target_family,connectivity_name,ConnectivityIndexType::connectivityIndex())
112  , m_custom_connectivity(nullptr)
113  {
114  }
115  ~ItemConnectivitySelectorT()
116  {
117  // NOTE: les connectivités sont détuites par les familles.
118  }
119 
120  public:
121 
122  IIncrementalItemConnectivity* customConnectivity() const override
123  {
124  return trueCustomConnectivity();
125  }
126 
127  void checkValidConnectivityList() const override
128  {
129  if (m_item_connectivity_index<0)
130  return;
131  auto x = m_item_connectivity_list;
132  auto current_list = x->connectivityList(m_item_connectivity_index);
133  auto ref_list = m_custom_connectivity->connectivityList();
134  auto current_indexes = x->connectivityIndex(m_item_connectivity_index);
135  auto ref_indexes = m_custom_connectivity->connectivityIndex();
136  auto* current_list_ptr = current_list.data();
137  auto* ref_list_ptr = ref_list.data();
138  if (current_list_ptr!=ref_list_ptr)
139  ARCANE_FATAL("Bad list base pointer current={0} ref={1}",current_list_ptr,ref_list_ptr);
140  if (current_list.size()!=ref_list.size())
141  ARCANE_FATAL("Bad list size current={0} ref={1}",current_list.size(),ref_list.size());
142  auto* current_indexes_ptr = current_indexes.data();
143  auto* ref_indexes_ptr = ref_indexes.data();
144  if (current_indexes_ptr!=ref_indexes_ptr)
145  ARCANE_FATAL("Bad indexes base pointer current={0} ref={1}",current_indexes_ptr,ref_indexes_ptr);
146  if (current_indexes.size()!=ref_indexes.size())
147  ARCANE_FATAL("Bad indexes size current={0} ref={1}",current_indexes.size(),ref_indexes.size());
148  }
149 
150  void compactConnectivities() override
151  {
152  if (m_custom_connectivity)
153  m_custom_connectivity->compactConnectivityList();
154  }
155 
156  public:
157 
158  void addConnectedItem(ItemLocalId item_lid,ItemLocalId sub_item_lid)
159  {
160  if (m_custom_connectivity)
161  m_custom_connectivity->addConnectedItem(item_lid,sub_item_lid);
162  }
163 
164  void removeConnectedItem(ItemLocalId item_lid,ItemLocalId sub_item_lid)
165  {
166  if (m_custom_connectivity)
167  m_custom_connectivity->removeConnectedItem(item_lid,sub_item_lid);
168  }
169 
170  void removeConnectedItems(ItemLocalId item_lid)
171  {
172  if (m_custom_connectivity)
173  m_custom_connectivity->removeConnectedItems(item_lid);
174  }
175 
176  void replaceItems(ItemLocalId item_lid,Int32ConstArrayView sub_item_lids)
177  {
178  if (m_custom_connectivity)
179  m_custom_connectivity->replaceConnectedItems(item_lid,sub_item_lids);
180  }
181 
182  void replaceItem(ItemLocalId item_lid,Integer index,ItemLocalId sub_item_lid)
183  {
184  if (m_custom_connectivity)
185  m_custom_connectivity->replaceConnectedItem(item_lid,index,sub_item_lid);
186  }
187 
188  bool hasConnectedItem(ItemLocalId source_item, ItemLocalId target_local_id) const
189  {
190  if (m_custom_connectivity)
191  return m_custom_connectivity->hasConnectedItem(source_item,target_local_id);
192 
193  return false;
194  }
195 
196  CustomType* trueCustomConnectivity() const { return m_custom_connectivity; }
197 
198  protected:
199 
200  void _createCustomConnectivity(const String& name) override
201  {
202  m_custom_connectivity = new CustomType(m_source_family,m_target_family,name);
203  }
204 
205  // used only with family dependencies. The concrete type of families are needed only for FaceToCellConnectivity
206  // where the IncrementalItemConnectivity has been overloaded (to handle back/front cell connectivity)
207  template <class SourceFamily, class TargetFamily>
208  void _createCustomConnectivity(const String& name)
209  {
210  m_custom_connectivity = new typename CustomConnectivity<SourceFamily,TargetFamily>::type(m_source_family,m_target_family,name);
211  }
212 
213  void _buildCustomConnectivity() override
214  {
215  // Indique à l'IncrementalItemConnectivity qu'elle doit mettre à jour
216  // les accesseurs de ItemInternal.
217  if (m_custom_connectivity)
218  m_custom_connectivity->setItemConnectivityList(m_item_connectivity_list,
219  m_item_connectivity_index);
220  }
221 
222  public:
223  // Build called when using family dependencies
224  template<class SourceFamily, class TargetFamily>
225  void build()
226  {
227  if (m_is_built)
228  return;
229 
230  _createCustomConnectivity<SourceFamily,TargetFamily>(m_connectivity_name);
231  info() << "Family: " << m_source_family->fullName()
232  << " create new connectivity: " << m_connectivity_name;
233 
234  _buildCustomConnectivity();
235  m_is_built = true;
236  }
237 
238  private:
239 
240  CustomType* m_custom_connectivity;
241 };
