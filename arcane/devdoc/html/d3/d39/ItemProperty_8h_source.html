106  template <class AbstractFamily>
107  void resize(AbstractFamily* item_family, const Integer nb_element_per_item, const DataType default_value) // SDC Template AbstractFamily necessaire pour applications IFPEN
108  {
109  Integer dim1_old_size = m_data.dim1Size();
110  Integer dim2_old_size = m_data.dim2Size();
111  Integer dim1_new_size = item_family->maxLocalId();
112  Integer dim2_new_size = nb_element_per_item;
113  m_data.resize(dim1_new_size,dim2_new_size);
114  // Fill new data
115  // if new dim 1 larger, enter loop
116  for (Integer i = dim1_old_size; i < dim1_new_size; ++i)
117  {
118  for (Integer j = 0; j < dim2_new_size; ++j) m_data[i][j] = default_value;
119  }
120  // test if dim2 larger before loop
121  if (dim2_new_size > dim2_old_size)
122  {
123  for (Arcane::Integer i = 0; i < dim1_old_size; ++i)
124  {
125  for (Integer j = dim2_old_size; j < dim2_new_size;++j) m_data[i][j] = default_value;
126  }
127  }
128  }
129 
130  Integer dim1Size() const {return m_data.dim1Size();}
131  Integer dim2Size() const {return m_data.dim2Size();}
132

136  template<class AbstractItem>
137  ConstArrayView<DataType> operator[](AbstractItem item) const { return m_data[item.localId()]; }

148  }
149  }

176  // nb_element per item est de taille IItemFamily::maxLocalId
177  template <class AbstractFamily>
178  void resize(AbstractFamily* item_family, const Arcane::IntegerConstArrayView nb_element_per_item, const DataType default_value) // SDC Template AbstractFamily necessaire pour applications IFPEN
179  {
180  ARCANE_ASSERT((nb_element_per_item.size() == item_family->maxLocalId()),
181  ("In item property resize : nb_element_per_item must have size IItemFamilyy::maxLocalId"))
182  Integer dim1_old_size = m_data.dim1Size();
183  Integer dim1_new_size = nb_element_per_item.size();
184  IntegerUniqueArray dim2_old_sizes(m_data.dim2Sizes());
185  IntegerConstArrayView dim2_new_sizes(nb_element_per_item);
186  m_data.resize(nb_element_per_item);
187  // Fill new data with default value
188  // Dim1 size larger
189  for (Arcane::Integer i = dim1_old_size; i < dim1_new_size; ++i)
190  {
191  for (Arcane::Integer j = 0; j < dim2_new_sizes[i]; ++j)
192  {
193  m_data[i][j] = default_value;
194  }
195  }
196  // Dim2 size larger (eventually, cannot know a priori, need to loop).
197  for (Arcane::Integer i = 0; i < dim1_old_size; ++i)
198  {
199  for (Arcane::Integer j = dim2_old_sizes[i]; j < dim2_new_sizes[i]; ++j)
200  {
201  m_data[i][j] = default_value;
202  }
203  }
204  }
205 
206  Integer dim1Size() const {return m_data.dim1Size();}
207  IntegerConstArrayView dim2Sizes() const {return m_data.dim2Sizes();}

211 
212  template<class AbstractItem>
213  ConstArrayView<DataType> operator[](const AbstractItem item) const {return m_data[item.localId()];}

215  void updateSupport(Int32ConstArrayView new_to_old_ids)
216  {
217  UniqueMultiArray2<DataType> old_data(m_data);
218  Integer new_dim1_size = new_to_old_ids.size();
219  IntegerConstArrayView dim2_sizes = m_data.dim2Sizes();
220  IntegerUniqueArray new_dim2_sizes(new_dim1_size);
221  // Compute new sizes
222  for (Integer i = 0; i < new_dim1_size; ++i)
223  {
224  new_dim2_sizes[i] = dim2_sizes[new_to_old_ids[i]];
225  }
226  m_data.resize(new_dim2_sizes);
227  for (Integer i = 0; i < new_dim1_size; ++i)
228  {
229  for (Integer j = 0; j < new_dim2_sizes[i];++j) m_data[i][j] = old_data[new_to_old_ids[i]][j];
230  }
231  }
232

234  {
235  m_data = item_property_from.m_data.constView(); // copy values
236  }

239

241  SharedMultiArray2<DataType> m_data; // ItemProperty used in IFPEN applications as a shared object (Arcane-like...).
242 };
