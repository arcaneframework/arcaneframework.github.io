34 /*
35  * \brief Variable de liens d'items de types quelconques.
36  *
37  * Par exemple :
38  *
39  * AnyItem::Variable<Real> variable(family); // Remplie
40  * AnyItem::LinkVariableArray<Real> link_variable(link_family); // Remplie de taille 3
41  *
42  * Real value = 0;
43  * ENUMERATE_ANY_ITEM_LINK(ilink, link_family) {
44  * if(ilink.index() < 10) {
45  * info() << "back item = [uid=" << family.concreteItem(ilink.back()).uniqueId()
46  * << ",lid=" << family.concreteItem(ilink.back()).localId() << ",kind="
47  * << family.concreteItem(ilink.back()).kind() << "]";
48  * info() << "front item = [uid=" << family.concreteItem(ilink.front()).uniqueId()
49  * << ",lid=" << family.concreteItem(ilink.front()).localId() << ",kind="
50  * << family.concreteItem(ilink.front()).kind() << "]";
51  * }
52  *
53  * for(Integer i = 0; i < 3; ++i)
54  * value += link_variable[ilink][i] + variable[ilink.back()][i] + variable[ilink.front()];
55  * }
56  *
57  */
58 template<typename DataType>
59 class LinkVariableArray
60  : public ILinkFamilyObserver
61 {
62 public:
63 
64  LinkVariableArray(const LinkFamily& family)
65  : m_size(0)
66  , m_family(family)
67  , m_values(m_family.capacity(),m_size)
68  {
69  m_family.registerObserver(*this);
70  }
71 
72  LinkVariableArray(const LinkVariableArray& v)
73  : m_size(v.m_size)
74  , m_family(v.m_family)
75  , m_values(v.m_values)
76  {
77  m_family.registerObserver(*this);
78  }
79 
80  ~LinkVariableArray()
81  {
82  m_family.removeObserver(*this);
83  }
84 
86  inline ArrayView<DataType> operator[](const LinkFamily::LinkIndex& item) {
87  return m_values[item.index()];
88  }
89 
91  inline ConstArrayView<DataType> operator[](const LinkFamily::LinkIndex& item) const {
92  return m_values[item.index()];
93  }
94 
96  inline void notifyFamilyIsInvalidate() {
97  // Si la famille change, on retaille
98  m_values.resize(m_family.capacity(),m_size);
99  }
100 
102  inline void notifyFamilyIsReserved() {
103  // Si la famille est reservée, on retaille simplement
104  m_values.resize(m_family.capacity(),m_size);
105  }
106 
108  inline void resize(Integer size)
109  {
110  m_size = size;
111  m_values.resize(m_family.capacity(),m_size);
112  }
113 
115  inline Integer size() const { return m_size; }
116 
117 private:
118 
120  Integer m_size;
121 
123  const LinkFamily m_family;
124 
126  Arcane::UniqueArray2<DataType> m_values;
127 };
