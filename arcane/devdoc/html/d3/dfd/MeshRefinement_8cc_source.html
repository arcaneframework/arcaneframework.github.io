163void
164MeshRefinement::
165init()
166{
167 // Recalcul le max uniqueId() des nodes/cells/faces.
168 CHECKPERF( m_perf_counter.start(PerfCounter::INIT) )
169 IParallelMng* pm = m_mesh->parallelMng();
170 {
171 Int64 max_node_uid = 0;
172 ENUMERATE_NODE(inode,m_mesh->allNodes())
173 {
174 const Node& node = *inode;
175 const Int64 uid = node.uniqueId();
176 if (uid>max_node_uid)
177 max_node_uid = uid;
178 }
179
180 if (pm->commSize() > 1)
181 m_max_node_uid = pm->reduce(Parallel::ReduceMax, max_node_uid);
182 else
183 m_max_node_uid = max_node_uid;
184 info() << "NODE_UID_INFO: MY_MAX_UID=" << max_node_uid << " GLOBAL=" << m_max_node_uid;
185 m_next_node_uid = m_max_node_uid + 1 + m_mesh->parallelMng()->commRank();
186 }
187 ItemTypeMng* itm = m_mesh->itemTypeMng();
188
189 {
190 Int64 max_cell_uid = 0;
191 Integer max_nb_hChildren = 0;
192 ENUMERATE_CELL(icell,m_mesh->allCells())
193 {
194 const Cell& cell = *icell;
195 const Int64 uid = cell.uniqueId();
196 const Int32 nb_hChildren = itm->nbHChildrenByItemType(cell.type());
197 if (uid>max_cell_uid)
198 max_cell_uid = uid;
199 if (nb_hChildren>max_nb_hChildren)
200 max_nb_hChildren = nb_hChildren;
201
202 }
203 if (pm->commSize() > 1)
204 {
205 m_max_cell_uid = pm->reduce(Parallel::ReduceMax, max_cell_uid);
206 m_max_nb_hChildren = pm->reduce(Parallel::ReduceMax, max_nb_hChildren);
207 }
208 else
209 {
210 m_max_cell_uid = max_cell_uid;
211 m_max_nb_hChildren = max_nb_hChildren;
212 }
213 info() << "CELL_UID_INFO: MY_MAX_UID=" << max_cell_uid << " GLOBAL=" << m_max_cell_uid;
214 m_next_cell_uid = m_max_cell_uid + 1 + pm->commRank() * m_max_nb_hChildren;
215 }
216
217 {
218 Int64 max_face_uid = 0;
219 ENUMERATE_FACE(iface,m_mesh->allFaces())
220 {
221 const Face& face = *iface;
222 const Int64 uid = face.uniqueId();
223 if (uid>max_face_uid)
224 max_face_uid = uid;
225 }
226
227 if (pm->commSize() > 1)
228 m_max_face_uid = pm->reduce(Parallel::ReduceMax, max_face_uid);
229 else
230 m_max_face_uid = max_face_uid;
231 info() << "FACE_UID_INFO: MY_MAX_UID=" << max_face_uid << " GLOBAL=" << m_max_face_uid;
232 m_next_face_uid = m_max_face_uid + 1 + pm->commRank();
233 }
234
235 m_parallel_amr_consistency->init() ;
236
237 CHECKPERF( m_perf_counter.stop(PerfCounter::INIT) )
238}

241void MeshRefinement::
242_updateMaxUid(ArrayView<ItemInternal*> cells)
243{
244 // Recalcul le max uniqueId() des nodes/cells/faces.
245 CHECKPERF( m_perf_counter.start(PerfCounter::INIT) )
246 IParallelMng* pm = m_mesh->parallelMng();
247 ItemTypeMng* itm = m_mesh->itemTypeMng();
248 {
249
250 Int64 max_node_uid = m_max_node_uid;
251 Int64 max_cell_uid = m_max_cell_uid;
252 Integer max_nb_hChildren = m_max_nb_hChildren;
253 Int64 max_face_uid = m_max_face_uid;
254
255 typedef std::set<Int64> set_type ;
256 typedef std::pair<set_type::iterator,bool> insert_return_type ;
257 set_type node_list ;
258 set_type face_list ;
259 for(Integer icell=0;icell<cells.size();++icell){
260 Cell cell = cells[icell];
261 for (UInt32 i = 0, nc = cell.nbHChildren(); i < nc; i++){
262 Cell child = cell.hChild(i);
263
264 //UPDATE MAX CELL UID
265 const Int64 cell_uid = child.uniqueId();
266 const Int32 nb_hChildren = itm->nbHChildrenByItemType(child.type());
267 if (cell_uid>max_cell_uid)
268 max_cell_uid = cell_uid;
269 if (nb_hChildren>max_nb_hChildren)
270 max_nb_hChildren = nb_hChildren;
271
272 //UPDATE MAX NODE UID
273 for( Node inode : child.nodes() ){
274 const Int64 uid = inode.uniqueId();
275 insert_return_type value = node_list.insert(uid) ;
276 if(value.second){
277 if (uid>max_node_uid)
278 max_node_uid = uid;
279 }
280 }
281
282
283 //UPDATE MAX FACE UID
284 for( Face iface : child.faces() ){
285 const Int64 uid = iface.uniqueId();
286 insert_return_type value = face_list.insert(uid) ;
287 if(value.second){
288 if (uid>max_face_uid)
289 max_face_uid = uid;
290 }
291 }
292 }
293 }
294
295
296 if (pm->commSize() > 1)
297 {
298 m_max_node_uid = pm->reduce(Parallel::ReduceMax, max_node_uid);
299 m_max_cell_uid = pm->reduce(Parallel::ReduceMax, max_cell_uid);
300 m_max_nb_hChildren = pm->reduce(Parallel::ReduceMax, max_nb_hChildren);
301 m_max_face_uid = pm->reduce(Parallel::ReduceMax, max_face_uid);
302 }
303 else
304 {
305 m_max_node_uid = max_node_uid;
306 m_max_cell_uid = max_cell_uid;
307 m_max_nb_hChildren = max_nb_hChildren;
308 m_max_face_uid = max_face_uid;
309 }
310 m_next_node_uid = m_max_node_uid + 1 + m_mesh->parallelMng()->commRank();
311 m_next_cell_uid = m_max_cell_uid + 1 + pm->commRank() * m_max_nb_hChildren;
312 m_next_face_uid = m_max_face_uid + 1 + pm->commRank();
313 }
314
315 CHECKPERF( m_perf_counter.stop(PerfCounter::INIT) )
316}

372bool
373MeshRefinement::
374refineAndCoarsenItems(const bool maintain_level_one)
375{
376 CHECKPERF( m_perf_counter.start(PerfCounter::INIT) )
377
378 bool _maintain_level_one = maintain_level_one;
379
380 // la règle de niveau-un est la seule condition implementée
381 if (!maintain_level_one)
382 {
383 warning() << "Warning, level one rule is the only condition accepted for AMR!";
384 }
385 else
386 _maintain_level_one = m_face_level_mismatch_limit;
387
388 // Nous ne pouvons pas encore transformer un maillage de non-niveau-un en un maillage de niveau-un
389 if (_maintain_level_one){
390 ARCANE_ASSERT((_checkLevelOne(true)), ("checkLevelOne failed"));
391 }
392
393 // Nettoyage des flags de raffinement d'une étape précédente
394 this->_cleanRefinementFlags();
395 CHECKPERF( m_perf_counter.stop(PerfCounter::INIT) )
396
397 // La consistence parallèle doit venir en premier, ou le déraffinement
398 // le long des interfaces entre processeurs pourrait de temps en temps être
399 // faussement empéché
400 if (m_mesh->parallelMng()->isParallel())
401 this->_makeFlagParallelConsistent();
402
403 CHECKPERF( m_perf_counter.start(PerfCounter::CONSIST) )
404 // Repete jusqu'au matching du changement de flags sur chaque processeur
405 Integer iter = 0 ;
406 do
407 {
408 // Repete jusqu'au matching des flags coarsen/refine localement
409 bool satisfied = false;
410 do
411 {
412 const bool coarsening_satisfied = this->_makeCoarseningCompatible(maintain_level_one);
413 const bool refinement_satisfied = this->_makeRefinementCompatible(maintain_level_one);
414 satisfied = (coarsening_satisfied && refinement_satisfied);
415#ifdef ARCANE_DEBUG
416 bool max_satisfied = satisfied,min_satisfied = satisfied;
417 max_satisfied = m_mesh->parallelMng()->reduce(Parallel::ReduceMax,max_satisfied);
418 min_satisfied = m_mesh->parallelMng()->reduce(Parallel::ReduceMin,min_satisfied);
419 ARCANE_ASSERT ( (satisfied == max_satisfied), ("parallel max_satisfied failed"));
420 ARCANE_ASSERT ( (satisfied == min_satisfied), ("parallel min_satisfied failed"));
421#endif
422 } while (!satisfied);
423 ++iter ;
424 } while (m_mesh->parallelMng()->isParallel() && !this->_makeFlagParallelConsistent() && iter<10 );
425 if(iter==AMRMAXCONSISTENCYITER) fatal()<<" MAX CONSISTENCY ITER REACHED";
426 CHECKPERF( m_perf_counter.stop(PerfCounter::CONSIST) )
427
428 // D'abord déraffine les items flaggés.
429 CHECKPERF( m_perf_counter.start(PerfCounter::COARSEN) )
430 const bool coarsening_changed_mesh = this->_coarsenItems();
431 CHECKPERF( m_perf_counter.stop(PerfCounter::COARSEN) )
432
433 // Maintenant, raffine les items flaggés. Ceci prendra
434 // plus de mémoire, et peut être plus de ce qui est libre.
435 Int64UniqueArray cells_to_refine;
436 const bool refining_changed_mesh = this->_refineItems(cells_to_refine);
437
438 // Finalement, préparation du nouveau maillage pour utilisation
439 if (refining_changed_mesh || coarsening_changed_mesh) {
440 bool do_compact = m_mesh->properties()->getBool("compact");
441 m_mesh->properties()->setBool("compact",true) ; // Forcing compaction prevents from bugs when using AMR
442
443 // Raffinement
444 CHECKPERF( m_perf_counter.start(PerfCounter::ENDUPDATE) )
445 m_mesh->modifier()->endUpdate();
446 m_mesh->properties()->setBool("compact",do_compact) ;
447 CHECKPERF( m_perf_counter.stop(PerfCounter::ENDUPDATE) )
448
449 // deraffinement
450 //bool remove_ghost_children = false;
451 if (coarsening_changed_mesh)
452 {
453 //remove_ghost_children=true;
454
455 CHECKPERF( m_perf_counter.start(PerfCounter::CONTRACT) )
456 this->_contract();
457 CHECKPERF( m_perf_counter.stop(PerfCounter::CONTRACT) )
458
459 CHECKPERF( m_perf_counter.start(PerfCounter::ENDUPDATE) )
460 m_mesh->properties()->setBool("compact",true) ; // Forcing compaction prevents from bugs when using AMR (leads to problems whith dof)
461 m_mesh->modifier()->endUpdate();
462 m_mesh->properties()->setBool("compact",do_compact) ;
463 CHECKPERF( m_perf_counter.stop(PerfCounter::ENDUPDATE) )
464
465 }
466
467 // callback pour transporter les variables sur le nouveau maillage
468 CHECKPERF( m_perf_counter.start(PerfCounter::INTERP) )
469 this->_interpolateData(cells_to_refine);
470 CHECKPERF( m_perf_counter.stop(PerfCounter::INTERP) )
471 //
472
473 if (!coarsening_changed_mesh && m_mesh->parallelMng()->isParallel())
474 this->_makeFlagParallelConsistent2();
475
476 if (m_mesh->parallelMng()->isParallel())
477 {
478 CHECKPERF( m_perf_counter.start(PerfCounter::PGHOST) )
479 m_mesh->modifier()->setDynamic(true);
480 UniqueArray<Int64> ghost_cell_to_refine ;
481 UniqueArray<Int64> ghost_cell_to_coarsen ;
482 m_mesh->modifier()->updateGhostLayerFromParent(ghost_cell_to_refine,
483 ghost_cell_to_coarsen,
484 false);
485 _update(ghost_cell_to_refine) ;
486 CHECKPERF( m_perf_counter.stop(PerfCounter::PGHOST) )
487 _checkOwner("refineAndCoarsenItems after ghost update");
488 }
489
490 /*
491 if(do_compact)
492 {
493 CHECKPERF( m_perf_counter.start(PerfCounter::COMPACT) )
494 m_mesh->properties()->setBool("compact",true) ;
495 DynamicMesh* mesh = dynamic_cast<DynamicMesh*> (m_mesh);
496 if(mesh)
497 mesh->compact() ;
498 CHECKPERF( m_perf_counter.stop(PerfCounter::COMPACT) )
499 }*/
500
501#ifdef ACTIVATE_PERF_COUNTER
502 info()<<"MESH REFINEMENT PERF INFO" ;
503 m_perf_counter.printInfo(info().file()) ;
504 info()<<"NODE FINDER PERF INFO" ;
505 m_node_finder.getPerfCounter().printInfo(info().file()) ;
506 info()<<"FACE FINDER PERF INFO" ;
507 m_face_finder.getPerfCounter().printInfo(info().file()) ;
508 info()<<"PARALLEL AMR CONSISTENCY PERF INFO" ;
509 m_parallel_amr_consistency->getPerfCounter().printInfo(info().file()) ;
510#endif
511 return true;
512 }
513 // Si il n'y avait aucun changement dans le maillage
514 return false;
515}

1070
1071 // la règle de niveau-un est la seule condition implementée
1072 if (!maintain_level_one){
1073 warning() << "Warning, level one rule is the only condition accepted for AMR!";
1074 }
1075 else
1076 _maintain_level_one = m_face_level_mismatch_limit;
1077
1078 // à moins que nous rencontrions une situation spécifique, la règle niveau-un
1079 // sera satisfaite aprês avoir exécuté cette boucle juste une fois
1080 bool level_one_satisfied = true;
1081
1082 // à moins que nous rencontrions une situation spéccifique, nous serons compatible
1083 // avec tous flags de raffinement choisis
1084 bool compatible_with_refinement = true;
1085
1086 // Trouver le niveau maximum dans le maillage
1087 Integer max_level = 0;
1088
1089 // d'abord nous regardons tous les items actifs de niveau 0. Puisque ca n'a pas de sens de
1090 // les déraffiner nous devons donc supprimer leur flags de déraffinement si
1091 // ils sont déjà positionnés.
1092 // itérer seulement sur les mailles actives
1093 ENUMERATE_CELL(icell,m_mesh->allActiveCells()){
1094 const Cell cell = *icell;
1095 max_level = std::max(max_level, cell.level());
1096
1097 Integer f = cell.itemBase().flags();
1098 if ((cell.level() == 0) && (f & ItemFlags::II_Coarsen)) {
1099 f &= ~ItemFlags::II_Coarsen;
1100 f |= ItemFlags::II_DoNothing;
1101 cell.mutableItemBase().setFlags(f);
1102 }
1103 }
1104 // Si il n'y a pas d'items à raffiner sur ce processeur alors
1105 // il n'y a pas de travail à faire pour nous
1106 if (max_level == 0){
1107 debug() << "makeCoarseningCompatible() done";

1111

1113 }

1116 // pour déraffinement on check ses voisins. Si un de ses voisins
1117 // est marqué pour raffinement et est de même niveau alors il y a un
1118 // conflit. Par convention raffinement gagne, alors on démarque l'item pour

1164 // de cet item veulent être déraffinés, il est impossible de savoir à ce stade.

1184 //notre changement doit être propager aux processeurs voisins

1218 //pour le déraffinement. Si tous les enfants ne
