146template<typename type> int
147_displayArray(const AbstractArray<type>* obj,const char* type_name)
148{
149 Arccore::Span<const type> view = *obj;
150 const _ArrayStruct* true_ptr = reinterpret_cast<const _ArrayStruct*>(obj);
151 _ArrayMetaData* true_obj = true_ptr->m_md;
152 char strtype[1024];
153 arcane_ttf_header();
154 TV_ttf_add_row("size","Arcane::Int64",&true_obj->size);
155 TV_ttf_add_row("capacity","Arcane::Int64",&true_obj->capacity);
156 snprintf(strtype,sizeof(strtype),"%s[%ld]",type_name,view.size());
157 TV_ttf_add_row("data",strtype,view.data());
158 return TV_ttf_format_ok;
159}
160}

233ATTR_USED
234int
235TV_ttf_display_type(const Arcane::ItemInternal * obj)
236{
237 // ATTENTION: cette implémentation aggresse directement les structures Arcane.
238 // En cas de changement d'implémentation d'Arcane des incompatibilités peuvent
239 // se produire
240
241 using Arcane::Integer;
242 char strtype[32];
243
244 arcane_ttf_header();
245 TV_ttf_add_row("local_id","Integer",reinterpret_cast<const int*>(obj));
246 _ItemSharedInfo* shared_info = reinterpret_cast<_ItemSharedInfo*>(Arcane::TotalviewAdapter::getItemSharedInfo(obj));
247 const Integer local_id = obj->localId();
248
249 TV_ttf_add_row("unique_id","Int64",shared_info->m_unique_ids.ptrAt(local_id));
250 TV_ttf_add_row("kind","eItemKind",&shared_info->m_item_kind);
251
252 snprintf(strtype,sizeof(strtype),"type=%s",obj->typeInfo()->typeName().localstr());
253 //TV_ttf_add_row(strtype,"ItemTypeInfo*",&shared_info->m_item_type);
254
255 {
256 // Nodes
257 for(Integer i=0;i<obj->nbNode();++i){
258 Arcane::ItemInternal * item = obj->internalNode(i);
259 snprintf(strtype,sizeof(strtype),"node[%d] (lid=%d uid=%ld)",i,item->localId(),item->uniqueId().asInt64());
260 TV_ttf_add_row(strtype,"ItemInternal*",item);
261 }
262 }
263 {
264 // Edges
265 for(Integer i=0;i<obj->nbEdge();++i){
266 Arcane::ItemInternal* item = obj->internalEdge(i);
267 snprintf(strtype,sizeof(strtype),"edge[%d] (lid=%d uid=%ld)",i,item->localId(),item->uniqueId().asInt64());
268 TV_ttf_add_row(strtype,"ItemInternal*",item);
269 }
270 }
271 {
272 // Faces
273 for(Integer i=0;i<obj->nbFace();++i) {
274 Arcane::ItemInternal * item = obj->internalFace(i);
275 snprintf(strtype,sizeof(strtype),"face[%d] (lid=%d uid=%ld)",i,item->localId(),item->uniqueId().asInt64());
276 TV_ttf_add_row(strtype,"ItemInternal*",item);
277 }
278 }
279 {
280 // Cells
281 for(Integer i=0;i<obj->nbCell();++i){
282 Arcane::ItemInternal * item = obj->internalCell(i);
283 snprintf(strtype,sizeof(strtype),"cell[%d] (lid=%d uid=%ld)",i,item->localId(),item->uniqueId().asInt64());
284 TV_ttf_add_row(strtype,"ItemInternal*",item);
285 }
286 }
287
288 // TODO pour les DualNode et Link, mais ce n'est pas si simple...
289
290 const Integer flags = obj->flags();
291 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_Boundary,"Boundary");
292 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_HasFrontCell,"HasFrontCell");
293 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_HasBackCell,"HasBackCell");
294 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_FrontCellIsFirst,"FrontCellIsFirst");
295 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_BackCellIsFirst,"BackCellIsFirst");
296 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_Own,"Own");
297 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_Added,"Added");
298 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_Suppressed,"Suppressed");
299 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_Shared,"Shared");
300 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_SubDomainBoundary,"SubDomainBoundary");
301 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_JustAdded,"JustAdded");
302 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_NeedRemove,"NeedRemove");
303 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_SlaveFace,"SlaveFace");
304 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_MasterFace,"MasterFace");
305 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_Detached,"Detached");
306 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_HasTrace,"HasTrace");
307 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_UserMark1,"UserMark1");
308 show_ttf_internal_flag(flags,Arcane::ItemInternal::II_UserMark2,"UserMark2");
309
310 return TV_ttf_format_ok;
311}

354// Structure symétrique pour permettre des accès à des données privées.
355// Il faut etre certain que cette structure soit identique a ItemEnumerator
356struct __MY_ItemEnumerator
357{
358 Arcane::ItemInternal** m_items;
359 const Arcane::Int32* ARCANE_RESTRICT m_local_ids;
360 Arcane::Integer m_index;
361 Arcane::Integer m_count;
362 const Arcane::ItemGroupImpl * m_group_impl; // pourrait être retiré en mode release si nécessaire
363};
364
365int
366TV_ttf_display_type(const Arcane::ItemEnumerator* obj)
367{
368 arcane_ttf_header();
369
370 __MY_ItemEnumerator* mobj = (__MY_ItemEnumerator*)obj;
371
372 Arcane::Integer index = mobj->m_index;
373
374 TV_ttf_add_row("index",TV_ttf_type_int,&mobj->m_index);
375 TV_ttf_add_row("count",TV_ttf_type_int,&mobj->m_count);
376 TV_ttf_add_row("local_id",TV_ttf_type_int,&mobj->m_local_ids[index]);
377 if (index<mobj->m_count)
378 TV_ttf_add_row("item","Arcane::ItemInternal",(mobj->m_items[mobj->m_local_ids[index]]));
379 return TV_ttf_format_ok;
380}
