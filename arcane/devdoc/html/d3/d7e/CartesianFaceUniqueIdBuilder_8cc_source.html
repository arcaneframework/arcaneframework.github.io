78void CartesianFaceUniqueIdBuilder::
79computeFacesUniqueIdAndOwner()
80{
81 auto* cmgi = ICartesianMeshGenerationInfo::getReference(m_mesh, false);
82 if (!cmgi)
83 ARCANE_FATAL("No 'ICartesianMeshGenerationInfo'. The mesh is not a cartesian mesh "
84 "or was not generated by 'CartesianMeshGenerator'");
85
86 Integer dimension = m_mesh->dimension();
87 if (dimension != 2 && dimension != 3)
88 ARCANE_THROW(NotSupportedException, "Bad value '{0}' for dimension. Only dimensions 2 or 3 are supported",
89 dimension);
90 Int32 my_rank = m_mesh->parallelMng()->commRank();
91
92 Int64ConstArrayView global_nb_cells = cmgi->globalNbCells();
93 info() << "Dim=" << dimension;
94
95 CartesianGridDimension grid_dimension(global_nb_cells[0], global_nb_cells[1], global_nb_cells[2]);
96 Int64x3 nb_cell = grid_dimension.nbCell();
97 Int64x3 nb_face = grid_dimension.nbFace();
98 Int64x3 nb_face_dir = grid_dimension.nbFaceParallelToDirection();
99
100 Int64 nb_cell_xy = nb_cell.x * nb_cell.y;
101
102 Int64 face_offset_z = nb_face_dir.x + nb_face_dir.y;
103 Int64 nb_face_xyz = face_offset_z + nb_cell_xy;
104 Int64 total_nb_face_xy = (nb_face_dir.x + nb_face_dir.y) * nb_cell.z;
105 Int64 total_nb_face_x = (nb_face_dir.x * nb_cell.z);
106
107 info() << "NB_Cell: X=" << nb_cell.x << " Y=" << nb_cell.y << " Z=" << nb_cell.z
108 << " XY=" << nb_cell_xy;
109 info() << "NB_Face: X=" << nb_face.x << " Y=" << nb_face.y << " Z=" << nb_face.z
110 << " NbDirX=" << nb_face_dir.x << " NbDirY=" << nb_face_dir.y << " NbDirZ=" << nb_face_dir.z
111 << " NbFaceXYZ=" << nb_face_xyz << " OffsetZ=" << face_offset_z
112 << " TotalNbFaceX=" << total_nb_face_x << " TotalNbFaceXY=" << total_nb_face_xy;
113
114 auto own_cells_offsets = cmgi->ownCellOffsets();
115 Int64 own_cell_offset_x = own_cells_offsets[0];
116 Int64 own_cell_offset_y = own_cells_offsets[1];
117 Int64 own_cell_offset_z = own_cells_offsets[2];
118 info() << "CellOffset: X=" << own_cell_offset_x << " Y=" << own_cell_offset_y << " Z=" << own_cell_offset_z;
119
120 auto sub_domain_offsets = cmgi->subDomainOffsets();
121 Int64 sub_domain_offset_x = sub_domain_offsets[0];
122 Int64 sub_domain_offset_y = sub_domain_offsets[1];
123 Int64 sub_domain_offset_z = sub_domain_offsets[2];
124 info() << "SubDomainOffset: X=" << sub_domain_offset_x << " Y=" << sub_domain_offset_y << " Z=" << sub_domain_offset_z;
125
126 auto nb_sub_domains = cmgi->nbSubDomains();
127 Int32 nb_sub_domain_x = nb_sub_domains[0];
128 Int32 nb_sub_domain_y = nb_sub_domains[1];
129 Int32 nb_sub_domain_z = nb_sub_domains[2];
130 info() << "NbSubDomain: X=" << nb_sub_domain_x << " Y=" << nb_sub_domain_y << " Z=" << nb_sub_domain_z;
131
132 // Détermine le rang inférieur dans chaque direction
133 Int32 previous_rank_x = my_rank;
134 Int32 previous_rank_y = my_rank;
135 Int32 previous_rank_z = my_rank;
136
137 ItemInternalMap& cells_map = m_mesh->cellsMap();
138 bool is_verbose = m_is_verbose;
139
140 if (dimension == 2) {
141 if (sub_domain_offset_x > 0)
142 previous_rank_x = my_rank - 1;
143 if (sub_domain_offset_y > 0)
144 previous_rank_y = my_rank - nb_sub_domain_x;
145 info() << "PreviousRank X=" << previous_rank_x << " Y=" << previous_rank_y;
146 CartesianGridDimension::CellUniqueIdComputer2D cell_uid_computer(0, nb_cell.x);
147 CartesianGridDimension::FaceUniqueIdComputer2D face_uid_computer(grid_dimension.getFaceComputer2D(0));
148
149 // Les mailles sont des quadrangles
150 std::array<Int64, 4> face_uids;
151 std::array<Int64, 4> face_uids2;
152 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid, cells_map)
153 {
154 // Récupère l'indice (I,J) de la maille
155 Cell cell{ iid->value() };
156 Int64 uid = cell.uniqueId();
157 const Int64 y2 = uid / nb_cell.x;
158 const Int64 x2 = uid % nb_cell.x;
159 Int64x3 xyz = cell_uid_computer.compute(uid);
160 Int64 x = xyz.x;
161 Int64 y = xyz.y;
162 // Pour test. A supprimer par la suite
163 if (x != x2)
164 ARCANE_FATAL("Bad X {0} {1}", x, x2);
165 if (y != y2)
166 ARCANE_FATAL("Bad Y {0} {1}", y, y2);
167 if (is_verbose)
168 info() << "CELL (UID=" << uid << ",XY=" << x << "," << y << ") "
169 << " N0=" << cell.node(0).uniqueId()
170 << " N1=" << cell.node(1).uniqueId()
171 << " N2=" << cell.node(2).uniqueId()
172 << " N3=" << cell.node(3).uniqueId();
173 // Pour test. A supprimer par la suite et garder celui de face_uid_computer.
174 // Faces selon Y
175 face_uids[0] = (x + 0) + ((y + 0) * nb_cell.x) + nb_face_dir.x;
176 face_uids[2] = (x + 0) + ((y + 1) * nb_cell.x) + nb_face_dir.x;
177
178 // Faces selon X
179 face_uids[1] = (x + 1) + (y + 0) * nb_face.x;
180 face_uids[3] = (x + 0) + (y + 0) * nb_face.x;
181 face_uids2 = face_uid_computer.computeForCell(x, y);
182 for (int i = 0; i < 4; ++i) {
183 if (face_uids[i] != face_uids2[i])
184 ARCANE_FATAL("Bad face uid i={0} ref={1} new={2}", i, face_uids[i], face_uids2[i]);
185 }
186 for (int i = 0; i < 4; ++i) {
187 Face face = cell.face(i);
188 if (is_verbose)
189 info() << "CELL=" << uid << " Face=" << i << " uid=" << face_uids[i]
190 << " N0=" << face.node(0).uniqueId()
191 << " N1=" << face.node(1).uniqueId();
192
193 face.mutableItemBase().setUniqueId(face_uids[i]);
194 }
195 // Positionne le propriétaire de la face inférieure en X
196 if (x == own_cell_offset_x && previous_rank_x != my_rank)
197 cell.face(3).mutableItemBase().setOwner(previous_rank_x, my_rank);
198 // Positionne le propriétaire de la face inférieure en Y
199 if (y == own_cell_offset_y && previous_rank_y != my_rank)
200 cell.face(0).mutableItemBase().setOwner(previous_rank_y, my_rank);
201 }
202 }
203 else if (dimension == 3) {
204 if (sub_domain_offset_x > 0)
205 previous_rank_x = my_rank - 1;
206 if (sub_domain_offset_y > 0)
207 previous_rank_y = my_rank - nb_sub_domain_x;
208 if (sub_domain_offset_z > 0)
209 previous_rank_z = my_rank - (nb_sub_domain_x * nb_sub_domain_y);
210 info() << "PreviousRank X=" << previous_rank_x << " Y=" << previous_rank_y << " Z=" << previous_rank_z;
211 CartesianGridDimension::CellUniqueIdComputer3D cell_uid_computer(0, nb_cell.x, nb_cell_xy);
212 CartesianGridDimension::FaceUniqueIdComputer3D face_uid_computer(grid_dimension.getFaceComputer3D(0));
213 // Les mailles sont des hexaèdres
214 std::array<Int64, 6> face_uids;
215 std::array<Int64, 6> face_uids2;
216 ENUMERATE_ITEM_INTERNAL_MAP_DATA(iid, cells_map)
217 {
218 // Récupère l'indice (I,J) de la maille
219 Cell cell{ iid->value() };
220 Int64 uid = cell.uniqueId();
221 Int64 z2 = uid / nb_cell_xy;
222 Int64 v2 = uid - (z2 * nb_cell_xy);
223 Int64 y2 = v2 / nb_cell.x;
224 Int64 x2 = v2 % nb_cell.x;
225 Int64x3 xyz = cell_uid_computer.compute(uid);
226 Int64 x = xyz.x;
227 Int64 y = xyz.y;
228 Int64 z = xyz.z;
229 // Pour test. A supprimer par la suite
230 if (x != x2)
231 ARCANE_FATAL("Bad X {0} {1}", x, x2);
232 if (y != y2)
233 ARCANE_FATAL("Bad Y {0} {1}", y, y2);
234 if (z != z2)
235 ARCANE_FATAL("Bad Z {0} {1}", z, z2);
236 if (is_verbose)
237 info() << "CELL (UID=" << uid << ",XYZ=" << x << "," << y << "," << z << ") "
238 << " N0=" << cell.node(0).uniqueId()
239 << " N1=" << cell.node(1).uniqueId()
240 << " N2=" << cell.node(2).uniqueId()
241 << " N3=" << cell.node(3).uniqueId()
242 << " N4=" << cell.node(4).uniqueId()
243 << " N5=" << cell.node(5).uniqueId()
244 << " N6=" << cell.node(6).uniqueId()
245 << " N7=" << cell.node(7).uniqueId();
246
247 // Pour test. A supprimer par la suite et garder celui de face_uid_computer.
248 // Faces selon Z
249 face_uids[0] = (x + 0) + ((y + 0) * nb_cell.x) + ((z + 0) * nb_face_dir.z) + total_nb_face_xy;
250 face_uids[3] = (x + 0) + ((y + 0) * nb_cell.x) + ((z + 1) * nb_face_dir.z) + total_nb_face_xy;
251
252 // Faces selon X
253 face_uids[1] = (x + 0) + ((y + 0) * nb_face.x) + ((z + 0) * nb_face_dir.x);
254 face_uids[4] = (x + 1) + ((y + 0) * nb_face.x) + ((z + 0) * nb_face_dir.x);
255
256 // Faces selon Y
257 face_uids[2] = (x + 0) + ((y + 0) * nb_cell.x) + ((z + 0) * nb_face_dir.y) + total_nb_face_x;
258 face_uids[5] = (x + 0) + ((y + 1) * nb_cell.x) + ((z + 0) * nb_face_dir.y) + total_nb_face_x;
259
260 face_uids2 = face_uid_computer.computeForCell(x, y, z);
261 for (int i = 0; i < 6; ++i) {
262 if (face_uids[i] != face_uids2[i])
263 ARCANE_FATAL("Bad face uid i={0} ref={1} new={2}", i, face_uids[i], face_uids2[i]);
264 }
265
266 for (int i = 0; i < 6; ++i) {
267 Face face = cell.face(i);
268 if (is_verbose)
269 info() << "CELL=" << uid << " Face=" << i << " uid=" << face_uids[i]
270 << " N0=" << face.node(0).uniqueId()
271 << " N1=" << face.node(1).uniqueId()
272 << " N2=" << face.node(2).uniqueId()
273 << " N3=" << face.node(3).uniqueId();
274 face.mutableItemBase().setUniqueId(face_uids[i]);
275 }
276
277 // Positionne le propriétaire de la face inférieure en X
278 if (x == own_cell_offset_x && previous_rank_x != my_rank)
279 cell.face(1).mutableItemBase().setOwner(previous_rank_x, my_rank);
280 // Positionne le propriétaire de la face inférieure en Y
281 if (y == own_cell_offset_y && previous_rank_y != my_rank)
282 cell.face(2).mutableItemBase().setOwner(previous_rank_y, my_rank);
283 // Positionne le propriétaire de la face inférieure en Z
284 if (z == own_cell_offset_z && previous_rank_z != my_rank)
285 cell.face(0).mutableItemBase().setOwner(previous_rank_z, my_rank);
286 }
287 }
288 else
289 ARCANE_FATAL("Invalid dimension");
290}
