73class GenericPartitionerIf
74{
75 // TODO: Faire le malloc sur le device associé à la queue.
76 // et aussi regarder si on peut utiliser mallocAsync().
77
78 public:
79
80 template <typename SelectLambda, typename InputIterator, typename OutputIterator>
81 void apply(GenericPartitionerBase& s, Int32 nb_item, InputIterator input_iter, OutputIterator output_iter,
82 const SelectLambda& select_lambda)
83 {
84 eExecutionPolicy exec_policy = eExecutionPolicy::Sequential;
85 RunQueue* queue = s.m_queue;
86 if (queue)
87 exec_policy = queue->executionPolicy();
88 switch (exec_policy) {
89#if defined(ARCANE_COMPILING_CUDA)
90 case eExecutionPolicy::CUDA: {
91 size_t temp_storage_size = 0;
92 cudaStream_t stream = impl::CudaUtils::toNativeStream(queue);
93 // Premier appel pour connaitre la taille pour l'allocation
94 int* nb_list1_ptr = nullptr;
95 ARCANE_CHECK_CUDA(::cub::DevicePartition::If(nullptr, temp_storage_size,
96 input_iter, output_iter, nb_list1_ptr, nb_item,
97 select_lambda, stream));
98
99 s.m_algo_storage.allocate(temp_storage_size);
100 nb_list1_ptr = s.m_device_nb_list1_storage.allocate();
101 ARCANE_CHECK_CUDA(::cub::DevicePartition::If(s.m_algo_storage.address(), temp_storage_size,
102 input_iter, output_iter, nb_list1_ptr, nb_item,
103 select_lambda, stream));
104 s.m_device_nb_list1_storage.copyToAsync(s.m_host_nb_list1_storage, queue);
105 } break;
106#endif
107#if defined(ARCANE_COMPILING_HIP)
108 case eExecutionPolicy::HIP: {
109 size_t temp_storage_size = 0;
110 // Premier appel pour connaitre la taille pour l'allocation
111 hipStream_t stream = impl::HipUtils::toNativeStream(queue);
112 int* nb_list1_ptr = nullptr;
113 ARCANE_CHECK_HIP(rocprim::partition(nullptr, temp_storage_size, input_iter, output_iter,
114 nb_list1_ptr, nb_item, select_lambda, stream));
115
116 s.m_algo_storage.allocate(temp_storage_size);
117 nb_list1_ptr = s.m_device_nb_list1_storage.allocate();
118
119 ARCANE_CHECK_HIP(rocprim::partition(s.m_algo_storage.address(), temp_storage_size, input_iter, output_iter,
120 nb_list1_ptr, nb_item, select_lambda, stream));
121 s.m_device_nb_list1_storage.copyToAsync(s.m_host_nb_list1_storage, queue);
122 } break;
123#endif
124 case eExecutionPolicy::Thread:
125 // Pas encore implémenté en multi-thread
126 [[fallthrough]];
127 case eExecutionPolicy::Sequential: {
128 UniqueArray<bool> filter_index(nb_item);
129 auto saved_output_iter = output_iter;
130 auto output2_iter = output_iter + nb_item;
131 for (Int32 i = 0; i < nb_item; ++i) {
132 auto v = *input_iter;
133 if (select_lambda(*input_iter)) {
134 *output_iter = v;
135 ++output_iter;
136 }
137 else {
138 --output2_iter;
139 *output2_iter = v;
140 }
141 ++input_iter;
142 }
143 Int32 nb_list1 = static_cast<Int32>(output_iter - saved_output_iter);
144 s.m_host_nb_list1_storage[0] = nb_list1;
145 } break;
146 default:
147 ARCANE_FATAL(getBadPolicyMessage(exec_policy));
148 }
149 }
150};

168class GenericPartitioner

173 explicit GenericPartitioner(RunQueue* queue)

180
181 template <typename InputIterator, typename OutputIterator, typename SelectLambda>
182 void applyIf(Int32 nb_item, InputIterator input_iter, OutputIterator output_iter,
183 const SelectLambda& select_lambda)
184 {
185 _setCalled();
186 impl::GenericPartitionerBase* base_ptr = this;
187 impl::GenericPartitionerIf gf;
188 gf.apply(*base_ptr, nb_item, input_iter, output_iter, select_lambda);
189 }
190
192 Int32 nbFirstPart()
193 {
194 m_is_already_called = false;
195 return _nbFirstPart();
196 }
197
198 private:
199
200 bool m_is_already_called = false;
201
202 private:
203
204 void _setCalled()
205 {
206 if (m_is_already_called)
207 ARCANE_FATAL("apply() has already been called for this instance");
208 m_is_already_called = true;
209 }
210};
