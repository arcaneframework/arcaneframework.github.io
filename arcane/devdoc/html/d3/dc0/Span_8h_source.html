660 _sampleSpan(SpanImpl<const DataType,SizeType> values,
661  SpanImpl<const IntegerType,SizeType> indexes,
662  SpanImpl<DataType,SizeType> result)
663 {
664  const Int64 result_size = indexes.size();
665  [[maybe_unused]] const Int64 my_size = values.size();

668  IntegerType index = indexes[i];
669  ARCCORE_CHECK_AT(index,my_size);
670  result[i] = ptr[index];
671  }
