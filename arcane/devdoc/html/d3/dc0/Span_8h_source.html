802template<typename DataType,typename IntegerType,typename SizeType> inline void
803_sampleSpan(SpanImpl<const DataType,SizeType> values,
804 SpanImpl<const IntegerType,SizeType> indexes,
805 SpanImpl<DataType,SizeType> result)
806{
807 const Int64 result_size = indexes.size();
808 [[maybe_unused]] const Int64 my_size = values.size();
809 const DataType* ptr = values.data();
810 for( Int64 i=0; i<result_size; ++i) {
811 IntegerType index = indexes[i];
812 ARCCORE_CHECK_AT(index,my_size);
813 result[i] = ptr[index];
814 }
815}

879template<typename ByteType, typename DataType,Int64 Extent> inline Span<DataType>
880asSpanInternal(Span<ByteType,Extent> bytes)
881{
882 Int64 size = bytes.size();
883 if (size==0)
884 return {};
885 static constexpr Int64 data_type_size = static_cast<Int64>(sizeof(DataType));
886 static_assert(data_type_size>0,"Bad datatype size");
887 ARCCORE_ASSERT((size%data_type_size)==0,("Size is not a multiple of sizeof(DataType)"));
888 auto* ptr = reinterpret_cast<DataType*>(bytes.data());
889 return { ptr, size / data_type_size };
890}
