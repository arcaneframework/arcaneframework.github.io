63void OneStat::
64print(std::ostream& o)
65{
66 Int64 div_time = static_cast<Int64>(m_total_time * 1000.0);
67 if (div_time == 0)
68 div_time = 1;
69 if (m_nb_msg > 0) {
70 Int64 average_time = (Int64)(m_total_time / (Real)m_nb_msg);
71 o << " MPIStat " << m_name << " :" << m_nb_msg << " messages" << '\n';
72 o << " MPIStat " << m_name << " :" << m_total_size << " bytes ("
73 << m_total_size / div_time << " Kb/s) (average size "
74 << m_total_size / m_nb_msg << " bytes)" << '\n';
75 o << " MPIStat " << m_name << " Time: " << m_total_time << " seconds"
76 << " (avg=" << average_time << ")" << '\n';
77 }
78}

141OneStat* Stat::
142_find(const String& name)
143{
144 auto i = m_list.find(name);
145 if (i != m_list.end())
146 return i->second;
147
148 OneStat* os = new OneStat(name);
149 // Important: utiliser os.m_name car m_list stocke juste un
150 // pointeur sur la chaîne de caractère.
151 m_list.insert(OneStatMap::value_type(os->name(), os));
152 return os;
153}

181void StatData::
182mergeData(OneStat one_stat)
183{
184 auto pos(std::find_if(m_stat_col.m_stats.begin(), m_stat_col.m_stats.end(),
185 [&one_stat](const OneStat& os) { return (one_stat.name() == os.name()); }));
186 if (pos == m_stat_col.end())
187 m_stat_col.m_stats.emplace_back(one_stat);
188 else
189 pos->addMessage(one_stat.totalSize(), one_stat.totalTime());
190}
