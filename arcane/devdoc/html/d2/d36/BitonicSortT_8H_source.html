51 template<typename KeyType,typename KeyTypeTraits> void BitonicSort<KeyType,KeyTypeTraits>::
52 _init(ConstArrayView<KeyType> keys)
53 {
54  m_init_size = keys.size();
55 
56  // Il est indispensable que tout les procs aient la même valeur
57  // de m_size
58  m_size = m_parallel_mng->reduce(Parallel::ReduceMax, m_init_size);
59 
60  m_keys.resize(m_size);
61  m_key_ranks.resize(m_size);
62  m_key_indexes.resize(m_size);
63 
64  Int32 rank = m_parallel_mng->commRank();
65 
66  // Les valeurs de la variable aux mailles tab sont stockées
67  // dans le tableau m_key
68  for( Integer i=0; i<m_init_size; ++i ){
69  m_keys[i] = keys[i];
70  m_key_indexes[i] = i;
71  m_key_ranks[i] = rank;
72  }
73 
74  // Remplit la fin du tableau avec des valeurs non significatives
75  // Afin que chaque processeur ait le meme nombre d'elements
76  KeyType max_value = KeyTypeTraits::maxValue();
77  for( Integer i=m_init_size; i<m_size; i++){
78  m_keys[i] = max_value;
79  m_key_indexes[i] = -1;
80  m_key_ranks[i] = rank;
81  }
82 }

97 template<typename KeyType,typename KeyTypeTraits> void
98 BitonicSort<KeyType,KeyTypeTraits>::
99 sort(ConstArrayView<KeyType> keys)
100 {
101  _init(keys);
102 
103  m_nb_merge = 0;
104 
105  info() << "BITONIC_SORT want_rank?=" << m_want_index_and_rank
106  << " size=" << m_size
107  << " memsize=" << sizeof(KeyType)*m_size
108  << " structsize=" << sizeof(KeyType);
109 
110  // Appel au tri local des cles
111  _localHeapSort();
112 
113  // Définition du nombre de niveaux pour l'algorithme
114  Int32 nb_level = 0;
115  Int32 test = 1;
116  Int32 nb_rank = m_parallel_mng->commSize();
117  while (nb_rank > test){
118  nb_level++;
119  test *= 2;
120  }
121  Int32 size = 1;
122 
123  // Boucle sur les niveaux de profondeurs de l'algorithme
124  for( Int32 ilevel=0; ilevel<nb_level; ++ilevel ){
125  size *= 2;
126 
127  // Boucle sur les groupes de processeurs à traiter
128  for( Int32 jproc=0; jproc<nb_rank; jproc += size ){
129  _mergeLevels(jproc, size);
130  }
131  }
132 
133  // Le nombre d'éléments valides de la liste peut-être inférieur
134  // au nombre d'éléments alloués. Dès qu'une valeur est invalide,
135  // on arrête la liste.
136  // TODO: partir de la fin
137  for( Integer i=0; i<m_size; ++i ){
138  if (!KeyTypeTraits::isValid(m_keys[i])){
139  m_size = i;
140  break;
141  }
142  }
143 
144  info() << "END_BITONIC_SORT nb_merge=" << m_nb_merge;
145 
146  m_keys.resize(m_size);
147  m_key_ranks.resize(m_size);
148  m_key_indexes.resize(m_size);
149 }

163 template <typename KeyType,typename KeyTypeTraits> void
164 BitonicSort<KeyType,KeyTypeTraits>::
165 _mergeLevels(Int32 begin,Int32 size)
166 {
167  //etage separateur
168  for( Int32 i=0; i<size/2; ++i )
169  _mergeProcessors(begin+i, begin+size-1-i);
170 
171  if (size<=2)
172  return;
173 
174  // Application de la trieuse bitonique de niveau n/2
175  Int32 div_size = size;
176  while (div_size>1){
177  div_size /= 2;
178  for( Int32 iproc=0; iproc<size; iproc += div_size)
179  _separator(begin+iproc,div_size);
180  }
181 }

211 template <typename KeyType,typename KeyTypeTraits> void BitonicSort<KeyType,KeyTypeTraits>::
212 _mergeProcessors(Int32 iproc1, Int32 iproc2)
213 {
214  if (iproc1>=iproc2)
215  fatal()<<" Erreur dans l'algorithme de tri bitonique";
216 
217  if (iproc2>=m_parallel_mng->commSize())
218  return;
219 
220  bool is_proc2 = (iproc2 == m_parallel_mng->commRank());
221  bool is_proc1 = (iproc1 == m_parallel_mng->commRank());
222 
223  if (!is_proc1 && !is_proc2)
224  return;
225 
226  info() << "SORT iproc1=" << iproc1 << " iproc2=" << iproc2;
227  ++m_nb_merge;
228 
229  Integer buf_size = m_size;
230 
232  //traitement du processeur 2
234 
235  if (is_proc2){
236  //envoi de la liste au proc 1
237  UniqueArray<KeyType> send_keys(m_keys);
238  UniqueArray<Request> requests;
239  requests.add(KeyTypeTraits::send(m_parallel_mng,iproc1,send_keys));
240  if (m_want_index_and_rank){
241  requests.add(m_parallel_mng->send(m_key_indexes,iproc1,false));
242  requests.add(m_parallel_mng->send(m_key_ranks,iproc1,false));
243  }
244  //reception de la fin de la liste fusionnee par le proc1
245  requests.add(KeyTypeTraits::recv(m_parallel_mng,iproc1,m_keys));
246  if (m_want_index_and_rank){
247  requests.add(m_parallel_mng->recv(m_key_indexes, iproc1,false));
248  requests.add(m_parallel_mng->recv(m_key_ranks, iproc1,false));
249  }
250  m_parallel_mng->waitAllRequests(requests);
251  }
252 
254  //traitement du processeur 1
256  if (is_proc1){
257  //reception de la liste du proc2
258 
259  UniqueArray<KeyType> buf2;
260  Int32UniqueArray buf2_proc;
261  Int32UniqueArray buf2_local_id;
262  //on alloue buf2 a la taille+1 afin de placer une sentinelle en
263  //buf2[size]
264  buf2.resize(buf_size);
265  buf2_proc.resize(buf_size);
266  buf2_local_id.resize(buf_size);
267  Request recv_request = KeyTypeTraits::recv(m_parallel_mng,iproc2,buf2);
268  m_parallel_mng->waitAllRequests(ArrayView<Request>(1,&recv_request));
269  if (m_want_index_and_rank){
270  m_parallel_mng->recv(buf2_local_id, iproc2);
271  m_parallel_mng->recv(buf2_proc, iproc2);
272  }
273 
274  //a remplacer par le plus grand nombre representable par un Integer
275  buf2.add(KeyTypeTraits::maxValue());
276  //buf2[buf_size] = KeyTypeTraits::maxValue();
277 
278  // allocation d'un buffer de travail pour la fusion des deux listes
279  Integer total_size = m_size+buf_size;
280  UniqueArray<KeyType> buf_merge(total_size);
281  Int32UniqueArray buf_proc_merge(total_size);
282  Int32UniqueArray buf_local_id_merge(total_size);
283 
284 #if 0
285  info() << "SIZE=" << buf_size << " total=" << total_size << " m_size=" << m_size
286  << " proc2=" << iproc2 << " init-size=" << m_init_size;
287  for( Integer i=0; i<buf_size; ++i ){
288  info() << "L1=" << i << " v=" << buf2[i];
289  }
290  for( Integer i=0; i<m_size; ++i ){
291  info() << "L2=" << i << " v=" << m_keys[i];
292  }
293 #endif
294 
295  // Fusion des deux listes en partant du principe que chacune est
296  // déjà triée.
297  Integer cursor1 = 0;
298  Integer cursor2 = 0;
299  for( Integer i=0; i<total_size; ++i ){
300  if (cursor1>=m_size || (KeyTypeTraits::compareLess(buf2[cursor2],m_keys[cursor1]) && cursor2<buf_size)){
301  buf_merge[i] = buf2[cursor2];
302  buf_local_id_merge[i] = buf2_local_id[cursor2];
303  buf_proc_merge[i] = buf2_proc[cursor2];
304  ++cursor2;
305  }
306  else{
307  buf_merge[i] = m_keys[cursor1];
308  buf_local_id_merge[i] = m_key_indexes[cursor1];
309  buf_proc_merge[i] = m_key_ranks[cursor1];
310  ++cursor1;
311  }
312  }
313  buf2.resize(buf_size);
314 
315  // Recopie et envoi de la fin de la liste vers le proc2
316  for (Integer ii=0; ii<buf_size; ii++){
317  buf2[ii] = buf_merge[m_size+ii];
318  buf2_local_id[ii] = buf_local_id_merge[m_size+ii];
319  buf2_proc[ii] = buf_proc_merge[m_size+ii];
320  }
321  Request send_request = KeyTypeTraits::send(m_parallel_mng,iproc2,buf2);
322  m_parallel_mng->waitAllRequests(ArrayView<Request>(1,&send_request));
323  if (m_want_index_and_rank){
324  m_parallel_mng->send(buf2_local_id,iproc2);
325  m_parallel_mng->send(buf2_proc,iproc2);
326  }
327 
328  // recopie du debut de la liste dans la variable m_work_unique_id
329  for (Integer i =0; i<m_size; i++){
330  m_keys[i]=buf_merge[i];
331  m_key_indexes[i]=buf_local_id_merge[i];
332  m_key_ranks[i]=buf_proc_merge[i];
333  }
334  }
335 }

345 template<typename KeyType,typename KeyTypeTraits> void BitonicSort<KeyType,KeyTypeTraits>::
346 _localHeapSort()
347 {
348  //cas particulier d'une liste reduite à zero ou un element
349  if (m_size<2)
350  return;
351 
352  //suivant du parent du dernier element
353  Integer l = m_size/2;
354  Integer ir = m_size-1;
355  KeyType rra;
356  Integer i,j;
357  Integer tmp_local_id, tmp_proc;
358 
359  for (;;){
360  if (l>0){
361  --l;
362  rra = m_keys[l];
363  tmp_local_id= m_key_indexes[l];
364  tmp_proc = m_key_ranks[l];
365  }
366  else{
367  rra = m_keys[ir];
368  tmp_local_id= m_key_indexes[ir];
369  tmp_proc = m_key_ranks[ir];
370 
371  m_keys[ir] = m_keys[0];
372  m_key_indexes[ir] = m_key_indexes[0];
373  m_key_ranks[ir] = m_key_ranks[0];
374 
375  if (--ir==0){
376  m_keys[0] = rra;
377  m_key_indexes[0] = tmp_local_id;
378  m_key_ranks[0] = tmp_proc;
379  break;
380  }
381  }
382  i=l;
383  j=l+l+1;
384  while(j <= ir){
385  if (j<ir && KeyTypeTraits::compareLess(m_keys[j],m_keys[j+1]))
386  ++j;
387  if (KeyTypeTraits::compareLess(rra,m_keys[j])){
388  m_keys[i] = m_keys[j];
389  m_key_indexes[i] = m_key_indexes[j];
390  m_key_ranks[i] = m_key_ranks[j];
391  i=j;
392  //enfant a gauche
393  j = j * 2 + 1;
394  }
395  else
396  j=ir+1;
397  }
398  m_keys[i] = rra;
399  m_key_indexes[i] = tmp_local_id;
400  m_key_ranks[i] = tmp_proc;
401  }
402 }
