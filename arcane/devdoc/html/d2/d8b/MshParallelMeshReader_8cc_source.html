270 inline std::pair<Int64, Int64>
271 _interval(Int32 index, Int32 nb_interval, Int64 size)
272 {
273 Int64 isize = size / nb_interval;
274 Int64 ibegin = index * isize;
275 // Pour le dernier interval, prend les elements restants
276 if ((index + 1) == nb_interval)
277 isize = size - ibegin;
278 return { ibegin, isize };
279 }

361{
362 switch (mshElemType) {
363 case IT_NullType: // used to decode IT_NullType: IT_HemiHexa7|IT_Line9
364 switch (nNodes) {
365 case 7:
366 return IT_HemiHexa7;
367 default:
368 info() << "Could not decode IT_NullType with nNodes=" << nNodes;
369 throw IOException("_switchMshType", "Could not decode IT_NullType with nNodes");
370 }
371 break;
372 case MSH_PNT:
373 nNodes = 1;
374 return IT_Vertex;
375 case MSH_LIN_2:
376 nNodes = 2;
377 return IT_Line2;
378 case MSH_TRI_3:
379 nNodes = 3;
380 return IT_Triangle3;
381 case MSH_QUA_4:
382 nNodes = 4;
383 return IT_Quad4;
384 case MSH_TET_4:
385 nNodes = 4;
386 return IT_Tetraedron4;
387 case MSH_HEX_8:
388 nNodes = 8;
389 return IT_Hexaedron8;
390 case MSH_PRI_6:
391 nNodes = 6;
392 return IT_Pentaedron6;
393 case MSH_PYR_5:
394 nNodes = 5;
395 return IT_Pyramid5;
396 case MSH_TRI_10:
397 nNodes = 10;
398 return IT_Heptaedron10;
399 case MSH_TRI_12:
400 nNodes = 12;

425void MshParallelMeshReader::
426_computeNodesPartition()
427{
428 // Détermine la bounding box de la maille
429 Real max_value = FloatInfo<Real>::maxValue();
430 Real min_value = -max_value;
431 Real3 min_box(max_value, max_value, max_value);
432 Real3 max_box(min_value, min_value, min_value);
433 const Int64 nb_node = m_mesh_info.nodes_coordinates.largeSize();
434 for (Real3 pos : m_mesh_info.nodes_coordinates) {
435 min_box = math::min(min_box, pos);
436 max_box = math::max(max_box, pos);
437 }
438 m_mesh_info.m_node_min_bounding_box = min_box;
439 m_mesh_info.m_node_max_bounding_box = max_box;
440
442 UniqueArray<Int32> nodes_part(nb_node, 0);
443
444 // Pour la partition, on ne prend en compte que la coordonnée X.
445 // On est sur qu'elle est valable pour toutes les dimensions du maillage.
446 // On partitionne avec des intervalles de même longueur.
447 // NOTE: Cela fonctionne bien si l'ensemble des noeuds est bien réparti.
448 // Si ce n'est pas le cas on pourrait utiliser une bisection en coupant
449 // à chaque fois sur la moyenne.
450 Real min_x = min_box.x;
451 Real max_x = max_box.x;
452 IParallelMng* pm = m_parallel_mng;
453 Real global_min_x = pm->reduce(Parallel::ReduceMin, min_x);
454 Real global_max_x = pm->reduce(Parallel::ReduceMax, max_x);
455 info() << "MIN_MAX_X=" << global_min_x << " " << global_max_x;
456
457 Real diff_v = (global_max_x - global_min_x) / static_cast<Real>(m_nb_part);
458 // Ne devrait pas arriver mais c'est nécessaire pour éviter d'éventuelles
459 // divisions par zéro.
460 if (!math::isNearlyEqual(global_min_x, global_max_x)) {
461 for (Int64 i = 0; i < nb_node; ++i) {
462 Int32 part = static_cast<Int32>((m_mesh_info.nodes_coordinates[i].x - global_min_x) / diff_v);
463 part = std::clamp(part, 0, m_nb_part - 1);
464 nodes_part[i] = part;
465 }
466 }
467 UniqueArray<Int32> nb_node_per_rank(m_nb_part, 0);
468 // Construit la table de hashage des rangs
469 for (Int64 i = 0; i < nb_node; ++i) {
470 Int32 rank = m_parts_rank[nodes_part[i]];
471 Int64 uid = m_mesh_info.nodes_unique_id[i];
472 ++nb_node_per_rank[rank];
473 m_mesh_info.nodes_rank_map.insert(std::make_pair(uid, rank));
474 }
475 pm->reduce(Parallel::ReduceSum, nb_node_per_rank);
476 info() << "NB_NODE_PER_RANK=" << nb_node_per_rank;
477}

537void MshParallelMeshReader::
538_readNodesOneEntity(Int32 entity_index)
539{
540 IosFile* ios_file = m_ios_file.get();
541 IParallelMng* pm = m_parallel_mng;
542 const Int32 my_rank = pm->commRank();
543
544 UniqueArray<Int64> nodes_uids;
545 UniqueArray<Real3> nodes_coordinates;
546
547 FixedArray<Int64, 4> entity_infos;
548 _getInt64ArrayAndBroadcast(entity_infos.view());
549
550 _goToNextLine();
551
552 // Dimension de l'entité (pas utile)
553 [[maybe_unused]] Int64 entity_dim = entity_infos[0];
554 // Tag de l'entité (pas utile)
555 [[maybe_unused]] Int64 entity_tag = entity_infos[1];
556 Int64 parametric_coordinates = entity_infos[2];
557 Int64 nb_node2 = entity_infos[3];
558
559 info() << "[Nodes] index=" << entity_index << " entity_dim=" << entity_dim << " entity_tag=" << entity_tag
560 << " parametric=" << parametric_coordinates
561 << " nb_node2=" << nb_node2;
562
563 if (parametric_coordinates != 0)
564 ARCANE_THROW(NotSupportedException, "Only 'parametric coordinates' value of '0' is supported (current={0})", parametric_coordinates);
565
566 // Il est possible que le nombre de noeuds soit 0.
567 // Dans ce cas, il faut directement passer à la ligne suivante
568 if (nb_node2 == 0)
569 return;
570
571 // Partitionne la lecture en \a m_nb_part
572 // Pour chaque i_entity , on a d'abord la liste des identifiants puis la liste des coordonnées
573
574 for (Int32 i_part = 0; i_part < m_nb_part; ++i_part) {
575 Int64 nb_to_read = _interval(i_part, m_nb_part, nb_node2).second;
576 Int32 dest_rank = m_parts_rank[i_part];
577 info() << "Reading UIDS part i=" << i_part << " dest_rank=" << dest_rank << " nb_to_read=" << nb_to_read;
578 if (my_rank == dest_rank || my_rank == m_master_io_rank) {
579 nodes_uids.resize(nb_to_read);
580 }
581
582 // Le rang maitre lit les informations des noeuds pour la partie concernée
583 // et les transfère au rang destination
584 if (ios_file) {
585 for (Integer i = 0; i < nb_to_read; ++i) {
586 // Conserve le uniqueId() du noeuds.
587 nodes_uids[i] = ios_file->getInt64();
588 //info() << "I=" << i << " ID=" << nodes_uids[i];
589 }
590 if (dest_rank != m_master_io_rank) {
591 pm->send(nodes_uids, dest_rank);
592 }
593 }
594 else if (my_rank == dest_rank) {
595 pm->recv(nodes_uids, m_master_io_rank);
596 }
597
598 // Conserve les informations de ma partie
599 if (my_rank == dest_rank) {
600 m_mesh_info.nodes_unique_id.addRange(nodes_uids);
601 }
602 }
603
604 // Lecture par partie des coordonnées
605 for (Int32 i_part = 0; i_part < m_nb_part; ++i_part) {
606 Int64 nb_to_read = _interval(i_part, m_nb_part, nb_node2).second;
607 Int32 dest_rank = m_parts_rank[i_part];
608 info() << "Reading COORDS part i=" << i_part << " dest_rank=" << dest_rank << " nb_to_read=" << nb_to_read;
609 if (my_rank == dest_rank || my_rank == m_master_io_rank) {
610 nodes_coordinates.resize(nb_to_read);
611 }
612
613 // Le rang maitre lit les informations des noeuds pour la partie concernée
614 // et les transfère au rang destination
615 if (ios_file) {
616 for (Integer i = 0; i < nb_to_read; ++i) {
617 nodes_coordinates[i] = _getReal3();
618 //info() << "I=" << i << " ID=" << nodes_uids[i] << " COORD=" << Real3(nx, ny, nz);
619 }
620 if (dest_rank != m_master_io_rank) {
621 pm->send(nodes_coordinates, dest_rank);
622 }
623 }
624 else if (my_rank == dest_rank) {
625 pm->recv(nodes_coordinates, m_master_io_rank);
626 }
627
628 // Conserve les informations de ma partie
629 if (my_rank == dest_rank) {
630 m_mesh_info.nodes_coordinates.addRange(nodes_coordinates);
631 }
632 }
633
634 _goToNextLine();
635}

642void MshParallelMeshReader::
643_readOneElementBlock(MeshV4ElementsBlock& block)
644{
645 IosFile* ios_file = m_ios_file.get();
646 IParallelMng* pm = m_parallel_mng;
647 const Int32 my_rank = pm->commRank();
648 const Int64 nb_entity_in_block = block.nb_entity;
649 const Int32 item_nb_node = block.item_nb_node;
650
651 info() << "Reading block nb_entity=" << nb_entity_in_block << " item_nb_node=" << item_nb_node;
652
653 UniqueArray<Int64> uids;
654 UniqueArray<Int64> connectivities;
655
656 for (Int32 i_part = 0; i_part < m_nb_part; ++i_part) {
657 const Int64 nb_to_read = _interval(i_part, m_nb_part, nb_entity_in_block).second;
658 const Int32 dest_rank = m_parts_rank[i_part];
659
660 info() << "Reading block part i_part=" << i_part
661 << " nb_to_read=" << nb_to_read << " dest_rank=" << dest_rank;
662
663 const Int64 nb_uid = nb_to_read;
664 const Int64 nb_connectivity = nb_uid * item_nb_node;
665 if (my_rank == dest_rank || my_rank == m_master_io_rank) {
666 uids.resize(nb_uid);
667 connectivities.resize(nb_connectivity);
668 }
669 if (ios_file) {
670 // Utilise des Int64 pour garantir qu'on ne déborde pas.
671 for (Int64 i = 0; i < nb_uid; ++i) {
672 Int64 item_unique_id = ios_file->getInt64();
673 uids[i] = item_unique_id;
674 for (Int32 j = 0; j < item_nb_node; ++j)
675 connectivities[(i * item_nb_node) + j] = ios_file->getInt64();
676 }
677 if (dest_rank != m_master_io_rank) {
678 pm->send(uids, dest_rank);
679 pm->send(connectivities, dest_rank);
680 }
681 }
682 else if (my_rank == dest_rank) {
683 pm->recv(uids, m_master_io_rank);
684 pm->recv(connectivities, m_master_io_rank);
685 }
686 if (my_rank == dest_rank) {
687 block.uids.addRange(uids);
688 block.connectivities.addRange(connectivities);
689 }
690 }
691}

723Integer MshParallelMeshReader::
724_readElementsFromFileAscii()
725{
726 IosFile* ios_file = m_ios_file.get();
727 IParallelMng* pm = m_parallel_mng;
728
729 FixedArray<Int64, 4> elements_info;
730 _getInt64ArrayAndBroadcast(elements_info.view());
731
732 Int64 nb_block = elements_info[0];
733 Int64 number_of_elements = elements_info[1];
734 Int64 min_element_tag = elements_info[2];
735 Int64 max_element_tag = elements_info[3];
736
737 _goToNextLine();
738
739 info() << "[Elements] nb_block=" << nb_block
740 << " nb_elements=" << number_of_elements
741 << " min_element_tag=" << min_element_tag
742 << " max_element_tag=" << max_element_tag;
743
744 if (number_of_elements < 0)
745 ARCANE_THROW(IOException, "Invalid number of elements: {0}", number_of_elements);
746
747 UniqueArray<MeshV4ElementsBlock>& blocks = m_mesh_info.blocks;
748 blocks.resize(nb_block);
749
750 {
751 // Numérote les blocs (pour le débug)
752 Integer index = 0;
753 for (MeshV4ElementsBlock& block : blocks) {
754 block.index = index;
755 ++index;
756 }
757 }
758
759 for (MeshV4ElementsBlock& block : blocks) {
760
761 FixedArray<Int64, 4> block_info;
762 _getInt64ArrayAndBroadcast(block_info.view());
763
764 Int32 entity_dim = CheckedConvert::toInt32(block_info[0]);
765 Int64 entity_tag = block_info[1];
766 Int64 entity_type = block_info[2];
767 Int64 nb_entity_in_block = block_info[3];
768
769 Integer item_nb_node = 0;
770 Integer item_type = _switchMshType(entity_type, item_nb_node);
771
772 info() << "[Elements] index=" << block.index << " entity_dim=" << entity_dim
773 << " entity_tag=" << entity_tag
774 << " entity_type=" << entity_type << " nb_in_block=" << nb_entity_in_block
775 << " item_type=" << item_type << " item_nb_node=" << item_nb_node;
776
777 block.nb_entity = nb_entity_in_block;
778 block.item_type = item_type;
779 block.item_nb_node = item_nb_node;
780 block.dimension = entity_dim;
781 block.entity_tag = entity_tag;
782
783 if (entity_type == MSH_PNT) {
784 // Si le type est un point, le traitement semble un peu particulier.
785 // Il y a dans ce cas deux entiers dans la ligne suivante:
786 // - un entier qui ne semble pas être utilisé
787 // - le numéro unique du noeud qui nous intéresse
788 Int64 item_unique_id = NULL_ITEM_UNIQUE_ID;
789 if (ios_file) {
790 [[maybe_unused]] Int64 unused_id = ios_file->getInt64();
791 item_unique_id = ios_file->getInt64();
792 info() << "Adding unique node uid=" << item_unique_id;
793 }
794 if (m_is_parallel)
795 pm->broadcast(ArrayView<Int64>(1, &item_unique_id), m_master_io_rank);
796 block.uids.add(item_unique_id);
797 }
798 else {
799 _readOneElementBlock(block);
800 }
801 _goToNextLine();
802 }
803
804 // Maintenant qu'on a tout les blocs, la dimension du maillage est
805 // la plus grande dimension des blocs
806 Integer mesh_dimension = -1;
807 for (const MeshV4ElementsBlock& block : blocks)
808 mesh_dimension = math::max(mesh_dimension, block.dimension);
809 if (mesh_dimension < 0)
810 ARCANE_FATAL("Invalid computed mesh dimension '{0}'", mesh_dimension);
811 if (mesh_dimension != 2 && mesh_dimension != 3)
812 ARCANE_THROW(NotSupportedException, "mesh dimension '{0}'. Only 2D or 3D meshes are supported", mesh_dimension);
813 info() << "Computed mesh dimension = " << mesh_dimension;
814
815 for (MeshV4ElementsBlock& block : blocks) {
816 if (block.dimension == mesh_dimension)
817 _computeOwnCells(block);
818 }
819
820 return mesh_dimension;
821}

863void MshParallelMeshReader::
864_computeOwnCells(MeshV4ElementsBlock& block)
865{
866 // On ne conserve que les mailles dont le premier noeud appartient à notre rang.
867
868 IParallelMng* pm = m_parallel_mng;
869 const Int32 my_rank = pm->commRank();
870
871 const Int32 item_type = block.item_type;
872 const Int32 item_nb_node = block.item_nb_node;
873
874 UniqueArray<Int64> connectivities;
875 UniqueArray<Int64> uids;
876 UniqueArray<Int32> nodes_rank;
877
878 const Int32 nb_part = m_parts_rank.size();
879 for (Int32 i_part = 0; i_part < nb_part; ++i_part) {
880 const Int32 dest_rank = m_parts_rank[i_part];
881 // Broadcast la i_part-ème partie des uids et connectivités des mailles
882 ArrayView<Int64> connectivities_view = _broadcastArray(pm, block.connectivities, connectivities, dest_rank);
883 ArrayView<Int64> uids_view = _broadcastArray(pm, block.uids, uids, dest_rank);
884
885 Int32 nb_item = uids_view.size();
886 nodes_rank.resize(nb_item);
887 nodes_rank.fill(-1);
888
889 // Parcours les mailles. Chaque maille appartiendra au rang
890 // de son premier noeud. Si cette partie correspond à mon rang, alors
891 // on conserve la maille.
892 for (Int32 i = 0; i < nb_item; ++i) {
893 Int64 first_node_uid = connectivities_view[i * item_nb_node];
894 auto x = m_mesh_info.nodes_rank_map.find(first_node_uid);
895 if (x == m_mesh_info.nodes_rank_map.end())
896 // Le noeud n'est pas dans ma liste
897 continue;
898 Int32 rank = x->second;
899 nodes_rank[i] = rank;
900 }
901 pm->reduce(Parallel::ReduceMax, nodes_rank);
902 for (Int32 i = 0; i < nb_item; ++i) {
903 const Int32 rank = nodes_rank[i];
904 if (rank != my_rank)
905 // Le noeud n'est pas dans ma partie
906 continue;
907 // Le noeud est chez moi, j'ajoute ma maille à la liste des
908 // mailles que je vais créer.
909 m_mesh_info.cells_type.add(item_type);
910 m_mesh_info.cells_nb_node.add(item_nb_node);
911 m_mesh_info.cells_uid.add(uids_view[i]);
912 auto v = connectivities_view.subView(i * item_nb_node, item_nb_node);
913 m_mesh_info.cells_connectivity.addRange(v);
914 }
915 }

1089void MshParallelMeshReader::
1090_addFaceGroupOnePart(ConstArrayView<Int64> connectivities, Int32 item_nb_node,
1091 const String& group_name, Int32 block_index)
1092{
1093 IMesh* mesh = m_mesh;
1094 const Int32 nb_entity = connectivities.size() / item_nb_node;
1095
1096 // Il peut y avoir plusieurs blocs pour le même groupe.
1097 // On récupère le groupe s'il existe déjà.
1098 FaceGroup face_group = mesh->faceFamily()->findGroup(group_name, true);
1099
1100 UniqueArray<Int32> faces_id; // Numéro de la face dans le maillage \a mesh
1101 faces_id.reserve(nb_entity);
1102
1103 const Int32 face_nb_node = nb_entity * item_nb_node;
1104
1105 UniqueArray<Int64> faces_first_node_unique_id(nb_entity);
1106 UniqueArray<Int32> faces_first_node_local_id(nb_entity);
1107 UniqueArray<Int64> faces_nodes_unique_id(face_nb_node);
1108 Integer faces_nodes_unique_id_index = 0;
1109
1110 UniqueArray<Int64> orig_nodes_id(item_nb_node);
1111 UniqueArray<Integer> face_nodes_index(item_nb_node);
1112
1113 IItemFamily* node_family = mesh->nodeFamily();
1114 NodeInfoListView mesh_nodes(node_family);
1115
1116 // Réordonne les identifiants des faces retrouver la face dans le maillage
1117 for (Integer i_face = 0; i_face < nb_entity; ++i_face) {
1118 for (Integer z = 0; z < item_nb_node; ++z)
1119 orig_nodes_id[z] = connectivities[faces_nodes_unique_id_index + z];
1120
1121 MeshUtils::reorderNodesOfFace2(orig_nodes_id, face_nodes_index);
1122 for (Integer z = 0; z < item_nb_node; ++z)
1123 faces_nodes_unique_id[faces_nodes_unique_id_index + z] = orig_nodes_id[face_nodes_index[z]];
1124 faces_first_node_unique_id[i_face] = orig_nodes_id[face_nodes_index[0]];
1125 faces_nodes_unique_id_index += item_nb_node;
1126 }
1127
1128 node_family->itemsUniqueIdToLocalId(faces_first_node_local_id, faces_first_node_unique_id, false);
1129
1130 faces_nodes_unique_id_index = 0;
1131 for (Integer i_face = 0; i_face < nb_entity; ++i_face) {
1132 const Integer n = item_nb_node;
1133 Int32 face_first_node_lid = faces_first_node_local_id[i_face];
1134 if (face_first_node_lid != NULL_ITEM_LOCAL_ID) {
1135 Int64ConstArrayView face_nodes_id(item_nb_node, &faces_nodes_unique_id[faces_nodes_unique_id_index]);
1136 Node current_node(mesh_nodes[faces_first_node_local_id[i_face]]);
1137 Face face = MeshUtils::getFaceFromNodesUnique(current_node, face_nodes_id);
1138
1139 // En parallèle, il est possible que la face ne soit pas dans notre sous-domaine
1140 // même si un de ses noeuds l'est
1141 if (face.null()) {
1142 if (!m_is_parallel) {
1143 OStringStream ostr;
1144 ostr() << "(Nodes:";
1145 for (Integer z = 0; z < n; ++z)
1146 ostr() << ' ' << face_nodes_id[z];
1147 ostr() << " - " << current_node.localId() << ")";
1148 ARCANE_FATAL("INTERNAL: MeshMeshReader face index={0} with nodes '{1}' is not in node/face connectivity",
1149 i_face, ostr.str());
1150 }
1151 }
1152 else
1153 faces_id.add(face.localId());
1154 }
1155
1156 faces_nodes_unique_id_index += n;
1157 }
1158 info(4) << "Adding " << faces_id.size() << " faces from block index=" << block_index
1159 << " to group '" << face_group.name() << "'";
1160 face_group.addItems(faces_id);
1161}
