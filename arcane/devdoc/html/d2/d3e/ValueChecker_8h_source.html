65  template<typename T>
66  void areEqualArray(Span<const T> values,Span<const T> expected_values,
67  const String& message)
68  {
69  Int64 nb_value = values.size();
70  Int64 nb_expected = expected_values.size();
71  if (nb_value!=nb_expected){
72  _addError(String::format("{0} bad array size n={1} expected={2}",
73  message,nb_value,nb_expected));
74  // Ne compare pas les éléments du tableau si les tailles
75  // sont différentes.
76  return;
77  }
78 
79  for( Int64 i=0; i<nb_value; ++i ){
80  const T& v = values[i];
81  const T& e = expected_values[i];
82  if (v!=e){
83  _addError(String::format("{0} index={1} value={2} expected={3}",message,i,v,e));
84  }
85  }
86  }
