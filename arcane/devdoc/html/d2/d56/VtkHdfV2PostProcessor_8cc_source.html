264void VtkHdfV2DataWriter::
265beginWrite(const VariableCollection& vars)
266{
267 ARCANE_UNUSED(vars);
268
269 IParallelMng* pm = m_mesh->parallelMng();
270 const Int32 nb_rank = pm->commSize();
271 m_is_parallel = nb_rank > 1;
272 m_is_master_io = pm->isMasterIO();
273
274 Int32 time_index = m_times.size();
275 const bool is_first_call = (time_index < 2);
276 m_is_first_call = is_first_call;
277 if (is_first_call)
278 pwarning() << "L'implémentation au format 'VtkHdfV2' est expérimentale";
279
280 String filename = _getFileName();
281
282 Directory dir(m_directory_name);
283
284 m_full_filename = dir.file(filename);
285 info(4) << "VtkHdfV2DataWriter::beginWrite() file=" << m_full_filename;
286
287 HInit();
288
289 // Il est possible d'utiliser le mode collectif de HDF5 via MPI-IO dans les cas suivants:
290 // - Hdf5 a été compilé avec MPI
291 // - on est en mode MPI pure (ni mode mémoire partagé, ni mode hybride)
292 m_is_collective_io = pm->isParallel() && HInit::hasParallelHdf5();
293 if (pm->isHybridImplementation() || pm->isThreadImplementation())
294 m_is_collective_io = false;
295
296 if (is_first_call)
297 info() << "VtkHdfV2DataWriter: using collective MPI/IO ?=" << m_is_collective_io;
298
299 // Vrai si on doit participer aux écritures
300 // Si on utilise MPI/IO avec HDF5, il faut tout de même que tous
301 // les rangs fassent toutes les opérations d'écriture pour garantir
302 // la cohérence des méta-données.
303 m_is_writer = m_is_master_io || m_is_collective_io;
304
305 // Indique qu'on utilise MPI/IO si demandé
306 HProperty plist_id;
307 if (m_is_collective_io)
308 plist_id.createFilePropertyMPIIO(pm);
309
310 if (is_first_call && m_is_master_io)
311 dir.createDirectory();
312
313 if (m_is_collective_io)
314 pm->barrier();
315
316 if (m_is_writer) {
317 m_standard_types.initialize();
318
319 if (is_first_call)
320 m_file_id.openTruncate(m_full_filename, plist_id.id());
321 else
322 m_file_id.openAppend(m_full_filename, plist_id.id());
323
324 _openOrCreateGroups();
325
326 if (is_first_call) {
327 std::array<Int64, 2> version = { 2, 0 };
328 _addInt64ArrayAttribute(m_top_group, "Version", version);
329 _addStringAttribute(m_top_group, "Type", "UnstructuredGrid");
330 }
331 }
332
333 CellGroup all_cells = m_mesh->allCells();
334 NodeGroup all_nodes = m_mesh->allNodes();
335
336 const Int32 nb_cell = all_cells.size();
337 const Int32 nb_node = all_nodes.size();
338
339 Int32 total_nb_connected_node = 0;
340 ENUMERATE_ (Cell, icell, all_cells) {
341 Cell cell = *icell;
342 total_nb_connected_node += cell.nodeIds().size();
343 }
344
345 // Pour les offsets, la taille du tableau est égal
346 // au nombre de mailles plus 1.
347 UniqueArray<Int64> cells_connectivity(total_nb_connected_node);
348 UniqueArray<Int64> cells_offset(nb_cell + 1);
349 UniqueArray<unsigned char> cells_ghost_type(nb_cell);
350 UniqueArray<unsigned char> cells_type(nb_cell);
351 UniqueArray<Int64> cells_uid(nb_cell);
352 cells_offset[0] = 0;
353 {
354 Int32 connected_node_index = 0;
355 ENUMERATE_CELL (icell, all_cells) {
356 Int32 index = icell.index();
357 Cell cell = *icell;
358
359 cells_uid[index] = cell.uniqueId();
360
361 Byte ghost_type = 0;
362 bool is_ghost = !cell.isOwn();
363 if (is_ghost)
364 ghost_type = VtkUtils::CellGhostTypes::DUPLICATECELL;
365 cells_ghost_type[index] = ghost_type;
366
367 unsigned char vtk_type = VtkUtils::arcaneToVtkCellType(cell.type());
368 cells_type[index] = vtk_type;
369 for (NodeLocalId node : cell.nodeIds()) {
370 cells_connectivity[connected_node_index] = node;
371 ++connected_node_index;
372 }
373 cells_offset[index + 1] = connected_node_index;
374 }
375 }
376
377 _initializeOffsets();
378
379 // TODO: faire un offset pour cet objet (ou regarder comment le calculer automatiquement
380 _writeDataSet1DCollective<Int64>({ { m_top_group, "Offsets" }, m_offset_for_cell_offset_info }, cells_offset);
381
382 _writeDataSet1DCollective<Int64>({ { m_top_group, "Connectivity" }, m_connectivity_offset_info },
383 cells_connectivity);
384 _writeDataSet1DCollective<unsigned char>({ { m_top_group, "Types" }, m_cell_offset_info }, cells_type);
385
386 {
387 Int64 nb_cell_int64 = nb_cell;
388 _writeDataSet1DCollective<Int64>({ { m_top_group, "NumberOfCells" }, m_part_offset_info },
389 asConstSpan(&nb_cell_int64));
390 Int64 nb_node_int64 = nb_node;
391 _writeDataSet1DCollective<Int64>({ { m_top_group, "NumberOfPoints" }, m_part_offset_info },
392 asConstSpan(&nb_node_int64));
393 Int64 number_of_connectivity_ids = cells_connectivity.size();
394 _writeDataSet1DCollective<Int64>({ { m_top_group, "NumberOfConnectivityIds" }, m_part_offset_info },
395 asConstSpan(&number_of_connectivity_ids));
396 }
397
398 // Sauve les uniqueIds, les types et les coordonnées des noeuds.
399 {
400 UniqueArray<Int64> nodes_uid(nb_node);
401 UniqueArray<unsigned char> nodes_ghost_type(nb_node);
402 VariableNodeReal3& nodes_coordinates(m_mesh->nodesCoordinates());
403 UniqueArray2<Real> points;
404 points.resize(nb_node, 3);
405 ENUMERATE_ (Node, inode, all_nodes) {
406 Int32 index = inode.index();
407 Node node = *inode;
408
409 nodes_uid[index] = node.uniqueId();
410
411 Byte ghost_type = 0;
412 bool is_ghost = !node.isOwn();
413 if (is_ghost)
414 ghost_type = VtkUtils::PointGhostTypes::DUPLICATEPOINT;
415 nodes_ghost_type[index] = ghost_type;
416
417 Real3 pos = nodes_coordinates[inode];
418 points[index][0] = pos.x;
419 points[index][1] = pos.y;
420 points[index][2] = pos.z;
421 }
422
423 // Sauve l'uniqueId de chaque noeud dans le dataset "GlobalNodeId".
424 _writeDataSet1DCollective<Int64>({ { m_node_data_group, "GlobalNodeId" }, m_cell_offset_info }, nodes_uid);
425
426 // Sauve les informations sur le type de noeud (réel ou fantôme).
427 _writeDataSet1DCollective<unsigned char>({ { m_node_data_group, "vtkGhostType" }, m_cell_offset_info }, nodes_ghost_type);
428
429 // Sauve les coordonnées des noeuds.
430 _writeDataSet2DCollective<Real>({ { m_top_group, "Points" }, m_point_offset_info }, points);
431 }
432
433 // Sauve les informations sur le type de maille (réel ou fantôme)
434 _writeDataSet1DCollective<unsigned char>({ { m_cell_data_group, "vtkGhostType" }, m_cell_offset_info }, cells_ghost_type);
435
436 // Sauve l'uniqueId de chaque maille dans le dataset "GlobalCellId".
437 // L'utilisation du dataset "vtkOriginalCellIds" ne fonctionne pas dans Paraview.
438 _writeDataSet1DCollective<Int64>({ { m_cell_data_group, "GlobalCellId" }, m_cell_offset_info }, cells_uid);
439
440 if (m_is_writer) {
441
442 // Liste des temps.
443 Real current_time = m_times[time_index - 1];
444 _writeDataSet1D<Real>({ { m_steps_group, "Values" }, m_time_offset_info }, asConstSpan(&current_time));
445
446 // Offset de la partie.
447 Int64 part_offset = (time_index - 1) * pm->commSize();
448 _writeDataSet1D<Int64>({ { m_steps_group, "PartOffsets" }, m_time_offset_info }, asConstSpan(&part_offset));
449
450 // Nombre de temps
451 _addInt64ttribute(m_steps_group, "NSteps", time_index);
452 }
453}

501template <typename DataType> void VtkHdfV2DataWriter::
502_writeDataSetGeneric(const DataInfo& data_info, Int32 nb_dim,
503 Int64 dim1_size, Int64 dim2_size, const DataType* values_data,
504 bool is_collective)
505{
506 HGroup& group = data_info.dataset.group;
507 const String& name = data_info.dataset.name;
508
509 // Si positif ou nul indique l'offset d'écriture. Sinon on écrit à la fin
510 Int64 wanted_offset = data_info.offset.value();
511
512 // TODO: utiliser une structure qui encapsule les dimensions pour vérifier
513 // les éventuels débordements de tableau.
514 static constexpr int MAX_DIM = 2;
515 HDataset dataset;
516
517 // En cas d'opération collective, local_dims et global_dims sont
518 // différents sur la première dimension. La deuxième dimension est toujours
519 // identique pour local_dims et global_dims et ne doit pas être modifiée durant
520 // tout le calcul.
521
522 // Dimensions du dataset que le rang courant va écrire.
523 hsize_t local_dims[MAX_DIM];
524 local_dims[0] = dim1_size;
525 local_dims[1] = dim2_size;
526
527 // Dimensions cumulées de tous les rangs pour l'écriture.
528 hsize_t global_dims[MAX_DIM];
529
530 // Dimensions maximales du DataSet
531 // Pour la deuxième dimension, on suppose qu'elle est constante au cours du temps.
532 hsize_t max_dims[MAX_DIM];
533 max_dims[0] = H5S_UNLIMITED;
534 max_dims[1] = dim2_size;
535
536 const hid_t hdf_type = HDFTraits<DataType>::hdfType();
537 herr_t herror = 0;
538 Int64 write_offset = 0;
539
540 Int64 my_index = 0;
541 Int64 global_dim1_size = dim1_size;
542 Int32 nb_participating_rank = 1;
543 if (is_collective) {
544 // En mode collectif il faut récupérer les index de chaque rang.
545 // TODO: pour les variables, ces indices ne dépendent que du groupe associé
546 // et on peut donc conserver l'information pour éviter le gather à chaque fois.
547 IParallelMng* pm = m_mesh->parallelMng();
548 nb_participating_rank = pm->commSize();
549 Int32 my_rank = pm->commRank();
550
551 UniqueArray<Int64> all_sizes(nb_participating_rank);
552 pm->allGather(ConstArrayView<Int64>(1, &dim1_size), all_sizes);
553
554 global_dim1_size = 0;
555 for (Integer i = 0; i < nb_participating_rank; ++i)
556 global_dim1_size += all_sizes[i];
557 my_index = 0;
558 for (Integer i = 0; i < my_rank; ++i)
559 my_index += all_sizes[i];
560 }
561
562 HProperty write_plist_id;
563 if (is_collective)
564 write_plist_id.createDatasetTransfertCollectiveMPIIO();
565
566 HSpace memory_space;
567 memory_space.createSimple(nb_dim, local_dims);
568
569 HSpace file_space;
570
571 if (m_is_first_call) {
572 // TODO: regarder comment mieux calculer le chunk
573 hsize_t chunk_dims[MAX_DIM];
574 global_dims[0] = global_dim1_size;
575 global_dims[1] = dim2_size;
576 // Il est important que tout le monde ait la même taille de chunk.
577 Int64 chunk_size = global_dim1_size / nb_participating_rank;
578 if (chunk_size < 1024)
579 chunk_size = 1024;
580 chunk_dims[0] = chunk_size;
581 chunk_dims[1] = dim2_size;
582 info(4) << "CHUNK nb_dim=" << nb_dim
583 << " chunk0=" << chunk_dims[0]
584 << " name=" << name
585 << " nb_byte=" << global_dim1_size * sizeof(DataType);
586 file_space.createSimple(nb_dim, global_dims, max_dims);
587 HProperty plist_id;
588 plist_id.create(H5P_DATASET_CREATE);
589 H5Pset_chunk(plist_id.id(), nb_dim, chunk_dims);
590
591 dataset.create(group, name.localstr(), hdf_type, file_space, HProperty{}, plist_id, HProperty{});
592
593 if (is_collective) {
594 hsize_t offset[MAX_DIM];
595 offset[0] = my_index;
596 offset[1] = 0;
597 if ((herror = H5Sselect_hyperslab(file_space.id(), H5S_SELECT_SET, offset, nullptr, local_dims, nullptr)) < 0)
598 ARCANE_THROW(IOException, "Can not select hyperslab '{0}' (err={1})", name, herror);
599 }
600 }
601 else {
602 // Agrandit la première dimension du dataset.
603 // On va ajouter 'global_dim1_size' à cette dimension.
604 dataset.open(group, name.localstr());
605 file_space = dataset.getSpace();
606 int nb_dimension = file_space.nbDimension();
607 if (nb_dimension != nb_dim)
608 ARCANE_THROW(IOException, "Bad dimension '{0}' for dataset '{1}' (should be 1)",
609 nb_dimension, name);
610 // TODO: Vérifier que la deuxième dimension est la même que celle sauvée.
611 hsize_t original_dims[MAX_DIM];
612 file_space.getDimensions(original_dims, nullptr);
613 hsize_t offset0 = original_dims[0];
614 // Si on a un offset positif issu de OffsetInfo alors on le prend.
615 // Cela signifie qu'on a fait un retour arrière.
616 if (wanted_offset >= 0) {
617 offset0 = wanted_offset;
618 info() << "Forcing offset to " << wanted_offset;
619 }
620 global_dims[0] = offset0 + global_dim1_size;
621 global_dims[1] = dim2_size;
622 write_offset = offset0;
623 // Agrandit le dataset.
624 // ATTENTION cela invalide file_space. Il faut donc le relire juste après.
625 if ((herror = dataset.setExtent(global_dims)) < 0)
626 ARCANE_THROW(IOException, "Can not extent dataset '{0}' (err={1})", name, herror);
627 file_space = dataset.getSpace();
628
629 hsize_t offsets[MAX_DIM];
630 offsets[0] = offset0 + my_index;
631 offsets[1] = 0;
632 info(4) << "APPEND nb_dim=" << nb_dim
633 << " dim0=" << global_dims[0]
634 << " count0=" << local_dims[0]
635 << " offsets0=" << offsets[0] << " name=" << name;
636
637 if ((herror = H5Sselect_hyperslab(file_space.id(), H5S_SELECT_SET, offsets, nullptr, local_dims, nullptr)) < 0)
638 ARCANE_THROW(IOException, "Can not select hyperslab '{0}' (err={1})", name, herror);
639 }
640
641 // Effectue l'écriture
642 if ((herror = dataset.write(hdf_type, values_data, memory_space, file_space, write_plist_id)) < 0)
643 ARCANE_THROW(IOException, "Can not write dataset '{0}' (err={1})", name, herror);
644
645 if (dataset.isBad())
646 ARCANE_THROW(IOException, "Can not write dataset '{0}'", name);
647
648 if (!data_info.offset.isNull())
649 m_offset_info_list.insert(std::make_pair(data_info.offset, write_offset));
650}

665_writeDataSet1DUsingCollectiveIO(const DataInfo& data_info, Span<const DataType> values)

667 _writeDataSetGeneric(data_info, 1, values.size(), 1, values.data(), true);

868
869void VtkHdfV2DataWriter::
870write(IVariable* var, IData* data)
871{
872 info(4) << "Write VtkHdfV2 var=" << var->name();
873
874 eItemKind item_kind = var->itemKind();
875
876 if (var->dimension() != 1)
877 ARCANE_FATAL("Only export of scalar item variable is implemented (name={0})", var->name());
878
879 HGroup* group = nullptr;
880 OffsetInfo offset_info;
881 switch (item_kind) {
882 case IK_Cell:
883 group = &m_cell_data_group;
884 offset_info = m_cell_offset_info;
885 break;
886 case IK_Node:
887 group = &m_node_data_group;
888 offset_info = m_point_offset_info;
889 break;
890 default:
891 ARCANE_FATAL("Only export of 'Cell' or 'Node' variable is implemented (name={0})", var->name());
892 }
893
894 ARCANE_CHECK_POINTER(group);
895
896 DataInfo data_info{ { *group, var->name() }, offset_info };
897 eDataType data_type = var->dataType();
898 switch (data_type) {
899 case DT_Real:
900 _writeBasicTypeDataset<Real>(data_info, data);
901 break;
902 case DT_Int64:
903 _writeBasicTypeDataset<Int64>(data_info, data);
904 break;
905 case DT_Int32:
906 _writeBasicTypeDataset<Int32>(data_info, data);
907 break;
908 case DT_Real3:
909 _writeReal3Dataset(data_info, data);
910 break;
911 case DT_Real2:
912 _writeReal2Dataset(data_info, data);
913 break;
914 default:
915 warning() << String::format("Export for datatype '{0}' is not supported (var_name={1})", data_type, var->name());
916 }
917}

991/*---------------------------------------------------------------------------*/
992/*---------------------------------------------------------------------------*/
993
994void VtkHdfV2DataWriter::
995_initializeOffsets()
996{
997 // Il y a 5 valeurs d'offset utilisées:
998 // - offset sur le nombre de mailles (CellOffsets). Cet offset a pour nombre d'éléments
999 // le nombre de temps sauvés et est augmenté à chaque sortie du nombre de mailles. Cet offset
1000 // est aussi utiliser pour les variables aux mailles
1001 // - offset sur le nombre de noeuds (PointOffsets). Il équivalent à 'CellOffsets' mais
1002 // pour les noeuds.
1003 // - offset pour "NumberOfCells", "NumberOfPoints" et "NumberOfConnectivityIds". Pour chacun
1004 // de ces champs il y a NbPart valeurs par temps, avec 'NbPart' le nombre de parties (donc
1005 // le nombre de sous-domaines si on ne fait pas de regroupement). Il y a donc au total
1006 // NbPart * NbTimeStep dans ce champ d'offset.
1007 // - offset pour le champ "Connectivity" qui s'appelle "ConnectivityIdOffsets".
1008 // Cet offset a pour nombre d'éléments le nombre de temps sauvés.
1009 // - offset pour le champ "Offsets". "Offset" contient pour chaque maille l'offset dans
1010 // "Connectivity" de la connectivité des noeuds de la maille. Cet offset n'est pas sauvés
1011 // mais comme ce champ à un nombre de valeur égale au nombre de mailles plus 1 il est possible
1012 // de le déduire de "CellOffsets" (il vaut "CellOffsets" plus l'index du temps courant).
1013
1014 m_cell_offset_info = OffsetInfo(m_steps_group, "CellOffsets");
1015 m_point_offset_info = OffsetInfo(m_steps_group, "PointOffsets");
1016 m_connectivity_offset_info = OffsetInfo(m_steps_group, "ConnectivityIdOffsets");
1017 // Ces trois offsets ne sont pas sauvegardés dans le format VTK
1018 m_offset_for_cell_offset_info = OffsetInfo("_OffsetForCellOffsetInfo");
1019 m_part_offset_info = OffsetInfo("_PartOffsetInfo");
1020 m_time_offset_info = OffsetInfo("_TimeOffsetInfo");
1021
1022 // Regarde si on n'a pas fait de retour-arrière.
1023 // C'est le cas si le nombre de temps sauvés est supérieur au nombre
1024 // de valeurs de \a m_times.
1025 if (m_is_writer && !m_is_first_call) {
1026 IParallelMng* pm = m_mesh->parallelMng();
1027 const Int32 nb_rank = pm->commSize();
1028 Int64 nb_current_step = _readInt64Attribute(m_steps_group, "NSteps");
1029 Int32 time_index = m_times.size();
1030 info(4) << "NB_STEP=" << nb_current_step << " time_index=" << time_index
1031 << " current_time=" << m_times.back();
1032 const bool debug_times = false;
1033 if (debug_times) {
1034 StandardArrayT<Real> a1(m_steps_group.id(), "Values");
1035 UniqueArray<Real> times;
1036 a1.directRead(m_standard_types, times);
1037 info() << "TIMES=" << times;
1038 }
1039 if ((nb_current_step + 1) != time_index) {
1040 info() << "[VtkHdf] go_backward detected";
1041 Int32 wanted_step = time_index - 1;
1042 // Signifie qu'on a fait un retour arrière.
1043 // Dans ce cas il faut relire les offsets
1044 _readAndSetOffset(m_cell_offset_info, wanted_step);
1045 _readAndSetOffset(m_point_offset_info, wanted_step);
1046 _readAndSetOffset(m_connectivity_offset_info, wanted_step);
1047 m_part_offset_info.setValue(wanted_step * nb_rank);
1048 m_time_offset_info.setValue(wanted_step);
1049 m_offset_for_cell_offset_info.setValue(m_cell_offset_info.value() + wanted_step * nb_rank);
1050 }
1051 }
1052}
