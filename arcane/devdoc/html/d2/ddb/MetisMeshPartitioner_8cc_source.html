138partitionMesh(bool initial_partition,Int32 nb_part)
139{
140 // Signal que le partitionnement peut planter car metis n'est pas toujours
141 // tres fiable.
142 initial_partition = (subDomain()->commonVariables().globalIteration() <= 2);

154void MetisMeshPartitioner::
155_partitionMesh(bool initial_partition,Int32 nb_part)
156{
157 ISubDomain* sd = subDomain();
158 IParallelMng* pm = m_parallel_mng;
159 Int32 my_rank = pm->commRank();
160 Int32 nb_rank = pm->commSize();
161 IMesh* mesh = this->mesh();
162 bool force_partition = false;
163
164 if (nb_part<nb_rank)
165 ARCANE_THROW(ArgumentException,"partition with nb_part ({0}) < nb_rank ({1})",
166 nb_part,nb_rank);
167
168 if (nb_rank==1){
169 info() << "INFO: Unable to test load balancing on a single sub-domain";
170 return;
171 }
172
173 // TODO : comprendre les modifs de l'IFPEN
174 // utilise toujours repartitionnement complet.
175 // info() << "Metis: params " << m_nb_refine << " " << imbalance() << " " << maxImbalance();
176 // info() << "Metis: params " << (m_nb_refine>=10) << " " << (imbalance()>4.0*maxImbalance()) << " " << (imbalance()>1.0);
177 bool force_full_repartition = false;
178 //force_full_repartition = true;
179 force_full_repartition |= (m_nb_refine < 0); // toujours la première fois, pour compatibilité avec l'ancienne version
180
181 if (nb_part != nb_rank) { // Pas de "repartitionnement" si le nombre de parties ne correspond pas au nombre de processeur.
182 force_full_repartition = true;
183 force_partition = true;
184 }
185
186 info() << "WARNING: compensating the potential lack of 'Metis' options in case of manual instanciation";
187 Integer max_diffusion_count = 10; // reprise des options par défaut du axl
188 Real imbalance_relative_tolerance = 4;
189 float tolerance_target = 1.05f;
190 bool dump_graph = false;
191 MetisCallStrategy call_strategy = MetisCallStrategy::one_processor_per_node;
192 bool in_out_digest = false;
193
194 // Variables d'environnement permettant de regler les options lorsqu'il n'y a pas de jeu de donnees
195
196 String call_strategy_env = platform::getEnvironmentVariable("ARCANE_METIS_CALL_STRATEGY");
197 if (call_strategy_env == "all-processors"){
198 call_strategy = MetisCallStrategy::all_processors;
199 } else if (call_strategy_env == "one-processor-per-node") {
200 call_strategy = MetisCallStrategy::one_processor_per_node;
201 } else if (call_strategy_env == "two-processors-two-nodes") {
202 call_strategy = MetisCallStrategy::two_processors_two_nodes;
203 } else if (call_strategy_env == "two-gathered-processors") {
204 call_strategy = MetisCallStrategy::two_gathered_processors;
205 } else if (call_strategy_env == "two-scattered-processors") {
206 call_strategy = MetisCallStrategy::two_scattered_processors;
207 } else if (!call_strategy_env.null()) {
208 ARCANE_FATAL("Invalid value '{0}' for ARCANE_METIS_CALL_STRATEGY environment variable",call_strategy_env);
209 }
210
211 String in_out_digest_env = platform::getEnvironmentVariable("ARCANE_METIS_INPUT_OUTPUT_DIGEST");
212 if (!in_out_digest_env.null()){
213 in_out_digest = true;
214 }
215
216 if (options()){
217 max_diffusion_count = options()->maxDiffusiveCount();
218 imbalance_relative_tolerance = options()->imbalanceRelativeTolerance();
219 tolerance_target = (float)(options()->toleranceTarget());
220 dump_graph = options()->dumpGraph();
221 call_strategy = options()->metisCallStrategy();
222 in_out_digest = options()->inputOutputDigest();
223 }
224
225 if (max_diffusion_count > 0)
226 force_full_repartition |= (m_nb_refine>=max_diffusion_count);
227 force_full_repartition |= (imbalance()>imbalance_relative_tolerance*maxImbalance());
228 force_full_repartition |= (imbalance()>1.0);
229
230
231 // initialisations pour la gestion des contraintes (sauf initUidRef)
232 initConstraints(false);
233
234 bool is_shared_memory = pm->isThreadImplementation();
235 // En mode mémoire partagé, pour l'instant on force le fait d'utiliser un seul
236 // processeur par noeud car on ne sait pas si on dispose de plusieurs rang par noeud.
237 // Notamment, en mode mémoire partagé sans MPI on n'a obligatoirement qu'un seul noeud
238 if (is_shared_memory)
239 call_strategy = MetisCallStrategy::one_processor_per_node;
240
241 idxtype nb_weight = nbCellWeight();
242
243 info() << "Load balancing with Metis nb_weight=" << nb_weight << " initial=" << initial_partition
244 << " call_strategy=" << (int)call_strategy
245 << " is_shared_memory?=" << is_shared_memory
246 << " disabling_fpe?=" << m_disable_floatingexception;
247 if (nb_weight==0)
248 initial_partition = true;
249
250 UniqueArray<idxtype> metis_vtkdist(nb_rank+1);
251 Integer total_nb_cell = 0;
252
253 // Contient les numéros uniques des entités dans la renumérotation
254 // propre à metis
255 VariableCellInteger cell_metis_uid(VariableBuildInfo(mesh,"CellsMetisUid",IVariable::PNoDump));
256
257 UniqueArray<Integer> global_nb_own_cell(nb_rank);
258 CellGroup own_cells = mesh->ownCells();
259 Integer nb_own_cell = nbOwnCellsWithConstraints(); // on tient compte des contraintes
260 pm->allGather(ConstArrayView<Integer>(1,&nb_own_cell),global_nb_own_cell);
261 Int32 nb_empty_part = 0;
262 {
263 //Integer total_first_uid = 0;
264 metis_vtkdist[0] = 0;
265 for( Integer i=0; i<nb_rank; ++i ){
266 //total_first_uid += global_nb_own_cell[i];
267 Int32 n = global_nb_own_cell[i];
268 if (n==0)
269 ++nb_empty_part;
270 total_nb_cell += n;
271 metis_vtkdist[i+1] = static_cast<idxtype>(total_nb_cell);
272 // info() << "METIS VTKDIST " << (i+1) << ' ' << metis_vtkdist[i+1];
273 }
274 }
275 // N'appelle pas Parmetis si on n'a pas de mailles car sinon cela provoque une erreur.
276 info() << "Total nb_cell=" << total_nb_cell << " nb_empty_partition=" << nb_empty_part;
277 if (total_nb_cell==0){
278 info() << "INFO: mesh '" << mesh->name() << " has no cell. No partitioning is needed";
279 freeConstraints();
280 return;
281 }
282
283 // HP: Skip this shortcut because it produces undefined itemsNewOwner variable.
284/*
285 if (total_nb_cell < nb_rank) { // Dans ce cas, pas la peine d'appeler ParMetis.
286 warning() << "There are no subdomain except cells, no reffinement";
287 freeConstraints();
288 return;
289 }
290*/
291
292 // Nombre max de mailles voisine connectés aux mailles
293 // en supposant les mailles connectées uniquement par les faces
294 // Cette valeur sert à préallouer la mémoire pour la liste des mailles voisines
295 Integer nb_max_face_neighbour_cell = 0;
296 {
297 // Renumérote les mailles pour Metis pour que chaque sous-domaine
298 // ait des mailles de numéro consécutifs
299 Integer mid = static_cast<Integer>(metis_vtkdist[my_rank]);
300 ENUMERATE_CELL(i_item,own_cells){
301 const Cell& item = *i_item;
302 if (cellUsedWithConstraints(item)){
303 cell_metis_uid[item] = mid;
304 ++mid;
305 }
306 nb_max_face_neighbour_cell += item.nbFace();
307 }
308 cell_metis_uid.synchronize();
309 }
310
311 _initUidRef(cell_metis_uid);
312
313 // libération mémoire
314 cell_metis_uid.setUsed(false);
315
316 SharedArray<idxtype> metis_xadj;
317 metis_xadj.reserve(nb_own_cell+1);
318
319 SharedArray<idxtype> metis_adjncy;
320 metis_adjncy.reserve(nb_max_face_neighbour_cell);
321
322
323 // Construction de la connectivité entre les cellules et leurs voisines en tenant compte des contraintes
324 // (La connectivité se fait suivant les faces)
325 Int64UniqueArray neighbour_cells;
326 UniqueArray<float> edge_weights;
327 edge_weights.resize(0);
328 ENUMERATE_CELL(i_item,own_cells){
329 Cell item = *i_item;
330
331 if (!cellUsedWithConstraints(item))
332 continue;
333
334 metis_xadj.add(metis_adjncy.size());
335
336 getNeighbourCellsUidWithConstraints(item, neighbour_cells, &edge_weights);
337 for( Integer z=0; z<neighbour_cells.size(); ++z )
338 metis_adjncy.add(static_cast<idxtype>(neighbour_cells[z]));
339 }
340 metis_xadj.add(metis_adjncy.size());
341
342 idxtype wgtflag = 3; // Sommets et aretes
343//2; // Indique uniquement des poids sur les sommets
344 idxtype numflags = 0;
345 idxtype nparts = static_cast<int>(nb_part);
346
347 // Poids aux sommets du graphe (les mailles)
348// if (initial_partition)
349// nb_weight = 1;
350
351 String s = platform::getEnvironmentVariable("ARCANE_LB_PARAM");
352 if (!s.null() && (s.upper() == "PARTICLES"))
353 nb_weight = 1; // Only one weight in this case !
354
355#if CEDRIC_LB_2
356 nb_weight = 1;
357#endif
358 bool cond = (nb_weight == 1);
359 // Real max_int = 1 << (sizeof(idxtype)*8-1);
360 UniqueArray<realtype> metis_ubvec(CheckedConvert::toInteger(nb_weight));
361 SharedArray<float> cells_weights;
362
363 PartitionConverter<float,idxtype> converter(pm, (Real)IDX_MAX, cond);
364
365
371 cells_weights = cellsWeightsWithConstraints(CheckedConvert::toInteger(nb_weight));
372
373 // Déséquilibre autorisé pour chaque contrainte
374 metis_ubvec.resize(CheckedConvert::toInteger(nb_weight));
375 metis_ubvec.fill(tolerance_target);
376
377 converter.reset(CheckedConvert::toInteger(nb_weight));
378
379 do {
380 cond = converter.isBalancable<realtype>(cells_weights, metis_ubvec, nb_part);
381 if (!cond) {
382 info() << "We have to increase imbalance :";
383 info() << metis_ubvec;
384 for (auto& tol: metis_ubvec ) {
385 tol *= 1.5f;
386 }
387 }
388 } while (!cond);
389
390 ArrayConverter<float,idxtype,PartitionConverter<float,idxtype> > metis_vwgtConvert(cells_weights, converter);
391
392 SharedArray<idxtype> metis_vwgt(metis_vwgtConvert.array().constView());
393
394 const bool do_print_weight = false;
395 if (do_print_weight){
396 std::ofstream dumpy;
397 StringBuilder fname;
398 Integer iteration = mesh->subDomain()->commonVariables().globalIteration();
399 fname = "weigth-";
400 fname += pm->commRank();
401 fname += "-";
402 fname += iteration;
403 String f(fname);
404 dumpy.open(f.localstr());
405 for (int i = 0; i < metis_xadj.size()-1 ; ++i) {
406 dumpy << " Weight uid=" << i;
407 for( int j=0 ; j < nb_weight ; ++j ){
408 dumpy << " w=" << *(metis_vwgt.begin()+i*nb_weight+j)
409 << "(" << cells_weights[i*nb_weight+j] <<")";
410 }
411 dumpy << '\n';
412 }
413 dumpy.close();
414 }
415
416 converter.reset(1); // Only one weight for communications !
417
418 converter.computeContrib(edge_weights);
419 ArrayConverter<float,idxtype,PartitionConverter<float,idxtype> > metis_ewgt_convert(edge_weights, converter);
420 SharedArray<idxtype> metis_ewgt((UniqueArray<idxtype>)metis_ewgt_convert.array());
421
422 SharedArray<float> cells_size;
423 realtype itr=50; // In average lb every 20 iterations ?
424 SharedArray<idxtype> metis_vsize;
425 if (!initial_partition) {
426 cells_size = cellsSizeWithConstraints();
427 converter.computeContrib(cells_size, (Real)itr);
428 metis_vsize.resize(metis_xadj.size()-1,1);
429 }
430
431
432 idxtype metis_options[4];
433 metis_options[0] = 0;
434
435 // By default RandomSeed is fixed
436
437#ifdef CEDRIC_LB_2
438 m_random_seed = Convert::toInteger((Real)MPI_Wtime());
439#endif // CEDRIC_LB_2
440
441 metis_options[0] = 1;
442 metis_options[1] = 0;
443 metis_options[2] = m_random_seed;
444 metis_options[3] = 1; // Initial distribution information is implicit
445 /*
446 * Comment to keep same initial random seed each time !
447 m_random_seed++;
448 */
449
450 idxtype metis_edgecut = 0;
451
452 // TODO: compute correct part number !
453 UniqueArray<idxtype> metis_part;
454
455 std::chrono::high_resolution_clock clock;
456 auto start_time = clock.now();
457
458 GraphDistributor gd(pm);
459
460 // Il y a actuellement 2 mecanismes de regroupement du graph : celui fait par "GraphDistributor" et
461 // celui fait par le wrapper ParMetis. Il est possible de combiner les 2 (two_processors_two_nodes).
462 // A terme, ces 2 mecanismes devraient fusionner et il ne faudrait conserver que le "GraphDistributor".
463 //
464 // La redistribution par le wrapper n'est faite que dans les 2 cas suivants :
465 // - on veut un regroupement sur 2 processeurs apres un regroupement par noeuds (two_processors_two_nodes)
466 // - on veut un regroupement direct sur 2 processeurs, en esperant qu'ils soient sur 2 noeuds distincts (two_scattered_processors)
467
468 bool redistribute = true; // redistribution par GraphDistributor
469 bool redistribute_by_wrapper = false; // redistribution par le wrapper ParMetis
470
471 if (call_strategy == MetisCallStrategy::all_processors || call_strategy == MetisCallStrategy::two_scattered_processors) {
472 redistribute = false;
473 }
474
475 if (call_strategy == MetisCallStrategy::two_processors_two_nodes || call_strategy == MetisCallStrategy::two_scattered_processors) {
476 redistribute_by_wrapper = true;
477 }
478 // Indique si on n'autorise de n'utiliser qu'un seul PE.
479 // historiquement (avant avril 2020) cela n'était pas autorisé car cela revenait
480 // à appeler 'ParMetis' sur un seul PE ce qui n'était pas supporté.
481 // Maintenant, on appelle directement Metis dans ce cas donc cela ne pose pas de
482 // problèmes. Cependant pour des raisons historiques on garde l'ancien comportement
483 // sauf pour deux cas:
484 // - si l'échange de messages est en mode mémoire partagé. Comme le partitionnement
485 // dans ce mode n'était pas supporté avant, il n'y a
486 // pas d'historique à conserver. De plus cela est indispensable si on n'utilise
487 // qu'un seul noeud de calcul car alors
488 // - lors du partionnement initial et s'il y a des partitions vides. Ce cas n'existait
489 // pas avant non plus car on utilisait 'MeshPartitionerTester' pour faire un
490 // premier pré-partitionnement qui garantit aucune partition vide. Cela permet
491 // d'éviter ce pré-partitionnement.
492
493 bool gd_allow_only_one_rank = false;
494 if (is_shared_memory || (nb_empty_part!=0 && initial_partition))
495 gd_allow_only_one_rank = true;
496
497 // S'il n'y a qu'une seule partie non-vide on n'utilise qu'un seul rang. Ce cas
498 // intervient normalement uniquement en cas de partitionnement initial et si
499 // un seul rang (en général le rang 0) a des mailles.
500 if ((nb_empty_part+1)==nb_rank){
501 info() << "Initialize GraphDistributor with max rank=1";
502 gd.initWithMaxRank(1);
503 }
504 else if (call_strategy == MetisCallStrategy::two_gathered_processors && (nb_rank > 2)) {
505 // Seuls les 2 premiers processeurs seront utilisés.
506 info() << "Initialize GraphDistributor with max rank=2";
507 gd.initWithMaxRank(2);
508 }
509 else {
510 info() << "Initialize GraphDistributor with one rank per node"
511 << " (allow_one?=" << gd_allow_only_one_rank << ")";
512 gd.initWithOneRankPerNode(gd_allow_only_one_rank);
513 }
514
515 MPI_Comm metis_mpicomm = MPI_COMM_NULL;
516 info() << "Using GraphDistributor?=" << redistribute;
517 if (redistribute){
518 metis_xadj = gd.convert<idxtype>(metis_xadj, &metis_part, true);
519 metis_vwgt = gd.convert<idxtype>(metis_vwgt);
520 metis_adjncy = gd.convert<idxtype>(metis_adjncy);
521 metis_ewgt = gd.convert<idxtype>(metis_ewgt);
522 if (!initial_partition)
523 metis_vsize = gd.convert<idxtype>(metis_vsize);
524 metis_mpicomm = gd.getCommunicator();
525
526 metis_vtkdist.resize(gd.size()+1);
527 if (gd.contribute()) {
528 UniqueArray<Integer> buffer(gd.size());
529 Integer nbVertices = metis_part.size();
530 gd.parallelManager()->allGather(ConstArrayView<Integer>(1,&nbVertices),buffer);
531 metis_vtkdist[0] = 0;
532 for (int i = 1 ; i < metis_vtkdist.size() ; ++i) {
533 metis_vtkdist[i] = metis_vtkdist[i-1] + buffer[i-1];
534 }
535 }
536 metis_options[3] = PARMETIS_PSR_UNCOUPLED ; // Initial distribution information is in metis_part
537 }
538 else{
539 metis_part = UniqueArray<idxtype>(nb_own_cell, my_rank);
540 Parallel::Communicator comm = this->communicator();
541 metis_mpicomm = (MPI_Comm)comm;
542 }
543
544 bool do_call_metis = true;
545 if (redistribute)
546 do_call_metis = gd.contribute();
547 if (do_call_metis) {
548
549 if (dump_graph) {
550 Integer iteration = sd->commonVariables().globalIteration();
551 StringBuilder name("graph-");
552 name += iteration;
553 _writeGraph(metis_vtkdist, metis_xadj, metis_adjncy, metis_vwgt, metis_ewgt, name.toString(),
554 metis_mpicomm);
555 }
556
557
558 // ParMetis >= 4 requires to define tpwgt.
559 const Integer tpwgt_size = CheckedConvert::toInteger(nb_part) * CheckedConvert::toInteger(nb_weight);
560 UniqueArray<realtype> tpwgt(tpwgt_size);
561 float fill_value = (float)(1.0/(double)nparts);
562 tpwgt.fill(fill_value);
563
564 int retval = METIS_OK;
565 Timer::Action ts(sd,"Metis");
566
567 MetisWrapper wrapper;
568 force_partition |= initial_partition;
569 force_full_repartition |= force_partition;
570 if (force_full_repartition){
571 m_nb_refine = 0;
572 if (force_partition) {
573 info() << "Metis: use a complete partitioning.";
574 retval = wrapper.callPartKway(traceMng(),
575 in_out_digest,
576 redistribute_by_wrapper,
577 metis_vtkdist.data(),
578 metis_xadj.data(),
579 metis_adjncy.data(),
580 metis_vwgt.data(),
581 metis_ewgt.data(),
582 &wgtflag,&numflags,&nb_weight,
583 &nparts, tpwgt.data(),
584 metis_ubvec.data(),
585 metis_options,&metis_edgecut,
586 metis_part.data(),
587 &metis_mpicomm);
588 }
589 else {
590 info() << "Metis: use a complete REpartitioning.";
591 retval = wrapper.callAdaptiveRepart(traceMng(),
592 in_out_digest,
593 redistribute_by_wrapper,
594 metis_vtkdist.data(),
595 metis_xadj.data(),
596 metis_adjncy.data(),
597 metis_vwgt.data(),
598 metis_vsize.data(), // Vsize !
599 metis_ewgt.data(),
600 &wgtflag,&numflags,&nb_weight,
601 &nparts,tpwgt.data(),
602 metis_ubvec.data(),
603 &itr, metis_options,&metis_edgecut,
604 metis_part.data(),
605 &metis_mpicomm);
606 }
607 }
608 else{
609 // TODO: mettre dans MetisWrapper et supprimer utilisation de .data()
610 // (cela doit être faire par le wrapper)
611 ++m_nb_refine;
612 info() << "Metis: use a diffusive REpartitioning";
613 retval = ParMETIS_V3_RefineKway(metis_vtkdist.data(),
614 metis_xadj.data(),
615 metis_adjncy.data(),
616 metis_vwgt.data(),
617 metis_ewgt.data(),
618 &wgtflag,&numflags,&nb_weight,
619 &nparts,tpwgt.data(),
620 metis_ubvec.data(),
621 metis_options,&metis_edgecut,
622 metis_part.data(),
623 &metis_mpicomm);
624 }
625 if (retval != METIS_OK) {
626 ARCANE_FATAL("Error while computing ParMetis partitioning r='{0}'",retval);
627 }
628 }
629 if (redistribute){
630 metis_part = gd.convertBack<idxtype>(metis_part, nb_own_cell);
631 }
632
633 auto end_time = clock.now();
634 std::chrono::microseconds duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
635 info() << "Time to partition using parmetis = " << duration.count() << " us ";
636
637 // Stratégie à adopter pour supprimer les partitions vides.
638 // A noter qu'il faut faire cela avant d'appliquer les éventuelles contraintes
639 // pour garantir que ces dernières seront bien respectées
640 if (options()){
641 switch(options()->emptyPartitionStrategy()){
642 case MetisEmptyPartitionStrategy::DoNothing:
643 break;
644 case MetisEmptyPartitionStrategy::TakeFromBiggestPartitionV1:
645 _removeEmptyPartsV1(nb_part,nb_own_cell,metis_part);
646 break;
647 case MetisEmptyPartitionStrategy::TakeFromBiggestPartitionV2:
648 _removeEmptyPartsV2(nb_part,metis_part);
649 break;
650 }
651 }
652 else
653 _removeEmptyPartsV2(nb_part,metis_part);
654
655 VariableItemInt32& cells_new_owner = mesh->toPrimaryMesh()->itemsNewOwner(IK_Cell);
656 {
657 Integer index = 0;
658 ENUMERATE_CELL(i_item,own_cells){
659 Cell item = *i_item;
660 if (!cellUsedWithConstraints(item))
661 continue;
662
663 Int32 new_owner = CheckedConvert::toInt32(metis_part[index]);
664 ++index;
665 changeCellOwner(item, cells_new_owner, new_owner);
666 }
667 }
668
669 // libération des tableaux temporaires
670 freeConstraints();
671
672 cells_new_owner.synchronize();
673
674 changeOwnersFromCells();
675
676 m_nb_refine = m_parallel_mng->reduce(Parallel::ReduceMax, m_nb_refine);
677}

691void MetisMeshPartitioner::
692_removeEmptyPartsV1(const Int32 nb_part,const Int32 nb_own_cell,ArrayView<idxtype> metis_part)
693{
694 IParallelMng* pm = m_parallel_mng;
695 Int32 my_rank = pm->commRank();
696
697 //The following code insures that no empty part will be created.
698 // TODO: faire une meilleure solution, qui prend des elements sur la partie la plus lourde.
699 UniqueArray<Int32> elem_by_part(nb_part,0);
700 UniqueArray<Int32> min_part(nb_part);
701 UniqueArray<Int32> max_part(nb_part);
702 UniqueArray<Int32> sum_part(nb_part);
703 UniqueArray<Int32> min_roc(nb_part);
704 UniqueArray<Int32> max_proc(nb_part);
705 for (int i =0 ; i < nb_own_cell ; i++) {
706 elem_by_part[CheckedConvert::toInteger(metis_part[i])]++;
707 }
708 pm->computeMinMaxSum(elem_by_part, min_part, max_part, sum_part, min_roc, max_proc);
709
710 int nb_hole=0;
711 Int32 max_part_id = -1;
712 Int32 max_part_nbr = -1;
713 // Compute number of empty parts
714 for(int i = 0; i < nb_part ; i++) {
715 if (sum_part[i] == 0) {
716 nb_hole ++;
717 }
718 if(max_part_nbr < sum_part[i]) {
719 max_part_nbr = sum_part[i];
720 max_part_id = i;
721 }
722 }
723 info() << "Parmetis: number empty parts " << nb_hole;
724
725 // Le processeur ayant la plus grosse partie de la plus
726 // grosse partition comble les trous
727 if(my_rank == max_proc[max_part_id]) {
728 int offset = 0;
729 for(int i = 0; i < nb_part ; i++) {
730 // On ne comble que s'il reste des mailles
731 if (sum_part[i] == 0 && offset < nb_own_cell) {
732 while(offset < nb_own_cell && metis_part[offset] != max_part_id) {
733 offset++;
734 }
735 // Si on est sorti du while car pas assez de mailles
736 if(offset == nb_own_cell)
737 break;
738 // Le trou est comblé par ajout d'une seule maille
739 metis_part[offset] = i;
740 offset++;
741 }
742 }
743 }
744}

755Int32 MetisMeshPartitioner::
756_removeEmptyPartsV2Helper(const Int32 nb_part,ArrayView<idxtype> metis_part,Int32 algo_iteration)
757{
758 IParallelMng* pm = m_parallel_mng;
759 Int32 my_rank = pm->commRank();
760 const Int32 nb_own_cell = metis_part.size();
761 // The following code insures that no empty part will be created.
762 UniqueArray<idxtype> elem_by_part(nb_part,0);
763 UniqueArray<idxtype> min_part(nb_part);
764 UniqueArray<idxtype> max_part(nb_part);
765 UniqueArray<idxtype> sum_part(nb_part);
766 UniqueArray<Int32> min_rank(nb_part);
767 UniqueArray<Int32> max_rank(nb_part);
768 for (int i =0 ; i < nb_own_cell ; i++) {
769 elem_by_part[metis_part[i]]++;
770 }
771 pm->computeMinMaxSum(elem_by_part, min_part, max_part, sum_part, min_rank, max_rank);
772
773 // Rang des parties vides
774 UniqueArray<Int32> empty_part_ranks;
775 int nb_hole = 0;
776 Int32 max_part_id = -1;
777 Int64 max_part_nbr = -1;
778 // Compute number of empty parts
779 Int64 total_nb_cell = 0;
780 for(int i = 0; i < nb_part ; i++) {
781 Int64 current_sum_part = sum_part[i];
782 total_nb_cell += current_sum_part;
783 if (current_sum_part == 0) {
784 empty_part_ranks.add(i);
785 nb_hole ++;
786 }
787 if (max_part_nbr < current_sum_part) {
788 max_part_nbr = current_sum_part;
789 max_part_id = i;
790 }
791 }
792 if (max_part_id<0)
793 ARCANE_FATAL("Bad value max_part_id ({0})",max_part_id);
794 info() << "Parmetis: check empty parts: (" << algo_iteration << ") nb_empty_parts=" << nb_hole
795 << " nb_max_part=" << max_part_nbr
796 << " max_part_rank=" << max_part_id
797 << " max_proc_max_part_id=" << max_rank[max_part_id]
798 << " empty_part_ranks=" << empty_part_ranks
799 << " total_nb_cell=" << total_nb_cell;
800
801 if (nb_hole==0)
802 return 0;
803
804 // Le processeur ayant la plus grosse partie de la plus
805 // grosse partition comble les trous
806 if (my_rank == max_rank[max_part_id]) {
807 // On garantit qu'on n'enlèvera pas toutes nos mailles
808 Int32 max_remove_cell = nb_own_cell - 1;
809 int offset = 0;
810 for(int i = 0; i < nb_part ; i++) {
811 // On ne comble que s'il reste des mailles
812 if (sum_part[i] == 0 && offset < nb_own_cell) {
813 while (offset < max_remove_cell && metis_part[offset] != max_part_id) {
814 offset++;
815 }
816 // Si on est sorti du while car pas assez de mailles
817 if (offset == max_remove_cell)
818 break;
819 // Le trou est comblé par ajout d'une seule maille
820 metis_part[offset] = i;
821 offset++;
822 }
823 }
824 }
825
826 return nb_hole;
827}

837void MetisMeshPartitioner::
838_removeEmptyPartsV2(const Int32 nb_part,ArrayView<idxtype> metis_part)
839{
840 bool do_continue = true;
841 Int32 last_nb_hole = 0;
842 while (do_continue){
843 Int32 nb_hole = _removeEmptyPartsV2Helper(nb_part,metis_part,0);
844 info() << "Parmetis: nb_empty_partition=" << nb_hole << " last_nb_partition=" << last_nb_hole;
845 if (nb_hole==0)
846 break;
847 // Garanti qu'on sort de la boucle si on a toujours le même nombre de trous
848 // qu'avant. Cela permet d'éviter les boucles infinies.
849 // Cela signifie aussi qu'on ne peut pas combler les trous et donc il y
850 // aura surement des partitions vides
851 if (last_nb_hole>0 && last_nb_hole<=nb_hole){
852 pwarning() << "Can not remove all empty partitions. This is probably because you try"
853 << " to cut in too many partitions";
854 break;
855 }
856 last_nb_hole = nb_hole;
857 }
858}
