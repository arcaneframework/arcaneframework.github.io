257// Calcul de la consommation mémoire pour chaque type d'entité.
258// Les mailles bénéficient ensuite des contributions des autres entités adjacentes.
259void CriteriaMng::
260computeMemory()
261{
262 Int32 length = m_family_names.size();
263 m_overall_memory.resize(length);
264 m_resident_memory.resize(length);
265 m_buffer.id = -1;
266 m_overall_memory.fill(0);
267 m_resident_memory.fill(0);
268
269 // For each variable, compute the size for one object.
270 for (VariableCollectionEnumerator vc(m_variable_mng->usedVariables()); ++vc;) {
271 const IVariable* var = *vc;
272 Integer memory = 0;
273 try {
274 if (var->dataType() != DT_String)
275 memory = dataTypeSize(var->dataType());
276 }
277 catch (const ArgumentException&) {
278 memory = 0; // Cannot know memory used for that ...
279 continue;
280 }
281 Int32 family_index = -1;
282 Integer kind = var->itemKind();
283 if (kind == IK_Particle) { // Not the same counter for all items
284 family_index = _findEntity(var->itemFamilyName());
285 if (family_index >= 0) {
286 m_overall_memory[family_index] += memory;
287 }
288 }
289 m_overall_memory[kind] += memory;
290
291 int properties = var->property();
292 if ((properties & IVariable::PNoExchange) ||
293 (properties & IVariable::PNoNeedSync) ||
294 (properties & IVariable::PTemporary) ||
295 (properties & IVariable::PSubDomainPrivate)) {
296 continue;
297 }
298 m_resident_memory[kind] += memory;
299 if (family_index >= 0) {
300 m_resident_memory[family_index] += memory;
301 }
302 }
303}

346// Pour les mailles, on calcule la contribution mémoire des noeuds, aretes et faces.
347void CriteriaMng::
348_computeMemCell(Cell cell)
349{
350 Real contrib;
351 if (cell.localId() == m_buffer.id) // already computed
352 return;
353 m_buffer.id = cell.localId();
354 m_buffer.overall_memory = m_overall_memory[IK_Cell];
355 m_buffer.resident_memory = m_resident_memory[IK_Cell];
356
357 contrib = _computeMemContrib<Node>(cell.nodes());
358 m_buffer.overall_memory += contrib * m_overall_memory[IK_Node];
359 m_buffer.resident_memory += contrib * m_resident_memory[IK_Node];
360
361 contrib = _computeMemContrib<Face>(cell.faces());
362 m_buffer.overall_memory += contrib * m_overall_memory[IK_Face];
363 m_buffer.resident_memory += contrib * m_resident_memory[IK_Face];
364
365 contrib = _computeMemContrib<Edge>(cell.edges());
366 m_buffer.overall_memory += contrib * m_overall_memory[IK_Edge];
367 m_buffer.resident_memory += contrib * m_resident_memory[IK_Edge];
368}

400
401void LoadBalanceMng::
402initAccess(IMesh* mesh)
403{
404 if (!mesh)
405 ARCANE_FATAL("Null mesh");
406 m_mesh_handle = mesh->handle();
407 int vflags = IVariable::PExecutionDepend | IVariable::PNoDump | IVariable::PTemporary;
408 m_cell_new_owner = std::make_unique<VariableCellInt32>(VariableBuildInfo(m_mesh_handle, "CellFamilyNewOwner", IVariable::PExecutionDepend | IVariable::PNoDump));
409 m_comm_costs = new VariableFaceReal(VariableBuildInfo(m_mesh_handle, "LbMngCommCost", vflags));
410 m_mass_over_weigth = new VariableCellReal(VariableBuildInfo(m_mesh_handle, "LbMngOverallMass", vflags));
411 m_mass_res_weight = new VariableCellReal(VariableBuildInfo(m_mesh_handle, "LbMngResidentMass", vflags));
412 m_event_weights = new VariableCellArrayReal(VariableBuildInfo(m_mesh_handle, "LbMngMCriteriaWgt", vflags));
413 m_comm_costs->fill(1);
414 m_mass_over_weigth->fill(1);
415 m_mass_res_weight->fill(1);
416
417 mesh->traceMng()->info() << "LoadBalanceMng::initAccess(): use_memory=" << m_mass_criterion;
418
419 if (m_compute_comm || m_mass_criterion) { // Memory useful only for communication cost or mass lb criterion
420 m_criteria->computeMemory();
421 _computeResidentMass();
422 }
423 if (m_compute_comm)
424 _computeComm();
425 if (m_mass_criterion)
426 _computeOverallMass();
427 _computeEvents();
428}

531/*---------------------------------------------------------------------------*/
532/*---------------------------------------------------------------------------*/
533
534void LoadBalanceMng::
535_computeComm()
536{
537 IMesh* mesh = m_mesh_handle.mesh();
538 Integer penalty = 2; // How many times we do synchronization ?
539
540 if (m_comm_vars.size())
541 m_comm_costs->fill(0);
542
543 for (int i = 0; i < m_comm_vars.size(); ++i) {
544 StoreIProxyItemVariable& commvar = m_comm_vars[i];
545 ENUMERATE_FACE (iface, mesh->ownFaces()) {
546 (*m_comm_costs)[iface] += commvar[iface] * m_criteria->getResidentMemory(commvar.getPos());
547 }
548 }
549 if (cellCommContrib()) {
550 ENUMERATE_CELL (icell, mesh->ownCells()) {
551 Real mem = (*m_mass_res_weight)[icell];
552 for (Face face : icell->faces()) {
553 (*m_comm_costs)[face] += mem * penalty;
554 }
555 }
556 }
557
558 IVariable* ivar;
559 // Make sure that ghosts contribution is used
560 ivar = m_comm_costs->variable();
561 ivar->itemFamily()->reduceFromGhostItems(ivar, Parallel::ReduceSum);
562 m_comm_costs->synchronize();
563}

565/*---------------------------------------------------------------------------*/
566/*---------------------------------------------------------------------------*/
567
568void LoadBalanceMng::
569_computeEvents()
570{
571 m_event_weights->resize(nbCriteria());
572
573 ArrayView<StoreIProxyItemVariable> eventVars = m_event_vars;
574 if (m_mass_criterion) {
575 StoreIProxyItemVariable cvar(m_mass_over_weigth->variable());
576 m_event_vars[0] = cvar;
577 }
578 else {
579 eventVars = m_event_vars.subView(1, m_event_vars.size());
580 }
581
582 for (Integer i = 0; i < eventVars.size(); ++i) {
583 ENUMERATE_CELL (icell, m_mesh_handle.mesh()->ownCells()) {
584 Integer count = i;
585 if (m_nb_criterion) {
586 count += 1;
587 (*m_event_weights)[icell][0] = 1;
588 }
589 (*m_event_weights)[icell][count] = eventVars[i][icell];
590 }
591 }
592}
