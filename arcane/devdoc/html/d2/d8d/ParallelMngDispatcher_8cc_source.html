99void ParallelMngDispatcherBuildInfo::

101{

131void ParallelMngDispatcher::DefaultControlDispatcher::
132waitSomeRequests(ArrayView<Request> requests,ArrayView<bool> indexes,
133 bool is_non_blocking)
134{
135 UniqueArray<Integer> done_requests;
136 if (is_non_blocking)
137 done_requests = m_parallel_mng->testSomeRequests(requests);
138 else

140 indexes.fill(false);

220class ParallelMngDispatcher::Impl
221: public IParallelMngInternal
222{
223 public:
224
225 explicit Impl(ParallelMngDispatcher* pm)
226 : m_parallel_mng(pm)
227 {
228 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_DISABLE_ACCELERATOR_AWARE_MESSAGE_PASSING", true))
229 m_is_accelerator_aware_disabled = (v.value()!=0);
230 }
231 ~Impl()
232 {
233 m_queue.reset();
234 }
235 public:
236
237 Runner* defaultRunner() const override { return m_runner; }
238 RunQueue* defaultQueue() const override { return m_queue.get(); }
239 bool isAcceleratorAware() const override
240 {
241 if (m_is_accelerator_aware_disabled)
242 return false;
243 if (!m_queue.get())
244 return false;
245 return m_parallel_mng->_isAcceleratorAware();
246 }
247 void setDefaultRunner(Runner* runner) override
248 {
249 m_runner = runner;
250 // Conserve une référence sur le Runner pour éviter sa destruction
251 m_runner_ref = (runner) ? *runner : Runner();
252 if (m_runner){
253 Accelerator::RunQueueBuildInfo build_info(-5);
254 m_queue = makeQueueRef(*m_runner,build_info);
255 m_queue->setAsync(true);
256 }
257 else
258 m_queue.reset();
259 }
260
261 private:
262
263 ParallelMngDispatcher* m_parallel_mng = nullptr;
264 Runner* m_runner = nullptr;
265 Ref<RunQueue> m_queue;
266 Runner m_runner_ref;
267 bool m_is_accelerator_aware_disabled = false;
268};

435void ParallelMngDispatcher::
436broadcastString(String& str,Int32 rank)
437{
438 Int64 len_info[1];
439 Int32 my_rank = commRank();
440 Span<const Byte> bytes = str.bytes();
441 if (rank==my_rank){
442 len_info[0] = bytes.size();
443 broadcast(Int64ArrayView(1,len_info),rank);
444 ByteUniqueArray utf8_array(bytes);
445 broadcast(utf8_array,rank);
446 }
447 else{
448 broadcast(Int64ArrayView(1,len_info),rank);
449 ByteUniqueArray utf8_array(len_info[0]);
450 broadcast(utf8_array,rank);
451 str = String::fromUtf8(utf8_array);
452 }
453}

457
458void ParallelMngDispatcher::
459broadcastMemoryBuffer(ByteArray& bytes,Int32 rank)
460{
461 Int64 size = 0;
462 if (commRank()==rank){
463 size = bytes.largeSize();
464 }
465 {
466 Int64ArrayView bs(1,&size);
467 broadcast(bs,rank);
468 }
469 if (commRank()!=rank)
470 bytes.resize(size);
471 if (size!=0)
472 broadcast(bytes,rank);
473}

487
488void ParallelMngDispatcher::
489processMessages(ConstArrayView<ISerializeMessage*> messages)
490{
491 TimeMetricSentry tphase(Timer::phaseAction(timeStats(),TP_Communication));
492 ScopedPtrT<ISerializeMessageList> message_list(createSerializeMessageList());
493
494 Integer nb_message = messages.size();
495 for( Integer i=0; i<nb_message; ++i ){
496 ISerializeMessage* m = messages[i];
497 message_list->addMessage(m);
498 }
499 message_list->processPendingMessages();
500
501 message_list->waitMessages(Parallel::WaitAll);
502}

543UniqueArray<Integer> ParallelMngDispatcher::
544_doWaitRequests(ArrayView<Request> requests,eWaitType wait_type)
545{
546 Timer::Phase tphase(timeStats(),TP_Communication);
547 Ref<IRequestList> request_list(createRequestListRef());
548 Integer nb_request = requests.size();
549 request_list->add(requests);
550 request_list->wait(wait_type);
551 // Ne pas oublier de recopier les requêtes car elles ont pu être modifiées
552 for (Integer i=0; i<nb_request; ++i )
553 requests[i] = request_list->request(i);
554 return request_list->doneRequestIndexes();
555}
