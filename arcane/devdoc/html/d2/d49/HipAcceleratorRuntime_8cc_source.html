49class HipRunQueueStream
50: public impl::IRunQueueStream
51{
52 public:
53
54 HipRunQueueStream(impl::IRunnerRuntime* runtime, const RunQueueBuildInfo& bi)
55 : m_runtime(runtime)
56 {
57 if (bi.isDefault())
58 ARCANE_CHECK_HIP(hipStreamCreate(&m_hip_stream));
59 else {
60 int priority = bi.priority();
61 ARCANE_CHECK_HIP(hipStreamCreateWithPriority(&m_hip_stream, hipStreamDefault, priority));
62 }
63 }
64 ~HipRunQueueStream() override
65 {
66 ARCANE_CHECK_HIP_NOTHROW(hipStreamDestroy(m_hip_stream));
67 }
68
69 public:
70
71 void notifyBeginLaunchKernel([[maybe_unused]] impl::RunCommandImpl& c) override
72 {
73#ifdef ARCANE_HAS_ROCTX
74 auto kname = c.kernelName();
75 if (kname.empty())
76 roctxRangePush(c.traceInfo().name());
77 else
78 roctxRangePush(kname.localstr());
79#endif
80 return m_runtime->notifyBeginLaunchKernel();
81 }
82 void notifyEndLaunchKernel(impl::RunCommandImpl&) override
83 {
84#ifdef ARCANE_HAS_ROCTX
85 roctxRangePop();
86#endif
87 return m_runtime->notifyEndLaunchKernel();
88 }
89 void barrier() override
90 {
91 ARCANE_CHECK_HIP(hipStreamSynchronize(m_hip_stream));
92 }
93 bool _barrierNoException() override
94 {
95 return hipStreamSynchronize(m_hip_stream) != HIP_SUCCESS;
96 }
97 void copyMemory(const MemoryCopyArgs& args) override
98 {
99 auto r = hipMemcpyAsync(args.destination().data(), args.source().data(),
100 args.source().bytes().size(), hipMemcpyDefault, m_hip_stream);
101 ARCANE_CHECK_HIP(r);
102 if (!args.isAsync())
103 barrier();
104 }
105 void prefetchMemory(const MemoryPrefetchArgs& args) override
106 {
107 auto src = args.source().bytes();
108 if (src.size()==0)
109 return;
110 DeviceId d = args.deviceId();
111 int device = hipCpuDeviceId;
112 if (!d.isHost())
113 device = d.asInt32();
114 auto r = hipMemPrefetchAsync(src.data(), src.size(), device, m_hip_stream);
115 ARCANE_CHECK_HIP(r);
116 if (!args.isAsync())
117 barrier();
118 }
119 void* _internalImpl() override
120 {
121 return &m_hip_stream;
122 }
123
124 public:
125
126 hipStream_t trueStream() const
127 {
128 return m_hip_stream;
129 }
130
131 private:
132
133 impl::IRunnerRuntime* m_runtime;
134 hipStream_t m_hip_stream;
135};

140class HipRunQueueEvent
141: public impl::IRunQueueEventImpl
142{
143 public:
144
145 explicit HipRunQueueEvent(bool has_timer)
146 {
147 if (has_timer)
148 ARCANE_CHECK_HIP(hipEventCreate(&m_hip_event));
149 else
150 ARCANE_CHECK_HIP(hipEventCreateWithFlags(&m_hip_event, hipEventDisableTiming));
151 }
152 ~HipRunQueueEvent() override
153 {
154 ARCANE_CHECK_HIP_NOTHROW(hipEventDestroy(m_hip_event));
155 }
156
157 public:
158
159 // Enregistre l'événement au sein d'une RunQueue
160 void recordQueue(impl::IRunQueueStream* stream) final
161 {
162 auto* rq = static_cast<HipRunQueueStream*>(stream);
163 ARCANE_CHECK_HIP(hipEventRecord(m_hip_event, rq->trueStream()));
164 }
165
166 void wait() final
167 {
168 ARCANE_CHECK_HIP(hipEventSynchronize(m_hip_event));
169 }
170
171 void waitForEvent(impl::IRunQueueStream* stream) final
172 {
173 auto* rq = static_cast<HipRunQueueStream*>(stream);
174 ARCANE_CHECK_HIP(hipStreamWaitEvent(rq->trueStream(), m_hip_event, 0));
175 }
176
177 Int64 elapsedTime(IRunQueueEventImpl* from_event) final
178 {
179 auto* true_from_event = static_cast<HipRunQueueEvent*>(from_event);
180 ARCANE_CHECK_POINTER(true_from_event);
181 float time_in_ms = 0.0;
182 ARCANE_CHECK_HIP(hipEventElapsedTime(&time_in_ms, true_from_event->m_hip_event, m_hip_event));
183 double x = time_in_ms * 1.0e6;
184 Int64 nano_time = static_cast<Int64>(x);
185 return nano_time;
186 }
187
188 private:
189
190 hipEvent_t m_hip_event;
191};

237 void setMemoryAdvice(ConstMemoryView buffer, eMemoryAdvice advice, DeviceId device_id) override
238 {
239 auto v = buffer.bytes();
240 const void* ptr = v.data();
241 size_t count = v.size();
242 int device = device_id.asInt32();
243 hipMemoryAdvise hip_advise;
244
245 if (advice == eMemoryAdvice::MostlyRead)
246 hip_advise = hipMemAdviseSetReadMostly;
247 else if (advice == eMemoryAdvice::PreferredLocationDevice)
248 hip_advise = hipMemAdviseSetPreferredLocation;
249 else if (advice == eMemoryAdvice::AccessedByDevice)
250 hip_advise = hipMemAdviseSetAccessedBy;
251 else if (advice == eMemoryAdvice::PreferredLocationHost) {
252 hip_advise = hipMemAdviseSetPreferredLocation;
253 device = hipCpuDeviceId;
254 }
255 else if (advice == eMemoryAdvice::AccessedByHost) {
256 hip_advise = hipMemAdviseSetAccessedBy;
257 device = hipCpuDeviceId;
258 }
259 else
260 return;
261 //std::cout << "MEMADVISE p=" << ptr << " size=" << count << " advise = " << hip_advise << " id = " << device << "\n";
262 ARCANE_CHECK_HIP(hipMemAdvise(ptr, count, hip_advise, device));
263 }
264 void unsetMemoryAdvice(ConstMemoryView buffer, eMemoryAdvice advice, DeviceId device_id) override
265 {
266 auto v = buffer.bytes();
267 const void* ptr = v.data();
268 size_t count = v.size();
269 int device = device_id.asInt32();
270 hipMemoryAdvise hip_advise;
271
272 if (advice == eMemoryAdvice::MostlyRead)
273 hip_advise = hipMemAdviseUnsetReadMostly;
274 else if (advice == eMemoryAdvice::PreferredLocationDevice)
275 hip_advise = hipMemAdviseUnsetPreferredLocation;
276 else if (advice == eMemoryAdvice::AccessedByDevice)
277 hip_advise = hipMemAdviseUnsetAccessedBy;
278 else if (advice == eMemoryAdvice::PreferredLocationHost) {
279 hip_advise = hipMemAdviseUnsetPreferredLocation;
280 device = hipCpuDeviceId;
281 }
282 else if (advice == eMemoryAdvice::AccessedByHost) {
283 hip_advise = hipMemAdviseUnsetAccessedBy;
284 device = hipCpuDeviceId;
285 }
286 else
287 return;
288 ARCANE_CHECK_HIP(hipMemAdvise(ptr, count, hip_advise, device));
289 }

300 void getPointerAttribute(PointerAttribute& attribute, const void* ptr) override
301 {
302 hipPointerAttribute_t pa;
303 hipError_t ret_value = hipPointerGetAttributes(&pa, ptr);
304 auto mem_type = ePointerMemoryType::Unregistered;
305 // Si \a ptr n'a pas été alloué dynamiquement (i.e: il est sur la pile),
306 // hipPointerGetAttribute() retourne une erreur. Dans ce cas on considère
307 // la mémoire comme non enregistrée.
308 if (ret_value==hipSuccess){
309#if HIP_VERSION_MAJOR >= 6
310 auto rocm_memory_type = pa.type;
311#else
312 auto rocm_memory_type = pa.memoryType;
313#endif
314 if (pa.isManaged)
315 mem_type = ePointerMemoryType::Managed;
316 else if (rocm_memory_type == hipMemoryTypeHost)
317 mem_type = ePointerMemoryType::Host;
318 else if (rocm_memory_type == hipMemoryTypeDevice)
319 mem_type = ePointerMemoryType::Device;
320 }
321
322 //std::cout << "HIP Info: hip_memory_type=" << (int)pa.memoryType << " is_managed?=" << pa.isManaged
323 // << " flags=" << pa.allocationFlags
324 // << " my_memory_type=" << (int)mem_type
325 // << "\n";
326 _fillPointerAttribute(attribute, mem_type, pa.device,
327 ptr, pa.devicePointer, pa.hostPointer);
328 }

342void HipRunnerRuntime::
343fillDevices()
344{
345 int nb_device = 0;
346 ARCANE_CHECK_HIP(hipGetDeviceCount(&nb_device));
347 std::ostream& omain = std::cout;
348 omain << "ArcaneHIP: Initialize Arcane HIP runtime nb_available_device=" << nb_device << "\n";
349 for (int i = 0; i < nb_device; ++i) {
350 OStringStream ostr;
351 std::ostream& o = ostr.stream();
352
353 hipDeviceProp_t dp;
354 ARCANE_CHECK_HIP(hipGetDeviceProperties(&dp, i));
355
356 int has_managed_memory = 0;
357 ARCANE_CHECK_HIP(hipDeviceGetAttribute(&has_managed_memory, hipDeviceAttributeManagedMemory, i));
358
359 o << "\nDevice " << i << " name=" << dp.name << "\n";
360 o << " computeCapability = " << dp.major << "." << dp.minor << "\n";
361 o << " totalGlobalMem = " << dp.totalGlobalMem << "\n";
362 o << " sharedMemPerBlock = " << dp.sharedMemPerBlock << "\n";
363 o << " regsPerBlock = " << dp.regsPerBlock << "\n";
364 o << " warpSize = " << dp.warpSize << "\n";
365 o << " memPitch = " << dp.memPitch << "\n";
366 o << " maxThreadsPerBlock = " << dp.maxThreadsPerBlock << "\n";
367 o << " totalConstMem = " << dp.totalConstMem << "\n";
368 o << " clockRate = " << dp.clockRate << "\n";
369 //o << " deviceOverlap = " << dp.deviceOverlap<< "\n";
370 o << " multiProcessorCount = " << dp.multiProcessorCount << "\n";
371 o << " kernelExecTimeoutEnabled = " << dp.kernelExecTimeoutEnabled << "\n";
372 o << " integrated = " << dp.integrated << "\n";
373 o << " canMapHostMemory = " << dp.canMapHostMemory << "\n";
374 o << " computeMode = " << dp.computeMode << "\n";
375 o << " maxThreadsDim = " << dp.maxThreadsDim[0] << " " << dp.maxThreadsDim[1]
376 << " " << dp.maxThreadsDim[2] << "\n";
377 o << " maxGridSize = " << dp.maxGridSize[0] << " " << dp.maxGridSize[1]
378 << " " << dp.maxGridSize[2] << "\n";
379 o << " hasManagedMemory = " << has_managed_memory << "\n";
380
381 String description(ostr.str());
382 omain << description;
383
384 DeviceInfo device_info;
385 device_info.setDescription(description);
386 device_info.setDeviceId(DeviceId(i));
387 device_info.setName(dp.name);
388 m_device_info_list.addDevice(device_info);
389 }

395class HipMemoryCopier
396: public IMemoryCopier
397{
398 void copy(ConstMemoryView from, [[maybe_unused]] eMemoryRessource from_mem,
399 MutableMemoryView to, [[maybe_unused]] eMemoryRessource to_mem,
400 const RunQueue* queue) override
401 {
402 if (queue){
403 queue->copyMemory(MemoryCopyArgs(to.bytes(),from.bytes()).addAsync(queue->isAsync()));
404 return;
405 }
406 // 'hipMemcpyDefault' sait automatiquement ce qu'il faut faire en tenant
407 // uniquement compte de la valeur des pointeurs. Il faudrait voir si
408 // utiliser \a from_mem et \a to_mem peut améliorer les performances.
409 ARCANE_CHECK_HIP(hipMemcpy(to.data(), from.data(), from.bytes().size(), hipMemcpyDefault));
410 }

441 mrm->setAllocator(eMemoryRessource::Device, getHipDeviceMemoryAllocator());
442 mrm->setCopier(&global_hip_memory_copier);
443 global_hip_runtime.fillDevices();
444}
