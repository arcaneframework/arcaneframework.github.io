39 template <int PaddingSize, typename SizeType> ARCCORE_HOST_DEVICE inline static SizeType
40 getSizeWithSpecificPadding(SizeType size)
41 {
42 if (size <= 0)
43 return 0;
44 SizeType modulo = size % PaddingSize;
45 if (modulo == 0)
46 return size;
47 // TODO: vérifier débordement.
48 SizeType padding_size = ((size / PaddingSize) + 1) * PaddingSize;
49 return padding_size;
50 }

57 template <typename SizeType> ARCCORE_HOST_DEVICE inline static SizeType

63 template <typename DataType>

72 template <typename DataType> ARCCORE_HOST_DEVICE static void applySimdPaddingView(Span<DataType> ids)
73 {
74 using SizeType = Int64;
75 const SizeType size = ids.size();
76 SizeType padding_size = getSizeWithPadding(size);
77 if (padding_size <= size)
78 return;
79
80 // Construit une vue avec la taille du padding
81 // pour éviter les débordement de tableau lors des tests
82 Span<DataType> padded_ids(ids.data(), padding_size);
83
84 DataType last_value = ids[size - 1];
85 for (SizeType k = size; k < padding_size; ++k)
86 padded_ids[k] = last_value;
87 }
88
89 template <typename DataType>
90 static void applySimdPadding(Array<DataType>& ids)
91 {
92 const Int64 size = ids.largeSize();
93 Int64 padding_size = getSizeWithPadding(size);
94 if (padding_size <= size)
95 return;
96 MemoryAllocationArgs args;
97 if (ids.allocator()->guaranteedAlignment(args) < AlignedMemoryAllocator::simdAlignment())
98 ARCCORE_FATAL("Allocator guaranted alignment ({0}) has to be greated than {1}",
99 ids.allocator()->guaranteedAlignment(args), AlignedMemoryAllocator::simdAlignment());
100 if (padding_size > ids.capacity())
101 ARCCORE_FATAL("Not enough capacity c={0} min_expected={1}", ids.capacity(),
102 padding_size);
103 applySimdPaddingView(ids.span());
104 }
105
106 template <typename DataType>
107 static void checkSimdPadding(Span<const DataType> ids)
108 {
109 using SizeType = Int64;
110 const Int64 size = ids.size();
111 SizeType padding_size = getSizeWithPadding(size);
112 if (padding_size <= size)
113 return;
114
115 // Construit une vue avec la taille du padding
116 // pour éviter les débordements de tableau lors des tests
117 Span<const DataType> padded_ids(ids.data(), padding_size);
118
119 // Vérifie que le padding est fait avec la dernière valeur valide.
120 SizeType last_id = ids[size - 1];
121 for (SizeType k = size; k < padding_size; ++k)
122 if (padded_ids[k] != last_id)
123 ARCCORE_FATAL("Bad padding value i={0} expected={1} value={2}",
124 k, last_id, padded_ids[k]);
125 }
126};
