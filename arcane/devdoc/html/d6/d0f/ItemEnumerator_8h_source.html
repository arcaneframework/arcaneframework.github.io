164 
165  void _init()
166  {
167  m_is_not_end = (m_index<m_count);
168  if (m_is_not_end){
169  Int32 lid = m_local_ids[m_index];
170  // Vérifie qu'on n'indexe pas 'm_items' avec un localId() nul.
171  // Cela n'est pas possible avec les groupes d'entités mais ca l'est avec
172  // les ItemVector par exemple.
173  // De même, 'm_items' peut-être nul si tous les localId() sont nuls.
174  // Si c'est le cas, alors on prend comme ItemSharedInfo l'instance nulle.
175  // A terme le ItemSharedInfo sera renseigné directement dans le constructeur de
176  // cette classe.
177  Int32 idx = lid;
178  if (idx==NULL_ITEM_LOCAL_ID)
179  idx = 0;
180  ItemSharedInfo* isi = (m_items) ? m_items[idx]->sharedInfo() : ItemSharedInfo::nullInstance();
181  m_base = ItemType(ItemBaseBuildInfo(lid,isi));
182  }
183  }
