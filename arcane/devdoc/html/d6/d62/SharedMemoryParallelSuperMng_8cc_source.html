131Ref<IParallelMng> SharedMemoryParallelMngContainer::
132_createParallelMng(Int32 local_rank,ITraceMng* tm)
133{
134 if (local_rank<0 || local_rank>=m_nb_local_rank)
135 ARCANE_THROW(ArgumentException,"Bad value '{0}' for local_rank (max={1})",
136 local_rank,m_nb_local_rank);
137
138 // Cette méthode n'est pas réentrante.
139 Mutex::ScopedLock sl(m_internal_create_mutex);
140
141 IParallelSuperMng* sm = m_application->sequentialParallelSuperMng();
142 SharedMemoryParallelMngBuildInfo build_info;
143 build_info.rank = local_rank;
144 build_info.nb_rank = m_nb_local_rank;
145 build_info.trace_mng = tm;
146 build_info.thread_mng = m_thread_mng;
147 build_info.sequential_parallel_mng = sm->internalCreateWorldParallelMng(0);
148 build_info.world_parallel_mng = nullptr;
149 build_info.message_queue = m_message_queue;
150 build_info.thread_barrier = m_thread_barrier;
151 build_info.all_dispatchers = m_all_dispatchers;
152 build_info.sub_builder_factory = m_sub_factory_builder;
153 build_info.container = makeRef<IParallelMngContainer>(this);
154 // Seul le rang 0 positionne l'éventuel communicateur sinon tous les PE
155 // vont se retrouver avec le même rang MPI
156 if (local_rank==0)
157 build_info.communicator = m_communicator;
158
159 IParallelMng* pm = new SharedMemoryParallelMng(build_info);
160 pm->build();
161
162 return makeRef(pm);
163}

236
237void SharedMemoryParallelSuperMng::
238build()
239{
240 if (!arcaneHasThread())
241 ARCANE_FATAL("Can not create SharedMemoryParallelSuperMng because threads are disabled");
242
243 // Si on a été initialisé avec MPI, alors les requêtes nulles et le communicateur
244 // nul ont déjà été positionnés.
245 if (!m_has_mpi_init){
246 Request::setNullRequest(Request(0,nullptr,0));
247 Parallel::Communicator::setNullCommunicator(Parallel::Communicator((void*)nullptr));
248 }
249
250 m_is_parallel = true;
251 Int32 n = m_application->applicationBuildInfo().nbSharedMemorySubDomain();
252 if (n==0)
253 ARCANE_FATAL("Number of shared memory sub-domains is not defined");
254
255 ITraceMng* app_tm = m_application->traceMng();
256 app_tm->info() << "SharedMemoryParallelSuperMng: nb_local_sub_domain=" << n;
257 app_tm->info() << "SharedMemoryParallelSuperMng: mpi_communicator=" << getMPICommunicator();
258 ServiceBuilder<IParallelMngContainerFactory> sb(m_application);
259 String service_name = "SharedMemoryParallelMngContainerFactory";
260 m_builder_factory = sb.createReference(service_name);
261 Ref<IParallelMngContainer> x = m_builder_factory->_createParallelMngBuilder(n,communicator());
262 m_main_builder = x;
263 m_container = dynamic_cast<SharedMemoryParallelMngContainer*>(x.get());
264 ARCANE_CHECK_POINTER(m_container);
265}

270Ref<IParallelMng> SharedMemoryParallelSuperMng::
271internalCreateWorldParallelMng(Int32 local_rank)
272{
273 Int32 max_rank = nbLocalSubDomain();
274 if (local_rank<0 || local_rank>=max_rank)
275 ARCANE_THROW(ArgumentException,"Bad value '{0}' for local_rank (max={1})",
276 local_rank,max_rank);
277
278 ITraceMng* tm = nullptr;
279 ITraceMng* app_tm = m_application->traceMng();
280 if (local_rank==0){
281 // Le premier sous-domaine créé utilise le traceMng() par défaut.
282 tm = app_tm;
283 }
284 else{
285 tm = m_application->createAndInitializeTraceMng(app_tm,String::fromNumber(local_rank));
286 }
287
288 Ref<IParallelMng> pm = m_container->_createParallelMng(local_rank,tm);
289 return pm;
290}
