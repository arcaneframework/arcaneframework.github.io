52  void initWithOneRankPerNode(bool allow_only_one_rank)
53  {
54  m_is_init = true;
55  m_targetRank = -1;
56  m_skip = true;
57  m_contribute = false;
58  if (!m_pm_ini)
59  return;
60 
61  auto topo { ParallelMngUtils::createTopologyRef(m_pm_ini) };
62  if (topo->isMasterMachine()) {
63  m_contribute = true;
64  }
65 
66  Int32 machineRank = topo->machineRank();
67  Int32ConstArrayView targetRanks = topo->masterMachineRanks();
68 
69  m_targetRank = targetRanks[machineRank];
70  m_targetSize = targetRanks.size();
71 
72  if ((m_targetSize != m_pm_ini->commSize()) // Only useful if number of processes change.
73  && (allow_only_one_rank || m_targetSize > 1)) { // And if more than 1 node for parmetis
74  m_skip = false;
75  m_pm_sub = m_pm_ini->createSubParallelMngRef(targetRanks);
76  }
77  else { // All ranks have to work.
78  // Still making a new communicator for safety when using with third party library
79  m_contribute = true;
80  m_targetRank = m_pm_ini->commRank();
81  m_targetSize = m_pm_ini->commSize();
82  Int32UniqueArray keptRanks(m_targetSize);
83  for (int i = 0 ; i < m_targetSize ; i++)
84  keptRanks[i] = i;
85  m_pm_sub = m_pm_ini->createSubParallelMngRef(keptRanks);
86  }
87 
88  m_pm_ini->traceMng()->info() << "Running on " << m_targetSize << " nodes";
89  }

91  // Max is hard-coded to work on integers.
92  void initWithMaxRank(Int32 targetSize)
93  {
94  m_is_init = true;
95  m_targetSize = targetSize;
96  m_targetRank = -1;
97  m_skip = true;
98  m_contribute = false;
99 
100  if (!m_pm_ini)
101  return;
102 
103  if (m_pm_ini->commRank() < targetSize) { // At this time, no duplication
104  m_contribute = 1;
105  }
106  Int64 x = m_pm_ini->commRank() * m_targetSize;
107  m_targetRank = CheckedConvert::toInt32(x / m_pm_ini->commSize());
108 
109  Int32UniqueArray keepProc(m_targetSize);
110  Int32 step = m_targetSize/m_pm_ini->commSize();
111  step = (step == 0)?1:step;
112  for (int i = 0 ; i < m_targetSize ; ++i) {
113  keepProc[i] = i*step;
114  }
115 
116  m_pm_sub = m_pm_ini->createSubParallelMngRef(keepProc);
117 
118  if (m_targetSize != m_pm_ini->commSize()) {
119  m_skip = false;
120  }
121  }

138  //< Do the redistribution pm -> newComm
139  template<typename DataT>
140  SharedArray<DataT> convert(ConstArrayView<DataT> in, Array<DataT> * pattern = nullptr,
141  bool is_indirection=false) const
142  {
143  if (!m_is_init)
144  ARCANE_FATAL("Missing initialisation");
145  if (m_skip) {
146  SharedArray<DataT> out(in);
147  if (pattern != NULL) {
148  Integer size = in.size();
149  if (is_indirection) size -= 1;
150  pattern->resize(size, m_targetRank);
151  }
152  return out;
153  }
154  ConstArrayView<DataT> toSnd;
155 
156  Int32 nInfos = 2;
157  if (is_indirection) {
158  toSnd = in.subView(0, in.size()-1);
159  nInfos += 1; // need to store end of array
160  }
161  else {
162  toSnd = in;
163  }
164 
165  Int32 commSize = m_pm_ini->commSize();
166  UniqueArray<Int32> sndCnt(nInfos*commSize,-1);
167  UniqueArray<Parallel::Request> req;
168  UniqueArray<Int32> n_wanted(nInfos);
169  n_wanted[0] = m_targetRank;
170  n_wanted[1] = toSnd.size();
171  if (is_indirection)
172  n_wanted[2] = static_cast<Int32>(in[in.size()-1]);
173 
174  m_pm_ini->allGather(n_wanted, sndCnt);
175 
176  UniqueArray<Int32> sndNbr (commSize, 0);
177  UniqueArray<Int32> rcvNbr (commSize, 0);
178  UniqueArray<Int32> sndDsp (commSize, 0);
179  UniqueArray<Int32> rcvDsp (commSize, 0);
180 
181  sndNbr[m_targetRank] = toSnd.size();
182 
183  if (pattern != NULL) {
184  pattern->resize(0);
185  }
186 
187  Int32 myRank = m_pm_ini->commRank();
188  Int32 begin = 0;
189  for (int i = 0 ; i < commSize ; ++i) {
190  if (sndCnt[nInfos*i] == myRank) { // We have to receive this message
191  rcvNbr[i] = sndCnt[nInfos*i+1];
192  rcvDsp[i] = begin;
193  begin += rcvNbr[i] ;
194 
195  if (pattern != NULL)
196  pattern->addRange(i, rcvNbr[i]);
197  }
198  }
199  if (contribute() && is_indirection) begin += 1; // Trick: add one to mark end of array
200  SharedArray<DataT> out(begin, -1);
201 
202  m_pm_ini->allToAllVariable(toSnd, sndNbr, sndDsp, out, rcvNbr, rcvDsp);
203 
204  if (contribute() && is_indirection) { // We have to update offsets
205  DataT offset = 0;
206  DataT* my_iter = out.data();
207  for (int i = 0 ; i < commSize ; ++i) {
208  if (sndCnt[nInfos*i] == myRank) { // We have to receive this message
209  Int32 nRecv = sndCnt[nInfos*i+1];
210  DataT* my_end(my_iter+nRecv);
211  for ( ; my_iter != my_end ; ++ my_iter)
212  (*my_iter) += offset;
213  offset += sndCnt[nInfos*i+2];
214  }
215  }
216  out[out.size()-1] = offset;
217  }
218 
219  return out;
220  }

222  //< Do the backward redistribution newComm -> pm
223  template<typename DataT>
224  SharedArray<DataT> convertBack(ConstArrayView<DataT> in, Int32 nRecv) const
225  {
226  if (!m_is_init)
227  ARCANE_FATAL("Missing initialisation");
228  if (m_skip) {
229  SharedArray<DataT> out(in);
230  return out;
231  }
232 
233  Int32 nInfos = 2;
234  Int32 commSize = m_pm_ini->commSize();
235  UniqueArray<Int32> sndCnt(nInfos*commSize,-1);
236  UniqueArray<Parallel::Request> req;
237  UniqueArray<Int32> n_wanted(nInfos);
238  n_wanted[0] = m_targetRank;
239  n_wanted[1] = nRecv;
240 
241  m_pm_ini->allGather(n_wanted, sndCnt);
242 
243  UniqueArray<Int32> sndNbr (commSize, 0);
244  UniqueArray<Int32> rcvNbr (commSize, 0);
245  UniqueArray<Int32> sndDsp (commSize, 0);
246  UniqueArray<Int32> rcvDsp (commSize, 0);
247 
248  rcvNbr[m_targetRank] = nRecv;
249 
250  Int32 myRank = m_pm_ini->commRank();
251  Int32 begin = 0;
252  for (int i = 0 ; i < commSize ; ++i) {
253  if (sndCnt[nInfos*i] == myRank) { // We have to receive this message
254  sndNbr[i] = sndCnt[nInfos*i+1];
255  sndDsp[i] = begin;
256  begin += sndNbr[i] ;
257  }
258  }
259  SharedArray<DataT> out(nRecv, -1);
260 
261  m_pm_ini->allToAllVariable(in, sndNbr, sndDsp, out, rcvNbr, rcvDsp);
262 
263  return out;
264  }

277  bool m_skip; // Pas de redistribution

279  bool m_is_init = false;
