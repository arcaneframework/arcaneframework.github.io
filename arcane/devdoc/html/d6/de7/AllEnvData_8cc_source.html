107void AllEnvData::
108_computeAndResizeEnvItemsInternal()
109{
110 // Calcule le nombre de milieux par maille, et pour chaque
111 // milieu le nombre de matériaux par maille
112 IMesh* mesh = m_material_mng->mesh();
113 const IItemFamily* cell_family = mesh->cellFamily();
114 ConstArrayView<MeshEnvironment*> true_environments(m_material_mng->trueEnvironments());
115
116 Integer nb_env = true_environments.size();
117 Integer total_env_cell = 0;
118 Integer total_mat_cell = 0;
119 info(4) << "NB_ENV = " << nb_env;
120 for (const MeshEnvironment* env : true_environments) {
121 CellGroup cells = env->cells();
122 Integer env_nb_cell = cells.size();
123 info(4) << "EnvName=" << cells.name() << " nb_env_cell=" << env_nb_cell << " nb_mat_cell=" << env->totalNbCellMat();
124 total_env_cell += env_nb_cell;
125 total_mat_cell += env->totalNbCellMat();
126 }
127
128 // Il faut ajouter les infos pour les mailles de type AllEnvCell
129 Int32 max_local_id = cell_family->maxLocalId();
130 info(4) << "RESIZE TotalEnvCell=" << total_env_cell
131 << " TotalMatCell=" << total_mat_cell
132 << " MaxLocalId=" << max_local_id;
133
134 // Redimensionne les tableaux des infos
135 // ATTENTION : ils ne doivent plus être redimensionnés par la suite sous peine
136 // de tout invalider.
137 m_item_internal_data.resizeComponentItemInternals(max_local_id, total_env_cell);
138
139 if (arcaneIsCheck()) {
140 Int32 computed_nb_mat = 0;
141 Int32 computed_nb_env = 0;
142 ConstArrayView<Int16> cells_nb_env = m_component_connectivity_list->cellsNbEnvironment();
143 ConstArrayView<Int16> cells_nb_mat = m_component_connectivity_list->cellsNbMaterial();
144 ENUMERATE_ (Cell, icell, cell_family->allItems()) {
145 Int32 lid = icell.itemLocalId();
146 computed_nb_env += cells_nb_env[lid];
147 computed_nb_mat += cells_nb_mat[lid];
148 }
149 Int32 computed_size = computed_nb_mat + computed_nb_env;
150 Int32 storage_size = total_mat_cell + total_env_cell;
151 info(4) << "storage_size=" << storage_size << " computed=" << computed_size
152 << " max_local_id=" << max_local_id << " internal_nb_mat=" << total_mat_cell << " internal_nb_env=" << total_env_cell
153 << " computed_nb_mat=" << computed_nb_mat << " computed_nb_env=" << computed_nb_env;
154 if (storage_size != computed_size)
155 ARCANE_FATAL("BAD STORAGE SIZE internal={0} connectivity={1}", storage_size, computed_size);
156 }
157}

164void AllEnvData::
165_rebuildIncrementalConnectivitiesFromGroups()
166{
167 RunQueue queue(makeQueue(m_material_mng->runner()));
168 ConstArrayView<MeshEnvironment*> true_environments(m_material_mng->trueEnvironments());
169 auto clist = m_component_connectivity_list;
170 clist->removeAllConnectivities();
171 for (MeshEnvironment* env : true_environments) {
172 clist->addCellsToEnvironment(env->componentId(), env->cells().view().localIds(), queue);
173 for (MeshMaterial* mat : env->trueMaterials())
174 clist->addCellsToMaterial(mat->componentId(), mat->cells().view().localIds(), queue);
175 }

181void AllEnvData::
182_rebuildMaterialsAndEnvironmentsFromGroups()
183{
184 ConstArrayView<MeshEnvironment*> true_environments(m_material_mng->trueEnvironments());
185 const bool is_full_verbose = _isFullVerbose();
186 ConstArrayView<Int16> cells_nb_env = m_component_connectivity_list->cellsNbEnvironment();
187 for (const MeshEnvironment* env : true_environments) {
188 MeshMaterialVariableIndexer* var_indexer = env->variableIndexer();
189 ComponentItemListBuilderOld list_builder(var_indexer, 0);
190 CellGroup cells = var_indexer->cells();
191 Integer var_nb_cell = cells.size();
192 info(4) << "ENV_INDEXER (V2) i=" << var_indexer->index() << " NB_CELL=" << var_nb_cell << " name=" << cells.name()
193 << " index=" << var_indexer->index();
194 if (is_full_verbose)
195 info(5) << "ENV_INDEXER (V2) name=" << cells.name() << " cells=" << cells.view().localIds();
196
197 ENUMERATE_CELL (icell, cells) {
198 if (cells_nb_env[icell.itemLocalId()] > 1)
199 list_builder.addPartialItem(icell.itemLocalId());
200 else
201 // Je suis le seul milieu de la maille donc je prends l'indice global
202 list_builder.addPureItem(icell.itemLocalId());
203 }
204 if (is_full_verbose)
205 info() << "MAT_NB_MULTIPLE_CELL (V2) mat=" << var_indexer->name()
206 << " nb_in_global=" << list_builder.pureMatVarIndexes().size()
207 << " (ids=" << list_builder.pureMatVarIndexes() << ")"
208 << " nb_in_multiple=" << list_builder.partialMatVarIndexes().size()
209 << " (ids=" << list_builder.partialLocalIds() << ")";
210 var_indexer->endUpdate(list_builder);
211 }
212
213 for (MeshEnvironment* env : true_environments)
214 env->computeItemListForMaterials(*m_component_connectivity_list);
215}

220void AllEnvData::
221_computeInfosForEnvCells()
222{
223 IMesh* mesh = m_material_mng->mesh();
224 IItemFamily* cell_family = mesh->cellFamily();
225 CellGroup all_cells = cell_family->allItems();
226 const Int32 nb_cell = all_cells.size();
227 ConstArrayView<MeshEnvironment*> true_environments(m_material_mng->trueEnvironments());
228
229 SmallSpan<const Int16> cells_nb_env = m_component_connectivity_list->cellsNbEnvironment();
230
231 // Calcule pour chaque maille sa position dans le tableau des milieux
232 // en considérant que les milieux de chaque maille sont rangés consécutivement
233 // dans m_env_items_internal.
234 const Int32 max_local_id = cell_family->maxLocalId();
235 UniqueArray<Int32> env_cell_indexes(platform::getDefaultDataAllocator());
236 env_cell_indexes.resize(cells_nb_env.size());
237
239 UniqueArray<Int16> cells_nb_material(platform::getDefaultDataAllocator());
240 cells_nb_material.resize(max_local_id);
241
242 RunQueue& queue(m_material_mng->runQueue());
243
244 bool do_old = (max_local_id != nb_cell);
245 if (do_old) {
246 Integer env_cell_index = 0;
247 ENUMERATE_CELL (icell, all_cells) {
248 Int32 lid = icell.itemLocalId();
249 Int32 nb_env = cells_nb_env[lid];
250 env_cell_indexes[lid] = env_cell_index;
251 env_cell_index += nb_env;
252 }
253 }
254 else {
255 // TODO: Cela ne fonctionne que si all_cells est compacté et
256 // local_id[i] <=> i.
257 Accelerator::GenericScanner scanner(queue);
258 SmallSpan<Int32> env_cell_indexes_view(env_cell_indexes);
259 Accelerator::ScannerSumOperator<Int32> op;
260 scanner.applyExclusive(0, cells_nb_env, env_cell_indexes_view, op, A_FUNCINFO);
261 }
262
263 // Positionne les infos pour les EnvCell
264 {
265 NumArray<Int32, MDDim1> current_pos;
266 {
267 MDSpan<Int32, MDDim1> s(env_cell_indexes.data(), ArrayIndex<1>{ env_cell_indexes.size() });
268 current_pos.copy(s);
269 }
270 for (MeshEnvironment* env : true_environments) {
271 const Int16 env_id = env->componentId();
272 const MeshMaterialVariableIndexer* var_indexer = env->variableIndexer();
273 CellGroup cells = env->cells();
274
275 env->resizeItemsInternal(var_indexer->nbItem());
276
277 info(4) << "COMPUTE (V2) env_cells env=" << env->name() << " nb_cell=" << cells.size()
278 << " index=" << var_indexer->index()
279 << " max_multiple_index=" << var_indexer->maxIndexInMultipleArray();
280
281 SmallSpan<const MatVarIndex> matvar_indexes(var_indexer->matvarIndexes());
282
283 Int32ConstArrayView local_ids = var_indexer->localIds();
284
285 SmallSpan<Int16> cells_nb_mat_view = cells_nb_material.view();
286 m_component_connectivity_list->fillCellsNbMaterial(local_ids, env_id, cells_nb_mat_view, queue);
287
288 auto command = makeCommand(queue);
289 SmallSpan<Int32> current_pos_view(current_pos);
290 const Int32 nb_id = matvar_indexes.size();
291 ComponentItemSharedInfo* env_shared_info = m_item_internal_data.envSharedInfo();
292
293 Span<Int32> env_cells_local_id = cells._internalApi()->itemsLocalId();
294 SmallSpan<ConstituentItemIndex> env_id_list = env->componentData()->m_constituent_local_id_list.mutableLocalIds();
295 command << RUNCOMMAND_LOOP1(iter, nb_id)
296 {
297 auto [z] = iter();
298 MatVarIndex mvi = matvar_indexes[z];
299
300 Int32 lid = local_ids[z];
301 Int32 pos = current_pos_view[lid];
302 ++current_pos_view[lid];
303 Int16 nb_mat = cells_nb_mat_view[z];
304
305 ConstituentItemIndex cii_pos(pos);
306 matimpl::ConstituentItemBase ref_ii(env_shared_info, cii_pos);
307 ConstituentItemIndex cii_lid(lid);
308 env_id_list[z] = cii_pos;
309
310 ref_ii._setSuperAndGlobalItem(cii_lid, ItemLocalId(lid));
311 ref_ii._setNbSubItem(nb_mat);
312 ref_ii._setVariableIndex(mvi);
313 ref_ii._setComponent(env_id);
314 // Le rang 0 met à jour le padding SIMD du groupe associé au matériau
315 if (z==0)
316 ArraySimdPadder::applySimdPaddingView(env_cells_local_id);
317 };
318 cells._internalApi()->notifySimdPaddingDone();
319 }
320 for (MeshEnvironment* env : true_environments) {
321 env->computeMaterialIndexes(&m_item_internal_data, queue);
322 }
323 }
324
325 // Positionne les infos pour les AllEnvCell.
326 {
327 ComponentItemSharedInfo* all_env_shared_info = m_item_internal_data.allEnvSharedInfo();
328 auto command = makeCommand(queue);
329 SmallSpan<Int32> env_cell_indexes_view(env_cell_indexes);
330 command << RUNCOMMAND_ENUMERATE (Cell, cell_id, all_cells)
331 {
332 Int32 lid = cell_id;
333 Int16 n = cells_nb_env[lid];
334 matimpl::ConstituentItemBase ref_ii(all_env_shared_info, ConstituentItemIndex(lid));
335 ref_ii._setSuperAndGlobalItem({}, cell_id);
336 ref_ii._setVariableIndex(MatVarIndex(0, lid));
337 ref_ii._setNbSubItem(n);
338 if (n != 0)
339 ref_ii._setFirstSubItem(ConstituentItemIndex(env_cell_indexes_view[lid]));
340 };
341 }
342}

350void AllEnvData::
351_checkLocalIdsCoherency() const
352{
353 for (MeshEnvironment* env : m_material_mng->trueEnvironments()) {
354 Int32 index = 0;
355 Int32ConstArrayView indexer_local_ids = env->variableIndexer()->localIds();
356 ENUMERATE_COMPONENTCELL (icitem, env) {
357 ComponentCell cc = *icitem;
358 Int32 matvar_lid = cc.globalCell().localId();
359 Int32 direct_lid = indexer_local_ids[index];
360 if (matvar_lid != direct_lid)
361 ARCANE_FATAL("Incoherent localId() matvar_lid={0} direct_lid={1} index={2}",
362 matvar_lid, direct_lid, index);
363 ++index;
364 }
365 }

386void AllEnvData::
387forceRecompute(bool compute_all)
388{
389 m_material_mng->incrementTimestamp();
390
391 ConstArrayView<MeshMaterialVariableIndexer*> vars_idx = m_material_mng->_internalApi()->variablesIndexer();
392 Integer nb_var = vars_idx.size();
393 Int64 mesh_timestamp = m_material_mng->mesh()->timestamp();
394 info(4) << "ForceRecompute NB_VAR_IDX=" << nb_var << " compute_all?=" << compute_all
395 << " mesh_timestamp=" << mesh_timestamp << " current_mesh_timestamp=" << m_current_mesh_timestamp;
396
397 // Si le maillage a changé, il y a certaines choses qu'il faut toujours recalculer
398 bool has_mesh_changed = m_current_mesh_timestamp != mesh_timestamp;
399 m_current_mesh_timestamp = mesh_timestamp;
400
401 const bool is_verbose_debug = m_verbose_debug_level > 0;
402
403 // Il faut compter le nombre total de mailles par milieu et par matériau
404
405 ConstArrayView<MeshEnvironment*> true_environments(m_material_mng->trueEnvironments());
406
407 if (compute_all)
408 _rebuildIncrementalConnectivitiesFromGroups();
409
410 // Calcule le nombre de milieux par maille, et pour chaque
411 // milieu le nombre de matériaux par maille
412 if (compute_all || has_mesh_changed)
413 _computeNbEnvAndNbMatPerCell();
414
415 _computeAndResizeEnvItemsInternal();
416
417 bool is_full_verbose = _isFullVerbose();
418
419 if (compute_all)
420 _rebuildMaterialsAndEnvironmentsFromGroups();
421
422 for (const MeshEnvironment* env : true_environments) {
423 const MeshMaterialVariableIndexer* var_indexer = env->variableIndexer();
424 CellGroup cells = var_indexer->cells();
425 Integer var_nb_cell = cells.size();
426 info(4) << "FINAL_INDEXER i=" << var_indexer->index() << " NB_CELL=" << var_nb_cell << " name=" << cells.name()
427 << " index=" << var_indexer->index();
428 if (is_full_verbose) {
429 Int32UniqueArray my_array(cells.view().localIds());
430 info(5) << "FINAL_INDEXER (V2) name=" << cells.name() << " cells=" << my_array;
431 info(4) << "FINAL_MAT_NB_MULTIPLE_CELL (V2) mat=" << var_indexer->name()
432 << " ids=" << var_indexer->matvarIndexes();
433 }
434 }
435
436 _computeInfosForEnvCells();
437
438 if (is_verbose_debug) {
439 _printAllEnvCells(m_material_mng->mesh()->allCells().view());
440 for (IMeshMaterial* material : m_material_mng->materials()) {
441 ENUMERATE_COMPONENTITEM (MatCell, imatcell, material) {
442 MatCell pmc = *imatcell;
443 info() << "CELL IN MAT vindex=" << pmc._varIndex();
444 }
445 }
446 }
447
448 {
449 RunQueue& queue(m_material_mng->runQueue());
450 for (MeshEnvironment* env : true_environments) {
451 env->componentData()->_rebuildPartData(queue);
452 for (MeshMaterial* mat : env->trueMaterials())
453 mat->componentData()->_rebuildPartData(queue);
454 }
455 }
456
457 if (arcaneIsCheck())
458 m_material_mng->checkValid();
459
460 m_material_mng->syncVariablesReferences(compute_all);
461
462 if (is_verbose_debug) {
463 OStringStream ostr;
464 m_material_mng->dumpInfos2(ostr());
465 info() << ostr.str();
466 }
467
468 // Vérifie la cohérence des localIds() du variableIndexer()
469 // avec la maille globale associée au milieu
470 if (arcaneIsCheck())
471 _checkLocalIdsCoherency();
472
473 // Met à jour le AllCellToAllEnvCell s'il a été initialisé si la fonctionnalité est activé
474 if (m_material_mng->isCellToAllEnvCellForRunCommand()) {
475 auto* all_cell_to_all_env_cell(m_material_mng->_internalApi()->getAllCellToAllEnvCell());
476 if (all_cell_to_all_env_cell)
477 all_cell_to_all_env_cell->bruteForceUpdate();
478 else
479 m_material_mng->_internalApi()->createAllCellToAllEnvCell(platform::getDefaultDataAllocator());
480 }
481}

499{
500 ConstArrayView<Int16> cells_nb_env = m_component_connectivity_list->cellsNbEnvironment();

508 for (Integer z = 0; z < cell_nb_env; ++z) {
509 EnvCell ec = all_env_cell.cell(z);
510 Integer cell_nb_mat = ec.nbMaterial();
511 info() << "CELL3 nb_mat=" << cell_nb_mat << " env_id=" << ec.environmentId();
512 for (Integer k = 0; k < cell_nb_mat; ++k) {
513 MatCell mc = ec.cell(k);
514 info() << "CELL4 mat_item=" << mc._varIndex() << " mat_id=" << mc.materialId();
515 }
516 }

529void AllEnvData::
530_copyBetweenPartialsAndGlobals(const MeshVariableCopyBetweenPartialAndGlobalArgs& args,
531 bool is_add_operation)
532{
533 if (args.m_local_ids.empty())
534 return;
535
536 RunQueue::ScopedAsync sc(args.m_queue);
537 // Comme on a modifié des mailles, il faut mettre à jour les valeurs
538 // correspondantes pour chaque variable.
539 //info(4) << "NB_TRANSFORM=" << nb_transform << " name=" << e->name();
540 //Integer indexer_index = indexer->index();
541 auto func = [&](IMeshMaterialVariable* mv) {
542 auto* mvi = mv->_internalApi();
543 if (is_add_operation){
544 mvi->resizeForIndexer(args.m_var_index, *args.m_queue);
545 mvi->copyGlobalToPartial(args);
546 }
547 else
548 mvi->copyPartialToGlobal(args);
549 };
550 functor::apply(m_material_mng, &MeshMaterialMng::visitVariables, func);
551 args.m_queue->barrier();
552}

556
557void AllEnvData::
558_checkConnectivityCoherency()
559{
560 info() << "AllEnvData: checkCoherency()";
561 ConstArrayView<Int16> nb_mat_v2 = m_component_connectivity_list->cellsNbMaterial();
562 ConstArrayView<MeshEnvironment*> true_environments(m_material_mng->trueEnvironments());
563
564 ItemGroup all_cells = m_material_mng->mesh()->allCells();
565
566 Int32 nb_error = 0;
567
568 // Vérifie le nombre de matériaux par maille
569 ENUMERATE_CELL (icell, all_cells) {
570 Int32 ref_nb_mat = 0;
571 for (MeshEnvironment* env : true_environments) {
572 Int16 env_id = env->componentId();
573 ref_nb_mat += m_component_connectivity_list->cellNbMaterial(icell, env_id);
574 }
575 Int32 current_nb_mat = nb_mat_v2[icell.itemLocalId()];
576 if (ref_nb_mat != current_nb_mat) {
577 ++nb_error;
578 if (nb_error < 10)
579 error() << "Invalid values for nb_material cell=" << icell->uniqueId()
580 << " ref=" << ref_nb_mat << " current=" << current_nb_mat;
581 }
582 }
583
584 if (nb_error != 0)
585 ARCANE_FATAL("Invalid values for component connectivity nb_error={0}", nb_error);
586}
