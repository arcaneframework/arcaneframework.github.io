37 template<int N,template<int T> class LoopBoundType,typename Lambda> void
38 _applyGenericLoop(RunCommand& command,LoopBoundType<N> bounds,const Lambda& func)
39 {
40  Int64 vsize = bounds.nbElement();
41  if (vsize==0)
42  return;
43  impl::RunCommandLaunchInfo launch_info(command);
44  const eExecutionPolicy exec_policy = launch_info.executionPolicy();
45  switch(exec_policy){
46  case eExecutionPolicy::CUDA:
47 #if defined(ARCANE_COMPILING_CUDA)
48  {
49  launch_info.beginExecute();
50  auto [b,t] = launch_info.computeThreadBlockInfo(vsize);
51  cudaStream_t* s = reinterpret_cast<cudaStream_t*>(launch_info._internalStreamImpl());
52  // TODO: utiliser cudaLaunchKernel() à la place.
53  impl::doDirectGPULambdaArrayBounds<LoopBoundType<N>,Lambda> <<<b, t, 0, *s>>>(bounds,func);
54  }
55 #else
56  ARCANE_FATAL("Requesting CUDA kernel execution but the kernel is not compiled with CUDA compiler");
57 #endif
58  break;
59  case eExecutionPolicy::HIP:
60 #if defined(ARCANE_COMPILING_HIP)
61  {
62  launch_info.beginExecute();
63  auto [b,t] = launch_info.computeThreadBlockInfo(vsize);
64  hipStream_t* s = reinterpret_cast<hipStream_t*>(launch_info._internalStreamImpl());
65  auto& loop_func = impl::doDirectGPULambdaArrayBounds<LoopBoundType<N>,Lambda>;
66  hipLaunchKernelGGL(loop_func, b, t, 0, *s, bounds, func);
67  }
68 #else
69  ARCANE_FATAL("Requesting HIP kernel execution but the kernel is not compiled with HIP compiler");
70 #endif
71  break;
72  case eExecutionPolicy::Sequential:
73  launch_info.beginExecute();
74  arcaneSequentialFor(bounds,func);
75  break;
76  case eExecutionPolicy::Thread:
77  launch_info.beginExecute();
78  arcaneParallelFor(bounds,func);
79  break;
80  default:
81  ARCANE_FATAL("Invalid execution policy '{0}'",exec_policy);
82  }
83  launch_info.endExecute();
84 }
