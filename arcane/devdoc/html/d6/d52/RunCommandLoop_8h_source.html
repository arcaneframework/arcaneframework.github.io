39template<int N,template<int T> class LoopBoundType,typename Lambda> void
40_applyGenericLoop(RunCommand& command,LoopBoundType<N> bounds,const Lambda& func)
41{
42 Int64 vsize = bounds.nbElement();
43 if (vsize==0)
44 return;
45 impl::RunCommandLaunchInfo launch_info(command,vsize);
46 const eExecutionPolicy exec_policy = launch_info.executionPolicy();
47 launch_info.beginExecute();
48 switch(exec_policy){
49 case eExecutionPolicy::CUDA:
50 _applyKernelCUDA(launch_info,ARCANE_KERNEL_CUDA_FUNC(impl::doDirectGPULambdaArrayBounds)<LoopBoundType<N>,Lambda>,func,bounds);
51 break;
52 case eExecutionPolicy::HIP:
53 _applyKernelHIP(launch_info,ARCANE_KERNEL_HIP_FUNC(impl::doDirectGPULambdaArrayBounds)<LoopBoundType<N>,Lambda>,func,bounds);
54 break;
55 case eExecutionPolicy::Sequential:
56 arcaneSequentialFor(bounds,func);
57 break;
58 case eExecutionPolicy::Thread:
59 arcaneParallelFor(bounds,launch_info.computeParallelLoopOptions(vsize),func);
60 break;
61 default:
62 ARCANE_FATAL("Invalid execution policy '{0}'",exec_policy);
63 }
64 launch_info.endExecute();
65}
