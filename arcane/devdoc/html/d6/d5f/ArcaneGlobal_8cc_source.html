207extern "C++" ARCANE_UTILS_EXPORT void
208arcaneCheckAlignment(const void* ptr,Integer alignment)
209{
210 if (alignment<=0)
211 return;
212 Int64 iptr = (intptr_t)ptr;
213 Int64 modulo = iptr % alignment;
214 if (modulo!=0)
215 throw BadAlignmentException(A_FUNCINFO,ptr,alignment);
216}

221extern "C++" ARCANE_UTILS_EXPORT Integer
222arcaneSizeWithPadding(Integer size)
223{
224 if (size<=0)
225 return 0;
226 Integer modulo = size % SIMD_PADDING_SIZE;
227 if (modulo==0)
228 return size;
229 // TODO: vérifier débordement.
230 Integer padding_size = ((size / SIMD_PADDING_SIZE) + 1) * SIMD_PADDING_SIZE;
231 return padding_size;
232}
