74void VariableSynchronizerComputeList::
75compute()
76{
77 const bool is_debug = m_is_debug;
78 IItemFamily* item_family = m_item_group.itemFamily();
79 Int32 my_rank = m_parallel_mng->commRank();
80 Int32 nb_rank = m_parallel_mng->commSize();
81
82 m_is_verbose = traceMng()->verbosityLevel() >= 4;
83
84 UniqueArray<SharedArray<Int32>> boundary_items(nb_rank);
85
86 info() << "Compute synchronize informations group=" << m_item_group.name()
87 << " family=" << item_family->fullName()
88 << " this=" << this
89 << " group size=" << m_item_group.size()
90 << " is_verbose=" << m_is_verbose;
91
92 {
93 Integer nb_error = 0;
94 Int64UniqueArray bad_items_uid;
95 ENUMERATE_ITEM (i, m_item_group) {
96 Item item = *i;
97 impl::ItemBase item_internal = item.itemBase();
98 Int32 owner = item_internal.owner();
99 if (owner == my_rank)
100 continue;
101 Int64 uid = item_internal.uniqueId().asInt64();
102 if (owner == A_NULL_RANK || owner >= nb_rank) {
103 ++nb_error;
104 if (nb_error < 10)
105 bad_items_uid.add(uid);
106 continue;
107 }
108 if (is_debug) {
109 info() << "Add entity uid=" << uid
110 << " lid=" << item_internal.localId() << " to the subdomain " << owner;
111 }
112 boundary_items[owner].add(item_internal.localId());
113 }
114 if (nb_error != 0) {
115 for (Integer i = 0, is = bad_items_uid.size(); i < is; ++i) {
116 info() << "ERROR: The entity uid=" << bad_items_uid[i]
117 << " group=" << m_item_group.name() << " doesn't belong to "
118 << "any subdomain or belong to an invalid subdomain";
119 }
120 ARCANE_FATAL("Error while creating synchronization information");
121 }
122 }
123
124 _createList(boundary_items);
125
126 if (is_debug)
127 _printSyncList();
128
129 if (m_is_verbose)
130 info() << "Begin compute dispatcher Date=" << platform::getCurrentDateTime();
131}

136void VariableSynchronizerComputeList::
137_createList(UniqueArray<SharedArray<Int32>>& boundary_items)
138{
139 const bool is_debug = m_is_debug;
140 DataSynchronizeInfo* sync_info = m_synchronizer->m_sync_info.get();
141
142 sync_info->clear();
143
144 IItemFamily* item_family = m_item_group.itemFamily();
145 IParallelMng* pm = m_parallel_mng;
146 Int32 my_rank = pm->commRank();
147 Int32 nb_rank = pm->commSize();
148 info(4) << "VariableSynchronizer::createList() begin for group=" << m_item_group.name();
149
150 // Table du voisinage connu par items fantomes.
151 // Ceci n'est pas obligatoirement la liste finale pour m_communicating_ranks dans le cas
152 // de relation non symétrique ghost/shared entre processeurs (si l'un des deux vaut 0)
153 // Le traitement complémentaire apparaît après la section "Réciprocité des communications"
154 m_synchronizer->m_communicating_ranks.clear();
155 Int32UniqueArray communicating_ghost_ranks;
156 for (Integer i = 0; i < nb_rank; ++i) {
157 if (boundary_items[i].empty())
158 continue;
159 communicating_ghost_ranks.add(i);
160 if (is_debug) {
161 ItemInfoListView items_internal(item_family);
162 for (Integer z = 0, zs = boundary_items[i].size(); z < zs; ++z) {
163 Item item = items_internal[boundary_items[i][z]];
164 info() << "Item uid=" << item.uniqueId() << ",lid=" << item.localId();
165 }
166 }
167 }
168
169 Integer nb_comm_rank = communicating_ghost_ranks.size();
170
171 Int32UniqueArray nb_ghost(nb_rank);
172 Int32UniqueArray nb_share(nb_rank);
173 nb_ghost.fill(0);
174
175 // Nombre maximum de sous-domaines connectés. Sert pour dimensionner
176 // les tableaux pour le allGather()
177 Integer max_comm_rank = pm->reduce(Parallel::ReduceMax, nb_comm_rank);
178 debug() << "communicating sub domains my=" << nb_comm_rank
179 << " max=" << max_comm_rank;
180
181 // Liste des groupes des mailles fantômes.
182 UniqueArray<SharedArray<Int32>> ghost_group_list(boundary_items.size());
183
184 // Récupère les listes des entités fantômes.
185 for (Integer i = 0; i < nb_comm_rank; ++i) {
186 Int32 current_rank = communicating_ghost_ranks[i];
187 SharedArray<Int32>& ghost_grp = boundary_items[current_rank];
188 ghost_group_list[i] = ghost_grp;
189 nb_ghost[current_rank] = ghost_grp.size();
190 }
191
192 UniqueArray<ShareRankInfo> share_rank_info;
193 UniqueArray<GhostRankInfo> ghost_rank_info;
194 {
195 Integer gather_size = 1 + (max_comm_rank * 2);
196 Int32UniqueArray global_ghost_info(gather_size * nb_rank);
197 {
198 // Chaque sous-domaine construit un tableau indiquand pour
199 // chaque groupe d'éléments fantômes, le processeur concerné et
200 // le nombre d'éléments de ce groupe.
201 // Ce tableau sera ensuite regroupé sur l'ensemble des sous-domaines
202 // (par un allGather()) afin que chaque sous-domaine puisse le parcourir
203 // et ensuite savoir qui possède ses mailles partagés.
204 Int32UniqueArray local_ghost_info(gather_size);
205 Integer pos = 0;
206 local_ghost_info[pos++] = nb_comm_rank; // Indique le nombre d'éléments du tableau
207 debug() << "Send local info " << nb_comm_rank;
208 for (Integer index = 0, s = communicating_ghost_ranks.size(); index < s; ++index) {
209 local_ghost_info[pos++] = communicating_ghost_ranks[index];
210 local_ghost_info[pos++] = ghost_group_list[index].size();
211 debug() << "Send local info i=" << index << " target=" << communicating_ghost_ranks[index]
212 << " nb=" << ghost_group_list[index].size();
213 }
214 if (m_is_verbose) {
215 info() << "AllGather size() " << local_ghost_info.size()
216 << ' ' << global_ghost_info.size()
217 << " begin_date=" << platform::getCurrentDateTime();
218 }
219 pm->allGather(local_ghost_info, global_ghost_info);
220 if (m_is_verbose) {
221 info() << "AllGather end_date=" << platform::getCurrentDateTime();
222 }
223 }
224 {
225 for (Integer index = 0, s = nb_rank; index < s; ++index) {
226 Integer pos = gather_size * index;
227 Integer sub_size = global_ghost_info[pos++];
228 for (Integer sub_index = 0; sub_index < sub_size; ++sub_index) {
229 Integer proc_id = global_ghost_info[pos++];
230 Integer nb_elem = global_ghost_info[pos++];
231 if (proc_id == my_rank) {
232 if (is_debug) {
233 info() << "Get for share group " << index << ' ' << nb_elem;
234 }
235 share_rank_info.add(ShareRankInfo(index, nb_elem));
236 }
237 }
238 }
239 }
240
241 {
242 // Créé les infos concernant les mailles fantômes
243 Integer nb_send = communicating_ghost_ranks.size();
244 ghost_rank_info.resize(nb_send);
245 for (Integer i = 0; i < nb_send; ++i) {
246 SharedArray<Int32> gr = ghost_group_list[i];
247 ghost_rank_info[i].setInfos(communicating_ghost_ranks[i], gr);
248 }
249 }
250 }
251 //pm->barrier();
252 ItemInfoListView items_internal(item_family);
253 {
254 {
255 // Réciprocité des communications
256 // Pour que les synchronisations se fassent bien, il faut que
257 // 'share_rank_info' et 'ghost_rank_info' aient
258 // le même nombre d'éléments. Si ce n'est pas le cas, c'est
259 // qu'un processeur 'n' possède des mailles partagés avec le proc 'm'
260 // sans que cela soit réciproque. Si c'est le cas, on rajoute
261 // dans 'share_rank_info' une référence à ce sous-domaine
262 // avec aucun élément à envoyer.
263 Integer nb_recv = share_rank_info.size();
264 Integer nb_send = ghost_rank_info.size();
265
266 if (is_debug) {
267 info() << "Infos before auto add: send " << nb_send << " recv " << nb_recv;
268 for (Integer i = 0; i < ghost_rank_info.size(); ++i) {
269 const GhostRankInfo& asdi = ghost_rank_info[i];
270 info() << "Ghost: " << i << asdi.nbItem() << ' ' << asdi.rank();
271 }
272 for (Integer i = 0; i < share_rank_info.size(); ++i) {
273 const ShareRankInfo& asdi = share_rank_info[i];
274 info() << "Shared: " << i << ' ' << asdi.nbItem() << ' ' << asdi.rank();
275 }
276 }
277
278 for (Integer i = 0; i < nb_send; ++i) {
279 Integer proc_id = ghost_rank_info[i].rank();
280 Integer z = 0;
281 for (; z < nb_recv; ++z)
282 if (share_rank_info[z].rank() == proc_id)
283 break;
284 debug(Trace::Highest) << "CHECKS " << proc_id << ' ' << z << ' ' << nb_recv;
285 if (z == nb_recv) {
286 debug() << "Add communication with the subdomain " << proc_id;
287 share_rank_info.add(ShareRankInfo(proc_id));
288 }
289 }
290
291 for (Integer i = 0; i < nb_recv; ++i) {
292 Integer proc_id = share_rank_info[i].rank();
293 Integer z = 0;
294 for (; z < nb_send; ++z)
295 if (ghost_rank_info[z].rank() == proc_id)
296 break;
297 debug(Trace::Highest) << "CHECKR " << proc_id << ' ' << z << ' ' << nb_send;
298 if (z == nb_send) {
299 debug() << "Add communication with subdomain " << proc_id;
300 ghost_rank_info.add(GhostRankInfo(proc_id));
301 }
302 }
303
304 if (ghost_rank_info.size() != share_rank_info.size()) {
305 ARCANE_FATAL("Problem with the number of subdomain shared ({0}) and ghosts ({1})",
306 share_rank_info.size(), ghost_rank_info.size());
307 }
308 // Trie le tableau par numéro croissant de sous-domaine.
309 std::sort(std::begin(share_rank_info), std::end(share_rank_info));
310 std::sort(std::begin(ghost_rank_info), std::end(ghost_rank_info));
311 }
312
313 // Ok, maintenant on connait la liste des sous-domaines qui possèdent
314 // les mailles partagées de ce sous-domaine et réciproquement, la liste des
315 // sous-domaines intéressé par les mailles propres de ce sous-domaine.
316 // Il ne reste qu'à envoyer et recevoir les informations correspondantes.
317 // Pour cela, et afin d'éviter les blocages, on envoie d'abord les infos
318 // pour les sous-domaines dont le numéro est inférieur à ce sous-domaine.
319 Integer nb_comm_proc = ghost_rank_info.size();
320 info(4) << "Number of communicating processors: " << nb_comm_proc;
321 UniqueArray<Parallel::Request> requests;
322 {
323 //Integer nb_recv = share_rank_info.size();
324
325 // Trie le tableau par numéro croissant de sous-domaine.
326 for (Integer i = 0; i < ghost_rank_info.size(); ++i) {
327 const GhostRankInfo& asdi = ghost_rank_info[i];
328 debug() << "Ghost: " << i << " " << asdi.nbItem() << ' ' << asdi.rank();
329 }
330 for (Integer i = 0; i < share_rank_info.size(); ++i) {
331 const ShareRankInfo& asdi = share_rank_info[i];
332 debug() << "Shared: " << i << " " << asdi.nbItem() << ' ' << asdi.rank();
333 }
334 Integer current_send_index = 0;
335 Integer current_recv_index = 0;
336 for (Integer i = 0; i < nb_comm_proc * 2; ++i) {
337 Integer send_proc = nb_rank;
338 Integer recv_proc = nb_rank;
339 if (current_send_index != nb_comm_proc)
340 send_proc = ghost_rank_info[current_send_index].rank();
341 if (current_recv_index != nb_comm_proc)
342 recv_proc = ghost_rank_info[current_recv_index].rank();
343 bool do_send = true;
344 if (send_proc == recv_proc) {
345 if (send_proc < my_rank)
346 do_send = true;
347 else
348 do_send = false;
349 }
350 else if (send_proc < recv_proc)
351 do_send = true;
352 else
353 do_send = false;
354 if (do_send) {
355 GhostRankInfo& asdi = ghost_rank_info[current_send_index];
356 asdi.resize();
357 Int64ArrayView uids = asdi.uniqueIds();
358 Int32ConstArrayView asdi_local_ids = asdi.localIds();
359 //Integer zindex = 0;
360 Integer nb_local = asdi_local_ids.size();
361 for (Integer z = 0, zs = nb_local; z < zs; ++z) {
362 //for( ItemGroup::const_iter z(asdi.group()); z.hasNext(); ++z, ++zindex ){
363 const Item& elem = items_internal[asdi_local_ids[z]];
364 uids[z] = elem.uniqueId().asInt64();
365 }
366 if (is_debug) {
367 info() << "Number of elements that will be sent to the subdomain " << send_proc
368 << " " << nb_local << " éléments";
369 for (Integer z = 0; z < nb_local; ++z) {
370 info() << "Unique id " << uids[z];
371 }
372 }
373 debug() << "Send proc " << send_proc;
374 if (!uids.empty())
375 requests.add(pm->send(uids, send_proc, false));
376 ++current_send_index;
377 }
378 else {
379 ShareRankInfo& asdi = share_rank_info[current_recv_index];
380 asdi.resize();
381 Int64ArrayView items_unique_id = asdi.uniqueIds();
382 debug() << "Recv proc " << recv_proc;
383 //TODO utiliser non bloquant.
384 if (!items_unique_id.empty())
385 pm->recv(items_unique_id, recv_proc);
386 //String group_name(share_name);
387 //group_name += recv_proc;
388
389 SharedArray<Int32> items_local_id(items_unique_id.size());
390 item_family->itemsUniqueIdToLocalId(items_local_id, items_unique_id);
391 SharedArray<Int32> share_group = items_local_id;
392 debug() << "Creating shared entities for the subdomain " << recv_proc
393 << " with " << items_local_id.size() << " entities";
394 //ItemGroup share_group = mesh->itemFamily(item_kind)->createGroup(group_name,items_local_id,true);
395 //share_group.setLocalToSubDomain(true);
396 asdi.setLocalIds(share_group);
397 if (is_debug) {
398 for (Integer z = 0, zs = share_group.size(); z < zs; ++z) {
399 const Item& item = items_internal[share_group[z]];
400 info() << "Item uid=" << item.uniqueId() << ",lid=" << item.localId();
401 }
402 }
403 ++current_recv_index;
404 }
405 }
406 if (m_is_verbose) {
407 info() << "Wait requests n=" << requests.size()
408 << " begin_date=" << platform::getCurrentDateTime();
409 }
410 pm->waitAllRequests(requests);
411 if (m_is_verbose) {
412 info() << "Wait requests end_date=" << platform::getCurrentDateTime();
413 }
414 }
415 }
416 _checkValid(ghost_rank_info, share_rank_info);
417 sync_info->recompute();
418
419 // Calcul de m_communicating_ranks qui synthétisent les processeurs communiquants
420 for (Integer i = 0, n = sync_info->size(); i < n; ++i) {
421 Int32 target_rank = sync_info->targetRank(i);
422 m_synchronizer->m_communicating_ranks.add(target_rank);
423 if (sync_info->receiveInfo().nbItem(i) != boundary_items[target_rank].size())
424 ARCANE_FATAL("Inconsistent ghost count");
425 }
426}

431void VariableSynchronizerComputeList::
432_checkValid(ArrayView<GhostRankInfo> ghost_rank_info,
433 ArrayView<ShareRankInfo> share_rank_info)
434{
435 const bool is_debug = m_is_debug;
436 Integer nb_comm_proc = ghost_rank_info.size();
437 Integer nb_error = 0;
438 bool has_error = false;
439 const Integer max_error = 10; // Nombre max d'erreurs affichées.
440 Int32 my_rank = m_parallel_mng->commRank();
441 IItemFamily* item_family = m_item_group.itemFamily();
442 ItemInfoListView items_internal(item_family);
443
444 // Tableau servant à marquer les éléments qui sont soit
445 // propres au sous-domaine, soit fantômes.
446 // Normalement, si les données sont cohérentes cela devrait marquer
447 // tous les éléments.
448 // NOTE: ceci n'est utile que si \a itemGroup() vaut allItems()
449 UniqueArray<bool> marked_elem(item_family->maxLocalId());
450 marked_elem.fill(false);
451 // Marque les éléments propres au sous-domaine
452 ENUMERATE_ITEM (i_item, m_item_group) {
453 Item item = *i_item;
454 if (item.isOwn()) {
455 marked_elem[item.localId()] = true;
456 if (is_debug) {
457 info() << "Own Item " << ItemPrinter(item);
458 }
459 }
460 }
461
462 for (Integer i_comm = 0; i_comm < nb_comm_proc; ++i_comm) {
463 GhostRankInfo& ghost_info = ghost_rank_info[i_comm];
464 ShareRankInfo& share_info = share_rank_info[i_comm];
465 if (ghost_info.rank() != share_info.rank()) {
466 ARCANE_FATAL("Inconsistency between the subdomain numbers ghost_rank={0} share_rank={1}",
467 ghost_info.rank(), share_info.rank());
468 }
469 Integer current_proc = ghost_info.rank();
470 Int32ConstArrayView ghost_grp = ghost_info.localIds();
471 Int32ConstArrayView share_grp = share_info.localIds();
472
473 if (share_grp.empty() && ghost_grp.empty()) {
474 error() << "Shared and ghosts groups null for the subdomain " << current_proc;
475 has_error = true;
476 continue;
477 }
478 if (current_proc == my_rank) {
479 error() << "Error in the communication pattern: "
480 << "the processor can't communicate with itself";
481 has_error = true;
482 continue;
483 }
484
485 // Marque les éléments du groupe partagé
486 for (Integer z = 0, zs = ghost_grp.size(); z < zs; ++z) {
487 const Item& elem = items_internal[ghost_grp[z]];
488 bool is_marked = marked_elem[elem.localId()];
489 if (is_marked) {
490 // L'élément ne doit pas déjà être marqué.
491 if (nb_error < max_error)
492 error() << "The entity " << ItemPrinter(elem) << " belongs to another ghost group "
493 << "or is owned by the subdomain.";
494 ++nb_error;
495 continue;
496 }
497 marked_elem[elem.localId()] = true;
498 }
499
500 m_synchronizer->m_sync_info->add(VariableSyncInfo(share_grp, ghost_grp, current_proc));
501 }
502
503 // Vérifie que tous les éléments sont marqués
504 ENUMERATE_ITEM (i, m_item_group) {
505 Item item = *i;
506 if (!marked_elem[item.localId()]) {
507 if (nb_error < max_error) {
508 error() << "The entity " << ItemPrinter(item)
509 << " doesn't belong to the subdomain or any ghost group.";
510 }
511 ++nb_error;
512 }
513 }
514
515 // En cas d'erreur, on s'arrête.
516 if (nb_error != 0) {
517 has_error = true;
518 if (nb_error >= max_error)
519 error() << nb_error << " total elements are incorrectly dealt with";
520 }
521 if (has_error)
522 ARCANE_FATAL("Error while creating the exchange structures of the family={0}",
523 item_family->fullName());
524}

529void VariableSynchronizerComputeList::
530_printSyncList()
531{
532 DataSynchronizeInfo* sync_info = m_synchronizer->m_sync_info.get();
533 Integer nb_comm = sync_info->size();
534 info() << "SYNC LIST FOR GROUP : " << m_item_group.fullName() << " N=" << nb_comm;
535 OStringStream ostr;
536 IItemFamily* item_family = m_item_group.itemFamily();
537 ItemInfoListView items_internal(item_family);
538 for (Integer i = 0; i < nb_comm; ++i) {
539 Int32 target_rank = sync_info->targetRank(i);
540 ostr() << " TARGET=" << target_rank << '\n';
541 Int32ConstArrayView share_ids = sync_info->sendInfo().localIds(i);
542 ostr() << "\t\tSHARE(lid,uid) n=" << share_ids.size() << " :";
543 for (Integer z = 0, zs = share_ids.size(); z < zs; ++z) {
544 Item item = items_internal[share_ids[z]];
545 ostr() << " (" << item.localId() << "," << item.uniqueId() << ")";
546 }
547 ostr() << "\n";
548 Int32ConstArrayView ghost_ids = sync_info->receiveInfo().localIds(i);
549 ostr() << "\t\tGHOST(lid,uid) n=" << ghost_ids.size() << " :";
550 for (Integer z = 0, zs = ghost_ids.size(); z < zs; ++z) {
551 Item item = items_internal[ghost_ids[z]];
552 ostr() << " (" << item.localId() << "," << item.uniqueId() << ")";
553 }
554 ostr() << "\n";
555 }
556 info() << ostr.str();
557}
