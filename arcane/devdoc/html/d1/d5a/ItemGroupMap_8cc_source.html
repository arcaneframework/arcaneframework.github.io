1163void
1164ItemGroupMapAbstract::
1165_init(const ItemGroup & group)
1166{
1167// if (group.isAllItems())
1168// throw Arcane::FatalErrorException(A_FUNCINFO,"AllItems group are not allowed in this release of ItemGroupMap");
1169
1170 if (!m_group->null())
1171 {
1172 if (m_group != group.internal()) {
1173 m_group->detachObserver(m_variable);
1174 dynamic_cast<ItemGroupMapAbstractVariable*>(m_variable)->variableMng()->removeVariable(m_variable);
1175 m_group = group.internal();
1176 m_variable = new ItemGroupMapAbstractVariable(this);
1177 dynamic_cast<ItemGroupMapAbstractVariable*>(m_variable)->variableMng()->addVariable(m_variable);
1178 m_group->attachObserver(m_variable,new ItemGroupItemGroupMapObserver(m_variable));
1179 }
1180 }
1181 else
1182 {
1183 m_group = group.internal();
1184 m_variable = new ItemGroupMapAbstractVariable(this);
1185 dynamic_cast<ItemGroupMapAbstractVariable*>(m_variable)->variableMng()->addVariable(m_variable);
1186 m_group->attachObserver(m_variable,new ItemGroupItemGroupMapObserver(m_variable));
1187 }
1188
1189 if (group.isAllItems())
1190 traceMng()->warning() << "ItemGroupMap on 'AllItems' groups is unstable";
1191}
