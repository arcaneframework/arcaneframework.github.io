488 split(StringContainer& str_array,char c) const
489 {
490 const String& str = *this;
491 //TODO: passer par String::bytes().
492 const char* str_str = str.localstr();
493 Int64 offset = 0;
494 Int64 len = str.length();
495 for( Int64 i=0; i<len; ++i ){
496 // GG: remet temporairement l'ancienne sémantique (équivalente à strtok())
497 // et supprime la modif IFPEN car cela cause trop d'incompatibilités avec
498 // le code existant. A noter que l'implémentation de l'ancienne sémantique
499 // a plusieurs bugs:
500 // 1. dans le cas où on répète 3 fois ou plus consécutivement le
501 // délimiteur. Par exemple 'X:::Y' retourne {'X',':','Y'} au lieu de
502 // {'X','Y'}
503 // 2. Si on commence par le délimiteur, ce dernier est retourné:
504 // Avec ':X:Y', on retourne {':X','Y'} au lieu de {'X','Y'}
505 //if (str_str[i]==c){
506 if (str_str[i]==c && i!=offset){
507 str_array.push_back(std::string_view(str_str+offset,i-offset));
508 offset = i+1;
509 }
510 }
511 if (len!=offset)
512 str_array.push_back(std::string_view(str_str+offset,len-offset));
513 }
