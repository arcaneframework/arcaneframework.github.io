405  template<typename StringContainer> void
406  split(StringContainer& str_array,char c) const
407  {
408  const String& str = *this;
409  //TODO: passer par String::bytes().
410  const char* str_str = str.localstr();
411  Int64 offset = 0;
412  Int64 len = str.length();
413  for( Int64 i=0; i<len; ++i ){
414  // GG: remet temporairement l'ancienne sémantique (équivalente à strtok())
415  // et supprime la modif IFPEN car cela cause trop d'incompatibilités avec
416  // le code existant. A noter que l'implémentation de l'ancienne sémantique
417  // a plusieurs bugs:
418  // 1. dans le cas où on répète 3 fois ou plus consécutivement le
419  // délimiteur. Par exemple 'X:::Y' retourne {'X',':','Y'} au lieu de
420  // {'X','Y'}
421  // 2. Si on commence par le délimiteur, ce dernier est retourné:
422  // Avec ':X:Y', on retourne {':X','Y'} au lieu de {'X','Y'}
423  //if (str_str[i]==c){
424  if (str_str[i]==c && i!=offset){
425  str_array.push_back(std::string_view(str_str+offset,i-offset));
426  offset = i+1;
427  }
428  }
429  if (len!=offset)
430  str_array.push_back(std::string_view(str_str+offset,len-offset));
431  }
