52void SharedMemoryMessageRequest::
53copyFromSender(SharedMemoryMessageRequest* sender)
54{
55 SendBufferInfo send_info = sender->sendBufferInfo();
56 ReceiveBufferInfo receive_info = this->receiveBufferInfo();
57
58 const ISerializer* send_serializer = send_info.serializer();
59 ISerializer* receive_serializer = receive_info.serializer();
60 if (receive_serializer){
61 if (!send_serializer)
62 ARCANE_FATAL("No send serializer for receive serializer");
63 receive_serializer->copy(send_serializer);
64 return;
65 }
66
67 ByteConstSpan send_span = send_info.memoryBuffer();
68 ByteSpan receive_span = receive_info.memoryBuffer();
69 Int64 send_size = send_span.size();
70 Int64 receive_size = receive_span.size();
71 if (send_size > receive_size)
72 ARCANE_FATAL("Not enough memory for receiving message receive={0} send={1}",
73 receive_size,send_size);
74
75 MemoryRessourceMng::genericCopy(ConstMemoryView(send_span), MutableMemoryView(receive_span));
76}

244
245SharedMemoryMessageRequest* SharedMemoryMessageQueue::SubQueue::
246addReceive(Int64 request_id,const PointToPointMessageInfo& message,ReceiveBufferInfo buf)
247{
248 SharedMemoryMessageRequest* tmr = nullptr;
249 if (message.isRankTag()){
250 MessageTag tag = message.tag();
251 MessageRank dest = message.destinationRank();
252 tmr = _createReceiveRequest(request_id,dest,tag,buf);
253 TRACE_DEBUG("** ADD RECV queue={0} id={1} ORIG={2} DEST={3} tag={4} tmr={5} size={6} serializer={7}",
254 this,request_id,m_rank,dest,tag,tmr,buf.memoryBuffer().size(),buf.serializer());
255 }
256 else if (message.isMessageId()){
257 MessageId message_id = message.messageId();
258 MessageId::SourceInfo si = message_id.sourceInfo();
259 MessageRank dest = si.rank();
260 MessageTag tag = si.tag();
261
262 // On connait la requête 'send' qui matche celle ci. Il faut donc
263 // la positionner dès maintenant dans \a tmr pour être sur qu'on utilisera la bonne.
264 // Pour cela, cherche le send correspondant à l'id de notre requête
265 Int64 req_id = (size_t)message_id;
266 SharedMemoryMessageRequest* send_request = nullptr;
267 for( Integer i=0, n=m_send_requests.size(); i<n; ++i )
268 if (m_send_requests[i]->id()==req_id){
269 send_request = m_send_requests[i];
270 break;
271 }
272 if (!send_request)
273 ARCANE_FATAL("Can not find matching send request from MessageId");
274
275 tmr = _createReceiveRequest(request_id,dest,tag,buf);
276 TRACE_DEBUG("** ADD RECV FromMessageId queue={0} id={1} ORIG={2} DEST={3} tag={4} tmr={5} size={6} serializer={7}",
277 this,request_id,m_rank,dest,tag,tmr,buf.memoryBuffer().size(),buf.serializer());
278 tmr->setMatchingSendRequest(send_request);
279 }
280 else
281 ARCANE_THROW(NotSupportedException,"Invalid 'MessageInfo'");
282
283 m_recv_requests.add(tmr);
284 return tmr;
285}

301 return tmr;

340void SharedMemoryMessageQueue::SubQueue::
341_testOrWaitRequestAvailable(bool is_blocking)
342{
343 SharedMemoryMessageRequest* sq = nullptr;
344 if (is_blocking)
345 sq = m_async_message_queue.pop();
346 else
347 sq = m_async_message_queue.tryPop();
348 if (sq){
349 if (sq->orig()==m_rank)
350 m_done_requests.add(sq);
351 else
352 m_send_requests.add(sq);
353 }

377
378void SharedMemoryMessageQueue::SubQueue::
379wait(SharedMemoryMessageRequest* tmr)
380{
381 if (tmr->queue()!=this)
382 ARCANE_FATAL("Bad queue");
383 TRACE_DEBUG("**** WAIT MESSAGE tmr={0} rank={1} recv?={2} dest={3}"
384 " nb_send={4} nb_done={5}",tmr,m_rank,tmr->isRecv(),
385 tmr->dest(),m_send_requests.size(),m_done_requests.size());
386 while (!tmr->isDone()){
387 _checkRequestDone(tmr);
388
389 if (!tmr->isDone()){
390 // Bloque tant qu'on n'a pas recu de message.
391 waitRequestAvailable();
392 }
393 }
394
395 _cleanupRequestIfDone(tmr);
396}

442/*---------------------------------------------------------------------------*/
443/*---------------------------------------------------------------------------*/
444/*
445 * \brief Vérifie qu'un message d'envoie correspond à la requête \a tmr_recv.
446 *
447 * Regarde si une requête dans la liste des messages envoyés correspond à
448 * \a tmr_recv. Cela est le cas si le couple origine/destination convient
449 * ou si la destination est A_NULL_RANK (ce qui correspond à MPI_ANY_SOURCE
450 * dans le cas MPI).
451 * Il est important de prendre les requêtes dans l'ordre d'arrivée pour se
452 * conformer à la norme MPI.
453 *
454 * \retval true si une requête correspondante a été trouvée.
455 * \retval false sinon.
456 */
457SharedMemoryMessageRequest* SharedMemoryMessageQueue::SubQueue::
458_getMatchingSendRequest(MessageRank recv_dest,MessageRank recv_orig,MessageTag tag)
459{
460 bool is_any_tag = tag.isNull();
461 bool is_any_dest = recv_dest.isNull();
462 for( Integer j=0, n=m_send_requests.size(); j<n; ++j ){
463 SharedMemoryMessageRequest* tmr_send = m_send_requests[j];
464 TRACE_DEBUG("CHECK RECV DONE id={7} tmr_send={0} recv_dest={1}"
465 " recv_orig={2} send_dest={3} send_orig={4} request={5}/{6}\n",
466 tmr_send,recv_dest,recv_orig,
467 tmr_send->dest(),tmr_send->orig(),j,n,m_rank);
468 if (recv_orig==tmr_send->dest()){
469 bool is_rank_ok = (recv_dest==tmr_send->orig()) || (is_any_dest && m_rank==recv_orig);
470 bool is_tag_ok = (is_any_tag || tmr_send->tag()==tag);
471 if (is_rank_ok && is_tag_ok) {
472 return tmr_send;
473 }
474 }
475 }
476 return nullptr;
477}

479/*---------------------------------------------------------------------------*/
480/*---------------------------------------------------------------------------*/
481
482bool SharedMemoryMessageQueue::SubQueue::
483_checkRecvDone(SharedMemoryMessageRequest* tmr_recv)
484{
485 // Regarde si le send est déjà associé à notre requête. C'est le cas
486 // si on a utiliser un appel à probe().
487 auto* tmr_send = tmr_recv->matchingSendRequest();
488 if (!tmr_send)
489 tmr_send = _getMatchingSendRequest(tmr_recv->dest(),tmr_recv->orig(),tmr_recv->tag());
490 if (tmr_send){
491 tmr_recv->setSource(tmr_send->orig());
492 tmr_recv->copyFromSender(tmr_send);
493 tmr_recv->setDone(true);
494 tmr_send->queue()->m_async_message_queue.push(tmr_send);
495 _removeRequest(tmr_send,m_send_requests);
496 return true;
497 }
498 return false;
499}

503
504void SharedMemoryMessageQueue::SubQueue::
505waitSome(ArrayView<Request> requests,ArrayView<bool> requests_done,
506 bool is_non_blocking)
507{
508 Integer nb_request = requests.size();
509 requests_done.fill(false);
510 bool one_request_done = false;
511 TRACE_DEBUG("** WAIT SOME REQUEST rank={0} nb_request={1}\n",m_rank,nb_request);
512 while (!one_request_done){
513 if (is_non_blocking)
514 checkRequestAvailable();
515 bool has_valid_request = false;
516 for( Integer i=0; i<nb_request; ++i ){
517 Request request = requests[i];
518 if (!request.isValid())
519 continue;
520 SharedMemoryMessageRequest* tmr = requests[i];
521 if (!tmr)
522 continue;
523 this->testRequest(tmr);
524 if (tmr->isDone()){
525 one_request_done = true;
526 requests_done[i] = true;
527 tmr->destroy();
528 if (requests[i].hasSubRequest())
529 ARCANE_THROW(NotImplementedException,"handling of sub requests");
530 requests[i].reset();
531 }
532 has_valid_request = true;
533 }
534 // Si au moins une requête est terminée, sort de la boucle.
535 if (one_request_done)
536 break;
537 // Si ici, aucune requête n'a abouti. On bloque jusqu'à ce qu'il y ait
538 // un message dans la file sauf s'il n'y a aucune requête valide
539 // (Ceci est possible si toutes les requêtes de la liste sont des
540 // requêtes nulles).
541 if (!has_valid_request)
542 break;
543 // En mode non bloquant, on a déjà testé en début de boucle si
544 // une requête est disponible. Si on est ici on peut sortir directement
545 // sinon on va bloquer.
546 if (is_non_blocking)
547 break;
548 TRACE_DEBUG("** WAIT REQUEST AVAILABLE rank={0}",m_rank);
549 waitRequestAvailable();
550 }
551}

555
556MP::MessageId SharedMemoryMessageQueue::SubQueue::
557probe(const MP::PointToPointMessageInfo& message)
558{
559 TRACE_DEBUG("Probe rank={0} nb_send={1} nb_receive={2} queue={3} is_valid={4}",
560 m_rank,m_send_requests.size(),m_recv_requests.size(),this,message.isValid());
561 if (!message.isValid())
562 return MessageId();
563
564 // Il faut avoir initialisé le message avec un couple (rang/tag).
565 if (!message.isRankTag())
566 ARCCORE_FATAL("Invalid message_info: message.isRankTag() is false");
567
568 MessageRank rank = message.destinationRank();
569 MessageTag tag = message.tag();
570 bool is_blocking = message.isBlocking();
571 if (is_blocking)
572 ARCANE_THROW(NotImplementedException,"blocking probe");
573
574 // TODO: regarder pour mettre une sécurité anti-bouclage
575 // TODO: il faudrait vérifier que si on appelle deux fois cette
576 // méthode avec les mêmes informations on ne récupère pas le même message.
577 // Lorsque ce sera aussi le cas il faudra modifier legacyProbe() en
578 // conséquence.
579 for(;;){
580 _testOrWaitRequestAvailable(is_blocking);
581 auto* req = _getMatchingSendRequest(rank,m_rank,tag);
582 if (req){
583 // TODO: Vérifier que la réquête est bien avec un buffer et pas un
584 // 'ISerializer'.
585 Int64 send_size = req->sendBufferInfo().memoryBuffer().size();
586 MessageId::SourceInfo si(req->orig(),req->tag(),send_size);
587 return MessageId(si,(size_t)req->id());
588 }
589 if (!is_blocking)
590 // En non bloquant, sort de la boucle même si on n'a pas de requête.
591 break;
592 }
593 return {};
594}

666void SharedMemoryMessageQueue::
667waitAll(ArrayView<Request> requests)
668{
669 Integer nb_request = requests.size();
670 // Pour ne pas que cela bloque, il faut que les requêtes soient faites
671 // par ordre croissant des files, et les 'receive' D'ABORD !!
672 UniqueArray<SharedMemoryMessageRequest*> sorted_requests(nb_request);
673 for( Integer i=0; i<nb_request; ++i ){
674 sorted_requests[i] = (SharedMemoryMessageRequest*)requests[i].requestAsVoidPtr();
675 }
676 std::sort(std::begin(sorted_requests),std::end(sorted_requests),
677 SharedMemoryMessageRequest::SortFunctor(m_nb_thread));
678
679#if 0
680 for( Integer i=0; i<nb_request; ++i ){
681 SharedMemoryMessageRequest* tmr = sorted_requests[i];
682 cout << String::format("** WAIT FOR REQUEST tmr={0}\n",tmr);
683 }
684#endif
685
686 for( Integer i=0; i<nb_request; ++i ){
687 SharedMemoryMessageRequest* tmr = sorted_requests[i];
688 if (tmr){
689 SubQueue* sub_queue = tmr->queue();
690 sub_queue->wait(tmr);
691 tmr->destroy();
692 }
693 if (requests[i].hasSubRequest())
694 ARCANE_THROW(NotImplementedException,"handling of sub requests");
695 requests[i].reset();
696 }
697}
