138eExecutionPolicy AcceleratorRuntimeInitialisationInfo::
139executionPolicy() const
140{
141 String a = acceleratorRuntime();
142 if (a == "cuda")
143 return eExecutionPolicy::CUDA;
144 if (a == "hip")
145 return eExecutionPolicy::HIP;
146 if (a == "sycl")
147 return eExecutionPolicy::SYCL;
148 if (!a.null())
149 return eExecutionPolicy::None;
150 if (ConcurrencyBase::maxAllowedThread() > 1)
151 return eExecutionPolicy::Thread;
152 return eExecutionPolicy::Sequential;
153}

158void Impl::
159arccoreInitializeRunner(Accelerator::Runner& runner, ITraceMng* tm,
160 const AcceleratorRuntimeInitialisationInfo& acc_info)
161{
162 using namespace Accelerator;
163 String accelerator_runtime = acc_info.acceleratorRuntime();
164 eExecutionPolicy policy = acc_info.executionPolicy();
165 if (policy == eExecutionPolicy::None)
166 ARCCORE_FATAL("Invalid policy eExecutionPolicy::None");
167 tm->info() << "AcceleratorRuntime=" << accelerator_runtime;
168 tm->info() << "DefaultDataAllocator MemoryResource=" << MemoryUtils::getDefaultDataMemoryResource();
169 if (isAcceleratorPolicy(policy)) {
170 tm->info() << "Using accelerator runtime=" << policy << " device=" << acc_info.deviceId();
171 runner.initialize(policy, acc_info.deviceId());
172 runner.setAsCurrentDevice();
173 }
174 else {
175 tm->info() << "Using accelerator runtime=" << policy;
176 runner.initialize(policy);
177 }
178}
