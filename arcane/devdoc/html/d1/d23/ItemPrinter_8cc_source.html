58void ItemPrinter::
59print(std::ostream& o) const
60{
61 if (m_item.null()) {
62 if (m_has_item_kind) {
63 o << "(null " << itemKindName(m_item_kind) << ")";
64 }
65 else {
66 o << "(null Item)";
67 }
68 }
69 else {
70 o << "(";
71 ItemPrinter::Internal::_printBasics(o, m_item);
72 ItemPrinter::Internal::_printFlags(o, m_item.flags());
73 ItemPrinter::Internal::_printParents(o, m_item);
74 ItemPrinter::Internal::_printErrors(o, m_item);
75 o << ")";
76 }
77}

82void FullItemPrinter::
83print(std::ostream& ostr) const
84{
85 if (m_item.null()) {
86 ostr << "(null Item)";
87 }
88 else {
89 eItemKind ik = m_item.kind();
90 ostr << "(";
91 ItemPrinter::Internal::_printBasics(ostr, m_item);
92 ItemPrinter::Internal::_printFlags(ostr, m_item.flags());
93 ItemPrinter::Internal::_printParents(ostr, m_item);
94 ItemPrinter::Internal::_printErrors(ostr, m_item);
95 ostr << ")";
96 ostr << "\n\t";
97 if (ik != IK_Node)
98 if (m_item.nbNode() != 0)
99 ItemPrinter::Internal::_printItemSubItems(ostr, "Nodes", m_item.nodeList());
100 ostr << "\n\t";
101 if (m_item.nbEdge() != 0)
102 ItemPrinter::Internal::_printItemSubItems(ostr, "Edges", m_item.edgeList());
103 ostr << "\n\t";
104 if (m_item.nbFace() != 0)
105 ItemPrinter::Internal::_printItemSubItems(ostr, "Faces", m_item.faceList());
106 ostr << "\n\t";
107 if (ik != IK_Cell)
108 if (m_item.nbCell() != 0)
109 ItemPrinter::Internal::_printItemSubItems(ostr, "Cells", m_item.cellList());
110 }
111}

116void NeighborItemPrinter::
117_printSubItems(std::ostream& ostr, Integer level, Integer levelmax,
118 ItemVectorView sub_items, const char* name)
119{
120 indent(ostr, levelmax - level) << String::plural(sub_items.size(), name) << ":\n";
121 for (Item sub_item : sub_items) {
122 indent(ostr, levelmax - level) << "\t" << name;
123 print(ostr, sub_item, level - 1, levelmax);
124 ostr << "\n";
125 }
126}

131void NeighborItemPrinter::
132print(std::ostream& ostr, Item xitem, Integer level, Integer levelmax)
133{
134 impl::ItemBase item = xitem.itemBase();
135 if (xitem.null()) {
136 ostr << "(null Item)";
137 }
138 else {
139 eItemKind ik = item.kind();
140 ostr << "(";
141 ItemPrinter::Internal::_printBasics(ostr, item);
142 ItemPrinter::Internal::_printFlags(ostr, item.flags());
143 ItemPrinter::Internal::_printParents(ostr, item);
144 ItemPrinter::Internal::_printErrors(ostr, item);
145 ostr << ")";
146 if (level > 0) {
147 ostr << "\n";
148 if (ik != IK_Node)
149 if (item.nbNode() != 0)
150 _printSubItems(ostr, level, levelmax, item.nodeList(), "node");
151 if (item.nbEdge() != 0)
152 _printSubItems(ostr, level, levelmax, item.edgeList(), "edge");
153 if (item.nbFace() != 0)
154 _printSubItems(ostr, level, levelmax, item.faceList(), "face");
155 if (ik != IK_Cell)
156 if (item.nbCell() != 0)
157 _printSubItems(ostr, level, levelmax, item.cellList(), "cell");
158 }
159 }

268void ItemPrinter::Internal::
269_printErrors(std::ostream& o, ItemBase item)
270{
271 if (item.isSuppressed())
272 return;
273 const char* str = ", error=";
274 Integer nerror = 0;
275 Int32 mesh_rank = item.family()->mesh()->meshPartInfo().partRank();
276 if (item.isOwn() != (mesh_rank == item.owner()))
277 o << ((nerror++ == 0) ? str : "|") << "WRONG ISOWN";
278 if (item.nbParent() > 0) {
279 // Les erreurs sont ici exclusives car invalidantes pour l'item
280 ItemBase parent = item.parentBase(0);
281 if (parent.uniqueId() != item.uniqueId())
282 o << ((nerror++ == 0) ? str : "|") << "PARENT UID MISMATCH";
283 else if (parent.isSuppressed())
284 o << ((nerror++ == 0) ? str : "|") << "SUPPRESSED PARENT";
285 else if (parent.owner() != item.owner())
286 o << ((nerror++ == 0) ? str : "|") << "PARENT OWNER MISMATCH";
287 }
288}

293void ItemPrinter::Internal::
294_printItemSubItems(std::ostream& ostr, String name, const ItemVectorView& enumerator)
295{
296 ostr << " " << name << " count=" << enumerator.size();
297 ostr << " (uids=";
298 for (Item item : enumerator) {
299 if (item.localId() != NULL_ITEM_ID) {
300 if (!item.null())
301 ostr << " " << item.uniqueId();
302 else
303 ostr << " (null)";
304 }
305 else
306 ostr << " (null)";
307 }
308 ostr << ", lids=";
309 for (Item item : enumerator) {
310 if (item.localId() != NULL_ITEM_ID) {
311 if (!item.null())
312 ostr << " " << item.localId();
313 else
314 ostr << " (null)";
315 }
316 else
317 ostr << " (null)";
318 }
319 ostr << ")";
320}
