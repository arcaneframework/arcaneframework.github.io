305void CartesianMeshImpl::
306recreateFromDump()
307{
308 info() << "Creating 'CartesianMesh' infos from dump";
309
310 // Sauve le numéro de version pour être sur que c'est OK en reprise
311 Int32 v = m_properties->getInt32("Version");
312 if (v!=SERIALIZE_VERSION)
313 ARCANE_FATAL("Bad serializer version: trying to read from incompatible checkpoint v={0} expected={1}",
314 v,SERIALIZE_VERSION);
315
316 // Récupère les noms des groupes des patchs
317 UniqueArray<String> patch_group_names;
318 m_properties->get("PatchGroupNames",patch_group_names);
319 info(4) << "Found n=" << patch_group_names.size() << " patchs";
320 m_amr_patch_cell_groups.clear();
321 IItemFamily* cell_family = m_mesh->cellFamily();
322 for( const String& x : patch_group_names ){
323 CellGroup group = cell_family->findGroup(x);
324 if (group.null())
325 ARCANE_FATAL("Can not find cell group '{0}'",x);
326 m_amr_patch_cell_groups.add(group);
327 }
328
329 computeDirections();
330}

349
350void CartesianMeshImpl::
351computeDirections()
352{
353 info() << "CartesianMesh: computeDirections()";
354
355 m_mesh_timestamp = mesh()->timestamp();
356 _checkAddObservableMeshChanged();
357
358 m_amr_patches.clear();
359 m_amr_patches_pointer.clear();
360 _addPatchInstance(m_all_items_direction_info);
361
362 m_is_amr = m_mesh->isAmrActivated();
363
364 VariableCellReal3 cells_center(VariableBuildInfo(m_mesh,"TemporaryCartesianMeshCellCenter"));
365 VariableFaceReal3 faces_center(VariableBuildInfo(m_mesh,"TemporaryCartesianMeshFaceCenter"));
366
367 // Calcule les coordonnées du centre des mailles.
368 VariableNodeReal3& nodes_coord = m_mesh->nodesCoordinates();
369 ENUMERATE_CELL(icell,m_mesh->allCells()){
370 Cell cell = *icell;
371 Real3 center;
372 for( NodeLocalId inode : cell.nodeIds() )
373 center += nodes_coord[inode];
374 center /= cell.nbNode();
375 cells_center[icell] = center;
376 }
377 ENUMERATE_FACE(iface,m_mesh->allFaces()){
378 Face face = *iface;
379 Real3 center;
380 for( NodeLocalId inode : face.nodeIds() )
381 center += nodes_coord[inode];
382 center /= face.nbNode();
383 faces_center[iface] = center;
384 }
385
386 IItemFamily* cell_family = m_mesh->cellFamily();
387 IItemFamily* node_family = m_mesh->nodeFamily();
388 Int32 next_face_x = -1;
389 Int32 next_face_y = -1;
390 Int32 next_face_z = -1;
391
392 CellVectorView cell_view = cell_family->allItems().view();
393 Cell cell0 = cell_view[0];
394 Integer nb_face = cell0.nbFace();
395 Integer nb_node = cell0.nbNode();
396 Real3 cell_center = cells_center[cell0];
397
398 info(4) << "sizeof(CellDirectionMng)=" << sizeof(CellDirectionMng)
399 << " sizeof(FaceDirectionMng)=" << sizeof(FaceDirectionMng)
400 << " sizeof(NodelDirectionMng)=" << sizeof(NodeDirectionMng);
401 info(4) << "sizeof(IndexedItemConnectivityViewBase)=" << sizeof(IndexedItemConnectivityViewBase)
402 << " sizeof(CellInfoListView)=" << sizeof(CellInfoListView);
403 info(4) << "Cartesian mesh compute directions is_amr=" << m_is_amr;
404
405 for( Integer i=0; i<nb_node; ++i ){
406 Node node = cell0.node(i);
407 info(4) << "Node I=" << i << " node=" << ItemPrinter(node) << " pos=" << nodes_coord[node];
408 }
409
410 // On suppose que toutes les mailles ont le même sens de numérotation dans le maillage.
411 // Par exemple, pour toutes les mailles, la face d'indice 0 est celle du haut, celle
412 // d'indice 1 celle de droite.
413 for( Integer i=0; i<nb_face; ++i ){
414 Face f = cell0.face(i);
415 if (f.isSubDomainBoundary())
416 continue;
417 Cell next_cell = (f.backCell()==cell0) ? f.frontCell() : f.backCell();
418 Real3 next_center = cells_center[next_cell];
419 info(4) << "NEXT_CELL=" << ItemPrinter(next_cell) << " center=" << next_center \
420 << " back=" << f.backCell().uniqueId()
421 << " front=" << f.frontCell().uniqueId();
422
423 Real diff_x = math::abs(next_center.x - cell_center.x);
424 Real diff_y = math::abs(next_center.y - cell_center.y);
425 Real diff_z = math::abs(next_center.z - cell_center.z);
426 info(4) << "NEXT_CELL=" << ItemPrinter(next_cell) << " diff=" << Real3(diff_x,diff_y,diff_z);
427 //TODO: Verifier qu'il s'agit bien de la maille apres et pas avant.
428 // (tenir compte du signe de diff)
429 if (diff_x>diff_y && diff_x>diff_z){
430 // INC X
431 next_face_x = i;
432 info(4) << "Advance in direction X -> " << next_face_x;
433 }
434 else if (diff_y>diff_x && diff_y>diff_z){
435 // INC Y
436 next_face_y = i;
437 info(4) << "Advance in direction Y -> " << next_face_y;
438 }
439 else if (diff_z>diff_x && diff_z>diff_y){
440 // INC Z
441 next_face_z = i;
442 info(4) << "Advance in direction Z -> " << next_face_z;
443 }
444 else
445 ARCANE_FATAL("Bad value for next cell");
446 }
447
448 bool is_3d = m_mesh->dimension()==3;
449 m_all_items_direction_info->_internalComputeNodeCellInformations(cell0,cells_center[cell0],nodes_coord);
450
451 info() << "Informations from IMesh properties:";
452
453 auto* cmgi = ICartesianMeshGenerationInfo::getReference(m_mesh,true);
454
455 info() << "GlobalNbCell = " << cmgi->globalNbCells();
456 info() << "OwnNbCell: " << cmgi->ownNbCells();
457 info() << "SubDomainOffset: " << cmgi->subDomainOffsets();
458 info() << "OwnCellOffset: " << cmgi->ownCellOffsets();
459
460 CellGroup all_cells = cell_family->allItems();
461 NodeGroup all_nodes = node_family->allItems();
462 if (m_is_amr){
463 auto x = _buildPatchGroups(mesh()->allLevelCells(0),0);
464 all_cells = std::get<0>(x);
465 all_nodes = std::get<1>(x);
466 }
467
468 if (next_face_x!=(-1)){
469 m_local_face_direction[MD_DirX] = next_face_x;
470 _computeMeshDirection(*m_all_items_direction_info.get(),MD_DirX,cells_center,faces_center,all_cells,all_nodes);
471 }
472 if (next_face_y!=(-1)){
473 m_local_face_direction[MD_DirY] = next_face_y;
474 _computeMeshDirection(*m_all_items_direction_info.get(),MD_DirY,cells_center,faces_center,all_cells,all_nodes);
475 }
476 if (next_face_z!=(-1)){
477 m_local_face_direction[MD_DirZ] = next_face_z;
478 _computeMeshDirection(*m_all_items_direction_info.get(),MD_DirZ,cells_center,faces_center,all_cells,all_nodes);
479 }
480
481 // Positionne les informations par direction
482 for( Integer idir=0, nb_dir=mesh()->dimension(); idir<nb_dir; ++idir ){
483 CellDirectionMng& cdm = m_all_items_direction_info->cellDirection(idir);
484 cdm._internalSetOffsetAndNbCellInfos(cmgi->globalNbCells()[idir], cmgi->ownNbCells()[idir],
485 cmgi->subDomainOffsets()[idir], cmgi->ownCellOffsets()[idir]);
486 }
487
488 info() << "Compute cartesian connectivity";
489
490 m_permutation_storage.resize(1);
491 m_permutation_storage[0].compute();
492 m_nodes_to_cell_storage.resize(mesh()->nodeFamily()->maxLocalId());
493 m_cells_to_node_storage.resize(mesh()->cellFamily()->maxLocalId());
494 m_connectivity._setStorage(m_nodes_to_cell_storage,m_cells_to_node_storage,&m_permutation_storage[0]);
495 m_connectivity._computeInfos(mesh(),nodes_coord,cells_center);
496
497 // Ajoute informations de connectivités pour les patchs AMR
498 // TODO: supporter plusieurs appels à cette méthode
499 for( const CellGroup& cells : m_amr_patch_cell_groups ){
500 Integer patch_index = m_amr_patches.size();
501 info() << "AMR Patch name=" << cells.name() << " size=" << cells.size() << " index=" << patch_index;
502 auto* cdi = new CartesianMeshPatch(this,patch_index);
503 _addPatchInstance(makeRef(cdi));
504 cdi->_internalComputeNodeCellInformations(cell0,cells_center[cell0],nodes_coord);
505 auto [ patch_cells, patch_nodes ] = _buildPatchGroups(cells,patch_index);
506 _computeMeshDirection(*cdi,MD_DirX,cells_center,faces_center,patch_cells,patch_nodes);
507 _computeMeshDirection(*cdi,MD_DirY,cells_center,faces_center,patch_cells,patch_nodes);
508 if (is_3d)
509 _computeMeshDirection(*cdi,MD_DirZ,cells_center,faces_center,patch_cells,patch_nodes);
510 }
511
512 if (arcaneIsCheck())
513 checkValid();
514
515 _saveInfosInProperties();
516}

521std::tuple<CellGroup,NodeGroup> CartesianMeshImpl::
522_buildPatchGroups(const CellGroup& cells,Integer patch_level)
523{
524 // On créé un groupe pour chaque patch en garantissant que l'ordre de parcours
525 // est celui des uniqueId() des entités
526 // TODO: à terme, il faudrait que l'ordre de parcours soit le même que
527 // celui du maillage cartésien. Pour cela, il faut soit que les uniqueId()
528 // des mailles/noeuds créés soient dans le même ordre que le maillage cartésien,
529 // soit que la fonction de tri soit spécifique à ce type de maillage.
530 NodeGroup nodes = cells.nodeGroup();
531 IItemFamily* cell_family = cells.itemFamily();
532 IItemFamily* node_family = nodes.itemFamily();
533
534 String cell_group_name = String("AMRPatchCells") + patch_level;
535 CellGroup patch_cells = cell_family->createGroup(cell_group_name,Int32ConstArrayView(),true);
536 // Met les mêmes mailles que \a cells mais force le tri
537 patch_cells.setItems(cells.view().localIds(),true);
538
539 String node_group_name = String("AMRPatchNodes") + patch_level;
540 NodeGroup patch_nodes = node_family->createGroup(node_group_name,Int32ConstArrayView(),true);
541 // Met les mêmes noeuds que \a nodes mais force le tri
542 patch_nodes.setItems(nodes.view().localIds(),true);
543 info(4) << "PATCH_CELLS name=" << patch_cells.name() << " size=" << patch_cells.size();
544 info(4) << "PATCH_NODES name=" << patch_nodes.name() << " size=" << patch_nodes.size();
545 return { patch_cells, patch_nodes };
546}

551void CartesianMeshImpl::
552_computeMeshDirection(CartesianMeshPatch& cdi,eMeshDirection dir,VariableCellReal3& cells_center,
553 VariableFaceReal3& faces_center,CellGroup all_cells,NodeGroup all_nodes)
554{
555 IItemFamily* cell_family = m_mesh->cellFamily();
556 IItemFamily* face_family = m_mesh->faceFamily();
557 IItemFamily* node_family = m_mesh->nodeFamily();
558
559 Int32 max_cell_id = cell_family->maxLocalId();
560 Int32 max_face_id = face_family->maxLocalId();
561 Int32 max_node_id = node_family->maxLocalId();
562
563 CellDirectionMng& cell_dm = cdi.cellDirection(dir);
564 cell_dm._internalResizeInfos(max_cell_id);
565
566 FaceDirectionMng& face_dm = cdi.faceDirection(dir);
567 face_dm._internalResizeInfos(max_face_id);
568
569 NodeDirectionMng& node_dm = cdi.nodeDirection(dir);
570 node_dm._internalResizeInfos(max_node_id);
571
572 //TODO: attention à remettre à jour après changement de maillage.
573 info(4) << "COMPUTE DIRECTION dir=" << dir;
574
575 Int32 prev_local_face = -1;
576 Int32 next_local_face = m_local_face_direction[dir];
577 Integer mesh_dim = m_mesh->dimension();
578 // Calcul le numero local de face oppose à la face suivante.
579 if (mesh_dim==2)
580 prev_local_face = (next_local_face + 2) % 4;
581 else if (mesh_dim==3)
582 prev_local_face = (next_local_face + 3) % 6;
583
584 cell_dm._internalSetLocalFaceIndex(next_local_face,prev_local_face);
585
586 // Positionne pour chaque maille les faces avant et après dans la direction.
587 // On s'assure que ces entités sont dans le groupe des entités de la direction correspondante
588 std::set<Int32> cells_set;
589 ENUMERATE_CELL(icell,all_cells){
590 cells_set.insert(icell.itemLocalId());
591 }
592
593 // Calcule les mailles devant/derrière. En cas de patch AMR, il faut que ces deux mailles
594 // soient de même niveau
595 ENUMERATE_CELL(icell,all_cells){
596 Cell cell = *icell;
597 Int32 my_level = cell.level();
598 Face next_face = cell.face(next_local_face);
599 Cell next_cell = next_face.backCell()==cell ? next_face.frontCell() : next_face.backCell();
600 if (cells_set.find(next_cell.localId())==cells_set.end())
601 next_cell = Cell();
602 else if (next_cell.level()!=my_level)
603 next_cell = Cell();
604
605 Face prev_face = cell.face(prev_local_face);
606 Cell prev_cell = prev_face.backCell()==cell ? prev_face.frontCell() : prev_face.backCell();
607 if (cells_set.find(prev_cell.localId())==cells_set.end())
608 prev_cell = Cell();
609 else if (prev_cell.level()!=my_level)
610 prev_cell = Cell();
611 cell_dm.m_infos_view[icell.itemLocalId()] = CellDirectionMng::ItemDirectionInfo(next_cell,prev_cell);
612 }
613 cell_dm._internalComputeInnerAndOuterItems(all_cells);
614 face_dm._internalComputeInfos(cell_dm,cells_center,faces_center);
615 node_dm._internalComputeInfos(cell_dm,all_nodes,cells_center);
616}

621void CartesianMeshImpl::
622_refinePatch(Real3 position,Real3 length,bool is_3d)
623{
624 VariableNodeReal3& nodes_coord = m_mesh->nodesCoordinates();
625 UniqueArray<Int32> cells_local_id;
626 // Parcours les mailles actives et ajoute dans la liste des mailles
627 // à raffiner celles qui sont contenues dans le boîte englobante
628 // spécifiée dans le jeu de données.
629 Real3 min_pos = position;
630 Real3 max_pos = min_pos + length;
631 cells_local_id.clear();
632 ENUMERATE_CELL(icell,m_mesh->allActiveCells()){
633 Cell cell = *icell;
634 Real3 center;
635 for( NodeLocalId inode : cell.nodeIds() )
636 center += nodes_coord[inode];
637 center /= cell.nbNode();
638 bool is_inside_x = center.x>min_pos.x && center.x<max_pos.x;
639 bool is_inside_y = center.y>min_pos.y && center.y<max_pos.y;
640 bool is_inside_z = (center.z>min_pos.z && center.z<max_pos.z) || !is_3d;
641 if (is_inside_x && is_inside_y && is_inside_z)
642 cells_local_id.add(icell.itemLocalId());
643 }
644 _applyRefine(cells_local_id);
645 _saveInfosInProperties();
646}

710
711void CartesianMeshImpl::
712_applyRefine(ConstArrayView<Int32> cells_local_id)
713{
714 IItemFamily* cell_family = m_mesh->cellFamily();
715 Integer nb_cell = cells_local_id.size();
716 info(4) << "Local_NbCellToRefine = " << nb_cell;
717 Integer index = m_amr_patch_cell_groups.size();
718 String parent_group_name = String("CartesianMeshPatchParentCells")+index;
719 CellGroup parent_cells = cell_family->createGroup(parent_group_name,cells_local_id,true);
720
721 IParallelMng* pm = m_mesh->parallelMng();
722 Int64 total_nb_cell = pm->reduce(Parallel::ReduceSum,nb_cell);
723 info(4) << "Global_NbCellToRefine = " << total_nb_cell;
724 if (total_nb_cell==0)
725 return;
726
727 if(m_amr_type == eMeshAMRKind::Cell) {
728 debug() << "Refine with modifier() (for all mesh types)";
729 m_mesh->modifier()->flagCellToRefine(cells_local_id);
730 m_mesh->modifier()->adapt();
731 }
732 else if(m_amr_type == eMeshAMRKind::PatchCartesianMeshOnly) {
733 debug() << "Refine with specific refiner (for cartesian mesh only)";
734 computeDirections();
735 m_amr_mng->flagCellToRefine(cells_local_id);
736 m_amr_mng->refine();
737 }
738 else if(m_amr_type == eMeshAMRKind::Patch) {
739 ARCANE_FATAL("General patch AMR is not implemented. Please use PatchCartesianMeshOnly (3)");
740 }
741 else{
742 ARCANE_FATAL("AMR is not enabled");
743 }
744
745 {
746 MeshStats ms(traceMng(),m_mesh,m_mesh->parallelMng());
747 ms.dumpStats();
748 }
749 _addPatch(parent_cells);
750}

768
769void CartesianMeshImpl::
770renumberItemsUniqueId(const CartesianMeshRenumberingInfo& v)
771{
772 auto* cmgi = ICartesianMeshGenerationInfo::getReference(m_mesh,true);
773
774 // Regarde d'abord si on renumérote les faces
775 Int32 face_method = v.renumberFaceMethod();
776 if (face_method!=0 && face_method!=1)
777 ARCANE_FATAL("Invalid value '{0}' for renumberFaceMethod(). Valid values are 0 or 1",
778 face_method);
779 if (face_method==1)
780 ARCANE_THROW(NotImplementedException,"Method 1 for face renumbering");
781
782 // Regarde ensuite les patchs si demandé.
783 Int32 patch_method = v.renumberPatchMethod();
784 if (patch_method < 0 || patch_method > 4)
785 ARCANE_FATAL("Invalid value '{0}' for renumberPatchMethod(). Valid values are 0, 1, 2, 3 or 4",
786 patch_method);
787
788 else if (patch_method == 1 || patch_method == 3 || patch_method == 4){
789 CartesianMeshUniqueIdRenumbering renumberer(this,cmgi,v.parentPatch(),patch_method);
790 renumberer.renumber();
791 }
792 else if (patch_method == 2){
793 warning() << "The patch method 2 is experimental!";
794 CartesianMeshUniqueIdRenumberingV2 renumberer(this,cmgi);
795 renumberer.renumber();
796 }
797
798 // Termine par un tri éventuel.
799 if (v.isSortAfterRenumbering()){
800 info() << "Compacting and Sorting after renumbering";
801 m_mesh->nodeFamily()->compactItems(true);
802 m_mesh->faceFamily()->compactItems(true);
803 m_mesh->cellFamily()->compactItems(true);
804 computeDirections();
805 }
806}
