789 void _addReference(const ThatClassType* new_ref)
790 {
791 ThatClassType* nf = const_cast<ThatClassType*>(new_ref);
792 ThatClassType* prev = nf->m_prev;
793 nf->m_prev = this;
794 m_prev = prev;
795 m_next = nf;
796 if (prev)
797 prev->m_next = this;
798 }
799
800 void _removeReference()
801 {
802 if (m_prev)
803 m_prev->m_next = m_next;
804 if (m_next)
805 m_next->m_prev = m_prev;
806 }

808 void _checkFreeMemory()

811 this->_destroy();
812 this->_internalDeallocate();

816 {
817 if (&rhs != this) {
818 _removeReference();
819 _addReference(&rhs);
820 ++rhs.m_md->nb_ref;
821 --m_md->nb_ref;
822 _checkFreeMemory();
823 this->_setMP2(rhs.m_ptr, rhs.m_md);
824 }
825 }
