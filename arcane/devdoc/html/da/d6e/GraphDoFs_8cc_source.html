61void GraphDoFs::
62_allocateGraph()
63{
64 if (m_graph_allocated)
65 return;
66
67 auto connectivity_index = 0;
68 m_connectivity_indexes_per_type.resize(NB_BASIC_ITEM_TYPE, -1);
69 m_connectivity_indexes_per_type[IT_DualNode] = connectivity_index++;
70 m_connectivity_indexes_per_type[IT_DualEdge] = connectivity_index++;
71 m_connectivity_indexes_per_type[IT_DualFace] = connectivity_index++;
72 m_connectivity_indexes_per_type[IT_DualCell] = connectivity_index++;
73 m_connectivity_indexes_per_type[IT_DualParticle] = connectivity_index++;
74
75 m_item_family_network = m_mesh->itemFamilyNetwork();
76 if (m_item_family_network == nullptr)
77 traceMng()->fatal() << "ARCANE_GRAPH_CONNECTIVITY_POLICY need to be activated";
78
79 m_dualnodes_incremental_connectivity =
80 new IncrementalItemConnectivity(dualNodeFamily(),
81 linkFamily(),
82 mesh::connectivityName(dualNodeFamily(), linkFamily()));
83
84 m_connectivity_mng.registerConnectivity(m_dualnodes_incremental_connectivity);
85
86 //if (m_item_family_network)
87 // m_item_family_network->addRelation(dualNodeFamily(), linkFamily(), m_dualnodes_incremental_connectivity);
88
89 m_links_incremental_connectivity =
90 new IncrementalItemConnectivity(linkFamily(),
91 dualNodeFamily(),
92 mesh::connectivityName(linkFamily(), dualNodeFamily()));
93 m_connectivity_mng.registerConnectivity(m_links_incremental_connectivity);
94
95 if (m_item_family_network)
96 //m_item_family_network->addRelation(linkFamily(),dualNodeFamily(),m_links_incremental_connectivity);
97 m_item_family_network->addDependency(linkFamily(), dualNodeFamily(), m_links_incremental_connectivity, false);
98
99 m_incremental_connectivities.resize(NB_DUAL_ITEM_TYPE, nullptr);
100 std::array<int, NB_DUAL_ITEM_TYPE> dual_node_kinds = { IT_DualCell, IT_DualFace, IT_DualEdge, IT_DualNode, IT_DualParticle };
101 for (auto dual_node_kind : dual_node_kinds) {
102 IItemFamily* dual_item_family = _dualItemFamily(dualItemKind(dual_node_kind));
103 if (dual_item_family) {
104 auto dof2dual_incremental_connectivity =
105 new IncrementalItemConnectivity(dualNodeFamily(),
106 dual_item_family,
107 mesh::connectivityName(dualNodeFamily(), dual_item_family));
108
109 m_connectivity_mng.registerConnectivity(dof2dual_incremental_connectivity);
110 m_incremental_connectivities[_connectivityIndex(dual_node_kind)] = dof2dual_incremental_connectivity;
111 if (m_item_family_network)
112 m_item_family_network->addDependency(dualNodeFamily(), dual_item_family, dof2dual_incremental_connectivity, false);
113 }
114 }
115 m_graph_connectivity.reset(new GraphIncrementalConnectivity(dualNodeFamily(),
116 linkFamily(),
117 m_dualnodes_incremental_connectivity,
118 m_links_incremental_connectivity,
119 m_incremental_connectivities,
120 m_dual_node_to_connectivity_index));
121
122 for (auto& obs : m_connectivity_observer)
123 if(obs.get())
124 obs->notifyUpdateConnectivity();
125
126 m_update_sync_info = false;
127 m_graph_allocated = true;
128}

133void GraphDoFs::
134addLinks(Integer nb_link,
135 Integer nb_dual_nodes_per_link,
136 Int64ConstArrayView links_infos)
137{
138 Trace::Setter mci(traceMng(), _className());
139 if (!m_graph_allocated)
140 _allocateGraph();
141
142 mesh::DoFFamily& link_family = m_dof_mng.family(GraphDoFs::linkFamilyName());
143 mesh::DoFFamily& dual_node_family = m_dof_mng.family(GraphDoFs::dualNodeFamilyName());
144
145 // Extract link infos
146 Int64UniqueArray link_uids, connected_dual_node_uids;
147 link_uids.reserve(nb_link);
148 connected_dual_node_uids.reserve(nb_link * nb_dual_nodes_per_link);
149 for (auto links_infos_index = 0; links_infos_index < links_infos.size();) {
150 auto link_uid = links_infos[links_infos_index++];
151 link_uids.add(link_uid);
152 connected_dual_node_uids.addRange(
153 links_infos.subConstView(links_infos_index, nb_dual_nodes_per_link));
154 links_infos_index += nb_dual_nodes_per_link;
155 }
156
157 Int32UniqueArray link_lids(link_uids.size());
158 link_family.addDoFs(link_uids, link_lids);
159 link_family.endUpdate();
160
161 // resize connectivity
162 // fill connectivity
163 Int32UniqueArray connected_dual_nodes_lids(nb_link * nb_dual_nodes_per_link);
164 dual_node_family.itemsUniqueIdToLocalId(
165 connected_dual_nodes_lids.view(), connected_dual_node_uids.constView(), true);
166
167 auto link_index = 0;
168 ENUMERATE_DOF (inewlink, link_family.view(link_lids)) {
169 m_links_incremental_connectivity->notifySourceItemAdded(ItemLocalId(*inewlink));
170 for (auto lid : connected_dual_nodes_lids.subConstView(link_index, nb_dual_nodes_per_link)) {
171 m_links_incremental_connectivity->addConnectedItem(ItemLocalId(*inewlink), ItemLocalId(lid));
172 m_dualnodes_incremental_connectivity->addConnectedItem(ItemLocalId(lid), ItemLocalId(*inewlink));
173 }
174 link_index += nb_dual_nodes_per_link;
175 }
176 m_connectivity_mng.setUpToDate(m_links_incremental_connectivity) ;
177 m_connectivity_mng.setUpToDate(m_dualnodes_incremental_connectivity) ;
178
179 m_update_sync_info = false;
180}

240void GraphDoFs::
241addDualNodes(Integer graph_nb_dual_node,
242 Int64ConstArrayView dual_nodes_infos)
243{
244
245 Trace::Setter mci(traceMng(), _className());
246 if (!m_graph_allocated)
247 _allocateGraph();
248
249 // Size m_connecitivities if not yet done
250 bool is_parallel = m_mesh->parallelMng()->isParallel();
251 Integer domain_rank = m_mesh->parallelMng()->commRank();
252
253 std::map<Int64, std::pair<Int64UniqueArray, Int64UniqueArray>> dual_info_per_kind;
254 for (auto infos_index = 0; infos_index < dual_nodes_infos.size();) {
255 Int64 dual_node_kind = dual_nodes_infos[infos_index++];
256 auto& info = dual_info_per_kind[dual_node_kind];
257 auto& dual_node_uids = info.first;
258 auto& dual_item_uids = info.second;
259 if (dual_node_uids.size() == 0) {
260 dual_node_uids.reserve(graph_nb_dual_node);
261 dual_item_uids.reserve(graph_nb_dual_node);
262 }
263 dual_node_uids.add(dual_nodes_infos[infos_index++]);
264 dual_item_uids.add(dual_nodes_infos[infos_index++]);
265 }
266
267 for (Integer index = 0; index < NB_DUAL_ITEM_TYPE; ++index) {
268 Integer dual_node_kind = m_dualnode_kinds[index];
269 auto& info = dual_info_per_kind[dual_node_kind];
270 auto& dual_node_uids = info.first;
271 auto& dual_item_uids = info.second;
272
273 auto& dual_node_family = m_dof_mng.family(GraphDoFs::dualNodeFamilyName());
274 Int32UniqueArray dual_node_lids(dual_node_uids.size());
275 if (is_parallel) {
276 IItemFamily* dual_item_family = _dualItemFamily(dualItemKind(dual_node_kind));
277 if (dual_item_family) {
278 Int32UniqueArray dual_item_lids(dual_item_uids.size());
279 dual_item_family->itemsUniqueIdToLocalId(dual_item_lids, dual_item_uids);
280 auto dual_item_view = dual_item_family->view(dual_item_lids);
281
282 Int32UniqueArray local_dual_node_lids;
283 Int64UniqueArray local_dual_node_uids;
284
285 Integer local_size = 0;
286 for (auto const& item : dual_item_view)
287 if (item.owner() == domain_rank)
288 ++local_size;
289 local_dual_node_lids.resize(local_size);
290 local_dual_node_uids.reserve(local_size);
291
292 Integer ghost_size = dual_node_uids.size() - local_size;
293 Int64UniqueArray ghost_dual_node_uids;
294 Int32UniqueArray ghost_dual_node_lids;
295 Int32UniqueArray ghost_dual_node_owner;
296 ghost_dual_node_lids.resize(ghost_size);
297 ghost_dual_node_uids.reserve(ghost_size);
298 ghost_dual_node_owner.reserve(ghost_size);
299
300 Integer icount = 0;
301 for (auto const& item : dual_item_view) {
302 if (item.owner() == domain_rank) {
303 local_dual_node_uids.add(dual_node_uids[icount]);
304 }
305 else {
306 ghost_dual_node_uids.add(dual_node_uids[icount]);
307 ghost_dual_node_owner.add(item.owner());
308 }
309 ++icount;
310 }
311
312 dual_node_family.addDoFs(local_dual_node_uids, local_dual_node_lids);
313 dual_node_family.addGhostDoFs(ghost_dual_node_uids, ghost_dual_node_lids, ghost_dual_node_owner);
314 dual_node_family.endUpdate();
315
316 icount = 0;
317 Integer local_icount = 0;
318 Integer ghost_icount = 0;
319 for (auto const& item : dual_item_view) {
320 if (item.owner() == domain_rank)
321 dual_node_lids[icount] = local_dual_node_lids[local_icount++];
322 else
323 dual_node_lids[icount] = ghost_dual_node_lids[ghost_icount++];
324 ++icount;
325 }
326 }
327 }
328 else {
329 dual_node_family.addDoFs(dual_node_uids, dual_node_lids);
330 dual_node_family.endUpdate();
331 }
332
333 IItemFamily* dual_item_family = _dualItemFamily(dualItemKind(dual_node_kind));
334 if (dual_item_family) {
335
336 auto incremental_dual_item_connectivity = m_incremental_connectivities[index];
337 if(incremental_dual_item_connectivity==nullptr)
338 {
339 incremental_dual_item_connectivity =
340 new IncrementalItemConnectivity(dualNodeFamily(),
341 dual_item_family,
342 mesh::connectivityName(dualNodeFamily(), dual_item_family));
343
344 m_connectivity_mng.registerConnectivity(incremental_dual_item_connectivity);
345 m_incremental_connectivities[index] = incremental_dual_item_connectivity;
346 if (m_item_family_network)
347 m_item_family_network->addDependency(dualNodeFamily(), dual_item_family, incremental_dual_item_connectivity, false);
348 }
349 Int32UniqueArray dual_item_lids(dual_item_uids.size());
350 dual_item_family->itemsUniqueIdToLocalId(dual_item_lids, dual_item_uids);
351
352 ENUMERATE_DOF (idual_node, dual_node_family.view(dual_node_lids)) {
353 incremental_dual_item_connectivity->notifySourceItemAdded(ItemLocalId(*idual_node));
354 incremental_dual_item_connectivity->addConnectedItem(ItemLocalId(*idual_node), ItemLocalId(dual_item_lids[idual_node.index()]));
355 }
356
357 //m_dual_node_to_connectivity_index.resize(&dual_node_family, _connectivityIndex(dual_node_kind));
358
359 m_connectivity_mng.setUpToDate(incremental_dual_item_connectivity) ;
360 }
361 }
362
363 m_update_sync_info = false;
364}

395bool GraphDoFs::
396isUpdated()
397{
398 if(!m_graph_allocated)
399 return true ;
400
401 Integer sync_info = (m_update_sync_info ? 0 : 1) ;
402
403 sync_info = m_mesh->parallelMng()->reduce(Parallel::ReduceSum,sync_info);
404 m_update_sync_info = sync_info==0 ;
405 if(!m_update_sync_info)
406 return false ;
407
408 for (Integer index = 0; index < NB_DUAL_ITEM_TYPE; ++index) {
409 Integer dual_node_kind = m_dualnode_kinds[index];
410 auto dof2dual_incremental_connectivity = m_incremental_connectivities[_connectivityIndex(dual_node_kind)] ;
411
412 if (dof2dual_incremental_connectivity && ! m_connectivity_mng.isUpToDate(dof2dual_incremental_connectivity))
413 return false ;
414
415 if (! m_connectivity_mng.isUpToDate(m_dualnodes_incremental_connectivity))
416 return false ;
417
418 if (! m_connectivity_mng.isUpToDate(m_links_incremental_connectivity))
419 return false ;
420 }
421 return true ;
422}
