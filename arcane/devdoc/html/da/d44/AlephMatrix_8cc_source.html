23/******************************************************************************
24 *****************************************************************************/
25AlephMatrix::
26AlephMatrix(AlephKernel* kernel)
27: TraceAccessor(kernel->parallel()->traceMng())
28, m_kernel(kernel)
29, m_index(kernel->index())
30, m_setValue_idx(0)
31, m_addValue_idx(0)
32{
33 ItacFunction(AlephMatrix);
34 if (kernel->isInitialized() == false) {
35 debug() << "\33[1;32m[AlephMatrix::AlephMatrix] New Aleph matrix, but kernel is not initialized!\33[0m";
36 return;
37 }
38 // Récupération des rangs utilisés pour cette résolution
39 m_ranks = kernel->solverRanks(m_index);
40 // Booléen pour savoir si on participe ou pas
41 m_participating_in_solver = kernel->subParallelMng(m_index) != NULL;
42 debug() << "\33[1;32m[AlephMatrix::AlephMatrix] New Aleph matrix\33[0m";
43 if (!m_participating_in_solver) {
44 debug() << "\33[1;32m[AlephMatrix::AlephMatrix] Not concerned by this one!\33[0m";
45 return;
46 }
47 debug() << "\33[1;32m[AlephMatrix::AlephMatrix] site size="
48 << m_kernel->subParallelMng(m_index)->commSize()
49 << " @"
50 << m_kernel->subParallelMng(m_index)->commRank() << "\33[0m";
51 // On va chercher une matrice depuis la factory qui fait l'interface aux bibliothèques externes
52 m_implementation = m_kernel->factory()->GetMatrix(m_kernel, m_index);
53 traceMng()->flush();
54}

72void AlephMatrix::create(void)
73{
74 Timer::Action ta(m_kernel->subDomain(), "AlephMatrix::create");
75 debug() << "\33[1;32m[AlephMatrix::create(void)]\33[0m";
76 // Si le kernel n'est pas initialisé, on a rien à faire
77 if (!m_kernel->isInitialized())
78 return;

80 // on broadcast qu'un 'create' est à faire

82 m_kernel->world()->broadcast(UniqueArray<unsigned long>(1, 0xfff06e2l).view(), 0);
83 // On flush en prévision du remplissage, il faut le faire même étant configuré
84 // Par contre, on ne flush pas celui du addValue
85 m_setValue_idx = 0;
86}

126{

154void AlephMatrix::
155rowMapMapCol(Integer row,
156 Integer col,
157 Real val)
158{
159 rowColMap::const_iterator iRowMap = m_row_col_map.find(row);
160 // Si la row n'est même pas encore connue
161 // On rajoute une entrée map(map(m_addValue_idx))
162 if (iRowMap == m_row_col_map.end()) {
163 colMap* jMap = new colMap();
164 /*debug()<<"\33[1;32m[AlephMatrix::rowMapMapCol] row "
165 <<row<<" inconue, m_addValue_idx="
166 <<m_addValue_idx<<"\33[0m";*/
167 m_row_col_map.insert(std::make_pair(row, jMap));
168 jMap->insert(std::make_pair(col, m_addValue_idx));
169 updateKnownRowCol(row, col, val);
170 return;
171 }
172 // On focus sur la seconde dimension
173 colMap* jMap = iRowMap->second;
174 colMap::const_iterator iColMap = jMap->find(col);
175 // Si cet col n'est pas connue de cette row
176 // On rajoute une entrée
177 if (iColMap == jMap->end()) {
178 /*debug()<<"\33[1;32m[AlephMatrix::rowMapMapCol] col "
179 <<col<<" inconue, m_addValue_idx="
180 <<m_addValue_idx<<"\33[0m";*/
181 jMap->insert(std::make_pair(col, m_addValue_idx));
182 updateKnownRowCol(row, col, val);
183 return;
184 }
185 // Sinon on ajoute
186 //debug()<<"\33[1;32m[AlephMatrix::rowMapMapCol] hit\33[0m";
187 //debug()<<"[AlephMatrix::rowMapMapCol] += for ["<<row<<","<<col<<"]="<<val; traceMng()->flush();
188 m_addValue_val[iColMap->second] += val;
189}

237void AlephMatrix::
238setValue(Integer row, Integer col, Real val)
239{
240 // Re-ordering si besoin
241 row = m_kernel->ordering()->swap(row);
242 col = m_kernel->ordering()->swap(col);
243 // Si le kernel a déjà été configuré,
244 // on s'assure que la 'géométrie/support' n'a pas changée entre les résolutions
245 if (m_kernel->configured()) {
246 if ((m_setValue_row[m_setValue_idx] != row) ||
247 (m_setValue_col[m_setValue_idx] != col))
248 throw FatalErrorException("Aleph::setValue", "Row|Col have changed!");
249 m_setValue_row[m_setValue_idx] = row;
250 m_setValue_col[m_setValue_idx] = col;
251 m_setValue_val[m_setValue_idx] = val;
252 }
253 else {
254 m_setValue_row.add(row);
255 m_setValue_col.add(col);
256 m_setValue_val.add(val);
257 }
258 m_setValue_idx += 1;
259}

303void AlephMatrix::
304assemble(void)
305{
306 ItacFunction(AlephMatrix);
307 Timer::Action ta(m_kernel->subDomain(), "AlephMatrix::assemble");
308 // Si le kernel n'est pas initialisé, on ne fait toujours rien
309 if (!m_kernel->isInitialized()) {
310 debug() << "\33[1;32m[AlephMatrix::assemble] Trying to assemble a matrix"
311 << "from an uninitialized kernel!\33[0m";
312 return;
313 }
314 // Si aucun [set|add]Value n'a été perçu, ce n'est pas normal
315 if (m_addValue_idx != 0 && m_setValue_idx != 0)
316 throw FatalErrorException("AlephMatrix::assemble", "Still exclusives [add||set]Value required!");
317 // Si des addValue ont été captés, il faut les 'rejouer'
318 // Attention: pour l'instant les add et les set sont disjoints!
319 if (m_addValue_idx != 0) {
320 debug() << "\33[1;32m[AlephMatrix::assemble] m_addValue_idx!=0\33[0m";
321 // On flush notre index des setValues
322 m_setValue_idx = 0;
323 Timer::Action ta(m_kernel->subDomain(), "Flatenning addValues");
324 debug() << "\t\33[32m[AlephMatrix::assemble] Flatenning addValues size=" << m_addValue_row.size() << "\33[0m";
325 for (Integer k = 0, kMx = m_addValue_row.size(); k < kMx; ++k) {
326 m_setValue_row[k] = m_addValue_row[k];
327 m_setValue_col[k] = m_addValue_col[k];
328 m_setValue_val[k] = m_addValue_val[k];
329 /*debug()<<"\t\33[32m[AlephMatrix::assemble] setValue ("<<m_setValue_row[k]
330 <<","<<m_setValue_col[k]<<")="<<m_setValue_val[k]<<"\33[0m";*/
331 m_setValue_idx += 1;
332 }
333 }
334 // S'il y a des 'autres' et qu'on en fait pas parti, on les informe de l'assemblage
335 if (m_kernel->thereIsOthers() && !m_kernel->isAnOther()) {
336 debug() << "\33[1;32m[AlephMatrix::assemble] On informe les autres kappa que l'on assemble"
337 << "\33[0m";
338 m_kernel->world()->broadcast(UniqueArray<unsigned long>(1, 0x74f253cal).view(), 0);
339 // Et on leur donne l'info du m_setValue_idx
340 m_kernel->world()->broadcast(UniqueArray<Integer>(1, m_setValue_idx).view(), 0);
341 }
342 // On initialise la topologie si cela n'a pas été déjà fait
343 if (!m_kernel->isAnOther()) {
344 debug() << "\33[1;32m[AlephMatrix::assemble] Initializing topology"
345 << "\33[0m";
346 ItacRegion(topology->create, AlephMatrix);
347 m_kernel->topology()->create(m_setValue_idx);
348 }
349 // Si on a pas déjà calculé le nombre d'éléments non nuls par lignes
350 // c'est le moment de le déclencher
351 debug() << "\33[1;32m[AlephMatrix::assemble] Updating row_nb_element"
352 << "\33[0m";
353 if (!m_kernel->topology()->hasSetRowNbElements()) {
354 UniqueArray<Integer> row_nb_element;
355 row_nb_element.resize(m_kernel->topology()->nb_row_rank());
356 row_nb_element.fill(0);
357 // Quand on est pas un Autre, il faut mettre à jour le row_nb_element si cela n'a pas été spécifié lors du matrice->create
358 if (m_kernel->thereIsOthers() && !m_kernel->isAnOther()) {
359 debug() << "\33[1;32m[AlephMatrix::assemble] Kernel's topology has not set its nb_row_elements, now doing it!"
360 << "\33[0m";
361 const Integer row_offset = m_kernel->topology()->part()[m_kernel->rank()];
362 debug() << "\33[1;32m[AlephMatrix::assemble] row_offset=" << row_offset << "\33[0m";
363 debug() << "\33[1;32m[AlephMatrix::assemble] filled, row_nb_element.size=" << row_nb_element.size() << "\33[0m";
364 // On le fait pour l'instant en une passe pour avoir une borne max
365 for (Integer i = 0, iMx = m_setValue_row.size(); i < iMx; ++i)
366 row_nb_element[m_setValue_row.at(i) - row_offset] += 1;
367 }
368 m_kernel->topology()->setRowNbElements(row_nb_element);
369 debug() << "\33[1;32m[AlephMatrix::assemble] done hasSetRowNbElements"
370 << "\33[0m";
371 }
372 // Dans le cas //, le solveur se prépare à récupérer les parties de matrices venant des autres
373 debug() << "\33[1;32m[AlephMatrix::assemble] Récupération des parties de matrices"
374 << "\33[0m";
375 if (m_participating_in_solver && (!m_kernel->configured())) {
376 UniqueArray<Integer> nbValues(m_kernel->size());
377 {
378 ItacRegion(gathered_nb_setValued, AlephMatrix);
379 nbValues.fill(0);
380 for (Integer iCpu = 0; iCpu < m_kernel->size(); ++iCpu) {
381 if (m_kernel->rank() != m_ranks[iCpu])
382 continue;
383 //debug()<<"\33[1;32m[AlephMatrix::assemble] Adding nb_values from iCpu "<<iCpu<<"\33[0m";
384 nbValues[iCpu] = m_kernel->topology()->gathered_nb_setValued(iCpu);
385 }
386 }
387 {
388 ItacRegion(resize, AlephMatrix);
389 m_aleph_matrix_buffer_rows.resize(nbValues);
390 m_aleph_matrix_buffer_cols.resize(nbValues);
391 m_aleph_matrix_buffer_vals.resize(nbValues);
392 }
393 }
394 // Si on est pas en //, on a rien d'autre à faire
395 if (!m_kernel->isParallel())
396 return;
397 // Si je participe à la résolution, je reçois les contributions des autres participants
398 if (m_participating_in_solver) {
399 ItacRegion(iRecv, AlephMatrix);
400 debug() << "\33[1;32m[AlephMatrix::assemble] I am part of the solver, let's iRecv"
401 << "\33[0m";
402 // Si je suis le solveur, je recv le reste des matrices provenant soit des autres coeurs, soit de moi-même
403 for (Integer iCpu = 0; iCpu < m_kernel->size(); ++iCpu) {
404 // Sauf de moi-même
405 if (iCpu == m_kernel->rank())
406 continue;
407 // Sauf de ceux qui ne participent pas
408 if (m_kernel->rank() != m_ranks[iCpu])
409 continue;
410 debug() << "\33[1;32m[AlephMatrix::assemble] "
411 << " recv " << m_kernel->rank()
412 << " <= " << iCpu
413 << " size=" << m_aleph_matrix_buffer_cols[iCpu].size() << "\33[0m";
414 m_aleph_matrix_mpi_data_requests.add(m_kernel->world()->recv(m_aleph_matrix_buffer_vals[iCpu], iCpu, false));
415 // Une fois configuré, nous connaissons tous les (i,j): pas besoin de les renvoyer
416 if (!m_kernel->configured()) {
417 m_aleph_matrix_mpi_data_requests.add(m_kernel->world()->recv(m_aleph_matrix_buffer_rows[iCpu], iCpu, false));
418 m_aleph_matrix_mpi_data_requests.add(m_kernel->world()->recv(m_aleph_matrix_buffer_cols[iCpu], iCpu, false));
419 }
420 }
421 }
422 // Si je suis un rang Arcane qui a des données à envoyer, je le fais
423 if ((m_kernel->rank() != m_ranks[m_kernel->rank()]) && (!m_kernel->isAnOther())) {
424 ItacRegion(iSend, AlephMatrix);
425 debug() << "\33[1;32m[AlephMatrix::assemble]"
426 << " send " << m_kernel->rank()
427 << " => " << m_ranks[m_kernel->rank()]
428 << " for " << m_setValue_val.size() << "\33[0m";
429 m_aleph_matrix_mpi_data_requests.add(m_kernel->world()->send(m_setValue_val, m_ranks[m_kernel->rank()], false));
430 if (!m_kernel->configured()) {
431 debug() << "\33[1;32m[AlephMatrix::assemble] iSend my row to " << m_ranks[m_kernel->rank()] << "\33[0m";
432 m_aleph_matrix_mpi_data_requests.add(m_kernel->world()->send(m_setValue_row, m_ranks[m_kernel->rank()], false));
433 debug() << "\33[1;32m[AlephMatrix::assemble] iSend my col to " << m_ranks[m_kernel->rank()] << "\33[0m";
434 m_aleph_matrix_mpi_data_requests.add(m_kernel->world()->send(m_setValue_col, m_ranks[m_kernel->rank()], false));
435 }
436 }
437}

461void AlephMatrix::
462assemble_waitAndFill(void)
463{
464 ItacFunction(AlephMatrix);
465 Timer::Action ta(m_kernel->subDomain(), "AlephMatrix::assemble_waitAndFill");
466 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill]"
467 << "\33[0m";
468 if (m_kernel->isParallel()) {
469 ItacRegion(Wait, AlephMatrix);
470 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] wait for "
471 << m_aleph_matrix_mpi_data_requests.size() << " Requests"
472 << "\33[0m";
473 m_kernel->world()->waitAllRequests(m_aleph_matrix_mpi_data_requests);
474 m_aleph_matrix_mpi_data_requests.clear();
475 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] clear"
476 << "\33[0m";
477 if (m_participating_in_solver == false) {
478 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] nothing more to do"
479 << "\33[0m";
480 }
481 }
482 // Si je ne participe pas, je ne participe pas
483 if (!m_participating_in_solver)
484 return;
485 // Sinon, on prend le temps de construire la matrice, les autres devraient le faire aussi
486 if (!m_kernel->configured()) {
487 ItacRegion(Create, AlephMatrix);
488 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] solver " << m_index << " create_really"
489 << "\33[0m";
490 create_really();
491 }
492 // Et on enchaîne alors avec le remplissage de la matrice
493 {
494 ItacRegion(Fill, AlephMatrix);
495 if (m_kernel->configured())
496 m_implementation->AlephMatrixSetFilled(false); // Activation de la protection de remplissage
497 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] " << m_index << " fill"
498 << "\33[0m";
499 int* bfr_row_implem;
500 int* bfr_col_implem;
501 double* bfr_val_implem;
502 for (int iCpu = 0; iCpu < m_kernel->size(); ++iCpu) {
503 if (m_kernel->rank() != m_ranks[iCpu])
504 continue;
505 if (iCpu == m_kernel->rank()) {
506 bfr_row_implem = reinterpret_cast<int*>(m_setValue_row.unguardedBasePointer());
507 bfr_col_implem = reinterpret_cast<int*>(m_setValue_col.unguardedBasePointer());
508 bfr_val_implem = reinterpret_cast<double*>(m_setValue_val.unguardedBasePointer());
509 m_implementation->AlephMatrixFill(m_setValue_val.size(),
510 bfr_row_implem,
511 bfr_col_implem,
512 bfr_val_implem);
513 }
514 else {
515 bfr_row_implem = reinterpret_cast<int*>(m_aleph_matrix_buffer_rows[iCpu].unguardedBasePointer());
516 bfr_col_implem = reinterpret_cast<int*>(m_aleph_matrix_buffer_cols[iCpu].unguardedBasePointer());
517 bfr_val_implem = reinterpret_cast<double*>(m_aleph_matrix_buffer_vals[iCpu].unguardedBasePointer());
518 m_implementation->AlephMatrixFill(m_aleph_matrix_buffer_vals[iCpu].size(),
519 bfr_row_implem,
520 bfr_col_implem,
521 bfr_val_implem);
522 }
523 }
524 }
525 { // On déclare alors la matrice comme remplie, et on lance la configuration
526 ItacRegion(Cfg, AlephMatrix);
527 m_implementation->AlephMatrixSetFilled(true); // Désactivation de la protection de remplissage
528 if (!m_kernel->configured()) {
529 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] " << m_index << " MATRIX ASSEMBLE"
530 << "\33[0m";
531 int assrtnd = 0;
532 assrtnd = m_implementation->AlephMatrixAssemble();
533 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] AlephMatrixAssemble=" << assrtnd << "\33[0m";
534 // throw FatalErrorException("AlephMatrix::assemble_waitAndFill", "configuration failed");
535 }
536 }
537 debug() << "\33[1;32m[AlephMatrix::assemble_waitAndFill] done"
538 << "\33[0m";
539}

544void AlephMatrix::
545solve(AlephVector* x,
546 AlephVector* b,
547 Integer& nb_iteration,
548 Real* residual_norm,
549 AlephParams* solver_param,
550 bool async)
551{
552 ItacFunction(AlephMatrix);
553 Timer::Action ta(m_kernel->subDomain(), "AlephMatrix::solve");
554 debug() << "\33[1;32m[AlephMatrix::solve] Queuing solver " << m_index << "\33[0m";
555 m_kernel->postSolver(solver_param, this, x, b);
556 // Si on nous a spécifié le post, on ne déclenche pas le mode synchrone

628void AlephMatrix::
629reassemble(Integer& nb_iteration,
630 Real* residual_norm)
631{
632 ItacFunction(AlephMatrix);
633 Timer::Action ta(m_kernel->subDomain(), "AlephMatrix::reassemble");
634 // Si on est pas en mode parallèle, on en a finit pour le solve
635 if (!m_kernel->isParallel())
636 return;
637 m_aleph_matrix_buffer_n_iteration.resize(1);
638 m_aleph_matrix_buffer_n_iteration[0] = nb_iteration;
639 m_aleph_matrix_buffer_residual_norm.resize(4);
640 m_aleph_matrix_buffer_residual_norm[0] = residual_norm[0];
641 m_aleph_matrix_buffer_residual_norm[1] = residual_norm[1];
642 m_aleph_matrix_buffer_residual_norm[2] = residual_norm[2];
643 m_aleph_matrix_buffer_residual_norm[3] = residual_norm[3];
644 // Il faut recevoir des données
645 if (m_kernel->rank() != m_ranks[m_kernel->rank()] && !m_kernel->isAnOther()) {
646 debug() << "\33[1;32m[AlephMatrix::REassemble] " << m_kernel->rank()
647 << "<=" << m_ranks[m_kernel->rank()] << "\33[0m";
648 m_aleph_matrix_mpi_results_requests.add(m_kernel->world()->recv(m_aleph_matrix_buffer_n_iteration,
649 m_ranks[m_kernel->rank()], false));
650 m_aleph_matrix_mpi_results_requests.add(m_kernel->world()->recv(m_aleph_matrix_buffer_residual_norm,
651 m_ranks[m_kernel->rank()], false));
652 }
653 if (m_participating_in_solver) {
654 debug() << "\33[1;32m[AlephMatrix::REassemble] have participated, should send:"
655 << "\33[0m";
656 for (Integer iCpu = 0; iCpu < m_kernel->size(); ++iCpu) {
657 if (iCpu == m_kernel->rank())
658 continue;
659 if (m_kernel->rank() != m_ranks[iCpu])
660 continue;
661 debug() << "\33[1;32m[AlephMatrix::REassemble] " << m_kernel->rank() << " => " << iCpu << "\33[0m";
662 m_aleph_matrix_mpi_results_requests.add(m_kernel->world()->send(m_aleph_matrix_buffer_n_iteration,
663 iCpu, false));
664 m_aleph_matrix_mpi_results_requests.add(m_kernel->world()->send(m_aleph_matrix_buffer_residual_norm,
665 iCpu, false));
666 }
667 }
