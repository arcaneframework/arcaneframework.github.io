276  ItemInternal* _allocOne(bool& need_alloc)
277  {
278  ItemInternal* new_item = 0;
279  Integer nb_free = m_free_internals.size();
280  Int32 lid = 0;
281  if (nb_free!=0){
282  new_item = m_internals[m_free_internals.back()];
283  m_free_internals.popBack();
284  _setAdded(new_item);
285  lid = new_item->localId();
286  need_alloc = false;
287  }
288  else{
289  Integer nb_free2 = m_free_internals_in_multi_buffer.size();
290  if (nb_free2!=0){
291  new_item = m_free_internals_in_multi_buffer.back();
292  m_free_internals_in_multi_buffer.popBack();
293  }
294  else
295  new_item = m_item_internals_buffer->allocOne();
296  lid = m_internals.size();
297  new_item->setLocalId(lid);
298  m_internals.add(new_item);
299  need_alloc = true;
300  }
301  m_added_items.add(lid);
302  ++m_nb_item;
303  return new_item;
304  }
