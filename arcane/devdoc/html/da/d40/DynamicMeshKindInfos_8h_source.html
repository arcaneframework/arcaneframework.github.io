272 ItemInternal* _allocOne(bool& need_alloc)
273 {
274 ItemInternal* new_item = 0;
275 Integer nb_free = m_free_internals.size();
276 Int32 lid = 0;
277 if (nb_free!=0){
278 new_item = m_internals[m_free_internals.back()];
279 m_free_internals.popBack();
280 _setAdded(new_item);
281 lid = new_item->localId();
282 need_alloc = false;
283 }
284 else{
285 Integer nb_free2 = m_free_internals_in_multi_buffer.size();
286 if (nb_free2!=0){
287 new_item = m_free_internals_in_multi_buffer.back();
288 m_free_internals_in_multi_buffer.popBack();
289 }
290 else
291 new_item = m_item_internals_buffer->allocOne();
292 lid = m_internals.size();
293 new_item->setLocalId(lid);
294 m_internals.add(new_item);
295 _updateItemSharedInfoInternalView();
296 need_alloc = true;
297 }
298 m_added_items.add(lid);
299 ++m_nb_item;
300 return new_item;
301 }
