513 void _internalRealloc(Int64 new_capacity,bool compute_capacity,PodType pod_type)
514 {
515 if (_isSharedNull()){
516 if (new_capacity!=0)
517 _internalAllocate(new_capacity);
518 return;
519 }

807 void _resizeAndCopyView(Span<const T> rhs)
808 {
809 const T* rhs_begin = rhs.data();
810 Int64 rhs_size = rhs.size();
811 const Int64 current_size = m_md->size;
812 T* abegin = m_ptr;
813 // Vérifie que \a rhs n'est pas un élément à l'intérieur de ce tableau
814 if (abegin>=rhs_begin && abegin<(rhs_begin+rhs_size))
815 ArrayMetaData::overlapError(abegin,m_md->size,rhs_begin,rhs_size);
816
817 if (rhs_size > current_size) {
818 this->_internalRealloc(rhs_size, false);
819 // Crée les nouveaux éléments
820 this->_createRange(m_md->size, rhs_size, rhs_begin + current_size);
821 // Copie les éléments déjà existant
822 _copy(rhs_begin);
823 m_md->size = rhs_size;
824 }
825 else{
826 this->_destroyRange(rhs_size,current_size,IsPODType{});
827 m_md->size = rhs_size;
828 _copy(rhs_begin);
