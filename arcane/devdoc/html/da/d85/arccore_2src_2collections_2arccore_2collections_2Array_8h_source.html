550 template<typename PodType>
551 void _internalRealloc(Int64 new_capacity, bool compute_capacity, PodType pod_type, RunQueue* queue = nullptr)
552 {
553 if (_isSharedNull()){
554 if (new_capacity!=0)
555 _internalAllocate(new_capacity,queue);
556 return;
557 }
558
559 Int64 acapacity = new_capacity;
560 if (compute_capacity){
561 acapacity = m_md->capacity;
562 //std::cout << " REALLOC: want=" << wanted_size << " current_capacity=" << capacity << '\n';
563 while (new_capacity>acapacity)
564 acapacity = (acapacity==0) ? 4 : (acapacity + 1 + acapacity / 2);
565 //std::cout << " REALLOC: want=" << wanted_size << " new_capacity=" << capacity << '\n';
566 }
567 // Si la nouvelle capacité est inférieure à la courante,ne fait rien.
568 if (acapacity <= m_md->capacity)

570 _internalReallocate(acapacity, pod_type, queue);
