44 toInteger(Real r)
45 {
46  double v = Convert::toDouble(r);
47  if (v>(double)std::numeric_limits<Integer>::max() || v<(double)std::numeric_limits<Integer>::min())
48  throw BadCastException(A_FUNCINFO,
49  String::format("Invalid conversion from '{0}' to type Integer",v));
50  return (Integer)v;
51 }
