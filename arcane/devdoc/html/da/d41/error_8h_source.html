107struct ParseResult {
109 typedef bool (ParseResult::*BooleanType)() const;
110public:
112 ParseResult() : code_(kParseErrorNone), offset_(0) {}
114 ParseResult(ParseErrorCode code, size_t offset) : code_(code), offset_(offset) {}
115
117 ParseErrorCode Code() const { return code_; }
119 size_t Offset() const { return offset_; }
120
122 operator BooleanType() const { return !IsError() ? &ParseResult::IsError : NULL; }
124 bool IsError() const { return code_ != kParseErrorNone; }
125
126 bool operator==(const ParseResult& that) const { return code_ == that.code_; }
127 bool operator==(ParseErrorCode code) const { return code_ == code; }
128 friend bool operator==(ParseErrorCode code, const ParseResult & err) { return code == err.code_; }
129
130 bool operator!=(const ParseResult& that) const { return !(*this == that); }
131 bool operator!=(ParseErrorCode code) const { return !(*this == code); }
132 friend bool operator!=(ParseErrorCode code, const ParseResult & err) { return err != code; }
133
135 void Clear() { Set(kParseErrorNone); }
137 void Set(ParseErrorCode code, size_t offset = 0) { code_ = code; offset_ = offset; }
138
139private:
140 ParseErrorCode code_;
141 size_t offset_;
142};
