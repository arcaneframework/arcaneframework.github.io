92template <int N>
93class ComplexForLoopRanges
94{
95 public:
96
97 using ArrayBoundsType = ArrayBounds<typename MDDimType<N>::DimType>;
98 using ArrayIndexType = typename ArrayBoundsType::IndexType;
99 using IndexType = ArrayIndexType;
100
101 public:
102
103 ComplexForLoopRanges(ArrayBoundsType lower,ArrayBoundsType extents)
104 : m_lower_bounds(lower.asStdArray()), m_extents(extents){}
105 ComplexForLoopRanges(const SimpleForLoopRanges<N>& bounds)
106 : m_extents(bounds.m_bounds){}
107
108 public:
109
110 template<Int32 I> constexpr Int32 lowerBound() const { return m_lower_bounds[I]; }
111 template<Int32 I> constexpr Int32 upperBound() const { return m_lower_bounds[I]+m_extents.template constExtent<I>(); }
112 constexpr Int64 nbElement() const { return m_extents.nbElement(); }
113 constexpr ArrayIndexType getIndices(Int32 i) const
114 {
115 auto x = m_extents.getIndices(i);
116 x.add(m_lower_bounds);
117 return x;
118 }
119 private:
120
121 ArrayIndexType m_lower_bounds;
122 ArrayBoundsType m_extents;
123};
