91 class ARCANE_THREAD_EXPORT SharedMemoryMessageRequest
92 {
93  public:
94  class SortFunctor
95  {
96  public:
97  explicit SortFunctor(Int32 nb_thread) : m_nb_thread(nb_thread){}
98  bool operator()(SharedMemoryMessageRequest* r1,SharedMemoryMessageRequest* r2) const
99  {
100  if (!r1){
101  if (!r2)
102  return true;
103  else
104  return false;
105  }
106  if (!r2)
107  return true;
108  if (r1->isRecv() && !r2->isRecv())
109  return true;
110  if (!r1->isRecv() && r2->isRecv())
111  return false;
112  Int32 i1 = _getQueueIndex(r1->orig(),r1->dest());
113  Int32 i2 = _getQueueIndex(r2->orig(),r2->dest());
114  if (i1==i2)
115  return r1->id() < r2->id();
116  return i1 < i2;
117  }
118  Int32 _getQueueIndex(MessageRank thread1,MessageRank thread2) const
119  {
120  if (thread1.isNull())
121  ARCANE_FATAL("Null rank for thread1");
122  if (thread2.isNull())
123  ARCANE_FATAL("Null rank for thread2");
124  // TODO: gérer dest()==A_NULL_RANK.
125  return thread1.value() + (thread2.value() * m_nb_thread);
126  }
127  Int32 m_nb_thread;
128  };
129  public:
130  using SubQueue = SharedMemoryMessageQueue::SubQueue;
131  public:
133  SharedMemoryMessageRequest(SubQueue* queue,Int64 request_id,MessageRank orig,
134  MessageRank dest,MessageTag tag,ReceiveBufferInfo buf)
135  : m_queue(queue), m_request_id(request_id), m_is_recv(true)
136  , m_orig(orig), m_dest(dest), m_tag(tag), m_receive_buffer_info(buf)
137  {
138  }
140  SharedMemoryMessageRequest(SubQueue* queue,Int64 request_id,MessageRank orig,
141  MessageRank dest,MessageTag tag,SendBufferInfo buf)
142  : m_queue(queue), m_request_id(request_id), m_is_recv(false)
143  , m_orig(orig), m_dest(dest), m_tag(tag), m_send_buffer_info(buf)
144  {
145  }
146  public:
147  MessageRank orig() { return m_orig; }
148  MessageRank dest() { return m_dest; }
149  MessageTag tag() { return m_tag; }
150  bool isRecv() { return m_is_recv; }

153  SendBufferInfo sendBufferInfo() { return m_send_buffer_info; }
154  ReceiveBufferInfo receiveBufferInfo() { return m_receive_buffer_info; }
155  SharedMemoryMessageQueue::SubQueue* queue() { return m_queue; }
156  void copyFromSender(SharedMemoryMessageRequest* sender);

158  void destroy();
159  ISerializer* recvSerializer() { return m_receive_buffer_info.serializer(); }
160  const ISerializer* sendSerializer() { return m_send_buffer_info.serializer(); }
161  // Dans le cas ou dest()==A_NULL_RANK, positionne une fois le message recu le rang d'origine.
162  void setSource(MessageRank s)
163  {
164  if (isRecv())
165  m_dest = s;
166  }
168  SharedMemoryMessageRequest* matchingSendRequest() { return m_matching_send_request; }
169  void setMatchingSendRequest(SharedMemoryMessageRequest* r) { m_matching_send_request = r; }
