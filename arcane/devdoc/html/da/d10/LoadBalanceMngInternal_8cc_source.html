72template <typename DataType> class ProxyItemVariable
73: public IProxyItemVariable
74{
75 public:
76
77 explicit ProxyItemVariable(IVariable* var, Integer pos = 0)
78 : m_var(var)
79 , m_pos(pos)
80 {
81 }
82 ~ProxyItemVariable() {}
83
84 Real operator[](ItemEnumerator i) const
85 {
86 return static_cast<Real>(m_var[i]);
87 }
88 Integer getPos() const
89 {
90 return m_pos;
91 }
92
93 private:
94
95 ItemVariableScalarRefT<DataType> m_var;
96 Integer m_pos;
97};

106IProxyItemVariable* StoreIProxyItemVariable::
107proxyItemVariableFactory(IVariable* var, Integer pos)
108{
109 if (!var)
110 return &nullProxy;
111 switch (var->dataType()) {
112 case DT_Real:
113 return new ProxyItemVariable<Real>(var, pos);
114 break;
115 case DT_Int32:
116 return new ProxyItemVariable<Int32>(var, pos);
117 break;
118 default:
119 // TODO : throw an exception
120 return &nullProxy;
121 }
122}

149
150void LoadBalanceMngInternal::
151initAccess(IMesh* mesh)
152{
153 if (!mesh)
154 ARCANE_FATAL("Null mesh");
155
156 m_mesh_handle = mesh->handle();
157
158 CriteriaMng& mesh_criterion = m_mesh_criterion[mesh];
159 mesh_criterion.init(mesh);
160 mesh_criterion.defaultMassCriterion(m_default_mass_criterion);
161
162 mesh->traceMng()->info() << "LoadBalanceMngInternal::initAccess(): use_memory=" << mesh_criterion.useMassAsCriterion();
163
164 mesh_criterion.computeCriteria();
165
166 mesh->traceMng()->debug() << "LoadBalanceInternal -- Mesh : " << mesh->name() << " -- initAccess()";
167}
