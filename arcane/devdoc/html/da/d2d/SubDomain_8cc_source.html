373extern "C++" ISubDomain*
374arcaneCreateSubDomain(ISession* session,const SubDomainBuildInfo& sdbi)
375{
376 Ref<IParallelMng> pm = sdbi.parallelMng();
377 Ref<IParallelMng> all_replica_pm = sdbi.allReplicaParallelMng();
378 String case_file_name = sdbi.caseFileName();
379 ByteConstArrayView bytes = sdbi.caseBytes();
380
381 ITraceMng* tm = pm->traceMng();
382 StringBuilder trace_id;
383 trace_id += String::fromNumber(pm->commRank());
384 if (all_replica_pm!=pm){
385 trace_id += ",r";
386 trace_id += pm->replication()->replicationRank();
387 }
388 trace_id += ",";
389 trace_id += platform::getHostName();
390 tm->setTraceId(trace_id.toString());
391
392 auto* sd = new SubDomain(session,pm,all_replica_pm,case_file_name,bytes);
393 sd->build();
394 //GG: l'init se fait par l'appelant
395 //mng->initialize();
396 global_sub_domain = sd;
397 return sd;
398}

460void SubDomain::
461initialize()
462{
463 namespace ax = Arcane::Accelerator;
464 // Initialisation du module parallèle
465 // TODO: a supprimer car plus utile
466 m_parallel_mng->initialize();
467
468 {
469 // Initialise le runner par défaut en fonction des paramètres
470 // donnés par l'utilisateur.
471 IApplication* app = application();
472 ax::AcceleratorRuntimeInitialisationInfo config = app->acceleratorRuntimeInitialisationInfo();
473 // TODO: faire cela ailleurs
474 if (auto v = Convert::Type<Int32>::tryParseFromEnvironment("ARCANE_ACCELERATOR_PARALLELMNG_RANK_FOR_DEVICE",true)){
475 Int32 modulo = v.value();
476 if (modulo==0)
477 modulo = 1;
478 info() << "Use commRank() to choose accelerator device with modulo=" << modulo;
479 Int32 device_rank = m_parallel_mng->commRank() % modulo;
480 config.setDeviceId(ax::DeviceId(device_rank));
481 }
482 m_accelerator_mng->initialize(config);
483 Runner* runner = m_accelerator_mng->defaultRunner();
484 const auto& device_info = runner->deviceInfo();
485 info() << "DeviceInfo: name=" << device_info.name();
486 info() << "DeviceInfo: description=" << device_info.description();
487
488 if (Accelerator::impl::isAcceleratorPolicy(runner->executionPolicy())){
489 m_parallel_mng->_internalApi()->setDefaultRunner(runner);
490 m_all_replica_parallel_mng->_internalApi()->setDefaultRunner(runner);
491 }
492 m_variable_mng->_internalApi()->setAcceleratorMng(m_accelerator_mng);
493 }
494
495 _printCPUAffinity();
496
497 IMainFactory* mf = m_application->mainFactory();
498
499 m_time_loop_mng->build();
500
501 m_has_mesh_service = false;
502
503 // Lit le jeu de données.
504 // Il s'agit uniquement de s'assurer de la validité syntaxique
505 // du jeu de données. Les options ne sont pas lues.
506 // Cela doit être fait avant la création du maillage
507 // car l'instance du maillage dépend du jeu de données
508 ICaseDocument* case_document = nullptr;
509 if (!m_case_bytes.empty()){
510 case_document = m_case_mng->readCaseDocument(m_case_full_file_name,m_case_bytes);
511 if (!case_document)
512 ARCANE_FATAL("Can not read case options");
513 // Ajoute à la configuration celle présente dans le jeu de données.
514 {
515 ConfigurationReader cr(traceMng(),m_configuration.get());
516 cr.addValuesFromXmlNode(case_document->configurationElement(),ConfigurationReader::P_CaseDocument);
517 }
518 }
519
520 properties::readFromConfiguration(m_configuration.get(),*this);
521
522 if (case_document){
523 // Regarde s'il y a un tag <meshes> indiquant que la création et
524 // la lecture du maillage sera gérée par un service
525 XmlNode meshes_elem = case_document->meshesElement();
526 if (!meshes_elem.null()){
527 info() << "Using mesh service to create and allocate meshes";
528 m_has_mesh_service = true;
529 }
530 // TODO: dans ce cas vérifier qu'on n'a pas l'élément 'maillage' historique.
531 }
532 // Créé le maillage par défaut. Il faut le faire avant la création
533 // des services, car ces derniers peuvent en avoir besoin.
534 if (!m_has_mesh_service)
535 if (!m_is_create_default_mesh_v2)
536 m_legacy_mesh_builder->createDefaultMesh();
537
538 ScopedPtrT<IServiceLoader> service_loader(mf->createServiceLoader());
539 service_loader->loadSubDomainServices(this);
540
541 // Le module maître doit toujours être créé avant les autres pour que
542 // ses points d'entrées soient appelés en premier et dernier
543 m_module_master = mf->createModuleMaster(this);
544 m_time_history_mng = mf->createTimeHistoryMng(this);
545
546 service_loader->initializeModuleFactories(this);
547 m_lb_mng = mf->createLoadBalanceMng(this);
548
549 m_property_mng_checkpoint = new PropertyMngCheckpoint(this);
550
551 m_observers.addObserver(this,
552 &SubDomain::_notifyWriteCheckpoint,
553 m_checkpoint_mng->writeObservable());
554 }

559void SubDomain::
560destroy()
561{
562 m_case_mesh_master_service.reset();
563
564 m_property_mng_checkpoint = nullptr;
565
566 m_on_destroy_observable.notifyAllObservers();
567 m_on_destroy_observable.detachAllObservers();
568
569 platform::callDotNETGarbageCollector();
570
571 // Normalement on devrait pouvoir supprimer ce test car il ne devrait plus
572 // rester de références sur des services ou module. Cela est le cas
573 // avec l'implémentation 'coreclr' mais pas avec 'mono'. Du coup on
574 // laisse pour l'instant ce test.
575 // A partir de la version 3.7.8, la plupart des problèmes potentiels dus au GC sont
576 // réglés donc il n'est pas nécessaire de retourner directement. Néanmmoins

811 _doInitialPartitionForMesh(mesh,test_service,true);
812 else
813 info() << "Mesh name=" << mesh->name() << " have cells. Do not use " << test_service;

824void SubDomain::
825_doInitialPartitionForMesh(IMesh* mesh,const String& service_name,bool is_required)
826{
827 info() << "DoInitialPartition. mesh=" << mesh->name() << " service=" << service_name;
828
829 String lib_name = service_name;
830
831 IMeshPartitionerBase* mesh_partitioner_base = nullptr;
832 Ref<IMeshPartitionerBase> mesh_partitioner_base_ref;
833 Ref<IMeshPartitioner> mesh_partitioner_ref;
834
835 ServiceBuilder<IMeshPartitionerBase> sbuilder(this);
836 mesh_partitioner_base_ref = sbuilder.createReference(service_name,mesh,SB_AllowNull);
837 mesh_partitioner_base = mesh_partitioner_base_ref.get();
838
839 if (!mesh_partitioner_base){
840 // Si pas trouvé, recherche avec l'ancienne interface 'IMeshPartitioner' pour des
841 // raisons de compatibilité
842 pwarning() << "No implementation for 'IMeshPartitionerBase' interface found. "
843 << "Searching implementation for legacy 'IMeshPartitioner' interface";
844 ServiceBuilder<IMeshPartitioner> sbuilder_legacy(this);
845 mesh_partitioner_ref = sbuilder_legacy.createReference(service_name,mesh,SB_AllowNull);
846 if (mesh_partitioner_ref.get())
847 mesh_partitioner_base = mesh_partitioner_ref.get();
848 }
849
850 if (!mesh_partitioner_base){
851 // Si pas trouvé, récupère la liste des valeurs possibles et les affiche.
852 StringUniqueArray valid_names;
853 sbuilder.getServicesNames(valid_names);
854 String valid_values = String::join(", ",valid_names);
855 String msg = String::format("The specified service for the initial mesh partitionment ({0}) "
856 "is not available (valid_values={1}). This service has to implement "
857 "interface Arcane::IMeshPartitionerBase",
858 lib_name,valid_values);
859 if (is_required){
860 pfatal() << msg;
861 }
862 else{
863 pwarning() << msg;
864 }
865 return;
866 }
867
868 bool is_dynamic = mesh->isDynamic();
869 mesh->modifier()->setDynamic(true);
870 mesh->utilities()->partitionAndExchangeMeshWithReplication(mesh_partitioner_base,true);
871 mesh->modifier()->setDynamic(is_dynamic);
