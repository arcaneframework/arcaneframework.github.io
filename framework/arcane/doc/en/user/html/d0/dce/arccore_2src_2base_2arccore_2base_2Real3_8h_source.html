107/*!
108 * \brief Class managing a 3-dimensional real vector.
109 *
110 * Le vector includes three components \a x, \a y, and \a z, which are of type \b Real.
111 *
112 * \code
113 * Real3 value (1.0,2.3,4.5); // Created a triplet (x=1.0, y=2.3, z=4.5)
114 * cout << value.x; // Prints component x
115 * value.y += 3.2; // Adds 3.2 to component \b y
116 * \endcode
117 *
118 * or equivalently
119 *
120 * \code
121 * Real3 value (1.0,2.3,4.5); // Created a triplet (x=1.0, y=2.3, z=4.5)
122 * cout << value[0]; // Prints component x
123 * value[1] += 3.2; // Adds 3.2 to component \b y
124 * \endcode
125 */
126class ARCCORE_BASE_EXPORT Real3
127: public Real3POD

129 public:

133 : Real3POD()

138 }

399 /*!
400 * \brief Comparison operator.
401 *
402 * This operator allows sorting Real3 for example
403 * in std::set
404 */
405 friend constexpr ARCCORE_HOST_DEVICE bool operator<(Real3 v1, Real3 v2)
406 {
407 if (v1.x == v2.x) {
408 if (v1.y == v2.y)
409 return v1.z < v2.z;
410 else
411 return v1.y < v2.y;
412 }
413 return (v1.x < v2.x);
414 }

423 friend std::istream& operator>>(std::istream& i, Real3& t)
424 {
425 return t.assign(i);
426 }

428 /*!
429 * \brief Compares the current instance component by component to b.
430 *
431 * \retval true if this.x==b.x and this.y==b.y and this.z==b.z.
432 * \retval false otherwise.
433 */
434 constexpr ARCCORE_HOST_DEVICE bool operator==(Real3 b) const
435 {
436 return _eq(x, b.x) && _eq(y, b.y) && _eq(z, b.z);
