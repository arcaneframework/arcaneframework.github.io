144/*!
145 * \brief Template class for converting a type.
146 *
147 * Currently, this is only available via a specialization
148 * for the types 'Int32', 'Int64', and 'Real'.
149 */
150template <typename T>
151class Type;
152
153template <typename T>
154class ScalarType
155{
156 public:
157
158 //! Converts \a s to type \a T
159 ARCCORE_BASE_EXPORT static std::optional<T> tryParse(StringView s);
160
161 /*!
162 * \brief Converts \a s to type \a T.
163 *
164 * If \a s.empty() is true, then it returns \a default_value.
165 */
166 static std::optional<T>
167 tryParseIfNotEmpty(StringView s, const T& default_value)
168 {
169 return (s.empty()) ? default_value : tryParse(s);
170 }
171
172 /*!
173 * \brief Converts the value of the environment variable \a s to type \a T.
174 *
175 * If platform::getEnvironmentVariable(s) is null, return std::nullopt.
176 * Otherwise, it returns this value converted to type \a T. If the conversion
177 * is not possible, it returns std::nullopt if \a throw_if_invalid is \a false or
178 * throws an exception if it is \a true.
179 */
180 ARCCORE_BASE_EXPORT static std::optional<T>
181 tryParseFromEnvironment(StringView s, bool throw_if_invalid);
182};
183
184//! Specialization for scalar types
185template <> class Type<Int64> : public ScalarType<Int64>
186{};
187template <> class Type<Int32> : public ScalarType<Int32>
188{};
189template <> class Type<Real> : public ScalarType<Real>
190{};
