51/*!
52 * \brief Iterator over Arccore array classes.
53 *
54 * This iterator is used for Array, ArrayView, and ConstArrayView classes.
55 *
56 * It is of type std::random_access_iterator_tag.
57 */
58template <typename Iterator_>
59class ArrayIterator
60{
61 private:
62
63 // For the case where C++14 is not supported.
64 template <bool B, class XX = void>
65 using Iterator_enable_if_t = typename std::enable_if<B, XX>::type;
66
67 protected:
68
69 Iterator_ m_ptr;
70
71 using TraitsType_ = Arcane::Impl::ArrayIteratorTraits<Iterator_>;
72
73 public:
74
75 //typedef typename std::random_access_iterator_tag iterator_category;
76 //using iterator_category = std::random_access_iterator_tag;
77 using value_type = TraitsType_::value_type;
78 using difference_type = TraitsType_::difference_type;
79 using reference = TraitsType_::reference;
80 using pointer = TraitsType_::pointer;
81
82 public:
83
84 constexpr ARCCORE_HOST_DEVICE ArrayIterator() noexcept : m_ptr(Iterator_()) {}
85
86 constexpr ARCCORE_HOST_DEVICE explicit ArrayIterator(const Iterator_& i) noexcept
87 : m_ptr(i) {}
88
89 // Allow iterator to const_iterator conversion
90 template <typename X, typename = Iterator_enable_if_t<std::is_same<X, value_type*>::value>>
91 constexpr ARCCORE_HOST_DEVICE ArrayIterator(const ArrayIterator<X>& iter) noexcept
92 : m_ptr(iter.base()) {}
93
94 // Forward iterator requirements
95 constexpr ARCCORE_HOST_DEVICE reference operator*() const noexcept { return *m_ptr; }
96 constexpr ARCCORE_HOST_DEVICE pointer operator->() const noexcept { return m_ptr; }
97 constexpr ARCCORE_HOST_DEVICE ArrayIterator& operator++() noexcept
98 {
99 ++m_ptr;
100 return *this;
101 }
102 constexpr ARCCORE_HOST_DEVICE ArrayIterator operator++(int) noexcept { return ArrayIterator(m_ptr++); }
103
104 // Bidirectional iterator requirements
105 constexpr ARCCORE_HOST_DEVICE ArrayIterator& operator--() noexcept
106 {
107 --m_ptr;
108 return *this;
109 }
110 constexpr ARCCORE_HOST_DEVICE ArrayIterator operator--(int) noexcept { return ArrayIterator(m_ptr--); }
111
112 // Random access iterator requirements
113 constexpr ARCCORE_HOST_DEVICE reference operator[](difference_type n) const noexcept { return m_ptr[n]; }
114 constexpr ARCCORE_HOST_DEVICE ArrayIterator& operator+=(difference_type n) noexcept
115 {
116 m_ptr += n;
117 return *this;
118 }
119 constexpr ARCCORE_HOST_DEVICE ArrayIterator operator+(difference_type n) const noexcept { return ArrayIterator(m_ptr + n); }
120 constexpr ARCCORE_HOST_DEVICE ArrayIterator& operator-=(difference_type n) noexcept
121 {
122 m_ptr -= n;
123 return *this;
124 }
125 constexpr ARCCORE_HOST_DEVICE ArrayIterator operator-(difference_type n) const noexcept { return ArrayIterator(m_ptr - n); }
126
127 constexpr ARCCORE_HOST_DEVICE const Iterator_& base() const noexcept { return m_ptr; }
128};
