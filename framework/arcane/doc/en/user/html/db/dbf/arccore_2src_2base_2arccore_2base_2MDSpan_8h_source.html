30/*!
31 * \brief Base class for multidimensional views.
32 *
33 * This class is inspired by the std::mdspan class currently being defined
34 * (see http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2021/p0009r12.html)
35 *
36 * This class is used to manage views on arrays such as
37 * NumArray. The methods of this class are accessible on the accelerator.
38 *
39 * For more information, refer to the page \ref arcanedoc_core_types_numarray.

282 /*!
283 * \brief Returns a dimension (N-1) view starting from index element \a i.
284 *
285 * For example:
286 * \code
287 * MDSpan<Real, MDDim3> span3 = ...;
288 * MDSpan<Real, MDDim2> sliced_span = span3.slice(5);
289 * // sliced_span(i,i) <=> span3(5,i,j);
290 * \endcode
291 *
292 * \warning This is only valid if \a LayoutPolicy is \a RightLayout.
293 */
294 ARCCORE_HOST_DEVICE MDSpan<DataType, RemovedFirstExtentsType, LayoutPolicy>
295 slice(ExtentIndexType i) const requires(Extents::rank() >= 2 && std::is_base_of_v<RightLayout, LayoutPolicy>)
296 {
297 auto new_extents = m_extents.extents().removeFirstExtent().dynamicExtents();
298 std::array<ExtentIndexType, ExtentsType::rank()> indexes = {};
299 indexes[0] = i;
300 DataType* base_ptr = this->ptrAt(MDIndexType(indexes));
301 return MDSpan<DataType, RemovedFirstExtentsType, LayoutPolicy>(base_ptr, new_extents);
302 }
