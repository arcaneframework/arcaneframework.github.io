28/*!
29 * \internal
30 * \brief Unicode string dictionary.
31 *
32 * Maintains a list of (key, value) pairs allowing one string to be associated with another. This type of dictionary is
33 * used, for example, for translations, where the key is the language and the value is the corresponding translation.
34 */
35class ARCANE_UTILS_EXPORT StringDictionary
36{
37 private:
38
39 class Impl; //!< Implementation
40 public:
41
42 //! Constructs a dictionary
43 StringDictionary();
44
45 //! Constructs a dictionary
46 StringDictionary(const StringDictionary& rhs);
47 ~StringDictionary(); //!< Releases resources
48
49 public:
50
51 /*! \brief Adds the (key, value) pair to the dictionary.
52 *
53 * If a value already exists for \a key, it is replaced by
54 * the new one.
55 */
56 void add(const String& key, const String& value);
57
58 /*! \brief Removes the value associated with \a key.
59 *
60 * If no value was associated with \a key, nothing happens.
61 * \return the removed value if one exists.
62 */
63 String remove(const String& key);
64
65 /*! \brief Returns the value associated with \a key.
66 *
67 * If no value is associated with \a key, the null string is returned.
68 * It is not possible to distinguish between a value
69 * corresponding to the null string and an unfound value unless
70 * \a throw_exception is true, in which case an exception is thrown
71 * if no value corresponding to \a key exists.
72 */
73 String find(const String& key, bool throw_exception = false) const;
74
75 //! Fills \a keys and \a values with the corresponding values from the dictionary
76 void fill(StringList& param_names, StringList& values) const;
77
78 private:
79
80 Impl* m_p; //!< Implementation
81};
