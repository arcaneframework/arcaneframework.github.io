39class ARCCORE_TRACE_EXPORT TraceAccessor
40{
41 public:
42
43 // NOTE: The 'default' versions of the constructors,

55
56 public:

60
61 //! Flow for an information message

63
64 /*! \brief Flow for a parallel information message.
65 *
66 * Unlike info(), all processors write this
67 * message to standard output.
68 */
69 TraceMessage pinfo() const;
70
71 //! Flow for an information message of a given category
72 TraceMessage info(char category) const;
73
74 //! Flow for a parallel information message of a given category
75 TraceMessage pinfo(char category) const;
76
77 /*!
78 * \brief Flow for an information message.
79 *

81 */
82 TraceMessage info(bool v) const;
83
84 //! Flow for a warning message
85 TraceMessage warning() const;
86
87 /*! Flow for a parallel warning message
88 *

90 */
91 TraceMessage pwarning() const;
92
93 //! Flow for an error message
94 TraceMessage error() const;
95
96 /*! Flow for a parallel error message
97 *
98 * Unlike error(), only the master processor writes this message.

100 TraceMessage perror() const;

102 //! Flow for a log message

106 TraceMessage plog() const;
107
108 //! Flow for a log message preceded by the date

110
111 //! Flow for a fatal error message
112 TraceMessage fatal() const;
113
114 //! Flow for a parallel fatal error message
115 TraceMessage pfatal() const;
116
117#ifdef ARCCORE_DEBUG
118 //! Flow for a debug message
119 TraceMessageDbg debug(Trace::eDebugLevel = Trace::Medium) const;
120#else
121 //! Flow for a debug message
122 TraceMessageDbg debug(Trace::eDebugLevel = Trace::Medium) const
123 {
124 return TraceMessageDbg();
125 }
126#endif
127
128 //! Debug level of the configuration file
129 Trace::eDebugLevel configDbgLevel() const;
130
131 //! Flow for an information message of a given level
132 TraceMessage info(Int32 verbose_level) const;
133
134 //! Flow for an information message with the local information level of this instance.
135 TraceMessage linfo() const
136 {
137 return info(m_local_verbose_level);
138 }
139
140 //! Flow for an information message with the local information level of this instance.
141 TraceMessage linfo(Int32 relative_level) const
142 {
143 return info(m_local_verbose_level + relative_level);
144 }
145
146 void fatalMessage [[noreturn]] (const StandaloneTraceMessage& o) const;
147
148 protected:
149
150 void _setLocalVerboseLevel(Int32 v) { m_local_verbose_level = v; }
151 Int32 _localVerboseLevel() const { return m_local_verbose_level; }
152
153 private:
154
155 Ref<ITraceMng> m_trace;
156 Int32 m_local_verbose_level;
