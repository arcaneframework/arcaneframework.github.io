109/*!
110 * \ingroup ArcaneCartesianMesh
111 * \brief Info on the faces of a specific direction X, Y, or Z
112 * of a structured mesh.
113 */
114class ARCANE_CARTESIANMESH_EXPORT FaceDirectionMng
115{
116 friend CartesianMeshImpl;
117 friend CartesianMeshPatch;
118 class Impl;
119
120 private:
121
122 using ItemDirectionInfo = impl::CartesianItemDirectionInfo;
123
124 public:
125
126 /*!
127 * \brief Creates an empty instance.
128 *
129 * The instance is not valid until _internalInit() has been called.
130 */
131 FaceDirectionMng();
132
133 public:
134
135 //! Direction face corresponding to face \a f.
136 DirFace face(Face f) const
137 {
138 return _face(f.localId());
139 }
140 //! Direction face corresponding to face \a f.
141 DirFace face(FaceLocalId f) const
142 {
143 return _face(f.localId());
144 }
145
146 //! Direction face corresponding to face \a f.
147 ARCCORE_HOST_DEVICE DirFaceLocalId dirFaceId(FaceLocalId f) const
148 {
149 return _dirFaceId(f);
150 }
151
152 //! Group of all faces in the direction.
153 FaceGroup allFaces() const;
154
155 /*!
156 * \brief Group of all overlap faces in the direction.
157 *
158 * These are all faces that have two overlap cells around them.
159 *
160 * 0 1 2 3 4
161 * ┌───┬──┬──┬──┬──┐
162 * │ │ │ │ │ │
163 * │ ├──┼──┼──┼──┤
164 * │ │ │ │ │ │
165 * └───┴──┴──┴──┴──┘
166 *
167 * 0 : level -1
168 * 1 and 2 : Overlap cells (overlapCells)
169 * 3 : Outer cells (outerCells)
170 * 4 : Inner cells (innerCells)
171 *
172 * The layer of overlap cells refers to the layer of cells of the same
173 * level around the patch. These cells may belong to one or more
174 * patches.
175 */
176 FaceGroup overlapFaces() const;
177
178 /*!
179 * \brief Group of all faces within the patch in the direction.
180 *
181 * These are all faces that do not have two overlap cells.
182 * (`innerFaces() + outerFaces()` or simply `!overlapFaces()`)
183 *
184 * \warning Faces at the domain boundary (thus having only one
185 * "outer" cell) are included in this group. Therefore, one must not assume
186 * that there are two cells around every face in this group (for that,
187 * one must stick with the innerFaces() group).
188 */
189 FaceGroup inPatchFaces() const;
190
191 /*!
192 * \brief Group of all internal faces in the direction.
193 *
194 * A face is considered internal if its cell
195 * in front and behind is not null and is not an overlap cell.
196 */
197 FaceGroup innerFaces() const;
198
199 /*!
200 * \brief Group of all external faces in the direction.
201 *
202 * A face is considered external if its cell
203 * in front or behind is an overlap cell or is null (if at the domain
204 * boundary or if there are no layers of overlap cells).
205 *
206 * \note Faces between patches are not duplicated. Therefore, some faces
207 * in this group may also be in an outerFaces() of another patch.
208 */
209 FaceGroup outerFaces() const;
210
211 //! Direction face corresponding to face \a f.
212 DirFace operator[](Face f) const
213 {
214 return _face(f.localId());
215 }
216
217 //! Direction face corresponding to face \a f.
218 DirFace operator[](FaceLocalId f) const
219 {
220 return _face(f.localId());
221 }
222
223 //! Direction face corresponding to the face iterator \a iface
224 DirFace operator[](FaceEnumerator iface) const
225 {
226 return _face(iface.itemLocalId());
227 }
228
229 //! Direction value
230 eMeshDirection direction() const
231 {
232 return m_direction;
233 }
234
235 private:
236
237 //! Direction face corresponding to local face number \a local_id
238 DirFace _face(Int32 local_id) const
239 {
240 ItemDirectionInfo d = m_infos_view[local_id];
241 return DirFace(m_cells[d.m_next_lid], m_cells[d.m_previous_lid]);
242 }
243
244 //! Direction face corresponding to local face number \a local_id
245 ARCCORE_HOST_DEVICE DirFaceLocalId _dirFaceId(FaceLocalId local_id) const
246 {
247 ItemDirectionInfo d = m_infos_view[local_id];
248 return DirFaceLocalId(CellLocalId(d.m_next_lid), CellLocalId(d.m_previous_lid));
249 }
250
251 private:
252
253 /*!
254 * \internal
255 * \brief Calculates the information on faces associated with the cells
256 * in the direction \a cell_dm.
257 * Assumes that _internalInit() has been called.
258 */
259 void _internalComputeInfos(const CellDirectionMng& cell_dm,
260 const VariableCellReal3& cells_center,
261 const VariableFaceReal3& faces_center);
262
263 void _internalComputeInfos(const CellDirectionMng& cell_dm);
264
265 /*!
266 * \internal
267 * Initializes the instance.
268 */
269 void _internalInit(ICartesianMesh* cm, eMeshDirection dir, Integer patch_index);
270
271 /*!
272 * \internal
273 * Destroys the resources associated with the instance.
274 */
275 void _internalDestroy();
276
277 /*!
278 * \brief Resizes the container holding the \a ItemDirectionInfo.
279 *
280 * This invalidates current instances of FaceDirectionMng.
281 */
282 void _internalResizeInfos(Int32 new_size);
283
284 void _computeCellInfos(const CellDirectionMng& cell_dm,
285 const VariableCellReal3& cells_center,
286 const VariableFaceReal3& faces_center);
287 void _computeCellInfos() const;
288 bool _hasFace(Cell cell, Int32 face_local_id) const;
289
290 private:
291
292 SmallSpan<ItemDirectionInfo> m_infos_view;
293 CellInfoListView m_cells;
294 eMeshDirection m_direction;
295 Impl* m_p;
296};
