55/*!
56 * \brief Read, write, or read/write view on a 'NumArray'.
57 *
58 * Views work up to rank 4 arrays.
59 */
60template <typename Accessor, typename Extents, typename LayoutType>
61class NumArrayView
62: public NumArrayViewBase
63{
64 public:
65
66 using DataType = typename Accessor::ValueType;
67 using SpanType = MDSpan<DataType, Extents, LayoutType>;
68 using AccessorReturnType = typename Accessor::AccessorReturnType;
69
70 public:
71
72 NumArrayView(const ViewBuildInfo& command, SpanType v)
73 : NumArrayViewBase(command, Arccore::asBytes(v.to1DSpan()))
74 , m_values(v)
75 {}

155/*!
156 * \brief Write view.
157 */
158template <typename DataType, typename Extents, typename LayoutPolicy> auto
159viewOut(const ViewBuildInfo& command, NumArray<DataType, Extents, LayoutPolicy>& var)
160{
161 using Accessor = DataViewSetter<DataType>;
162 return NumArrayView<Accessor, Extents, LayoutPolicy>(command, var.mdspan());
163}
