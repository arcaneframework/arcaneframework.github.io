33/*!
34 * \ingroup ArcaneMaterials
35 * \brief Represents a material in a multi-material cell.
36 *
37 * This object represents a material in a multi-material cell.
38 *
39 * There is a special cell, called the null cell, for which
40 * null() is true and which represents an invalid cell. In the
41 * case of the invalid cell, none of the other
42 * class methods should be called, under penalty of causing a crash.
43 *
44 * \warning These meshes are invalidated as soon as the list of meshes of a
45 * material or environment changes. Therefore, one must not
46 * keep a cell of this type between two changes of this list.
47 */
48class MatCell
49: public ComponentCell
50{
51 public:
52
53 ARCCORE_HOST_DEVICE MatCell(const matimpl::ConstituentItemBase& item_base)
54 : ComponentCell(item_base)
55 {
56#ifdef ARCANE_CHECK
57 _checkLevel(item_base, LEVEL_MATERIAL);
58#endif
59 }
60
61 explicit ARCCORE_HOST_DEVICE MatCell(const ComponentCell& item)
62 : MatCell(item.constituentItemBase())
63 {
64 }
65
66 MatCell() = default;
67
68 public:
69
70 //! Environment cell to which this material cell belongs.
71 ARCCORE_HOST_DEVICE inline EnvCell envCell() const;
72
73 //! Associated material
74 IMeshMaterial* material() const { return _material(); }
75
76 //! Associated user material
77 IUserMeshMaterial* userMaterial() const { return _material()->userMaterial(); }
78
79 //! Material identifier
80 ARCCORE_HOST_DEVICE Int32 materialId() const { return componentId(); }
81
82 private:
83
84 IMeshMaterial* _material() const { return static_cast<IMeshMaterial*>(component()); }
85};

90/*!
91 * \ingroup ArcaneMaterials
92 * \brief Arcane cell of an environment.
93 *
94 * Such a cell contains information about the materials
95 * of an environment for a given cell.
96 *
97 * There is a special cell, called the null cell, for which
98 * null() is true and which represents an invalid cell. In the
99 * case of the invalid cell, none of the other
100 * class methods should be called, under penalty of causing a crash.
101 *
102 * \warning These meshes are invalidated as soon as the list of meshes of a
103 * material or environment changes. Therefore, one must not
104 * keep a cell of this type between two changes of this list.
105 */
106class EnvCell
107: public ComponentCell
108{
109 public:
110
111 explicit ARCCORE_HOST_DEVICE EnvCell(const matimpl::ConstituentItemBase& item_base)
112 : ComponentCell(item_base)
113 {
114#ifdef ARCANE_CHECK
115 _checkLevel(item_base, LEVEL_ENVIRONMENT);
116#endif
117 }
118 explicit ARCCORE_HOST_DEVICE EnvCell(const ComponentCell& item)
119 : EnvCell(item.constituentItemBase())
120 {
121 }
122 EnvCell() = default;
123
124 public:
125
126 // Number of environment materials present in the cell
127 ARCCORE_HOST_DEVICE Int32 nbMaterial() const { return nbSubItem(); }
128
129 //! Cell containing information about all environments
130 ARCCORE_HOST_DEVICE inline AllEnvCell allEnvCell() const;
131
132 //! i-th material cell of this cell
133 ARCCORE_HOST_DEVICE inline MatCell cell(Integer i) const { return _subItemBase(i); }
134
135 //! Associated environment
136 IMeshEnvironment* environment() const { return _environment(); }
137
138 //! Environment identifier
139 ARCCORE_HOST_DEVICE Int32 environmentId() const { return componentId(); }
140
141 //! Enumerator over the material cells of this cell
142 ARCCORE_HOST_DEVICE CellMatCellEnumerator subMatItems() const
143 {
144 return CellMatCellEnumerator(*this);
145 }
146
147 private:
148
149 IMeshEnvironment* _environment() const { return static_cast<IMeshEnvironment*>(component()); }
150};

155/*!
156 * \ingroup ArcaneMaterials
157 * \brief Arcane cell with material and environment information.
158 *
159 * Such a cell contains information about the environments
160 * for a given cell. It allows, for example, knowing the number
161 * of environments and for each the list of materials.
162 *
163 * \warning These meshes are invalidated as soon as the list of meshes of a
164 * material or environment changes. Therefore, one must not
165 * keep a cell of this type between two changes of this list.
166 */
167class AllEnvCell
168: public ComponentCell
169{
170 public:
171
172 explicit ARCCORE_HOST_DEVICE AllEnvCell(const matimpl::ConstituentItemBase& item_base)
173 : ComponentCell(item_base)
174 {
175#if defined(ARCANE_CHECK)
176 _checkLevel(item_base, LEVEL_ALLENVIRONMENT);
177#endif
178 }
179
180 explicit ARCCORE_HOST_DEVICE AllEnvCell(const ComponentCell& item)
181 : AllEnvCell(item.constituentItemBase())
182 {
183 }
184
185 AllEnvCell() = default;
186
187 public:
188
189 //! Number of environments present in the cell
190 ARCCORE_HOST_DEVICE Int32 nbEnvironment() const { return nbSubItem(); }
191
192 //! i-th environment cell
193 EnvCell cell(Int32 i) const { return EnvCell(_subItemBase(i)); }
194
195 //! Enumerator over the environment cells of this cell
196 ARCCORE_HOST_DEVICE CellEnvCellEnumerator subEnvItems() const
197 {
198 return CellEnvCellEnumerator(*this);
199 }
200};
