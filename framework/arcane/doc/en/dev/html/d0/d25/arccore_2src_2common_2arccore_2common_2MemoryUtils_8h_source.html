198template <typename DataType> inline void
199copy(Span<DataType> destination, Span<const DataType> source,
200 const RunQueue* queue = nullptr)
201{
202 ConstMemoryView input(asBytes(source));
203 MutableMemoryView output(asWritableBytes(destination));
204 copy(output, input, queue);
205}
