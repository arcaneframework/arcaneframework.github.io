31 // To indicate that Span2<T>::view() returns an Array2View
32 // and Span2<const T>::view() returns a ConstArray2View.

34 class View2TypeT
35 {
36 public:

39 };
40 template <typename T>
41 class View2TypeT<const T>
42 {
43 public:
44
