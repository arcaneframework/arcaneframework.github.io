111

117 }
118
119 public:
120
121 void resize(Int64 new_size) { BaseClass::_resizeNoInit(new_size); }
122 Span<DataType> to1DSpan() { return BaseClass::span(); }
123 Span<const DataType> to1DSpan() const { return BaseClass::constSpan(); }
124 Span<std::byte> bytes() { return asWritableBytes(BaseClass::span()); }
125 Span<const std::byte> bytes() const { return asBytes(BaseClass::constSpan()); }
126 void swap(NumArrayContainer<DataType>& rhs)
127 {
128 BaseClass::_swap(rhs);

130 }

133 //TODO: deprecate by mid 2026

135 eMemoryResource memoryResource() const { return m_memory_ressource; }
136 void copyInitializerList(std::initializer_list<DataType> alist)
137 {
138 Span<DataType> s = to1DSpan();
139 Int64 s1 = s.size();
140 Int32 index = 0;
141 for (auto x : alist) {
142 s[index] = x;
143 ++index;
144 // Ensures no overflow
145 if (index >= s1)
146 break;
147 }
148 }

168
171 void fill(const DataType& v, const RunQueue* queue)
172 {
173 Span<DataType> destination = to1DSpan();
174 NumArrayBaseCommon::_memoryAwareFill(asWritableBytes(destination), destination.size(), &v, _typeSize(), queue);
175 }
176
177 private:
178
179 void _memoryAwareCopy(const Span<const DataType>& v, eMemoryResource input_ressource, const RunQueue* queue)
180 {
181 NumArrayBaseCommon::_memoryAwareCopy(asBytes(v), input_ressource,
182 asWritableBytes(to1DSpan()), m_memory_ressource, queue);
183 }
184 void _resizeAndCopy(const ThatClass& v)
185 {
186 this->_resizeNoInit(v.to1DSpan().size());
187 _memoryAwareCopy(v, v.memoryRessource(), nullptr);
188 }
189
190 private:
191
