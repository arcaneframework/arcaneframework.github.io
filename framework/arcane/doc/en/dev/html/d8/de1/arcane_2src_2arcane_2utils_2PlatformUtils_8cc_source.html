254extern "C++" ARCANE_UTILS_EXPORT void platform::
255resetAlarmTimer(Integer nb_second)
256{
257#ifdef ARCANE_OS_UNIX
258 struct itimerval time_val;
259 struct itimerval otime_val;
260 time_val.it_value.tv_sec = nb_second;
261 time_val.it_value.tv_usec = 0;
262 time_val.it_interval.tv_sec = 0;
263 time_val.it_interval.tv_usec = 0;
264 // Use virtual time and not real time.
265 // This allows temporarily suspending a job (for example
266 // to fix system issues) without triggering the alarm.
267 int r = setitimer(ITIMER_VIRTUAL, &time_val, &otime_val);
268 if (r != 0)
269 cout << "** ERROR in setitimer r=" << r << '\n';
270#endif
271}

296 template <typename ByteType> bool
297 _readAllFile(StringView filename, bool is_binary, Array<ByteType>& out_bytes)
298 {
299 using namespace std;
300 long unsigned int file_length = platform::getFileLength(filename);
301 if (file_length == 0) {
302 //cerr << "** FAIL LENGTH\n";
303 return true;
304 }
305 ifstream ifile;
306 ios::openmode mode = ios::in;
307 if (is_binary)
308 mode |= ios::binary;
309 ifile.open(filename.toStdStringView().data());
310 if (ifile.fail()) {
311 //cerr << "** FAIL OPEN\n";
312 return true;
313 }
314 out_bytes.resize(file_length);
315 ifile.read((char*)(out_bytes.data()), file_length);
316 if (ifile.bad()) {
317 // cerr << "** BAD READ\n";
318 return true;
319 }
320 // It is possible that the number of bytes read is less
321 // than the file length, especially on Windows with text files
322 // and carriage return conversion. Therefore, it is necessary to resize
323 // the bytes to the correct length.
324 size_t nb_read = ifile.gcount();
325 out_bytes.resize(nb_read);
326 //cerr << "** READ " << file_length << " bytes " << (const char*)(bytes.begin()) << "\n";
327 return false;
328 }
329} // namespace

368extern "C++" String platform::
369getExeFullPath()
370{
371 String full_path;
372#if defined(ARCANE_OS_LINUX)
373 char* buf = ::realpath("/proc/self/exe", nullptr);
374 if (buf) {
375 full_path = StringView(buf);
376 ::free(buf);
377 }
378#elif defined(ARCANE_OS_WIN32)
379 char buf[2048];
380 int r = GetModuleFileNameA(NULL, buf, 2000);
381 if (r > 0) {
382 full_path = StringView(buf);
383 }
384#elif defined(ARCANE_OS_MACOS)
385 char buf[2048];
386 uint32_t bufSize = 2000;
387 int r = _NSGetExecutablePath(buf, &bufSize);
388 if (r == 0) { // success returns 0
389 full_path = StringView(buf);
390 }
391#else
392#error "platform::getExeFullPath() not implemented for this platform"
393#endif
394 return full_path;
395}

400extern "C++" void platform::
401fillCommandLineArguments(StringList& arg_list)
402{
403 arg_list.clear();
404#if defined(ARCANE_OS_LINUX)
405
406 const int BUFSIZE = 1024;
407 char buffer[BUFSIZE + 1];
408
409 UniqueArray<char> bytes;
410 bytes.reserve(1024);
411
412 {
413 const char* filename = "/proc/self/cmdline";
414 int fd = open(filename, O_RDONLY);
415 if (fd < 0)
416 return;
417 ssize_t nb_read = 0;
418 // TODO: handle interruptions
419 while ((nb_read = read(fd, buffer, BUFSIZE)) > 0) {
420 buffer[BUFSIZE] = '\0';
421 bytes.addRange(Span<const char>(buffer, nb_read));
422 }
423 close(fd);
424 }
425
426 int size = bytes.size();
427 const char* ptr = bytes.data();
428 const char* end = ptr + size;
429 while (ptr < end) {
430 arg_list.add(StringView(ptr));
431 while (*ptr++ && ptr < end)
432 ;
433 }
434#elif defined(ARCANE_OS_WIN32)
435 LPWSTR* w_arg_list = nullptr;
436 int nb_arg = 0;
437
438 w_arg_list = ::CommandLineToArgvW(GetCommandLineW(), &nb_arg);
439 if (!w_arg_list)
440 ARCANE_FATAL("Can not get arguments from command line");
441
442 for (int i = 0; i < nb_arg; i++) {
443 std::wstring_view wstr_view(w_arg_list[i]);
444 String str = StringUtils::convertToArcaneString(wstr_view);
445 arg_list.add(str);
446 }
447
448 ::LocalFree(w_arg_list);
449#elif defined(ARCANE_OS_MACOS)
450 int argc = *_NSGetArgc();
451 char** argv = *_NSGetArgv();
452 for (int i = 0; i < argc; i++) {
453 arg_list.add(StringView(argv[i]));
454 }
455#else
456 ARCANE_THROW(NotImplementedException, "not implemented for this platform");
457#endif
458}

465 String _getDebuggerStack(const char* command)
466 {
467 char filename[4096];
468 long pid = (long)getpid();
469 sprintf(filename, "errlog.%ld", pid);
470 int ret_value = system(command);
471 if (ret_value != 0) {
472 UniqueArray<Byte> bytes;
473 if (!platform::readAllFile(filename, false, bytes))
474 return String(bytes);
475 }
476 return {};
477 }

480extern "C++" ARCANE_UTILS_EXPORT String platform::
481getGDBStack()
482{
483 String result;
484#if defined(ARCANE_OS_LINUX)
485 const size_t cmd_size = 4096;
486 char cmd[cmd_size + 1];
487 //sprintf (cmd, "gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt'", (long)getpid ());
488 //sprintf (cmd, "gdb --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt' --batch", (long)getpid ());
489 long pid = (long)getpid();
490 snprintf(cmd, cmd_size, "gdb --ex 'set debuginfod enabled off' --ex 'attach %ld' --ex 'info threads' --ex 'thread apply all bt full' --batch", pid);
491 result = _getDebuggerStack(cmd);
492#endif
493 return result;
494}

499extern "C++" ARCANE_UTILS_EXPORT String platform::
500getLLDBStack()
501{
502 String result;
503#if defined(ARCANE_OS_LINUX)
504 const size_t cmd_size = 4096;
505 char cmd[cmd_size + 1];
506 long pid = (long)getpid();
507 // The commands 'clrthreads', 'clrstack', and 'dumpstack' require
508 // having installed 'dotnet-sos'.
509 snprintf(cmd, cmd_size, "lldb -p %ld -o 'bt' -o 'bt all' -o 'clrthreads' -o 'clrstack' -o 'dumpstack' --batch", pid);
510 result = _getDebuggerStack(cmd);
511#endif
512 return result;
513}
