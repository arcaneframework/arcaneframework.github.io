56Profiler::delta_type Profiler::
57toc(const std::string&)
58{
59 profile_unit* top = stack.back();
60 stack.pop_back();
61
62 value_type current = Platform::getRealTime();
63 delta_type delta = current - top->begin;
64
65 top->length += delta;
66 root.length = current - root.begin;
67
68 return delta;
69}
