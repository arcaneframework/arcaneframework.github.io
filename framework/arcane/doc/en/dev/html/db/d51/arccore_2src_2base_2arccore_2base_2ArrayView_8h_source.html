389 template <class U>
390 void copy(const U& copy_array)
391 {
392 auto copy_size = copy_array.size();
393 const_pointer copy_begin = copy_array.data();
394 pointer to_ptr = m_ptr;
395 Integer n = m_size;
396 if (copy_size < m_size)
397 n = (Integer)copy_size;
398 for (Integer i = 0; i < n; ++i)
399 to_ptr[i] = copy_begin[i];
400 }
