25void buildFaceGroupSurface(FaceGroup group,
26 GeometryKernel::TriangulationDataStructurePtr surface,
27 Array<Node>& node_array,
28 Array<Face>& face_array,
29 Array<bool>& face_reorient,
30 Real3& normal)
31{
32 surface->clear();
33 face_array.clear();
34 face_array.reserve(group.size()); // at least one per face
35 node_array.clear();
36 normal = Real3(0., 0., 0.);
37
38 NodeMapping nodeMapping(group.mesh(), node_array, *surface);
39
40 ENUMERATE_FACE (iface, group) {
41 const Face& face = *iface;
42 const Real face_factor = ((face.isBoundaryOutside()) ? 1 : -1);
43 switch (face.type()) {
44 case IT_Triangle3:
45 surface->newFace(nodeMapping.getNodeId(face.node(0)),
46 nodeMapping.getNodeId(face.node(1)),
47 nodeMapping.getNodeId(face.node(2)));
48 face_array.add(face);
49 if (face.isOwn())
50 normal += face_factor * nodeMapping.computeNormal(face.node(0), face.node(1), face.node(2));
51 break;
52 case IT_Quad4:
53 surface->newFace(nodeMapping.getNodeId(face.node(0)),
54 nodeMapping.getNodeId(face.node(1)),
55 nodeMapping.getNodeId(face.node(2)));
56 face_array.add(face);
57 if (face.isOwn())
58 normal += face_factor * nodeMapping.computeNormal(face.node(0), face.node(1), face.node(2));
59 surface->newFace(nodeMapping.getNodeId(face.node(2)),
60 nodeMapping.getNodeId(face.node(3)),
61 nodeMapping.getNodeId(face.node(0)));
62 face_array.add(face);
63 if (face.isOwn())
64 normal += face_factor * nodeMapping.computeNormal(face.node(2), face.node(3), face.node(0));
65 break;
66 default:
67 throw NotImplementedException(A_FUNCINFO, "Non Tri3 or Quad4 face type");
68 }
69 }
70
71 // Surface orientation and saving orientation changes
72 GeometryKernel::TriangulationTopoTools topology(*surface);
73 face_reorient.resize(face_array.size());
74 topology.computeMicroTopology(face_reorient.begin(), face_reorient.end());
75}

80void saveSurface(const char* filename, GeometryKernel::TriangulationDataStructure& tr)
81{
82 typedef GeometryKernel::Vector Vector;
83 typedef GeometryKernel::TriangulationFace Face;
84 typedef GeometryKernel::TObjectId TObjectId;
85
86 std::ofstream file;
87 file.open(filename, std::ios::out | std::ios::binary);
88 const int nVertices = tr.numberOfVertices();
89 file.write((const char*)&nVertices, sizeof(int));
90 for (Integer i = 0; i < nVertices; ++i) {
91 const Vector& point = tr.vertex(i).point();
92 file.write((const char*)&point, sizeof(Vector));
93 }
94 const int nFaces = tr.numberOfFaces();
95 file.write((const char*)&nFaces, sizeof(int));
96 for (Integer i = 0; i < nFaces; ++i) {
97 const Face& face = tr.face(i);
98 for (int j = 0; j < 3; ++j) {
99 const TObjectId k = face.vertex(j);
100 file.write((const char*)&k, sizeof(TObjectId));
101 }
102 }
103 file.close();
104}

109void loadSurface(const char* filename, GeometryKernel::TriangulationDataStructure& tr)
110{
111 typedef GeometryKernel::Vector Vector;
112 typedef GeometryKernel::TriangulationFace Face;
113 typedef GeometryKernel::TObjectId TObjectId;
114
115 tr.clear();
116
117 std::ifstream file;
118 file.open(filename, std::ios::in | std::ios::binary);
119 int nVertices;
120 file.read((char*)&nVertices, sizeof(int));
121 for (Integer i = 0; i < nVertices; ++i) {
122 Vector point;
123 file.read((char*)&point, sizeof(Vector));
124 tr.newVertex(point);
125 }
126 int nFaces;
127 file.read((char*)&nFaces, sizeof(int));
128 for (Integer i = 0; i < nFaces; ++i) {
129 TObjectId k[3];
130 for (int j = 0; j < 3; ++j) {
131 file.read((char*)&k[j], sizeof(TObjectId));
132 }
133 tr.newFace(k[0], k[1], k[2]);
134 }
135 file.close();
136}
