227 ARCCORE_HOST_DEVICE MDSpan<DataType, RemovedFirstExtentsType, LayoutPolicy>
228 slice(Int32 i) const requires(Extents::rank() >= 2 && std::is_base_of_v<RightLayout, LayoutPolicy>)
229 {
230 auto new_extents = m_extents.extents().removeFirstExtent().dynamicExtents();
231 std::array<Int32, ExtentsType::rank()> indexes = {};
232 indexes[0] = i;
233 DataType* base_ptr = this->ptrAt(MDIndexType(indexes));
234 return MDSpan<DataType, RemovedFirstExtentsType, LayoutPolicy>(base_ptr, new_extents);
235 }
