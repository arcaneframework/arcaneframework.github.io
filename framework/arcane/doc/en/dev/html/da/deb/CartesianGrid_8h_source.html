32template <typename IdType>
33class CartesianGrid
34{
35 public:
36
38 using IdType3 = IdType[3];
39
41 using CartesianNumberingType = CartesianNumbering<IdType>;
42
44 using CartesianNumberingType3 = CartesianNumberingType[3];
45
46 public:
47
49 // param[in] dimension Dimension of the Cartesian grid (up to 3).
50 CartesianGrid(const IdType3& ncells_dir, Integer dimension)
51 : m_dimension(dimension)
52 {
53
54 for (Integer d(0); d < m_dimension; ++d) {
55 m_ncells_dir[d] = ncells_dir[d];
56 m_nnodes_dir[d] = m_ncells_dir[d] + 1;
57 }
58 m_cart_num_cell.initNumbering(m_ncells_dir, dimension);
59
60 for (Integer d(m_dimension); d < 3; ++d) {
61 m_ncells_dir[d] = 1;
62 m_nnodes_dir[d] = 1;
63 }
64 m_cart_num_node.initNumbering(m_nnodes_dir, dimension);
65
66 // For faces, we will number them according to X, then according to Y and according to Z.
67 // Faces are distinguished by their orientations (dnorm).
68 IdType nfaces_norm_first = 0; // First face number according to the dnorm normal.
69 for (Integer dnorm(0); dnorm < m_dimension; ++dnorm) {
70
71 // Directions orthogonal to the normal.
72 Integer d1 = (dnorm + 1) % 3;
73 Integer d2 = (dnorm + 2) % 3;
74
75 // In the direction of the normal, we have m_ncells_dir[dnorm]+1 faces
76 // and in the directions m_ncells_dir[d] faces
77 m_nfaces_norm_dir[dnorm][dnorm] = m_ncells_dir[dnorm] + 1;
78 m_nfaces_norm_dir[dnorm][d1] = m_ncells_dir[d1];
79 m_nfaces_norm_dir[dnorm][d2] = m_ncells_dir[d2];
80
81 m_cart_num_face[dnorm].initNumbering(m_nfaces_norm_dir[dnorm], dimension, nfaces_norm_first);
82 nfaces_norm_first += m_cart_num_face[dnorm].nbItem();
83 }
84
85 for (Integer dnorm(m_dimension); dnorm < 3; ++dnorm) {
86 m_nfaces_norm_dir[dnorm][0] = 1;
87 m_nfaces_norm_dir[dnorm][1] = 1;
88 m_nfaces_norm_dir[dnorm][2] = 1;
89 }
90 }
91

145 return m_dimension;
146 }
147
148 protected:
149
150 IdType3 m_ncells_dir = { 1, 1, 1 }; // Number of cells per direction.
151 IdType3 m_nnodes_dir = { 1, 1, 1 }; // Number of nodes per direction.
152 IdType3 m_nfaces_norm_dir[3];
153
154 Integer m_dimension = 0;
155

157 CartesianNumberingType m_cart_num_node;
158 CartesianNumberingType3 m_cart_num_face;
159};
