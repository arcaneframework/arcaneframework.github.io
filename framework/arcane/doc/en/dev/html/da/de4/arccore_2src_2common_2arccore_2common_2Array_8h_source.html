751 void _initReference(const ThatClassType& rhs)

757 ++m_md->nb_ref;

760 void _updateReferences() final
761 {
762 // TODO merge with SharedArray2 implementation
763 for (ThatClassType* i = m_prev; i; i = i->m_prev)
764 i->_setMP2(m_ptr, m_md);
765 for (ThatClassType* i = m_next; i; i = i->m_next)
766 i->_setMP2(m_ptr, m_md);
767 }
768
769 Integer _getNbRef() final
770 {
771 // NOTE: to be checked, but when this method is called
772 // there is always only one reference.
773 // TODO merge with SharedArray2 implementation
774 Integer nb_ref = 1;
775 for (ThatClassType* i = m_prev; i; i = i->m_prev)
776 ++nb_ref;
777 for (ThatClassType* i = m_next; i; i = i->m_next)
778 ++nb_ref;
779 return nb_ref;
780 }
781 bool _isUseOwnMetaData() const final
782 {
783 return false;
784 }
785
791 void _addReference(const ThatClassType* new_ref)
792 {
793 ThatClassType* nf = const_cast<ThatClassType*>(new_ref);
794 ThatClassType* prev = nf->m_prev;
795 nf->m_prev = this;
796 m_prev = prev;
797 m_next = nf;
798 if (prev)
799 prev->m_next = this;
800 }
801
802 void _removeReference()
803 {
804 if (m_prev)
805 m_prev->m_next = m_next;
806 if (m_next)
807 m_next->m_prev = m_prev;
808 }
809
810 void _checkFreeMemory()
811 {
812 if (m_md->nb_ref == 0) {
813 this->_destroy();
814 this->_internalDeallocate();
815 }
816 }
817 void _operatorEqual(const ThatClassType& rhs)
818 {
819 if (&rhs != this) {
820 _removeReference();
821 _addReference(&rhs);
822 ++rhs.m_md->nb_ref;
823 --m_md->nb_ref;
824 _checkFreeMemory();
825 this->_setMP2(rhs.m_ptr, rhs.m_md);
826 }
827 }
828
829 private:
830
831 ThatClassType* m_next = nullptr;
832 ThatClassType* m_prev = nullptr;
833
834 private:
