337// Definition of the pragma to indicate iteration independence
338
344
349
350// For definitions, GCC must be last because Clang and ICC define
351// the macro __GNU__
352// For CLANG, there is no equivalent to the ICC pragma ivdep yet.
353// The closest one is:
354// #pragma clang loop vectorize(enable)
355// but it does not force vectorization.
356#ifdef __clang__
357#define ARCANE_PRAGMA_IVDEP_VALUE "clang loop vectorize(enable)"
358#else
359#ifdef __INTEL_COMPILER
360#define ARCANE_PRAGMA_IVDEP_VALUE "ivdep"
361#else
362#ifdef __GNUC__
363#if (__GNUC__ >= 5)
364#define ARCANE_PRAGMA_IVDEP_VALUE "GCC ivdep"
365#endif
366#endif
367#endif
368#endif
369
370#ifdef ARCANE_PRAGMA_IVDEP_VALUE
371#define ARCANE_PRAGMA_IVDEP _Pragma(ARCANE_PRAGMA_IVDEP_VALUE)
372#else
373#define ARCANE_PRAGMA_IVDEP
374#define ARCANE_PRAGMA_IVDEP_VALUE ""
375#endif
