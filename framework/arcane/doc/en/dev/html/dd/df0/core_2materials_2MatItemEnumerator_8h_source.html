113class ARCANE_CORE_EXPORT ComponentCellEnumerator
114{
115 friend class EnumeratorTracer;
116
117 protected:
118
119 explicit ComponentCellEnumerator(const ComponentItemVectorView& v);
120
121 public:
122
123 static ComponentCellEnumerator create(IMeshComponent* component);
124 static ComponentCellEnumerator create(const ComponentItemVector& v);
125 static ComponentCellEnumerator create(ComponentItemVectorView v);
126
127 public:
128
129 void operator++()
130 {
131 ++m_index;
132#ifdef ARCANE_CHECK
133 if (m_index < m_size)
134 _check();
135#endif
136 }
137 bool hasNext() const { return m_index < m_size; }
138
139 ComponentCell operator*() const
140 {
141 return ComponentCell(m_constituent_list_view._constituenItemBase(m_index));
142 }
143
144 Integer index() const { return m_index; }
145 MatVarIndex _varIndex() const { return m_matvar_indexes[m_index]; }
146
147 operator ComponentItemLocalId() const
148 {
149 return ComponentItemLocalId(m_matvar_indexes[m_index]);
150 }
151
152 protected:
153
154 Int32 _varArrayIndex() const { return m_matvar_indexes[m_index].arrayIndex(); }
155 Int32 _varValueIndex() const { return m_matvar_indexes[m_index].valueIndex(); }
156
157 protected:
158
159 void _check() const
160 {
161 MatVarIndex mvi = m_constituent_list_view._matVarIndex(m_index);
162 Int32 i_var_array_index = mvi.arrayIndex();
163 Int32 mv_array_index = _varArrayIndex();
164 if (i_var_array_index != mv_array_index)
165 ARCANE_FATAL("Bad 'var_array_index' in ComponentCell matvar='{0}' registered='{1}' index={2}",
166 mvi, m_matvar_indexes[m_index], m_index);
167 Int32 i_var_value_index = mvi.valueIndex();
168 Int32 mv_value_index = _varValueIndex();
169 if (i_var_value_index != mv_value_index)
170 ARCANE_FATAL("Bad 'var_value_index' for ComponentCell matvar='{0}' registered='{1}' index={2}",
171 mvi, m_matvar_indexes[m_index], m_index);
172 }
173
174 protected:
175
176 Int32 m_index;
177 Int32 m_size;
178 ConstituentItemLocalIdListView m_constituent_list_view;
179 ConstArrayView<MatVarIndex> m_matvar_indexes;
180 IMeshComponent* m_component;
181
182 protected:
183
184 matimpl::ConstituentItemBase _currentConstituentItemBase() const
185 {
186 return m_constituent_list_view._constituenItemBase(m_index);
187 }
188};
