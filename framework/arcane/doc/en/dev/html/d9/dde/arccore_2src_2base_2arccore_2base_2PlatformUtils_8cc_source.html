270 isDirectoryExist(const String& dir_name, bool& can_create)
271 {
272 can_create = true;
273 struct stat dirstat;
274 const char* dirname = dir_name.localstr();
275 int stat_val = ::stat(dirname, &dirstat);
276 if (stat_val == 0) {
277 if (dirstat.st_mode & S_IFDIR)
278 return true;
279 // The file exists but is not a directory
280 can_create = false;
281 return false;
282 }
283 return false;
284 }

290extern "C++" ARCCORE_BASE_EXPORT bool Platform::
291recursiveCreateDirectory(const String& dir_name)
292{
293 //cerr << "** REC CRE DIR " << dir_name << '\n';
294 bool can_create = false;
295 bool is_dir = isDirectoryExist(dir_name, can_create);
296 if (is_dir)
297 return false;
298 if (!can_create)
299 return true;
300 Int64 pos = 0;
301 const char* dir_name_str = dir_name.localstr();
302 for (Int64 i = 0, is = dir_name.length(); i < is; ++i)
303 if (dir_name_str[i] == '/')
304 pos = i;
305 if (pos != 0) {
306 String parent_dir(std::string_view(dir_name_str, pos));
307 bool is_bad = recursiveCreateDirectory(parent_dir);
308 if (is_bad)
309 return true;
310 }
311 //cerr << "** REAL DIR " << dir_name << '\n';
312#ifdef ARCCORE_OS_WIN32
313 int ret = mkdir(dir_name_str);
314#else
315 int ret = mkdir(dir_name_str, S_IRWXU | S_IRWXG);
316#endif
317 if (ret == (-1))
318 return true;
319 return false;
320}

351extern "C++" ARCCORE_BASE_EXPORT String Platform::
352getFileDirName(const String& file_name)
353{
354 // On Windows, checks if there are '/'. In this case, this character is taken as
355 // separator. Otherwise, '\\' is taken.
356 char separator = '/';
357 const char* file_name_str = file_name.localstr();
358#if defined(ARCCORE_OS_WIN32)
359 {
360 bool has_slash = false;
361 for (Int64 i = 0, n = file_name.length(); i < n; ++i)
362 if (file_name_str[i] == '/') {
363 has_slash = true;
364 break;
365 }
366 if (!has_slash)
367 separator = '\\';
368 }
369#endif
370 Int64 pos = 0;
371 for (Int64 i = 0, n = file_name.length(); i < n; ++i)
372 if (file_name_str[i] == separator)
373 pos = i;
374
375 if (pos == 0)
376 return String(".");
377
378 return std::string_view(file_name_str, pos);
379}

492#ifdef ARCCORE_OS_LINUX
493static long _getRSSMemoryLinux()
494{
495 FILE* f = ::fopen("/proc/self/stat", "r");
496 if (!f)
497 return 0;
498
499 // See proc(1), category 'stat'
500 int z_pid;
501 char z_comm[4096];
502 char z_state;
503 int z_ppid;
504 int z_pgrp;
505 int z_session;
506 int z_tty_nr;
507 int z_tpgid;
508 unsigned long z_flags;
509 unsigned long z_minflt;
510 unsigned long z_cminflt;
511 unsigned long z_majflt;
512 unsigned long z_cmajflt;
513 unsigned long z_utime;
514 unsigned long z_stime;
515 long z_cutime;
516 long z_cstime;
517 long z_priority;
518 long z_nice;
519 long z_zero;
520 long z_itrealvalue;
521 long z_starttime;
522 unsigned long z_vsize;
523 long z_rss;
524 int r = fscanf(f, "%d %s %c %d %d %d %d %d"
525 "%lu %lu %lu %lu %lu %lu %lu"
526 "%ld %ld %ld %ld %ld %ld %ld %lu %ld",
527 &z_pid, z_comm, &z_state, &z_ppid, &z_pgrp, &z_session, &z_tty_nr, &z_tpgid,
528 &z_flags, &z_minflt, &z_cminflt, &z_majflt, &z_cmajflt, &z_utime, &z_stime,
529 &z_cutime, &z_cstime, &z_priority, &z_nice, &z_zero, &z_itrealvalue,
530 &z_starttime, &z_vsize, &z_rss);
531 if (r != 24)
532 z_rss = 0;
533
534 ::fclose(f);
535 return z_rss;
536}
537#endif

560extern "C++" ARCCORE_BASE_EXPORT Int64 Platform::
561getCPUTime()
562{
563#if defined(ARCCORE_OS_LINUX)
564 struct timespec ts;
565 if (clock_gettime(CLOCK_PROCESS_CPUTIME_ID, &ts) == 0) {
566 return (ts.tv_sec * 1000000) + (ts.tv_nsec / 1000);
567 }
568 return 1;
569#else
570 // TODO Remove the use of clock() for Win32 and MacOS.
571 static Int64 orig_clock = 0;
572 static clock_t call_clock = 0;
573 clock_t current_clock = ::clock();
574 if (orig_clock == 0 && call_clock == 0) {
575 call_clock = current_clock;
576 }
577 else {
578 if (current_clock < call_clock) {
579 // We have exceeded the max value of clock
580 //cout << " WARNING: CLOCK exceeds INT_MAX: " << call_clock
581 // << " current=" << current_clock << '\n';
582 Int64 v = ARCCORE_INT64_MAX;
583 v *= 2;
584 orig_clock += v;
585 }
586 call_clock = current_clock;
587 }
588 //cout << " INFO: CLOCK orig=" << orig_clock << " call=" << call_clock
589 // << " current=" << current_clock << '\n';
590 return orig_clock + call_clock;
591#endif
592}

610extern "C++" ARCCORE_BASE_EXPORT String Platform::
611timeToHourMinuteSecond(Real t)
612{
613 Int64 s = (Int64)(t);
614 Int64 hour = s / 3600;
615 Int64 remaining_hour = s - (3600 * hour);
616 Int64 minute = remaining_hour / 60;
617 Int64 second = remaining_hour - (60 * minute);
618
619 StringBuilder sb;
620 sb += hour;
621 sb += "h";
622 sb += minute;
623 sb += "m";
624 sb += second;
625 sb += "s";
626
627 return sb.toString();
628}

756{
757#ifdef ARCCORE_OS_UNIX
758 String force_color = Platform::getEnvironmentVariable("ARCCORE_COLORTERM");
759 if (!force_color.null())
760 return true;

762 if (!::isatty(::fileno(stdout)))
763 return false;

766 //std::cout << "CHECK ENV term=" << term << '\n';
767

769 return true;
770
771 if (term == "xterm" || term == "rxvt" || term == "rxvt-unicode") {
772 String color_term = Platform::getEnvironmentVariable("COLORTERM");
773 //std::cout << "CHECK ENV color_term=" << color_term << '\n';
774 if (!color_term.null())
775 return true;
776 }
777#endif
778 return false;
779}

784void Platform::
785platformInitialize(bool enable_fpe)
786{
787 // For now, the only specific initialization depends
788 // on i386 processors. It consists of changing the default value of
789 // the FPU flags to generate an exception
790 // during an arithmetic error (such as division by zero).
791 if (enable_fpe)
792 enableFloatingException(true);
793
794 getCPUTime();
795
796 global_has_color_console = _getHasColorTerminal();
797
798 if (getEnvironmentVariable("ARCCORE_PAUSE_ON_EXCEPTION") == "1")
799 arccoreSetPauseOnException(true);
800 if (getEnvironmentVariable("ARCCORE_PRINT_ON_EXCEPTION") == "1")
801 arccoreCallExplainInExceptionConstructor(true);
802}

833extern "C++" String Platform::
834getLoadedSharedLibraryFullPath(const String& dll_name)
835{
836 String full_path;
837 if (dll_name.null())
838 return full_path;
839#if defined(ARCCORE_OS_LINUX)
840 {
841 std::ifstream ifile("/proc/self/maps");
842 String v;
843 String true_name = "lib" + dll_name + ".so";
844 while (ifile.good()) {
845 ifile >> v;
846 Span<const Byte> vb = v.bytes();
847 if (vb.size() > 0 && vb[0] == '/') {
848 if (v.endsWith(true_name)) {
849 full_path = v;
850 //std::cout << "V='" << v << "'\n";
851 break;
852 }
853 }
854 }
855 }
856#elif defined(ARCCORE_OS_WIN32)
857 HMODULE hModule = GetModuleHandleA(dll_name.localstr());
858 if (!hModule)
859 return full_path;
860 TCHAR dllPath[_MAX_PATH];
861 GetModuleFileName(hModule, dllPath, _MAX_PATH);
862 full_path = StringView(dllPath);
863#elif defined(ARCCORE_OS_MACOS)
864 {
865 String true_name = "lib" + dll_name + ".dylib";
866 uint32_t count = _dyld_image_count();
867 for (uint32_t i = 0; i < count; i++) {
868 const char* image_name = _dyld_get_image_name(i);
869 if (image_name) {
870 String image_path(image_name);
871 if (image_path.endsWith(true_name)) {
872 full_path = image_path;
873 break;
874 }
875 }
876 }
877 }
878#else
879 throw NotSupportedException(A_FUNCINFO);
880//#error "platform::getSymbolFullPath() not implemented for this platform"
881#endif
882 return full_path;
883}
