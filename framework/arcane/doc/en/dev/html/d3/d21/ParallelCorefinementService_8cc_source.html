129 struct NodeComparator
130 {
131 typedef std::pair<Integer, Integer> IntPair;
132 bool operator()(const IntPair& a, const IntPair& b) const
133 {
134 if (a.first == b.first)
135 return a.second < b.second;
136 else
137 return a.first < b.first;
138 }
139 static IntPair order(const IntPair& p)
140 {
141 if (p.first > p.second)
142 return IntPair(p.second, p.first);
143 else
144 return p;
145 }
146 };
147
148 static void surfaceSetup(std::map<NodeComparator::IntPair, Integer>& edges, FaceGroup group, IGeometry* geometry)
149 {
150 typedef NodeComparator::IntPair IntPair;
151 Real3 normal(0., 0., 0.);
152 ENUMERATE_FACE (iface, group) {
153 const Face face = *iface;
154 const NodeVectorView nodes = face.nodes();
155 const Integer nnodes = nodes.size();
156 for (Integer i = 0; i < nnodes; ++i) {
157 const IntPair p = NodeComparator::order(IntPair(nodes[i].localId(), nodes[(i + 1) % nnodes].localId()));
158 edges[p]++;
159 }
160 }
161 }

163 class AddEdges
164 {
165 public:
166
167 void operator()(const NodeComparator::IntPair& p)
168 {
169 const Integer indexA = mapNodes[p.first];
170 const Integer indexB = mapNodes[p.second];
171 if (not indexA and not indexB) {
172 mapNodes[p.first] = mapNodes[p.second] = newRef();
173 }
174 else if (not indexA and indexB) {
175 mapNodes[p.first] = mapNodes[p.second];
176 }
177 else if (indexA and not indexB) {
178 mapNodes[p.second] = mapNodes[p.first];
179 }
180 else if (indexA and indexB) {
181 allRefs[mapNodes[p.second] - 1] = allRefs[mapNodes[p.first] - 1];
182 }
183 }
184
185 Integer connexCount()
186 {
187 std::set<Integer> refSet;
188 for (Integer i = 0; i < (Integer)allRefs.size(); ++i)
189 refSet.insert(*allRefs[i]);
190 return refSet.size();
191 }
192
193 private:
194
195 std::vector<std::shared_ptr<Integer>> allRefs;
196 typedef std::map<Integer, Integer> MapNodes;
197 MapNodes mapNodes;
198
199 Integer newRef()
200 {
201 Integer ref = allRefs.size() + 1; // reserve 0 pour la non-valeur
202 allRefs.push_back(std::shared_ptr<Integer>(new Integer(ref)));
203 return ref;
204 }
205 };

207 static void exchange_items(IMesh* mesh,
208 UniqueArray<std::set<Int32>>& nodes_to_send,
209 UniqueArray<std::set<Int32>>& faces_to_send,
210 UniqueArray<std::set<Int32>>& cells_to_send)
211 {
212 ITraceMng* traceMng = mesh->traceMng();
213 if (mesh == NULL)
214 traceMng->fatal() << "Incompatible Mesh type";
215 IParallelMng* parallel_mng = mesh->parallelMng();
216
217 ScopedPtrT<IParallelExchanger> exchanger(parallel_mng->createExchanger());
218
219 const Integer nbSubDomain = parallel_mng->commSize();
220
221 // Initialization of the data exchanger
222 for (Integer isd = 0; isd < nbSubDomain; ++isd)
223 if (not cells_to_send[isd].empty())
224 exchanger->addSender(isd);
225 exchanger->initializeCommunicationsMessages();
226
227 for (Integer i = 0, ns = exchanger->nbSender(); i < ns; ++i) {
228 ISerializeMessage* sm = exchanger->messageToSend(i);
229 Int32 rank = sm->destRank();
230 ISerializer* s = sm->serializer();
231 const std::set<Int32>& cell_set = cells_to_send[rank];
232 Int32UniqueArray items_to_send(cell_set.size());
233 std::copy(cell_set.begin(), cell_set.end(), items_to_send.begin());
234 mesh->/*modifier()->*/ serializeCells(s, items_to_send);
235 }
236 exchanger->processExchange();
237
238 for (Integer i = 0, ns = exchanger->nbReceiver(); i < ns; ++i) {
239 ISerializeMessage* sm = exchanger->messageToReceive(i);
240 ISerializer* s = sm->serializer();
241 mesh->modifier()->addCells(s);
242 }
243
244 mesh->modifier()->endUpdate(true, false); // recalcul des ghosts sans suppression des anciens
245 }
246
247 Timer migrationTimer;
248 Timer corefinementTimer;
249};

289void ParallelCorefinementService::
290update()
291{
292 // info() << "All groups list : ";
293 // ItemGroupCollection coll = mesh()->groups();
294 // for(ItemGroupCollection::Iterator i=coll.begin(); i !=coll.end(); ++i) {
295 // ItemGroup g = *i;
296 // info() << "\tGroup : " << g.name() << " size=" << g.size();
297 // }
298 m_internal->corefinementTimer.start();
299 m_internal->migrationTimer.start();
300
301 pinfo() << "Surface infos berfore migration "
302 << m_master_group.name() << "(" << m_master_group.size() << ") and "
303 << m_slave_group.name() << "(" << m_slave_group.size() << ")";
304
305 IParallelMng* parallel_mng = subDomain()->parallelMng();
306
307 // 1- Calcul des boites englobantes coté Master
308 // + Slave pour minimiser (et non eviter) de multiples composantes connexes
309 // (non supporté par le coraffinement simple)
310 const Real maxValue = FloatInfo<Real>::maxValue();
311 const VariableNodeReal3& nodesCoordinates = PRIMARYMESH_CAST(mesh())->nodesCoordinates();
312 Real3
313 master_lower_left(maxValue, maxValue, maxValue),
314 master_upper_right(-maxValue, -maxValue, -maxValue),
315 slave_lower_left(maxValue, maxValue, maxValue),
316 slave_upper_right(-maxValue, -maxValue, -maxValue);
317
318 // Not optimized box computation (many nodes are min/max several times)
319 // Master BOX
320 ENUMERATE_FACE (iface, m_master_group) {
321 for (Node node : iface->nodes()) {
322 master_lower_left = math::min(master_lower_left, nodesCoordinates[node]);
323 master_upper_right = math::max(master_upper_right, nodesCoordinates[node]);
324 }
325 }
326 ENUMERATE_FACE (iface, m_slave_group) {
327 for (Node node : iface->nodes()) {
328 master_lower_left = math::min(master_lower_left, nodesCoordinates[node]);
329 master_upper_right = math::max(master_upper_right, nodesCoordinates[node]);
330 }
331 }
332
333 // Slave BOX
334 ENUMERATE_FACE (iface, m_slave_group) {
335 for (Node node : iface->nodes()) {
336 slave_lower_left = math::min(slave_lower_left, nodesCoordinates[node]);
337 slave_upper_right = math::max(slave_upper_right, nodesCoordinates[node]);
338 }
339 }
340
341 // Extend current box
342 master_lower_left.subSame(m_box_tolerance);
343 master_upper_right.addSame(m_box_tolerance);
344 slave_lower_left.subSame(m_box_tolerance);
345 slave_upper_right.addSame(m_box_tolerance);
346 debug(Trace::Medium) << "Local Box sizes :"
347 << " M= " << master_lower_left << " - " << master_upper_right
348 << " S= " << slave_lower_left << " - " << slave_upper_right;
349
350 // 2- migration des faces esclaves en fonction des boites maitres
351 // + migration des faces maitres 'utiles' vers les processeurs esclaves
352 UniqueArray<Real3> myBox;
353 myBox.add(master_lower_left);
354 myBox.add(master_upper_right);
355 myBox.add(slave_lower_left);
356 myBox.add(slave_upper_right);
357
358 const Integer nbSubDomain = subDomain()->nbSubDomain();
359 UniqueArray<Real3> allBoxes(4 * nbSubDomain);
360 parallel_mng->allGather(myBox, allBoxes);
361
362 // NB: ne traite pas les aretes.
363 // exchangers used in slave->master mode then master->slave
364 UniqueArray<std::set<Int32>> nodes_to_send;
365 nodes_to_send.resize(nbSubDomain);
366 UniqueArray<std::set<Int32>> faces_to_send;
367 faces_to_send.resize(nbSubDomain);
368 UniqueArray<std::set<Int32>> cells_to_send;
369 cells_to_send.resize(nbSubDomain);
370
371 // Inclusion calculations can be optimized by already having the local bounding box
372 // If there is no intersection between the local and remote boxes => nothing to do.
373#ifndef NO_USER_WARNING
374#warning "optimisations possibles"
375#endif /* NO_USER_WARNING */
376 for (Integer i_sub_domain = 0; i_sub_domain < nbSubDomain; ++i_sub_domain) {
377 if (i_sub_domain == subDomain()->subDomainId())
378 continue;
379 Internal::Box masterBox(allBoxes[4 * i_sub_domain], allBoxes[4 * i_sub_domain + 1], nodesCoordinates);
380 masterBox.m_trace = traceMng();
381 ENUMERATE_FACE (iface, m_slave_group) {
382 if (masterBox.isInside(*iface)) {
383 faces_to_send[i_sub_domain].insert(iface->localId());
384 Cell boundaryCell = iface->boundaryCell();
385 if (boundaryCell.null())
386 fatal() << "Non boundary face used in co-refinement";
387 cells_to_send[i_sub_domain].insert(boundaryCell.localId());
388 for (Node node : boundaryCell.nodes())
389 nodes_to_send[i_sub_domain].insert(node.localId());
390 }
391 }
392
393 Internal::Box slaveBox(allBoxes[4 * i_sub_domain + 2], allBoxes[4 * i_sub_domain + 3], nodesCoordinates);
394 slaveBox.m_trace = traceMng();
395 ENUMERATE_FACE (iface, m_master_group) {
396 if (masterBox.isInside(*iface)) {
397 faces_to_send[i_sub_domain].insert(iface->localId());
398 Cell boundaryCell = iface->boundaryCell();
399 if (boundaryCell.null())
400 fatal() << "Non boundary face used in co-refinement";
401 cells_to_send[i_sub_domain].insert(boundaryCell.localId());
402 for (Node node : boundaryCell.nodes())
403 nodes_to_send[i_sub_domain].insert(node.localId());
404 }
405 }
406 }
407
408 Internal::exchange_items(mesh(), nodes_to_send, faces_to_send, cells_to_send);
409
410 debug(Trace::Medium) << "Surface infos after migration "
411 << m_master_group.name() << "(" << m_master_group.size() << ") and "
412 << m_slave_group.name() << "(" << m_slave_group.size() << ")";
413
414 m_internal->migrationTimer.stop();
415
416 // 3- Calculating connected components
417 // For now, we check that locally we only have one connected component.
418 // To do this, we check that the boundary of the surfaces forms only a curve
419
420 // Selection of surfaces for local co-refinement
421 FaceGroup slave_coref_group = m_slave_group;
422 FaceGroup master_coref_group = m_master_group;
423
424 typedef Internal::NodeComparator::IntPair IntPair;
425 std::map<IntPair, Integer> slave_edges; // pairs of vertices indexed by their localIds
426 std::map<IntPair, Integer> master_edges; // pairs of vertices indexed by their localIds
427 Internal::surfaceSetup(slave_edges, slave_coref_group, m_geometry);
428 Internal::surfaceSetup(master_edges, master_coref_group, m_geometry);
429
430 {
431 Internal::AddEdges addEdges;
432 for (std::map<IntPair, Integer>::iterator i = slave_edges.begin(); i != slave_edges.end(); ++i)
433 if (i->second == 1)
434 addEdges(i->first);
435 else if (i->second > 2)
436 fatal() << "Not manifold slave surface";
437 if (addEdges.connexCount() > 1)
438 fatal() << "Slave surface is not connex";
439 }
440 {
441 Internal::AddEdges addEdges;
442 for (std::map<IntPair, Integer>::iterator i = master_edges.begin(); i != master_edges.end(); ++i)
443 if (i->second == 1)
444 addEdges(i->first);
445 else if (i->second > 2)
446 fatal() << "Not manifold master surface";
447 if (addEdges.connexCount() > 1)
448 fatal() << "Master surface is not connex";
449 }
450
451 // 4- Calculating co-refinement by connected components
452 // Currently one component per surface
453 // From this step onwards, if the mesh is modified, the co-refinement result is invalidated
454 // because a renumbering of local items is possible
455
456 ISurfaceUtils* surfaceUtils = options()->surfaceUtils();
457 ISurface* masterSurface = surfaceUtils->createSurface();
458 surfaceUtils->setFaceToSurface(masterSurface, master_coref_group);
459 ISurface* slaveSurface = surfaceUtils->createSurface();
460 surfaceUtils->setFaceToSurface(slaveSurface, slave_coref_group);
461
462 FaceFaceContactList coarse_contacts;
463 surfaceUtils->computeSurfaceContact(masterSurface, slaveSurface, coarse_contacts);
464
465 // { // local co-refinement integrity test
466 // Real3 normalA_coref(0,0,0), normalA_local(0,0,0);
467 // Real3 normalB_coref(0,0,0), normalB_local(0,0,0);
468
469 // ENUMERATE_FACE(iface,m_master_group.own())
470 // normalA_local += m_geometry->computeOrientedMeasure(*iface);
471 // ENUMERATE_FACE(iface,m_slave_group.own())
472 // normalB_local += m_geometry->computeOrientedMeasure(*iface);
473
474 // for(Integer i=0;i<coarse_contacts.size(); ++i) {
475 // const ISurfaceUtils::FaceFaceContact & contact = coarse_contacts[i];
476 // if (not contact.faceA.null() and contact.faceA.isOwn())
477 // normalA_coref += contact.normalA;
478 // if (not contact.faceB.null() and contact.faceB.isOwn())
479 // normalB_coref += contact.normalB;
480 // }
481 // pinfo() << "Global co-refinement stats0 : "
482 // << math::normeR3(normalA_coref) << " vs " << math::normeR3(normalA_local) << " ; "
483 // << math::normeR3(normalB_coref) << " vs " << math::normeR3(normalB_local);
484 // }
485
486#ifndef NO_USER_WARNING
487#warning "utiliser un destroySurface ???"
488#endif /* NO_USER_WARNING */
489 delete masterSurface;
490 delete slaveSurface;
491
492 // 5- Filtering contacts by the distance between their 2 faces
493 // and diffusion of useful master faces to the slave side.
494 m_contacts.clear();
495
496 // Registration of refinement needs for void-faces
497 ItemInternalList internals = mesh()->faceFamily()->itemsInternal();
498 // The key of SharedCorefinement refers to the previous internals
499 typedef std::map<Integer, Real3> SharedCorefinement;
500 SharedCorefinement voidface_shared_corefinement;
501 ARCANE_ASSERT((coarse_contacts.empty() == master_coref_group.empty()), ("Incompatible emptiness"));
502
503 if (coarse_contacts.empty()) {
504 ENUMERATE_FACE (iface, m_slave_group.own()) {
505 voidface_shared_corefinement[iface->localId()];
506 }
507 }
508 else {
509 Internal::CheckCloseFaces checkFaces(nodesCoordinates, m_box_tolerance);
510 SharedCorefinement facevoid_shared_corefinement; // for aggregation on the master
511 for (FaceFaceContactList::iterator i = coarse_contacts.begin(); i != coarse_contacts.end(); ++i) {
512 const ISurfaceUtils::FaceFaceContact& contact = *i;
513 const Face master_face = contact.faceA;
514 const Face slave_face = contact.faceB;
515 if (slave_face.null()) {
516 if (master_face.isOwn()) {
517 facevoid_shared_corefinement[master_face.localId()] += contact.normalA;
518 } // else not use since non local
519 }
520 else if (master_face.null()) {
521 if (slave_face.isOwn()) {
522 // This face must be post-processed in parallel
523 voidface_shared_corefinement[slave_face.localId()];
524 } // else not use since non local
525 }
526 else if (master_face.isOwn() or slave_face.isOwn()) {
527 if (checkFaces(master_face, slave_face)) {
528 // Distance compatible with the box selection
529 // Admissible contact (storage + preparation for diffusion)
530 if (master_face.isOwn()) // if !master_face.isOwn() it will receive the info via its owner
531 m_contacts.add(contact);
532 }
533 else {
534 // Completion of co-refinement for non-contact
535 if (master_face.isOwn()) {
536 facevoid_shared_corefinement[master_face.localId()] += contact.normalA;
537 }
538 if (slave_face.isOwn()) {
539 voidface_shared_corefinement[slave_face.localId()];
540 }
541 }
542 }
543 }
544
545 // Integration of aggregated face-voids
546 for (SharedCorefinement::const_iterator i = facevoid_shared_corefinement.begin();
547 i != facevoid_shared_corefinement.end(); ++i) {
548 Face faceA(internals[i->first]);
549 Face faceB; // null
550 ARCANE_ASSERT((faceA.isOwn()), ("Non local facevoid"));
551 ISurfaceUtils::FaceFaceContact contact(faceA, faceB);
552 contact.normalA = i->second;
553 // info() << "Global co-refinement info : facevoid add " << faceA.uniqueId() << " " << contact.normalA;
554 m_contacts.add(contact);
555 }
556 }
557
558 // 6- Serialization of data to slaves; preparation of void-face synthesis
559
560 // Compatibility of data to send
561 // Counts bi-parent sub-faces (faceface)
562 UniqueArray<Integer> info_to_send(nbSubDomain, 0);
563 for (FaceFaceContactList::iterator i = m_contacts.begin(); i != m_contacts.end(); ++i) {
564 const ISurfaceUtils::FaceFaceContact& contact = *i;
565 const Face master_face = contact.faceA;
566 const Face slave_face = contact.faceB;
567 if (not slave_face.null()) {
568 ARCANE_ASSERT((not master_face.null()), ("Integrity error")); // no construction of m_contacts filling
569 if (slave_face.isOwn()) {
570 if (master_face.isOwn()) {
571 // Local processing of void-faces
572 SharedCorefinement::iterator vf_finder = voidface_shared_corefinement.find(slave_face.localId());
573 if (vf_finder != voidface_shared_corefinement.end()) {
574 vf_finder->second += contact.normalB;
575 // info() << "Global co-ref Voidface add1 to slave " << slave_face.uniqueId() << " " << contact.normalB << " from " << master_face.uniqueId() << " " << contact.normalA;
576 }
577 }
578 }
579 else { // has to be sent
580 if (master_face.isOwn())
581 ++info_to_send[slave_face.owner()];
582 }
583 }
584 }
585
586 // List of synthesis messages (emissions / receptions)
587 ISerializeMessageList* messageList = NULL;
588 if (parallel_mng->isParallel())
589 messageList = parallel_mng->createSerializeMessageList();
590
591 // Preparation of reception
592 UniqueArray<Integer> info_to_recv(nbSubDomain, 0);
593 parallel_mng->allToAll(info_to_send, info_to_recv, 1);
594 std::list<std::shared_ptr<SerializeMessage>> recv_messages;
595 for (Integer i = 0; i < nbSubDomain; ++i) {
596 if (info_to_recv[i] > 0) {
597 SerializeMessage* message = new SerializeMessage(parallel_mng->commRank(), i, ISerializeMessage::MT_Recv);
598 recv_messages.push_back(std::shared_ptr<SerializeMessage>(message));
599 messageList->addMessage(message);
600 }
601 }
602
603 // Preparation of emissions
604 UniqueArray<std::shared_ptr<SerializeMessage>> sent_messages(nbSubDomain);
605 for (Integer i = 0; i < nbSubDomain; ++i) {
606 const Integer faceface_info_to_send = info_to_send[i];
607 if (faceface_info_to_send > 0) {
608 SerializeMessage* message = new SerializeMessage(parallel_mng->commRank(), i, ISerializeMessage::MT_Send);
609 sent_messages[i].reset(message);
610 messageList->addMessage(message);
611 SerializeBuffer& sbuf = message->buffer();
612 sbuf.setMode(ISerializer::ModeReserve); // preparatory phase
613 sbuf.reserveInteger(2); // Nb d'item faceface puis voidface
614 sbuf.reserve(DT_Int64, 2 * faceface_info_to_send); // The uids
615 sbuf.reserve(DT_Real, 6 * faceface_info_to_send); // The normals (nothing for voidface)
616 sbuf.reserve(DT_Real, 6 * faceface_info_to_send); // The centers (nothing for voidface)
617 sbuf.allocateBuffer(); // memory allocation
618 sbuf.setMode(ISerializer::ModePut);
619 sbuf.putInteger(faceface_info_to_send);
620 }
621 }
622
623 // Filling outgoing messages
624 for (FaceFaceContactList::iterator i = m_contacts.begin(); i != m_contacts.end(); ++i) {
625 const ISurfaceUtils::FaceFaceContact& contact = *i;
626 const Face master_face = contact.faceA;
627 const Face slave_face = contact.faceB;
628 if (not slave_face.null() and not slave_face.isOwn() and master_face.isOwn()) {
629 std::shared_ptr<SerializeMessage> message = sent_messages[slave_face.owner()];
630 SerializeBuffer& sbuf = message->buffer();
631 // Not possible to put Real3 ??? (even if we can reserve them)
632 sbuf.put(master_face.uniqueId().asInt64());
633 sbuf.putReal(contact.normalA.x);
634 sbuf.putReal(contact.normalA.y);
635 sbuf.putReal(contact.normalA.z);
636 sbuf.putReal(contact.centerA.x);
637 sbuf.putReal(contact.centerA.y);
638 sbuf.putReal(contact.centerA.z);
639 sbuf.put(slave_face.uniqueId().asInt64());
640 sbuf.putReal(contact.normalB.x);
641 sbuf.putReal(contact.normalB.y);
642 sbuf.putReal(contact.normalB.z);
643 sbuf.putReal(contact.centerB.x);
644 sbuf.putReal(contact.centerB.y);
645 sbuf.putReal(contact.centerB.z);
646
647 // pinfo() << "Global co-ref " << parallel_mng->commRank() << " push to "
648 // << slave_face.owner() << " " << master_face.uniqueId() << " - " << slave_face.uniqueId()
649 // << " : " << contact.normalA << " " << contact.normalB;
650 }
651 }
652
653 // Processing communications
654 if (messageList) {
655 messageList->processPendingMessages();
656 messageList->waitMessages(Parallel::WaitAll);
657 delete messageList;
658 messageList = NULL; // Clean destruction
659 }
660
661 // Processing received messages
662 for (std::list<std::shared_ptr<SerializeMessage>>::iterator i = recv_messages.begin(); i != recv_messages.end(); ++i) {
663 std::shared_ptr<SerializeMessage>& message = *i;
664 const Integer origDomainId = message->destRank();
665 ARCANE_ASSERT((origDomainId != subDomain()->subDomainId()), ("Local to local sent"));
666 SerializeBuffer& sbuf = message->buffer();
667 sbuf.setMode(ISerializer::ModeGet);
668 const Integer faceface_count = sbuf.getInteger();
669 ARCANE_ASSERT((faceface_count == info_to_recv[origDomainId]), ("Bad recv faceface count"));
670 Int64UniqueArray uids(2 * faceface_count);
671 Int32UniqueArray lids(2 * faceface_count);
672 sbuf.get(uids);
673 mesh()->faceFamily()->itemsUniqueIdToLocalId(lids, uids, true);
674
675 // Processing face-face pairs
676 for (Integer j = 0; j < faceface_count; ++j) {
677 Face faceA(internals[lids[2 * j]]);
678 Face faceB(internals[lids[2 * j + 1]]);
679 ARCANE_ASSERT((faceB.isOwn()), ("Not local information recieved"));
680 ISurfaceUtils::FaceFaceContact contact(faceA, faceB);
681 contact.normalA.x = sbuf.getReal();
682 contact.normalA.y = sbuf.getReal();
683 contact.normalA.z = sbuf.getReal();
684 contact.centerA.x = sbuf.getReal();
685 contact.centerA.y = sbuf.getReal();
686 contact.centerA.z = sbuf.getReal();
687 contact.normalB.x = sbuf.getReal();
688 contact.normalB.y = sbuf.getReal();
689 contact.normalB.z = sbuf.getReal();
690 contact.centerB.x = sbuf.getReal();
691 contact.centerB.y = sbuf.getReal();
692 contact.centerB.z = sbuf.getReal();
693 m_contacts.add(contact);
694
695 SharedCorefinement::iterator vf_finder = voidface_shared_corefinement.find(faceB.localId());
696 if (vf_finder != voidface_shared_corefinement.end()) {
697 vf_finder->second += contact.normalB;
698 // info() << "Global co-ref Voidface add2 to slave " << faceB.uniqueId() << " " << contact.normalB << " from " << faceA.uniqueId() << " " << contact.normalA;
699 }
700 }
701 }
702
703 // 7- Final processing of void-face pairs
704 for (SharedCorefinement::const_iterator i = voidface_shared_corefinement.begin();
705 i != voidface_shared_corefinement.end(); ++i) {
706 Face faceA; // null
707 Face faceB(internals[i->first]);
708 ARCANE_ASSERT((faceB.isOwn()), ("Non local voidface"));
709 Real3 faceB_normal = m_geometry->computeOrientedMeasure(faceB);
710 Real faceB_area = math::normeR3(faceB_normal);
711 if (math::normeR3(i->second) < faceB_area) {
712 ISurfaceUtils::FaceFaceContact contact(faceA, faceB);
713 contact.normalB = faceB_normal - i->second;
714 // info() << "Global co-refinement info : voidface add " << faceB.uniqueId() << " " << contact.normalB << " " << faceB_normal;
715 m_contacts.add(contact);
716 }
717 }
718
719 // 8- Cleanup of unnecessary replicas
720 // Not implemented and NOTE this can change local numbering
721
722 m_internal->corefinementTimer.stop();
723
724 UniqueArray<Real> times(2);
725 times[0] = m_internal->corefinementTimer.totalTime();
726 times[1] = m_internal->migrationTimer.totalTime();
727 parallel_mng->reduce(Parallel::ReduceMin, times);
728 info() << "Corefinement timers : all=" << times[0] << " migration=" << times[1] << " #domains=" << parallel_mng->commSize();
729}
