148template <typename type> int
149_displayArray(const AbstractArray<type>* obj, const char* type_name)
150{
151 Arccore::Span<const type> view = *obj;
152 const _ArrayStruct* true_ptr = reinterpret_cast<const _ArrayStruct*>(obj);
153 _ArrayMetaData* true_obj = true_ptr->m_md;
154 char strtype[1024];
155 arcane_ttf_header();
156 TV_ttf_add_row("size", "Arcane::Int64", &true_obj->size);
157 TV_ttf_add_row("capacity", "Arcane::Int64", &true_obj->capacity);
158 snprintf(strtype, sizeof(strtype), "%s[%ld]", type_name, view.size());
159 TV_ttf_add_row("data", strtype, view.data());
160 return TV_ttf_format_ok;
161}
162} // namespace Arcane

239ATTR_USED
240int TV_ttf_display_type(const Arcane::ItemInternal* obj)
241{
242 // WARNING: this implementation directly accesses Arcane structures.
243 // If the Arcane implementation changes, incompatibilities may
244 // occur
245
246 using Arcane::Integer;
247 char strtype[32];
248
249 arcane_ttf_header();
250 TV_ttf_add_row("local_id", "Integer", reinterpret_cast<const int*>(obj));
251 _ItemSharedInfo* shared_info = reinterpret_cast<_ItemSharedInfo*>(Arcane::TotalviewAdapter::getItemSharedInfo(obj));
252 const Integer local_id = obj->localId();
253
254 TV_ttf_add_row("unique_id", "Int64", shared_info->m_unique_ids.ptrAt(local_id));
255 TV_ttf_add_row("kind", "eItemKind", &shared_info->m_item_kind);
256
257 snprintf(strtype, sizeof(strtype), "type=%s", obj->typeInfo()->typeName().localstr());
258 //TV_ttf_add_row(strtype,"ItemTypeInfo*",&shared_info->m_item_type);
259
260 {
261 // Nodes
262 for (Integer i = 0; i < obj->nbNode(); ++i) {
263 Arcane::ItemInternal* item = obj->internalNode(i);
264 snprintf(strtype, sizeof(strtype), "node[%d] (lid=%d uid=%ld)", i, item->localId(), item->uniqueId().asInt64());
265 TV_ttf_add_row(strtype, "ItemInternal*", item);
266 }
267 }
268 {
269 // Edges
270 for (Integer i = 0; i < obj->nbEdge(); ++i) {
271 Arcane::ItemInternal* item = obj->internalEdge(i);
272 snprintf(strtype, sizeof(strtype), "edge[%d] (lid=%d uid=%ld)", i, item->localId(), item->uniqueId().asInt64());
273 TV_ttf_add_row(strtype, "ItemInternal*", item);
274 }
275 }
276 {
277 // Faces
278 for (Integer i = 0; i < obj->nbFace(); ++i) {
279 Arcane::ItemInternal* item = obj->internalFace(i);
280 snprintf(strtype, sizeof(strtype), "face[%d] (lid=%d uid=%ld)", i, item->localId(), item->uniqueId().asInt64());
281 TV_ttf_add_row(strtype, "ItemInternal*", item);
282 }
283 }
284 {
285 // Cells
286 for (Integer i = 0; i < obj->nbCell(); ++i) {
287 Arcane::ItemInternal* item = obj->internalCell(i);
288 snprintf(strtype, sizeof(strtype), "cell[%d] (lid=%d uid=%ld)", i, item->localId(), item->uniqueId().asInt64());
289 TV_ttf_add_row(strtype, "ItemInternal*", item);
290 }
291 }
292
293 // TODO for DualNode and Link, but it's not that simple...
294
295 const Integer flags = obj->flags();
296 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_Boundary, "Boundary");
297 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_HasFrontCell, "HasFrontCell");
298 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_HasBackCell, "HasBackCell");
299 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_FrontCellIsFirst, "FrontCellIsFirst");
300 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_BackCellIsFirst, "BackCellIsFirst");
301 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_Own, "Own");
302 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_Added, "Added");
303 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_Suppressed, "Suppressed");
304 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_Shared, "Shared");
305 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_SubDomainBoundary, "SubDomainBoundary");
306 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_JustAdded, "JustAdded");
307 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_NeedRemove, "NeedRemove");
308 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_SlaveFace, "SlaveFace");
309 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_MasterFace, "MasterFace");
310 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_Detached, "Detached");
311 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_HasEdgeFor1DItems, "HasEdgeFor1DItems");
312 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_UserMark1, "UserMark1");
313 show_ttf_internal_flag(flags, Arcane::ItemInternal::II_UserMark2, "UserMark2");
314
315 return TV_ttf_format_ok;
316}

354/* Arcane::ItemEnumerator */
355/*---------------------------------------------------------------------------*/
356
357// Symmetric structure to allow access to private data.
358// It must be certain that this structure is identical to ItemEnumerator
359struct __MY_ItemEnumerator
360{
361 Arcane::ItemInternal** m_items;
362 const Arcane::Int32* ARCANE_RESTRICT m_local_ids;
363 Arcane::Integer m_index;
364 Arcane::Integer m_count;
365 const Arcane::ItemGroupImpl* m_group_impl; // could be removed in release mode if necessary
366};
367
368int TV_ttf_display_type(const Arcane::ItemEnumerator* obj)
369{
370 arcane_ttf_header();
371
372 __MY_ItemEnumerator* mobj = (__MY_ItemEnumerator*)obj;
373
374 Arcane::Integer index = mobj->m_index;
375
376 TV_ttf_add_row("index", TV_ttf_type_int, &mobj->m_index);
377 TV_ttf_add_row("count", TV_ttf_type_int, &mobj->m_count);
378 TV_ttf_add_row("local_id", TV_ttf_type_int, &mobj->m_local_ids[index]);
379 if (index < mobj->m_count)
380 TV_ttf_add_row("item", "Arcane::ItemInternal", (mobj->m_items[mobj->m_local_ids[index]]));
381 return TV_ttf_format_ok;
382}
