38void DoKtoSimpleCSRMatrixConverter::convert(
39const IMatrixImpl* sourceImpl, IMatrixImpl* targetImpl) const
40{
41 const SrcMatrix& src = cast<SrcMatrix>(sourceImpl, sourceBackend());
42 TgtMatrix& tgt = cast<TgtMatrix>(targetImpl, targetBackend());
43
44 alien_debug([&] {
45 cout() << "Converting DoKMatrix: " << &src << " to SimpleCSRMatrix: " << &tgt;
46 });
47
48 if (sourceImpl->block()) {
49 throw FatalErrorException(A_FUNCINFO, "Block matrices are not handled yet");
50 }
51 else if (sourceImpl->vblock()) {
52 throw FatalErrorException(A_FUNCINFO, "Variable block matrices are not handled yet");
53 }
54 else
55 _build(src, tgt);
56}

63void DoKtoSimpleCSRMatrixConverter::_buildProfile(const SrcMatrix& src, TgtMatrix& tgt) const
64{
65 const ISpace& space = src.rowSpace();
66 const MatrixDistribution& dist = src.distribution();
67 IMessagePassingMng* parallel_mng = dist.parallelMng();
68
69 if (parallel_mng == nullptr)
70 return;
71
72 // if (space != src.colSpace())
73 // throw FatalErrorException("Matrix profiler must be used with square matrix");
74
75 Integer nproc = parallel_mng->commSize();
76
77 // Ugly const_cast but in need for DoKReverseIndexer
78 const_cast<SrcMatrix&>(src).compact();
79
80 const Integer local_size = dist.localRowSize();
81 const Integer global_size = dist.globalRowSize();
82 const Integer local_offset = dist.rowOffset();
83
84 const DoKLocalMatrixT<Real>& dokMatrix = src.data();
85 IReverseIndexer* dokMatrixRIndexer = dokMatrix.getReverseIndexer();
86
87 UniqueArray<Integer> offsets(nproc + 1);
88 for (int p = 0; p < nproc; p++) {
89 offsets[p] = dist.rowOffset(p);
90 }
91 offsets[nproc] = global_size;
92
93 SimpleCSRInternal::CSRStructInfo& profile = tgt.internal()->getCSRProfile();
94 profile.init(local_size);
95
96 ArrayView<Integer> row_offsets = profile.getRowOffset();
97 row_offsets.fill(0);
98 for (Integer i = 0; i < dokMatrixRIndexer->size(); ++i) {
99 // FIXME: check if index is correct.
100 const Integer localRow = (*dokMatrixRIndexer)[i].value().first + 1 - local_offset;
101 ++row_offsets[localRow];
102 }
103 for (Integer i = 0; i < local_size; ++i)
104 row_offsets[i + 1] += row_offsets[i];
105
106 profile.allocate();
107 ArrayView<Integer> cols = profile.getCols();
108
109 tgt.allocate();
110 ArrayView<Real> values = tgt.internal()->getValues();
111 values.fill(0);
112
113 for (Integer i = 0; i < dokMatrixRIndexer->size(); ++i) {
114 // This is not useful currently. offsetOfIJ == i !
115 // const Integer offsetOfIJ =
116 // dokMatrix.getIndexer()->findOffset((*dokMatrixRIndexer)[i].first,
117 // (*dokMatrixRIndexer)[i].second); cols[offsetOfIJ] =
118 // (*dokMatrixRIndexer)[i].second;
119 // FIXME: check if id exists
120 cols[i] = (*dokMatrixRIndexer)[i].value().second;
121 }
122
123 ConstArrayView<Real> dokValues = dokMatrix.getValues();
124 UniqueArray<Real>& csrValues = tgt.internal()->getValues();
125 for (Integer i = 0; i < dokMatrixRIndexer->size(); ++i) {
126 csrValues[i] = dokValues[i];
127 }
128
129 if (nproc > 1)
130 tgt.parallelStart(offsets, parallel_mng, true);
131 else
132 tgt.sequentialStart();
133
134 profile.getColOrdering() = SimpleCSRInternal::CSRStructInfo::eFull;
135}
