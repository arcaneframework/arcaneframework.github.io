50void SimpleCSRtoSYCLVectorConverter::convert(
51const IVectorImpl* sourceImpl, IVectorImpl* targetImpl) const
52{
53 const SimpleCSRVector<double>& v =
54 cast<SimpleCSRVector<double>>(sourceImpl, sourceBackend());
55 SYCLVector<double>& v2 =
56 cast<SYCLVector<double>>(targetImpl, targetBackend());
57
58 alien_debug(
59 [&] { cout() << "Converting SimpleCSRVector: " << &v << " to SYCLVector " << &v2; });
60
61 ConstArrayView<Real> values = v.values();
62 v2.setValues(values.size(), values.data());
63}
