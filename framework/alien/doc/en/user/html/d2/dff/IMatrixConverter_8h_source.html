23/*!
24 * \ingroup core
25 * \brief Matrices converter
26 *
27 * Interface to convert a matrix in another format, provided a converter between those two
28 * formats has been registered.
29 * If there is no direct converter between the source and target format, a third
30 * intermediate format will be used if available.
31 *
32 * \todo IMatrixImpl.h is needed only for the template definition. Could be removed
33 */
34class IMatrixConverter : public Alien::ObjectWithTrace
35{
36 public:
37 //! Type of matrix implementation
38 typedef Alien::IMatrixImpl IMatrixImpl;
39 //! Type of matrix backend
40 typedef Alien::BackEndId BackEndId;
41
42 public:
43 //! Free resources
44 virtual ~IMatrixConverter() {}
45
46 public:
47 /*!
48 * \brief Get the source backend id
49 * \returns The source backend id
50 */
51 virtual BackEndId sourceBackend() const = 0;
52
53 /*!
54 * \brief Get the target backend id
55 * \returns The target backend id
56 */
57 virtual BackEndId targetBackend() const = 0;
58
59 /*!
60 * \brief Convert a matrix from one format to another
61 * \param[in] sourceImpl Implementation of the source matrix
62 * \param[in,out] targetImpl Implementation of the target matrix
63 */
64 virtual void convert(const IMatrixImpl* sourceImpl, IMatrixImpl* targetImpl) const = 0;
65
66 public:
67 /*!
68 * \brief Cast a matrix implementation in its actual type
69 * \param[in] impl Matrix implementation
70 * \param[in] backend Backend id
71 * \returns The actual matrix implementation
72 *
73 * \todo Check backend using T
74 */
75 template <typename T>
76 static T& cast(IMatrixImpl* impl,[[maybe_unused]] BackEndId backend)
77 {
78 ALIEN_ASSERT((impl != NULL), ("Null implementation"));
79 ALIEN_ASSERT((impl->backend() == backend), ("Bad backend"));
80 T* t = dynamic_cast<T*>(impl);
81 ALIEN_ASSERT((t != NULL), ("Bad dynamic cast"));
82 return *t;
83 }
84
85 /*!
86 * \brief Const cast a matrix implementation in its actual type
87 * \param[in] impl Matrix implemenantation
88 * \param[in] backend Backend id
89 * \returns The actual matrix implementation
90 *
91 * \todo Check backend using T
92 */
93 template <typename T>
94 static const T& cast(const IMatrixImpl* impl,[[maybe_unused]] BackEndId backend)
95 {
96 ALIEN_ASSERT((impl != NULL), ("Null implementation"));
97 ALIEN_ASSERT((impl->backend() == backend), ("Bad backend"));
98 const T* t = dynamic_cast<const T*>(impl);
99 ALIEN_ASSERT((t != NULL), ("Bad dynamic cast"));
100 return *t;
101 }
102};
