49void DoKToSimpleCSRVectorConverter::convert(
50const IVectorImpl* sourceImpl, IVectorImpl* targetImpl) const
51{
52 const SrcVector& src = cast<SrcVector>(sourceImpl, sourceBackend());
53 TgtVector& tgt = cast<TgtVector>(targetImpl, targetBackend());
54
55 alien_debug([&] {
56 cout() << "Converting : DoK " << &src << " to SimpleCSRVector " << &tgt;
57 });
58
59 if (sourceImpl->block()) {
60 throw FatalErrorException(A_FUNCINFO, "Block vectors are not handled yet");
61 }
62
63 tgt.resize(src.m_data.size());
64 const auto offset = src.distribution().offset();
65 for (const auto [key, val] : src.m_data) {
66 tgt[key - offset] = val;
67 }
68}
