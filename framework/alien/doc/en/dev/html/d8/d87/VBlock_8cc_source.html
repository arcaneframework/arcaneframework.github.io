161Integer
162VBlock::size(Integer index) const
163{
164 auto it = m_internal->m_all_sizes.find(index);
165
166 if (it == m_internal->m_all_sizes.end())
167 throw FatalErrorException(A_FUNCINFO, "index is not registered");
168 return it.value();
169}
